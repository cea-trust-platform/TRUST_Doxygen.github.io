66//! Calcule la viscosite dynamique de mélange (depend des fraction massique)<br>Voir Wilke https://aip.scitation.org/doi/pdf/10.1063/1.1747673
67void Loi_Etat_Binaire_GP_base::calculer_mu_wilke()
68{
69 const DoubleTab& tab_Y1 = le_fluide->inco_chaleur().valeurs();
70 DoubleTab& tab_mu = le_fluide->viscosite_dynamique().valeurs();
71
72 int i, n=tab_mu.size();
73 for (i=0 ; i<n ; i++)
74 {
75 double m1om2 = massmol1_/massmol2_;
76 double m2om1 = 1./m1om2;
77 double mu1omu2 = mu1_/mu2_;
78 double mu2omu1 = 1./mu1omu2;
79
80// XXX : Which is better ?
81 double a1 = 1. + sqrt(mu1omu2) * pow(mu2omu1,0.25);
82 double a2 = 1. + sqrt(mu2omu1) * pow(mu1omu2,0.25);
83// double a1 = 1. + sqrt(mu1omu2) * exp(0.25*log(mu2omu1));
84// double a2 = 1. + sqrt(mu2omu1) * exp(0.25*log(mu1omu2));
85 double b1 = sqrt(8.*(1.+m1om2));
86 double b2 = sqrt(8.*(1.+m2om1));
87 double phi_12 = m1om2*a1*a1/b1;
88 double phi_21 = m2om1*a2*a2/b2;
89
90 double y1 = tab_Y1(i,0);
91 double y2 = 1. - y1; //!< All about binary mixture !
92
93 tab_mu(i,0) = (mu1_*y1)/(y1+y2*phi_12) + (mu2_*y2)/(y2+y1*phi_21);
94 }
95 tab_mu.echange_espace_virtuel();
96 Debog::verifier("calculer_mu_wilke",tab_mu);
97}

151//! Calcule la viscosite dynamique sur Schmidt
152void Loi_Etat_Binaire_GP_base::calculer_mu_sur_Sc()
153{
154 /*
155 * ====================================================================
156 * Schmidt number : Sc = nu / D = mu / ( rho * D )
157 * => rho * D = mu/Sc
158 * species equation : div ( rho * D * grad Y1) = div ( mu/Sc * grad Y1)
159 * ====================================================================
160 */
161
162 Champ_Don& mu_sur_Sc = le_fluide->mu_sur_Schmidt();
163 const Champ_base& rho = le_fluide->masse_volumique().valeur();
164 DoubleTab& tab_mu_sur_Sc = mu_sur_Sc.valeurs();
165 const DoubleTab& tab_rho = rho.valeurs();
166 const int n=tab_mu_sur_Sc.size();
167
168 if (!sub_type(Champ_Uniforme,mu_sur_Sc.valeur()))
169 {
170 if (sub_type(Champ_Uniforme,rho))
171 {
172 Cerr << "We should not have a density field of type Champ_Uniforme !" << finl;
173 Process::exit();
174 }
175 else
176 for (int i=0 ; i<n ; i++) tab_mu_sur_Sc(i,0) = tab_rho(i,0)*diff_coeff_;
177 }
178 else
179 {
180 Cerr << "We should not have a mu_sur_Sc of type Champ_Uniforme !" << finl;
181 Process::exit();
182 }
183
184 double temps_champ = rho.temps();
185 mu_sur_Sc.valeur().changer_temps(temps_champ);
186 tab_mu_sur_Sc.echange_espace_virtuel();
187}

292double Loi_Etat_Binaire_GP_base::calculer_masse_volumique(double P, double Y1) const
293{
294 /*
295 * Mixing species state equation
296 * ===========================================================================
297 * rho = ( P * M_mix ) / ( R * T)
298 * with M_mix = 1 / (Y1/M1 + Y2/M2) = = M2 / (M2Y1/M1 + Y2)
299 * But Y2 = 1 - Y1 => M_mix = M2 / (M2Y1/M1 + 1 -Y1) = M2 / (1 + Y1(M2/M1 -1))
300 * ===========================================================================
301 * See https://doi.org/10.1016/j.ijheatmasstransfer.2020.120470
302 */
303 const double PM2 = P*massmol2_;
304 const double RT = R_GAS*tempr_;
305 const double mix = 1.0+Y1*(massmol2_/massmol1_-1.0);
306 const double rh = PM2/(RT*mix);
307
308 return rh;
309}
