183inline int Parser::searchVar(const char * sv)
184{
185 std::string s(sv);
186 for (auto & c: s) c = toupper(c);
187 for (int i=0; i<ivar; i++)
188 if ((les_var[i]->getString()).compare(s) == 0 ) return i;
189 return -1;
190}

201inline double Parser::eval(PNode* node)
202{
203 assert(node!=NULL);
204 switch(node->type)
205 {
206 case 1 :
207 return evalOp(node); //!< PNode::OP
208 case 2 :
209 return node->nvalue; //!< PNode::VALUE
210 case 3 :
211 return evalFunc(node);//!< PNode::FUNCTION
212 case 4 :
213 return evalVar(node); //!< PNode::VAR
214 default:
215 Cerr << "method eval : Unknown type for this node !!!" << finl;
216 Process::exit();
217 return 0;
218 }
219}
