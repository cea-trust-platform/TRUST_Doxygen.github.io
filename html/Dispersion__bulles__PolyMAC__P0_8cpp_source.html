54void Dispersion_bulles_PolyMAC_P0::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const //!< 100% explicit
55{
56 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
57 if (!matrices.count(ch.le_nom().getString())) return; //rien a faire
58 Matrice_Morse& mat = *matrices.at(ch.le_nom().getString()), mat2;
59 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0, equation().zone_dis().valeur());
60 const DoubleTab& inco = ch.valeurs();
61 const IntTab& fcl = ch.fcl();
62
63 /* stencil : diagonal par bloc pour les vitesses aux faces, puis chaque composante des vitesses aux elems */
64 IntTrav stencil(0, 2);
65 stencil.set_smart_resize(1);
66 int i, j, e, f, k, l, N = inco.line_size(), d, db, D = dimension, nf_tot = zone.nb_faces_tot();
67
68 /*faces*/
69 for (f = 0; f < zone.nb_faces(); f++)
70 if (fcl(f, 0) < 2)
71 for (k = 0; k < N; k++)
72 for (l = 0; l < N; l++) stencil.append_line(N * f + k, N * f + l);
73 /* elements */
74 for (e = 0, i = nf_tot; e < zone.nb_elem_tot(); e++)
75 for (d = 0; d < D; d++, i++)
76 for (db = 0, j = nf_tot + D * e; db < D; db++, j++)
77 for (k = 0; k < N; k++)
78 for (l = 0; l < N; l++) stencil.append_line(N * i + k, N * j + l);
79
80
81 tableau_trier_retirer_doublons(stencil);
82 Matrix_tools::allocate_morse_matrix(inco.size_totale(), inco.size_totale(), stencil, mat2);
83 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
84}
