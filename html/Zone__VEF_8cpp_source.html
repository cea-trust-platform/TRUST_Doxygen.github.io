110//! Methode appelee par Zone_VF::discretiser apres la creation<br>des faces reelles.<br>On reordonne les faces de sorte a placer les faces "non standard"<br>au debut de la liste des faces. Les faces non standard sont celles<br>dont les volumes de controles sont modifies par les conditions aux<br>limites.
111void Zone_VEF::reordonner(Faces& les_faces)
112{
113 if (Process::je_suis_maitre())
114 Cerr << "Zone_VEF::reordonner les_faces " << finl;
115
116// Construction de rang_elem_non_std_ :
117// C'est un vecteur indexe par les elements du domaine.
118// size() = nb_elem()
119// size_tot() = nb_elem_tot()
120// Valeurs dans le tableau :
121// rang_elem_non_std_[i] = -1 si l'element i est standard,
122// sinon
123// rang_elem_non_std_[i] = j, ou j est l'indice de l'element dans
124// les tableaux indexes par les elements non standards (par exemple
125// le tableau Zone_Cl_VEF::type_elem_Cl_).
126// Un element est non standard s'il est voisin d'une face frontiere.
127 {
128 const int nb_faces_front = zone().nb_faces_frontiere();
129 zone().creer_tableau_elements(rang_elem_non_std_, ArrOfInt::NOCOPY_NOINIT);
130 rang_elem_non_std_ = -1;
131// D'abord on marque les elements non standards avec rang_elem_non_std_[i] = 0
132 for (int i_face = 0; i_face < nb_faces_front; i_face++)
133 {
134 const int elem = les_faces.voisin(i_face, 0);
135 rang_elem_non_std_[elem] = 0;
136 }
137 rang_elem_non_std_.echange_espace_virtuel();
138
139// On construit le md_vector des elements non standards, mais il n'est pas utilise.
140// L'important, c'est que rang_elem_non_std_ soit rempli correctement pour etre
141// utilise par creer_md_vect_renum() dans Zone_Cl_VEF::associer()
142 MD_Vector md_vect_elems_non_std;
143 MD_Vector_tools::creer_md_vect_renum_auto(rang_elem_non_std_, md_vect_elems_non_std);
144 }
145
146// Construction du tableau de renumerotation des faces. Ce tableau,
147// une fois trie dans l'ordre croissant donne l'ordre des faces dans
148// la zone_VF. La cle de tri est construite de sorte a pouvoir retrouver
149// l'indice de la face a partir de la cle par la formule :
150// indice_face = cle % nb_faces
151 const int nbr_faces = les_faces.nb_faces();
152 ArrOfInt sort_key(nbr_faces);
153 {
154 nb_faces_std_ =0;
155 const int nb_faces_front = zone().nb_faces_frontiere();
156// Attention : face_voisins_ n'est pas encore initialise, il
157// faut passer par les_faces.voisins() :
158 const IntTab& facevoisins = les_faces.voisins();
159// On place en premier les faces de bord:
160 int i_face;
161 for (i_face = 0; i_face < nbr_faces; i_face++)
162 {
163 int key = -1;
164 if (i_face < nb_faces_front)
165 {
166// Si la face est au bord, elle doit etre placee au debut
167// (en fait elle ne doit pas etre renumerotee)
168 key = i_face;
169 }
170 else
171 {
172 const int elem0 = facevoisins(i_face, 0);
173 const int elem1 = facevoisins(i_face, 1);
174// Ces faces ont toujours deux voisins.
175 assert(elem0 >= 0 && elem1 >= 0);
176 if (rang_elem_non_std_[elem0] >= 0 || rang_elem_non_std_[elem1] >= 0)
177 {
178// Si la face est voisine d'un element non standard, elle
179// doit etre classee juste apres les faces de bord:
180 key = i_face;
181 }
182 else
183 {
184// Face standard : a la fin du tableau
185 key = i_face + nbr_faces;
186 nb_faces_std_++;
187 }
188 }
189 sort_key[i_face] = key;
190 }
191 sort_key.ordonne_array();
192
193// On transforme a nouveau la cle en numero de face:
194 for (i_face = 0; i_face < nbr_faces; i_face++)
195 {
196 const int key = sort_key[i_face] % nbr_faces;
197 sort_key[i_face] = key;
198 }
199 }
200// On reordonne les faces:
201 {
202 IntTab& faces_sommets = les_faces.les_sommets();
203 IntTab old_tab(faces_sommets);
204 const int nb_som_faces = faces_sommets.line_size();
205 for (int i = 0; i < nbr_faces; i++)
206 {
207 const int old_i = sort_key[i];
208 for (int j = 0; j < nb_som_faces; j++)
209 faces_sommets(i, j) = old_tab(old_i, j);
210 }
211 }
212
213 {
214 IntTab& faces_voisins = les_faces.voisins();
215 IntTab old_tab(faces_voisins);
216 for (int i = 0; i < nbr_faces; i++)
217 {
218 const int old_i = sort_key[i];
219 faces_voisins(i, 0) = old_tab(old_i, 0);
220 faces_voisins(i, 1) = old_tab(old_i, 1);
221 }
222 }
223
224// Calcul de la table inversee: reverse_index[ancien_numero] = nouveau numero
225 ArrOfInt reverse_index(nbr_faces);
226 {
227 for (int i = 0; i < nbr_faces; i++)
228 {
229 const int j = sort_key[i];
230 reverse_index[j] = i;
231 }
232 }
233// Renumerotation de elem_faces:
234 {
235// Nombre d'indices de faces dans le tableau
236 const int nb_items = elem_faces_.size();
237 ArrOfInt& array = elem_faces_;
238 for (int i = 0; i < nb_items; i++)
239 {
240 const int old = array[i];
241 array[i] = reverse_index[old];
242 }
243 }
244// Mise a jour des indices des faces de joint:
245 {
246 Joints& joints = zone().faces_joint();
247 const int nbr_joints = joints.size();
248 for (int i_joint = 0; i_joint < nbr_joints; i_joint++)
249 {
250 Joint& un_joint = joints[i_joint];
251 ArrOfInt& indices_faces = un_joint.set_joint_item(Joint::FACE).set_items_communs();
252 const int nbr_indfaces = indices_faces.size_array();
253 assert(nbr_indfaces == un_joint.nb_faces()); //!< renum_items_communs rempli ?
254 for (int i = 0; i < nbr_indfaces; i++)
255 {
256 const int old = indices_faces[i]; //!< ancien indice local
257 indices_faces[i] = reverse_index[old];
258 }
259// Les faces de joint ne sont plus consecutives dans le
260// tableau: num_premiere_face n'a plus ne sens
261 un_joint.fixer_num_premiere_face(-1);
262 }
263 }
264}

266void Zone_VEF::discretiser()
267{
268 Zone& zone_geom=zone();
269 typer_elem(zone_geom);
270 Elem_geom_base& elem_geom = zone_geom.type_elem().valeur();
271 Zone_VF::discretiser();
272
273// Correction du tableau face_voisins:
274// A l'issue de Zone_VF::discretiser(), les elements voisins 0 et 1 d'une
275// face sont les memes sur tous les processeurs qui possedent la face.
276// Si la face est virtuelle et qu'un des deux elements voisins n'est
277// pas connu (il n'est pas dans l'epaisseur du joint), l'element voisin
278// vaut -1. Cela peut etre un voisin 0 ou un voisin 1.
279// On corrige les faces virtuelles pour que, si un element voisin n'est
280// pas connu, alors il est voisin1. Le voisin0 est donc toujours valide.
281 {
282 IntTab& face_vois = face_voisins();
283 const int debut = nb_faces();
284 const int fin = nb_faces_tot();
285 for (int i = debut; i < fin; i++)
286 {
287 if (face_voisins(i, 0) == -1)
288 {
289 face_vois(i, 0) = face_vois(i, 1);
290 face_vois(i, 1) = -1;
291 }
292 }
293 }
294
295// Verification de la coherence entre l'element geometrique et
296// l'elemnt de discretisation
297
298
299
300 if (sub_type(Tri_VEF,type_elem_.valeur()))
301 {
302 if (!sub_type(Triangle,elem_geom))
303 {
304 Cerr << " Le type de l'element geometrique " <<
305 elem_geom.que_suis_je() << " est incorrect" << finl;
306 Cerr << " Seul le type Triangle est compatible avec la discretisation VEF en dimension 2" << finl;
307 Cerr << " Il faut trianguler le domaine lorsqu'on utilise le mailleur interne" ;
308 Cerr << " en utilisant l'instruction: Trianguler nom_dom" <<
309 finl;
310 exit();
311 }
312 }
313 else if (sub_type(Tetra_VEF,type_elem_.valeur()))
314 {
315 if (!sub_type(Tetraedre,elem_geom))
316 {
317 Cerr << " Le type de l'element geometrique " <<
318 elem_geom.que_suis_je() << " est incorrect" << finl;
319 Cerr << " Seul le type Tetraedre est compatible avec la discretisation VEF en dimension 3" << finl;
320 Cerr << " Il faut tetraedriser le domaine lorsqu'on utilise le mailleur interne";
321 Cerr << " en utilisant l'instruction: Tetraedriser nom_dom" <<
322 finl;
323 exit();
324 }
325 }
326
327 else if (sub_type(Quadri_VEF,type_elem_.valeur()))
328 {
329
330 if (!sub_type(Quadrangle_VEF,elem_geom))
331 {
332 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
333 exit();
334 }
335 }
336 else if (sub_type(Hexa_VEF,type_elem_.valeur()))
337 {
338
339 if (!sub_type(Hexaedre_VEF,elem_geom))
340 {
341 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
342 exit();
343 }
344 }
345
346 int num_face;
347
348
349// On remplit le tableau face_normales_;
350// Attention : le tableau face_voisins n'est pas exactement un
351// tableau distribue. Une face n'a pas ses deux voisins dans le
352// meme ordre sur tous les processeurs qui possedent la face.
353// Donc la normale a la face peut changer de direction d'un
354// processeur a l'autre, y compris pour les faces de joint.
355 {
356 const int n = nb_faces();
357 face_normales_.resize(n, dimension);
358 creer_tableau_faces(face_normales_, Array_base::NOCOPY_NOINIT);
359 const IntTab& face_som = face_sommets();
360 const IntTab& face_vois = face_voisins();
361 const IntTab& elem_face = elem_faces();
362 const int n_tot = nb_faces_tot();
363 for (num_face = 0; num_face < n_tot; num_face++)
364 {
365 type_elem_.normale(num_face,
366 face_normales_,
367 face_som, face_vois, elem_face, zone_geom);
368 }
369 }
370
371// Calcul de facette_normales_
372 type_elem_.creer_facette_normales(zone_geom,
373 facette_normales(),
374 rang_elem_non_std());
375
376 calculer_volumes_entrelaces();
377 Cerr << "Informations of the Zone VEF of the domain " << zone().domaine().le_nom() << " : " << finl;
378
379 calculer_h_carre();
380// Calcul des porosites
381 Zone_VF::calculer_porosites();
382}

384void Zone_VEF::calculer_h_carre()
385{
386 const int nbe=nb_elem();
387// Calcul de h_carre
388 h_carre = 1.e30;
389 h_carre_.resize(nbe);
390// Calcul des surfaces
391 const DoubleVect& surfaces=face_surfaces();
392 const int nb_faces_elem=zone().nb_faces_elem();
393 for (int num_elem=0; num_elem<nbe; num_elem++)
394 {
395 double surf_max = 0;
396 for (int i=0; i<nb_faces_elem; i++)
397 {
398 double surf = surfaces(elem_faces(num_elem,i));
399 surf_max = (surf > surf_max)? surf : surf_max;
400 }
401 double vol = volumes(num_elem)/surf_max;
402 vol *= vol;
403 h_carre_(num_elem) = vol;
404 h_carre = ( vol < h_carre )? vol : h_carre;
405 }
406// Modif BM: je pense qu'avec mp_min(h_carre) on va eviter beaucoup de differences seq/par
407 h_carre = mp_min(h_carre);
408 Cerr << "Lowest cell size h=(Volume/max(Surface))= " << sqrt(h_carre) << finl;
409 double moyenne = 0.;
410 for (int i = 0; i < nbe; i++)
411 moyenne += h_carre_(i);
412 moyenne = mp_sum(moyenne);
413 double h_carre_moyen = moyenne / mp_sum((double)nbe);
414 Cerr << "Average cell size <h>= " << sqrt(h_carre_moyen) << finl;
415 if (h_carre*1e6<h_carre_moyen)
416 Cerr << "Warning, a cell size is more than 1000 times smaller than the average cell size of the domain. Check your mesh." << finl;
417 Cerr<<"=============================================="<<finl;
418}

420void Zone_VEF::calculer_volumes_entrelaces()
421{
422// Cerr << "les normales aux faces " << face_normales() << finl;
423// On calcule les volumes entrelaces;
424
425// Si domaine dynamique, le tableau peut eventuellement deja avoir la bonne structure, ou pas.
426// S'il n'est pas deformable, on n'est pas cense passer ici deux fois.
427 assert(zone().domaine().deformable()
428 || !(volumes_entrelaces_.get_md_vector().non_nul()));
429 if (!(volumes_entrelaces_.get_md_vector() == md_vector_faces()))
430 {
431 volumes_entrelaces_.reset();
432 creer_tableau_faces(volumes_entrelaces_, Array_base::NOCOPY_NOINIT);
433 }
434 int nb_faces_elem=zone().nb_faces_elem();
435 int num_face;
436 int nmax = premiere_face_int();
437 const double facteur = 1. / nb_faces_elem;
438 for (num_face = 0; num_face < nmax; num_face++)
439 {
440 int elem1 = face_voisins_(num_face,0);
441 volumes_entrelaces_[num_face] = volumes(elem1) * facteur;
442 }
443 nmax = nb_faces();
444 for ( ; num_face < nmax; num_face++)
445 {
446 int elem1 = face_voisins_(num_face,0);
447 int elem2 = face_voisins_(num_face,1);
448 volumes_entrelaces_[num_face] = (volumes(elem1) + volumes(elem2)) * facteur;
449 }
450 volumes_entrelaces_.echange_espace_virtuel();
451}

458void Zone_VEF::modifier_pour_Cl(const Conds_lim& conds_lim)
459{
460 Journal() << "Zone_VEF::Modifier_pour_Cl" << finl;
461 int nb_cond_lim=conds_lim.size();
462 int num_cond_lim=0;
463 for (; num_cond_lim<nb_cond_lim; num_cond_lim++)
464 {
465// for cl
466 const Cond_lim_base& cl = conds_lim[num_cond_lim].valeur();
467 if (sub_type(Periodique, cl))
468 {
469// if Perio
470 const Periodique& la_cl_period = ref_cast(Periodique,cl);
471 int nb_faces_elem = zone().nb_faces_elem();
472 const Front_VF& la_front_dis = ref_cast(Front_VF,cl.frontiere_dis());
473 int ndeb = 0;
474 int nfin = la_front_dis.nb_faces_tot();
475#ifndef NDEBUG
476 int num_premiere_face = la_front_dis.num_premiere_face();
477 int num_derniere_face = num_premiere_face+nfin;
478#endif
479 int nbr_faces_bord = la_front_dis.nb_faces();
480 assert((nb_faces()==0)||(ndeb<nb_faces()));
481 assert(nfin>=ndeb);
482 int elem1,elem2,k;
483 int face;
484// Modification des tableaux face_voisins_ , face_normales_ , volumes_entrelaces_
485// On change l'orientation de certaines normales
486// de sorte que les normales aux faces de periodicite soient orientees
487// de face_voisins(la_face_en_question,0) vers face_voisins(la_face_en_question,1)
488// comme le sont les faces internes d'ailleurs
489
490 DoubleVect C1C2(dimension);
491 double vol,psc=0;
492
493 for (int ind_face=ndeb; ind_face<nfin; ind_face++)
494 {
495// for ind_face
496 face = la_front_dis.num_face(ind_face);
497 if ( (face_voisins_(face,0) == -1) || (face_voisins_(face,1) == -1) )
498 {
499 int faassociee = la_front_dis.num_face(la_cl_period.face_associee(ind_face));
500 if (ind_face<nbr_faces_bord)
501 {
502 assert(faassociee>=num_premiere_face);
503 assert(faassociee<num_derniere_face);
504 }
505
506 elem1 = face_voisins_(face,0);
507 elem2 = face_voisins_(faassociee,0);
508 vol = (volumes(elem1) + volumes(elem2))/nb_faces_elem;
509 volumes_entrelaces_[face] = vol;
510 volumes_entrelaces_[faassociee] = vol;
511 face_voisins_(face,1) = elem2;
512 face_voisins_(faassociee,0) = elem1;
513 face_voisins_(faassociee,1) = elem2;
514 psc = 0;
515 for (k=0; k<dimension; k++)
516 {
517 C1C2[k] = xv_(face,k) - xp_(face_voisins_(face,0),k);
518 psc += face_normales_(face,k)*C1C2[k];
519 }
520
521 if (psc < 0)
522 for (k=0; k<dimension; k++)
523 face_normales_(face,k) *= -1;
524
525 for (k=0; k<dimension; k++)
526 face_normales_(faassociee,k) = face_normales_(face,k);
527 }
528 }
529 }
530 }

622{

624// ou si le maillage est deformable
625 if (vecteur_face_facette_.size()==0 || zone().domaine().deformable())
626 {
627// Taille 8*n*4*3*2=192n
628 const int nfa7 = type_elem().nb_facette();
629 const int nb_poly_tot = nb_elem_tot();
630 vecteur_face_facette_.resize(nb_poly_tot,nfa7,dimension,2);

632 const IntTab& les_Polys = zone().les_elems();

640// Boucle sur les facettes du polyedre non standard:
641 for (int fa7=0; fa7<nfa7; fa7++)
642 {
643 num1 = elem_faces(poly,KEL(0,fa7));
644 num2 = elem_faces(poly,KEL(1,fa7));
645

648 {
649// Calcul de la ieme coordonnee du centre de la fa7

654 int isom_glob = les_Polys(poly,isom_loc);
655 coord_centre_fa7 += coord(isom_glob,i);
656 }
657 coord_centre_fa7 /= nb_som_facette;
658// Fin calcul de la ieme coordonnee du centre de la fa7
659
660// Calcul de la ieme coordonnee du centre des faces num1 et num2
661 double coord_centre_num1 = 0;

663 for (int num_som_face=0; num_som_face<nb_som_facette; num_som_face++)
664 {

666 int s2 = face_sommets(num2,num_som_face);
667 coord_centre_num1 += coord(s1,i);

669 }
670 coord_centre_num1 /= nb_som_facette;
671 coord_centre_num2 /= nb_som_facette;
672// Fin de Calcul de la ieme coordonnee du centre des faces num1 et num2

674 vecteur_face_facette_(poly,fa7,i,1) = coord_centre_fa7 - coord_centre_num2;
675 }
676// Fin de Calcul des rx0 et rx1
677 }
678 }
679 }
