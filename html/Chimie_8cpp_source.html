59double Chimie::calculer_pas_de_temps() const
60{
61 double dt=1e30;
62 double dt_reaction;
63 for (int i=0; i<reactions_.size(); i++) //!< boucle sur l'ensemble des reactions
64 {
65 dt_reaction=reactions_[i].calculer_pas_de_temps();
66 if(i==0) Cout<<"Pour le(s) reaction(s), dt_max = ";
67 Cout<<dt_reaction<<" ";
68 if (i==(reactions_.size()-1)) Cout<<finl;
69 dt=std::min(dt,dt_reaction);
70 }
71// double dt_n=pb_->schema_temps().pas_de_temps();
72// filtre Butterworth :
73// alpha=1 -> pas de filtre
74// alpha->0 -> tres filtre
75// double alpha_Butterworth=1.;
76// return (dt_n*(1.-alpha_Butterworth)+alpha_Butterworth*dt);
77 return dt;
78}

82void Chimie::completer(const Probleme_base& pb)
83{
84// si pas de reaction on ne fait rien
85 if (reactions_.size()==0)
86 return;
87 pb_=pb;
88// 1) determination du nombre de FM
89 int nb_fm=0;
90 int nb_grains=0;
91 int nb_concentration=0;
92// Motcles alias;
93 ArrOfDouble masses_molaires;
94 for (int n=0; n<pb.nombre_d_equations(); n ++)
95 {
96
97 if (sub_type(Convection_Diffusion_Concentration,pb.equation(n)))
98 nb_concentration++;
99
100 }
101
102// 2 stockage des alias et des masses molaires
103 masses_molaires.resize_array(nb_grains+nb_fm+nb_concentration);
104 alias.dimensionner(nb_grains+nb_fm+nb_concentration);
105 nb_grains_=nb_grains;
106 int nb=0;
107 for (int n=0; n<pb.nombre_d_equations(); n ++)
108 {
109 if (sub_type(Convection_Diffusion_Concentration,pb.equation(n)))
110 {
111 const Convection_Diffusion_Concentration& eq= ref_cast(Convection_Diffusion_Concentration,pb.equation(n));
112 masses_molaires[nb]=eq.masse_molaire();
113 alias[nb]=eq.inconnue().valeur().le_nom();
114 REF(Champ_Inc_base) inco;
115 inco = eq.inconnue().valeur();
116 liste_C_.add(inco);
117 nb++;
118 }
119 }
120 assert(liste_ai_.size()==nb_grains_);
121// verification de l'unicite des alias
122 marqueur_espece_en_competition_micro_melange_=-1;
123 {
124 LIST(Nom) test;
125 for (int n=0; n<alias.size(); n++)
126 {
127 int deja=test.contient(alias[n]);
128 if (!deja)
129 test.add(alias[n]);
130 else
131 {
132 Cerr<<alias[n] <<" defini plusieurs fois !!!!!"<<finl;
133// Cerr<<test<< " "<<alias<<finl;
134 exit();
135 }
136 if (espece_en_competition_micro_melange_!=Nom())
137 {
138 if (espece_en_competition_micro_melange_==alias[n])
139 marqueur_espece_en_competition_micro_melange_=n;
140 }
141 }
142 assert(test.size()==nb_fm+nb_grains_+nb_concentration);
143 }
144 if (espece_en_competition_micro_melange_!=Nom())
145 {
146 if (marqueur_espece_en_competition_micro_melange_==-1)
147 {
148 Cerr<<" espece_en_competition_micro_melange non trouve: "<< espece_en_competition_micro_melange_<<finl;
149 exit();
150 }
151 else
152 Cerr<<" marqueur_espece_en_competition_micro_melange "<<marqueur_espece_en_competition_micro_melange_<<finl;
153 }
154 if(min_array(masses_molaires)<0)
155 {
156 Cerr<< "pb avec les masses_molaires "<<masses_molaires<<finl;
157 exit();
158 }
159 if (liste_Y_.size()>0)
160 {
161 Puissance_volumique_=(liste_Y_[0].valeur().valeurs()); //!< dimensionnement du tableau...
162 Puissance_volumique_=0.;
163 }
164 else
165 {
166 assert(liste_C_.size()!=0);
167 }
168// alias_=alias;
169 for (int i=0; i<reactions_.size(); i++)
170 {
171 reactions_[i].completer(alias,masses_molaires);
172 }
