86void Champ_front_pression_from_u::mettre_a_jour(double tps)
87{
88 const Champ_Inc_base& inco=ref_inco_.valeur();
89// on recupere le flux_bord de l'op div c'est donc u
90 const Navier_Stokes_std& eqns=ref_cast(Navier_Stokes_std,inco.equation());
91 const Operateur_Div& opdiv=eqns.operateur_divergence();
92 const Zone_VF& zone_VF = ref_cast(Zone_VF,zone_dis());
93
94 const DoubleTab& Flux0=opdiv.valeur().flux_bords();
95 if (Flux0.size()==0)
96 {
97 Cerr<<" Div pas encore pres...."<<finl;
98 return;
99// DoubleTrav resu(eqns.div());
100// opdiv.calculer(inco.valeurs(),resu);
101 }
102 const DoubleTab& Flux=opdiv.valeur().flux_bords();
103 const Front_VF& le_bord= ref_cast(Front_VF,frontiere_dis());
104 int premiere=le_bord.num_premiere_face();
105 double surf=0;
106 double fluxb=0;
107 for(int i=0; i<le_bord.nb_faces(); i++)
108 {
109 int face=premiere+i;
110 surf+=zone_VF.face_surfaces(face);
111 fluxb+=Flux(face,0);

113 surf=mp_sum(surf);
114 fluxb=mp_sum(fluxb);
115 double u_moy=fluxb/surf;
