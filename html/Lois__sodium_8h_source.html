20// proprietes physiques
21/* Lois physiques du sodium issues de fits sur DEN/DANS/DM2S/STMF/LMES/RT/12-018/A */
22#define Tk ((T) + 273.15)
23const double Ksil = 1e-9;
24
25/* inverse de la densite du liquide (hors pression) */
26inline double IRhoL(double T)
27{
28 return (9.829728e-4 + Tk * (2.641186e-7 + Tk * (-3.340743e-11 + Tk * 6.680973e-14)));
29}
30
31/* sa derivee */
32inline double DTIRhoL(double T)
33{
34 return (2.641186e-7 + Tk * (-3.340743e-11 * 2 + Tk * 6.680973e-14 * 3));
35}
36
37/* densite du liquide */
38inline double RhoL(double T, double P)
39{
40 return exp(Ksil * (P - 1e5)) / IRhoL(T);
41}
42
43/* enthalpie du liquide */
44inline double HL(double T, double P)
45{
46 return (2992600 / Tk - 365487.2 + Tk * (1658.2 + Tk * (-.42395 + Tk * 1.4847e-4)))
47 + (IRhoL(T) - Tk * DTIRhoL(T)) * (1 - exp(Ksil*(1e5 - P))) / Ksil;
48}
49
50/* derivees */
51inline double DTHL(double T, double P)
52{
53 return (-2992600 / (Tk*Tk) + 1658.2 + Tk * (-.42395 * 2 + Tk * 1.4847e-4 * 3))
54 - Tk * (-3.340743e-11 * 2 + Tk * 6.680973e-14 * 6) * (1 - exp(Ksil*(1e5 - P))) / Ksil;
55}
56
57/* inverses par methode de Newton */
58inline double TLh(double h, double T0, double P)
59{
60 double T = T0, sec;
61 int i;
62 for (i = 0; i < 100 && std::abs( sec = HL(T, P) - h ) > 1e-8; i++)
63 T -= sec / DTHL(T, P);
64 return i < 100 ? T : -1e10;
65}
66/* derivees */
67inline double DTRhoL(double T, double P)
68{
69 return - exp(Ksil * (P - 1e5)) * DTIRhoL(T) / (IRhoL(T) * IRhoL(T));
70}
71inline double DPRhoL(double T, double P)
72{
73 return Ksil * RhoL(T, P);
74}
75
76/* derivees */
77inline double DPHL(double T, double P)
78{
79 return (IRhoL(T) - Tk * DTIRhoL(T)) * exp(Ksil * (1e5 - P));
80}
81
82/* energie volumique du liquide */
83inline double EL(double T, double P)
84{
85 return HL(T, P) * RhoL(T, P);
86}
87/* derivees */
88inline double DTEL(double T, double P)
89{
90 return DTHL(T, P) * RhoL(T, P) + HL(T, P) * DTRhoL(T, P);
91}
92inline double DPEL(double T, double P)
93{
94 return DPHL(T, P) * RhoL(T, P) + HL(T, P) * DPRhoL(T, P);
95}
96
97inline double TLe(double e, double T0, double P)
98{
99 double T = T0, sec;
100 int i;
101 for (i = 0; i < 100 && std::abs( sec = EL(T, P) - e ) > 1e-3; i++)
102 T -= sec / DTEL(T, P);
103 return i < 100 ? T : -1e10;
104}

106/* viscosite du liquide*/
107inline double MuL( double T )
108{
109 return exp( -6.4406 - 0.3958 * log(Tk) + 556.835 / Tk );
110}
111/* conductivite du liquide */
112inline double LambdaL( double T )
113{
114 return std::max(124.67 + Tk * (-.11381 + Tk * (5.5226e-5 - Tk * 1.1848e-8)), 0.045);
115}
116/* tension superficielle */
117inline double SigmaL(double T)
118{
119 return 0.2405 * pow(1 - Tk / 2503.7, 1.126);
120}
121/* Nusselt du liquide -> Skupinski */
122inline double NuL(double T, double P, double w, double Dh)
123{
124 double Pe = RhoL(T, P) * std::abs(w) * Dh * DTHL(T, P) / LambdaL(T);
125 return 4.82 + 0.0185 * pow(Pe, 0.827);
126}
127/* temperature de saturation */
128inline double Tsat_Na( double P )
129{
130 double A = 7.8130, B = 11209, C = 5.249e5;
131 return 2 * C / ( -B + sqrt(B*B + 4 * A * C - 4 * C * log(P / 1e6))) - 273.15;
132}
133/* sa derivee */
134inline double DTsat_Na( double P )
135{
136 double A = 7.8130, B = 11209, C = 5.249e5, Tsk = Tsat_Na(P) + 273.15;
137 return Tsk * Tsk / ( P * sqrt(B*B + 4 * A * C - 4 * C * log(P / 1e6)));
138}
139/* pression de vapeur saturante */
140inline double Psat_Na( double T )
141{
142 double A = 7.8130, B = 11209, C = 5.249e5;
143 return 1e6 * exp(A - B / Tk - C / (Tk * Tk));
144}
145/* sa derivee */
146inline double DPsat_Na( double T )
147{
148 double B = 11209, C = 5.249e5;
149 return (B / (Tk * Tk) + 2 * C / (Tk * Tk * Tk)) * Psat_Na(T);
150}
151/* enthalpie massique de saturation */
152inline double Hsat( double P )
153{
154 return HL(Tsat_Na(P), P);
155}
156/* sa derivee */
157inline double DHsat( double P )
158{
159 return DTsat_Na(P) * DTHL(Tsat_Na(P), P) + DPHL(Tsat_Na(P), P);
160}
161/* chaleur latente, a prendre a Tsat_Na */
162inline double Lvap_Na( double P )
163{
164 double Tc = 2503.7, Tsk = Tsat_Na(P) + 273.15;
165 return 3.9337e5 * ( 1 - Tsk / Tc) + 4.3986e6 * pow( 1 - Tsk / Tc, .29302);
166}
167
168/* sa derivee */
169inline double DLvap_Na( double P )
170{
171 double Tc = 2503.7, Tsk = Tsat_Na(P) + 273.15;
172 return DTsat_Na(P) * (-3.9337e5 / Tc - 4.3986e6 * .29302 * pow( 1 - Tsk / Tc, .29302 - 1) / Tc);
173}

175/* densite de la vapeur : (gaz parfait) * f1(Tsat_Na) * f2(DTsat_Na)*/
176#define f1(Ts) (2.49121 + Ts * (-5.53796e-3 + Ts * (7.5465e-6 + Ts * (-4.20217e-9 + Ts * 8.59212e-13))))
177#define Df1(Ts) (-5.53796e-3 + Ts * (2 * 7.5465e-6 + Ts * (-3 * 4.20217e-9 + Ts * 4 * 8.59212e-13)))
178#define f2(dT) (1 + dT * (-5e-4 + dT * (6.25e-7 - dT * 4.1359e-25)))
179#define Df2(dT) (-5e-4 + dT * (2 * 6.25e-7 - dT * 3 * 4.1359e-25))
180inline double RhoV( double T, double P )
181{
182 double Tsk = Tsat_Na(P) + 273.15, dT = Tk - Tsk;
183 return P * 2.7650313e-3 * f1(Tsk) * f2(dT) / Tk;
184}
185/* ses derivees */
186inline double DTRhoV( double T, double P )
187{
188 double Tsk = Tsat_Na(P) + 273.15, dT = Tk - Tsk;
189 return P * 2.7650313e-3 * f1(Tsk) * (Df2(dT) - f2(dT) / Tk) / Tk;
190}
191inline double DPRhoV( double T, double P )
192{
193 double Tsk = Tsat_Na(P) + 273.15, dT = Tk - Tsk;
194 return 2.7650313e-3 * (f1(Tsk) * f2(dT) + P * DTsat_Na(P) * (Df1(Tsk) * f2(dT) - f1(Tsk) * Df2(dT))) / Tk;
195}
196#undef f1
197#undef Df1
198#undef f2
199#undef Df2
200/* inverse de la densite de la vapeur */
201inline double IRhoV( double T, double P )
202{
203 return 1 / RhoV(T, P);
204}
205/* ses derivees */
206inline double DTIRhoV( double T, double P )
207{
208 return -DTRhoV(T, P) / (RhoV(T, P) * RhoV(T, P));
209}
210inline double DPIRhoV( double T, double P )
211{
212 return -DPRhoV(T, P) / (RhoV(T, P) * RhoV(T, P));
213}
214/* enthalpie de la vapeur */
215#define CpVs(Ts) (-1223.89 + Ts * (14.1191 + Ts * (-1.62025e-2 + Ts * 5.76923e-6)))
216#define DCpVs(Ts) (14.1191 + Ts * (- 2 * 1.62025e-2 + Ts * 3 * 5.76923e-6))
217inline double HV( double T, double P)
218{
219 double Ts = Tsat_Na(P), dT = T - Ts, Cp0 = 910, k = -4.6e-3;
220 return HL(Ts, P) + Lvap_Na(P) + Cp0 * dT + (Cp0 - CpVs(Ts)) * (1 - exp(k * dT)) / k;
221}
222/* ses derivees */
223inline double DTHV( double T, double P)
224{
225 double Ts = Tsat_Na(P), dT = T - Ts, Cp0 = 910, k = -4.6e-3;
226 return Cp0 - (Cp0 - CpVs(Ts)) * exp(k * dT);
227}
228inline double DPHV( double T, double P)
229{
230 double Ts = Tsat_Na(P), dT = T - Ts, Cp0 = 910, k = -4.6e-3;
231 return DPHL(Ts, P) + DLvap_Na(P) + DTsat_Na(P) * (DTHL(Ts, P) - Cp0 - DCpVs(Ts) * (1 - exp(k * dT)) / k + (Cp0 - CpVs(Ts)) * exp(k * dT));
232}
233#undef CpVs
234#undef DCpVs
235/* energie volumique de la vapeur */
236inline double EV( double T, double P)
237{
238 return RhoV(T, P) * HV(T, P);
239}
240/* ses derivees */
241inline double DTEV(double T, double P)
242{
243 return DTRhoV(T, P) * HV(T, P) + RhoV(T, P) * DTHV(T, P);
244}
245inline double DPEV(double T, double P)
246{
247 return DPRhoV(T, P) * HV(T, P) + RhoV(T, P) * DPHV(T, P);
248}

262/* Nusselt de la vapeur -> Dittus/ Boetler */
263inline double NuV(double T, double P, double w, double Dh)
264{
265 double Re = RhoV(T, P) * std::abs(w) * Dh / MuV(T), Pr = MuV(T) * DTHV(T, P) / LambdaV(T);
266 return 0.023 * pow(Re, 0.8) * pow(Pr, 0.4);
267}

269inline double Rho( double T, double x, double P )
270{
271 return 1 / ( (1-x)/RhoL(T, P) + x / RhoV(T, P) );
