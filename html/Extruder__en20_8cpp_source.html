122//! Triangule tous les element d'une zone: transforme<br>les elements goemetriques de la zone en triangles.<br>Pour l'instant on ne sait raffiner que des Rectangles<br>(on les coupe en 4).
123void Extruder_en20::extruder(Domaine& dom)
124{
125 Zone& zone = dom.zone(0);
126
127 if (zone.type_elem()->que_suis_je() == "Rectangle" || zone.type_elem()->que_suis_je() == "Quadrangle" )
128 {
129 Cerr << " The extrusion of quadrangle is made by Extruder and not by Extruder_en20 " << finl;
130 exit();
131 }
132 else if( zone.type_elem()->que_suis_je() == "Triangle")
133 {
134 int oldnbsom = zone.nb_som();
135 IntTab& les_elems=zone.les_elems();
136 int oldsz=les_elems.dimension(0);
137 double dx = direction[0]/NZ;
138 double dy = direction[1]/NZ;
139 double dz = direction[2]/NZ;
140
141 Faces les_faces;
142// zone.creer_faces(les_faces);
143 {
144// bloc a factoriser avec Zone_VF.cpp :
145 Type_Face type_face = zone.type_elem().type_face(0);
146 les_faces.typer(type_face);
147 les_faces.associer_zone(zone);
148
149 Static_Int_Lists connectivite_som_elem;
150 const int nb_sommets_tot = zone.domaine().nb_som_tot();
151 const IntTab& elements = zone.les_elems();
152
153 construire_connectivite_som_elem(nb_sommets_tot,
154 elements,
155 connectivite_som_elem,
156 1 /* include virtual elements */);
157
158 Faces_builder faces_builder;
159 IntTab elem_faces; //!< Tableau dont on aura pas besoin
160 faces_builder.creer_faces_reeles(zone,
161 connectivite_som_elem,
162 les_faces,
163 elem_faces);
164 }
165 const int nbfaces2D = les_faces.nb_faces();
166
167
168 int newnbsom = oldnbsom*(NZ+1)+NZ*oldsz+nbfaces2D*NZ+oldnbsom*NZ;
169 DoubleTab new_soms(newnbsom, 3);
170 DoubleTab& coord_sommets=dom.les_sommets();
171 Objet_U::dimension=3;
172
173
174// les sommets du maillage 2D sont translates en premier
175 for (int i=0; i<oldnbsom; i++)
176 {
177 double x = coord_sommets(i,0);
178 double y = coord_sommets(i,1);
179 double z=0;
180 if (coord_sommets.dimension(1)>2)
181 z=coord_sommets(i,2);
182 for (int k=0; k<=NZ; k++)
183 {
184 new_soms(k*oldnbsom+i,0)=x;
185 new_soms(k*oldnbsom+i,1)=y;
186 new_soms(k*oldnbsom+i,2)=z;
187
188 x += dx;
189 y += dy;
190 z += dz;
191 }
192 }
193
194
195// creation des centres de gravite des elements 2D puis translation de ces points
196 for (int i=0; i<oldsz; i++)
197 {
198 int i0=les_elems(i,0);
199 int i1=les_elems(i,1);
200 int i2=les_elems(i,2);
201
202 double xg = 1./3.*(coord_sommets(i0,0)+coord_sommets(i1,0)+coord_sommets(i2,0))+0.5*dx;
203 double yg = 1./3.*(coord_sommets(i0,1)+coord_sommets(i1,1)+coord_sommets(i2,1))+0.5*dy;
204 double z = 0.5*dz;
205 if (coord_sommets.dimension(1)>2)
206 z = 1./3.*(coord_sommets(i0,2)+coord_sommets(i1,2)+coord_sommets(i2,2))+0.5*dz;
207 for (int k=0; k<NZ; k++)
208 {
209
210 new_soms(oldnbsom*(NZ+1)+k*oldsz+i,0)=xg;
211 new_soms(oldnbsom*(NZ+1)+k*oldsz+i,1)=yg;
212 new_soms(oldnbsom*(NZ+1)+k*oldsz+i,2)=z;
213
214 xg += dx;
215 yg += dy;
216 z += dz;
217 }
218 }
219
220
221// creation des centres des faces du maillage 2D puis translation de ces points
222 for (int i=0; i<nbfaces2D; i++)
223 {
224 int i0=les_faces.sommet(i,0);
225 int i1=les_faces.sommet(i,1);
226
227 double x01 = 0.5*(coord_sommets(i0,0)+coord_sommets(i1,0))+0.5*dx;
228 double y01 = 0.5*(coord_sommets(i0,1)+coord_sommets(i1,1))+0.5*dy;
229 double z = 0.5*dz;
230 if (coord_sommets.dimension(1)>2)
231 z = 0.5*(coord_sommets(i0,2)+coord_sommets(i1,2))+0.5*dz;
232 for (int k=0; k<NZ; k++)
233 {
234 new_soms(oldnbsom*(NZ+1)+NZ*oldsz+k*nbfaces2D+i,0)=x01;
235 new_soms(oldnbsom*(NZ+1)+NZ*oldsz+k*nbfaces2D+i,1)=y01;
236 new_soms(oldnbsom*(NZ+1)+NZ*oldsz+k*nbfaces2D+i,2)=z;
237 x01 += dx;
238 y01 += dy;
239 z += dz;
240 }
241 }
242
243// creation des centres des aretes du maillage 2D puis translation de ces points
244 for (int i=0; i<oldnbsom; i++)
245 {
246 double x = coord_sommets(i,0)+0.5*dx;
247 double y = coord_sommets(i,1)+ 0.5*dy;
248 double z = 0.5*dz;
249 if (coord_sommets.dimension(1)>2)
250 z=coord_sommets(i,2)+0.5*dz;
251 for (int k=0; k<NZ; k++)
252 {
253 new_soms(oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i,0)=x;
254 new_soms(oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i,1)=y;
255 new_soms(oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i,2)=z;
256 x += dx;
257 y += dy;
258 z += dz;
259 }
260 }
261
262
263 coord_sommets.reset();
264 dom.ajouter(new_soms);
265
266 int newnbelem = 20*NZ*oldsz;
267 IntTab new_elems(newnbelem, 4); //!< les nouveaux elements
268 int cpt=0;
269
270
271// en premier, on stocke les tetra du haut et du bas : NZ*nb_triangle*2 tetra
272 for (int i=0; i<oldsz; i++)
273 {
274 int i0=les_elems(i,0);
275 int i1=les_elems(i,1);
276 int i2=les_elems(i,2);
277
278 int ig=oldnbsom*(NZ+1)+i;
279
280 for (int k=0; k<NZ; k++)
281 {
282 new_elems(2*k*oldsz+2*i,0) = i0;
283 new_elems(2*k*oldsz+2*i,1) = i1;
284 new_elems(2*k*oldsz+2*i,2) = i2;
285 new_elems(2*k*oldsz+2*i,3) = ig;
286 cpt++;
287
288 new_elems(2*k*oldsz+2*i+1,0) = i0+oldnbsom;
289 new_elems(2*k*oldsz+2*i+1,1) = i1+oldnbsom;
290 new_elems(2*k*oldsz+2*i+1,2) = i2+oldnbsom;
291 new_elems(2*k*oldsz+2*i+1,3) = ig;
292 cpt++;
293
294 mettre_a_jour_sous_zone(zone,i,2*k*oldsz+2*i,2);
295
296 i0+=oldnbsom;
297 i1+=oldnbsom;
298 i2+=oldnbsom;
299 ig+=oldsz;
300 }
301 }
302
303
304
305// puis les autres tetras
306 for (int i=0; i<nbfaces2D; i++)
307 {
308 for (int ivois=0; ivois<2; ivois++)
309 {
310 int elem = les_faces.voisin(i,ivois);
311
312 if (elem>=0)
313 {
314 int i0=les_faces.sommet(i,0);
315 int i1=les_faces.sommet(i,1);
316 int i01=oldnbsom*(NZ+1)+NZ*oldsz+i;
317
318 for (int k=0; k<NZ; k++)
319 {
320 int ig=oldnbsom*(NZ+1)+k*oldsz+elem;
321
322 int i00=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+i0;
323 int i11=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+i1;
324// int i00=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i0;
325// int i11=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i1;
326
327 new_elems(cpt,0) = i0;
328 new_elems(cpt,1) = i1;
329 new_elems(cpt,2) = i01;
330 new_elems(cpt++,3) = ig;
331
332 new_elems(cpt,0) = i0+oldnbsom;
333 new_elems(cpt,1) = i1+oldnbsom;
334 new_elems(cpt,2) = i01;
335 new_elems(cpt++,3) = ig;
336
337 new_elems(cpt,0) = i1;
338 new_elems(cpt,1) = i11;
339 new_elems(cpt,2) = i01;
340 new_elems(cpt++,3) = ig;
341
342 new_elems(cpt,0) = i11;
343 new_elems(cpt,1) = i1+oldnbsom;
344 new_elems(cpt,2) = i01;
345 new_elems(cpt++,3) = ig;
346
347 new_elems(cpt,0) = i0;
348 new_elems(cpt,1) = i00;
349 new_elems(cpt,2) = i01;
350 new_elems(cpt++,3) = ig;
351
352 new_elems(cpt,0) = i00;
353 new_elems(cpt,1) = i0+oldnbsom;
354 new_elems(cpt,2) = i01;
355 new_elems(cpt++,3) = ig;
356
357
358 i0+=oldnbsom;
359 i1+=oldnbsom;
360 i01+=nbfaces2D;
361 ig+=oldsz;
362 }
363 }
364 }
365 }
366
367 les_elems.ref(new_elems);
368
369// Reconstruction de l'octree
370 zone.invalide_octree();
371 zone.typer("Tetraedre");
372
373 extruder_dvt(dom, les_faces,oldnbsom, oldsz);
374
375 }
376 else
377 {
378 Cerr << "It is not known yet how to extrude "
379 << zone.type_elem()->que_suis_je() <<"s"<<finl;
380 exit();
381 }
382}

386void Extruder_en20::traiter_faces_dvt(Faces& les_faces_bord, Faces& les_faces, int oldnbsom, int oldsz, int nbfaces2D )
387{
388 int size_2D = les_faces_bord.nb_faces();
389
390 IntTab les_sommets(6*size_2D*NZ, 3);
391
392 for (int i=0; i<size_2D; i++)
393 {
394 int i0=les_faces_bord.sommet(i,0);
395 int i1=les_faces_bord.sommet(i,1);
396
397// double x01 = 0.5*(coord_sommets(i0,0)+coord_sommets(i1,0));
398// double y01 = 0.5*(coord_sommets(i0,1)+coord_sommets(i1,1));
399
400// on recherche le numero de cette face de bord: pas top!
401 int jface=-1;
402 for (int iface=0; iface<nbfaces2D; iface++)
403 {
404 int j0=les_faces.sommet(iface,0);
405 int j1=les_faces.sommet(iface,1);
406
407 if (((i0==j0) &&(i1==j1)) || ((i0==j1) &&(i1==j0)))
408 {
409 jface=iface;
410 break;
411 }
412 }
413 assert(jface>=0);
414
415 for (int k=0; k<NZ; k++)
416 {
417// double z = (k+0.5)*dz;
418// int i01 = zone.chercher_sommets(x01, y01, z);
419 int j01 = oldnbsom*(NZ+1)+NZ*oldsz+k*nbfaces2D+jface;
420
421 int i00=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+i0;
422 int i11=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+i1;
423// int i00=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i0;
424// int i11=oldnbsom*(NZ+1)+NZ*oldsz+NZ*nbfaces2D+k*oldnbsom+i1;
425
426 les_sommets(k*6*size_2D+6*i,0) = i0;
427 les_sommets(k*6*size_2D+6*i,1) = i1;
428 les_sommets(k*6*size_2D+6*i,2) = j01;
429
430 les_sommets(k*6*size_2D+6*i+1,0) = j01;
431 les_sommets(k*6*size_2D+6*i+1,1) = i1;
432 les_sommets(k*6*size_2D+6*i+1,2) = i11;
433
434 les_sommets(k*6*size_2D+6*i+2,0) = j01;
435 les_sommets(k*6*size_2D+6*i+2,1) = i11;
436 les_sommets(k*6*size_2D+6*i+2,2) = i1+oldnbsom;
437
438 les_sommets(k*6*size_2D+6*i+3,0) = j01;
439 les_sommets(k*6*size_2D+6*i+3,1) = i0+oldnbsom;
440 les_sommets(k*6*size_2D+6*i+3,2) = i1+oldnbsom;
441
442 les_sommets(k*6*size_2D+6*i+4,0) = j01;
443 les_sommets(k*6*size_2D+6*i+4,1) = i0+oldnbsom;
444 les_sommets(k*6*size_2D+6*i+4,2) = i00;
445
446 les_sommets(k*6*size_2D+6*i+5,0) = j01;
447 les_sommets(k*6*size_2D+6*i+5,1) = i00;
448 les_sommets(k*6*size_2D+6*i+5,2) = i0;
449
450 i0+=oldnbsom;
451 i1+=oldnbsom;
452 }
453 }
454
455 les_faces_bord.typer(Faces::triangle_3D);
456 les_faces_bord.les_sommets().ref(les_sommets);
457 les_faces_bord.voisins().resize(6*size_2D*NZ, 2);
458 les_faces_bord.voisins()=-1;
459}

463void Extruder_en20::extruder_dvt(Domaine& dom, Faces& les_faces, int oldnbsom, int oldsz)
464{
465 Zone& zone = dom.zone(0);
466 const int nbfaces2D = les_faces.nb_faces();
467 IntTab& les_elems=zone.les_elems();
468 {
469 LIST_CURSEUR(Bord) curseur(zone.faces_bord());
470 while(curseur)
471 {
472 Faces& les_faces_bord=curseur->faces();
473 traiter_faces_dvt(les_faces_bord, les_faces, oldnbsom, oldsz, nbfaces2D );
474 ++curseur;
475 }
476 }
477 {
478 LIST_CURSEUR(Raccord) curseur(zone.faces_raccord());
479 while(curseur)
480 {
481 Faces& les_faces_bord=curseur.valeur()->faces();
482 traiter_faces_dvt(les_faces_bord, les_faces, oldnbsom, oldsz, nbfaces2D );
483 ++curseur;
484 }
485 }
486
487 Bord& devant = zone.faces_bord().add(Bord());
488 devant.nommer("devant");
489 Faces& les_faces_dvt=devant.faces();
490 les_faces_dvt.typer(Faces::triangle_3D);
491
492 IntTab som_dvt(oldsz, 3);
493 les_faces_dvt.voisins().resize(oldsz, 2);
494 les_faces_dvt.voisins()=-1;
495
496 Bord& derriere = zone.faces_bord().add(Bord());
497 derriere.nommer("derriere");
498 Faces& les_faces_der=derriere.faces();
499 les_faces_der.typer(Faces::triangle_3D);
500
501 IntTab som_der(oldsz, 3);
502 les_faces_der.voisins().resize(oldsz, 2);
503 les_faces_der.voisins()=-1;
504
505 for (int i=0; i<oldsz; i++)
506 {
507 int i0=les_elems(2*i,0);
508 int i1=les_elems(2*i,1);
509 int i2=les_elems(2*i,2);
510
511 som_dvt(i,0) = i0;
512 som_dvt(i,1) = i1;
513 som_dvt(i,2) = i2;
514
515 som_der(i,0) = i0+oldnbsom*NZ;
516 som_der(i,1) = i1+oldnbsom*NZ;
517 som_der(i,2) = i2+oldnbsom*NZ;
518
519 }
520
521
522 les_faces_dvt.les_sommets().ref(som_dvt);
523 les_faces_der.les_sommets().ref(som_der);
524
525}
