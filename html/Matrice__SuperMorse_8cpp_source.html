24//! Calcul de "resu += MATRICE * x" et d'un produit scalaire<br>(c'est une brique de base pour le gradient conjugue, voir class Solv_GCP)<br>Valeur de retour:<br>partie locale a ce processeur de "(MATRICE * x) scalaire x"<br>(attention: le produit scalaire compte toutes les lignes de la matrice<br>les items communs ne sont pas supprimes !)<br>(attention: c'est different de resu scalaire x !)
25double Matrice_SuperMorse::ajouter_mult_vect_et_prodscal(const DoubleVect& x, DoubleVect& resu) const
26{
27 assert(resu.size() == x.size());
28
29 const int nb_lignes = lignes_non_vides_.size_array();
30 const int *tab_lignes = lignes_non_vides_.addr();
31// Le premier indice de tab1_ qui va nous interesser est le deuxieme du tableau
32// (le premier du tableau est forcement 1)
33 assert(tab1_[0] == 1);
34 const int *tab1_ptr = tab1_.addr() + 1;
35 assert(tab1_.size_array() == nb_lignes + 1);
36 const int *tab2_ptr = tab2_.addr();
37 const double *coeff_ptr = coeff_.addr();
38 const double *x_ptr = x.addr() - 1; //!< decalage de 1 car on va indexer par des indices fortran
39 double *resu_ptr = resu.addr() - 1; //!< idem
40 int n = 1; //!< indice du coeff courant dans tab2 et coeff
41
42 double prod_scal_local = 0;
43
44 for (int i = 0; i < nb_lignes; i++)
45 {
46// Numero de la ligne de la matrice:
47 const int i_ligne = *(tab_lignes++);
48 assert(i_ligne >= 1 && i_ligne <= resu.size_array());
49 double r = 0.;
50// Indice de fin des coeffs de cette ligne dans tab2 et coeff
51 const int n_fin = *(tab1_ptr++);
52 for (; n < n_fin; n++)
53 {
54 const int colonne = *(tab2_ptr++);
55 assert(colonne >= 1 && colonne <= x.size_array());
56 const double coef = *(coeff_ptr++);
57 const double xb = x_ptr[colonne];
58 r += coef * xb;
59 }
60 resu_ptr[i_ligne] += r;
61// attention: on ne veut calculer que le produit scalaire entre x et la partie ajoutee a resu
62// (pas avec les valeurs existantes de resu avant l'appel)
63 prod_scal_local += r * x_ptr[i_ligne];
64 }
65 return prod_scal_local;
66}
