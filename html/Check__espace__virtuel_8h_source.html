27template <typename _TYPE_>
28inline void remplir_items_non_calcules_(TRUSTVect<_TYPE_>& v, _TYPE_ valeur)
29{
30 if (v.get_md_vector().non_nul())
31 {
32 const ArrOfInt& blocs = v.get_md_vector().valeur().get_items_to_compute();
33 const int sz = blocs.size_array() / 2, line_size = v.line_size();
34 int j = 0;
35// Ne pas passer par operator[], sinon plantage si la valeur actuelle est invalide
36 _TYPE_ *ptr = v.addr();
37 for (int i = 0; i < sz; i++)
38 {
39// remplir les elements jusqu'au debut du bloc:
40 const int j_fin = blocs[i*2] * line_size;
41 assert(j >= 0 && j_fin <= v.size_array());
42 for (; j < j_fin; j++) ptr[j] = valeur;
43// Sauter a la fin du bloc
44 j = blocs[i*2+1] * line_size;
45 }
46// Remplir les elements entre la fin du dernier bloc et la fin du vecteur
47 const int j_fin = v.size_array();
48 for (; j < j_fin; j++) ptr[j] = valeur;
49 }
50}

55//! Remplit les "items non calcules" du tableau avec une valeur invalide. Ce sont tous les items qui ne sont pas repertories<br>dans v.get_md_vector().valeurs().get_items_to_compute(). (items non calcules par les operations par defaut sur les vecteurs, en general ce sont les items virtuels)<br>Il est conseille d'appliquer cette methode a la fin des fonctions qui ne renvoient pas un espace virtuel a jour avec declare_espace_virtuel_invalide(...))<br>de sorte a provoquer une erreur si l'espace virtuel est utilise.
56template<typename _TYPE_>
57inline void remplir_items_non_calcules(TRUSTVect<_TYPE_>& v, _TYPE_ valeur = 0)
58{
59 remplir_items_non_calcules_(v, valeur);
60}

63//! en mode comm_check_enabled(), verifie si l'espace virtuel du vecteur est a jour, si ce n'est pas le cas, exit().<br>Ce test n'est fait qu'en mode comm_check_enabled() car il necessite des communications.
64template<typename _TYPE_>
65inline void assert_espace_virtuel_vect(const TRUSTVect<_TYPE_>& v)
66{
67 if (Comm_Group::check_enabled())
68 {
69 if (! check_espace_virtuel_vect(v))
70 {
71 Cerr << "Fatal error in assert_espace_virtuel_vect: virtual space of this vector is not up to date." << finl;
72 Process::barrier();
73 Process::exit();
74 }
75 }
76}
