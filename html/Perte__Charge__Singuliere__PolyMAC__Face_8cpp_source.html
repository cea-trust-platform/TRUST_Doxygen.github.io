65void Perte_Charge_Singuliere_PolyMAC_Face::remplir_num_faces(Entree& s)
66{
67 const Domaine& le_domaine = equation().probleme().domaine();
68 const Zone_Poly_base& zone_poly = ref_cast(Zone_Poly_base,equation().zone_dis().valeur());
69 int taille_bloc = zone_poly.nb_elem();
70 num_faces.resize(taille_bloc);
71 lire_surfaces(s,le_domaine,zone_poly,num_faces, sgn);
72// int nfac_tot = mp_sum(num_faces.size());
73 int nfac_max = (int)mp_max(num_faces.size()); //!< not to count several (number of processes) times the same face
74
75 if(je_suis_maitre() && nfac_max == 0)
76 {
77 Cerr << "Error when defining the surface plane for the singular porosity :" << finl;
78 Cerr << "No mesh faces has been found for the surface plane." << finl;
79 Cerr << "Check the coordinate of the surface plane which should match mesh coordinates." << finl;
80 exit();
81 }
82}

84void Perte_Charge_Singuliere_PolyMAC_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
85{
86 const Zone_Poly_base& zone = ref_cast(Zone_Poly_base, equation().zone_dis().valeur());
87 const IntTab& f_e = zone.face_voisins(), &fcl = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur()).fcl();
88 const std::string& nom_inco = equation().inconnue().le_nom().getString();
89 if (!matrices.count(nom_inco) || !sub_type(Zone_PolyMAC_P0, zone)) return;
90 Matrice_Morse& mat = *matrices.at(nom_inco), mat2;
91 int i, j, e, f, n, N = equation().inconnue().valeurs().line_size(), d, D = dimension, nf_tot = zone.nb_faces_tot();
92 DoubleTrav aar_f(N); //alpha * alpha * rho a chaque face
93 IntTrav stencil(0, 2);
94 stencil.set_smart_resize(1);
95
96 for (i = 0; i < num_faces.size(); i++)
97 if ((f = num_faces(i)) < zone.nb_faces() && fcl(f, 0) < 2)
98 for (j = 0; j < 2 && (e = f_e(f, j)) >= 0; j++) //elem amont / aval si PolyMAC V2
99 if (e < zone.nb_elem())
100 for (d = 0; d < D; d++)
101 for (n = 0; n < N; n++) stencil.append_line(N * (nf_tot + D * e + d) + n, N * f + n);
102 tableau_trier_retirer_doublons(stencil);
103 Matrix_tools::allocate_morse_matrix(equation().inconnue().valeurs().size_totale(), equation().inconnue().valeurs().size_totale(), stencil, mat2);
104 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
105}

107void Perte_Charge_Singuliere_PolyMAC_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
108{
109 const Zone_Poly_base& zone = ref_cast(Zone_Poly_base, equation().zone_dis().valeur());
110// const DoubleVect& volumes_entrelaces = zone_PolyMAC.volumes_entrelaces();
111 const DoubleVect& pf = zone.porosite_face(), &fs = zone.face_surfaces(), &vf = zone.volumes_entrelaces();
112 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
113 const DoubleTab& vit = la_vitesse->valeurs(), &nf = zone.face_normales(), &vfd = zone.volumes_entrelaces_dir(), &xv = zone.xv(), &xp = zone.xp(),
114 *alpha = pbm ? &pbm->eq_masse.inconnue().passe() : NULL, *a_r = pbm ? &pbm->eq_masse.champ_conserve().passe() : NULL;
115 const IntTab& f_e = zone.face_voisins(), &fcl = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur()).fcl();
116 const std::string& nom_inco = equation().inconnue().le_nom().getString();
117 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
118 int i, j, e, f, n, N = equation().inconnue().valeurs().line_size(), poly_v2 = sub_type(Zone_PolyMAC_P0, zone), semi = semi_impl.count(nom_inco), d, D = dimension, nf_tot = zone.nb_faces_tot();
119 DoubleTrav aar_f(N); //alpha * alpha * rho a chaque face
120 for (i = 0; i < num_faces.size(); i++)
121 if ((f = num_faces(i)) < zone.nb_faces())
122 {
123 double fac = (direction_perte_charge() < 0 ? fs(f) : std::fabs(nf(f,direction_perte_charge()))) * pf(f) * K();
124 if (pbm)
125 for (aar_f = 0, j = 0; j < 2 && (e = f_e(f, j)) >= 0; j++)
126 for (n = 0; n < N; n++) aar_f(n) += vfd(f, j) / vf(f) * (*a_r)(e, n) * (*alpha)(e, n);
127 else aar_f = 1;
128
129 if (!poly_v2 || fcl(f, 0) < 2) //contrib a la face : sauf si face de Dirichlet/Neumann en V2
130 {
131 for (n = 0; n < N; n++) secmem(f, n) -= 0.5 * fac * aar_f(n) * vit(f, n) * std::fabs(vit(f, n));
132 if (mat)
133 for (n = 0; n < N; n++) (*mat)(N * f + n, N * f + n) += fac * aar_f(n) * std::fabs(vit(f, n));
134 }
135 if (poly_v2)
136 for (j = 0; j < 2 && (e = f_e(f, j)) >= 0; j++)
137 if (e < zone.nb_elem()) //elem amont / aval si PolyMAC V2
138 {
139 for (d = 0; d < D; d++)
140 for (n = 0; n < N; n++) /* second membre */
141 secmem(nf_tot + D * e + d, n) += (j ? -1 : 1) * fs(f) * (xv(f, d) - xp(e, d)) * 0.5 * fac * aar_f(n) * vit(f, n) * std::fabs(vit(f, n));
142 if (mat && !semi && fcl(f, 0) < 2)
143 for (d = 0; d < D; d++)
144 for (n = 0; n < N; n++) /* derivee (pas possible en semi-implicite) */
145 (*mat)(N * (nf_tot + D * e + d) + n, N * f + n) -= (j ? -1 : 1) * fs(f) * (xv(f, d) - xp(e, d)) * fac * aar_f(n) * std::fabs(vit(f, n));
146 }
147 }
