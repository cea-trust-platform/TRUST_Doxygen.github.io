186//! Fonction outil utilisee dans calculer_Ec. Calcule la somme<br>de 0.5*v^2*rho*volumes_entrelaces.
187static double trait_part_calculer_ec_faces(const int face_debut,
188 const int nb_faces,
189 const int frontiere,
190 const DoubleTab& vitesse,
191 const DoubleVect& volumes_entrelaces,
192 const DoubleTab& xv,
193 const DoubleTab& masse_volumique,
194 const ArrOfDouble& translation,
195 const ArrOfDouble& rotation,
196 const int repere_mobile_,
197 const ArrOfInt& faces_doubles
198 )
199{
200 const int face_fin = face_debut + nb_faces;
201 double ec = 0.;
202 double rho = 0.;
203 const int nb_dim_1 = (vitesse.line_size() == 1);
204 const int dim = Objet_U::dimension;
205 ArrOfDouble ve(Objet_U::dimension);
206 for (int face = face_debut; face < face_fin; face++)
207 {
208// Calcul de la vitesse d'entrainement
209 if (repere_mobile_)
210 {
211 ve[0]=translation[0];
212 ve[1]=translation[1];
213 if (Objet_U::dimension==3)
214 {
215 ve[2]=translation[2];
216 ve[0]+=rotation[1]*xv(face,2)-rotation[2]*xv(face,1);
217 ve[1]+=rotation[2]*xv(face,0)-rotation[0]*xv(face,2);
218 ve[2]+=rotation[0]*xv(face,1)-rotation[1]*xv(face,0);
219 }
220 }
221 else
222 ve=0;
223
224 double v2;
225 double volume;
226 if (nb_dim_1)
227 {
228// Une composante de vitesse a la face (VDF)
229 const double v = vitesse(face);
230 if (repere_mobile_)
231 {
232 Cerr << "Le codage de l'energie cinetique calculee dans un repere fixe" <<finl;
233 Cerr << "n'est pas fait en VDF." << finl;
234 Process::exit(); //!< En effet probleme de conception, il faudrait avoir l'orientation des faces VDF
235 }
236 v2 = v * v;
237// En VDF, sur les frontieres, on ne prend que le 1/2 volume entrelace
238 volume = (frontiere ? 0.5 : 1) * volumes_entrelaces(face);
239 }
240 else
241 {
242// Deux ou trois composantes (VEFP1B)
243 v2 = 0.;
244 for (int i = 0; i < dim; i++)
245 {
246 const double v_i = vitesse(face, i);
247 v2 += (v_i + ve[i]) * (v_i + ve[i]);
248 }
249// En VEF, cela est incorrect, il faudrait les volumes etendus:
250 volume = volumes_entrelaces(face);
251 }
252 const int k = (masse_volumique.dimension(0)==1) ? 0 : face;
253 rho = masse_volumique(k, 0);
254 double contribution = (faces_doubles[face]==1) ? 0.5 : 1 ;
255 ec += contribution * 0.5 * v2 * volume * rho;
256 }
257 return ec;
258}

269//! Meme methode de calcul en VDF et en VEF.<br>Si aucun champ de masse volumique n'a ete associe, on calcule<br>INTEGRALE de 1/2 * u^2 sur le domaine.<br>Si un champ de masse volumique a ete associe, il doit etre<br>de type "champ aux faces". On calcule INTEGRALE de 1/2 * rho * u^2.<br>rho est un champ P0 aux elements, u est P0 sur les volumes entrelaces<br>Valeur de retour:<br>La somme sur tous les processeurs de l'integrale de 1/2*rho*u*u.
270void Traitement_particulier_NS_EC::calculer_Ec(double& energie_cinetique)
271{
272
273 const Zone_dis_base& zdisbase = mon_equation->inconnue().zone_dis_base();
274 const Zone_VF& zone_VF = ref_cast(Zone_VF, zdisbase);
275 const DoubleVect& volumes_entrelaces = zone_VF.volumes_entrelaces();
276 const DoubleTab& xv = zone_VF.xv();
277 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
278 REF(ArrOfDouble) translation(xv);
279 REF(ArrOfDouble) rotation(xv);
280 REF(DoubleTab) rho(xv);
281 DoubleVect rotation_nulle(dimension);
282 rotation_nulle=0;
283 if (repere_mobile_)
284 {
285 int ok=0;
286// Verification de l'existence d'un terme source Acceleration dans Navier Stokes
287 const Sources& les_sources=mon_equation->sources();
288 CONST_LIST_CURSEUR(Source) curseur(les_sources);
289 while(curseur)
290 {
291 if (sub_type(Terme_Source_Acceleration,curseur->valeur()))
292 {
293 const Terme_Source_Acceleration& terme_source_acceleration=ref_cast(Terme_Source_Acceleration,curseur->valeur());
294// Verification que les vitesses de translation du repere
295// mobiles sont bien definies et association des tableaux translation et
296// eventuellement rotation
297 if (terme_source_acceleration.champ_vitesse().non_nul())
298 {
299 translation=terme_source_acceleration.champ_vitesse().valeurs();
300 if (terme_source_acceleration.omega().non_nul())
301 rotation=terme_source_acceleration.omega().valeurs();
302 else
303 rotation=rotation_nulle;
304 ok=1;
305 }
306 }
307 ++curseur;
308 }
309 if (!ok)
310 {
311 Cerr << "Vous ne pouvez calculer l'energie cinetique dans un repere fixe" << finl;
312 Cerr << "que si le repere de calcul est mobile, c'est a dire que vous avez" << finl;
313 Cerr << "defini un terme source d'acceleration dans l'equation Navier Stokes." << finl;
314 Cerr << "Ou bien il manque dans ce terme source la definition de la vitesse" << finl;
315 Cerr << "du repere mobile dans le repere fixe. Modifier votre jeu de donnees." << finl;
316 exit();
317 }
318 }
319 double ec = 0.;
320
321 if (has_champ_masse_volumique())
322 {
323 const Champ_base& champ_rho = get_champ_masse_volumique();
324 rho = champ_rho.valeurs();
325 if (rho->dimension(0) != zone_VF.nb_faces() || rho->line_size() != 1)
326 {
327 Cerr << "Erreur dans Traitement_particulier_NS_EC::calculer_Ec" << finl;
328 Cerr << "le champ de masse volumique n'est pas un champ scalaire aux faces" << finl;
329 Process::exit();
330 }
331 }
332 else
333 {
334 rho = mon_equation->milieu().masse_volumique().valeurs();
335 }
336 const int nb_front = zone_VF.nb_front_Cl();
337 const ArrOfInt& faces_doubles = zone_VF.faces_doubles();
338// Faces de bord
339 for (int i = 0; i < nb_front; i++)
340 {
341 const Frontiere& fr = zone_VF.front_VF(i).frontiere();
342 const int debut = fr.num_premiere_face();
343 const int nb_faces = fr.nb_faces();
344 ec += trait_part_calculer_ec_faces(debut, nb_faces, 1,
345 vitesse, volumes_entrelaces, xv, rho, translation, rotation, repere_mobile_, faces_doubles);
346 }
347// Faces a l'interieur du domaine (y compris les faces de joint)
348 {
349 const int debut = zone_VF.premiere_face_int();
350 const int nb_faces = zone_VF.nb_faces_internes();
351 ec += trait_part_calculer_ec_faces(debut, nb_faces, 0,
352 vitesse, volumes_entrelaces, xv, rho, translation, rotation, repere_mobile_, faces_doubles);
353 }
354 ec = Process::mp_sum(ec);
355 energie_cinetique = ec;
356}
