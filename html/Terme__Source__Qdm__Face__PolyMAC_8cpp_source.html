49void Terme_Source_Qdm_Face_PolyMAC::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
50{
51 const Zone_Poly_base& zone = ref_cast(Zone_Poly_base, equation().zone_dis().valeur());
52 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur());
53 const DoubleTab& vals = la_source->valeurs(), &vfd = zone.volumes_entrelaces_dir(),
54 &rho = equation().milieu().masse_volumique().passe(), &nf = zone.face_normales(),
55 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
56 const DoubleVect& pe = zone.porosite_elem(), &ve = zone.volumes(), &pf = zone.porosite_face(), &vf = zone.volumes_entrelaces(), &fs = zone.face_surfaces();
57 const IntTab& f_e = zone.face_voisins(), &fcl = ch.fcl();
58 int e, f, i, cS = (vals.dimension_tot(0) == 1), cR = (rho.dimension_tot(0) == 1), nf_tot = zone.nb_faces_tot(),
59 n, N = equation().inconnue().valeurs().line_size(), d, D = dimension, calc_cl = !sub_type(Zone_PolyMAC_P0, zone); //en PolyMAC V1, on calcule aux CL
60
61 /* contributions aux faces (par chaque voisin), aux elems */
62 DoubleTrav a_f(N), rho_f(N), val_f(N), rho_m(2);
63 for (a_f = 1, f = 0; f < zone.nb_faces(); f++)
64 if (!fcl(f, 0)) //face interne
65 {
66 if (1)
67 {
68 for (i = 0; i < 2; i++)
69 for (e = f_e(f, i), n = 0; n < N; n++)
70 for (d = 0; d < D; d++)
71 secmem(f, n) += vfd(f, i) * pf(f) * (alp ? (*alp)(e, n) * rho(!cR * e, n) : 1) * nf(f, d) / fs(f) * vals(!cS * e, N * d + n);
72 }
73
74 if (0)
75 {
76 if (alp)
77 for (a_f = 0, i = 0; i < 2; i++)
78 for (e = f_e(f, i), n = 0; n < N; n++) a_f(n) += vfd(f, i) / vf(f) * (*alp)(e, n);
79 for (rho_m = 0, i = 0; i < 2; i++)
80 for (e = f_e(f, i), n = 0; n < N; n++) rho_m(i) += (alp ? (*alp)(e, n) : 1) * rho(!cR * e, n);
81 for (i = 0; i < 2; i++)
82 for (e = f_e(f, i), n = 0; n < N; n++)
83 {
84 double vnf = 0;
85 for (d = 0; d < D; d++) vnf += nf(f, d) / fs(f) * vals(!cS * e, N * d + n);
86 int strat = (i ? 1 : -1) * (rho_m(i) - rho(!cR * e, n)) * vnf > 0;
87 double R = alp && strat ? ((*alp)(e, n) < 1e-4 ? 1 : 0) /* min(max(1 - (*alp)(e, n) / 1e-4, 0.), 1.) */ : 0;
88 secmem(f, n) += vfd(f, i) * pf(f) * a_f(n) * (R * rho_m(i) + (1 - R) * rho(!cR * e, n)) * vnf;
89// Cerr << "f " << f << " i " << i << " n " << n << " a " << (*alp)(e, n) << " r " << rho(!cR * e, n) << " R " << R << finl;
90 }
91 }
92 }
93 else if (calc_cl || fcl(f, 0) < 2)
94 for (e = f_e(f, 0), n = 0; n < N; n++)
95 for (d = 0; d < D; d++) //face de bord non imposee -> avec le (alpha rho) de la maille
96 secmem(f, n) += pf(f) * vf(f) * (alp ? (*alp)(e, n) * rho(!cR * e, n) : 1) * nf(f, d) / fs(f) * vals(!cS * e, N * d + n);
97
98 /* en PolyMAC V2 : partie aux elements */
99 if (sub_type(Zone_PolyMAC_P0, zone))
100 for (e = 0; e < zone.nb_elem_tot(); e++)
101 for (d = 0; d < D; d++)
102 for (n = 0; n < N; n++)
103 secmem(nf_tot + D * e + d, n) += pe(e) * ve(e) * (alp ? rho(!cR * e, n) * (*alp)(e, n) : 1) * vals(!cS * e, N * d + n);
104}
