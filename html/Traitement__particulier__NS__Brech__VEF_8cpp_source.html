264void Traitement_particulier_NS_Brech_VEF::post_traitement_particulier_calcul_flux()
265{
266
267 for (int ii =0; ii<nb; ii++)
268 {
269 int taille = 1 ;
270 DoubleTab coord_trace (taille,3) ;
271 DoubleTab Surf_trace(taille,3) ;
272 DoubleTab valeurs_(taille,3) ;
273
274// Modifs VB pour prise en compte rho et calcul du flux enthalpique + Tmoy
275 REF(Champ_base) rch1 ;
276 REF(Champ_Inc_base) l_inco ;
277 const Probleme_base& pb = mon_equation->probleme();
278 /*
279 pb.a_pour_Champ_Inc("Temperature", rch1 ) ;
280 l_inco = ref_cast(Champ_Inc_base, rch1.valeur()) ;
281 const Champ_Inc_base& temp = l_inco.valeur();
282 */
283 rch1 = pb.get_champ("Temperature");
284 l_inco = ref_cast(Champ_Inc_base, rch1.valeur()) ;
285 const Champ_Inc_base& temp = l_inco.valeur();
286
287 DoubleTab temper_(taille,1);
288 DoubleTab rho_(taille,1) ;
289 DoubleTab cp_(taille,1) ;
290// Fin Modifs VB
291//
292 double rad, teta ;
293 double dr = ( r_out(ii) - r_int(ii) ) /delta_r(ii) ;
294 double dteta = 2.*3.14159/delta_teta(ii) ;
295
296 taille = (int)(delta_r(ii)) * (int)(delta_teta(ii)) ;
297 coord_trace.resize(taille,3) ;
298 Surf_trace.resize(taille,3) ;
299 valeurs_.resize(taille,3);
300
301// Modifs VB pour prise en compte rho et calcul du flux enthalpique + Tmoy
302 rho_.resize(taille,1) ;
303 temper_.resize(taille,1);
304 cp_.resize(taille,1);
305// Fin Modifs VB
306
307 double a,b,c ;
308
309// Modifs VB pour calculer le flux sur une face inclinee
310// anglex est l'angle que fait la normale a la face par rapport a l'axe des x (angle aigu)
311 double anglex = acos(C_trans(ii,0)/sqrt( C_trans(ii,0) * C_trans(ii,0)
312 + C_trans(ii,1) * C_trans(ii,1)
313 + C_trans(ii,2) * C_trans(ii,2) )) ;
314// Fin Modifs VB
315
316 int i =-1 ;
317
318 for ( rad = (r_int(ii)+(dr/2.)) ; rad < r_out(ii) ; rad += dr )
319 for (teta = dteta/2. ; teta < 2.*3.14159 ; teta += dteta )
320 {
321 i++ ;
322// Modifs VB pour calculer le flux sur une face inclinee
323// a = 0. ;
324// b = rad * cos(teta) ;
325 a = - rad * cos(teta) * sin(anglex) ;
326 b = rad * cos(teta) * cos(anglex) ;
327// Fin Modifs VB
328 c = rad * sin(teta) ;
329 coord_trace(i,0) = a + R_loc(ii,0) ;
330 coord_trace(i,1) = b + R_loc(ii,1) ;
331 coord_trace(i,2) = c + R_loc(ii,2) ;
332 double ra = (dr/2.+rad) * (dr/2.+rad) ;
333 double ri = (rad - (dr/2.))* (rad - (dr/2.)) ;
334 Surf_trace(i,0) = 3.14159 * (ra -ri) / delta_teta(ii) ;
335 }
336
337 double flux_pos = 0. ;
338 double flux_neg = 0. ;
339 double flux ;
340// Modifs VB pour prise en compte rho et calcul du flux enthalpique + Tmoy
341 double tempmoy = 0. ;
342 double massvol = 0. ;
343 double fluxE_pos = 0. ;
344 double fluxE_neg = 0. ;
345 double fluxE ;
346// fin modifs VB
347
348 const Zone_dis_base& zdis=mon_equation->inconnue().zone_dis_base();
349 const Zone& zone=zdis.zone();
350 IntVect les_polys(coord_trace.dimension(0));
351 zone.chercher_elements(coord_trace, les_polys);
352 mon_equation->inconnue()->valeur_aux_elems(coord_trace, les_polys, valeurs_);
353
354// Modifs VB pour prise en compte rho et calcul du flux enthalpique + Tmoy
355 mon_equation->fluide().masse_volumique()->valeur_aux_elems(coord_trace, les_polys, rho_);
356 mon_equation->fluide().capacite_calorifique()->valeur_aux_elems(coord_trace, les_polys, cp_);
357 temp.valeur_aux_elems(coord_trace, les_polys, temper_);
358// Fin Modifs VB
359
360// On annulle la partie virtuelle
361 for (i=0 ; i<taille; i++)
362 if (les_polys(i)>=zone.nb_elem())
363 {
364 for (int k=0; k<dimension; k++)
365 valeurs_(i,k)=0;
366// Modifs VB pour prise en compte rho et calcul du flux enthalpique + Tmoy
367 rho_(i)=0.;
368 temper_(i)=0.;
369 cp_(i)=0.;
370 }
371// Fin Modifs VB
372
373 for (i=0 ; i<taille; i++ )
374 {
375// Modifs VB pour prise en compte rho et calcul du flux enthalpique + Tmoy
376// flux = valeurs_(i,0)*Surf_trace(i,0) ;
377 massvol += rho_(i) ;
378 tempmoy += temper_(i) ;
379 flux = ( valeurs_(i,0)*cos(anglex) + valeurs_(i,1)*sin(anglex) )
380 * Surf_trace(i,0)*rho_(i) ;
381 fluxE = ( valeurs_(i,0)*cos(anglex) + valeurs_(i,1)*sin(anglex) )
382 * Surf_trace(i,0)*rho_(i)*cp_(i)*temper_(i) ;
383 flux=Process::mp_sum(flux);
384 fluxE=Process::mp_sum(fluxE);
385 if ( flux > 0. ) flux_pos += flux ;
386 if ( flux < 0. ) flux_neg += flux ;
387 if ( fluxE > 0. ) fluxE_pos += fluxE ;
388 if ( fluxE < 0. ) fluxE_neg += fluxE ;
389 }
390
391 tempmoy=Process::mp_sum(tempmoy);
392 massvol=Process::mp_sum(massvol);
393 tempmoy /= taille ;
394 massvol /= taille ;
395
396 if (je_suis_maitre())
397 {
398 Cout << finl ;
399 Cout << " ----------------------- " << finl ;
400 Cout << " Traitement_particulier flux en x (Plan Oyz) a: "<< R_loc(ii,0) << " "
401 << R_loc(ii,1) << " "
402 << R_loc(ii,2) << finl ;
403 Cout << " angle de la normale du plan par rapport a x = " << anglex/3.1415*180. << " (deg) " << finl ;
404 Cout << " flux_pos = " << flux_pos << " flux_neg = " << flux_neg << " (kg/s) " << finl ;
405 Cout << " fluxEnthalpique_pos = " << fluxE_pos << " fluxEnthalpique_neg = " << fluxE_neg << " (Watts) " << finl ;
406 Cout << " masse volumique moyenne sur le plan de coupe : " << massvol << finl ;
407 Cout << " temperature moyenne sur le plan de coupe : " << tempmoy << finl ;
408 }
409// Fin Modifs VB
410 }
411}

430

432 const DoubleTab& temperature = temp.valeurs();
433
434 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
435 const DoubleVect& gravite = mon_equation->fluide().gravite().valeurs();
436 const DoubleVect& beta = mon_equation->fluide().beta_t().valeurs();
437
438 const int nb_faces = zone_VEF.nb_faces();
439
440 DoubleVect P(nb_faces) ;
441 DoubleVect G(nb_faces) ;
442
443 DoubleVect beta_i(nb_faces) ;
444 beta_i = beta(0) ;
445
446 calculer_terme_production_K( zone_VEF, zone_Cl_VEF,
447 P, vitesse);
448
449 calculer_terme_destruction_K( zone_VEF, zone_Cl_VEF,
450 G, temperature,
451 beta_i, gravite) ;
452
453 DoubleVect& richard_loc = ch_ri.valeurs();
454 richard_loc = 0. ;
455
456 for ( int i=0; i<nb_faces; i++ )
457 {
458 if (std::fabs (P[i]) > 1.e-9 )richard_loc [i] = -G[i]/P[i] ;
459 else richard_loc [i] = 0. ;
460 }
461
462 ch_ri.changer_temps(mon_equation->inconnue().temps());
463
464}

527void Traitement_particulier_NS_Brech_VEF::
528calculer_terme_production_K(const Zone_VEF& zone_VEF,const Zone_Cl_VEF& zcl_VEF,
529 DoubleVect& P,
530 const DoubleTab& vit) const
531{
532// P est discretise comme K et Eps i.e au centre des elements
533//
534// P(elem) = -(2/3)*k(i)*div_U(i) + nu_t(i) * F(u,v,w)
535//
536// 2 2 2
537// avec F(u,v,w) = 2[(du/dx) + (dv/dy) + (dw/dz) ] +
538//
539// 2 2 2
540// (du/dy+dv/dx) + (du/dz+dw/dx) + (dw/dy+dv/dz)
541//
542// Rqs: On se place dans le cadre incompressible donc on neglige
543// le terme (2/3)*k(i)*div_U(i)
544
545 P= 0;
546
547// Calcul de F(u,v,w):
548
549// const Zone& zone = zone_VEF.zone();
550 int nb_elem = zone_VEF.nb_elem();
551// const IntTab& elem_faces = zone_VEF.elem_faces();
552 const IntTab& face_voisins = zone_VEF.face_voisins();
553// const int nb_faces = zone_VEF.nb_faces();
554 const DoubleVect& volumes = zone_VEF.volumes();
555// int premiere_face_entier = zone_VEF.premiere_face_int();
556// const IntTab& les_Polys = zone.les_elems();
557
558 int fac=0, poly1, poly2;
559 int nb_faces_ = zone_VEF.nb_faces();
560
561// (du/dx du/dy dv/dx dv/dy ...) pour un poly
562
563//
564// <
565// calcul des gradients; < [ Ujp*np/vol(j) ]
566// j
567//
568 int n_bord;
569
570 DoubleTab gradient_elem(nb_elem,dimension,dimension);
571 Champ_P1NC::calcul_gradient(vit,gradient_elem,zcl_VEF);
572
573// On a les gradient_elem par elements
574//
575
576 double du_dx;
577 double du_dy;
578 double du_dz;
579 double dv_dx;
580 double dv_dy;
581 double dv_dz;
582 double dw_dx;
583 double dw_dy;
584 double dw_dz;
585
586//
587// Calcul des du/dx dv/dy et des derivees croisees sur les faces de chaque elements dans le cas 2D
588//
589
590// Boucle sur les bords pour traiter les faces de bord
591// en distinguant le cas periodicite
592
593 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
594 {
595 const Cond_lim& la_cl = zcl_VEF.les_conditions_limites(n_bord);
596 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
597 int ndeb = le_bord.num_premiere_face();
598 int nfin = ndeb + le_bord.nb_faces();
599
600 if (sub_type(Periodique,la_cl.valeur()))
601 {
602 for (fac=ndeb; fac<nfin; fac++)
603 {
604 poly1 = face_voisins(fac,0);
605 poly2 = face_voisins(fac,1);
606 double a=volumes(poly1)/(volumes(poly1)+volumes(poly2));
607 double b=volumes(poly2)/(volumes(poly1)+volumes(poly2));
608
609 du_dx =a*gradient_elem(poly1,0,0) + b*gradient_elem(poly2,0,0);
610 du_dy =a*gradient_elem(poly1,0,1) + b*gradient_elem(poly2,0,1);

612 dv_dy =a*gradient_elem(poly1,1,1) + b*gradient_elem(poly2,1,1);
613

615

619 {

621 dv_dz=a*gradient_elem(poly1,1,2) + b*gradient_elem(poly2,1,2);

623 dw_dy=a*gradient_elem(poly1,2,1) + b*gradient_elem(poly2,2,1);
624 dw_dz=a*gradient_elem(poly1,2,2) + b*gradient_elem(poly2,2,2);
625
626// Determination du terme de production
627
628 P(fac) = (2*( du_dx*du_dx + dv_dy*dv_dy + dw_dz*dw_dz )
629 + ( (du_dy+dv_dx)*(du_dy+dv_dx)
630 + (du_dz+dw_dx)*(du_dz+dw_dx)
631 + (dw_dy+dv_dz)*(dw_dy+dv_dz)));
632 }
633 }

652 dv_dz=gradient_elem(poly1,1,2);
653 dw_dx=gradient_elem(poly1,2,0);
654 dw_dy=gradient_elem(poly1,2,1);
655 dw_dz=gradient_elem(poly1,2,2);
656
657// Determination du terme de production
658
659 P(fac) = (2*( du_dx*du_dx + dv_dy*dv_dy + dw_dz*dw_dz )
660 + ( (du_dy+dv_dx)*(du_dy+dv_dx)
661 + (du_dz+dw_dx)*(du_dz+dw_dx)
662 + (dw_dy+dv_dz)*(dw_dy+dv_dz)));
663 }
664 }
665 }
666 }

668
669// Traitement des faces internes
670
671 for (; fac<nb_faces_; fac++)
672 {
673 poly1 = face_voisins(fac,0);
674 poly2 = face_voisins(fac,1);
675 double a=volumes(poly1)/(volumes(poly1)+volumes(poly2));
676 double b=volumes(poly2)/(volumes(poly1)+volumes(poly2));
677
678 du_dx=a*gradient_elem(poly1,0,0) + b*gradient_elem(poly2,0,0);
679 du_dy=a*gradient_elem(poly1,0,1) + b*gradient_elem(poly2,0,1);
680 dv_dx=a*gradient_elem(poly1,1,0) + b*gradient_elem(poly2,1,0);
681 dv_dy=a*gradient_elem(poly1,1,1) + b*gradient_elem(poly2,1,1);
682
683// Determination du terme de production
684
685 P(fac) = ( 2*(du_dx*du_dx + dv_dy*dv_dy) + ((du_dy+dv_dx)*(du_dy+dv_dx)));
686
687 if (dimension == 3)
688 {
689 du_dz=a*gradient_elem(poly1,0,2) + b*gradient_elem(poly2,0,2);
690 dv_dz=a*gradient_elem(poly1,1,2) + b*gradient_elem(poly2,1,2);
691 dw_dx=a*gradient_elem(poly1,2,0) + b*gradient_elem(poly2,2,0);
692 dw_dy=a*gradient_elem(poly1,2,1) + b*gradient_elem(poly2,2,1);
693 dw_dz=a*gradient_elem(poly1,2,2) + b*gradient_elem(poly2,2,2);
694
695// Determination du terme de production
696
697 P(fac) = (2*( du_dx*du_dx + dv_dy*dv_dy + dw_dz*dw_dz )
698 + ( (du_dy+dv_dx)*(du_dy+dv_dx)
699 + (du_dz+dw_dx)*(du_dz+dw_dx)
700 + (dw_dy+dv_dz)*(dw_dy+dv_dz)));

706void Traitement_particulier_NS_Brech_VEF::
707calculer_terme_destruction_K(const Zone_VEF& zone_VEF,
708 const Zone_Cl_VEF& zcl_VEF,
709 DoubleVect& G,const DoubleTab& temper,
710 const DoubleVect& beta,const DoubleVect& gravite) const
711{
712//
713// G est discretise comme K et Eps i.e au centre des faces
714//
715// --> ----->
716// G(face) = beta alpha_t(face) G . gradT(face)
717//
718 int nb_elem = zone_VEF.nb_elem();
719 int nb_faces= zone_VEF.nb_faces();
720 DoubleTrav u_teta(nb_faces,dimension);
721 DoubleTrav gradient_elem(nb_elem,dimension);
722 u_teta=0;
723 gradient_elem=0;
724
725// const Zone& la_zone=zone_VEF.zone();
726// int nb_faces_elem = la_zone.nb_faces_elem();
727// const IntTab& elem_faces = zone_VEF.elem_faces();
728 const IntTab& face_voisins = zone_VEF.face_voisins();
729 const DoubleTab& face_normales = zone_VEF.face_normales();
730 int premiere_face_entier = zone_VEF.premiere_face_int();
731 const DoubleVect& volumes = zone_VEF.volumes();
732 int nb_faces_ = zone_VEF.nb_faces();
733 int elem1,elem2,fac;
734
735 DoubleVect coef(Objet_U::dimension);
736// const IntTab& les_elem_faces = zone_VEF.elem_faces();
737
738// Calcul du gradient de temperature :
739
740// On traite les bords
741 int n_bord;
742 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
743 {
744 const Cond_lim& la_cl = zcl_VEF.les_conditions_limites(n_bord);
745 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
746 int ndeb = le_bord.num_premiere_face();
747 int nfin = ndeb + le_bord.nb_faces();
748
749 if (sub_type(Periodique,la_cl.valeur()))
750 {
751 for (fac=ndeb; fac<nfin; fac++)
752 {
753 elem1=face_voisins(fac,0);
754 elem2=face_voisins(fac,1);
755 for (int i=0; i<dimension; i++)
756 {
757 gradient_elem(elem1,i) += 0.5*face_normales(fac,i)*temper(fac);
758 gradient_elem(elem2,i) -= 0.5*face_normales(fac,i)*temper(fac);
759 }
760 }
761 }
762 else
763 {
764 for (int fac2=ndeb; fac2<nfin; fac2++)
765 {
766 elem1=face_voisins(fac2,0);
767 for (int i=0; i<dimension; i++)
768 gradient_elem(elem1,i) += face_normales(fac2,i)*temper(fac2);
769 }
770 }
771 }
772
773// On traite les faces internes
774 for (fac = premiere_face_entier ; fac<nb_faces_; fac++)
775 {
776 elem1=face_voisins(fac,0);
777 elem2=face_voisins(fac,1);
778 for (int i=0; i<dimension; i++)
779 {
780 gradient_elem(elem1,i) += face_normales(fac,i)*temper(fac);
781 gradient_elem(elem2,i) -= face_normales(fac,i)*temper(fac);
782 }
783 }
784
785 for (int elem=0; elem<nb_elem; elem++)
786 for (int i=0; i<dimension; i++)
787 gradient_elem(elem,i) /= volumes(elem);
788
789// Calcul de u_teta :
790
791// On traite les bords
792
793 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
794 {
795 const Cond_lim& la_cl = zcl_VEF.les_conditions_limites(n_bord);
796 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
797 int ndeb = le_bord.num_premiere_face();
798 int nfin = ndeb + le_bord.nb_faces();
799
800 if (sub_type(Periodique,la_cl.valeur()))
801 {
802 for (fac=ndeb; fac<nfin; fac++)
803 {
804 elem1 = face_voisins(fac,0);
805 elem2 = face_voisins(fac,1);
806 double a=volumes(elem1)/(volumes(elem1)+volumes(elem2));
807 double b=volumes(elem2)/(volumes(elem1)+volumes(elem2));
808 for (int i=0; i<dimension; i++)
809 u_teta(fac,i)=a*beta(elem1)*gradient_elem(elem1,i)
810 +b*beta(elem2)*gradient_elem(elem2,i);
811 }
812 }
813 else
814 {
815 for (fac=ndeb; fac< nfin; fac++)
816 {
817 elem1 = face_voisins(fac,0);
818 for (int i=0; i<dimension; i++)
819 u_teta(fac,i)=beta(elem1)*gradient_elem(elem1,i);
820 }
821 }
822 }
823
824 for (; fac<nb_faces_; fac++)
825 {
826 elem1 = face_voisins(fac,0);
827 elem2 = face_voisins(fac,1);
828 double a=volumes(elem1)/(volumes(elem1)+volumes(elem2));
829 double b=volumes(elem2)/(volumes(elem1)+volumes(elem2));
830 for (int i=0; i<dimension; i++)
831 u_teta(fac,i)=a*beta(elem1)*gradient_elem(elem1,i)+b*beta(elem2)*gradient_elem(elem2,i);
832 }
833// -------> ------>
834// Calcul de gravite . u_teta
835//
836 G = 0;
837 for (fac=0; fac< nb_faces_; fac++)
838 {
839 if (dimension == 2)
840 G[fac] = gravite(0)*u_teta(fac,0) + gravite(1)*u_teta(fac,1);
841 else if (dimension == 3)
842 G[fac] = gravite(0)*u_teta(fac,0) + gravite(1)*u_teta(fac,1) + gravite(2)*u_teta(fac,2);
843 }
844}
