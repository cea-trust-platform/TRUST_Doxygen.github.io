44void Op_Moyenne::completer(const Probleme_base& Pb)
45{
46 Nom nom_pour_post("Moyenne_");
47
48 const REF(Champ_Generique_base)& mon_champ=integrale_champ.le_champ();
49 const Noms noms = mon_champ->get_property("nom");
50 nom_pour_post += Motcle(noms[0]);
51
52 integrale_champ->nommer(nom_pour_post);
53// Dimensionnement du champ integrale_champ a la meme taille que mon_champ
54
55 Champ espace_stockage_source;
56 const Champ_base& source = mon_champ->get_champ(espace_stockage_source);
57 const DoubleTab& tab1 = source.valeurs();
58
59// La methode valeurs pour cette classe est surchargee et renvoie
60// integrale_champ->valeurs(). Il n'y a pas de stockage en memoire
61// de la moyenne. On stocke l'integrale en temps et on postraite la moyenne
62// Copie de la structure du vecteur, initialisation a zero.
63 valeurs().copy(tab1, ArrOfDouble::NOCOPY_NOINIT);
64 if (!ch_moyenne_convergee_.non_nul())
65 valeurs() = 0.;
66 int nb_comp = source.nb_comp();
67 integrale_champ->associer_zone_dis_base(Pb.domaine_dis().zone_dis(0).valeur());
68 integrale_champ->fixer_nb_comp(nb_comp);
69 if (nb_comp == 1)
70 integrale_champ->fixer_unite(source.unite());
71 else
72 integrale_champ->fixer_unites(source.unites());
73 integrale_champ->changer_temps(Pb.schema_temps().temps_courant());
74}

76DoubleTab Op_Moyenne::calculer_valeurs() const
77{
78 double dt = dt_integration();
79 DoubleTab moyenne(valeurs());
80 if (dt > 0)
81 moyenne /= dt;
82 return moyenne;
83}
