127//! On agglomere les Zones de meme type
128void Zones::comprimer()
129{
130 if (Process::nproc() > 1)
131 {
132 Cerr << "Error in Zones::comprimer() : compression prohibited in parallel mode" << finl;
133 exit();
134 }
135 Cerr << "Compressing objects of type Zone in progress... " << finl;
136
137 IntVect fait(size());
138 int rang1=0, rang2=0;
139 {
140 LIST_CURSEUR(Zone) curseur(*this);;;
141 while(curseur)
142 {
143 curseur->comprimer();
144 ++curseur;
145 }
146 }
147
148 LIST_CURSEUR(Zone) curseur1(*this);;;
149 while(curseur1)
150 {
151 if (!fait(rang1) )
152 {
153 fait(rang1)=1;
154 Zone& zone1=curseur1.valeur();
155 rang2=rang1;
156 LIST_CURSEUR(Zone) curseur2(curseur1.list());;;
157 while(curseur2)
158 {
159 if (!fait(rang2) )
160 {
161 Zone& zone2=curseur2.valeur();
162 if ((zone2.type_elem()->que_suis_je())
163 ==
164 (zone1.type_elem()->que_suis_je()) )
165 {
166 Nom nom1=zone1.le_nom();
167 Nom nom2=zone2.le_nom();
168 Nom nom=nom1;
169 Cerr << "agglomeration of Zones "<< nom1
170 << " and " << nom2 << finl;;
171 fait(rang2)=1;
172 if( (nom1.longueur()+nom2.longueur()) < 20 )
173 {
174 nom+="_";
175 nom+=nom2;
176 }
177 zone1.nommer(nom);
178 int sz1=zone1.les_elems().dimension(0);
179 int sz2=zone2.les_elems().dimension(0);
180 int nb_ccord=zone1.les_elems().dimension(1);
181 IntTab& elems1=zone1.les_elems();
182 IntTab& elems2=zone2.les_elems();
183 elems1.resize(sz1+sz2, nb_ccord);
184 for(int i=0; i<sz2; i++)
185 for(int j=0; j<nb_ccord; j++)
186 elems1(sz1+i,j)=elems2(i,j);
187 zone2.faces_bord().associer_zone(zone1);
188 zone1.faces_bord().add(zone2.faces_bord());
189 zone2.faces_joint().associer_zone(zone1);
190 zone1.faces_joint().add(zone2.faces_joint());
191 zone2.faces_raccord().associer_zone(zone1);
192 zone1.faces_raccord().add(zone2.faces_raccord());
193 zone2.faces_int().associer_zone(zone1);
194 zone1.faces_int().add(zone2.faces_int());
195 elems2.resize(0);
196 zone1.comprimer();
197 }
198 }
199 ++curseur2;
200 rang2++;
201 }
202 }
203 ++curseur1;
204 ++rang1;
205 }
206 curseur1=*this;
207 while(curseur1)
208 {
209 if (curseur1->les_elems().size()==0)
210 {
211// Cerr << &(curseur1.valeur()) << finl;
212 suppr(curseur1.valeur());
213 curseur1=*this;
214 }
215 else
216 ++curseur1;
217 }
218 (*this).valeur().invalide_octree();
219 Cerr << "Zones::comprimer() - End" << finl;
220
221}

238//! Merge all the zones of my processor in a single one<br>is not tested when the zones are dispatched on multiple processors!
239void Zones::merge()
240{
241 LIST_CURSEUR(Zone) curseur(*this);;;
242 Zone& zone1=curseur.valeur();
243
244 LIST_CURSEUR(Zone) curseur2(curseur.list());;;
245 while(!curseur2.list().est_dernier())
246 {
247 Zone& zone2=curseur2.list().suivant().valeur();
248 assert(zone1.le_nom() == zone2.le_nom());
249 int sz1=zone1.les_elems().dimension(0);
250 int sz2=zone2.les_elems().dimension(0);
251 int nb_coord=zone1.les_elems().dimension(1);
252 IntTab& elems1=zone1.les_elems();
253 IntTab& elems2=zone2.les_elems();
254 elems1.resize(sz1+sz2, nb_coord);
255 for(int i=0; i<sz2; i++)
256 for(int j=0; j<nb_coord; j++)
257 elems1(sz1+i,j)=elems2(i,j);
258 zone2.faces_bord().associer_zone(zone1);
259 zone1.faces_bord().add(zone2.faces_bord());
260 zone2.faces_joint().associer_zone(zone1);
261 zone1.faces_joint().add(zone2.faces_joint());
262 zone2.faces_raccord().associer_zone(zone1);
263 zone1.faces_raccord().add(zone2.faces_raccord());
264 zone2.faces_int().associer_zone(zone1);
265 zone1.faces_int().add(zone2.faces_int());
266 elems2.resize(0);
267 zone1.correct_type_of_borders_after_merge();
268 zone1.comprimer();
269 zone1.comprimer_joints();
270 curseur2.list().supprimer();
271
272 }
