19void QuickSort(const DoubleVect& a,int g, int d, ArrOfInt& c)
20{
21 int t,i,j;
22 double v;
23 if (g < d)
24 {
25 if(a[c[g]]>a[c[d]])
26 {
27 t=c[g];
28 c[g]=c[d];
29 c[d]=t;
30 }
31 v = a[c[d]];
32 i = g-1;
33 j = d;
34 do
35 {
36 do i++;
37 while (a[c[i]] < v);
38 do j--;
39 while (a[c[j]] > v);
40 t = c[i];
41 c[i] = c[j];
42 c[j] = t;
43 }
44 while (j > i);
45 c[j] = c[i];
46 c[i] = c[d];
47 c[d] = t;
48 QuickSort (a, g, i-1,c);
49 QuickSort (a, i+1, d,c);
50 }
51}
