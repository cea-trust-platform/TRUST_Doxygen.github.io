257int Pb_MED::comprend_champ(const Motcle& mot) const
258{
259// Cerr<<" Pb_MED::comprend_champ"<<mot <<" " <<nomschampmed<<finl;
260 int nbchampmed=nomschampmed.size();

280void Pb_MED::creer_champ(const Motcle& motlu)
281{
282 Cerr<<"Pb_MED::creer_champ "<< motlu<<finl;
283 Noms liste_noms;
284 get_noms_champs_postraitables(liste_noms);
285 Champ_Fonc toto; //!< on ajoute toto et on le type apres pour eviter des copies qui ne marchent pas ...
286 Champ_Fonc& le_ch_fonc= champs_fonc_post.add(toto);
287 le_ch_fonc.typer("Champ_Fonc_MED");
288 int nbchampmed=nomschampmed.size();
289
290 Nom localisation;
291 Motcle es;
292 le_ch_fonc.valeur().nommer(motlu);
293
294 int flag=0;
295 Nom test("_");
296 test+=domaine().le_nom();
297
298 for (int ch=0; ch<nbchampmed; ch++)
299 {
300 Nom pp(nomschampmed[ch]);
301 pp.prefix(test);
302// Cerr<<test<<" ICI "<<pp<<" "<<motlu<<finl;
303 if (motlu==pp)
304 {
305 es=pp;
306 es.prefix("_SOM");
307 if (es!=pp)
308 localisation="som";
309 else
310 {
311 es.prefix("_ELEM");
312 if (es!=pp)
313 localisation="elem";
314 }
315 Cerr<<"One wishes to read the field "<<es<<" choice "<<localisation<<" readen word in the data set "<<motlu<<finl;
316 le_ch_fonc.valeur().nommer(es);
317 flag=1;
318 }
319 }
320 Champ_Fonc_MED& chmed=ref_cast(Champ_Fonc_MED,le_ch_fonc.valeur());
321 const Domaine& dom_med = domaine();
322 chmed.creer(nom_fic,dom_med,localisation, temps_sauv_);
323
324 if (flag)
325 {
326 chmed.le_champ().nommer(le_ch_fonc.valeur().le_nom());
327 chmed.le_champ().corriger_unite_nom_compo();
328 chmed.le_champ().nommer(motlu);
329 }
330
331 le_ch_fonc.valeur().nommer(motlu);
332 chmed.le_champ().nommer(motlu);
333 if (!le_ch_fonc.non_nul())
334 {
335 Cerr << "Error when reading data for post-processing" << finl;
336 Cerr << "The creation of the field " << motlu << " has failed." << finl;
337 exit();
338 }

341const Champ_base& Pb_MED::get_champ(const Motcle& un_nom) const
342{
343 REF(Champ_base) ref_champ;
344
345 double temps_courant = schema_temps().temps_courant();
346
347 CONST_LIST_CURSEUR(Champ_Fonc) curseur = champs_fonc_post ;
348 Motcle nom_champ;
349 int ok_post=1;
350 while(curseur)
351 {
352 Champ_Fonc_MED& ch_med = ref_cast_non_const(Champ_Fonc_MED,curseur.valeur().valeur());
353 nom_champ = Motcle(curseur.valeur().le_nom());
354// nom_champ=ch_med.le_champ().le_nom();
355// Cerr<<(int)__LINE__<<nom_champ<<finl;
356// nom_champ = Motcle(curseur.valeur().valeur().le_nom());
357 if ((nom_champ==un_nom) && (ok_post==1))
358 {
359 ref_champ = curseur.valeur();
360 if (ch_med.temps()!=temps_courant)
361 ch_med.mettre_a_jour(temps_courant);
362 return ch_med.le_champ();
363// return ref_champ.valeur();
364 }
365 else
366 {
367 int nb_composantes = curseur.valeur().valeur().nb_comp();
368 for (int i=0; i<nb_composantes; i++)
369 {
370 nom_champ = Motcle(curseur.valeur().valeur().nom_compo(i));
371 if ((nom_champ==un_nom) && (ok_post==1))
372 {
373 ref_champ = curseur.valeur();
374 if (ch_med.temps()!=temps_courant)
375 ch_med.mettre_a_jour(temps_courant);
376// return ref_champ.valeur();
377 return ch_med.le_champ();
378 }
379 }
380 }
381 ++curseur;
382 }
383
384 Cerr<<"The indicated name "<<un_nom<<" do not correspond to a field understood by the problem"<<finl;
385 Cerr<<"Check the field name to indicate in the post-processing set"<<finl;
386 exit();
387
388// Pour compilation
389 return ref_champ;
390}
