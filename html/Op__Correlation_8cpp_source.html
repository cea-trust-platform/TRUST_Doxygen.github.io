45void Op_Correlation::completer(const Probleme_base& Pb)
46{
47 const REF(Champ_Generique_base)& a = la_moyenne_a().integrale().le_champ();
48 const REF(Champ_Generique_base)& b = la_moyenne_b().integrale().le_champ();
49 integrale_tps_a_ = la_moyenne_a().integrale();
50 integrale_tps_b_ = la_moyenne_b().integrale();
51
52// On recupere la zone discretisee
53 const Zone_dis_base& zone = a->get_ref_zone_dis_base();
54
55// Dimensionnement du champ integrale_champ a la meme taille que mon_champ
56 const DoubleTab& tab_a = valeurs_a();
57 const DoubleTab& tab_b = valeurs_b();
58
59 Champ espace_stockage_source_a, espace_stockage_source_b;
60 const Champ_base& source_a = a->get_champ(espace_stockage_source_a);
61 const Champ_base& source_b = b->get_champ(espace_stockage_source_b);
62
63 const int nb_compo_a = source_a.nb_comp();
64 const int nb_compo_b = source_b.nb_comp();
65
66// Pour utiliser nom_cible uniquement si a et b sont des Champ_Generique_refChamp
67// pour pouvoir reproduire le nom des variables dans le lml
68 Noms noms_a, noms_b;
69
70 if (!(sub_type(Champ_Generique_refChamp,a.valeur()) && (sub_type(Champ_Generique_refChamp,b.valeur()))))
71 {
72 noms_a = a->get_property("nom");
73 noms_b = b->get_property("nom");
74 }
75 else
76 {
77 noms_a = a->get_property("nom_cible");
78 noms_b = b->get_property("nom_cible");
79 }
80
81 const Nom nom_a = noms_a[0];
82 const Nom nom_b = noms_b[0];
83
84// Cas particulier si a et b sont des Champ_Generique_refChamp
85// pour pouvoir reproduire le nom des variables dans le lml
86 Noms compo_a, compo_b;
87
88 if (!(sub_type(Champ_Generique_refChamp,a.valeur()) && (sub_type(Champ_Generique_refChamp,b.valeur()))))
89 {
90 compo_a = a->get_property("composantes");
91 compo_b = b->get_property("composantes");
92 }
93 else
94 {
95 compo_a = source_a.noms_compo();
96 compo_b = source_b.noms_compo();
97 }
98
99 const Noms unites_a = a->get_property("unites");
100 const Noms unites_b = b->get_property("unites");
101
102 int nb_valeurs_nodales = -1;
103
104// Note B.M.: integrale_tps_ab_ est deja type dans Correlation::associer
105// En revanche fixer_nb_valeurs_nodales n'est pas fait.
106
107 if ( ( tab_a.get_md_vector() != tab_b.get_md_vector() ) ||
108 ( tab_a.dimension_tot(0) != tab_b.dimension_tot(0) ) ||
109 ( nb_compo_a > 1 && ( tab_a.nb_dim() == 1 || tab_a.dimension(1) != nb_compo_a ) ) ||
110 ( nb_compo_b > 1 && ( tab_b.nb_dim() == 1 || tab_b.dimension(1) != nb_compo_b ) ) )
111 {
112// Support different pour les champs a et b ou pour leurs composantes
113// Alors on ramene aux elements en creant un Champ_Fonc_PO_VDF ou VEF
114 integrale_tps_ab_.support_different()=1;
115 Nom Champ_Fonc_P0="Champ_Fonc_P0_";
116 Champ_Fonc_P0+=Pb.discretisation().que_suis_je().substr_old(1,3);
117 integrale_tps_ab_.typer(Champ_Fonc_P0);
118 nb_valeurs_nodales = zone.nb_elem();
119 }
120 else
121 {
122// Meme support pour les champs a et b
123 integrale_tps_ab_.support_different()=0;
124 if (tab_a.size_reelle_ok())
125 nb_valeurs_nodales = tab_a.dimension(0);
126 else
127 nb_valeurs_nodales = -1; //!< B.M.: quoi donner a fixer_nb_valeurs_nodales ???
128 }
129 int nb_comp= nb_compo_a * nb_compo_b;
130 DoubleTab& val = valeurs_ab();
131 Noms noms_composantes;
132 if (nb_compo_a == 1)
133 if (nb_compo_b == 1)
134 {
135// A et B champs scalaires
136 }
137 else
138 {
139// A champ scalaire, B champ vectoriel
140 noms_composantes.dimensionner(nb_comp);
141 Nom debut("Correlation_");
142 debut += nom_a+"_";
143 for (int i=0; i<nb_comp; i++)
144 noms_composantes[i]=debut+compo_b[i];
145 }
146 else if (nb_compo_b == 1)
147 {
148// A champ vectoriel, B champ scalaire
149 noms_composantes.dimensionner(nb_comp);
150 Nom debut("Correlation_");
151 debut+=nom_b+"_";
152 for (int i=0; i<nb_comp; i++)
153 noms_composantes[i]=debut+compo_a[i];
154 }
155 else
156 {
157// A et B champs vectoriels
158 noms_composantes.dimensionner(nb_comp);
159 Nom debut("Correlation_");
160 for (int i=0; i<nb_compo_a; i++)
161 for (int j=0; j<nb_compo_b; j++)
162 noms_composantes[i*nb_compo_b+j]=debut+compo_a[i]+"_"+compo_b[j];
163 }
164
165 if (nb_comp > 1)
166 val.resize(0, nb_comp);
167
168 Nom nom_pour_post("Correlation_");
169 nom_pour_post+=nom_a+"_"+nom_b;
170 integrale_tps_ab_->nommer(nom_pour_post);
171
172 integrale_tps_ab_.associer_zone_dis_base(zone);
173 integrale_tps_ab_->fixer_nb_comp(nb_comp);
174
175// BM: le parametre de fixer_nb_valeurs_nodales est inutile,
176// le champ sait determiner tout seul combien de ddl il a, mais
177// il y a un test sur le parametre: il faut que la valeur soit la bonne.
178 integrale_tps_ab_.fixer_nb_valeurs_nodales(nb_valeurs_nodales);
179
180// On fixe les unites
181 Nom unite(unites_a[0]);
182 unite+=".";
183 unite+=unites_b[0];
184
185 if (nb_comp == 1)
186 integrale_tps_ab_->fixer_unite(unite);
187 else
188 {
189 Noms unites(nb_comp);
190 for (int i=0; i<unites.size(); i++)
191 unites[i]=unite;
192 integrale_tps_ab_->fixer_noms_compo(noms_composantes);
193 integrale_tps_ab_->fixer_unites(unites);
194 }
195 integrale_tps_ab_.changer_temps(Pb.schema_temps().temps_courant());
196}

198DoubleTab Op_Correlation::calculer_valeurs() const
199{
200 Integrale_tps_produit_champs correlation(integrale_tps_ab_);
201 const double dt_ab = dt_integration_ab();
202 if ( dt_ab > 0 )
203 {
204// On calcule Moyenne(a'b')=Moyenne(ab)-Moyenne(a)*Moyenne(b)
205 correlation.valeurs() /= dt_ab;
206 const double dt_a = dt_integration_a();
207 const double dt_b = dt_integration_b();
208 assert(est_egal(dt_a,dt_ab));
209 assert(est_egal(dt_b,dt_ab));
210 correlation.ajoute_produit_tensoriel(-1/(dt_a*dt_b),integrale_tps_a_.valeur().valeur(),integrale_tps_b_.valeur().valeur());
211 }
212 return correlation.valeurs();
213}
