75void Op_Grad_PolyMAC_P0_Face::update_grad(int full_stencil) const
76{
77 const Zone_PolyMAC_P0& zone = ref_zone.valeur();
78 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
79 const DoubleTab& press = le_champ_inco.non_nul() ? le_champ_inco->valeurs() : ref_cast(Navier_Stokes_std, equation()).pression().valeurs(),
80 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
81 const int M = press.line_size();
82 double t_past = equation().inconnue().valeur().recuperer_temps_passe();

85 /* gradient */
86 zone.fgrad(M, 1, ref_zcl->les_conditions_limites(), ch.fcl(), NULL, NULL, 1, full_stencil, fgrad_d, fgrad_e, fgrad_c);
87 last_gradp_ = t_past;

90void Op_Grad_PolyMAC_P0_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
91{
92 const Zone_PolyMAC_P0& zone = ref_zone.valeur();
93 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
94 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces(), &fcl = ch.fcl();
95 const DoubleTab& nf = zone.face_normales(), &xp = zone.xp(), &xv = zone.xv();
96 const DoubleVect& fs = zone.face_surfaces(), &ve = zone.volumes();
97 int i, j, e, eb, f, ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), d, D = dimension, n, N = ch.valeurs().line_size(),
98 m, M = (le_champ_inco.non_nul() ? le_champ_inco->valeurs() : ref_cast(Navier_Stokes_std, equation()).pression().valeurs()).line_size();
99 update_grad(sub_type(Pb_Multiphase, equation().probleme())); //provoque le calcul du gradient
100
101 IntTrav sten_p(0, 2), sten_v(0, 2); //stencils (NS, pression), (NS, vitesse)
102 sten_p.set_smart_resize(1), sten_v.set_smart_resize(1);
103
104 const std::string& nom_inc = ch.le_nom().getString();
105 Matrice_Morse *mat_p = matrices["pression"], *mat_v = !semi_impl.count(nom_inc) && matrices.count(nom_inc) ? matrices.at(nom_inc) : NULL, mat2_p, mat2_v;
106 std::map<int, std::set<int>> dpb_v, dgp_pb; //dependances vitesses -(dpb_v)-> pressions au bord -(dgp_pb)-> gradient
107 if (mat_v)
108 for (f = 0; f < zone.nb_faces_tot(); f++)
109 if (fcl(f, 0) > 1)
110 for (e = f_e(f, 0), n = 0, m = 0; n < N; n++, m += (M > 1))
111 for (d = 0, i = nf_tot + D * e; d < D; d++, i++)
112 if (std::fabs(nf(f, d)) > 1e-6 * fs(f))
113 for (dpb_v[M * f + m].insert(N * i + n), j = mat_v->get_tab1()(N * i + n) - 1; j < mat_v->get_tab1()(N * i + n + 1) - 1; j++)
114 dpb_v[M * f + m].insert(mat_v->get_tab2()(j) - 1);
115
116 /* aux faces : gradient aux faces + remplissage de dgp_pb */
117 std::vector<std::set<int>> dfgpf(N); //dfgpf[n][idx] = coeff : dependance en les pfb variables (presents dans dpf_ve)
118 for (f = 0; f < zone.nb_faces_tot(); f++)
119 {
120 /* |f| grad p */
121 for (i = fgrad_d(f); i < fgrad_d(f + 1); i++) //face interne -> flux multipoints
122 for (e = fgrad_e(i), n = 0, m = 0; n < N; n++, m += (M > 1))
123 {
124 if (f < zone.nb_faces() && e < ne_tot) sten_p.append_line(N * f + n, M * e + m);
125 if (e >= ne_tot && dpb_v.count(M * (e - ne_tot) + m)) dfgpf[n].insert(M * (e - ne_tot) + m);
126 }
127
128 /* face -> vf(f) * phi grad p */
129 if (fcl(f, 0) < 2 && f < zone.nb_faces())
130 for (n = 0, m = 0; n < N; n++, m += (M > 1))
131 for (auto &c : dfgpf[n]) dgp_pb[N * f + n].insert(M * c + m);
132 /* elems amont/aval -> ve(e) * phi grad p */
133 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
134 if (e < zone.nb_elem())
135 for (d = 0; d < D; d++)
136 if (fs(f) * std::fabs(xv(f, d) - xp(e, d)) > 1e-6 * ve(e))
137 for (n = 0, m = 0; n < N; n++, m += (M > 1))
138 for (auto &c : dfgpf[n])
139 dgp_pb[N * (nf_tot + D * e + d) + n].insert(M * c + m);
140 for (n = 0; n < N; n++) dfgpf[n].clear();
141 }
142
143 /* aux elements : gradient aux elems */
144 for (e = 0; e < zone.nb_elem(); e++)
145 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
146 for (j = fgrad_d(f); j < fgrad_d(f + 1); j++)
147 if ((eb = fgrad_e(j)) < ne_tot)
148 for (d = 0; d < D; d++)
149 if (std::fabs(fs(f) * (xv(f, d) - xp(e, d))) > 1e-6 * ve(e))
150 for (n = 0, m = 0; n < N; n++, m += (M > 1))
151 sten_p.append_line(N * (nf_tot + D * e + d) + n, M * eb + m);
152
153 /* sten_v en une ligne */
154 for (auto &i_sc : dgp_pb)
155 for (auto &c : i_sc.second)
156 for (auto &k : dpb_v.at(c)) sten_v.append_line(i_sc.first, k);
157
158 /* allocation / remplissage */
159 tableau_trier_retirer_doublons(sten_p), tableau_trier_retirer_doublons(sten_v);
160 if (mat_p) Matrix_tools::allocate_morse_matrix(N * (nf_tot + ne_tot * D), M * ne_tot , sten_p, mat2_p), mat_p->nb_colonnes() ? *mat_p += mat2_p : *mat_p = mat2_p;
161 if (mat_v) Matrix_tools::allocate_morse_matrix(N * (nf_tot + ne_tot * D), N * (nf_tot + ne_tot * D), sten_v, mat2_v), mat_v->nb_colonnes() ? *mat_v += mat2_v : *mat_v = mat2_v;
162}

164void Op_Grad_PolyMAC_P0_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
165{
166 const Zone_PolyMAC_P0& zone = ref_zone.valeur();
167 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
168 const Conds_lim& cls = ref_zcl->les_conditions_limites();
169 const IntTab& f_e = zone.face_voisins(), &fcl = ch.fcl();
170 const DoubleTab& nf = zone.face_normales(), &xp = zone.xp(), &xv = zone.xv(), &vfd = zone.volumes_entrelaces_dir(),
171 &press = semi_impl.count("pression") ? semi_impl.at("pression") : (le_champ_inco.non_nul() ? le_champ_inco->valeurs() : ref_cast(Navier_Stokes_std, equation()).pression().valeurs()),
172 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
173 const DoubleVect& fs = zone.face_surfaces(), &ve = zone.volumes(), &pe = zone.porosite_elem(), &pf = zone.porosite_face();
174 int i, j, e, f, fb, ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), d, D = dimension, n, N = secmem.line_size(), m, M = press.line_size();
175 update_grad();
176 const std::string& nom_inc = ch.le_nom().getString();
177 Matrice_Morse *mat_p = !semi_impl.count("pression") && matrices.count("pression") ? matrices.at("pression") : NULL,
178 *mat_v = !semi_impl.count(nom_inc) && matrices.count(nom_inc) ? matrices.at(nom_inc) : NULL;
179
180 DoubleTrav gb(nf_tot, M), gf(N), alpha(N); //-grad p aux bords , alpha |f| (grad p)_f
181 std::map<int, std::map<int, double>> dgp_gb, dgb_v; //dependances vitesses -(dgb_v)-> -grad p aux bords -(dgp_gb)-> grad p ailleurs
182 for (f = 0; f < zone.nb_faces_tot(); f++)
183 if (fcl(f, 0) > 1) //Dirichlet/Symetrie : pression du voisin + correction en regardant l'eq de NS dans celui-ci
184 for (e = f_e(f, 0), n = 0, m = 0; n < N; n++, m += (M > 1))
185 {
186 double fac = 1. / (fs(f) * ve(e));
187 std::map<int, double> *dv = mat_v ? &dgb_v[M * f + m] : NULL; //dv[indice dans mat_NS] = coeff
188 for (d = 0, i = nf_tot + D * e; d < D; d++, i++)
189 if (std::fabs(nf(f, d)) > 1e-6 * fs(f)) //boucle sur la direction : i est l'indice dans mat_v
190 {
191 gb(f, m) += fac * nf(f, d) * secmem(i, n); //partie constante -> directement dans pfb
192 if (dv)
193 for (j = mat_v->get_tab1()(N * i + n) - 1; j < mat_v->get_tab1()(N * i + n + 1) - 1; j++) //partie lineaire -> dans dgb_v
194 if (mat_v->get_coeff()(j)) (*dv)[mat_v->get_tab2()(j) - 1] -= fac * nf(f, d) * mat_v->get_coeff()(j);
195 }
196 }
197
198 for (f = 0; f < zone.nb_faces(); f++)
199 if (fgrad_d(f + 1) == fgrad_d(f)) abort();
200 /* aux faces */
201 std::vector<std::map<int, double>> dgf_pe(N), dgf_gb(N); //dependance de [grad p]_f en les pressions aux elements, en les grad p aux faces de bord
202 for (f = 0; f < zone.nb_faces_tot(); f++)
203 {
204 for (alpha = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
205 for (n = 0; n < N; n++) alpha(n) += vfd(f, i) * (alp ? (*alp)(e, n) : 1);
206
207 /* |f| grad p */
208 for (gf = 0, i = fgrad_d(f); i < fgrad_d(f + 1); i++)
209 for (e = fgrad_e(i), fb = e - ne_tot, n = 0, m = 0; n < N; n++, m += (M > 1))
210 {
211 double fac = fgrad_c(i, m);
212 gf(n) += fac * (e < ne_tot ? press(e, m) : fcl(fb, 0) == 1 ? ref_cast(Neumann, cls[fcl(fb, 1)].valeur()).flux_impose(fcl(fb, 2), m) : gb(e - ne_tot, m));
213 if (mat_p && e < ne_tot) dgf_pe[n][e] += fac;
214 if (mat_v && fb >= 0 && dgb_v.count(M * fb + m)) dgf_gb[n][M * fb + m] += fac;
215 }
216
217 /* face -> vf(f) * phi grad p */
218 if (fcl(f, 0) < 2 && f < zone.nb_faces())
219 for (n = 0, m = 0; n < N; n++, m += (M > 1))
220 {
221 double fac = alpha(n) * pf(f);
222 secmem(f, n) -= fac * gf(n);
223 for (auto &&i_c : dgf_pe[n]) (*mat_p)(N * f + n, M * i_c.first + m) += fac * i_c.second;
224 for (auto &&i_c : dgf_gb[n]) dgp_gb[N * f + n][M * i_c.first + m] += fac * i_c.second;
225 }
226 /* elems amont/aval -> ve(e) * phi grad p */
227 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
228 if (e < zone.nb_elem())
229 for (d = 0; d < D; d++)
230 if (fs(f) * std::fabs(xv(f, d) - xp(e, d)) > 1e-6 * ve(e))
231 for (n = 0, m = 0; n < N; n++, m += (M > 1))
232 {
233 double fac = (i ? -1 : 1) * fs(f) * pe(e) * (xv(f, d) - xp(e, d)) * (alp ? (*alp)(e, n) : 1);
234 secmem(nf_tot + D * e + d, n) -= fac * gf(n);
235 for (auto &i_c : dgf_pe[n]) (*mat_p)(N * (nf_tot + D * e + d) + n, M * i_c.first + m) += fac * i_c.second;
236 for (auto &i_c : dgf_gb[n]) dgp_gb[N * (nf_tot + D * e + d) + n][M * i_c.first + m] += fac * i_c.second;
237 }
238 for (n = 0; n < N; n++) dgf_pe[n].clear(), dgf_gb[n].clear();
239 }
240
241 /* correction de mat_NS : en une seule ligne! */
242 if (mat_v)
243 for (auto &i_jc : dgp_gb)
244 for (auto &j_c : i_jc.second)
245 if (dgb_v.count(j_c.first))
246 for (auto &k_d : dgb_v.at(j_c.first))
247 (*mat_v)(i_jc.first, k_d.first) += j_c.second * k_d.second;
248}
