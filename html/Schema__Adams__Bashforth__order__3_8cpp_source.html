103static double Ln02(const double time_n_plus_1, const double time_n, const double time_n_minus_1, const double time_n_minus_2)
104{
105// Integration of first degree-2 lagrangian polynomial with Simpson's formulae
106// Ln02(time_n_minus_2) = 0
107// Ln02(time_n_minus_1) = 0
108// Ln02(time_n) = 1
109// Integration bound : time_n and time_n_plus_1
110 double result = 0.;
111
112 double midpoint = (time_n_plus_1+time_n);
113 midpoint *= 0.5;
114
115 double scaling_coeff = (time_n_plus_1-time_n);
116 scaling_coeff /= 6.;
117
118 double denominator = (time_n-time_n_minus_1);
119 denominator *= (time_n-time_n_minus_2);
120 denominator = 1./denominator;
121
122 double numerator = (time_n-time_n_minus_1)*(time_n-time_n_minus_2);
123 numerator += 4*(midpoint-time_n_minus_1)*(midpoint-time_n_minus_2);
124 numerator += (time_n_plus_1-time_n_minus_1)*(time_n_plus_1-time_n_minus_2);
125
126 result = scaling_coeff * numerator * denominator;
127 return result;
128}

130static double Ln12(const double time_n_plus_1, const double time_n, const double time_n_minus_1, const double time_n_minus_2)
131{
132// Integration of second degree-2 lagrangian polynomial with Simpson's formulae
133// Ln12(time_n_minus_2) = 0
134// Ln12(time_n_minus_1) = 1
135// Ln12(time_n) = 0
136// Integration bound : time_n and time_n_plus_1
137 double result = 0.;
138
139 double midpoint = (time_n_plus_1+time_n);
140 midpoint *= 0.5;
141
142 double scaling_coeff = (time_n_plus_1-time_n);
143 scaling_coeff /= 6.;
144
145 double denominator = (time_n_minus_1-time_n);
146 denominator *= (time_n_minus_1-time_n_minus_2);
147 denominator = 1./denominator;
148
149 double numerator = 0.;
150 numerator += 4*(midpoint-time_n)*(midpoint-time_n_minus_2);
151 numerator += (time_n_plus_1-time_n)*(time_n_plus_1-time_n_minus_2);
152
153 result = scaling_coeff * numerator * denominator;
154 return result;
155}

157void Schema_Adams_Bashforth_order_3::compute_adams_bashforth_coefficients(double time_step, const DoubleTab& times)
158{
159 if (nb_pas_dt()<100 && limpr() && facteur_securite_pas()==1.0)
160 {
161 Cerr << finl;
162 Cerr << "********************* Warning (printed only on the first 100 time steps) **********************"<< finl;
163 Cerr << "The Adams Bashforth order 3 scheme is recommended with a facsec lower than 1.0 for example 0.5."<< finl;
164 Cerr << "***********************************************************************************************"<< finl;
165// See $TRUST_ROOT/tests/Reference/Schema_Adams_Bashforth_order_3 test case.
166 }
167
168 assert(times.size_array() == 4); //2 past times, present and future times, stored in ascending order inside "times" table
169
170 if (coefficients().size_array() != 3)
171 {
172 coefficients().resize(3,Array_base::NOCOPY_NOINIT);
173 }
174
175 double inv_time_step = 1./time_step;
176 double time_n_minus_2 = times[0];
177 double time_n_minus_1 = times[1];
178 double time_n = times[2];
179 double time_n_plus_1 = times[3];
180
181 coefficients()[2] = Ln02(time_n_plus_1,time_n,time_n_minus_1,time_n_minus_2) * inv_time_step;
182 coefficients()[1] = Ln12(time_n_plus_1,time_n,time_n_minus_1,time_n_minus_2) * inv_time_step;
183 coefficients()[0] = 1-coefficients()[2]-coefficients()[1];
184}
