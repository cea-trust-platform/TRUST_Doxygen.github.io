101 Cerr << motlu << finl;
102 no_qdm_ = 1;
103 }
104 else if (motlu=="solveur")
105 {
106 le_solveur_est_lu = 1;

108// redefinition de motlu pour garder les minuscules/majuscules
109 Nom motlubis;

150
151 if (le_solveur_est_lu==0)
152 {
153 SChaine toto;
154 toto<<"Gmres { diag seuil "<<param_defaut_.seuil_generation_solveur()<<" nb_it_max "<< param_defaut_.nb_it_max() << " controle_residu " <<controle_residu_<<" } "<<finl;
155 le_solveur_lu_=Nom(toto.get_str());

159 EChaine titi(le_solveur_lu_);

161 }
162 param_defaut_.solveur().nommer("solveur_implicite");
163 return is;

179 exit();

185 param_impl.seuil_convergence_implicite() = param_defaut_.seuil_convergence_implicite();
186 if (param_impl.seuil_verification_solveur()<0)
187 param_impl.seuil_verification_solveur() = param_defaut_.seuil_verification_solveur();
188 if (param_impl.seuil_test_preliminaire_lu()==0)
189 param_impl.seuil_test_preliminaire_solveur() = param_defaut_.seuil_test_preliminaire_solveur();

191 param_impl.solveur() = param_defaut_.solveur();
192
193// Some checks:

207 Cerr << finl << "Error!" << finl;

216void Simpler_Base::assembler_matrice_pression_implicite(Equation_base& eqn_NS,const Matrice_Morse& matrice,Matrice& matrice_en_pression_2)
217{
218 Navier_Stokes_std& eqnNS = ref_cast(Navier_Stokes_std,eqn_NS);
219 const IntVect& tab1 = matrice.get_tab1();
220 const DoubleVect& coeff = matrice.get_coeff();
221 const DoubleTab& present = eqn_NS.inconnue().valeurs();
222 int nb_comp = 1;
223 int deux_entrees = 0;
224
225 if (present.nb_dim()==2)
226 {
227 deux_entrees = 1;
228 nb_comp = present.dimension(1);
229 }
230
231 const Zone_VF& la_zone = ref_cast(Zone_VF,eqnNS.zone_dis().valeur());
232 if (deux_entrees==0)
233 {
234 DoubleVect vol2 = la_zone.volumes_entrelaces();
235 int ns = vol2.size();
236 for (int i=0; i<ns; i++)
237 {
238 int idiag = tab1[i*nb_comp]-1;
239 double ref = coeff[idiag];
240 for (int c=1; c<nb_comp; c++)
241 if (!est_egal(ref,coeff[tab1[i*nb_comp+c]-1]))
242 {
243 Cerr<<"Pb dans Piso sur la diagonale case"<<i<<" comp "<< c<<" ref "<<ref<<" valeurs "<<coeff[tab1[i*nb_comp+c]-1]<<finl;
244 exit();
245 }
246 vol2[i] = coeff[idiag];
247 }
248 vol2.echange_espace_virtuel();
249 eqnNS.assembleur_pression().valeur().assembler_mat(matrice_en_pression_2,vol2,1,1);
250 }
251 else
252 {
253 DoubleTab vol2(present);
254 int ns = vol2.dimension(0);
255 for (int i=0; i<ns; i++)
256 for (int c=0; c<nb_comp; c++)
257 vol2(i,c) = matrice(i*nb_comp+c,i*nb_comp+c);
258
259 vol2.echange_espace_virtuel();
260 eqnNS.assembleur_pression().valeur().assembler_mat(matrice_en_pression_2,vol2,1,1);
261 }
262
263 SolveurSys& solveur_pression_ = eqnNS.solveur_pression();
264 solveur_pression_.valeur().reinit();
265}
