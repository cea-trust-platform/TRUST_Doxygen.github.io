69DoubleTab& Masse_PolyMAC_P0_Face::appliquer_impl(DoubleTab& sm) const
70{
71 const Zone_PolyMAC_P0& zone = la_zone_PolyMAC_P0.valeur();
72 const IntTab& f_e = zone.face_voisins();
73 const DoubleVect& pf = zone.porosite_face(), &pe = zone.porosite_elem(), &vf = zone.volumes_entrelaces(), &ve = zone.volumes();
74 int i, e, f, nf_tot = zone.nb_faces_tot(), n, N = equation().inconnue().valeurs().line_size(), d, D = dimension;
75 const DoubleTab *a_r = sub_type(QDM_Multiphase, equation()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.champ_conserve().passe() : NULL, &vfd = zone.volumes_entrelaces_dir();
76 double fac;
77
78// vitesses aux faces
79 for (f = 0; f < zone.nb_faces(); f++)
80 for (n = 0; n < N; n++)
81 {
82 for (fac = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++) fac += vfd(f, i) / vf(f) * (a_r ? (*a_r)(e, n) : 1);
83 sm(f, n) /= pf(f) * vf(f) * fac; //vitesse calculee
84 }
85
86// vitesses aux elements
87 if (sm.dimension_tot(0) > N * nf_tot)
88 for (e = 0; e < zone.nb_elem(); e++)
89 for (d = 0; d < D; d++)
90 for (n = 0; n < N; n++)
91 sm(nf_tot + D * e + d, n) /= pe(e) * ve(e) * (a_r ? (*a_r)(e, n) : 1);
92
93 sm.echange_espace_virtuel();
94 return sm;
95}

97void Masse_PolyMAC_P0_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
98{
99 const std::string& nom_inc = equation().inconnue().le_nom().getString();
100 if (!matrices.count(nom_inc)) return; //rien a faire
101 Matrice_Morse& mat = *matrices.at(nom_inc), mat2;
102 const Zone_PolyMAC_P0& zone = la_zone_PolyMAC_P0;
103 const DoubleTab& inco = equation().inconnue().valeurs();
104 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
105 const Masse_ajoutee_base *corr = pbm && pbm->has_correlation("masse_ajoutee") ? &ref_cast(Masse_ajoutee_base, pbm->get_correlation("masse_ajoutee").valeur()) : NULL;
106 int i, e, f, nf_tot = zone.nb_faces_tot(), m, n, N = inco.line_size(), d, D = dimension;
107
108 IntTrav sten(0, 2);
109 sten.set_smart_resize(1);
110 for (f = 0, i = 0; f < zone.nb_faces(); f++)
111 for (n = 0; n < N; n++, i++) //faces reelles
112 if (corr)
113 for (m = 0; m < N; m++) sten.append_line(i, N * f + m);
114 else sten.append_line(i, i);
115 for (e = 0, i = N * nf_tot; e < zone.nb_elem_tot(); e++)
116 for (d = 0; d < D; d++)
117 for (n = 0; n < N; n++, i++) //tous les elems (pour Op_Grad_PolyMAC_P0_Face)
118 if (corr)
119 for (m = 0; m < N; m++) sten.append_line(i, N * (nf_tot + D * e + d) + m);
120 else sten.append_line(i, i);
121 Matrix_tools::allocate_morse_matrix(inco.size_totale(), inco.size_totale(), sten, mat2);
122 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
123}

125void Masse_PolyMAC_P0_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, double dt, const tabs_t& semi_impl, int resoudre_en_increments) const
126{
127 const DoubleTab& inco = equation().inconnue().valeurs(), &passe = equation().inconnue().passe();
128 Matrice_Morse *mat = matrices[equation().inconnue().le_nom().getString()]; //facultatif
129 const Zone_PolyMAC_P0& zone = la_zone_PolyMAC_P0;
130 const Conds_lim& cls = la_Zone_Cl_PolyMAC->les_conditions_limites();
131 const IntTab& f_e = zone.face_voisins(), &fcl = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur()).fcl();
132 const DoubleVect& pf = zone.porosite_face(), &pe = zone.porosite_elem(), &vf = zone.volumes_entrelaces(), &ve = zone.volumes(), &fs = zone.face_surfaces();
133 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
134 const DoubleTab& nf = zone.face_normales(), &rho = equation().milieu().masse_volumique().passe(),
135 *alpha = pbm ? &pbm->eq_masse.inconnue().passe() : NULL, *a_r = pbm ? &pbm->eq_masse.champ_conserve().passe() : NULL, &vfd = zone.volumes_entrelaces_dir();
136 const Masse_ajoutee_base *corr = pbm && pbm->has_correlation("masse_ajoutee") ? &ref_cast(Masse_ajoutee_base, pbm->get_correlation("masse_ajoutee").valeur()) : NULL;
137 int i, e, f, nf_tot = zone.nb_faces_tot(), m, n, N = inco.line_size(), d, D = dimension, cR = rho.dimension_tot(0) == 1;
138
139 /* faces : si CLs, pas de produit par alpha * rho en multiphase */
140 DoubleTrav masse(N, N), masse_e(N, N); //masse alpha * rho, contribution
141 for (f = 0; f < zone.nb_faces(); f++) //faces reelles
142 {
143 if (!pbm || fcl(f, 0) >= 2)
144 for (masse = 0, n = 0; n < N; n++) masse(n, n) = 1; //pas Pb_Multiphase ou CL -> pas de alpha * rho
145 else for (masse = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
146 {
147 for (masse_e = 0, n = 0; n < N; n++) masse_e(n, n) = (*a_r)(e, n); //partie diagonale
148 if (corr) corr->ajouter(&(*alpha)(e, 0), &rho(!cR * e, 0), masse_e); //partie masse ajoutee
149 for (n = 0; n < N; n++)
150 for (m = 0; m < N; m++) masse(n, m) += vfd(f, i) / vf(f) * masse_e(n, m); //contribution au alpha * rho de la face
151 }
152 for (n = 0; n < N; n++)
153 {
154 double fac = pf(f) * vf(f) / dt;
155 for (m = 0; m < N; m++) secmem(f, n) -= fac * resoudre_en_increments * masse(n, m) * inco(f, m);
156 if (fcl(f, 0) < 2)
157 for (m = 0; m < N; m++) secmem(f, n) += fac * masse(n, m) * passe(f, m);
158 else if (fcl(f, 0) == 3)
159 for (d = 0; d < D; d++)
160 secmem(f, n) += fac * masse(n, n) * ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), N * d + n) * nf(f, d) / fs(f);
161 if (mat)
162 for (m = 0; m < N; m++)
163 if (masse(n, m)) (*mat)(N * f + n, N * f + m) += fac * masse(n, m);
164 }
165 }
166
167 for (e = 0, i = nf_tot; e < zone.nb_elem_tot(); e++) //tous les elems (pour Op_Grad_PolyMAC_P0_Face)
168 {
169 for (masse = 0, n = 0; n < N; n++) masse(n, n) = a_r ? (*a_r)(e, n) : 1; //partie diagonale
170 if (corr) corr->ajouter(&(*alpha)(e, 0), &rho(!cR * e, 0), masse); //partie masse ajoutee
171 for (d = 0; d < D; d++, i++)
172 for (n = 0; n < N; n++)
173 {
174 double fac = pe(e) * ve(e) / dt;
175 for (m = 0; m < N; m++) secmem(i, n) -= fac * masse(n, m) * (resoudre_en_increments * inco(i, m) - passe(i, m));
176 if (mat)
177 for (m = 0; m < N; m++)
178 if (masse(n, m)) (*mat)(N * i + n, N * i + m) += fac * masse(n, m);
179 }
180 }
181 i++;
182}
