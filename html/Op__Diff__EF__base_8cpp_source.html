90double Op_Diff_EF_base::calculer_dt_stab() const
91{
92 remplir_nu(nu_);
93// La diffusivite est constante dans le domaine donc
94//
95// dt_diff = h*h/diffusivite
96
97 double dt_stab = DMAXFLOAT;
98 const Zone_EF& zone_EF = la_zone_EF.valeur();
99 if (! has_champ_masse_volumique())
100 {
101 if (1)
102 {
103// Methode "standard" de calcul du pas de temps
104// Ce calcul est tres conservatif: si le max de la diffusivite
105// n'est pas atteint a l'endroit ou le min de delta_h_carre est atteint,
106// le pas de temps est sous-estime.
107 const Champ_base& champ_diffusivite = diffusivite_pour_pas_de_temps();
108 const DoubleVect& valeurs_diffusivite = champ_diffusivite.valeurs();
109 double alpha_max = local_max_vect(valeurs_diffusivite);
110
111 if ((alpha_max == 0.)||(zone_EF.nb_elem()==0))
112 dt_stab = DMAXFLOAT;
113 else
114 {
115 const double min_delta_h_carre = zone_EF.carre_pas_du_maillage();
116 dt_stab = min_delta_h_carre / (2. * dimension * alpha_max);
117 }
118 }
119 else
120 {
121 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
122 const DoubleTab& coord=zone_ef.zone().domaine().les_sommets();
123 const IntTab& elems=zone_ef.zone().les_elems() ;
124 const Champ_base& champ_diffusivite = diffusivite_pour_pas_de_temps();
125 const DoubleVect& valeurs_diffusivite = champ_diffusivite.valeurs();
126 const int nb_elem = zone_EF.nb_elem();
127 int elem;
128// Champ de masse volumique variable.
129 for (elem = 0; elem < nb_elem; elem++)
130 {
131
132 const double diffu = valeurs_diffusivite(elem)+DMINFLOAT;
133 double dx2=0,dl;
134 for (int d=0; d<3; d++)
135 {
136 dl=coord(elems(elem,0),d)-coord(elems(elem,7),d);
137 dx2+=(dl*dl);
138 }
139 const double dt = dx2 / diffu;
140 if (dt_stab > dt)
141 {
142 dt_stab = dt;
143// Cerr<<" ii "<<elem<<" "<<diffu<<" "<<dx2<<finl;
144 }
145 }
146 }
147 }
148 else
149 {
150 const double deux_dim = 2. * Objet_U::dimension;
151 const Champ_base& champ_diffu = diffusivite();
152 const DoubleTab& valeurs_diffu = champ_diffu.valeurs();
153 const Champ_base& champ_rho = get_champ_masse_volumique();
154 const DoubleTab& valeurs_rho = champ_rho.valeurs();
155 assert(sub_type(Champ_Fonc_P0_base, champ_rho));
156 assert(sub_type(Champ_Fonc_P0_base, champ_diffu));
157 const int nb_elem = zone_EF.nb_elem();
158 int elem;
159// Champ de masse volumique variable.
160 for (elem = 0; elem < nb_elem; elem++)
161 {
162 const double h_carre = zone_EF.carre_pas_maille(elem);
163 const double diffu = valeurs_diffu(elem);
164 const double rho = valeurs_rho(elem);
165 const double dt = h_carre * rho / (deux_dim * diffu);
166 if (dt_stab > dt)
167 dt_stab = dt;
168 }
169 }
170 dt_stab = Process::mp_min(dt_stab);
171 return dt_stab;
172}
