28void Eval_Puiss_Neutr_VDF_Elem::associer_repartition(const Nom& n, const Nom& nom_ssz)
29{
30 const int nb_elem = la_zone->nb_elem();
31 const DoubleTab& xp = la_zone->xp();
32 const Sous_Zone& ssz = la_zone->zone().domaine().ss_zone(nom_ssz);
33
34 fxyz = n;
35
36 std::string sfxyz(fxyz.getString());
37 for (auto & c: sfxyz) c = toupper(c);
38 Parser p(sfxyz,3);
39 p.addVar("x");
40 p.addVar("y");
41 p.addVar("z");
42 p.parseString();
43
44 rep.resize(nb_elem);
45 rep = 0.;
46 for(int i=0; i<ssz.nb_elem_tot(); i++) rep(ssz(i)) = 1.;
47
48 for (int i=0; i<nb_elem; i++)
49 {
50 double x,y,z;
51 x = xp(i,0);
52 y = xp(i,1);
53 if (Objet_U::dimension == 3) z = xp(i,2);
54 else z=0.;
55 p.setVar("x",x);
56 p.setVar("y",y);
57 p.setVar("z",z);
58 rep(i) *= p.eval();
59 }
60}
