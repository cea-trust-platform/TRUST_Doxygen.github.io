70void Op_Dift_VEF_base::completer()
71{
72 Operateur_base::completer();
73
74 const RefObjU& modele_turbulence = equation().get_modele(TURBULENCE);
75 if (modele_turbulence.non_nul() && sub_type(Mod_turb_hyd_base,modele_turbulence.valeur()))
76 {
77 const Mod_turb_hyd_base& mod_turb = ref_cast(Mod_turb_hyd_base,modele_turbulence.valeur());
78 const Champ_Fonc& viscosite_turbulente = mod_turb.viscosite_turbulente();
79 associer_diffusivite_turbulente(viscosite_turbulente);
80 associer_modele_turbulence(mod_turb);
81 }
82 else if (sub_type(Modele_turbulence_scal_base,modele_turbulence.valeur()))
83 {
84 const Modele_turbulence_scal_base& modele_turbulence_scalaire = ref_cast(Modele_turbulence_scal_base,modele_turbulence.valeur());
85 const Champ_Fonc& conductivite_turbulente = modele_turbulence_scalaire.conductivite_turbulente();
86 associer_diffusivite_turbulente(conductivite_turbulente);
87 }
88 else
89 {
90 Cerr << "Error in Op_Dift_VEF_Face::completer() " << finl;
91 Cerr << que_suis_je() <<" operator is presently associated to "<<equation().que_suis_je()<<finl;
92 Cerr << "instead of being associated to an equation dedicated to a turbulent flow."<<finl;
93 exit();
94 }
95}

97void Op_Dift_VEF_base::calculer_borne_locale(DoubleVect& borne_visco_turb, double dt_conv, double dt_diff_sur_dt_conv) const
98{
99 const Zone_VEF& la_zone_VEF = la_zone_vef.valeur();
100 int nb_elem = la_zone_VEF.nb_elem();
101 int flag = diffusivite().valeurs().dimension(0)>1 ? 1 : 0;
102 for (int elem=0; elem<nb_elem; elem++)
103 {
104 double h_inv = 1./la_zone_VEF.carre_pas_maille(elem);
105// C'est pas tres propre pour recuperer diffu mais ca evite de coder cette methode dans plusieurs classes:
106 double diffu = (flag ? diffusivite().valeurs()(elem) : diffusivite().valeurs()(0,0));
107//
108 double coef = 1./(2*(dt_conv+DMINFLOAT)*dimension*h_inv*dt_diff_sur_dt_conv) - diffu;
109 if (coef>0 && coef<borne_visco_turb(elem))
110 borne_visco_turb(elem) = coef;
111 }
112}
