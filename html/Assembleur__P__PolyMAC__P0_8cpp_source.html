52int Assembleur_P_PolyMAC_P0::assembler_mat(Matrice& la_matrice,const DoubleVect& diag,int incr_pression,int resoudre_en_u)
53{
54 set_resoudre_increment_pression(incr_pression);
55 set_resoudre_en_u(resoudre_en_u);
56 la_matrice.typer("Matrice_Morse");
57 Matrice_Morse& mat = ref_cast(Matrice_Morse, la_matrice.valeur());
58
59 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0, la_zone_PolyMAC.valeur());
60 const Op_Grad_PolyMAC_P0_Face& grad = ref_cast(Op_Grad_PolyMAC_P0_Face, ref_cast(Navier_Stokes_std, equation()).operateur_gradient().valeur());
61 grad.update_grad();
62 const DoubleTab& fgrad_c = grad.fgrad_c;
63 const IntTab& f_e = zone.face_voisins(), &fgrad_d = grad.fgrad_d, &fgrad_e = grad.fgrad_e;
64 const DoubleVect& pf = zone.porosite_face(), &fs = zone.face_surfaces();
65 int i, j, e, eb, f, ne = zone.nb_elem(), ne_tot = zone.nb_elem_tot();
66
67// en l'absence de CLs en pression, on ajoute P(0) = 0 sur le process 0
68 has_P_ref=0;
69 for (int n_bord=0; n_bord<la_zone_PolyMAC->nb_front_Cl(); n_bord++)
70 if (sub_type(Neumann_sortie_libre, la_zone_Cl_PolyMAC->les_conditions_limites(n_bord).valeur()) )
71 has_P_ref=1;
72
73 /* 1. stencil de la matrice en pression : seulement au premier passage */
74 if (!stencil_done)
75 {
76 IntTrav stencil(0, 2);
77 stencil.set_smart_resize(1);
78 for (f = 0; f < zone.nb_faces(); f++)
79 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
80 if (e < ne)
81 for (j = fgrad_d(f); j < fgrad_d(f + 1); j++)
82 if ((eb = fgrad_e(j)) < ne_tot) stencil.append_line(e, eb);
83
84 tableau_trier_retirer_doublons(stencil);
85 Matrix_tools::allocate_morse_matrix(ne_tot, ne_tot, stencil, mat);
86 mat.sort_stencil();
87 tab1.ref_array(mat.get_set_tab1()), tab2.ref_array(mat.get_set_tab2());
88 stencil_done = 1;
89 }
90 else //sinon, on recycle
91 {
92 mat.get_set_tab1().ref_array(tab1);
93 mat.get_set_tab2().ref_array(tab2);
94 mat.get_set_coeff().resize(tab2.size());
95 mat.set_nb_columns(ne_tot);
96 }
97
98 /* 2. remplissage des coefficients : style Op_Diff_PolyMAC_Elem */
99 for (f = 0; f < zone.nb_faces(); f++)
100 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
101 if (e < ne)
102 for (j = fgrad_d(f); j < fgrad_d(f + 1); j++)
103 if ((eb = fgrad_e(j)) < ne_tot)
104 mat(e, eb) += (i ? 1 : -1) * pf(f) * fs(f) * fgrad_c(j, 0);
105
106 if (!has_P_ref && !Process::me()) mat(0, 0) *= 2;
107
108 return 1;
109}

113void Assembleur_P_PolyMAC_P0::dimensionner_continuite(matrices_t matrices, int aux_only) const
114{
115 if (aux_only) return; //rien a faire
116 int e, n, N = ref_cast(Pb_Multiphase, equation().probleme()).nb_phases(), ne_tot = la_zone_PolyMAC->nb_elem_tot();
117 IntTrav stencil(0, 2);
118 stencil.set_smart_resize(1);
119 for (e = 0; e < la_zone_PolyMAC->nb_elem(); e++)
120 for (n = 0; n < N; n++) stencil.append_line(e, N * e + n);
121 Matrix_tools::allocate_morse_matrix(ne_tot, N * ne_tot, stencil, *matrices.at("alpha"));
122}

124void Assembleur_P_PolyMAC_P0::assembler_continuite(matrices_t matrices, DoubleTab& secmem, int aux_only) const
125{
126 if (aux_only) return;
127 const DoubleTab& alpha = ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().valeurs();
128 Matrice_Morse& mat = *matrices.at("alpha");
129 const DoubleVect& ve = la_zone_PolyMAC->volumes(), &pe = la_zone_PolyMAC->porosite_elem();
130 int e, n, N = alpha.line_size();
131 /* second membre : on multiplie par porosite * volume pour que le systeme en P soit symetrique en cartesien */
132 for (e = 0; e < la_zone_PolyMAC->nb_elem(); e++)
133 for (secmem(e) = -pe(e) * ve(e), n = 0; n < N; n++) secmem(e) += pe(e) * ve(e) * alpha(e, n);
134 /* matrice */
135 for (e = 0; e < la_zone_PolyMAC->nb_elem(); e++)
136 for (n = 0; n < N; n++) mat(e, N * e + n) = -pe(e) * ve(e);
137}
