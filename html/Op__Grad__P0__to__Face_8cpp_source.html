34void Op_Grad_P0_to_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
35{
36 statistiques().begin_count(gradient_counter_);
37 const DoubleTab& inco = semi_impl.count("pression") ? semi_impl.at("pression") : equation().inconnue().valeur().valeurs();
38 assert_espace_virtuel_vect(inco);
39 const Zone_VDF& zvdf = la_zone_vdf.valeur();
40 const Zone_Cl_VDF& zclvdf = la_zcl_vdf.valeur();
41 const DoubleVect& face_surfaces = zvdf.face_surfaces();
42
43 double dist;
44 int n0, n1, ori;
45
46// Boucle sur les bords pour traiter les conditions aux limites
47 for (int n_bord = 0; n_bord<zvdf.nb_front_Cl(); n_bord++)
48 {
49 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
50 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
51 const int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
52
53 if (sub_type(Periodique,la_cl.valeur())) //!< Correction periodicite
54 {
55 for (int num_face = ndeb; num_face < nfin; num_face++)
56 {
57 n0 = face_voisins(num_face,0), n1 = face_voisins(num_face,1);
58 dist = volume_entrelaces(num_face) / face_surfaces(num_face);
59 secmem(num_face) += ((inco(n1) - inco(n0)))/dist;
60 }
61 }
62 else
63 {
64 for (int num_face = ndeb; num_face < nfin; num_face++)
65 {
66 n0 = face_voisins(num_face,0);
67 if (n0<0) n0 = face_voisins(num_face,1);
68
69 ori = orientation(num_face);
70 int face_opposee = zvdf.elem_faces(n0,ori);
71 if (face_opposee == num_face) face_opposee = zvdf.elem_faces(n0,ori+dimension);
72
73 n1 = face_voisins(face_opposee,0);
74 if (n1==n0) n1 = face_voisins(face_opposee,1);
75
76 secmem(num_face) += (inco(n1)-inco(n0)) / (xp(n1,ori)- xp(n0,ori));
77 }
78 }
79 }
80
81// Boucle sur les faces internes
82 for (int num_face = zvdf.premiere_face_int(); num_face < zvdf.nb_faces(); num_face++)
83 {
84 n0 = face_voisins(num_face,0), n1 = face_voisins(num_face,1), ori = orientation(num_face);
85 secmem(num_face) += (inco(n1)-inco(n0))/(xp(n1,ori)- xp(n0,ori));
86 }
87
88 secmem.echange_espace_virtuel();
89 statistiques().end_count(gradient_counter_);
90
91}
