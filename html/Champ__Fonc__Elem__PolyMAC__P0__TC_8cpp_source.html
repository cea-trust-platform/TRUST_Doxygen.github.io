40void Champ_Fonc_Elem_PolyMAC_P0_TC::me_calculer(double tps) //See Pope 2000 page 367 for ref
41{
42 const Champ_Face_PolyMAC_P0& vitesse = ref_cast(Champ_Face_PolyMAC_P0,champ_.valeur());
43 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0,vitesse.zone_vf());
44 int e, d_U, d_X, n ;
45 int D = dimension, N = champ_a_deriver().valeurs().line_size();
46 int ne = zone.nb_elem(), nf_tot = zone.nb_faces_tot();
47
48 const Navier_Stokes_std& eq = ref_cast(Navier_Stokes_std, vitesse.equation());
49 DoubleTab& tab_tc = valeurs();
50 const grad_Champ_Face_PolyMAC_P0& grad = ref_cast(grad_Champ_Face_PolyMAC_P0, eq.get_champ("gradient_vitesse"));
51 const DoubleTab& tab_grad = grad.valeurs();
52
53 for (e = 0; e < ne; e++)
54 for (n = 0; n < N; n++)
55 {
56 tab_tc(e, n) = 0;
57 for (d_U = 0; d_U < D; d_U++)
58 for (d_X = 0; d_X < D; d_X++)
59 {
60 double Sij = 0.5 * (tab_grad(nf_tot + d_X + e * D , D * n + d_U) + tab_grad(nf_tot + d_U + e * D , D * n + d_X)) ;
61 if (d_U == d_X)
62 for (int i = 0 ; i <D ; i++) Sij += -1./D*tab_grad(nf_tot + i + e * D , D * n + i) ; //!< Substract the divergence : this term is zero in incompressible NS
63 tab_tc(e, n) += Sij * Sij ;
64 }
65 tab_tc(e, n) = sqrt(2 * tab_tc(e, n));
66 }
67
68 tab_tc.echange_espace_virtuel();
69}
