26//! renvoit le tableau res, res=org si flag=0, res= porosite*org sinon<br>attention ne pas modifier res car sinon on ne sait pas ce que l'on fait sur org (d'ou le renvoi const)
27const DoubleTab& modif_par_porosite_si_flag(const DoubleTab& org, DoubleTab& res,int flag,const DoubleVect& porosite)
28{
29// Switch mp_min_vect to local_min_abs_vect to avoid mp_min
30// call during the calculation of operators (convection, diffusion,...)
31 if ((flag==0)||(local_min_abs_vect(porosite)==1))
32 res.ref(org);
33 else
34 {
35 assert(org.dimension_tot(0)==porosite.size_totale());
36 res = org;
37 tab_multiply_any_shape(res, porosite, VECT_ALL_ITEMS);
38 }
39 return res;
40}

54//! Porosites_champ nom_pb champ affecte le champ champ a la porosite volumique du domaine du probleme de nom_pb, puis calcule les porosites surfacqiues comme moyenne harmonique
55Entree& Porosites_champ::interpreter(Entree& is)
56{
57 Nom nom_pb;
58 is >> nom_pb;
59 if(! sub_type(Probleme_base, objet(nom_pb)))
60 {
61 Cerr << nom_pb << " is of type " << objet(nom_pb).que_suis_je() << finl;
62 Cerr << "We waited an object of type Probleme_base" << finl;
63 Process::exit();
64 }
65 Probleme_base& pb=ref_cast(Probleme_base, objet(nom_pb));
66 Champ le_champ;
67 is>>le_champ;
68
69// On passe par la zone_VF
70 Zone_VF& zvf=ref_cast(Zone_VF, pb.equation(0).zone_dis().valeur());
71 DoubleVect& porosite_elem = zvf.porosite_elem();
72 DoubleVect& porosite_face_r = zvf.porosite_face();
73 const IntTab& face_voisins = zvf.face_voisins();
74 int nb_elem_tot = zvf.zone().nb_elem_tot();
75
76// on discretise des champs equivalents pour pouvoir faire affecter
77// puis on recopie les valeurs
78 Champ_Don porosite_elem_p;
79 pb.discretisation().discretiser_champ("CHAMP_ELEM", pb.equation(0).zone_dis(), "pp", "1",1,0., porosite_elem_p);
80
81 porosite_elem_p.valeur().valeurs()=1;
82 porosite_elem_p.valeur().affecter(le_champ);
83 for (int elem=0; elem<nb_elem_tot; elem++)
84 porosite_elem(elem)=porosite_elem_p.valeurs()(elem);
85
86// calcul de la porosite surfacique comme la moyenne harmonique
87// cf D.J pour justification
88 Champ_Don porosite_face_p;
89 pb.discretisation().discretiser_champ("VITESSE", pb.equation(0).zone_dis(), "pp", "1",1,0., porosite_face_p);
90 DoubleTab& porosite_face=porosite_face_p.valeurs();
91 int nb_face_tot = zvf.nb_faces_tot();
92 if (porosite_face_r.size()==porosite_face.size())
93 {
94 for (int face=0; face<nb_face_tot; face++)
95 {
96 int elem1 = face_voisins(face,1);
97 int elem2 = face_voisins(face,0);
98 if ((elem1 != -1) && (elem2 != -1))
99 {
100 porosite_face(face)=2./(1./porosite_elem(elem1)+1./porosite_elem(elem2));
101 }
102 else if (elem1!=-1)
103 porosite_face(face) = porosite_elem(elem1) ;
104 else
105 porosite_face(face) = porosite_elem(elem2) ;
106 }
107 porosite_face.echange_espace_virtuel();
108 for (int face=0; face<nb_face_tot; face++)
109 porosite_face_r(face) = porosite_face(face);
110 }
111 return is;
112}
