61// checking stencil
62bool Matrix_tools::is_normalized_stencil( const IntTab& stencil )
63{
64 const int size = stencil.dimension( 0 );
65 for ( int i=1; i<size; ++i )
66 {
67 int delta1 = stencil( i-1, 0 ) - stencil( i, 0 );
68 int delta2 = stencil( i-1, 1 ) - stencil( i, 1 );
69 int delta = ( delta1 == 0 ) ? delta2 : delta1;
70
71 if ( delta >= 0 )
72 {
73 return false;
74 }
75 }
76 return true;
77}

79// checking symmetric stencil
80bool Matrix_tools::is_normalized_symmetric_stencil( const IntTab& stencil )
81{
82 if ( ! ( is_normalized_stencil( stencil ) ) )
83 {
84 Cerr << "non-normalized stencil" << finl;
85 return false;
86 }
87
88 const int size = stencil.dimension( 0 );
89 for ( int i=0; i<size; ++i )
90 {
91 int delta = stencil( i, 0 ) - stencil( i, 1 );
92 if ( delta > 0 )
93 {
94 Cerr << "( " << i << " ) line index > column index : ( " << stencil( i, 0 ) << ", " << stencil( i, 1 ) << " )" << finl;
95 return false;
96 }
97 }
98
99 return true;
100}

103// building morse matrices
104void Matrix_tools::allocate_morse_matrix( const int nb_lines,
105 const int nb_columns,
106 const IntTab& stencil,
107 Matrice_Morse& matrix ,
108 const bool& attach_stencil_to_matrix )
109{
110 assert( is_normalized_stencil( stencil ) );
111
112 const int nb_coefficients = stencil.dimension( 0 );
113
114 matrix.dimensionner( nb_lines,
115 nb_columns,
116 nb_coefficients );
117
118 if ( nb_coefficients > 0 )
119 {
120 matrix.get_set_tab1( ) = 0 ;
121 matrix.get_set_tab1( )( 0 ) = 1;
122 for ( int i=0; i<nb_coefficients; ++i )
123 {
124 assert( stencil( i ,0 ) >= 0 );
125 assert( stencil( i ,0 ) < nb_lines );
126 assert( stencil( i ,1 ) >= 0 );
127 assert( stencil( i ,1 ) < nb_columns );
128
129 matrix.get_set_tab1( )( stencil( i, 0 ) + 1 ) += 1;
130 matrix.get_set_tab2( )( i ) = stencil( i, 1 ) + 1;
131 }
132 for ( int i=0; i<nb_lines; ++i )
133 {
134 matrix.get_set_tab1()( i + 1 ) += matrix.get_tab1()( i );
135 }
136 }
137
138 if( attach_stencil_to_matrix )
139 {
140 matrix.set_stencil( stencil );
141 }
142}

259// allocation for scaled addition
260void Matrix_tools::allocate_for_scaled_addition( const Matrice& A,
261 const Matrice& B,
262 Matrice& C )
263{
264 assert( A.valeur( ).nb_lignes( ) == B.valeur( ).nb_lignes( ) );
265 assert( A.valeur( ).nb_colonnes( ) == B.valeur( ).nb_colonnes( ) );
266
267 IntTab A_stencil;
268 A.valeur( ).get_stencil( A_stencil );
269 const int A_size = A_stencil.dimension( 0 );
270
271 IntTab B_stencil;
272 B.valeur( ).get_stencil( B_stencil );
273 const int B_size = B_stencil.dimension( 0 );
274
275 int size = A_size + B_size;
276 IntTab stencil;
277 stencil.set_smart_resize( 1 );
278 stencil.resize( size, 2 );
279
280 for ( int i=0; i<A_size; ++i )
281 {
282 stencil( i , 0 ) = A_stencil( i, 0 );
283 stencil( i , 1 ) = A_stencil( i, 1 ) ;
284 }
285
286 for ( int i=0; i<B_size; ++i )
287 {
288 stencil( i+A_size , 0 ) = B_stencil( i, 0 );
289 stencil( i+A_size , 1 ) = B_stencil( i, 1 ) ;
290 }
291
292 tableau_trier_retirer_doublons( stencil );
293
294 C.typer( "Matrice_Morse" );
295 Matrice_Morse& C_ = ref_cast( Matrice_Morse, C.valeur( ) );
296
297 allocate_morse_matrix( A.valeur( ).nb_lignes( ),
298 A.valeur( ).nb_colonnes( ),
299 stencil,
300 C_ );
301}

347// scaled addition
348void Matrix_tools::add_scaled_matrices( const Matrice& A,
349 const double alpha,
350 const Matrice& B,
351 const double beta,
352 Matrice& C )
353{
354 assert( sub_type( Matrice_Morse, C.valeur( ) ) );
355 Matrice_Morse& C_ = ref_cast( Matrice_Morse, C.valeur( ) );
356
357 assert( C_.check_sorted_morse_matrix_structure( ) );
358
359 IntTab A_stencil;
360 ArrOfDouble A_coefficients;
361 A.valeur( ).get_stencil_and_coefficients( A_stencil,
362 A_coefficients );
363
364 IntTab B_stencil;
365 ArrOfDouble B_coefficients;
366 B.valeur( ).get_stencil_and_coefficients( B_stencil,
367 B_coefficients );
368
369 int A_k = 0;
370 int B_k = 0;
371
372 C_.get_set_coeff() = 0.0 ;
373
374 const int nb_lines = C_.nb_lignes( );
375 for ( int i=0; i<nb_lines; ++i )
376 {
377 int k0 = C_.get_tab1()( i ) - 1;
378 int k1 = C_.get_tab1()( i + 1 ) - 1;
379
380 for ( int k=k0; k<k1; ++k )
381 {
382 int j = C_.get_tab2()( k ) - 1;
383
384 if ( ( A_k < A_stencil.dimension( 0 ) ) && ( A_stencil( A_k, 0 ) == i ) && ( A_stencil( A_k, 1 ) == j ) )
385 {
386 C_.get_set_coeff()( k ) += alpha * A_coefficients[ A_k ];
387 ++A_k;
388 }
389
390 if ( ( B_k < B_stencil.dimension( 0 ) ) && ( B_stencil( B_k, 0 ) == i ) && ( B_stencil( B_k, 1 ) == j ) )
391 {
392 C_.get_set_coeff()( k ) += beta * B_coefficients[ B_k ];
393 ++B_k;
394 }
395 }
396 }
397
398 assert( A_k == A_coefficients.size_array( ) );
399 assert( B_k == B_coefficients.size_array( ) );
400}

462bool Matrix_tools::is_diagonal_stencil( const int nb_lines,
463 const int nb_columns,
464 const IntTab& stencil )
465{
466 if ( nb_lines != nb_columns )
467 {
468 return false;
469 }
470
471 const int size = stencil.dimension( 0 );
472
473 if ( size == 0 )
474 {
475 return false;
476 }
477
478 for ( int i=0; i<size; ++i )
479 {
480 if ( stencil( i, 0 ) != stencil( i, 1 ) )
481 {
482 return false;
483 }
484
485 if ( stencil( i, 0 ) >= nb_lines )
486 {
487 return false;
488 }
489
490 if ( stencil( i, 1 ) >= nb_columns )
491 {
492 return false;
493 }
494 }
495
496 return true;
497}

536void Matrix_tools::extend_matrix_stencil( const IntTab& stencil,
537 Matrice& matrix ,
538 const bool& attach_stencil_to_matrix )
539{
540 if ( ! ( is_null_stencil( stencil ) ) )
541 {
542 const int nb_lines = matrix.valeur( ).nb_lignes( );

544

547 full_stencil.set_smart_resize( 1 );
548 const int size = stencil.dimension( 0 );

553

557 full_stencil( old_size + i , 1 ) = stencil( i, 1 );

561
562 allocate_from_stencil( nb_lines,
563 nb_columns,

565 matrix ,

573 Matrice_Morse& mat,
574 const bool& inverse )
575{
576 const int nb_lignes = mat.nb_lignes( );
577 const IntVect& tab1 = mat.get_tab1( );
578 const IntVect& tab2 = mat.get_tab2( );
579 int nnz_tot = 0;
580 for( int i=0; i<nb_lignes; i++ )
581 {
582 const int nnz_i = tab1[ i+1 ] - tab1[ i ]; //!< nnz sur la ligne i
583 for(int k=0; k<nnz_i; k++)
584 {
585 const int j = tab2[nnz_tot + k] - 1 ; //!< indice de la colonne
586 double& coefficient = mat.coef( i, j );
587 if( inverse )
588 coefficient *= 1./diag[ i ] ;
589 else
590 coefficient *= diag[ i ] ;
591 }
592 nnz_tot += nnz_i;
593 }
