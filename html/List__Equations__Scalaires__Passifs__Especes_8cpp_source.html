28Entree& List_Equations_Scalaires_Passifs_Especes::readOn(Entree& is)
29{
30// Ici se situe un gros piratage
31 Motcle accolade_ouverte("{");
32 Motcle accolade_fermee("}");
33 Motcle motlu;
34 is >> motlu;
35 if (motlu!=accolade_ouverte)
36 {
37 Cerr<<"Big problem in List_Equations_Scalaires_Passifs_Especes::readOn , we expected "<<accolade_ouverte;
38 Process::exit();
39 }
40 is >> motlu;
41 while (motlu!=accolade_fermee)
42 {
43 Equation toto;
44 Nom nume(list_eq.size());
45 Equation& Eqn_nvelle=list_eq.add(toto);
46 Eqn_nvelle.typer(motlu);

48
49// maintenant on associe le pb
50 Eqn.associer_pb_base(probleme());
51 Eqn.associer_milieu_base(mil.valeur());
52 Eqn.associer_sch_tps_base(schema_temps());
53
54// on la discretise (pas fait par discretiser)
55 Eqn.associer_zone_dis(probleme().domaine_dis().zone_dis(0));
56 Eqn.discretiser();
57
58// on change le nom de l'inconnue et de l equation
59 Nom nom;
60 nom=Eqn.inconnue().valeur().le_nom();
61 Nom nom_eq;
62 nom_eq = Eqn.le_nom();
63 nom+=nume;
64 Cerr<<"The unknown name is modified : new name "<<nom<<finl;
65 Eqn.inconnue().valeur().nommer(nom);
66 nom_eq+=nume;
67 Cerr<<"The equation name is modified : new name "<<nom_eq<<finl;
68 Eqn.nommer(nom_eq);
69
70// enfin on lit
71 is >> Eqn;
72 is >> motlu;
73 }
