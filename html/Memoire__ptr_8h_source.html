51// Supprime les donnees attachees a la zone de travail

77//! Ajout d'elements dans la zone de travail<br>Parametre: int n<br>Signification: le nombre d'element a creer dans la zone de travail
78/** \return Double_ptr_trav*<br>
79Signification: pointeur sur le zone de travail contenant les elements crees */
80template<typename _TYPE_>
81inline TRUST_ptr_trav<_TYPE_>* TRUST_ptr_trav<_TYPE_>::add(int n)
82{
83 if(sz==-1)
84 {
85 sz=n;
86 data=new _TYPE_[sz];
87 if(!data)
88 {
89 Cerr << "Not enough memory " << finl;
90 Process::exit();
91 }
92 lock_=1;
93 next=0;
94 return this;
95 }
96 if ( (!lock_) && (sz==n))
97 {
98 lock_=1;
99 return this;
100 }
101 TRUST_ptr_trav* curseur=this;
102 while(curseur->next)
103 {
104 curseur=curseur->next;
105 if ( (!curseur->lock_) && (curseur->sz==n))
106 {
107 curseur->lock_=1;
108 return curseur;
109 }
110 }
111 curseur->next= new TRUST_ptr_trav();
112 if(!curseur->next)
113 {
114 Cerr << "Not enough memory " << finl;
115 Process::exit();
116 }
117 curseur=curseur->next;
118 curseur->sz=n;
119 curseur->data=new _TYPE_[n];
120 if(!data)
121 {
122 Cerr << "Not enough memory " << finl;
123 Process::exit();
124 }
125 curseur->lock_=1;
126 curseur->next=0;
127 return curseur;
128}
