199 }

233DoubleTab& Op_Grad_EF::ajouter(const DoubleTab& pression, DoubleTab& grad) const
234{
235
236 test(pression,grad,(*this));
237// Gradient d'un champ scalaire localise aux centres des elements
238
239// On initialise grad a zero
240 grad = 0;
241 assert_espace_virtuel_vect(pression);
242 Debog::verifier("pression dans Op_Grad_EF",pression);
243 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
244
245
246 const DoubleTab& Bij_thilde=zone_ef.Bij_thilde();
247 int nb_elem_tot=zone_ef.zone().nb_elem_tot();
248 int nb_som_elem=zone_ef.zone().nb_som_elem();
249 const IntTab& elems=zone_ef.zone().les_elems() ;
250
251 for (int elem=0; elem<nb_elem_tot; elem++)
252 {
253 double pe=pression(elem);
254 for (int s=0; s<nb_som_elem; s++)
255 {
256 int som=elems(elem,s);
257 for (int i=0; i<dimension; i++)
258 {
259 grad(som,i)-=Bij_thilde(elem,s,i)*pe;
260 }
261 }
262 }
263
264 const IntTab& faces_sommets=zone_ef.face_sommets();
265 int nb_som_face=faces_sommets.dimension(1);
266// const IntTab& face_voisins = zone_ef.face_voisins();
267 const DoubleTab& face_normales = zone_ef.face_normales();
268 const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
269
270
271
272// prise en compte des " cl " ajout du temr - int P sur le bord
273// test ...
274 for (int n_bord=0; n_bord<zone_ef.nb_front_Cl()*1; n_bord++)
275 {
276 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
277 if (sub_type(Neumann_sortie_libre,la_cl.valeur()) )
278 continue;
279
280 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
281 int num1 = 0;
282 int num2 = + le_bord.nb_faces_tot();
283 if (sub_type(Dirichlet,la_cl.valeur()) )
284// ca ne sert a rien
285 for (int ind_face=num1; ind_face<num2; ind_face++)
286 {
287 int face=le_bord.num_face(ind_face);
288 for (int s=0; s<nb_som_face; s++)
289 {
290 int som=faces_sommets(face,s);
291 for (int comp=0; comp<dimension; comp++)
292 grad(som,comp) = 0;
293 }
294 }
295
296
297 else
298 for (int ind_face=num1; ind_face<num2*0; ind_face++)
299 {
300 int face=le_bord.num_face(ind_face);
301// int elem1 = face_voisins(face,0);
302// double diff = - pression[elem1]/nb_som_face;
303
304 for (int s=0; s<nb_som_face; s++)
305 {
306 int som=faces_sommets(face,s);
307// on calcule grad.S /S2
308 double grad_n=0;
309 for (int comp=0; comp<dimension; comp++)
310 grad_n+=grad(som,comp)*face_normales(face,comp);
311 grad_n/=(zone_ef.surface(face)*zone_ef.surface(face));
312 for (int comp=0; comp<dimension; comp++)
313 grad(som,comp) -= grad_n*face_normales(face,comp);
314 }
315 }
316 }
317 ajouter_bord(grad,zone_ef,la_zcl_EF.valeur(),int_P_bord_);
318 grad.echange_espace_virtuel();
319 return grad;
320}

329int Op_Grad_EF::impr(Sortie& os) const
330{
331 const int impr_mom=la_zone_EF->zone().Moments_a_imprimer();
332 const int impr_sum=(la_zone_EF->zone().Bords_a_imprimer_sum().est_vide() ? 0:1);
333 const int impr_bord=(la_zone_EF->zone().Bords_a_imprimer().est_vide() ? 0:1);
334 const Schema_Temps_base& sch = equation().probleme().schema_temps();
335 double temps = sch.temps_courant();
336 const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
337 const Zone_EF& zone_EF = la_zone_EF.valeur();
338 const IntTab& face_voisins = zone_EF.face_voisins();
339 const DoubleTab& face_normales = zone_EF.face_normales();
340 const Equation_base& eqn = equation();
341 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,eqn);
342 const Champ_P0_EF& la_pression_P0 = ref_cast(Champ_P0_EF,eqn_hydr.pression_pa().valeur());
343 const DoubleTab& pression_P0 = la_pression_P0.valeurs();
344 int n_bord, elem ;
345 int face;
346 double n0, n1, n2 ;
347 const ArrOfDouble& c_grav=la_zone_EF->zone().cg_moments();
348 flux_bords_.resize(zone_EF.nb_faces_bord(),dimension);
349 flux_bords_ = 0.;
350 int flag=je_suis_maitre();
351 SFichier Flux_grad;
352 ouvrir_fichier(Flux_grad,"",flag);
353 SFichier Flux_grad_moment;
354 ouvrir_fichier( Flux_grad_moment,"moment",impr_mom&&flag);
355 SFichier Flux_grad_sum;
356 ouvrir_fichier(Flux_grad_sum,"sum",impr_sum&&flag);
357 EcrFicPartage Flux_grad_face;
358 ouvrir_fichier_partage(Flux_grad_face,"",impr_bord);
359 if (je_suis_maitre())
360 {
361 Flux_grad.add_col(temps);
362 if (impr_mom) Flux_grad_moment.add_col(temps);
363 if (impr_sum) Flux_grad_sum.add_col(temps);
364 }
365 for ( n_bord=0; n_bord<zone_EF.nb_front_Cl(); n_bord++)
366 {
367 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
368 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
369
370 int ndeb = le_bord.num_premiere_face();
371 int nfin = ndeb + le_bord.nb_faces();
372
373 double fluxx_s = 0. ;
374 double fluxy_s = 0. ;
375 double fluxz_s = 0. ;
376 double fluxx_sum_s = 0. ;
377 double fluxy_sum_s = 0. ;
378 double fluxz_sum_s = 0. ;
379 double moment_x = 0.;
380 double moment_y = 0.;
381 double moment_z = 0.;
382 for (face=ndeb; face<nfin; face++)
383 {
384
385 elem = face_voisins(face,0);
386
387
388 if (dimension == 2)
389 {
390 n0 = face_normales(face,0) ;
391 n1 = face_normales(face,1) ;
392
393 flux_bords_(face,0) = pression_P0(elem)*n0 ;
394 flux_bords_(face,1) = pression_P0(elem)*n1 ;
395
396 fluxx_s += flux_bords_(face,0) ;
397 fluxy_s += flux_bords_(face,1) ;
398
399// Calcul du moment exerce par le fluide sur le bord (OM/\F)
400 if (impr_mom) moment_z+=flux_bords_(face,1)*c_grav[0]-flux_bords_(face,0)*c_grav[1];
401
402 if (zone_EF.zone().Bords_a_imprimer_sum().contient(le_bord.le_nom()))
403 {
404 fluxx_sum_s += flux_bords_(face,0) ;
405 fluxy_sum_s += flux_bords_(face,1) ;
406 }
407 }
408 else if (dimension == 3)
409 {
410 n0 = face_normales(face,0) ;
411 n1 = face_normales(face,1) ;
412 n2 = face_normales(face,2) ;
413
414 flux_bords_(face,0) = pression_P0(elem)*n0 ;
415 flux_bords_(face,1) = pression_P0(elem)*n1 ;
416 flux_bords_(face,2) = pression_P0(elem)*n2 ;
417
418 fluxx_s += flux_bords_(face,0) ;
419 fluxy_s += flux_bords_(face,1) ;
420 fluxz_s += flux_bords_(face,2) ;
421
422// Calcul du moment exerce par le fluide sur le bord (OM/\F)
423 moment_x+=flux_bords_(face,2)*c_grav[1]-flux_bords_(face,1)*c_grav[2];
424 moment_y+=flux_bords_(face,0)*c_grav[2]-flux_bords_(face,2)*c_grav[0];
425 moment_z+=flux_bords_(face,1)*c_grav[0]-flux_bords_(face,0)*c_grav[1];
426 if (zone_EF.zone().Bords_a_imprimer_sum().contient(le_bord.le_nom()))
427 {
428 fluxx_sum_s += flux_bords_(face,0) ;
429 fluxy_sum_s += flux_bords_(face,1) ;
430 fluxz_sum_s += flux_bords_(face,2) ;
431 }
432 }
433 }
434 if (dimension == 2)
435 {
436 fluxx_s=Process::mp_sum(fluxx_s);
437 fluxy_s=Process::mp_sum(fluxy_s);
438 fluxx_sum_s=Process::mp_sum(fluxx_sum_s);
439 fluxy_sum_s=Process::mp_sum(fluxy_sum_s);
440 moment_z=Process::mp_sum(moment_z);
441 if (je_suis_maitre())
442 {
443 Flux_grad.add_col(fluxx_s);
444 Flux_grad.add_col(fluxy_s);
445 if (impr_sum)
446 {
447 Flux_grad_sum.add_col(fluxx_sum_s);
448 Flux_grad_sum.add_col(fluxy_sum_s);
449 }
450 if (impr_mom) Flux_grad_moment.add_col(moment_z);
451 }
452 }
453 if (dimension == 3)
454 {
455 fluxx_s=Process::mp_sum(fluxx_s);
456 fluxy_s=Process::mp_sum(fluxy_s);
457 fluxz_s=Process::mp_sum(fluxz_s);
458 fluxx_sum_s=Process::mp_sum(fluxx_sum_s);
459 fluxy_sum_s=Process::mp_sum(fluxy_sum_s);
460 fluxz_sum_s=Process::mp_sum(fluxz_sum_s);
461 moment_x=Process::mp_sum(moment_x);
462 moment_y=Process::mp_sum(moment_y);
463 moment_z=Process::mp_sum(moment_z);
464 if(je_suis_maitre())
465 {
466 Flux_grad.add_col(fluxx_s);
467 Flux_grad.add_col(fluxy_s);
468 Flux_grad.add_col(fluxz_s);
469 if (impr_sum)
470 {
471 Flux_grad_sum.add_col(fluxx_sum_s);
472 Flux_grad_sum.add_col(fluxy_sum_s);
473 Flux_grad_sum.add_col(fluxz_sum_s);
474 }
475 if (impr_mom)
476 {
477 Flux_grad_moment.add_col(moment_x);
478 Flux_grad_moment.add_col(moment_y);
479 Flux_grad_moment.add_col(moment_z);
480 }
481 }
482 }
483 }
484 if (je_suis_maitre())
485 {
486 Flux_grad << finl;
487 if (impr_mom) Flux_grad_moment<<finl;
488 if (impr_sum) Flux_grad_sum<<finl;
489 }
490 for ( n_bord=0; n_bord<zone_EF.nb_front_Cl(); n_bord++)
491 {
492 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
493 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
494
495 int ndeb = le_bord.num_premiere_face();
496 int nfin = ndeb + le_bord.nb_faces();
497 if (zone_EF.zone().Bords_a_imprimer().contient(le_bord.le_nom()))
498 {
499 if (je_suis_maitre()) Flux_grad_face << "# Force par face sur " << le_bord.le_nom() << " au temps " << temps << " : " << finl;
500 for (face=ndeb; face<nfin; face++)
501 {
502 if (dimension==2)
503 Flux_grad_face << "# Face a x= " << zone_EF.xv(face,0) << " y= " << zone_EF.xv(face,1) << " flux_x= " << flux_bords_(face,0) << " flux_y= " << flux_bords_(face,1) << finl;
504 else if (dimension == 3)
505 Flux_grad_face << "# Face a x= " << zone_EF.xv(face,0) << " y= " << zone_EF.xv(face,1) << " z= " << zone_EF.xv(face,3) << " flux_x= " << flux_bords_(face,0) << " flux_y= " << flux_bords_(face,1) << " flux_z= " << flux_bords_(face,2) << finl;
506 }
507 Flux_grad_face.syncfile();
508 }
509 }
510
511 return 1;
512}
