62// -Cas CL periodique : assure que les valeurs sur des faces periodiques
63// en vis a vis sont identiques. Pour cela on prend la demi somme des deux valeurs.
64void Champ_Q1NC::verifie_valeurs_cl()
65{
66 const Zone_Cl_dis_base& zcl = equation().zone_Cl_dis().valeur();
67 int nb_cl = zcl.nb_cond_lim();
68 DoubleTab& ch_tab = valeurs();
69 int nb_compo = nb_comp();
70 int ndeb,nfin,num_face;
71
72 for (int i=0; i<nb_cl; i++)
73 {
74 const Cond_lim_base& la_cl = zcl.les_conditions_limites(i).valeur();
75 if (sub_type(Periodique,la_cl))
76 {
77 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl);
78 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
79 ndeb = le_bord.num_premiere_face();
80 nfin = ndeb + le_bord.nb_faces();
81 int voisine;
82 double moy;
83
84 for (num_face=ndeb; num_face<nfin; num_face++)
85 {
86 voisine = la_cl_perio.face_associee(num_face-ndeb) + ndeb;
87 for (int comp=0; comp<nb_compo; comp++)
88 {
89 if (ch_tab(num_face,comp) != ch_tab(voisine,comp))
90 {
91 moy = 0.5*(ch_tab(num_face,comp) + ch_tab(voisine,comp));
92 ch_tab(num_face,comp) = moy;
93 ch_tab(voisine,comp) = moy;
94 }
95 }
96 }
97 }
98 }
99 ch_tab.echange_espace_virtuel();
100}

103{

116void Champ_Q1NC::cal_rot_ordre1(DoubleTab& vorticite)
117{
118 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
119 const int nb_elem = zone_VEF.nb_elem();
120
121 DoubleTab gradient_elem(0, dimension, dimension);
122// le tableau est initialise dans la methode gradient():
123 zone_VEF.zone().creer_tableau_elements(gradient_elem, Array_base::NOCOPY_NOINIT);
124
125 gradient(gradient_elem);
126 Debog::verifier("apres calcul gradient",gradient_elem);
127 int num_elem;
128
129 switch(dimension)
130 {
131 case 2 :
132 {
133 for (num_elem=0; num_elem<nb_elem; num_elem++)
134 {
135 vorticite(num_elem)=gradient_elem(num_elem,1,0)-gradient_elem(num_elem,0,1);
136 }
137 }
138 break;
139 case 3 :
140 {
141 for (num_elem=0; num_elem<nb_elem; num_elem++)
142 {
143 vorticite(num_elem,0)=gradient_elem(num_elem,2,1)-gradient_elem(num_elem,1,2);
144 vorticite(num_elem,1)=gradient_elem(num_elem,0,2)-gradient_elem(num_elem,2,0);
145 vorticite(num_elem,2)=gradient_elem(num_elem,1,0)-gradient_elem(num_elem,0,1);
146 }
147 }
148 }
149
150 Debog::verifier("apres calcul vort",vorticite);
151 return ;
152}

154void Champ_Q1NC::gradient(DoubleTab& gradient_elem)
155{
156// Calcul du gradient de la vitesse pour le calcul de la vorticite
157 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
158 const DoubleTab& vitesse = equation().inconnue().valeurs();
159
160 const DoubleTab& face_normales = zone_VEF.face_normales();
161 const int nb_faces = zone_VEF.nb_faces();
162 const int nb_elem = zone_VEF.nb_elem();
163 const IntTab& face_voisins = zone_VEF.face_voisins();
164 int premiere_face_int = zone_VEF.premiere_face_int();
165 const DoubleVect& volumes = zone_VEF.volumes();
166
167 assert(gradient_elem.dimension(0) == nb_elem);
168 assert(gradient_elem.dimension(1) == dimension);
169 assert(gradient_elem.dimension(2) == dimension);
170 operator_egal(gradient_elem, 0.); //!< Espace reel uniquement
171 int icomp,fac,i,elem1,elem2;
172
173// Boucle sur les faces
174
175 for (fac=0; fac< premiere_face_int; fac++)
176 {
177 elem1=face_voisins(fac,0);
178 for (icomp=0; icomp<dimension; icomp++)
179 for (i=0; i<dimension; i++)
180 {
181 gradient_elem(elem1, icomp, i) +=
182 face_normales(fac,i)*vitesse(fac,icomp);
183 }
184 }
185 for (; fac<nb_faces; fac++)
186 {
187 elem1=face_voisins(fac,0);
188 elem2=face_voisins(fac,1);
189 for (icomp=0; icomp<dimension; icomp++)
190 for (i=0; i<dimension; i++)
191 {
192 gradient_elem(elem1, icomp, i) +=
193 face_normales(fac,i)*vitesse(fac,icomp);
194 if(elem2<nb_elem)
195 gradient_elem(elem2, icomp, i) -=
196 face_normales(fac,i)*vitesse(fac,icomp);
197 }
198 }
199
200 tab_divide_any_shape(gradient_elem, volumes);
201}
