37void Source_Generique_VDF_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
38{
39 Champ espace_stockage;
40 const Champ_base& champ_calc = ch_source_->get_champ(espace_stockage);
41 const DoubleTab& valeurs_calc = champ_calc.valeurs();
42
43 int nb_faces = la_zone_VDF->nb_faces();
44 const DoubleVect& vol_entrelaces = la_zone_VDF->volumes_entrelaces();
45 const DoubleVect& poro_face = la_zone_VDF->porosite_face();
46 int nb_front_cl = la_zone_VDF->nb_front_Cl();
47 int premiere_face_interne = la_zone_VDF->premiere_face_int();
48 int num_face;
49
50 for (int num_cl =0; num_cl<nb_front_cl; num_cl++)
51 {
52 const Cond_lim& la_cl = la_zcl_VDF->les_conditions_limites(num_cl);
53 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
54 int ndeb = le_bord.num_premiere_face();
55 int nfin = ndeb + le_bord.nb_faces();
56 if ((sub_type(Dirichlet,la_cl.valeur()))
57 ||
58 (sub_type(Dirichlet_homogene,la_cl.valeur()))
59 )
60 ;
61 else
62 for (num_face=ndeb; num_face<nfin; num_face++)
63 {
64 secmem(num_face) += valeurs_calc(num_face)*vol_entrelaces(num_face)*poro_face(num_face);
65 }
66 }
67
68 for (num_face=premiere_face_interne; num_face<nb_faces; num_face++)
69 secmem(num_face) += valeurs_calc(num_face)*vol_entrelaces(num_face)*poro_face(num_face);
70
71 secmem.echange_espace_virtuel();
72}
