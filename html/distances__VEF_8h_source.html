50inline double vitesse_tangentielle(const double v0,const double v1,const double r0,const double r1)
51{
52// On prend std::fabs car mathematiquement la valeur est >=0
53 return sqrt(std::fabs(carre(v0)+carre(v1)-carre(v0*r0+v1*r1)));
54}

61inline void calcule_r0r1(const DoubleTab& face_normale, int& fac, double& r0, double& r1)
62{
63 r0=face_normale(fac,0);
64 r1=face_normale(fac,1);
65 double tmp=1/sqrt(r0*r0+r1*r1);
66 r0*=tmp;
67 r1*=tmp;

81inline double distance_face(int fac,int fac1,const Zone_VEF& zone)
82{
83 int dimension=Objet_U::dimension;
84 const DoubleTab& xv = zone.xv(); //!< centre de gravite des faces
85 const DoubleTab& face_normale = zone.face_normales();
86 double r0,r1;
87 double x0=xv(fac,0);
88 double y0=xv(fac,1);
89 double x1=xv(fac1,0);
90 double y1=xv(fac1,1);
91 if (dimension==3)
92 {
93 double r2;
94 double z0=xv(fac,2);
95 double z1=xv(fac1,2);
96 calcule_r0r1r2(face_normale,fac,r0,r1,r2);
97 return std::fabs(r0*(x1-x0)+r1*(y1-y0)+r2*(z1-z0));
98 }
99 else
100 {
101 calcule_r0r1(face_normale,fac,r0,r1);
102 return std::fabs(r0*(x1-x0)+r1*(y1-y0));
103 }
104}
