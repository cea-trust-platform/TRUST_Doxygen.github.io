90int PE_Groups::enter_group(const Comm_Group& group)
91{
92 assert(&group != &current_group());
93 if (ngroups >= max_ngroups-1)
94 {
95 Cerr << "Comm_Group::enter_group : fatal, too many groups" << finl;
96 Process::exit();
97 }
98 int my_rank_in_group = rank_translate(current_group().rank(), current_group(), group);
99 if (my_rank_in_group >= 0)
100 {
101// Sauvegarde du pointeur sur le groupe actuel et changement du
102// current_group() :
103 groups[ngroups] = group;
104 current_group_ = &group;
105 ngroups++;
106// Attention, on a change de current_group() !
107
108// On verifie que tous les processeurs du nouveau groupe sont la:
109 if (Comm_Group::check_enabled())
110 current_group().barrier(0);
111
112 return 1;
113 }
114 else
115 {
116 return 0;
117 }
118}

124void PE_Groups::exit_group()
125{
126 if (Comm_Group::check_enabled())
127 current_group().barrier(0);
128
129 if (ngroups <= 1)
130 {
131 Cerr << "Comm_Group::exit_group() error : trying to exit from TRUST main group." << finl;
132 Process::exit();
133 }
134 ngroups--;
135 current_group_ = &(groups[ngroups-1].valeur());
136}
