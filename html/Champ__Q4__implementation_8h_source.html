85//! Calcule la coordonnee barycentrique d'un point (x,y) par<br>rapport au sommet specifie d'un triangle ou d'un rectange (un element)<br>Ce calcul concerne un point 2D.
86/** \return double<br>
87Signification: la coordonnee barycentrique du point (x,y) par rapport<br>
88au sommet specifie (i) dans l'element specifie (le_poly)<br>
89Contraintes: 0 <= valeur <= 1<br>
90Excpetion: un triangle n'a pas plus de 3 sommets<br>
91Excpetion: un rectangle n'a pas plus de 4 sommets */
92inline double coord_barycentrique(const IntTab& polys, const DoubleTab& coord, double x, double y, int le_poly, int i)
93{
94 int som0,som1,som2;
95 int nb_som_elem = polys.dimension(1);
96// Distinction du calcul de la coordonnee barycentrique en fonction du type de l element
97// Cas Triangle
98 if (nb_som_elem==3)
99 {
100 switch(i)
101 {
102 case 0 :
103 {
104 som0=polys(le_poly , 0);
105 som1=polys(le_poly , 1);
106 som2=polys(le_poly , 2);
107 break;
108 }
109 case 1 :
110 {
111 som0=polys(le_poly , 1);
112 som1=polys(le_poly , 2);
113 som2=polys(le_poly , 0);
114 break;
115 }
116 case 2 :
117 {
118 som0=polys(le_poly , 2);
119 som1=polys(le_poly , 0);
120 som2=polys(le_poly , 1);
121 break;
122 }
123 default :
124 {
125 som0=-1;
126 som1=-1;
127 som2=-1;
128 Cerr << "Error in Champ_P1::coord_barycentrique : " << finl;
129 Cerr << "A triangle does not have " << i << "nodes " << finl;
130 Process::exit();
131 }
132
133 }
134 double den = (coord(som2,0)-coord(som1,0))*(coord(som0,1)-coord(som1,1))
135 - (coord(som2,1)-coord(som1,1))*(coord(som0,0)-coord(som1,0));
136
137 double num = (coord(som2,0)-coord(som1,0))*(y-coord(som1,1))
138 - (coord(som2,1)-coord(som1,1))*(x-coord(som1,0));
139
140 assert (den != 0.);
141 double coord_bary=num/den;
142 if ((coord_bary < -Objet_U::precision_geom) || (coord_bary > 1.+Objet_U::precision_geom))
143 {
144 Cerr << "WARNING: The barycentric coordinate of point :" << finl;
145 Cerr << "x= " << x << " y=" << y << finl;
146 Cerr << "is not between 0 and 1 : " << coord_bary << finl;
147 Cerr << "On the element " << le_poly << " of the processor " << Process::me() << finl;
148 }
149 return coord_bary;
150 }
151// Cas Rectangle
152 else if (nb_som_elem==4)
153 {
154 double alpha_x=0.;
155 double alpha_y=0.;
156 double delta_x, delta_y;
157// int som0,som1,som2;
158 double x0,y0;
159
160 switch(i)
161 {
162 case 0 :
163 {
164 alpha_x = -1.;
165 alpha_y = -1;
166
167 break;
168 }
169 case 1 :
170 {
171 alpha_x = 1.;
172 alpha_y = -1.;
173
174 break;
175 }
176 case 2 :
177 {
178 alpha_x = -1.;
179 alpha_y = 1.;
180
181 break;
182 }
183 case 3 :
184 {
185 alpha_x = 1.;
186 alpha_y = 1.;
187
188 break;
189 }
190 default :
191 {
192 Cerr << "Error in Champ_P1::coord_barycentrique : " << finl;
193 Process::exit();
194 }
195
196 }
197
198 som0=polys(le_poly , 0);
199 som1=polys(le_poly , 1);
200 som2=polys(le_poly , 2);
201 delta_x = coord(som1,0)-coord(som0,0);
202 delta_y = coord(som2,1)-coord(som0,1);
203 x0 = coord(som0,0)+delta_x/2.;
204 y0 = coord(som0,1)+delta_y/2.;
205 double coord_bary = 0.25*(1.+2.*alpha_x*(x-x0)/delta_x)*(1.+2.*alpha_y*(y-y0)/delta_y);
206
207
208 if ((coord_bary < -Objet_U::precision_geom) || (coord_bary > 1.+Objet_U::precision_geom))
209 {
210 Cerr << "WARNING: The barycentric coordinate of point :" << finl;
211 Cerr << "x= " << x << " y=" << y << finl;
212 Cerr << "is not between 0 and 1 : " << coord_bary << finl;
213 Cerr << "On the element " << le_poly << " of the processor " << Process::me() << finl;
214 }
215 return coord_bary;
216 }
217
218 Cerr<<"The number of nodes by element " <<nb_som_elem<<" does not correspond to a treated situation."<<finl;
219 Process::exit();
220 return 0.;
221}

279//! Calcule la coordonnee barycentrique d'un point (x,y,z) par<br>rapport au sommet specifie d'un tetraedre ou d'un hexaedre (un element)<br>Ce calcul concerne un point 3D.
280/** \return double<br>
281Signification: la coordonnee barycentrique du point (x,y,z) par rapport<br>
282au sommet specifie (i) dans l'element specifie (le_poly)<br>
283Contraintes: 0 <= valeur <= 1 */
284inline double coord_barycentrique(const IntTab& polys, const DoubleTab& coord, double x, double y, double z, int le_poly, int i)
285{
286 int som0,som1,som2,som3;
287 int nb_som_elem = polys.dimension(1);
288// Distinction du calcul de la coordonnee barycentrique en fonction du type de l element
289// Cas Tetraedre
290 if (nb_som_elem==4)
291 {
292 switch(i)
293 {
294 case 0 :
295 {
296 som0=polys(le_poly , 0);
297 som1=polys(le_poly , 1);
298 som2=polys(le_poly , 2);
299 som3=polys(le_poly , 3);
300 break;
301 }
302 case 1 :
303 {
304 som0=polys(le_poly , 1);
305 som1=polys(le_poly , 2);
306 som2=polys(le_poly , 3);
307 som3=polys(le_poly , 0);
308 break;
309 }
310 case 2 :
311 {
312 som0=polys(le_poly , 2);
313 som1=polys(le_poly , 3);
314 som2=polys(le_poly , 0);
315 som3=polys(le_poly , 1);
316 break;
317 }
318 case 3 :
319 {
320 som0=polys(le_poly , 3);
321 som1=polys(le_poly , 0);
322 som2=polys(le_poly , 1);
323 som3=polys(le_poly , 2);
324 break;
325 }
326 default :
327 {
328 som0=-1;
329 som1=-1;
330 som2=-1;
331 som3=-1;
332 Cerr << "Error in Champ_P1::coord_barycentrique : " << finl;
333 Cerr << "A tetrahedron does not have " << i << "nodes " << finl;
334 Process::exit();
335 }
336 }
337
338 double xp = (coord(som2,1)-coord(som1,1))*(coord(som0,2)-coord(som1,2))
339 - (coord(som2,2)-coord(som1,2))*(coord(som0,1)-coord(som1,1));
340 double yp = (coord(som2,2)-coord(som1,2))*(coord(som0,0)-coord(som1,0))
341 - (coord(som2,0)-coord(som1,0))*(coord(som0,2)-coord(som1,2));
342 double zp = (coord(som2,0)-coord(som1,0))*(coord(som0,1)-coord(som1,1))
343 - (coord(som2,1)-coord(som1,1))*(coord(som0,0)-coord(som1,0));
344 double den = xp * (coord(som3,0)-coord(som1,0))
345 + yp * (coord(som3,1)-coord(som1,1))
346 + zp * (coord(som3,2)-coord(som1,2));
347
348 xp = (coord(som2,1)-coord(som1,1))*(z-coord(som1,2))
349 - (coord(som2,2)-coord(som1,2))*(y-coord(som1,1));
350 yp = (coord(som2,2)-coord(som1,2))*(x-coord(som1,0))
351 - (coord(som2,0)-coord(som1,0))*(z-coord(som1,2));
352 zp = (coord(som2,0)-coord(som1,0))*(y-coord(som1,1))
353 - (coord(som2,1)-coord(som1,1))*(x-coord(som1,0));
354 double num = xp * (coord(som3,0)-coord(som1,0))
355 + yp * (coord(som3,1)-coord(som1,1))
356 + zp * (coord(som3,2)-coord(som1,2));
357
358 assert (den != 0.);
359 double coord_bary = num/den;
360 if ((coord_bary < -Objet_U::precision_geom) || (coord_bary > 1.+Objet_U::precision_geom))
361 {
362 Cerr << "WARNING: The barycentric coordinate of point :" << finl;
363 Cerr << "x= " << x << " y=" << y << " z=" << z << finl;
364 Cerr << "is not between 0 and 1 : " << coord_bary << finl;
365 Cerr << "On the element " << le_poly << " of the processor " << Process::me() << finl;
366 }
367 return coord_bary;
368 }
369// Cas Hexaedre
370 else if (nb_som_elem==8)
371 {
372 double alpha_x=0.;
373 double alpha_y=0.;
374 double alpha_z=0.;
375 double delta_x, delta_y, delta_z;
376// int som0,som1,som2;
377 double x0,y0,z0;
378
379 switch(i)
380 {
381 case 0 :
382 {
383 alpha_x = -1.;
384 alpha_y = -1.;
385 alpha_z = -1.;
386 break;
387 }
388 case 1 :
389 {
390 alpha_x = 1.;
391 alpha_y = -1.;
392 alpha_z = -1.;
393 break;
394 }
395 case 2 :
396 {
397 alpha_x = -1.;
398 alpha_y = 1.;
399 alpha_z = -1.;
400 break;
401 }
402 case 3 :
403 {
404 alpha_x = 1.;
405 alpha_y = 1.;
406 alpha_z = -1.;
407 break;
408 }
409 case 4 :
410 {
411 alpha_x = -1.;
412 alpha_y = -1;
413 alpha_z = 1.;
414 break;
415 }
416 case 5 :
417 {
418 alpha_x = 1.;
419 alpha_y = -1.;
420 alpha_z = 1.;
421 break;
422 }
423 case 6 :
424 {
425 alpha_x = -1.;
426 alpha_y = 1.;
427 alpha_z = 1.;
428 break;
429 }
430 case 7 :
431 {
432 alpha_x = 1.;
433 alpha_y = 1.;
434 alpha_z = 1.;
435 break;
436 }
437 default :
438 {
439 Cerr << "Error in Champ_P1::coord_barycentrique : " << finl;
440 Process::exit();
441 }
442 }
443
444 som0=polys(le_poly , 0);
445 som1=polys(le_poly , 1);
446 som2=polys(le_poly , 2);
447 som3=polys(le_poly , 4);
448
449 delta_x = coord(som1,0)-coord(som0,0);
450 delta_y = coord(som2,1)-coord(som0,1);
451 delta_z = coord(som3,2)-coord(som0,2);
452
453 x0 = coord(som0,0)+delta_x/2.;
454 y0 = coord(som0,1)+delta_y/2.;
455 z0 = coord(som0,2)+delta_z/2.;
456
457 double coord_bary = (1./8.)*(1.+2.*alpha_x*(x-x0)/delta_x)*(1.+2.*alpha_y*(y-y0)/delta_y)*(1.+2.*alpha_z*(z-z0)/delta_z);
458
459
460 if ((coord_bary < -Objet_U::precision_geom) || (coord_bary > 1.+Objet_U::precision_geom))
461 {
462 Cerr << "WARNING: The barycentric coordinate of point :" << finl;
463 Cerr << "x= " << x << " y=" << y <<" z="<< z <<finl;
464 Cerr << "is not between 0 and 1 : " << coord_bary << finl;
465 Cerr << "On the element " << le_poly << " of the processor " << Process::me() << finl;
466 }
467 return coord_bary;
468 }
469
470 Cerr<<"The number of nodes by element " <<nb_som_elem<<" does not correspond to a treated situation."<<finl;
471 Process::exit();
472 return 0.;
