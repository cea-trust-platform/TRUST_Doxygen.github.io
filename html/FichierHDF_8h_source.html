47//! This abstract class provides all the functionalities to open and manipulate HDF files and related<br>concepts (datasets, groups, etc ...)<br>It intentionally does not place itself in the Sortie_Fichier_base or Entree_Fichier_base hierarchy<br>since it is rather the datasets inside the HDF file that are regarded as TRUST Fichier objects.
48//
49class FichierHDF
50{
51public:
52 FichierHDF();
53 virtual ~FichierHDF();
54
55// Creates (and open) the HDF file.
56 virtual void create(Nom filename);
57 virtual void open(Nom filename, bool readOnly);
58 virtual void close();
59
60// Multiple Writer methods:
61// every processor writes their own dataset into the file
62 virtual void create_and_fill_dataset_MW(Nom dataset_basename, Sortie_Brute& sortie);
63 virtual void create_and_fill_dataset_MW(Nom dataset_basename, SChaine& sortie);
64
65// Method to write the dataset owned by proc #proc_rank in the given file
66// (to use when a single proc writes all the datasets for example)
67 virtual void create_and_fill_dataset_SW(Nom datasetname, Sortie_Brute& sortie);
68
69// Method to read the dataset owned by proc #proc_rank in the given file
70 virtual void read_dataset(Nom dataset_basename, int proc_rank, Entree_Brute& entree);

72// Method to create datasets of size length and whose names are given by dataset_names

90 virtual void prepare_dataset_props();

93
94 hid_t file_id_;
95 hid_t file_access_plst_;
96 hid_t dataset_transfer_plst_;
97 bool is_different_int_size_=false; //!< File with int32 (or int64) whereas version is int64 (or int32)

99
