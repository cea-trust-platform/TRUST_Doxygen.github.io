42template <typename Type_Double>
43void Eval_Forchheimer_VDF_Face::calculer_terme_source(const int num_face, Type_Double& source) const
44{
45 const int size = source.size_array();
46 for (int i = 0; i < size; i++)
47 {
48 const double U = (vitesse->valeurs())(num_face,i);
49 source[i] = -Cf/sqrt(modK->getK(porosite))*volumes_entrelaces(num_face)*porosite_surf(num_face)*std::fabs(U)*U;
50 }
51}
