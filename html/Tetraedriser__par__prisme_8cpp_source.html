77//! Fonction hors classe<br>Decoupe toutes les faces d'un objet Faces<br>dont les faces ont 4 sommets<br>en 2 faces a 3 sommets.
78static void decoupe(Zone& zone, Faces& faces)
79{
80
81 const Domaine& dom=zone.domaine();
82 const DoubleTab& coord=dom.coord_sommets();
83
84
85 IntTab& sommets=faces.les_sommets();
86 int nb_faces=sommets.dimension(0);
87 assert(sommets.dimension(1)==4);
88 IntTab nouveaux(2*nb_faces, 3);
89 faces.voisins().resize(2*nb_faces, 2);
90 faces.voisins()=-1;
91 for(int i=0; i<nb_faces; i++)
92 {
93 int i1 = sommets(i,0);
94 int i2 = sommets(i,1);
95 int i3 = sommets(i,2);
96 int i4 = sommets(i,3);
97
98// C'est une face a X constant
99 if((coord(i1,0)==coord(i2,0)) && (coord(i1,0)==coord(i3,0)))
100 {
101 nouveaux(i,0) = i1;
102 nouveaux(i,1) = i2;
103 nouveaux(i,2) = i3;
104 nouveaux(nb_faces+i,0) = i2;
105 nouveaux(nb_faces+i,1) = i3;
106 nouveaux(nb_faces+i,2) = i4;
107 }
108
109// C'est une face a Z constant
110 if((coord(i1,2)==coord(i2,2)) && (coord(i1,2)==coord(i3,2)))
111 {
112 nouveaux(i,0) = i1;
113 nouveaux(i,1) = i2;
114 nouveaux(i,2) = i3;
115 nouveaux(nb_faces+i,0) = i2;
116 nouveaux(nb_faces+i,1) = i3;
117 nouveaux(nb_faces+i,2) = i4;
118 }
119// C'est une face a Y constant
120 if((coord(i1,1)==coord(i2,1)) && (coord(i1,1)==coord(i3,1)))
121 {
122 nouveaux(i,0) = i1;
123 nouveaux(i,1) = i2;
124 nouveaux(i,2) = i4;
125 nouveaux(nb_faces+i,0) = i1;
126 nouveaux(nb_faces+i,1) = i3;
127 nouveaux(nb_faces+i,2) = i4;
128 }
129
130 }
131 sommets.ref(nouveaux);
132}

155//! Tetraedrise tous les elements d'une zone : transforme<br>les elements goemetriques de la zone en tetraedres IDENTIQUES<br>par la methode des prismes.<br>Pour l'instant on ne sait tetraedriser que des Hexaedre.<br>(on les coupe en 2).<br>Les elements sont tetraedrises et tous les bords<br>sont types en Triangle_3D.
156void Tetraedriser_par_prisme::trianguler(Zone& zone) const
157{
158 if(zone.type_elem()->que_suis_je() == "Hexaedre")
159 {
160 zone.typer("Tetraedre");
161 IntTab& les_elems=zone.les_elems();
162 int oldsz=les_elems.dimension(0);
163 IntTab new_elems(6*oldsz, 4);
164 for(int i=0; i< oldsz; i++)
165 {
166 int i0=les_elems(i,0);
167 int i1=les_elems(i,1);
168 int i2=les_elems(i,2);
169 int i3=les_elems(i,3);
170 int i4=les_elems(i,4);
171 int i5=les_elems(i,5);
172 int i6=les_elems(i,6);
173 int i7=les_elems(i,7);
174 {
175 new_elems(i,0)=i0;
176 new_elems(i,1)=i1;
177 new_elems(i,2)=i2;
178 new_elems(i,3)=i5;
179
180 new_elems(i+oldsz,0)=i0;
181 new_elems(i+oldsz,1)=i2;
182 new_elems(i+oldsz,2)=i4;
183 new_elems(i+oldsz,3)=i5;
184 mettre_a_jour_sous_zone(zone,i,i+oldsz,1);
185
186 new_elems(i+2*oldsz,0)=i1;
187 new_elems(i+2*oldsz,1)=i2;
188 new_elems(i+2*oldsz,2)=i3;
189 new_elems(i+2*oldsz,3)=i5;
190 mettre_a_jour_sous_zone(zone,i,i+2*oldsz,1);
191
192 new_elems(i+3*oldsz,0)=i2;
193 new_elems(i+3*oldsz,1)=i3;
194 new_elems(i+3*oldsz,2)=i5;
195 new_elems(i+3*oldsz,3)=i7;
196 mettre_a_jour_sous_zone(zone,i,i+3*oldsz,1);
197
198 new_elems(i+4*oldsz,0)=i2;
199 new_elems(i+4*oldsz,1)=i4;
200 new_elems(i+4*oldsz,2)=i5;
201 new_elems(i+4*oldsz,3)=i6;
202 mettre_a_jour_sous_zone(zone,i,i+4*oldsz,1);
203
204 new_elems(i+5*oldsz,0)=i2;
205 new_elems(i+5*oldsz,1)=i5;
206 new_elems(i+5*oldsz,2)=i6;
207 new_elems(i+5*oldsz,3)=i7;
208 mettre_a_jour_sous_zone(zone,i,i+5*oldsz,1);
209 }
210 }
211 les_elems.ref(new_elems);
212 }
213 else
214 {
215 Cerr << "We do not yet know how to Tetraedriser_par_prisme the "
216 << zone.type_elem()->que_suis_je() <<"s"<<finl;
217 }
218 {
219 LIST_CURSEUR(Bord) curseur(zone.faces_bord());;;
220 while(curseur)
221 {
222 Faces& les_faces=curseur->faces();
223 les_faces.typer(Faces::triangle_3D);
224// decoupe(les_faces);
225 decoupe(zone,les_faces);
226 ++curseur;
227 }
228 }
229
230 {

232 LIST_CURSEUR(Raccord) curseur(zone.faces_raccord());;;
233 while(curseur)
234 {
235 Faces& les_faces=curseur.valeur()->faces();
236 les_faces.typer(Faces::triangle_3D);
237// decoupe(les_faces);
238 decoupe(zone,les_faces);
239 ++curseur;
240 }
241 }
242 {
243// Les Faces internes
244 LIST_CURSEUR(Faces_Interne) curseur(zone.faces_int());;;
245 while(curseur)
246 {
247 Faces& les_faces=curseur->faces();
248 les_faces.typer(Faces::triangle_3D);
249// decoupe(les_faces);
250 decoupe(zone,les_faces);
251 ++curseur;
252 }
253 }
