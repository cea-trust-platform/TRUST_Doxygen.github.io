52template <typename Type_Double>
53void Eval_Echange_Himp_VDF_Elem::calculer_terme_source(const int num_elem, Type_Double& source) const
54{
55 const int size = source.size_array();
56 for (int i = 0; i < size; i++) source[i] = h_*(Tvois(num_elem,i)-Tcourant(num_elem,i)) * volumes(num_elem)*porosite_vol(num_elem);
