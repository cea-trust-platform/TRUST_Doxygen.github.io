39void Champ_front_fonc_gradient_VDF::mettre_a_jour(double temps)
40{
41// Cerr << "Champ_front_fonc_gradient_VDF::mettre_a_jour" << finl;
42 const Front_VF& la_front=ref_cast(Front_VF, frontiere_dis());
43 int nb_faces = la_front.nb_faces();
44 int ndeb = la_front.num_premiere_face();
45
46 DoubleTab& tab=valeurs_au_temps(temps);
47
48 const Zone_VDF& la_zone_VDF = ref_cast(Zone_VDF,zone_dis());
49 const IntVect& orientation = la_zone_VDF.orientation();
50// On recupere la CL et les champs associes
51 const Paroi_Knudsen_non_negligeable& la_condition_limite=ref_cast(Paroi_Knudsen_non_negligeable,condition_limite(la_front.le_nom()));
52 const DoubleTab& la_inconnue_paroi_=la_condition_limite.vitesse_paroi();
53 const DoubleTab& k_=la_condition_limite.k();
54 const DoubleVect& la_inconnue=inconnue->valeurs();
55 int elem,iro,ori,i,j;
56// double inconnue_moyenne1,inconnue_moyenne2;
57
58 DoubleTab la_inconnue_paroi(nb_faces,dimension);
59 DoubleVect k(nb_faces);
60 DoubleVect inconnue_moyenne(dimension-1);
61 if (la_inconnue_paroi_.dimension(0)==1)
62 for (i=0; i<nb_faces; i++)
63 for (j=0; j<dimension; j++)
64 la_inconnue_paroi(i,j)=la_inconnue_paroi_(0,j);
65 else
66 for (i=0; i<nb_faces; i++)
67 for (j=0; j<dimension; j++)
68 la_inconnue_paroi(i,j)=la_inconnue_paroi_(i,j);
69 if (k_.dimension(0)==1)
70 for (i=0; i<nb_faces; i++)
71 k(i)=k_(0,0);
72 else
73 for (i=0; i<nb_faces; i++)
74 k(i)=k_(i,0);
75 for (i=0; i<nb_faces; i++)
76 {
77 int num_face=ndeb+i;
78 double k_sur_e;
79 if (axi)
80 k_sur_e=k(i)/la_zone_VDF.dist_norm_bord_axi(num_face);
81 else
82 k_sur_e=k(i)/la_zone_VDF.dist_norm_bord(num_face);
83 elem = la_zone_VDF.face_voisins(num_face,0);
84 if ( elem == -1 )
85 elem = la_zone_VDF.face_voisins(num_face,1);
86// Calcul de la inconnue a la paroi en fonction
87// de l'orientation
88 ori = orientation(num_face);
89 tab(i,ori) = la_inconnue_paroi(i,ori);
90 if (dimension==2)
91 {
92 moy_2D_vit(la_inconnue,elem,ori,la_zone_VDF,inconnue_moyenne[0]);
93 iro=1-ori;
94 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[0])/(1+k_sur_e);
95 }
96 else if (dimension==3)
97 {
98 moy_3D_vit(la_inconnue,elem,ori,la_zone_VDF,inconnue_moyenne[0],inconnue_moyenne[1]);
99 if (ori==0)
100 {
101 iro=1;
102 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[0])/(1+k_sur_e);
103 iro=2;
104 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[1])/(1+k_sur_e);
105 }
106 else if (ori==1)
107 {
108 iro=0;
109 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[0])/(1+k_sur_e);
110 iro=2;
111 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[1])/(1+k_sur_e);
112 }
113 else if (ori==2)
114 {
115 iro=0;
116 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[0])/(1+k_sur_e);
117 iro=1;
118 tab(i,iro)=(la_inconnue_paroi(i,iro)+k_sur_e*inconnue_moyenne[1])/(1+k_sur_e);
119 }
120 }
121 }
122 tab.echange_espace_virtuel();
123}
