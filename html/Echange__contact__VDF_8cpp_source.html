90void calculer_h_local(DoubleTab& tab,const Equation_base& une_eqn,const Zone_VDF& zvdf_2,const Front_VF& front_vf,const Milieu_base& le_milieu,double invhparoi,int opt)
91{
92 DoubleVect e;
93 const IntTab& face_voisins = zvdf_2.face_voisins();
94 int i;
95 int nb_faces=front_vf.nb_faces();
96 int nb_comp = le_milieu.conductivite()->nb_comp();
97 int ndeb = front_vf.num_premiere_face();
98 int nfin = ndeb + nb_faces;
99 bool dequiv=false;
100
101 e.resize(front_vf.nb_faces());
102
103 const RefObjU& modele_turbulence = une_eqn.get_modele(TURBULENCE);
104
105 if (modele_turbulence.non_nul() && sub_type(Modele_turbulence_scal_base,modele_turbulence.valeur()) && opt!=1 )
106 {
107 const Modele_turbulence_scal_base& mod_turb_scal = ref_cast(Modele_turbulence_scal_base,modele_turbulence.valeur());
108 const Turbulence_paroi_scal& loi_par = mod_turb_scal.loi_paroi();
109 if( loi_par.valeur().use_equivalent_distance() )
110 {
111 int boundary_index=-1;
112 dequiv=true;
113// Looking for the good boundary
114 int nb_boundaries=zvdf_2.zone().nb_front_Cl();
115 for (int n_bord=0; n_bord<nb_boundaries; n_bord++)
116 {
117 if (zvdf_2.front_VF(n_bord).le_nom() == front_vf.le_nom())
118 boundary_index=n_bord;
119 }
120 for (int ind_face=0; ind_face<nb_faces; ind_face++)
121 {
122 e(ind_face)=loi_par.valeur().equivalent_distance(boundary_index,ind_face);
123 }
124 }
125 }
126 if( ! dequiv )
127 {
128 for (int face=ndeb; face<nfin; face++)
129 {
130 e(face-ndeb) = zvdf_2.dist_norm_bord(face);
131 }
132 }
133
134// Calcul de tab = 1/(e/lambda + 1/h_paroi) =1/(e/lambda+invhparoi)
135 if(!sub_type(Champ_Uniforme,le_milieu.conductivite().valeur()))
136 {
137 const DoubleTab& lambda = le_milieu.conductivite().valeurs();
138 for (int face=ndeb; face<nfin; face++)
139 {
140 int elem = face_voisins(face,0);
141 if (elem == -1)
142 elem = face_voisins(face,1);
143 for(i=0; i<nb_comp; i++)
144 {
145 assert(le_milieu.conductivite()(elem,i)!=0.);
146 tab(face-ndeb,i) = 1./(e(face-ndeb)/lambda(elem,i)+invhparoi);
147 }
148 }
149 }
150 else //!< la conductivite est un Champ uniforme
151 {
152 for (int face=ndeb; face<nfin; face++)
153 {
154 for(i=0; i<nb_comp; i++)
155 {
156 assert(le_milieu.conductivite()(0,i)!=0.);
157 tab(face-ndeb,i) = 1./(e(face-ndeb)/le_milieu.conductivite()(0,i)+invhparoi);
158 }
159 }
160 }
161
162} //!< fin du cas Raccord_local_homogene

165void calculer_h_distant(DoubleTab& tab,const Equation_base& une_eqn,const Zone_VDF& zvdf_2,const Front_VF& front_vf,const Milieu_base& le_milieu,double invhparoi,int opt,const Nom& nom_racc2)

168 int i;
169 int nb_comp = le_milieu.conductivite()->nb_comp();
170 int nb_faces_raccord1 =tab.dimension(0); //= zone_dis1.zone().raccord(nom_racc1).valeur().nb_faces();
171 bool dequiv=false;
172

182 DoubleVect d_equiv_tmp;
183 front_vf.frontiere().trace_face_distant( loi_par.valeur().equivalent_distance_name(d_equiv_tmp,nom_racc2),e);
184// const Paroi_scal_hyd_base_VDF& loip = ref_cast(Paroi_scal_hyd_base_VDF,loi_par.valeur());
185// front_vf.frontiere().trace_face(loip.d_equiv_nom(d_equiv_tmp,nom_racc2),e);
186 }
187 }

189 {
190 DoubleVect dist;
191 front_vf.frontiere().trace_face_distant(zvdf_2.dist_norm_bord(dist,nom_racc2),e);
192 }
193

195

198 DoubleTab lambda;
199 front_vf.frontiere().trace_elem_distant(le_milieu.conductivite().valeurs(),lambda);

201 for(i=0; i<nb_comp; i++)
202 {
203 assert(lambda(face,i)!=0.);
204 tab(face,i) = 1./(e(face)/lambda(face,i)+invhparoi);
205 }
206 }
207 else //!< la conductivite est un Champ uniforme
208 {
209 const DoubleTab& lambda = le_milieu.conductivite().valeurs();
210 for (int face=0; face<nb_faces_raccord1; face++)
211 for(i=0; i<nb_comp; i++)
212 {
213 assert(lambda(0,i)!=0.);
214 tab(face,i) = 1./(e(face)/lambda(0,i)+invhparoi);
215 }
216 }
217} //!< fin du cas Raccord_distant_homogene

262int Echange_contact_VDF::initialiser(double temps)
263{
264 if (!Echange_global_impose::initialiser(temps))
265 return 0;
266
267 Champ_front_calc& ch=ref_cast(Champ_front_calc, T_autre_pb().valeur());
268 const Equation_base& o_eqn = ch.equation();
269 const Front_VF& fvf = ref_cast(Front_VF, frontiere_dis()), o_fvf = ref_cast(Front_VF, ch.front_dis());
270 const Zone_VDF& o_zone = ref_cast(Zone_VDF, ch.zone_dis());
271 const IntTab& o_f_e = o_zone.face_voisins();
272 const Milieu_base& le_milieu=ch.milieu();
273 int nb_comp = le_milieu.conductivite()->nb_comp();
274 Nom nom_racc1=frontiere_dis().frontiere().le_nom();
275 Zone_dis_base& zone_dis1 = zone_Cl_dis().zone_dis().valeur();
276 int nb_faces_raccord1 = zone_dis1.zone().raccord(nom_racc1).valeur().nb_faces();
277
278 h_imp_.typer("Champ_front_fonc");
279 h_imp_->fixer_nb_comp(nb_comp);
280 h_imp_.valeurs().resize(nb_faces_raccord1,nb_comp);
281
282 if (zone_dis1.zone().raccord(nom_racc1).valeur().que_suis_je() !="Raccord_distant_homogene")
283 verifier_correspondance();
284
285 autre_h.resize(nb_faces_raccord1,nb_comp);
286
287 ch.initialiser(temps,zone_Cl_dis().equation().inconnue());
288
289 monolithic = sub_type(Schema_Euler_Implicite, o_eqn.schema_temps()) ?
290 ref_cast(Schema_Euler_Implicite, o_eqn.schema_temps()).resolution_monolithique(o_eqn.domaine_application()) : 0;
291 if (!monolithic) return 1; //pas besoin du reste
292 o_zone.init_virt_e_map();
293
294 /* src(i) = (proc, j) : source de l'item i de mdv_elem */
295 IntTab src(0, 2);
296 MD_Vector_tools::creer_tableau_distribue(o_zone.zone().md_vector_elements(), src);
297 for (int i = 0; i < src.dimension_tot(0); i++) src(i, 0) = Process::me(), src(i, 1) = i;
298 src.echange_espace_virtuel();
299
300 /* o_proc, o_item -> processeur/item de l'element pour chaque face de la frontiere */
301 DoubleTrav o_proc, o_item, proc, l_item;
302 o_zone.creer_tableau_faces(o_proc), o_zone.creer_tableau_faces(o_item);
303 fvf.frontiere().creer_tableau_faces(proc), fvf.frontiere().creer_tableau_faces(l_item);
304 for (int i = 0; i < o_fvf.nb_faces(); i++)
305 {
306 int f = o_fvf.num_face(i), e = o_f_e(f, 0) == -1 ? o_f_e(f, 1) : o_f_e(f, 0);
307 o_proc(f) = src(e, 0), o_item(f) = src(e, 1); //element
308 }
309
310// projection sur la frontiere locale
311 if (o_fvf.frontiere().que_suis_je() == "Raccord_distant_homogene")
312 o_fvf.frontiere().trace_face_distant(o_proc, proc), o_fvf.frontiere().trace_face_distant(o_item, l_item);
313 else o_fvf.frontiere().trace_face_local(o_proc, proc), o_fvf.frontiere().trace_face_local(o_item, l_item);
314
315// remplissage
316 item.resize(fvf.nb_faces()), item = -1;
317 for (int i = 0; i < fvf.nb_faces(); i++)
318 if (l_item(i) >= 0)
319 {
320 if (proc(i) == Process::me()) item(i) = l_item(i); //item local (reel)
321 else
322 {
323 if (o_zone.virt_e_map.count({{ (int) proc(i), (int) l_item(i) }})) //item local (virtuel)
324 item(i) = o_zone.virt_e_map.at({{ (int) proc(i), (int) l_item(i) }});
325 else Process::exit(Nom("Echange_contact_VDF : missing element opposite face ") + Nom(fvf.num_face(i)) + " in " + fvf.le_nom() + " ! Have you used Decouper_multi?");
326 }
327 }
328 return 1;
329}

331void Echange_contact_VDF::mettre_a_jour(double temps)
332{
333 Champ_front_calc& ch=ref_cast(Champ_front_calc, T_autre_pb().valeur());
334 const Milieu_base& le_milieu=ch.milieu();
335 int nb_comp = le_milieu.conductivite()->nb_comp();
336 assert(nb_comp==1);

339

341
342 DoubleTab& mon_h= h_imp_->valeurs();
343
344 int opt=0;
345 assert(h_paroi!=0.);
346 double invhparoi=1./h_paroi;
347 calculer_h_autre_pb( autre_h, invhparoi, opt);
348 calculer_h_mon_pb(mon_h,0.,opt);
349
350 calculer_Teta_paroi(T_wall_,mon_h,autre_h,is_pb_fluide,temps);
351 calculer_Teta_equiv(T_ext()->valeurs_au_temps(temps),mon_h,autre_h,is_pb_fluide,temps);
352// on a calculer Teta paroi, on peut calculer htot dans himp (= mon_h)
353 int taille=mon_h.dimension(0);
354 for (int ii=0; ii<taille; ii++)
355 for (int jj=0; jj<nb_comp; jj++)
356 {
357 mon_h(ii,jj)=1./(1./autre_h(ii,jj)+1./mon_h(ii,jj));
358 }
359 Echange_global_impose::mettre_a_jour(temps);
360}

406// En VDF, les faces de deux raccords doivent etre numerotes de la meme facon
407int Echange_contact_VDF::verifier_correspondance() const
408{
409 const Champ_front_calc& ch=ref_cast(Champ_front_calc, T_autre_pb().valeur());
410 const Zone_VDF& ma_zvdf = ref_cast(Zone_VDF,zone_Cl_dis().zone_dis().valeur());
411 const Front_VF& ma_front_vf = ref_cast(Front_VF,frontiere_dis());
412 const Zone_VDF& zvdf_2=ref_cast(Zone_VDF, ch.zone_dis());
413 const Front_VF& front_vf=ref_cast(Front_VF, ch.front_dis());
414 int ndeb = ma_front_vf.num_premiere_face();
415 int ndeb2 = front_vf.num_premiere_face();
416 int nb_faces_bord = ma_front_vf.nb_faces();
417 int ind_fac,ind_fac_2;
418 const DoubleTab& xv=ma_zvdf.xv();
419 const DoubleTab& xv2=zvdf_2.xv();
420
421 DoubleVect x1(dimension);
422 DoubleVect x2(dimension);
423 for (int numfa=0; numfa<nb_faces_bord; numfa++)
424 {
425 ind_fac = numfa+ndeb;
426 ind_fac_2=numfa+ndeb2;
427 for (int k=0; k<dimension; k++)
428 {
429 x1[k] = xv(ind_fac,k);
430 x2[k] = xv2(ind_fac_2,k);
431 }
432 if (!meme_point2(x1,x2))
433 {
434 Cerr<<finl;
435 Cerr<<"Correspondence problem: the two 'Raccords' named "<<ma_front_vf.le_nom() <<" are not facing."<<finl;
436 Cerr<<"Problem on the following faces:" << finl;
437 if ( dimension==2 )
438 {
439 Cerr << "Coordinates of the face 1: (" << x1[0] << " " << x1[1] << ") on the domain " << ma_zvdf.domaine_dis().domaine().le_nom() << finl;
440 Cerr << "Coordinates of the face 2: (" << x2[0] << " " << x2[1] << ") on the domain " << zvdf_2.domaine_dis().domaine().le_nom() << finl;
441 }
442 else
443 {
444 Cerr << "Coordinates of the face 1: (" << x1[0] << " " << x1[1] << x1[2] << ") on the domain " << ma_zvdf.domaine_dis().domaine().le_nom() << finl;
445 Cerr << "Coordinates of the face 2: (" << x2[0] << " " << x2[1] << x2[2] << ") on the domain " << zvdf_2.domaine_dis().domaine().le_nom() << finl;
446 }
447 Cerr << "\nIn VDF, numbering of faces on each 'Raccord' must be identical (to a constant)." << finl;
448 Cerr << "The faces of each sub frontier of the boundary " << ma_front_vf.le_nom() << " must be created in the same order on each domain, this process is sensitive to the reading sense of the faces." << finl;
449 exit();
450 return 0;
451 }
452 }
453 return 1;

456int meme_point2(const DoubleVect& x,const DoubleVect& y)
457{
458 int dim = x.size();
459 assert(dim==y.size());
460 int ind =1;
461 for (int k=0; k<dim; k++)
462 if (std::fabs(x(k)-y(k))>=Objet_U::precision_geom)
463 {
464 ind = 0;
465 return ind;
466 }
467 return ind;
