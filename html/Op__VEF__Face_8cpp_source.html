37// Description
38// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de
39// la convection, de la diffusion pour le cas des faces.
40// Cette matrice a une structure de matrice morse.
41// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
42
43void Op_VEF_Face::dimensionner(const Zone_VEF& la_zone,
44 const Zone_Cl_VEF& la_zone_cl,
45 Matrice_Morse& la_matrice) const
46{
47// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de
48// la convection, de la diffusion pour le cas des faces.
49// Cette matrice a une structure de matrice morse.
50// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
51// Pour ce faire il faut chercher les faces voisines de la face consideree.
52
53 int num_face;
54 int ndeb = 0;
55 int nfin = la_zone.nb_faces();
56 int nnnn = la_zone.nb_faces_tot();
57 nfin=nnnn;
58 int i,j,k,kk;
59 int elem1,elem2;
60 int nb_faces_elem = la_zone.zone().nb_faces_elem();
61// const Conds_lim& les_cl = la_zone_cl.les_conditions_limites();
62 const int nb_comp = la_zone_cl.equation().inconnue().valeurs().line_size();
63 la_matrice.dimensionner(nfin*nb_comp,nfin*nb_comp,0);
64
65 IntVect& tab1=la_matrice.get_set_tab1();
66 IntVect& tab2=la_matrice.get_set_tab2();
67 DoubleVect& coeff = la_matrice.get_set_coeff();
68 coeff=0;
69
70
71 const IntTab& elem_faces = la_zone.elem_faces();
72 const IntTab& face_voisins = la_zone.face_voisins();
73
74// A chaque face on associe un tableau d'entiers et une liste de reels:
75// voisines[i] = {j t.q j>i et M(i,j) est non nul }
76
77// IntVect rang_voisin(nfin*nb_comp);
78 IntVect rang_voisin(nnnn*nb_comp);
79 rang_voisin=nb_comp;
80
81
82// On traite toutes les faces
83
84 for (num_face= 0; num_face<nfin; num_face++)
85 {
86 elem1 = face_voisins(num_face,0);
87 elem2 = face_voisins(num_face,1);
88
89 for (i=0; i<nb_faces_elem; i++)
90 {
91 if ( (j=elem_faces(elem1,i)) != num_face )
92 {
93 for (k=0; k<nb_comp; k++)
94 {
95 rang_voisin(num_face*nb_comp+k)+=nb_comp;
96 }
97 }
98 if (elem2!=-1)
99 if ( (j=elem_faces(elem2,i)) != num_face )
100 {
101 for (k=0; k<nb_comp; k++)
102 {
103 rang_voisin(num_face*nb_comp+k)+=nb_comp;
104 }
105 }
106 }
107 }
108
109
110// les faces voisines de num_face etant desormais comtabilisees
111// nous dimensionnons tab1 et tab2 au nombre de faces
112
113 tab1(0)=1;
114 for (num_face=ndeb; num_face<nfin; num_face++)
115 {
116 for (k=0; k< nb_comp; k++)
117 {
118 tab1(num_face*nb_comp+1+k)=rang_voisin(num_face*nb_comp+k)+tab1(num_face*nb_comp+k);
119 }
120 }
121 la_matrice.dimensionner(nfin*nb_comp,tab1(nfin*nb_comp)-1);
122
123 for (num_face = 0; num_face < nfin; num_face++ )
124 {
125 for (k=0; k< nb_comp; k++)
126 {
127 for (kk=0; kk<nb_comp; kk++)
128 {
129 int modulo = (k+kk)%nb_comp;
130 tab2[tab1[num_face*nb_comp+k]-1+kk]=num_face*nb_comp+1+modulo;
131 }
132 rang_voisin[num_face*nb_comp+k]=tab1[num_face*nb_comp+k]+nb_comp-1;
133 }
134 }
135
136
137// On traite toutes les faces
138 for (num_face= 0; num_face<nfin; num_face++)
139 {
140 elem1 = face_voisins(num_face,0);
141 elem2 = face_voisins(num_face,1);
142
143 for (i=0; i<nb_faces_elem; i++)
144 {
145 if ( (j=elem_faces(elem1,i)) != num_face )
146 {
147 for (k=0; k<nb_comp; k++)
148 {
149 for (kk=0; kk<nb_comp; kk++)
150 {
151 int modulo = (k+kk)%nb_comp;
152 tab2[rang_voisin[num_face*nb_comp+k]+kk]=j*nb_comp+1+modulo;
153 }
154 rang_voisin[num_face*nb_comp+k]+=nb_comp;
155 }
156 }
157 if (elem2!=-1)
158 if ( (j=elem_faces(elem2,i)) != num_face )
159 {
160 for (k=0; k<nb_comp; k++)
161 {
162 for (kk=0; kk<nb_comp; kk++)
163 {
164 int modulo = (k+kk)%nb_comp;
165 tab2[rang_voisin[num_face*nb_comp+k]+kk]=j*nb_comp+1+modulo;
166 }
167 rang_voisin[num_face*nb_comp+k]+=nb_comp;
168 }
169 }
170 }
171 }
172}

178void Op_VEF_Face::modifier_pour_Cl(const Zone_VEF& la_zone,
179 const Zone_Cl_VEF& la_zone_cl,
180 Matrice_Morse& la_matrice, DoubleTab& secmem) const
181{
182// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de
183// la convection, de la diffusion pour le cas des faces.
184// Cette matrice a une structure de matrice morse.
185// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
186 const Conds_lim& les_cl = la_zone_cl.les_conditions_limites();
187 const IntVect& tab1=la_matrice.get_tab1();
188 DoubleVect& coeff = la_matrice.get_set_coeff();
189 const DoubleTab& champ_inconnue = la_zone_cl.equation().inconnue().valeurs();
190 const int nb_comp = champ_inconnue.line_size();
191 ArrOfDouble normale(nb_comp);
192 int size = les_cl.size();
193 for (int i=0; i<size; i++)
194 {
195 const Cond_lim& la_cl = les_cl[i];
196 if (sub_type(Dirichlet,la_cl.valeur()))
197 {
198 const Dirichlet& la_cl_Dirichlet = ref_cast(Dirichlet,la_cl.valeur());
199 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
200 int nfaces = la_front_dis.nb_faces();
201 for (int ind_face=0; ind_face < nfaces; ind_face++)
202 {
203 int face = la_front_dis.num_face(ind_face);
204 for (int comp=0; comp<nb_comp; comp++)
205 {
206 int idiag = tab1[face*nb_comp+comp]-1;
207 coeff[idiag]=1;
208// pour les voisins
209 int nbvois = tab1[face*nb_comp+1+comp] - tab1[face*nb_comp+comp];
210 for (int k=1; k < nbvois; k++)
211 {
212 coeff[idiag+k]=0;
213 }
214// pour le second membre
215 if (nb_comp == 1)
216 secmem(face) = la_cl_Dirichlet.val_imp(ind_face,0);
217 else
218 secmem(face,comp)= la_cl_Dirichlet.val_imp(ind_face,comp);
219 }
220 }
221 }
222 if (sub_type(Dirichlet_homogene,la_cl.valeur()))
223 {
224 const Dirichlet_homogene& la_cl_Dirichlet_homogene = ref_cast(Dirichlet_homogene,la_cl.valeur());
225 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
226 int nfaces = la_front_dis.nb_faces_tot();
227 for (int ind_face=0; ind_face < nfaces; ind_face++)
228 {
229 int face = la_front_dis.num_face(ind_face);
230 for (int comp=0; comp<nb_comp; comp++)
231 {
232 int idiag = tab1[face*nb_comp+comp]-1;
233 coeff[idiag]=1;
234// pour les voisins
235 int nbvois = tab1[face*nb_comp+1+comp] - tab1[face*nb_comp+comp];
236 for (int k=1; k < nbvois; k++)
237 {
238 coeff[idiag+k]=0;
239 }
240// pour le second membre
241 if (nb_comp == 1)
242 secmem(face) = la_cl_Dirichlet_homogene.val_imp(ind_face,0);
243 else
244 secmem(face,comp)= la_cl_Dirichlet_homogene.val_imp(ind_face,comp);
245 }
246 }
247 }
248 if (sub_type(Symetrie,la_cl.valeur()))
249 if (la_zone_cl.equation().inconnue().valeur().nature_du_champ()==vectoriel)
250 {
251 const IntVect& tab2=la_matrice.get_tab2();
252 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
253 const DoubleTab& face_normales = la_zone.face_normales();
254 int nfaces = la_front_dis.nb_faces_tot();
255 ArrOfDouble somme(la_matrice.nb_colonnes()); //!< On dimensionne au plus grand
256 for (int ind_face=0; ind_face < nfaces; ind_face++)
257 {
258 int face = la_front_dis.num_face(ind_face);
259 double max_coef=0;
260 int ind_max=-1;
261 double n2=0;
262 for (int comp=0; comp<nb_comp; comp++)
263 {
264 normale[comp]=face_normales(face,comp);
265 if (std::fabs(normale[comp])>std::fabs(max_coef))
266 {
267 max_coef=normale[comp];
268 ind_max=comp;
269 }
270 n2+=normale[comp]*normale[comp];
271 }
272 normale/=sqrt(n2);
273 max_coef=normale[ind_max];
274
275// On commence par recalculer secmem=secmem-A *present pour pouvoir modifier A (on en profite pour projeter)
276 int nb_coeff_ligne=tab1[face*nb_comp+1] - tab1[face*nb_comp];
277 for (int k=0; k<nb_coeff_ligne; k++)
278 {
279 for (int comp=0; comp<nb_comp; comp++)
280 {
281 int j=tab2[tab1[face*nb_comp+comp]-1+k]-1;
282// assert(j!=(face*nb_comp+comp));
283// if ((j>=(face*nb_comp))&&(j<(face*nb_comp+nb_comp)))
284 const double coef_ij=la_matrice(face*nb_comp+comp,j);
285 int face2=j/nb_comp;
286 int comp2=j-face2*nb_comp;
287 secmem(face,comp)-=coef_ij*champ_inconnue(face2,comp2);
288 }
289 }
290 double somme_b=0;
291
292 for (int comp=0; comp<nb_comp; comp++)
293 somme_b+=secmem(face,comp)*normale[comp];
294
295// on retire secmem.n n
296 for (int comp=0; comp<nb_comp; comp++)
297 secmem(face,comp)-=somme_b*normale[comp];
298
299// on doit remettre la meme diagonale partout on prend la moyenne
300 double ref=0;
301 for (int comp=0; comp<nb_comp; comp++)
302 {
303 int j0=face*nb_comp+comp;
304 ref+=la_matrice(j0,j0);
305 }
306 ref/=nb_comp;
307
308 for (int comp=0; comp<nb_comp; comp++)
309 {
310 int j0=face*nb_comp+comp;
311 double rap=ref/la_matrice(j0,j0);
312 for (int k=0; k<nb_coeff_ligne; k++)
313 {
314 int j=tab2[tab1[j0]-1+k]-1;
315 la_matrice(j0,j)*=rap;
316 }
317 assert(est_egal(la_matrice(j0,j0),ref));
318 }
319// on annule tous les coef extra diagonaux du bloc
320//
321 for (int k=1; k<nb_coeff_ligne; k++)
322 {
323 for (int comp=0; comp<nb_comp; comp++)
324 {
325 int j=tab2[tab1[face*nb_comp+comp]-1+k]-1;
326 assert(j!=(face*nb_comp+comp));
327 if ((j>=(face*nb_comp))&&(j<(face*nb_comp+nb_comp)))
328 la_matrice(face*nb_comp+comp,j)=0;
329 }
330 }
331
332// pour les blocs extra diagonaux on assure que Aij.ni=0
333// ArrOfDouble somme(nb_coeff_ligne);
334 for (int k=0; k<nb_coeff_ligne; k++)
335 {
336 somme[k]=0;
337 int j=tab2[tab1[face*nb_comp]-1+k]-1;
338
339// le coeff j doit exister sur les nb_comp lignes
340 double dsomme=0;
341 for (int comp=0; comp<nb_comp; comp++)
342 dsomme+=la_matrice(face*nb_comp+comp,j)*normale[comp];
343
344// on retire somme ni
345
346 for (int comp=0; comp<nb_comp; comp++)
347// on modifie que les coefficients ne faisant pas intervenir u(face,comp)
348 if ((j<(face*nb_comp))||(j>=(face*nb_comp+nb_comp)))
349 la_matrice(face*nb_comp+comp,j)-=(dsomme)*normale[comp];
350 }
351// Finalement on recalcule secmem=secmem+A*champ_inconnue (A a ete beaucoup modiife)
352 for (int k=0; k<nb_coeff_ligne; k++)
353 {
354 for (int comp=0; comp<nb_comp; comp++)
355 {
356 int j=tab2[tab1[face*nb_comp+comp]-1+k]-1;
357 int face2=j/nb_comp;
358 int comp2=j-face2*nb_comp;
359 const double coef_ij=la_matrice(face*nb_comp+comp,j);
360 secmem(face,comp)+=coef_ij*champ_inconnue(face2,comp2);
361 }
362 }
363 {
364// verification
365 double somme_c=0;
366 for (int comp=0; comp<nb_comp; comp++)
367 somme_c+=secmem(face,comp)*normale[comp];
368// on retire secmem.n n
369 for (int comp=0; comp<nb_comp; comp++)
370 secmem(face,comp)-=somme_c*normale[comp];
371 }
372 }
373 }
374 }

377void Op_VEF_Face::modifier_flux( const Operateur_base& op) const
378{
379 controle_modifier_flux_=1;
380 DoubleTab& flux_bords_=op.flux_bords();
381 if (flux_bords_.nb_dim()!=2)
382 return;
383 const Probleme_base& pb=op.equation().probleme();
384
385 const Zone_VEF& la_zone_vef=ref_cast(Zone_VEF,op.equation().zone_dis().valeur());
386 int nb_compo=flux_bords_.dimension(1);
387// On multiplie le flux au bord par rho*Cp sauf si c'est un operateur de diffusion avec la conductivite comme champ
388 if (op.equation().inconnue().le_nom()=="temperature"
389 && !( sub_type(Operateur_Diff_base,op) && ref_cast(Operateur_Diff_base,op).diffusivite().le_nom() == "conductivite" ) )
390 {
391 const Champ_base& rho = (op.equation()).milieu().masse_volumique().valeur();
392 const Champ_Don& Cp = (op.equation()).milieu().capacite_calorifique();
393 int rho_uniforme=(sub_type(Champ_Uniforme,rho) ? 1:0);
394 int cp_uniforme=(sub_type(Champ_Uniforme,Cp.valeur()) ? 1:0);
395 double Cp_=0,rho_=0;
396 int is_rho_u=pb.is_dilatable();
397 if (is_rho_u)
398 {
399 is_rho_u=0;
400 if (sub_type(Op_Conv_VEF_base,op))
401 {
402 if (ref_cast(Op_Conv_VEF_base,op).vitesse().le_nom()=="rho_u")
403 is_rho_u=1;
404 }
405 }
406 const int nb_faces_bords=la_zone_vef.nb_faces_bord();
407 for (int face=0; face<nb_faces_bords; face++)
408 {
409 if (cp_uniforme) Cp_=Cp(0,0);
410 else
411 {
412 if (Cp.nb_comp()==1) Cp_=Cp(face);
413 else Cp_=Cp(face,0);
414 }
415 if (rho_uniforme) rho_=rho(0,0);
416 else
417 {
418 if (rho.nb_comp()==1) rho_=rho(face);
419 else rho_=rho(face,0);
420 }
421// si on est en QC temperature et si on a calcule div(rhou * T)
422// il ne faut pas remultiplier par rho
423 if (is_rho_u) rho_=1;
424 flux_bords_(face,0) *= (rho_*Cp_);
425 }
426 }
427
428
429// On multiplie par rho si Navier Stokes incompressible
430 Nom nom_eqn=op.equation().que_suis_je();
431 if (nom_eqn.debute_par("Navier_Stokes") && pb.milieu().que_suis_je()=="Fluide_Incompressible")
432 {
433 const Champ_base& rho = op.equation().milieu().masse_volumique().valeur();
434 if (sub_type(Champ_Uniforme,rho))
435 {
436 double coef = rho(0,0);
437 int nb_faces_bord=la_zone_vef.nb_faces_bord();
438 for (int face=0; face<nb_faces_bord; face++)
439 for(int k=0; k<nb_compo; k++)
440 flux_bords_(face,k) *= coef;
441 }
442 }
443
444}

446// Description
447// Impression des flux d'un operateur VEF aux faces (ie: diffusion, convection)
448//
449int Op_VEF_Face::impr(Sortie& os, const Operateur_base& op) const
450{
451 const Zone_VEF& la_zone_vef=ref_cast(Zone_VEF,op.equation().zone_dis().valeur());
452 DoubleTab& flux_bords_=op.flux_bords();
453 if (flux_bords_.nb_dim()!=2)
454 {
455 Cout << "L'impression des flux n'est pas codee pour l'operateur " << op.que_suis_je() << finl;
456 return 1;
457 }
458 if (controle_modifier_flux_==0)
459 if (max_abs_array(flux_bords_)!=0)
460 {
461 Cerr<<op.que_suis_je()<<" appelle Op_VEF_Face::impr sans avoir appeler Op_VEF_Face::modifier_flux, on arrete tout "<<finl;
462 Process::exit();
463 }
464 int nb_compo=flux_bords_.dimension(1);
465 const Probleme_base& pb=op.equation().probleme();
466 const Schema_Temps_base& sch=pb.schema_temps();
467// On n'imprime les moments que si demande et si on traite l'operateur de diffusion de la vitesse
468 int impr_mom=0;
469 if (la_zone_vef.zone().Moments_a_imprimer() && sub_type(Operateur_Diff_base,op) && op.equation().inconnue().le_nom()=="vitesse")
470 impr_mom=1;
471
472 const int impr_sum=(la_zone_vef.zone().Bords_a_imprimer_sum().est_vide() ? 0:1);
473 const int impr_bord=(la_zone_vef.zone().Bords_a_imprimer().est_vide() ? 0:1);
474
475// Calcul des moments
476 const int nb_faces = la_zone_vef.nb_faces_tot();
477 DoubleTab xgr(nb_faces,Objet_U::dimension);
478 xgr=0.;
479 if (impr_mom)
480 {
481 const DoubleTab& xgrav = la_zone_vef.xv();
482 const ArrOfDouble& c_grav=la_zone_vef.zone().cg_moments();
483 for (int num_face=0; num_face <nb_faces; num_face++)
484 for (int i=0; i<Objet_U::dimension; i++)
485 xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
486 }
487
488// On parcours les frontieres pour sommer les flux par frontiere dans le tableau flux_bord
489 DoubleVect bilan(nb_compo);
490 bilan = 0;
491 int nb_cl = la_zone_vef.nb_front_Cl();
492 DoubleTrav flux_bords(4,nb_cl,nb_compo);
493 flux_bords=0.;
494 /*
495 flux_bord(k) -> flux_bords(0,num_cl,k)
496 flux_bord_perio1(k) -> flux_bords(1,num_cl,k)
497 flux_bord_perio2(k) -> flux_bords(2,num_cl,k)
498 moment(k) -> flux_bords(3,num_cl,k)
499 */
500 for (int num_cl=0; num_cl<nb_cl; num_cl++)
501 {
502 const Cond_lim& la_cl = op.equation().zone_Cl_dis().les_conditions_limites(num_cl);
503 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
504 int ndeb = frontiere_dis.num_premiere_face();
505 int nfin = ndeb + frontiere_dis.nb_faces();
506 int perio = (sub_type(Periodique,la_cl.valeur())?1:0);
507 for (int face=ndeb; face<nfin; face++)
508 {
509 for(int k=0; k<nb_compo; k++)
510 {
511 flux_bords(0,num_cl,k)+=flux_bords_(face, k);
512 if(perio)
513 {
514 if(face<(ndeb+frontiere_dis.nb_faces()/2))
515 flux_bords(1,num_cl,k)+=flux_bords_(face, k);
516 else
517 flux_bords(2,num_cl,k)+=flux_bords_(face, k);
518 }
519 }
520 if (impr_mom)
521 {
522// Calcul du moment exerce par le fluide sur le bord (OM/\F)
523 if (Objet_U::dimension==2)
524 flux_bords(3,num_cl,0)+=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
525 else
526 {
527 flux_bords(3,num_cl,0)+=flux_bords_(face,2)*xgr(face,1)-flux_bords_(face,1)*xgr(face,2);
528 flux_bords(3,num_cl,1)+=flux_bords_(face,0)*xgr(face,2)-flux_bords_(face,2)*xgr(face,0);
529 flux_bords(3,num_cl,2)+=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
530 }
531 }
532 } //!< fin for face
533 } //!< fin for num_cl
534
535// On somme les contributions de chaque processeur
536 mp_sum_for_each_item(flux_bords);
537
538// Ecriture dans les fichiers
539 if (Process::je_suis_maitre())
540 {
541// Open files if needed
542// SFichier Flux;
543 if (!Flux.is_open()) op.ouvrir_fichier(Flux,"",1);
544// SFichier Flux_moment;
545 if (!Flux_moment.is_open()) op.ouvrir_fichier(Flux_moment,"moment",impr_mom);
