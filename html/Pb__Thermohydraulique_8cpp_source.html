161//! Associe le milieu au probleme<br>Le milieu doit etre de type fluide incompressible<br>Parametre: Milieu_base\& mil<br>Signification: le milieu physique a associer au probleme<br>Valeurs par defaut:<br>Contraintes: doit etre de type fluide incompressible<br>reference constante<br>Acces: entree
162void Pb_Thermohydraulique::associer_milieu_base(const Milieu_base& mil)
163{
164 if (sub_type(Fluide_Incompressible,mil))
165 {
166 eq_hydraulique.associer_milieu_base(mil);
167 eq_thermique.associer_milieu_base(mil);
168 }
169 else if (sub_type(Fluide_Ostwald,mil))
170 {
171 eq_hydraulique.associer_milieu_base(mil);
172 eq_thermique.associer_milieu_base(mil);
173 Cerr << "Fluide non newtonien" << finl;
174 }
175 else
176 {
177 Cerr << "Un milieu de type " << mil.que_suis_je() << " ne peut etre associe a "<< finl;
178 Cerr << "un probleme de type Pb_Thermohydraulique " << finl;
179 exit();
180 }
181}
