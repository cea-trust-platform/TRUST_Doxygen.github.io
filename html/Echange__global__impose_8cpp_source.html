147double Echange_global_impose::champ_exterieur(int i, const Champ_front& champ_ext) const
148{
149 if (ma_zone_cl_dis->equation().que_suis_je()!="Convection_Diffusion_Concentration")
150 {
151 const Milieu_base& mil=ma_zone_cl_dis->equation().milieu();
152 Nom nom_pb=ma_zone_cl_dis->equation().probleme().que_suis_je();
153 double d_Cp;
154 double d_rho;
155 const Champ_base& rho=mil.masse_volumique().valeur();
156 if ((nom_pb.debute_par("Probleme_Interface")|| nom_pb==Nom("Probleme_Thermo_Front_Tracking"))||!sub_type(Champ_Uniforme,rho)||nom_pb==Nom("Pb_Conduction"))
157 {
158// Pour le front tracking, on ne divise pas par Rho*Cp
159// ni pour QC
160 d_rho=1.;
161 d_Cp=1.;
162 }
163 else
164 {
165 const Champ_Don& Cp =mil.capacite_calorifique();
166
167 if (sub_type(Champ_Uniforme,rho))
168 {
169 d_rho= rho(0,0);
170 }
171 else
172 {
173 d_rho= rho(i);
174 }
175 if (sub_type(Champ_Uniforme,Cp.valeur()))
176 {
177 d_Cp= Cp(0,0);
178 }
179 else
180 {
181 d_Cp= Cp.valeur()(i);
182 }
183 }
184
185 if (champ_ext.valeurs().size()==1)
186 return champ_ext(0,0)/(d_rho*d_Cp);
187 else if (champ_ext.valeurs().dimension(1)==1)
188 return champ_ext(i,0)/(d_rho*d_Cp);
189 else
190 Cerr << "Echange_global_impose::flux_impose erreur" << finl;
191 exit();
192 return 0.;
193 }
194 else
195 {
196 if (champ_ext.valeurs().size()==1)
197 return champ_ext(0,0);
198 else if (champ_ext.valeurs().dimension(1)==1)
199 return champ_ext(i,0);
200 else
201 Cerr << "Echange_global_impose::flux_impose erreur" << finl;
202 exit();
203 return 0.;
204
205 }
206}
