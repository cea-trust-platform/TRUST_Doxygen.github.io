49void Op_Diff_RotRot::
50associer(const Zone_dis& zone_dis, const Zone_Cl_dis& zone_Cl_dis,
51 const Champ_Inc& inco)
52{
53 const Zone_VEF& zvef = ref_cast(Zone_VEF, zone_dis.valeur());
54 const Zone_Cl_VEF& zclvef = ref_cast(Zone_Cl_VEF, zone_Cl_dis.valeur());
55 la_zone_vef = zvef;
56 la_zcl_vef = zclvef;
57
58 curl_.associer(zone_dis,zone_Cl_dis,inco);
59 rot_.associer(zone_dis,zone_Cl_dis,inco);
60
61//
62// On definit le champ vorticite
63 vorticite_.typer("Champ_P1_isoP1Bulle");
64 Champ_P1_isoP1Bulle& vorticite =
65 ref_cast(Champ_P1_isoP1Bulle,vorticite_.valeur());
66
67 vorticite.associer_zone_dis_base(zvef);
68 vorticite.nommer("vorticite");
69
70 if (dimension == 2) vorticite.fixer_nb_comp(1);
71 else vorticite.fixer_nb_comp(dimension);
72
73 int nb_tot = zvef.nb_elem() + zvef.nb_som() -1;
74 vorticite.fixer_nb_valeurs_nodales(nb_tot);
75
76 vorticite.fixer_unite("s-1");
77
78//
79
80 solveur_.typer("Solv_GCP");
81 Precond p;
82 p.typer("SSOR");
83 ref_cast(Solv_GCP,solveur.valeur()).set_precond(p);
84 assembler_matrice(matrice_vorticite_);
85 tester();
86
87}

101DoubleTab& Op_Diff_RotRot::
102ajouter(const DoubleTab& vitesse, DoubleTab& diffusion) const
103{
104 Cerr << "je suis dans OpDiffRotRot" << finl;
105 DoubleTab curl(matrice_vorticite_.valeur().ordre());
106
107// Marche car Champ_Inc est mutable
108 curl_.calculer(vitesse,curl);
109// curl=-1*curl;
110 calculer_vorticite(vorticite_.valeurs(),curl);

112
113 Cerr << "je sors de OpDiffRotRot" << finl;
114
115 return diffusion;

119int Op_Diff_RotRot::
120calculer_vorticite(DoubleTab& solution, const DoubleTab& curl) const
121{
122 const Zone& zone = zone_Vef().zone();
123 static int nb_appel2=0;
124
125// Resolution en vorticite: on ne considere que le cas
126// sequentiel pour le moment.
127// Pour le cas parallele, il faut s'inspirer de la classe N_S.cpp.
128
129 if (Process::nproc()==1)
130 {
131// On considere que la matrice de vorticite est une matrice
132// morse par defaut, et on applique la methode inverse()
133// pour resoudre le systeme lineaire.
134 const Matrice_Morse_Sym& la_matrice=ref_cast(Matrice_Morse_Sym,
135 matrice_vorticite_.valeur());
136// Matrice_Morse& la_matrice = (Matrice_Morse&) matrice_vorticite_.valeur();
137 DoubleTab solution_temporaire(la_matrice.ordre());
138
139 assert(solution_temporaire.size() == solution.size()-1);
140 assert(curl.size() == solution_temporaire.size());
141
142// On resoud un systeme lineaire pour calculer la vorticite
143// La vorticite est alors stockee dans la variable solution
144// la_matrice.inverse(curl,solution_temporaire,1e-15);
145 Solv_GCP& solv = ref_cast_non_const(Solv_GCP,solveur_.valeur());
146 solv.set_seuil(1e-17);
147 solv.resoudre_systeme(la_matrice,curl,solution_temporaire);
148
149// On recopie les valeurs de solution temoraire dans solution
150// Une case n'est pas remplie: on le fait plus tard.
151 for (int i=0; i<solution_temporaire.size(); i++)
152 solution[i]=solution_temporaire[i];
153
154// On remplit le dernier element de "solution"
155 int sommet = zone.nb_som()-1;
156
157 for (int i=0; i<curl_.elem_som_size(sommet); i++)
158 solution[sommet] +=
159 solution_temporaire[curl_.elements_pour_sommet(sommet,i)];
160
161 nb_appel2++;
162
163 }
164
165 return 1;
166}

169/* Fonctions permettant l'assemblage de la matrice */
170/* de vorticite */
171//
172
173/* On calcule la matrice de vorticite pour le probleme triple:
174 / on considere le cas sans condition au limite pour la vorticite
175 / pour le moment, et on ne code pas la partie parallele de
176 / l'algo.
177 / REM: pour l'instant, ne fonctionne qu'en 2D
178*/
179int Op_Diff_RotRot::assembler_matrice(Matrice& matrice)
180{
181 const Zone& zone = zone_Vef().zone();
182
183 int colonne_a_remplir_tab2,colonne_a_remplir_coeff;
184 int nombre_coeff_non_nuls;
185
186 Cerr << "Assemblage de la matrice de vorticite en cours..." << finl;
187 matrice.typer("Matrice_Morse_Sym");
188// Matrice_Morse& la_matrice=(Matrice_Morse&) matrice.valeur();
189
190// On dimensionne la matrice de maniere convenable.
191// La matrice est carree et de taille (nombre_elem+nombre_som)
192// Le nombre de coeff non nuls n'excedent pas
193// (2* (dimension+1) + 1)* nb_elem + (nb_som-1) * (nb_som-1)
194// ATTENTION: une BASE de mon espace est les fonctions indicatrices
195// des elements + les fonctions chapeaux - 1 de ces fonctions
196// Sinon la somme des toutes les fontions chapeaux - la somme
197// de toutes les fonctions indicatrices = 0
198// Par defaut, on enleve la derniere fonction chapeau pour
199// former notre base.
200// Cf. Papier dans Latex/Vorticity
201 nombre_coeff_non_nuls = (2*dimension + 3) * zone.nb_elem()
202 + (zone.nb_som()-1)* (zone.nb_som()-1);
203// la_matrice.dimensionner(zone.nb_elem()+zone.nb_som()-1,nombre_coeff_non_nuls);
204 Matrice_Morse la_matrice(zone.nb_elem()+ zone.nb_som()-1,nombre_coeff_non_nuls);
205
206// Au cas ou la matrice existe deja dans un fichier
207 Nom nomfic("Vorticite.sv");
208 LecFicDistribueBin vorticite;
209 int fic_vorticite_existe;
210
211 if(vorticite.ouvrir(nomfic)) fic_vorticite_existe=1;
212 else fic_vorticite_existe=0;
213
214 if(fic_vorticite_existe)
215 {
216 Cerr << "Relecture de la matrice de vorticite sur le fichier: "
217 << nomfic << finl;
218 vorticite >> la_matrice;
219 vorticite.close();
220 Cerr << "Fin de la relecture de la matrice de vorticite." << finl;
221 return 1;
222 }
223
224 Cerr << "Assemblage de la matrice de vorticite " << nomfic << finl;
225
226// Maintenant on remplit la matrice ligne par ligne
227// On rappelle que les elements et les sommets sont numerotes
228// a partir de 0.
229// Mais les indices de colonnes et de lignes des tableaux
230// d'une matrice morse FORTRAN commencent a 1.
231// Ex de construction: matrice[numerotation_C++] = numerotation_FORTRAN
232
233// Parametres importants pour remplir les tableaux
234 nombre_coeff_non_nuls = 1;
235 colonne_a_remplir_tab2 = 0; //pour le C++
236 colonne_a_remplir_coeff = 0; //pour le C++
237
238// On commence par remplir les lignes de la sous-matrice
239// de taille nb_elem * (nb_elem + nb_som) : cf. structure de la matrice
240 for (int numero_elem = 0; numero_elem < zone.nb_elem(); numero_elem++)
241 {
242// Pour un element donne "numero_elem", on calcule la liste
243// des sommets qui appartiennent a cet element.
244 IntList sommets_pour_elem = sommets_pour_element(numero_elem);
245 Tri(sommets_pour_elem); //tableau trie
246
247// On remplit tab1
248 la_matrice.get_set_tab1()(numero_elem) = nombre_coeff_non_nuls;
249
250// On remplit tab2 et coeff
251
252// Les nb_elem premiers elements de la ligne "numero_elem"
253// a cause du FORTRAN
254 la_matrice.get_set_tab2()(colonne_a_remplir_tab2) = numero_elem + 1;
255 la_matrice.get_set_coeff()(colonne_a_remplir_coeff) =
256 remplir_elem_elem_EF(numero_elem);
257
258// On incremente nb_coeff_non_nuls car on vient de remplir
259// une ligne par l'un de ces elements non nul
260 nombre_coeff_non_nuls++;
261
262// On incremente les compteurs puisque l'on vient de remplir
263// des cases du tableau
264 colonne_a_remplir_tab2++;
265 colonne_a_remplir_coeff++;
266
267// Les nb_som elements de la ligne "numero_elem"
268 for (int i = 0; i < zone.nb_som_elem(); i++)
269 {
270// Si ce "sommet_pour_elem[i]" est different du numero
271// du dernier sommet alors on stocke le bon coefficient
272 if (sommets_pour_elem[i] != zone.nb_som()-1)
273 {
274 la_matrice.get_set_tab2()(colonne_a_remplir_tab2) =
275 zone.nb_elem()+sommets_pour_elem[i]+1; //pour FORTRAN
276 la_matrice.get_set_coeff()(colonne_a_remplir_coeff) =
277 remplir_elem_som_EF(numero_elem,sommets_pour_elem[i]);
278
279// On incremente nb_coeff_non_nuls car on vient de remplir
280// une ligne avec l'un de ces elements non nul
281 nombre_coeff_non_nuls++;
282
283// Enfin on incremente les colonne_* pour eviter de reecrire
284// sur des coefficients deja stockes
285 colonne_a_remplir_tab2++;
286 colonne_a_remplir_coeff++;
287 }
288
289 }
290
291// Pas besoin de modifier les entiers nombre_coeff_non_nuls et
292// colonne_* : ils sont a la bonne valeur.
293
294 }
295
296// On remplit les lignes de la sous-matrice de taille
297// nb_som *( nb_elem+nb_som) :cf. structure de la matrice
298 for (int numero_som = 0; numero_som < zone.nb_som()-1; numero_som++)
299 {
300// On stocke les tableaux qui nous sont utiles
301// Ainsi que leur taille respective
302 IntList Elem_pour_sommet = elements_pour_sommet(numero_som);
303 IntList Sommets_voisins = sommets_voisins(numero_som,Elem_pour_sommet);
304 Tri(Elem_pour_sommet); //liste triee
305 Tri(Sommets_voisins); //liste triee
306
307// On remplit tab1
308// pour FORTRAN
309 la_matrice.get_set_tab1()(zone.nb_elem()+numero_som) = nombre_coeff_non_nuls;
310
311// On remplit tab2 et coeff
312
313// Les nb_elem premiers elements de la ligne "numero_som"
314 for (int i=0; i < Elem_pour_sommet.size(); i++)
315 {
316// A cause du FORTRAN
317 la_matrice.get_set_tab2()(colonne_a_remplir_tab2) = Elem_pour_sommet[i]+1;
318 la_matrice.get_set_coeff()(colonne_a_remplir_coeff) =
319 remplir_som_elem_EF(Elem_pour_sommet[i],numero_som);
320
321// On incremente nb_coeff_non_nuls car on vient de remplir
322// une ligne avec l'un de ces elements non nul
323 nombre_coeff_non_nuls++;
324
325// On incremente convenablement les indices de colonnes
326 colonne_a_remplir_tab2++;
327 colonne_a_remplir_coeff++;
328 }
329
330// Pas besoin de modifier les entiers nb_coeff_non_nuls et
331// colonne_* : ils sont a la bonne valeur.
332
333// Les nb_som elements de la ligne "numero_som"
334 for (int i=0; i < Sommets_voisins.size(); i++)
335 {
336// On recupere le numero global du sommet_voisin
337 int numero_som_global = Sommets_voisins[i];
338
339// Et si "numero_som_global" est different du numero
340// du dernier sommet, alors on stocke le bon coefficient
341
342 if (numero_som_global != zone.nb_som()-1)
343 {
344// A cause du FORTRAN
345 la_matrice.get_set_tab2()(colonne_a_remplir_tab2) =
346 zone.nb_elem()+Sommets_voisins[i]+1;
347 la_matrice.get_set_coeff()(colonne_a_remplir_coeff) =
348 remplir_som_som_EF(numero_som,Sommets_voisins[i],Elem_pour_sommet);
349
350// On incremente nb_coeff_non_nuls car on vient de remplir
351// une ligne avec l'un de ces elements non nul
352 nombre_coeff_non_nuls++;
353
354// On incremente les indices
355 colonne_a_remplir_tab2++;
356 colonne_a_remplir_coeff++;
357 }
358
359 }
360
361// Pas besoin d'incrementer nombre_coeff_non_nuls et colonne_*
362// : ils sont deja a la bonne valeur
363 }
364
365// Par convention pour les matrices morses, le dernier element
366// de tab1 est tab1[zone.nb_elem()+zone.nb_som()+1] et vaut
367// le nombre total de coefficients non nuls +1
368// soit avec notre algorithme: nombre_coeff_non_nuls
369 la_matrice.get_set_tab1()(zone.nb_elem()+zone.nb_som()-1) = nombre_coeff_non_nuls;
370
371// if(Debog::mode_db==2) Debog::save_matrix_seq(la_matrice);
372// else if(Debog::mode_db==3)
373// Debog::save_and_distribute_matrix_seq(la_matrice);
374
375 Matrice_Morse_Sym& la_matrice_sym = ref_cast(Matrice_Morse_Sym,
376 matrice.valeur());
377 la_matrice_sym = la_matrice;
378
379 Cerr << "Fin de l'assemblage de la matrice de vorticite. " << finl;
380
381 return 1;
382}

384/* Pour un element donne "numero_elem" retourne */
385/* la liste des sommets appartenant a cet element */
386IntList Op_Diff_RotRot::sommets_pour_element(int numero_elem) const
387{
388 IntList resultat;
389 int numero_global_sommet = 0;
390 const Zone& zone = zone_Vef().zone();
391
392// Par precaution mais normalement inutile
393 if (! resultat.est_vide() ) resultat.vide();
394
395 for (int i=0 ; i< zone.nb_som_elem() ; i++)
396 {
397 numero_global_sommet = zone.sommet_elem(numero_elem,i);
398 resultat.add_if_not(numero_global_sommet);
399 }
400
401 return resultat;
402}

404/* Pour un sommet donne "numero_sommet" retourne */
405/* la liste des elements contenant ce sommet */
406IntList
407Op_Diff_RotRot::elements_pour_sommet(int numero_sommet) const
408{
409 IntList resultat;
410 int numero_global_som;
411 const Zone& zone = zone_Vef().zone();
412
413// Par precaution mais normalement inutile
414 if (! resultat.est_vide() ) resultat.vide();
415
416// Pas tres efficace mais marche quelle que soit la dimension:
417// on boucle sur les elements, on regarde pour chaque element
418// ses sommets, puis on compare ces sommets au parametre d'entree
419// et si l'un d'eux coincide avec le parametre d'entree, on stocke
420// l'element.
421 for (int numero_elem = 0 ; numero_elem < zone.nb_elem() ; numero_elem++)
422 for (int numero_som = 0 ; numero_som < zone.nb_som_elem() ; numero_som++)
423 {
424 numero_global_som = zone.sommet_elem(numero_elem,numero_som);
425 if ( numero_sommet == numero_global_som )
426 resultat.add_if_not(numero_elem);
427 }
428
429 return resultat;
430
431}

464/* Fonction de tri d'une IntList */
465/* Le tri s'effectue par ordre croissant */
466/* REM: on n'a aucun doublon dans la liste triee */
467void Op_Diff_RotRot::Tri(IntList& liste_a_trier) const
468{
469 if (liste_a_trier.est_vide())
470 {
471 Cerr << "Erreur dans Op_Diff_RotRot::Tri()." << finl;
472 Cerr << "La liste a trier est vide: sortie du programme." << finl;
473 Process::exit();
474 }
475
476 IntList temporaire;
477 int minimum;
478
479 while ( !liste_a_trier.est_vide() )
480 {
481 minimum = liste_a_trier[0];
482
483 for (int i = 0; i < liste_a_trier.size(); i++)
484 minimum = (minimum <= liste_a_trier[i] ? minimum : liste_a_trier[i]);
485
486 temporaire.add_if_not(minimum);
487 liste_a_trier.suppr(minimum);
488 }
489
490 for (int i = 0 ; i < temporaire.size() ; i++)
491 liste_a_trier.add_if_not(temporaire[i]);
492
493}

527/* Pour l'element "numero_som" retourne le coefficient */
528/* a placer dans la sous matrice de taille nb_som * nb_som */
529/* a la ligne "numero_som" , colonne "sommet_voisin"*/
530/* "elem_voisins" est le tableau des elements contenant "numero_som" */
531/* Matrice EF */
532double
533Op_Diff_RotRot::remplir_som_som_EF(const int numero_som,
534 const int sommet_voisin,
535 const IntList& elem_voisins) const
536{
537 double resultat = 0.;
538 int test = 0;
539
540// Premier test: si numero_som = sommet_voisin
541// Alors on peut tout de suite retourner le resultat
542 if (numero_som == sommet_voisin)
543 {
544 for (int i=0; i<elem_voisins.size(); i++)
545 {
546 resultat += zone_Vef().volumes(elem_voisins[i]);
547 }
548
549// On tient compte de la dimension dans nos calculs
550 resultat *= 2./((dimension+1)*(dimension+2));
551
552 return resultat;
553 }
554
555// Sinon:
556// On calcule la contribution au resultat
557
558 for (int i=0; i<elem_voisins.size() ; i++)
559 {
560 if (sommets_pour_element(elem_voisins[i]).contient(sommet_voisin))
561 {
562 resultat += zone_Vef().volumes(elem_voisins[i]);
563 test++;
564 }
565 }
566
567// On tient compte de la dimension dans laquelle on evolue
568 resultat *= 1./((dimension+1)*(dimension+2));
569
570// On verifier que sommet_voisin etait bien dans le voisinage de numero_voisin
571 if (test == 0)
572 {
573 Cerr << "Erreur Op_Diff_RotRot::remplir_som_som_EF." << finl;
574 Cerr << "sommet_voisin n'est pas dans le voisinage de numero_voisin."
