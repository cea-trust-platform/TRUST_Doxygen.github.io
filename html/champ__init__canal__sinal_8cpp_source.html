23Entree& champ_init_canal_sinal::readOn(Entree& is)
24{
25 int dim = lire_dimension(is, que_suis_je());
26 dimensionner(1, dim);
27 if (abs(dim) > 3)
28 {
29 Cerr << "Problem while reading the dimension of field champ_init_canal_sinal" << finl;
30 Cerr << "Check that it was correctly indicated : dim = " << dim << finl;
31 Process::exit();
32 }
33
34 ArrOfInt Verif(9);
35 Verif[3] = 1;
36 Verif[5] = 1;
37 Verif[6] = 1;
38 Verif[7] = 1;
39 Verif[8] = 1;
40 Motcles les_mots(9);
41 Motcle mot_lu;
42 {
43 les_mots[0] = "Ucent";
44 les_mots[1] = "h";
45 les_mots[2] = "ampli_bruit";
46 les_mots[3] = "ampli_sin";
47 les_mots[4] = "omega";
48 les_mots[5] = "dir_flow";
49 les_mots[6] = "dir_wall";
50 les_mots[7] = "min_dir_flow";
51 les_mots[8] = "min_dir_wall";
52 }
53 Motcle acc_ouverte("{");
54 Motcle acc_fermee("}");
55 is >> mot_lu;
56 if (mot_lu != acc_ouverte)
57 {
58 Cerr << "Expecting a { instead of " << mot_lu << "while reading intial condition" << finl;
59 }
60 is >> mot_lu;
61 while (mot_lu != acc_fermee)
62 {
63 int rang = les_mots.search(mot_lu);
64 switch(rang)
65 {
66 case 0:
67 is >> Ucent;
68 Cerr << "Centerline Longitudinal Velocity = " << Ucent << finl;
69// valeur par defaut de ampli_sin = Ucent/10.
70 ampli_sin = Ucent / 10.;
71 Verif[0] = 1;
72 break;
73 case 1:
74 is >> h;
75 Cerr << "Channel Half-height = " << h << finl;
76 Verif[1] = 1;
77 break;
78 case 2:
79 is >> ampli_bruit;
80 Cerr << "Noise Amplitude = " << ampli_bruit << finl;
81 Verif[2] = 1;
82 break;
83 case 3:
84 is >> ampli_sin;
85 Cerr << "Sine Perturbation Amplitude = " << ampli_sin << finl;
86 Verif[3] = 1;
87 break;
88 case 4:
89 is >> omega;
90 Cerr << "Omega = " << omega << finl;
91 Verif[4] = 1;
92 break;
93 case 5:
94 is >> dir_flow;
95 Cerr << "Flow direction : " << dir_flow << finl;
96 Verif[5] = 1;
97 break;
98 case 6:
99 is >> dir_wall;
100 Cerr << "The specified Wall direction : " << dir_wall << finl;
101 Verif[6] = 1;
102 break;
103 case 7:
104 is >> min_dir_flow;
105 Cerr << "Minimum value position in flow direction : " << min_dir_flow << finl;
106 Verif[7] = 1;
107 break;
108 case 8:
109 is >> min_dir_wall;
110 Cerr << "Minimum value position in wall direction : " << min_dir_wall << finl;
111 Verif[8] = 1;
112 break;
113 default:
114 {
115 Cerr << "Error in champ_init_canal_sinal::readOn " << finl;
116 Cerr << mot_lu << " is not understood by velocity initial condition generation." << finl;
117 Cerr << "The understood keywords are : " << les_mots << finl;
118 Process::exit();
119 }
120 }
121 is >> mot_lu;
122 }
123
124 if (dir_flow == dir_wall)
125 {
126 Cerr << "Error in champ_init_canal_sinal::readOn " << finl;
127 Cerr << "You gave the same direction for a wall and the flow direction : Impossible !" << finl;
128 Process::exit();
129 }
130
131 if ((dir_wall > 2) || (dir_flow > 2))
132 {
133 Cerr << "Error in champ_init_canal_sinal::readOn " << finl;
134 Cerr << "One of the specified directions are not possible." << finl;
135 Cerr << "Use : 0 for X, 1 for Y and 2 for Z !" << finl;
136 Process::exit();
137 }
138
139 int valmin = min_array(Verif);
140 if (valmin == 0)
141 {
142 for (int i = 0; i < 5; i++)
143 if (Verif[i] == 0) Cerr << "You didn't give a value for " << les_mots[i] << finl;
144
145 Cerr << "Error in champ_init_canal_sinal::readOn " << finl;
146 Cerr << "User needs to specify values for the parameters below.";
147 Cerr << "The syntax is : Ucent val_Ucent h val_h ampli_bruit val_ampli_bruit [ampli_sin val_ampli_sin] omega val_omega" << finl;
148 Cerr << "dir_flow val_dir_flow dir_wall val_dir_wall" << finl;
149 Process::exit();
150 }
151
152 return is;
153}

155DoubleVect& champ_init_canal_sinal::valeur_a(const DoubleVect& positions, DoubleVect& tab_valeurs) const
156{
157 Cerr << "values = " << tab_valeurs << finl;
158 Cerr << "In champ_init_canal_sinal::valeur_a nb_compo_=" << nb_compo_ << finl;
159 tab_valeurs.resize(nb_compo_);
160 switch(dimension)
161 {
162 case (1):
163 {
164 assert(nb_compo_ == 1);
165 tab_valeurs(dir_flow) = fx(positions(dir_flow));
166 break;
167 }
168 case (2):
169 {
170 assert(nb_compo_ <= 2);
171 tab_valeurs(dir_flow) = fx(positions(dir_flow), positions(dir_wall));
172 if (nb_compo_ == 2) tab_valeurs(dir_wall) = fy(positions(dir_flow), positions(dir_wall));
173 break;
174 }
175 case (3):
176 {
177 assert(nb_compo_ <= 3);
178 tab_valeurs(dir_flow) = fx(positions(dir_flow), positions(dir_wall), positions(3 - dir_flow - dir_wall));
179 if (nb_compo_ >= 2) tab_valeurs(dir_wall) = fy(positions(dir_flow), positions(dir_wall), positions(3 - dir_flow - dir_wall));
180 if (nb_compo_ >= 3) tab_valeurs(3 - dir_flow - dir_wall) = fz(positions(dir_flow), positions(dir_wall), positions(3 - dir_flow - dir_wall));
181 break;
182 }
183 }
184 return tab_valeurs;
185}

189 double val = 0.;
190 switch(dimension)

192 case (1):
193 val = fx(positions(dir_flow));
194 break;
195 case (2):
196 {
197 if (ncomp == 1 + dir_flow) val = fx(positions(dir_flow), positions(dir_wall));
198 else val = fy(positions(dir_flow), positions(dir_wall));
199 }
200 break;
201 case (3):
202 {
203 if (ncomp == 1 + dir_flow) val = fx(positions(dir_flow), positions(dir_wall), positions(3 - dir_flow - dir_wall));
204 else if (ncomp == 1 + dir_wall) val = fy(positions(dir_flow), positions(dir_wall), positions(3 - dir_flow - dir_wall));
205 else val = fz(positions(dir_flow), positions(dir_wall), positions(3 - dir_flow - dir_wall));
206 }
207 break;
208 default:
209 val = -1;
210 Process::exit();
211 break;
212 }
