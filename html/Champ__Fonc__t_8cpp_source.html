45Entree& Champ_Fonc_t::readOn(Entree& is)
46{
47 int nb_compo;
48 nb_compo=lire_dimension(is,que_suis_je());
49 if (nb_compo <= 0)
50 {
51 Cerr << "Error in Champ_Fonc_t::readOn : nb_compo <= 0" << finl;
52 assert(0);
53 exit();
54 }
55 dimensionner(1, nb_compo);
56 ft.dimensionner(nb_compo);
57 for (int i_parser = 0; i_parser < nb_compo; i_parser++)
58 {
59 Nom expression;
60 is >> expression;
61 const char* s = expression.getChar();
62 std::string ss(s);
63 for (auto & c: ss) c = toupper(c);
64 ft[i_parser].setNbVar(1);
65 ft[i_parser].setString(ss);
66 ft[i_parser].addVar("t");
67 ft[i_parser].parseString();
68 }
69
70 me_calculer(0.);
71
72 return is;
73}
