165 {
166 int elem0=face_voisin(fac,0);
167 int elem1=face_voisin(fac,1);
168 int val0=-1;
169 if (elem0!=-1) val0=marq_elem(elem0);
170 int val1=-1;
171 if (elem1!=-1) val1=marq_elem(elem1);
172
173 if (val0!=val1)
174 {
175 if ((val0==1)||(val1==1))
176 {
177 if (marq[fac]!=-1) //pas un joint
178 {
179 marq[fac]=1;
180 nb_t++;
181 }
182 }
183 }
184 }
185
186 Cerr<<me()<<" nb_elem_m "<<nb_elem_m<<finl;
187 Cerr<<me()<<" nb_t "<<nb_t<<finl;
188 IntTab& les_elems=zone.les_elems();
189 const IntTab& les_elems_old=zone_vf.zone().les_elems();
190 int nb_som_elem=les_elems_old.dimension(1);
191 les_elems.resize(nb_elem_m,nb_som_elem);
192 const IntTab& face_sommets=zone_vf.face_sommets();
193 int nb=0;
194 for (int ele=0; ele<nb_elem; ele++)
195 {
196 if (marq_elem(ele)==1)
197 {
198 for (int k=0; k<nb_som_elem; k++)
199 les_elems(nb,k)=les_elems_old(ele,k);

201 }
202 }
203 assert(nb==nb_elem_m);

211 indfaces.resize(nb_t,nb_som_face);
212 IntTab& facesv=les_faces.voisins();
213 facesv.resize(nb_t,2);
214 facesv=-1;
215
216
217 nb=0;
218 for (int fac=0; fac<nb_faces; fac++)
219 if (marq[fac]==1)
220 {
221 for (int s=0; s<nb_som_face; s++)
222 indfaces(nb,s)=face_sommets(fac,s);
223// on calcule la normale
224 nb++;
225 }
226 Cerr<<finl;;
227 zone.faces_bord().add(faces);
228 zone.faces_bord().associer_zone(zone);
229 zone.type_elem().associer_zone(zone);
230 zone.fixer_premieres_faces_frontiere();
231
232
233
234 assert(nb==nb_t);
235 NettoieNoeuds::nettoie(dom);
236
237 return is;
