43void Interprete_geometrique_base::associer_domaine(Entree& is)
44{
45 Cerr << "Association of the domain on which the interpreter applies " << que_suis_je() << finl;
46 Nom nom_dom;
47 is >> nom_dom;
48 if(!sub_type(Domaine, objet(nom_dom)))
49 {
50 Cerr << nom_dom << " is of type " << objet(nom_dom).que_suis_je() << finl;
51 Cerr << "we know to treat only one domain with the interpreter " << que_suis_je() << finl;
52 exit();
53 }
54 domains_.add(ref_cast(Domaine, objet(nom_dom)));
55}

57Entree& Interprete_geometrique_base::interpreter(Entree& is)
58{
59 Cerr << "Execution of the interpreter " << que_suis_je() << finl;
60 Process::imprimer_ram_totale();
61// Appel a l'interprete fils
62 interpreter_(is);
63
64// On met a jour un certain nombre de choses
65// suite a la modification des domaines
66 for (int j=0; j<domains_.size(); j++)
67 for(int i=0; i<domaine(j).nb_zones() ; i++)
68 {
69 Zone& zone = domaine(j).zone(i);
70 zone.invalide_octree();
71 zone.faces_bord().associer_zone(zone);
72 zone.faces_joint().associer_zone(zone);
73 zone.faces_raccord().associer_zone(zone);
74 zone.faces_int().associer_zone(zone);
75 zone.type_elem().associer_zone(zone);
76 zone.fixer_premieres_faces_frontiere();
77 zone.associer_domaine(domaine(j));
78 }
79 return is;
80}

83void Interprete_geometrique_base::mettre_a_jour_sous_zone(Zone& zone, int& elem, int num_premier_elem, int nb_elem) const
84{
85 Domaine& dom=zone.domaine();
86 for (int ssz=0; ssz<dom.nb_ss_zones(); ssz++)
87 {
88 Sous_Zone& sous_zone=dom.ss_zone(ssz);
89 if ((que_suis_je()=="Raffiner_anisotrope")
90 || (que_suis_je()=="Raffiner_isotrope")
91 || (que_suis_je()=="Remove_elem")
92 || (que_suis_je()=="Tetraedriser_homogene")
93 || (que_suis_je()=="Tetraedriser_homogene_compact")
94 || (que_suis_je()=="Tetraedriser_homogene_fin")
95 || (que_suis_je()=="Tetraedriser_par_prisme")
96 || (que_suis_je()=="Tetraedriser")
97 || (que_suis_je()=="Trianguler")
98 || (que_suis_je()=="Trianguler_fin")
99 || (que_suis_je()=="Trianguler_H")
100 || (que_suis_je()=="VerifierCoin"))
101 {
102 const int nb_poly=sous_zone.nb_elem_tot();
103 for (int nb_p=0; nb_p<nb_poly; nb_p++)
104 {
105 if (elem==sous_zone[nb_p])
106 {
107 Cerr << " The element " << elem << " which will be modified, is included to the subarea \"" << sous_zone.le_nom() << "\". So the element";
108 if (nb_elem>1) Cerr << "s";
109 Cerr << " generated [";
110 for(int num_elem=num_premier_elem; num_elem<num_premier_elem+nb_elem; num_elem++)
111 {
112 sous_zone.add_poly(num_elem);
113 if (num_elem!=num_premier_elem) Cerr << " ";
114 Cerr << num_elem;
115 }
116 Cerr << "] will be added to this subarea." << finl;
117 }
118 }
119 }
120 else
121 {
122 Cerr << finl;
123 Cerr << "You have defined the subarea \"" << sous_zone.le_nom() << "\" before using the interpreter \"" << que_suis_je() << "\"." << finl;
124 Cerr << "To correctly mark the good elements, set the subarea after to have used the interpreter." << finl;
125 exit();
126 }
127 }
128}
