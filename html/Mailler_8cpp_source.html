95//! Fonction principale de l'interprete Mailler<br>Structure du jeu de donnee (en dimension 2) :<br>Mailler dom<br>{<br>[Epsilon eps]<br>[objet1]<br>,<br>[objet2]<br>...<br>}<br>Deux points seront confondus des que la distance entre eux est<br>inferieure a Epsilon.
96/** \return Entree\&<br>
97Signification: le flot d'entree<br>
98Contraintes: */
99Entree& Mailler::interpreter_(Entree& is)
100{
101 if (Process::nproc()>1)
102 {
103 Cerr << "Mailler keyword can't be used in parallel calculation!" << finl;
104 Cerr << "Use only Scatter keyword to read partitioned mesh." << finl;
105 exit();
106 }
107 double precision_geom_sa=precision_geom;
108 associer_domaine(is);
109 Domaine& dom=domaine();
110
111// On debloque les structures pour modifier le domaine
112 Scatter::uninit_sequential_domain(dom);
113
114 Motcle motlu;
115 is >> motlu;
116 if (motlu != "{")
117 {
118 Cerr << "We expected a {" << finl;
119 exit();
120 }
121// Valeur par defaut: precision_geom
122 dom.fixer_epsilon(Objet_U::precision_geom);
123 Nom typ_zone;
124 do
125 {
126 is >> typ_zone;
127 motlu = typ_zone;
128 if(motlu=="epsilon")
129 {
130 double eps;
131 is >> eps;
132 dom.fixer_epsilon(eps);
133// GF le eps du domaine ne sert plus quand on cherche les sommets
134// doubles;
135 precision_geom=eps;
136 is >> motlu;
137 if ((motlu!="}") && (motlu!=",") )
138 {
139 Cerr<<"The syntax of \"Mailler\" has changed. We expected \",\" after the definition of epsilon and not \"" <<motlu << "\"" <<finl;
140 exit();
141 }
142 verifie_syntaxe(motlu);
143 }
144 else if (motlu=="Domaine")
145 {
146 Cerr<<"The syntax of \"Mailler\" has changed. The keyword \"Domaine\" has been replaced by \"Domain\"" <<finl;
147 exit();
148 }
149 else if(motlu=="Domain")
150 {
151 Nom nom_dom;
152 is >> nom_dom;
153 Cerr << "Adding a domain " << nom_dom << finl;
154 Domaine& added_dom=ref_cast(Domaine, objet(nom_dom));
155 IntVect nums;
156 Zone& zone=dom.add(added_dom.zone(0));
157 zone.associer_domaine(dom);
158 dom.ajouter(added_dom.coord_sommets(), nums);
159 Scatter::uninit_sequential_domain(dom);
160 zone.renum(nums);
161 zone.associer_domaine(dom);
162 dom.comprimer();
163 dom.zone(0).fixer_premieres_faces_frontiere();
164 dom.zone(0).type_elem().associer_zone(dom.zone(0));
165 is >> motlu;
166 verifie_syntaxe(motlu);
167 }
168 else
169 {
170 DERIV(Zone) une_zone;
171 une_zone.typer(typ_zone);
172 Zone& ajoutee = dom.add(une_zone.valeur());
173 ajoutee.associer_domaine(dom);
174 is >> ajoutee;
175 ajoutee.associer_domaine(dom); //!< On renumerote!
176 is >> motlu;
177 verifie_syntaxe(motlu);
178 }
179 }
180 while(motlu != "}");
181// Dans le cas ou on a modifie precision geom, on le remet a la valeur
182// d'origine
183 precision_geom=precision_geom_sa;
184 dom.comprimer();
185 dom.zone(0).fixer_premieres_faces_frontiere();
186
187
188 NettoieNoeuds::nettoie(dom);
189
190 Scatter::init_sequential_domain(dom);
191 return is;
192}
