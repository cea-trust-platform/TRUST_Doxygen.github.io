50DoubleTab& Masse_VDF_Face::appliquer_impl(DoubleTab& sm) const
51{
52 assert(la_zone_VDF.non_nul());
53 assert(la_zone_Cl_VDF.non_nul());
54 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
55 const DoubleVect& porosite_face = zone_VDF.porosite_face();
56 const DoubleVect& volumes_entrelaces = zone_VDF.volumes_entrelaces();
57 const int nb_faces = zone_VDF.nb_faces(), N = sm.line_size();
58
59 if (sm.dimension(0) != nb_faces)
60 {
61 Cerr << "Masse_VDF_Face::appliquer : erreur dans la taille de sm" << finl;
62 exit();
63 }
64
65
66// Boucle sur les faces joint
67
68// Boucle sur les bords
69// Sur les faces qui portent des conditions aux limites de Dirichlet ou de Symetrie
70// la vitesse normale reste egale a sa valeur initiale.
71// Donc sur ces faces vpoint doit rester a 0.
72
73 for (int n_bord=0; n_bord<zone_VDF.nb_front_Cl(); n_bord++)
74 {
75
76// pour chaque Condition Limite on regarde son type
77 const Cond_lim& la_cl = la_zone_Cl_VDF->les_conditions_limites(n_bord);
78 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
79 const int ndeb = la_front_dis.num_premiere_face();
80 const int nfin = ndeb + la_front_dis.nb_faces();
81
82 if ( sub_type(Dirichlet,la_cl.valeur()) || sub_type(Dirichlet_homogene,la_cl.valeur()) )
83// Pour les faces de Dirichlet on met sm a 0
84 for (int f = ndeb; f < nfin; f++)
85 for (int n = 0; n < N; n++)
86 sm(f, n) = 0;
87 else if (sub_type(Symetrie,la_cl.valeur()))
88// Pour les faces de Symetrie on met vpoint a 0
89 for (int f = ndeb; f < nfin; f++)
90 for (int n = 0; n < N; n++)
91 sm(f, n) = 0;
92 else
93 for (int f = ndeb; f < nfin; f++)
94 for (int n = 0; n < N; n++)
95 sm(f, n) /= (volumes_entrelaces(f) * porosite_face(f));
96
97 }
98
99// Boucle sur les faces internes
100 const int ndeb = zone_VDF.premiere_face_int();
101 for (int f = ndeb; f < nb_faces; f++)
102 for (int n = 0; n < N; n++)
103 sm(f, n) /= (volumes_entrelaces(f) * porosite_face(f));
104// sm.echange_espace_virtuel();
105// Debog::verifier("Masse_VDF_Face::appliquer sm",sm);
106 return sm;
107}
