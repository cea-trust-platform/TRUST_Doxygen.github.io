30//! Lit les valeurs du champ uniforme par morceaux a partir d'un flot d'entree.<br>On lit le nom du domaine (nom_domaine) le nombre de composantes du champ (nb_comp) la valeur par defaut<br>du champ ainsi que les valeurs sur les sous zones.<br>Format:<br>Champ_Uniforme_Morceaux nom_domaine nb_comp<br>{ Defaut val_def sous_zone_1 val_1 ... sous_zone_i val_i }
31Entree& Champ_Uniforme_Morceaux::readOn(Entree& is)
32{
33 int dim;
34 Motcle motlu;
35 Nom nom;
36 int k, poly;
37 is >> nom;
38 interprete_get_domaine(nom);
39 Domaine& le_domaine=mon_domaine.valeur();
40 const IntTab& les_elems=le_domaine.zone(0).les_elems();
41 const int nb_som_elem = le_domaine.zone(0).nb_som_elem();
42 double x=0,y=0,z=0;
43
44 dim=lire_dimension(is,que_suis_je());
45 fixer_nb_comp(dim);
46 valeurs_.resize(0, dim);
47 le_domaine.creer_tableau_elements(valeurs_);
48 is >> nom;
49 motlu=nom;
50 if(motlu != Motcle("{") )
51 {
52 Cerr << "Error while reading a " << que_suis_je() << finl;
53 Cerr << "We expected a { instead of " << nom << finl;
54 exit();
55 }
56 is >> nom;
57 motlu=nom;
58 if(motlu != Motcle("defaut") )
59 {
60 Cerr << "Error while reading a " << que_suis_je() << finl;
61 Cerr << "We expected defaut instead of " << nom << finl;
62 exit();
63 }
64
65 VECT(Parser_U) fxyz(dim);
66 for( k=0; k< dim; k++)
67 {
68 Nom tmp;
69 is >> tmp;
70 fxyz[k].setNbVar(3);
71 fxyz[k].setString(tmp);
72 fxyz[k].addVar("x");
73 fxyz[k].addVar("y");
74 fxyz[k].addVar("z");
75 fxyz[k].parseString();
76 }
77
78 for( poly=0; poly<le_domaine.zone(0).nb_elem(); poly++)
79 {
80 x = y = z = 0;
81 int nsom = 0, e=-1;
82 for (int isom = 0; isom<nb_som_elem; isom++)
83 if ((e = les_elems(poly,isom)) >= 0)
84 {
85 x += le_domaine.coord(e,0);
86 y += le_domaine.coord(e,1);
87 if (dimension == 3) z += le_domaine.coord(e,2);
88 nsom++;
89 }
90 x = x/nsom;
91 y = y/nsom;
92 z = z/nsom;
93 for( k=0; k< dim; k++)
94 {
95 fxyz[k].setVar("x",x);
96 fxyz[k].setVar("y",y);
97 fxyz[k].setVar("z",z);
98 valeurs_(poly,k)=fxyz[k].eval();
99 }
100 }
101
102 is >> nom;
103 motlu=nom;
104 while (motlu != Motcle("}") )
105 {
106 REF(Sous_Zone) refssz=les_sous_zones.add(le_domaine.ss_zone(nom));
107 Sous_Zone& ssz = refssz.valeur();
108 for( k=0; k< dim; k++)
109 {
110 Nom tmp;
111 is >> tmp;
112 fxyz[k].setString(tmp);
113 fxyz[k].parseString();
114 }
115 for( poly=0; poly<ssz.nb_elem_tot(); poly++)
116 {
117 x = y = z = 0;
118 int nsom = 0, e=-1;
119 for (int isom = 0; isom<nb_som_elem; isom++)
120 if ((e = les_elems(ssz(poly),isom)) >= 0)
121 {
122 x += le_domaine.coord(e,0);
123 y += le_domaine.coord(e,1);
124 if (dimension == 3) z += le_domaine.coord(e,2);
125 nsom++;
126 }
127 x = x/nsom;
128 y = y/nsom;
129 z = z/nsom;
130 for( k=0; k< dim; k++)
131 {
132 fxyz[k].setVar("x",x);
133 fxyz[k].setVar("y",y);
134 fxyz[k].setVar("z",z);
135 valeurs_(ssz(poly),k)=fxyz[k].eval();
136 }
137 }
138 is >> nom;
139 motlu=nom;
140 }
141 valeurs().echange_espace_virtuel();
142 return is;
143}

145Champ_base& Champ_Uniforme_Morceaux::affecter_(const Champ_base& ch)
146{
147 if (sub_type(Champ_Uniforme_Morceaux, ch))
148 {
149 const Champ_Uniforme_Morceaux& chum = ref_cast(Champ_Uniforme_Morceaux, ch);
150 les_sous_zones = chum.sous_zones();
151 mon_domaine = chum.domaine();
152 valeurs_ = chum.valeurs();
153 }
154 else if (sub_type(Champ_Uniforme, ch))
155 {
156 const Champ_Uniforme& chu = ref_cast(Champ_Uniforme, ch);
157 const int nb_poly = valeurs_.dimension(0), dim = valeurs_.dimension(1);
158 const DoubleTab& ch_val = chu.valeurs();
159 for (int k = 0; k < dim; k++)
160 for (int poly = 0; poly < nb_poly; poly++) valeurs_(poly, k) = ch_val(0, k);
161 }
162 else
163 {
164 Cerr << "It is not known to affect a " << ch.que_suis_je() << " at a " << que_suis_je() << finl;
165 Process::exit();
166 }
167 return *this;
168}
