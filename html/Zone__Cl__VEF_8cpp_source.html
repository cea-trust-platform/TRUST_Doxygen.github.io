85//! etape de discretisation : dimensionnement des tableaux
86void Zone_Cl_VEF::associer(const Zone_VEF& la_zone_VEF)
87{
88 int nb_faces_non_std = la_zone_VEF.nb_faces_non_std();
89 const Elem_VEF& type_elem= la_zone_VEF.type_elem();
90 int nb_fa7_elem = type_elem.nb_facette();
91
92 {
93 IntVect renum;
94 la_zone_VEF.creer_tableau_faces(renum, ArrOfInt::NOCOPY_NOINIT);
95 renum = -1;
96// Marquer les faces non standard reelles (premieres faces dans la zone)
97 int i;
98 for (i = 0; i < nb_faces_non_std; i++)
99 renum[i] = 0;
100// Echange espace virtuel => marquer les faces virtuelles non standard
101 renum.echange_espace_virtuel();
102 MD_Vector md_vect;
103 MD_Vector_tools::creer_md_vect_renum_auto(renum, md_vect);
104 MD_Vector_tools::creer_tableau_distribue(md_vect, volumes_entrelaces_Cl_, Array_base::NOCOPY_NOINIT);
105 }
106 {
107// Construction du descripteur pour le tableau des elements "Cl":
108 MD_Vector md_vect;
109 MD_Vector_tools::creer_md_vect_renum(la_zone_VEF.rang_elem_non_std(), md_vect);
110
111// Creation de tableaux aux elements "Cl":
112 normales_facettes_Cl_.resize(0, nb_fa7_elem, dimension);
113 MD_Vector_tools::creer_tableau_distribue(md_vect, normales_facettes_Cl_, Array_base::NOCOPY_NOINIT);
114
115 vecteur_face_facette_Cl_.resize(0, nb_fa7_elem, dimension, 2);
116 MD_Vector_tools::creer_tableau_distribue(md_vect, vecteur_face_facette_Cl_, Array_base::NOCOPY_NOINIT);
117
118 MD_Vector_tools::creer_tableau_distribue(md_vect, type_elem_Cl_, Array_base::NOCOPY_NOINIT);
119 }
120}

143//! appele par completer() : remplissage de type_elem_Cl_ et<br>volumes_entrelaces_Cl_
144void Zone_Cl_VEF::remplir_volumes_entrelaces_Cl(const Zone_VEF& la_zone_VEF)
145{
146 const DoubleVect& volumes_entrelaces = la_zone_VEF.volumes_entrelaces();
147 const IntVect& rang_elem = la_zone_VEF.rang_elem_non_std();
148
149// Initialisation du tableau volumes_entrelaces_Cl
150// a priori les faces non standard ne sont pas touchees par les C.L
151
152 for (int i=0; i<la_zone_VEF.nb_faces_non_std(); i++)
153 volumes_entrelaces_Cl_[i] = volumes_entrelaces(i);
154
155// Calcul des valeurs des volumes entrelaces modifiees par les C.L:
156
157 int nb_poly_tot = la_zone_VEF.zone().nb_elem_tot();
158 ArrOfInt poly_fait(nb_poly_tot);
159 poly_fait = 0;
160 for(int i=0; i<les_conditions_limites_.size(); i++)
161 {
162 const Cond_lim_base& la_cl=les_conditions_limites_[i].valeur();
163 if ( (sub_type(Dirichlet,la_cl))
164 ||
165 (sub_type(Dirichlet_homogene,la_cl))
166 )
167 {
168 const Front_VF& le_bord= la_zone_VEF.front_VF(i);
169 const Frontiere& front=le_bord.frontiere();
170 int premiere=le_bord.num_premiere_face();
171 int derniere=premiere+le_bord.nb_faces();
172 for(int j=premiere; j<derniere; j++)
173 {
174 volumes_entrelaces_Cl_[j] =0;
175 int elem=la_zone_VEF.face_voisins(j,0);
176 int n_poly=rang_elem(elem);
177 if(!poly_fait[n_poly])
178 {
179 poly_fait[n_poly]=1;
180 const Elem_VEF& type_elem= la_zone_VEF.type_elem();
181 type_elem.modif_volumes_entrelaces(j,elem,la_zone_VEF,
182 volumes_entrelaces_Cl(),
183 type_elem_Cl(n_poly)) ;
184 }
185 }
186// faces virtuelles de bord :
187 const ArrOfInt& faces_virt = front.get_faces_virt();
188 int nb_faces_virt = faces_virt.size_array();
189 for (int ind_face=0; ind_face<nb_faces_virt; ind_face++)
190 {
191 int num_face = faces_virt[ind_face];
192 int elem=la_zone_VEF.face_voisins(num_face,0);
193 int n_poly=rang_elem(elem);
194 if(n_poly!=-1)
195 {
196 if(!poly_fait[n_poly])
197 {
198 poly_fait[n_poly]=1;
199 const Elem_VEF& type_elem= la_zone_VEF.type_elem();
200 type_elem.modif_volumes_entrelaces_faces_joints(num_face,elem,la_zone_VEF,
201 volumes_entrelaces_Cl(),
202 type_elem_Cl(n_poly)) ;
203 }
204 }
205 }
206 }
207 }
208}

213//! appele par completer() : remplissage de normales_facettes_Cl_<br>et vecteur_face_caette_Cl_<br>CHANGER LE NOM
214void Zone_Cl_VEF::remplir_normales_facettes_Cl(const Zone_VEF& la_zone_VEF)
215{
216 const Zone& z = la_zone_VEF.zone();
217 const Domaine& dom = z.domaine();
218 const DoubleTab& les_coords =dom.coord_sommets();
219 const IntTab& les_Polys = z.les_elems() ;
220 const Elem_VEF& elemvef = la_zone_VEF.type_elem();
221 const IntVect& rang_elem=la_zone_VEF.rang_elem_non_std();
222 const IntTab& elem_faces = la_zone_VEF.elem_faces();
223 const DoubleTab& xv=la_zone_VEF.xv();
224 const IntTab& KEL=elemvef.valeur().KEL();
225
226
227 int num_elem, fa7;
228 int isom,ncomp;
229 int idirichlet=-1,n1=-1,n2=-1,n3=-1;
230 int nb_elem = z.nb_elem();
231 int nsom = z.nb_som_elem();
232 int nfa7 = elemvef.nb_facette();
233// int nb_som_facette = zone().type_elem().nb_som_face();
234// if ( sub_type(Hexaedre_VEF,zone().type_elem().valeur())) { nb_som_facette--; }
235 int nb_som_facette=dimension;
236
237 IntVect num_som(nsom);
238 DoubleTab x(nsom,dimension);
239 DoubleVect xg(dimension);
240 DoubleVect xj0(dimension);
241 DoubleVect u(dimension);
242 DoubleVect v(dimension);
243
244// Calcul des valeurs des normales aux facettes modifiees par les C.L:
245
246 for (int elem=0; elem<nb_elem; elem++)
247 {
248
249 if (rang_elem(elem)!= -1)
250 {
251 num_elem = rang_elem(elem);
252
253// numero des sommets du polyedre:
254 for (isom = 0; isom<nsom; isom++)
255 num_som[isom] = les_Polys(elem,isom);
256
257
258// Calcul des coordonnees des sommets du polyedre:
259 for (isom = 0; isom<nsom; isom++)
260 for (ncomp =0; ncomp<dimension; ncomp++)
261 x(isom,ncomp) = les_coords(num_som[isom],ncomp);
262
263
264// Calcul des coordonnees du point G en fonction du type du polyedre
265// idirichlet= nb de faces de dirichlet de l'elem
266// pour triangle si idirichlet=2, n1=sommet confondu avec G
267// pour quadrangle se reporter a Quadri_VEF.cpp
268
269 elemvef.calcul_xg(xg,x,type_elem_Cl_[num_elem],
270 idirichlet,n1,n2,n3);
271// Calcul des valeurs du tableau normales_facettes_Cl:
272
273
274 for (fa7=0; fa7<nfa7; fa7++)
275 {
276 elemvef.creer_normales_facettes_Cl(normales_facettes_Cl(),fa7, num_elem,x,xg,z);
277// Correction des valeurs du tableau normales_facettes_Cl:
278 elemvef.modif_normales_facettes_Cl(normales_facettes_Cl(),fa7,num_elem,idirichlet,n1,n2,n3);
279 int num1= elem_faces(elem,KEL(0,fa7));
280 int num2= elem_faces(elem,KEL(1,fa7));
281 for (int dir=0; dir<dimension; dir++)
282 {
283 double coord_centre_fa7= xg(dir);
284
285
286 for (int num_som_fa7=0; num_som_fa7<nb_som_facette-1; num_som_fa7++)
287 {
288 int isom_loc = KEL(num_som_fa7+2,fa7);
289 int isom_glob = les_Polys(elem,isom_loc);
290 coord_centre_fa7 += les_coords(isom_glob,dir);
291 }
292 coord_centre_fa7 /= nb_som_facette;
293
294 vecteur_face_facette_Cl_(num_elem,fa7,dir,0)=coord_centre_fa7-xv(num1,dir);
295 vecteur_face_facette_Cl_(num_elem,fa7,dir,1)=coord_centre_fa7-xv(num2,dir);
296 }
297
298 }
299 }
300 }
301 normales_facettes_Cl_.echange_espace_virtuel();
302 vecteur_face_facette_Cl_.echange_espace_virtuel();
303}

323//! appele par remplir_volumes_entrelaces_Cl() : remplissage de type_elem_Cl_
324void Zone_Cl_VEF::remplir_type_elem_Cl(const Zone_VEF& la_zone_VEF)
325{
326 const Zone& z = la_zone_VEF.zone();
327 int nfac= z.type_elem().nb_faces(); //!< dans Elem_geom
328 const IntTab& elem_faces = la_zone_VEF.elem_faces();
329 const IntTab& face_voisins = la_zone_VEF.face_voisins();
330 const IntVect& rang_elem = la_zone_VEF.rang_elem_non_std();
331 type_elem_Cl_ = 0;
332 int elem, num_elem;
333
334 for(int i=0; i<les_conditions_limites_.size(); i++)
335 {
336 Cond_lim_base& la_cl=les_conditions_limites_[i].valeur();
337 if ( (sub_type(Dirichlet,la_cl))
338 ||
339 (sub_type(Dirichlet_homogene,la_cl))
340 )
341 {
342 const Front_VF& le_bord= la_zone_VEF.front_VF(i);
343 int premiere = le_bord.num_premiere_face();
344 int derniere = premiere+le_bord.nb_faces();
345 for(int j=premiere; j<derniere; j++)
346 {
347 elem=face_voisins(j,0);
348 int numero1,numero2 =-1;
349 for(numero1=0; numero1<nfac; numero1++)
350 if(elem_faces(elem,numero1) == j)
351 numero2 = numero1;
352 if(numero2 == -1)
353 Cerr << "Zone_Cl_VEF::creer_type_elem_Cl() PAS POSSIBLE!! " << finl;
354 num_elem = rang_elem(elem);
355 assert(num_elem!=-1);
356 if (sub_type(Tri_VEF,la_zone_VEF.type_elem().valeur()) || sub_type(Tetra_VEF,la_zone_VEF.type_elem().valeur()))
357 type_elem_Cl_[num_elem]+= Deux_Puissance((int)(nfac-1-numero2));
358 else if (sub_type(Quadri_VEF,la_zone_VEF.type_elem().valeur()) || sub_type(Hexa_VEF,la_zone_VEF.type_elem().valeur()) )
359 type_elem_Cl_[num_elem]+= trois_puissance((int)(nfac-1-numero2));
360 else
361 {
362 Cerr << "erreur dans Zone_Cl_VEF::remplir_type_elem_Cl mauvais typage d'element " << finl;
363 exit();
364 }
365
366 }
367 }
368 }
369
370 type_elem_Cl_.echange_espace_virtuel();
371}

376//! Impose les conditions aux limites a la valeur temporelle "temps" du<br>Champ_Inc
377void Zone_Cl_VEF::imposer_cond_lim(Champ_Inc& ch, double temps)
378{
379 DoubleTab& ch_tab = ch->valeurs(temps);
380 if (sub_type(Champ_P0_VEF,ch.valeur()))
381 {
382 ;
383 }
384 else if (sub_type(Champ_P1NC,ch.valeur()) || sub_type(Champ_Q1NC,ch.valeur()))
385 {
386 int nb_comp;
387
388 if (sub_type(Champ_P1NC,ch.valeur()))
389 {
390 const Champ_P1NC& chnc = ref_cast(Champ_P1NC,ch.valeur());
391 nb_comp = chnc.nb_comp();
392 int ndeb,nfin,num_face,ncomp;
393 for(int i=0; i<nb_cond_lim(); i++)
394 {
395 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
396 if (sub_type(Periodique,la_cl))
397 {
398
399// On fait en sorte que le champ ait la meme valeur
400// sur deux faces de periodicite qui sont en face l'une de l'autre
401 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl);
402 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
403 ndeb = le_bord.num_premiere_face();
404 nfin = ndeb + le_bord.nb_faces();
405 int voisine;
406 double moy;
407 for (num_face=ndeb; num_face<nfin; num_face++)
408 {
409 voisine = la_cl_perio.face_associee(num_face-ndeb) + ndeb;
410 if (nb_comp == 1)
411 {
412 if ( ch_tab[num_face] != ch_tab[voisine] )
413 {
414 moy = 0.5*(ch_tab[num_face] + ch_tab[voisine]);
415 ch_tab[num_face] = moy;
416 ch_tab[voisine] = moy;
417 }
418 }
419 else
420 {
421 for (ncomp=0; ncomp<nb_comp; ncomp++)
422 if ( ch_tab(num_face,ncomp) != ch_tab(voisine,ncomp) )
423 {
424 moy = 0.5*(ch_tab(num_face,ncomp) + ch_tab(voisine,ncomp));
425 ch_tab(num_face,ncomp) = moy;
426 ch_tab(voisine,ncomp) = moy;
427 }
428 }
429 }
430 }
431 else if ( (sub_type(Symetrie,la_cl) ) &&
432 (ch->nature_du_champ()==vectoriel) )
433 {
434 const Zone_VEF& zvef = chnc.zone_vef();
435 const DoubleTab& face_normales = zvef.face_normales();
436 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
437 ndeb = le_bord.num_premiere_face();
438 nfin = ndeb + le_bord.nb_faces();
439 if (nb_comp == dimension)
440 {
441 for (num_face=ndeb; num_face<nfin; num_face++)
442 {
443 double face_n, surf=0, flux=0;
444 for (ncomp=0; ncomp<nb_comp; ncomp++)
445 {
446 face_n = face_normales(num_face,ncomp);
447 flux += ch_tab(num_face,ncomp)*face_n;
448 surf += face_n*face_n;
449 }
450// flux /= surf; // Fixed bug: Arithmetic exception
451 if (std::fabs(surf)>=DMINFLOAT) flux /= surf;
452 for (ncomp=0; ncomp<nb_comp; ncomp++)
453 ch_tab(num_face,ncomp) -= flux*face_normales(num_face,ncomp);
454 }
455 }
456 }
457 else if ( sub_type(Dirichlet_entree_fluide,la_cl) )
458 {
459 const Dirichlet_entree_fluide& la_cl_diri = ref_cast(Dirichlet_entree_fluide,la_cl);
460 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
461 ndeb = le_bord.num_premiere_face();
462 nfin = ndeb + le_bord.nb_faces();
463 if (nb_comp == 1)
464 for (num_face=ndeb; num_face<nfin; num_face++)
465 ch_tab[num_face] = la_cl_diri.val_imp_au_temps(temps,num_face-ndeb);
466 else
467 {
468 for (ncomp=0; ncomp<nb_comp; ncomp++)
469 for (num_face=ndeb; num_face<nfin; num_face++)
470 ch_tab(num_face,ncomp) =la_cl_diri.val_imp_au_temps(temps,num_face-ndeb,ncomp);
471 }
472 }
473 else if ( sub_type(Scalaire_impose_paroi,la_cl) )
474 {
475 const Scalaire_impose_paroi& la_cl_diri = ref_cast(Scalaire_impose_paroi,la_cl);
476 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
477 ndeb = le_bord.num_premiere_face();
478 nfin = ndeb + le_bord.nb_faces();
479 if (nb_comp == 1)
480 for (num_face=ndeb; num_face<nfin; num_face++)
481 {
482 ch_tab[num_face] = la_cl_diri.val_imp_au_temps(temps,num_face-ndeb);
483 }
484 else
485 {
486 for (ncomp=0; ncomp<nb_comp; ncomp++)
487 for (num_face=ndeb; num_face<nfin; num_face++)
488 ch_tab(num_face,ncomp) =la_cl_diri.val_imp_au_temps(temps,num_face-ndeb,ncomp);
489 }
490 }
491 else if ( (sub_type(Dirichlet_paroi_fixe,la_cl) ) &&
492 (ch->nature_du_champ()==multi_scalaire) )
493 {
494 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
495 ndeb = le_bord.num_premiere_face();
496 nfin = ndeb + le_bord.nb_faces();
497 if (nb_comp == 1)
498 for (num_face=ndeb; num_face<nfin; num_face++)
499 ch_tab[num_face] = 0;
500 else
501 {
502 for (ncomp=0; ncomp<nb_comp; ncomp++)
503 for (num_face=ndeb; num_face<nfin; num_face++)
504 {
505 ch_tab(num_face,0) = 0;
506 ch_tab(num_face,1) = 0;
507 }
508 }
509 }
510 else if ( sub_type(Dirichlet_paroi_fixe,la_cl) )
511 {
512 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
513 ndeb = le_bord.num_premiere_face();
514 nfin = ndeb + le_bord.nb_faces();
515 if (nb_comp == 1)
516 for (num_face=ndeb; num_face<nfin; num_face++)
517 ch_tab[num_face] = 0;
518 else
519 {
520 for (ncomp=0; ncomp<nb_comp; ncomp++)
521 for (num_face=ndeb; num_face<nfin; num_face++)
522 ch_tab(num_face,ncomp) = 0;
523 }
524 }
525 else if ( sub_type(Dirichlet_paroi_defilante,la_cl) )
526 {
527 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
528 const Dirichlet_paroi_defilante& la_cl_diri=ref_cast(Dirichlet_paroi_defilante,la_cl);
529 const DoubleTab& face_normales = chnc.zone_vef().face_normales();
530 ndeb = le_bord.num_premiere_face();
531 nfin = ndeb + le_bord.nb_faces();
532 if (nb_comp != dimension)
533 {
534 Cerr << "Cas non prevu dans Zone_Cl_VEF::imposer_cond_lim." << finl;
535 exit();
536 }
537 else
538 {
539 for (num_face=ndeb; num_face<nfin; num_face++)
540 {
541 double surf=0, flux=0;
542 for (ncomp=0; ncomp<nb_comp; ncomp++)
543 {
544 double face_n = face_normales(num_face,ncomp);
545 flux += la_cl_diri.val_imp_au_temps(temps,num_face-ndeb, ncomp)*face_n;
546 surf += face_n*face_n;
547 }
548// flux /= surf; // Fixed bug: Arithmetic exception
549 if (std::fabs(surf)>=DMINFLOAT) flux /= surf;
550 for (ncomp=0; ncomp<nb_comp; ncomp++)
551 ch_tab(num_face,ncomp) = la_cl_diri.val_imp_au_temps(temps,num_face-ndeb, ncomp) - flux * face_normales(num_face,ncomp);
552 }
553 }
554 }
555 }
556 }
557 else if (sub_type(Champ_Q1NC,ch.valeur()))
558 {
559 const Champ_Q1NC& chnc = ref_cast(Champ_Q1NC,ch.valeur());
560 nb_comp = chnc.nb_comp();
561 int ndeb,nfin,num_face,ncomp;
562 for(int i=0; i<nb_cond_lim(); i++)
563 {
564 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
565 if (sub_type(Periodique,la_cl))
566 {
567// On fait en sorte que le champ ait la meme valeur
568// sur deux faces de periodicite qui sont en face l'une de l'autre
569 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl);
570 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());

576 {

578 if (nb_comp == 1)
579 {

581 {
582 moy = 0.5*(ch_tab[num_face] + ch_tab[voisine]);

584 ch_tab[voisine] = moy;
585 }
586 }
587 else
588 {
589 for (ncomp=0; ncomp<nb_comp; ncomp++)
590 if ( ch_tab(num_face,ncomp) != ch_tab(voisine,ncomp) )
591 {
592 moy = 0.5*(ch_tab(num_face,ncomp) + ch_tab(voisine,ncomp));
593 ch_tab(num_face,ncomp) = moy;
594 ch_tab(voisine,ncomp) = moy;
595 }
596 }

598 }

605 ndeb = le_bord.num_premiere_face();
606 nfin = ndeb + le_bord.nb_faces();
607 if (nb_comp == dimension)
608 {
609 for (num_face=ndeb; num_face<nfin; num_face++)
610 {
611 double face_n, surf=0, flux=0;
612 for (ncomp=0; ncomp<nb_comp; ncomp++)
613 {
614 face_n = face_normales(num_face,ncomp);
615 flux += ch_tab(num_face,ncomp)*face_n;
616 surf += face_n*face_n;
617 }
618// flux /= surf; // Fixed bug: Arithmetic exception
619 if (std::fabs(surf)>=DMINFLOAT) flux /= surf;
620 for (ncomp=0; ncomp<nb_comp; ncomp++)
621 ch_tab(num_face,ncomp) -= flux*face_normales(num_face,ncomp);
622 }
623 }
624 }
625 else if ( sub_type(Dirichlet_entree_fluide,la_cl) )
626 {

629 ndeb = le_bord.num_premiere_face();
630 nfin = ndeb + le_bord.nb_faces();
631 if (nb_comp == 1)
632 for (num_face=ndeb; num_face<nfin; num_face++)
633 ch_tab[num_face] = la_cl_diri.val_imp_au_temps(temps,num_face-ndeb);
634 else
635 {
636 for (ncomp=0; ncomp<nb_comp; ncomp++)
637 for (num_face=ndeb; num_face<nfin; num_face++)
638 ch_tab(num_face,ncomp) =la_cl_diri.val_imp_au_temps(temps,num_face-ndeb,ncomp);
639 }
640 }
641 else if ( sub_type(Scalaire_impose_paroi,la_cl) )
642 {
643 const Scalaire_impose_paroi& la_cl_diri = ref_cast(Scalaire_impose_paroi,la_cl);
644 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
645 ndeb = le_bord.num_premiere_face();
646 nfin = ndeb + le_bord.nb_faces();
647 if (nb_comp == 1)
648 for (num_face=ndeb; num_face<nfin; num_face++)
649 {

662 ndeb = le_bord.num_premiere_face();
663 nfin = ndeb + le_bord.nb_faces();
664 if (nb_comp == 1)
665 for (num_face=ndeb; num_face<nfin; num_face++)
666 ch_tab[num_face] = 0;
667 else
668 {
669 for (ncomp=0; ncomp<nb_comp; ncomp++)
670 for (num_face=ndeb; num_face<nfin; num_face++)
671 ch_tab(num_face,ncomp) = 0;
672 }
673 }
674 else if ( sub_type(Dirichlet_paroi_defilante,la_cl) )
675 {
676 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
677 const Dirichlet_paroi_defilante& la_cl_diri=ref_cast(Dirichlet_paroi_defilante,la_cl);
678 ndeb = le_bord.num_premiere_face();

688 }

695 Cerr << "Le type de Champ_Inc " << ch->que_suis_je() << " n'est pas prevu en VEF\n" << finl;

698 }
