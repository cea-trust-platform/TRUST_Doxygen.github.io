97//! Calcule la conductivite
98void Loi_Etat_Multi_GP_base::calculer_lambda()
99{
100 const Champ_Don& mu = le_fluide->viscosite_dynamique();
101 const DoubleTab& tab_mu = mu.valeurs();
102 Champ_Don& lambda = le_fluide->conductivite();
103 DoubleTab& tab_lambda = lambda.valeurs();
104 const DoubleTab& tab_Cp = le_fluide->capacite_calorifique().valeurs();
105 int i, n = tab_lambda.size();
106
107// La conductivite est soit un champ uniforme soit calculee a partir de la viscosite dynamique et du Pr
108 if (sub_type(Champ_Fonc_Tabule,lambda.valeur()))
109 {
110 lambda.valeur().mettre_a_jour(temperature_.valeur().temps());
111 return;
112 }
113 if (!sub_type(Champ_Uniforme,lambda.valeur()))
114 {
115 if (sub_type(Champ_Uniforme,mu.valeur()))
116 {
117 double mu0 = tab_mu(0,0);
118 for (i=0 ; i<n ; i++) tab_lambda(i,0) = mu0 * Cp_ / Pr_;
119 }
120 else
121 for (i=0 ; i<n ; i++) tab_lambda(i,0) = tab_mu(i,0) * tab_Cp(i,0) / Pr_;
122 }
123 tab_lambda.echange_espace_virtuel();
124 Debog::verifier("tab_lambda",tab_lambda);
125}

140//! Calcule la diffusivite
141void Loi_Etat_Multi_GP_base::calculer_alpha()
142{
143 const Champ_Don& lambda = le_fluide->conductivite();
144 const DoubleTab& tab_lambda = lambda.valeurs();
145 Champ_Don& alpha=le_fluide->diffusivite();
146 DoubleTab& tab_alpha = le_fluide->diffusivite().valeurs();
147 const DoubleTab& tab_rho = le_fluide->masse_volumique().valeurs();
148 const DoubleTab& tab_Cp = le_fluide->capacite_calorifique().valeurs();
149 int i, n = tab_alpha.size(), isVDF = 0;
150 if (alpha.valeur().que_suis_je()=="Champ_Fonc_P0_VDF") isVDF = 1;
151
152 if (isVDF)
153 {
154 if (sub_type(Champ_Uniforme,lambda.valeur()))
155 {
156 double lambda0 = tab_lambda(0,0);
157 for (i=0 ; i<n ; i++) tab_alpha(i,0) = lambda0 / ( tab_rho(i,0) * tab_Cp(i,0) );
158 }
159 else
160 {
161 for (i=0 ; i<n ; i++) tab_alpha(i,0) = tab_lambda(i,0) / ( tab_rho(i,0) * tab_Cp(i,0) );
162 }
163 }
164 else //!< VEF
165 {
166 const IntTab& elem_faces=ref_cast(Zone_VF,le_fluide->vitesse().zone_dis_base()).elem_faces();
167 double rhoelem, Cpelem;
168 int face, nfe = elem_faces.line_size();
169
170 if (sub_type(Champ_Uniforme,lambda.valeur()))
171 {
172 double lambda0 = tab_lambda(0,0);
173 for (i=0 ; i<n ; i++)
174 {
175 rhoelem = 0;
176 Cpelem = 0;
177 for (face=0; face<nfe; face++)
178 {
179 rhoelem += tab_rho(elem_faces(i,face),0);
180 Cpelem += tab_Cp(elem_faces(i,face),0);
181 }
182 rhoelem /= nfe;
183 Cpelem /= nfe;
184 tab_alpha(i,0) = lambda0 / ( rhoelem * Cpelem );
185 }
186 }
187 else
188 {
189 for (i=0 ; i<n ; i++)
190 {
191 rhoelem = 0;
192 Cpelem = 0;
193 for (face=0; face<nfe; face++)
194 {
195 rhoelem += tab_rho(elem_faces(i,face),0);
196 Cpelem += tab_Cp(elem_faces(i,face),0);
197 }
198 rhoelem /= nfe;
199 Cpelem /= nfe;
200 tab_alpha(i,0) = tab_lambda(i,0) / ( rhoelem * Cpelem );
201 }
202 }
203 }
204 tab_alpha.echange_espace_virtuel();
205 Debog::verifier("alpha",tab_alpha);
206}

250void Loi_Etat_Multi_GP_base::calculer_tab_mu(const DoubleTab& mu, int size)
251{
252// Dynamic viscosity is on elem
253 DoubleTab& tab_mu = le_fluide->viscosite_dynamique().valeurs();
254 const int nb_elem = tab_mu.size();
255 if (nb_elem==size) //!< VDF
256 {
257 for (int elem=0; elem < nb_elem; elem++) tab_mu(elem,0) = mu(elem);
258 }
259 else //!< VEF (average on elem from values on face);
260 {
261 const IntTab& elem_faces=ref_cast(Zone_VF,le_fluide->vitesse().zone_dis_base()).elem_faces();
262 const int nfe = elem_faces.line_size();
263 tab_mu = 0;
264 for (int elem=0; elem<nb_elem; elem++)
265 {
266 for (int face = 0; face < nfe; face++) tab_mu(elem,0) += mu(elem_faces(elem, face));
267
268 tab_mu(elem,0) /= nfe;
269 }
270 }
271 tab_mu.echange_espace_virtuel();
272 Debog::verifier("tab_mu", tab_mu);
273}
