42static inline char char_uppercase(char c)
43{
44 if (c >= 'a' && c <= 'z')
45 c += 'A' - 'a';
46 return c;
47}
48
49static inline int strcmp_uppercase(const char *n1, const char *n2)
50{
51 int i = 0;
52 unsigned char c1, c2;
53 int delta;
54 do
55 {
56 c1 = (unsigned char) char_uppercase(n1[i]);
57 c2 = (unsigned char) char_uppercase(n2[i]);
58 delta = c1 - c2;
59 i++;
60 }
61 while ((delta == 0) && (c1 != 0) && (c2 != 0));
62 return delta;
63}

105//! Recherche le type de nom "nom" dans la liste de types enregistres<br>par une recherche binaire.<br>On compare les chaines converties en majuscules.<br>On range dans "index" l'indice du type, s'il a ete trouve<br>et sinon l'indice du type juste apres (dans ce cas, on a<br>les_types[index-1]->n < nom < les_types[index]->n )<br>Si le type a ete trouve on renvoie 1, sinon 0.
106int Type_info::search_type_info_name(const char *nom, int& index)
107{
108 assert(nom != 0);
109// [imin..imax] est l'intervalle ou se trouve l'index recherche
110 int imin = 0;
111 int imax = nb_classes;
112 while (imax > imin)
113 {
114// On a toujours milieu < imax
115 int milieu = (imin + imax) / 2;
116 int comparaison = strcmp_uppercase(nom, les_types[milieu]->name());
117 if (comparaison == 0)
118 {
119 index = milieu;
120 return 1;
121 }
122 if (comparaison < 0)
123 {
124// nom < les_types[milieu]
125// l'index recherche est donc inferieur ou egal a "milieu"
126 imax = milieu;
127 }
128 else
129 {
130// nom > les_types[milieu]
131// l'index recherche est donc strictement superieur a "milieu"
132 imin = milieu + 1;
133 }
134 }
135 index = imax;
136 return 0;
137}

236//! Methode statique appelee par les constructeurs de Type_info pour<br>ajouter un nouveau type a la liste des types enregistres.<br>Verifie que le nom du type n'existe pas encore.
237void Type_info::ajouter_type(const Type_info& type_info)
238{
239// Split type_info.names_ into A if | is found (eg: n=A|B)
240// and add a synonym B
241 Nom A("");
242 Nom B("");
243 int synonym_found=0;
244 int i = 0;
245 unsigned char c;
246 do //!< Start loop
247 {
248 c = type_info.names_[i];
249// Find a |
250 if (c==124)
251 {
252 if (synonym_found==1)
253 {
254 Cerr << "More than 1 synonym found in " << type_info.names_ << finl;
255 Cerr << "Not supported yet !" << finl;
256 Process::exit();
257 }
258 else
259 {
260 synonym_found=1;
261 i++;
262 c = type_info.names_[i];
263 if (c==0)
264 {
265 Cerr << "Error in a classname which can't finished by | : " << type_info.names_ << finl;
266 Process::exit();
267 }
268 }
269 }
270// Build the synonym name:
271 if (synonym_found)
272 B+=c;
273 else
274 A+=c;
275 i++;
276 }
277 while (c!=0); //!< End loop
278
279 name_ = new Nom(A);
280 if (synonym_found)
281 {
282// Commented cause too verbose:
283// Cerr << "Keyword " << A << " has a synonym: " << B << finl;
284 synonym_name_ = new Nom(B);
285 synonym_ = new Synonyme_info(synonym_name_->getChar(),name_->getChar());
286 }
287// Verifie qu'il y a assez de place dans le tableau :
288 if (les_types_memsize <= nb_classes + 1)
289 {
290 static const int INCREMENT = 512;
291// Plus assez de place dans le tableau: on redimensionne.
292 les_types_memsize += INCREMENT;
293 const Type_info** nouveau = new const Type_info*[les_types_memsize];
294 for (int j = 0; j < nb_classes; j++)
295 nouveau[j] = les_types[j];
296 delete[] les_types;
297 les_types = nouveau;
298
299 int * temp = new int[les_types_memsize];
300 for (int j = 0; j < nb_classes; j++)
301 temp[j] = types_homonymes[j];
302 delete[] types_homonymes;
303 types_homonymes = temp;
304 }
305
306// On cherche ou mettre le type dans le tableau :
307 int existe_deja=Synonyme_info::est_un_synonyme(type_info.name());
308 if (existe_deja)
309 {
310 Cerr<<" class "<<type_info.name()<<" already exists as a synonym which is forbidden!!!!"<<finl;
311 Process::exit();
312 }
313 int index;
314 existe_deja = search_type_info_name(type_info.name(), index);
315 if (existe_deja)
316 {
317 types_homonymes[index] = 1;
318// GF: si on a un homonyme que la macro string_macro_trio fonctionne
319// on a un pb sauf pour les iterateurs
320 if (strcmp(string_macro_trio("VECT",titi),"VECT"))
321 {
322 if (strncmp(type_info.name(),"Iterateur_",10))
323 {
324 Cerr<<" type "<<type_info.name()<<" is in double and it is not allowed!!!!"<<finl;
325 Process::exit();
326 }
327 }
328 }
329 else
330 {
331// Ajout du type dans le tableau a l'indice "index":
332 for (int j = nb_classes; j > index; j--)
333 {
334 les_types[j] = les_types[j-1];
335 types_homonymes[j] = types_homonymes[j-1];
336 }
337 les_types[index] = &type_info;
338 types_homonymes[index] = 0;
339 nb_classes++;
340 }
341}

485//! Test d'appartenance d'un type dans les types de bases du type considere<br>si direct == 0<br>renvoie 1 si (*p) fait partie des bases de (*this)<br>renvoie 0 sinon.<br>si direct != 0<br>renvoie 1 si (*p) fait partie des bases de (*this)<br>ou des meres directes ou non de (*this)<br>renvoie 0 sinon.
486/** \return int<br>
487Signification: code de retour (0 ou 1)<br>
488Contraintes: */
489int Type_info::has_base(const Type_info* p, int direct) const
490{
491// recherche de p->name() dans b
492// si trouve return 1
493// sinon si pas direct return 0;
494// sinon recherche dans les bases de b
495// Modif de B. Mathieu: test uniquement sur l'adresse du type_info,
496// pas sur le nom du type...
497 if (p == 0)
498 {
499 return 0;
500 }
501 else
502 {
503 if (p == this)
504 {
505 return 1;
506 }
507 else
508 {
509 for (int i = 0; i < nb_bases_; i++)
510 if (b[i] == p)
511 return 1;
512 if (!direct)
513 {
514// Verifier les ancetres de niveau superieur
515 for (int i = 0; i < nb_bases_; i++)
516 if (b[i]->has_base(p, direct)) return 1;
517 }
518 }
519 }
520 return 0;
521}

599int Type_info::les_sous_types(const Type_info& mere, Noms& les_sous_types)
600{
601 int compteur=0;
602 int i= nb_classes;
603// Modif B. Mathieu: name() ne renvoie plus un static.

605// Cerr << "---------" << (const char*) nom_mere << finl;

608 if( les_types[i]->has_base(nom_mere, 0) )
609 if (!les_types[i]->same(nom_mere))
610 {
611 compteur++;
612 }
613 }
614// Cerr << compteur << finl ;
615 if(compteur==0) return 0;
616 les_sous_types.dimensionner(compteur);
617 compteur=0;
618 i= nb_classes;
619 while(i--)
620 {
621 if( les_types[i]->has_base(nom_mere, 0) )
622 if (!les_types[i]->same(nom_mere))
623 {
624 les_sous_types[compteur++]=les_types[i]->name();
625 }
626 }
627 return compteur;
628}
