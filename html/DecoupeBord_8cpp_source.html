35void create_listb_from_domaine2(const Domaine& dom1, const Domaine& dom2,const Noms& nomdec)
36{
37 const DoubleTab& xs1=dom1.coord_sommets();
38 const Zone& zone1=dom1.zone(0);
39 const DoubleTab& xs2=dom2.coord_sommets();
40 const Zone& zone2=dom2.zone(0);
41
42 int nbfr=zone1.nb_front_Cl();
43
44 SFichier listb(dom1.le_nom()+".boundary_list");
45
46 for (int l=0; l<nbfr; l++)
47 {
48 const Frontiere& fr1=zone1.frontiere(l);
49 const Nom& nomfr1=fr1.le_nom();
50 int l2=zone2.rang_frontiere(nomfr1);
51
52 if(l2<0)
53 {
54 Cerr << " The domains " << dom1.le_nom() << " and " << dom2.le_nom() << " do not have the same names of boundaries " << finl;
55 Cerr<<nomfr1<<" not found in "<<dom1.le_nom() <<finl;
56 Cerr << " Make sure they were generated in a similar way (almost to the mesh size) " << finl;
57 Process::exit();
58 }
59 const Frontiere& fr2=zone2.frontiere(l2);
60
61 if (nomdec.search(nomfr1)!=-1)
62 {
63 int nbfaces1=fr1.nb_faces();
64 const IntTab& sommets_face1 = (ref_cast(Frontiere,fr1)).les_sommets_des_faces();
65
66 int nbfaces2=fr2.nb_faces();
67 const IntTab& sommets_face2 = (ref_cast(Frontiere,fr2)).les_sommets_des_faces();
68
69 ArrsOfInt faces_associees(nbfaces2);
70
71 ArrOfDouble xg1(Objet_U::dimension); //!< centre de gravite face de fr1
72 ArrOfDouble xg2(Objet_U::dimension); //!< centre de gravite face de fr2
73
74 int face1,face2,i,j;
75 int nb_som_face=sommets_face1.dimension(1);
76
77
78 for (face1=0; face1<nbfaces1; face1++) //!< Generation de Liste_faces_associees
79 {
80 xg1=0.;
81
82 for (i=0; i<nb_som_face; i++)
83 for (j=0; j<Objet_U::dimension; j++)
84 xg1[j]+=xs1(sommets_face1(face1,i),j); //!< centre de gravite de la face1 (non divise par nb_som_face)
85
86// recherche de la face de dom2 correspondante
87
88 double dist_min=1.e6;
89 int face_min=-1;
90
91 for (face2=0; face2<nbfaces2; face2++)
92 {
93 xg2=0.;
94
95 for (i=0; i<nb_som_face; i++)
96 for (j=0; j<Objet_U::dimension; j++)
97 xg2[j]+=xs2(sommets_face2(face2,i),j); //centre de gravite de la face2 (non divise par nb_som_face)
98
99 double dist=0.;
100
101 for (j=0; j<Objet_U::dimension; j++) dist+=(xg2[j]-xg1[j])*(xg2[j]-xg1[j]);
102
103 if(dist<dist_min)
104 {
105 dist_min=dist ;
106 face_min=face2;
107 }
108 }
109
110 faces_associees[face_min].set_smart_resize(1);
111 faces_associees[face_min].append_array(face1);
112 }
113
114 for (face2=0; face2<nbfaces2; face2++) //!< Ecriture des fichiers de sortie
115 {
116 Nom nomfic(nomfr1);
117 nomfic+="%";
118 Nom n;
119 n=face2;
120 nomfic+=n;
121 SFichier s(nomfic);
122 s<<faces_associees[face2]<<finl;
123 listb<<nomfic<<finl;
124 s.close();
125 }
126
127 }
128 }
129
130 listb.close();

133void create_listb_from_xyz(const Domaine& dom1,const Noms& nomdec,const Noms& expr)
134{
135 int n1=nomdec.size();
136 int n2=expr.size();
137 if(n1!=n2)

144 const Zone& zone1=dom1.zone(0);

172

182

204

208 nomfic+="%";
209 Nom n;
210 n=face2;
211 nomfic+=n;

214 listb<<nomfic<<finl;
215 s.close();
216 }
217 }
218 }

223void create_listb_naif(const Domaine& dom1,const Noms& nomdec,const ArrOfInt& nbdec)
224{
225 int n1=nomdec.size();
226 int n2=nbdec.size_array();
227 if(n1!=n2)
228 {
229 Cerr << "The numbers of boundaries in 'bords_a_decouper' and 'nb_parts_naif' are not the same : check your data file"<<finl;
230 Process::exit();
231 }
232
233 SFichier listb(dom1.le_nom()+".boundary_list");
234 const Zone& zone=dom1.zone(0);
235 int nbfr=zone.nb_front_Cl();
236 for (int l=0; l<nbfr; l++)
237 {
238 const Frontiere& fr=zone.frontiere(l);
239 int nbfaces=fr.nb_faces();
240 int compt=0;
241 const Nom& nomfr=fr.le_nom();
242 int inc=nomdec.search(nomfr);
243 if (inc!=-1)
244 {
245 int nb=nbdec[inc];
246 nb=std::min(nb,nbfaces);
247 if (nbfaces!=0)
248 {
249 int nc=nbfaces/nb;
250
251// Cerr<<nbfaces<<finl;
252 if ((nb*nc)<nbfaces) nc++;
253// Cerr<<"nc "<<nc<<finl;
254 Cerr<<nomfr<<" splitting in "<<nb<<" parts, average number of faces "<<nbfaces/(1.*nb)<<finl;
255// Cerr<<"nombre moyen de faces "<<nc<<" bord decoupe "<<nomfr<<" en "<<nbfaces/nc<<" parties"<<finl;
256 for (int i=0; i<nb; i++)
257 {
258 if (i==nb-1) nc=nbfaces-compt;
259 IntVect p(nc);
260 for (int j=0; j<nc; j++) p[j]=compt++;
261 Nom nomfic(nomfr);
262 nomfic+="%";
263 Nom n;
264 n=i;
265 nomfic+=n;
266 SFichier s(nomfic);
267 s<<p<<finl;
268 listb<<nomfic<<finl;
269 s.close();
270 }
271 }
272
273 }
274 }
275 listb.close();

278void create_listb_geom(const Domaine& dom1,const Noms& nomdec,const ArrOfInt& nbdec)
279{
280 if((Objet_U::dimension)!=3)
281 {
282 Cerr<<"nb_parts_geom has been defined for dimension 3 only"<<finl;;
283 Process::exit();
284 }
285
286 int n1=nomdec.size();
287 int n2=nbdec.size_array();
288 if((2*n1)!=n2)
289 {
290 Cerr << "The 'nb_parts_geom' has to be twice the number of boundaries in 'bords_a_decouper' : check your data file"<<finl;
291 Process::exit();
292 }
293
294 const DoubleTab& xs=dom1.coord_sommets();
295 const Zone& zone=dom1.zone(0);
296 int nbfr=zone.nb_front_Cl();
297
298 Parser_U parser;
299 parser.setNbVar(3);
300 parser.addVar("x");
301 parser.addVar("y");
302 parser.addVar("z");
303 Noms expr_tot(n1);
304
305 for (int l=0; l<nbfr; l++)
306 {
307 const Frontiere& fr=zone.frontiere(l);
308 int nbfaces=fr.nb_faces();
309 const Nom& nomfr=fr.le_nom();
310 int inc=nomdec.search(nomfr);
311 if (inc!=-1)
312 {
313 int nb1=nbdec[2*inc];
314 int nb2=nbdec[2*inc+1];
315
316 if (nbfaces!=0)
317 {
318
319// int nbfaces=fr.nb_faces();
320 const IntTab& sommets_face = (ref_cast(Frontiere,fr)).les_sommets_des_faces();
321 int nb_som_face=dom1.les_sommets().dimension(1);
322
323 ArrOfDouble coord_min(3);
324 coord_min=1e6;
325 ArrOfDouble coord_max(3);
326 coord_max=-1e6;
327
328 for (int face=0; face<nbfaces; face++)
329 {
330 for (int i=0; i<nb_som_face; i++)
331 for (int j=0; j<3; j++)
332 {
333 coord_min[j] = std::min(coord_min[j],xs(sommets_face(face,i),j));
334 coord_max[j] = std::max(coord_max[j],xs(sommets_face(face,i),j));
335 }
336 }
337
338 Nom expr="0.+";
339
340 if(coord_min[0]==coord_max[0]) //!< plan x=cste
341 {
342 double dy=(coord_max[1]-coord_min[1])/nb1;
343 double dz=(coord_max[2]-coord_min[2])/nb2;
344
345 for (int j=1; j<nb1; j++)
346 {
347 double pas=j*dy;
348 expr+="(y>";
349 expr+=Nom(pas);
350 expr+=")+";
351 }
352 expr+=Nom(nb1);
353 expr+="*(";
354 for (int k=1; k<nb2; k++)
355 {
356 double pas=k*dz;
357 expr+="(z>";
358 expr+=Nom(pas);
359 expr+=")+";
360 }
361 }
362 else if (coord_min[1]==coord_max[1]) //!< plan y=cste
363 {
364 double dx=(coord_max[0]-coord_min[0])/nb1;
365 double dz=(coord_max[2]-coord_min[2])/nb2;
366
367 for (int i=1; i<nb1; i++)
368 {
369 double pas=i*dx;
370 expr+="(x>";
371 expr+=Nom(pas);
372 expr+=")+";
373 }
374 expr+=Nom(nb1);
375 expr+="*(";
376 for (int k=1; k<nb2; k++)
377 {
378 double pas=k*dz;
379 expr+="(z>";
380 expr+=Nom(pas);
381 expr+=")+";
382 }
383 }
384 else if (coord_min[2]==coord_max[2]) //!< plan z=cste
385 {
386 double dx=(coord_max[0]-coord_min[0])/nb1;
387 double dy=(coord_max[1]-coord_min[1])/nb2;
388
389 for (int i=1; i<nb1; i++)
390 {
391 double pas=i*dx;
392 expr+="(x>";
393 expr+=Nom(pas);
394 expr+=")+";
395 }
396 expr+=Nom(nb1);
397 expr+="*(";
398 for (int j=1; j<nb2; j++)
399 {
400 double pas=j*dy;
401 expr+="(y>";
402 expr+=Nom(pas);
403 expr+=")+";
404 }
405 }
406 else
407 {
408 Cerr <<"the boundary "<<nomfr<<" is not plane : nb_parts_geom can't be used"<<finl;
409 Process::exit();
410 }
411
412 expr+="0.)";
413 Cerr << "fontiere : "<<nomfr<<" - expression : " <<expr<<finl;
414 parser.setString(expr);
415 parser.parseString();
416 expr_tot[inc]=expr;
417
418 }//if (nbfaces!=0)
419 }//if (inc!=-1)
420 }//for (int l=0;l<nbfr;l++)
421
422 Cerr <<"create_listb_from_xyz .... "<<finl;
423 create_listb_from_xyz(dom1,nomdec,expr_tot);
424 Cerr <<" ... ok "<<finl;
425}

506void DecoupeBord::Decouper(Domaine& dom, const Nom& nom_file)
507{
508 Cerr << "Splitting to the radiation of the boundaries of the domain " << dom.le_nom() << finl;
509 int nb_faces_bord_sa=dom.zone(0). nb_faces_bord();
510 int nb_faces_raccord_sa=dom.zone(0). nb_faces_raccord();
511 if (nproc()!=1)
512 {
513 Cerr << " The splitting DecoupeBord can be done only in sequential " << finl;
514 exit();
515 }
516
517
518// on ecrit la nvelle geom en sequentiel
519 int nbtot=0;
520 {
521// on regarde le nombre de lignes de nom_file
522 EFichier listb2(nom_file);
523 Nom toto;
524 listb2>>toto;
525 while(listb2.good())
526 {
527 listb2>>toto;
528 nbtot++;
529 }
530 }
531
532 EFichier listb2(nom_file);
533 Noms nomborddec(nbtot);
534 for (int nf=0; nf<nbtot; nf++) listb2>>nomborddec[nf];
535
536 Zone& zone=dom.zone(0);
537
538 int nbfr=zone.nb_front_Cl();
539 Bords& listbord=zone.faces_bord();
540 Raccords& listrac=zone.faces_raccord();
541 Bords listbord2;
542 Raccords listrac2;
543 int nbbord=zone.nb_bords();
544 IntVect decoup(nbfr);
545 Noms nomsdesbordsorg(nbfr);
546// on parcourt toutes les frontieres
547 for (int b=0; b<nbfr; b++)
548 {
549 Frontiere& org=zone.frontiere(b);
550 const Nom& nombord=org.le_nom();
551 nomsdesbordsorg[b]=nombord;
552 decoup(b)=0;
553 SFichier bord_xv(dom.le_nom()+"."+nomsdesbordsorg[b]+"_xv");
554 bord_xv.setf(ios::scientific);
555 bord_xv.precision(12);
556 for (int nf=0; nf<nbtot; nf++)
557 {
558 Nom nombord_dec=nomborddec[nf];
559 const char* marq=strchr(nombord_dec,'%');
560 if (marq)
561 {
562 nombord_dec.prefix(marq);
563// on regarde dans les bords decoupes si ils correspondent a la frontiere courante
564 if (nombord==nombord_dec)
565 {
566
567// exit();
568 decoup(b)=1;
569 Bord toto_bord;
570 Raccord toto_rac;
571 if (b>=nbbord) toto_rac.typer(org.le_type());
572 Frontiere& toto=(b<nbbord?ref_cast(Frontiere,toto_bord):ref_cast(Frontiere,toto_rac.valeur()));
573 Nom nombord_dec_bis=nomborddec[nf];
574 toto.nommer(nombord_dec_bis);
575 IntVect listfaces;
576// on recupere la liste des faces
577 EFichier lis(nombord_dec_bis);
578 lis>>listfaces;
579// on genere les faces associes a ce nouveau bord
580 Faces& Facesorg=org.faces();
581 Faces& newfaces=toto.faces();
582 newfaces.typer(Facesorg.type_face());
583 int newnbfaces=listfaces.size();
584 int nbsom=Facesorg.nb_som_faces();
585 newfaces.dimensionner(newnbfaces);
586 const IntTab& sommetsorg=Facesorg.les_sommets();
587 IntTab& newsommet=newfaces.les_sommets();
588 for (int i=0; i<newnbfaces; i++)
589 for (int j=0; j<nbsom; j++)
590 newsommet(i,j)=sommetsorg(listfaces[i],j);
591 DoubleTab position;
592
593 calculer_centres_gravite(position,newfaces.type_face(),dom.coord_sommets(),newsommet);
594
595 bord_xv<<nombord_dec_bis<<finl;
596 bord_xv<<position<<finl;
597 if (b<nbbord)
598 listbord2.add(toto_bord);
599 else
600 listrac2.add(toto_rac);
601 }
602 }
603 }
604 bord_xv.close();
605 }
606// on supprime les bords decoupes du domaine
607 for (int b2=0; b2<nbbord; b2++)
608 if (decoup(b2))
609 listbord.suppr(zone.bord(nomsdesbordsorg[b2]));
610 for (int r2=nbbord; r2<nbfr; r2++)
611 if (decoup(r2))
612 listrac.suppr(zone.raccord(nomsdesbordsorg[r2]));
613 listbord.add(listbord2);
614 listrac.add(listrac2);
615
616 int nb_faces_bord=dom.zone(0).nb_faces_bord();
617 if (nb_faces_bord!=nb_faces_bord_sa)
618 {
619 Cerr<<"the new number of boundary faces is false :"<<nb_faces_bord;
620 Cerr<<" instead of "<<nb_faces_bord_sa<<finl;
621 exit();
622 }
623 int nb_faces_raccord=dom.zone(0).nb_faces_raccord();
624 if (nb_faces_raccord!=nb_faces_raccord_sa)
625 {
626 Cerr<<"the new number of connection faces is false :"<<nb_faces_raccord;
627 Cerr<<" instead of "<<nb_faces_raccord_sa<<finl;
628 exit();
629 }
630
631
