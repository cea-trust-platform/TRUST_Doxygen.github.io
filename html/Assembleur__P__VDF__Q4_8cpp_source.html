33int Assembleur_P_VDF_Q4::assembler(Matrice& la_matrice)
34{
35 Cerr << "Assembleur_P_VDF_Q4::assembler " << finl;
36 Cerr << "Assemblage de la matrice Q4 en cours..." << finl;
37 const IntTab& FaceSoms = la_zone_VDF->face_sommets();
38 const IntTab& FaceVois = la_zone_VDF->face_voisins();
39 const IntVect& Orientation = la_zone_VDF->orientation();
40 int face, som0,som1;
41 int nbfaces = la_zone_VDF->nb_faces(), nbsom = la_zone_VDF->nb_som();
42 const DoubleVect& volumes_entrelaces = la_zone_VDF->volumes_entrelaces();
43 const DoubleVect& porosite_face = la_zone_VDF->porosite_face();
44 const DoubleTab& CoordSom = la_zone_VDF->zone().domaine().coord_sommets();
45 double surf,vol,por,val, r;
46 int ori, elem0,elem1;
47// construction matrice
48 la_matrice.typer("Matrice_Morse");
49 Matrice_Morse& mat = ref_cast(Matrice_Morse,la_matrice.valeur());
50 mat.dimensionner(nbsom,0);
51 IntVect rang_voisins(nbsom);
52 IntVect& tab1 = mat.get_set_tab1();
53 IntVect& tab2 = mat.get_set_tab2();
54 DoubleVect& coeff = mat.get_set_coeff();
55
56 tab1 = 0;
57 tab2 = 0;
58 coeff = 0;
59 rang_voisins = 1;
60 for (face=0 ; face<nbfaces ; face++)
61 {
62 som0 = FaceSoms(face,0);
63 som1 = FaceSoms(face,1);
64 rang_voisins(som0)++;
65 rang_voisins(som1)++;
66 }
67 tab1(0) = 1;
68 for (som0=0 ; som0<nbsom ; som0++)
69 {
70 tab1(som0+1) = rang_voisins(som0) + tab1(som0);
71 }
72 rang_voisins = 1;
73 mat.dimensionner(nbsom,tab1(nbsom)-1);
74// calcul coefficient matrice
75 for (face=0 ; face<nbfaces ; face++)
76 {
77 ori = Orientation(face);
78 som0 = FaceSoms(face,0);
79 som1 = FaceSoms(face,1);
80 elem0 = FaceVois(face,0);
81 elem1 = FaceVois(face,1);
82 if (elem0>-1 && elem1>-1)
83 {
84 surf = 0.5* (la_zone_VDF->dim_elem(elem0,ori) + la_zone_VDF->dim_elem(elem1,ori));
85 }
86 else if (elem0>-1)
87 {
88 surf = la_zone_VDF->dim_elem(elem0,ori);
89 }
90 else
91 {
92 surf = 0.;
93 surf = la_zone_VDF->dim_elem(elem1,ori);
94 }
95 if (bidim_axi/* && ori==1*/)
96 {
97 r = 0.5*(CoordSom(som0,0)+CoordSom(som1,0));
98 surf = r * surf * 2 * M_PI;
99 }
100 vol = volumes_entrelaces(face);
101 por = porosite_face(face);
102 val = surf*surf /vol * por;
103#ifdef _AFFDEBUG
104 if (la_zone_VDF->nb_som()<400)
105 {
106 Cerr<<"face="<<face<<" ori="<<ori<<" soms="<<som0<<"/"<<som1<<" coordsom="<<CoordSom(som0,0)<<"/"<<CoordSom(som0,1);
107 Cerr<<" "<<CoordSom(som1,0)<<"/"<<CoordSom(som1,1)<<" elems="<<FaceVois(face,0)<<"/"<<FaceVois(face,1)<<" dim_elems/2=";
108 if (FaceVois(face,0)>-1)
109 Cerr<<la_zone_VDF->dim_elem(FaceVois(face,0),ori)/2.<<" / ";
110 else
111 Cerr<<"--- / ";
112 if (FaceVois(face,1)>-1)
113 Cerr<<la_zone_VDF->dim_elem(FaceVois(face,1),ori)/2.<<" ";
114 else
115 Cerr<<"--- ";
116 Cerr<<" surf="<<surf<<" vol="<<vol<<" val="<<val<<finl;
117 }
118#endif
119
120 tab2(tab1(som0)-1) = som0+1;
121 tab2(tab1(som1)-1) = som1+1;
122 mat(som0,som0) += val;
123 mat(som1,som1) += val;
124// if (coeff(tab1(som0)+rang_voisins(som0)-1)!=-1) {
125 tab2(tab1(som0)+rang_voisins(som0)-1) = som1+1;
126 mat(som0,som1) -= val;
127 rang_voisins(som0)++;
128 tab2(tab1(som1)+rang_voisins(som1)-1) = som0+1;
129 mat(som1,som0) -= val;
130 rang_voisins(som1)++;
131// }
132 }
133 Cerr << "Fin de l'assemblage de la matrice Q4" << finl;
134 return 1;
135}

142int Assembleur_P_VDF_Q4::modifier_secmem(const DoubleTab& tab_secmem_, DoubleVect& secmem)
143{
144 const DoubleTab& CoordSommets = la_zone_VDF->zone().domaine().coord_sommets();
145 const IntTab& FaceSoms = la_zone_VDF->face_sommets();
146 const IntTab& FaceVois = la_zone_VDF->face_voisins();
147 const IntVect& Orientation = la_zone_VDF->orientation();
148 const DoubleTab& CoordSom = la_zone_VDF->zone().domaine().coord_sommets();
149 int face, som0,som1, ori, invori, elem0,elem1;
150 int nbfaces = la_zone_VDF->nb_faces(), nbsom = la_zone_VDF->nb_som();
151 double surf, r;
152
153 secmem.resize(nbsom);
154 secmem = 0.;
155
156 for (face=0 ; face<nbfaces ; face++)
157 {
158 surf = 0.;
159 ori = Orientation(face);
160 som0 = FaceSoms(face,0);
161 som1 = FaceSoms(face,1);
162 elem0 = FaceVois(face,0);
163 elem1 = FaceVois(face,1);
164 if (elem0>-1 && elem1>-1)
165 {
166 surf = 0.5* (la_zone_VDF->dim_elem(elem0,ori) + la_zone_VDF->dim_elem(elem1,ori));
167 }
168 else if (elem0>-1)
169 {
170 surf = la_zone_VDF->dim_elem(elem0,ori);
171 }
172 else if (elem1>-1)
173 {
174 surf = la_zone_VDF->dim_elem(elem1,ori);
175 }
176 else
177 {
178 assert(0==1);
179 }
180 if (bidim_axi/* && ori==1*/)
181 {
182 r = 0.5*(CoordSom(som0,0)+CoordSom(som1,0));
183 surf = r * surf * 2 * M_PI;
184 }
185 if (ori==0)
186 invori = 1;
187 else
188 invori = 0;
189 if (CoordSommets(som1,invori)>CoordSommets(som0,invori))
190 {
191 secmem(som0) += tab_secmem_(face) * surf;
192 secmem(som1) -= tab_secmem_(face) * surf;
193 }
194 else
195 {
196 secmem(som0) -= tab_secmem_(face) * surf;
197 secmem(som1) += tab_secmem_(face) * surf;
198 }
199 secmem.echange_espace_virtuel();
200 }
201
202 return 1;
203}
