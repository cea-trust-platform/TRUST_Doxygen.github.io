127//! Effectue un pas de temps Leap_frog, sur l'equation<br>passee en parametre.
128/** \return int<br>
129Signification: renvoie toujours 1<br>
130Contraintes: */
131int Leap_frog::faire_un_pas_de_temps_eqn_base(Equation_base& eq)
132{
133// EXEMPLE POUR UN SCHEMA EXPLICITE UTILISANT U(n) et U(n+1)
134// Un+1=Un+dt*F
135// Un
136 DoubleTab& present = eq.inconnue().valeurs();
137
138// Un+1
139 DoubleTab& futur = eq.inconnue().futur();
140
141// Un+1=F
142 eq.derivee_en_temps_inco(futur);
143 futur.echange_espace_virtuel();
144
145// ECRIRE ICI LE CALCUL de U(n+1) EN FONCTION DE U(n) ET DE derivee
146// futur*=dt; Un+1=dt*F
147// futur+=present; Un+1=Un+dt*F
148// SI IL S'AGISSAIT D'UN SCHEMA QUI FAIT INTERVENIR U(n-1)
149// if ((nb_pas_dt_>4) && ((nb_pas_dt_ % 10) != 0)) {
150// if (nb_pas_dt_>4) {
151 if ((nb_pas_dt_>4) && ((nb_pas_dt_ % 10) != 0))
152 {
153 DoubleTab& passe = eq.inconnue().passe();
154 double gamma=0.2;
155 DoubleTab& passe2 = eq.inconnue().passe(2);
156 DoubleTab& correcteur=passe;
157 correcteur*=(1.-2.*gamma);
158 correcteur.ajoute(gamma, present, VECT_REAL_ITEMS);
159 correcteur.ajoute(gamma, passe2, VECT_REAL_ITEMS);
160 futur*=(2*dt_);
161 futur+=correcteur;
162 }
163 else
164 {
165 futur*=dt_;
166 futur+=present;
167 }
168
169 eq.zone_Cl_dis().imposer_cond_lim(eq.inconnue(),temps_courant()+pas_de_temps());
170 DoubleTab tmp(futur);
171 tmp -= present;
172 tmp /= dt_;
173 update_critere_statio(tmp, eq);
174 futur.echange_espace_virtuel();
175
176 return 1;
177}
