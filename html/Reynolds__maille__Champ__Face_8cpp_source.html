46// Methode de calcul de la valeur sur une face d'un champ uniforme ou non a plusieurs composantes
47inline double valeur(const DoubleTab& champ, const int face, const int compo, const Zone_VDF& la_zone_VDF)
48{
49 if (champ.dimension(0)==1)
50 return champ(0,compo); //!< Champ uniforme
51 else
52 {
53 int elem0 = la_zone_VDF.face_voisins(face,0);
54 int elem1 = la_zone_VDF.face_voisins(face,1);
55 if (elem0<0) elem0 = elem1; //!< face frontiere
56 if (elem1<0) elem1 = elem0; //!< face frontiere
57 return 0.5*(champ(elem0,compo)+champ(elem1,compo));
58 }
59}

61void Reynolds_maille_Champ_Face::mettre_a_jour(double tps)
62{
63 const int nb_faces = zone_vdf().nb_faces();
64 DoubleTab& re = valeurs(); //!< Reynolds de maille Boucle sur les faces
65//
66 for (int face=0; face<nb_faces; face++)
67 {
68// Calcul de la viscosite face
69 double nu_face = valeur(nu_->valeurs(), face, 0, zone_vdf());
70// Calcul de la taille de maille entourant la face
71 double taille_maille = zone_vdf().volumes_entrelaces()(face) / zone_vdf().face_surfaces(face);
72// Calcul du Reynolds de maille
73 re(face) = std::fabs(vitesse_->valeurs()(face)) * taille_maille / nu_face; //!< Reynolds_maille = |Uface| * taille_maille / nu_face
74 }
75
76 changer_temps(tps);
77 Champ_Fonc_base::mettre_a_jour(tps);
78}
