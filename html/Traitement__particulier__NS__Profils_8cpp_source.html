64Entree& Traitement_particulier_NS_Profils::lire(Entree& is)
65{
66// Initialisation
67 oui_u_inst = 0;
68 oui_stat = 0;
69 oui_repr = 0;
70 oui_repr_nu_t = 0;
71 oui_profil_nu_t = 0;
72 IntVect Verif(8);
73 n_probes = 1;
74 Verif=0;
75
76// END Initialisation
77 Motcle accouverte = "{" , accfermee = "}" ;
78 Motcle motbidon, motlu;
79 is >> motbidon ;
80 if (motbidon == accouverte)
81 {
82 Motcles les_mots(8);
83 les_mots[0] = "const_profile";
84 les_mots[1] = "num_probes";
85 les_mots[2] = "dt_output";
86 les_mots[3] = "stats";
87 les_mots[4] = "profil_nu_t";
88 les_mots[5] = "reprise";
89 les_mots[6] = "reprise_nu_t";
90 les_mots[7] = "homog_dir";
91 is >> motlu;
92 while(motlu != accfermee)
93 {
94 int rang=les_mots.search(motlu);
95 switch(rang)
96 {
97 case 0 :
98 {
99// For const_profile Constant Profile value : X, Y, Z
100 is >> motlu; //!< Direction of constant profile
101 if((motlu=="x")||(motlu=="X")) dir_profil = 0;
102 else if((motlu=="y")||(motlu=="Y")) dir_profil = 1;
103 else if((motlu=="z")||(motlu=="Z")) dir_profil = 2;
104 else
105 {
106 Cerr << "Wrong value for direction of profile !" << finl;
107 exit();
108 }
109 Cerr << "Profiles at constant " << motlu << finl;
110 break;
111 }
112 case 1 :
113 {
114// Number of probes and reading of the values of the postions
115 is >> n_probes;
116 if(n_probes<1)
117 {
118 Cerr << "Error in Traitement_particulier_NS_Profils::lire" << finl;
119 Cerr << "Wrong value for number of probes (" << n_probes << ") :" << finl;
120 Cerr << "There must be at least one profile !" << finl;
121 exit();
122 }
123 positions.resize(n_probes);
124 for(int i=0; i<n_probes; i++)
125 {
126 is >> positions(i);
127 Cerr << " Probe_" << i+1 << " = " << positions(i) << finl;
128 }
129 break;
130 }
131 case 2 :
132 {
133// Pour u_inst
134 is >> dt_post_inst; //!< intervalle de temps de sortie des moyennes spatiales initialisation pour le calcul des stats spat.
135//
136 oui_u_inst = 1;
137 Cerr << "Spatial average parameters read..." << finl;
138 Cerr << "dt_post_inst : " << dt_post_inst << finl;
139 break;
140 }
141 case 3 :
142 {
143// For Time average
144 is >> temps_deb; //!< Initial time for time average statistics.
145 is >> temps_fin; //!< Final time for time average statistics.
146 is >> dt_post_stat; //!< Output intervalfor time average statistics.
147 oui_stat=1; //!< =1 : on calcule des stats temp.
148 Cerr << "Reading time statitics parameters..." << finl;
149 Cerr << "Initial time : " << temps_deb << " End time : " << temps_fin << finl;
150 Cerr << "Time interval for output : " << dt_post_stat << finl;
151// Verif ensuite pour voir si on a rentrer les valeurs specifiques aux calculs des stats spat.
152 break;
153 }
154 case 4 :
155 {
156// For eddy viscosity profile
157 oui_profil_nu_t = 1;
158 Cerr << "User asked for statistics on eddy-viscosity..." << finl;
159 const RefObjU& modele_turbulence = mon_equation.valeur().get_modele(TURBULENCE);
160 const Mod_turb_hyd_base& mod_turb = ref_cast(Mod_turb_hyd_base,modele_turbulence.valeur());
161 if(sub_type(Mod_turb_hyd_base,mod_turb))
162 Cerr << "Statistics for eddy-viscosity : OK !" << finl;
163 else
164 {
165 Cerr << "User asked for eddy-viscosity profile," << finl;
166 Cerr << "but is not defined in laminar case !" << finl;
167 Cerr << "Please check the problem type..." << finl;
168 exit();
169 }
170 break;
171 }
172 case 5 :
173 {
174 oui_repr = 1;
175 is >> fich_repr_u ; //!< indication du nom du fichier de reprise des stats
176 Cerr << "The time statistics file is : " << fich_repr_u << finl;
177 break;
178 }
179 case 6 :
180 {
181// For eddy viscosity profile Reprise
182 oui_repr_nu_t = 1;
183 is >> fich_repr_nu_t ; //!< indication du nom du fichier de reprise des stats de nut
184 Cerr << "Continuing statistics on eddy-viscosity : OK!" << finl;
185 break;
186 }
187 case 7 :
188 {
189// For homogeneous direction specification : X, Y, Z
190 is >> motlu; //!< Direction of homogeneity
191 if((motlu=="x")||(motlu=="X")) homo_dir = 0;
192 else if((motlu=="y")||(motlu=="Y")) homo_dir = 1;
193 else if((motlu=="z")||(motlu=="Z")) homo_dir = 2;
194 else
195 {
196 Cerr << "Wrong value for homogeneous direction !" << finl;
197 exit();
198 }
199 Cerr << "Homogeneous direction : " << motlu << finl;
200 break;
201 }
202 default :
203 {
204 lire(motlu,is);
205 break;
206 }
207 }
208 is >> motlu;
209 }
210 is >> motlu;
211 if (motlu != accfermee)
212 {
213 Cerr << "Error while reading in Traitement_particulier_NS_Profils::lire" << finl;
214 Cerr << "We were expecting a }" << finl;
215 exit();
216 }
217 }
218 else
219 {
220 Cerr << "Error while reading in Traitement_particulier_NS_Profils" << finl;
221 Cerr << "We were expecting a {" << finl;
222 exit();
223 }
224
225 if ((oui_u_inst!=1)&&(oui_stat!=0))
226 {
227 Cerr << "User asked for time statistics :" << finl;
228 Cerr << "To do so, you also need to specify the parameters for space statistics," << finl;
229 Cerr << "since it is used for the time stats.." << finl;
230 exit();
231 }
232 if ((oui_profil_nu_t!=1)&&(oui_repr_nu_t!=0))
233 {
234 Cerr << "User asked for time statistics on nu_t :" << finl;
235 Cerr << "pour cela il faut egalement demander le calcul des stats spatiales," << finl;
236 Cerr << "puisque il est necessaire au calcul des stats temp." << finl;
237 exit();
238 }
239 return is;
240}

295void Traitement_particulier_NS_Profils::reprendre_stat()
296{
297 double tps = mon_equation->inconnue().temps();
298 int j,num_prof;
299 double ti,ti2;
300 Nom temps;
301 if (oui_stat == 1)
302 {
303 ifstream ficu0(fich_repr_u);
304 if (ficu0)
305 {
306 LecFicDistribueBin ficu (fich_repr_u);
307 ficu >> temps;
308 u_moy_temp_x.resize(n_probes,Nap);
309 u_moy_temp_y.resize(n_probes,Nap);
310 u_moy_temp_z.resize(n_probes,Nap);
311
312 uv_moy_temp.resize(n_probes,Nap);
313 u2_moy_temp.resize(n_probes,Nap);
314 v2_moy_temp.resize(n_probes,Nap);
315 w2_moy_temp.resize(n_probes,Nap);
316
317 u3_moy_temp.resize(n_probes,Nap);
318 v3_moy_temp.resize(n_probes,Nap);
319 w3_moy_temp.resize(n_probes,Nap);
320
321 u4_moy_temp.resize(n_probes,Nap);
322 v4_moy_temp.resize(n_probes,Nap);
323 w4_moy_temp.resize(n_probes,Nap);
324
325 for(num_prof=0; num_prof<n_probes; num_prof++)
326 {
327 ficu >> Nxy(num_prof) >> Nyy(num_prof) >> Nzy(num_prof) >> Nuv(num_prof);
328 for (j=0; j<Nxy(num_prof); j++)
329 {
330 ficu >> u_moy_temp_x(num_prof,j) ;
331 ficu >> u2_moy_temp(num_prof,j);
332 }
333 for (j=0; j<Nyy(num_prof); j++)
334 {
335 ficu >> u_moy_temp_y(num_prof,j) ;
336 ficu >> v2_moy_temp(num_prof,j);
337 }
338 for (j=0; j<Nzy(num_prof); j++)
339 {
340 ficu >> u_moy_temp_z(num_prof,j);
341 ficu >> w2_moy_temp(num_prof,j);
342 }
343 for (j=0; j<Nuv(num_prof); j++)
344 ficu >> uv_moy_temp(num_prof,j) ;
345 }
346// END reading "n_probes" profiles
347 ficu >> ti ;
348 Cerr << "ti=" << ti << finl;
349 Nom chc_ti = Nom(ti), chc_temps_deb = Nom(temps_deb);
350 Cerr << "chc_ti=" << chc_ti << " chc_temps_deb=" << chc_temps_deb << finl;
351 if (chc_ti!=chc_temps_deb)
352 {
353 if (ti > temps_deb)
354 {
355 Cerr << "pb de reprise des stats de la vitesse!!" << finl;
356 Cerr << "Le temps de debut des stats demande " << temps_deb << finl;
357 Cerr << "est inferieur a celui de debut des stats sauvees !" << ti << finl;
358 exit();
359 }
360 else
361 {
362 if (temps_deb>=tps)
363 {
364 Cerr << "On recommence le calcul des stats de la vitesse a partir de t=" << temps_deb << "s" << finl;
365 u_moy_temp_x = 0.;
366 u_moy_temp_y = 0.;
367 u_moy_temp_z = 0.;
368 uv_moy_temp = 0.;
369 u2_moy_temp = 0.;
370 v2_moy_temp = 0.;
371 w2_moy_temp = 0.;
372 oui_repr = 0;
373 }
374 else

376 Cerr << "On a deja depasse le temps auquel vous voulez commencer les stats!!" << finl;
377 exit();
378 }
379 }

397 exit();

405 {

407 ficu >> temps;
408 nu_t_temp.resize(n_probes,Nap);
409 for(num_prof=0; num_prof<n_probes; num_prof++)
410 {
411 ficu >> Nuv(num_prof);
412 for (j=0; j<Nuv(num_prof); j++)
413 ficu >> nu_t_temp(num_prof,j) ;
414 }
415 ficu >> ti2 ;
416 if (ti2!=ti)
417 {
