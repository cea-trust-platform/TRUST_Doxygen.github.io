51// On calcule la valeur moyenne entre borne_inf et borne_sup du polynome de lagrange de degre 2 :
52// - qui vaut 0 en a et en b
53// - qui vaut 1 en c
54static double integrate_lagrangian_basis(double a, double b, double c, double borne_inf, double borne_sup)
55{
56 double result = 0.;
57 double scaling = (c-a)*(c-b);
58 double dt = borne_sup-borne_inf;
59
60 double prod = a*b;
61 double sum = -0.5*(a+b);
62
63 double tmp_sup = borne_sup*(prod+borne_sup*(borne_sup/3.+sum));
64 double tmp_inf = borne_inf*(prod+borne_inf*(borne_inf/3.+sum));
65
66 result = tmp_sup-tmp_inf;
67 result/=(scaling*dt);
68
69 return result;
70}

92void Schema_Adams_Moulton_order_3::compute_adams_moulton_coefficients(double time_step, const DoubleTab& times) const
93{
94 if (nb_pas_dt()<100 && limpr() && (facteur_securite_pas()>1.0 || facsec_max_>1.0))
95 {
96 Cerr << finl;
97 Cerr << "********************* Warning (printed only on the first 100 time steps) ************************"<< finl;
98 Cerr << "The Adams Moulton order 3 scheme is recommended with a facsec and facsec_max set to 1.0 or lower."<< finl;
99 Cerr << "*************************************************************************************************"<< finl;
100// See $TRUST_ROOT/tests/Reference/Schema_Adams_Moulton_order_3 test case.
101 }
102 assert(times.size_array() == 3); //past, present and future times, stored in ascending order inside "times" table
103
104 if ( adams_moulton_coefficients_.size_array() != 3)
105 {
106 adams_moulton_coefficients_.resize(3,Array_base::NOCOPY_NOINIT);
107 }
108
109 double past_time = times[0];
110 double present_time = times[1];
111 double future_time = times[2];
112
113 adams_moulton_coefficients_[0] = integrate_lagrangian_basis(present_time,future_time,past_time,present_time,future_time);
114 adams_moulton_coefficients_[1] = integrate_lagrangian_basis(past_time,future_time,present_time,present_time,future_time);
115 adams_moulton_coefficients_[2] = 1 - adams_moulton_coefficients_[1] - adams_moulton_coefficients_[0];
116}
