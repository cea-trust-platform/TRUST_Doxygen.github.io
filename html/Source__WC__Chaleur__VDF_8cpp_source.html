38void Source_WC_Chaleur_VDF::compute_interpolate_gradP(DoubleTab& UgradP_elem, const DoubleTab& Ptot) const
39{
40 /*
41 * NOTA BENE :
42 * On a discretis√© P_EOS avec une directive = temperature qui donne en VDF :
43 * Ptot => aux elems
44 * la_vitesse => faces car vdf
45 * grad_Ptot => faces car vdf
46 *
47 * Donc u*grad(P_tot) => faces
48 * L'equation est une equation de scalaire => aux elems car vdf
49 * On a d P_tot / d t = del P / del t + u.grad(P_tot)
50 *
51 * del P / del t => aux elem
52 * u.grad(P_tot) !!! faut l'interpoler aux elems
53 */
54
55 const Navier_Stokes_WC& eqHyd = ref_cast(Navier_Stokes_WC,mon_equation->probleme().equation(0));
56 const DoubleTab& la_vitesse = eqHyd.vitesse().valeurs();
57 DoubleTab grad_Ptot(eqHyd.grad_P().valeurs()); //!< initialise avec grad(P) car face
58 const Convection_Diffusion_Chaleur_WC& eq_chal = ref_cast(Convection_Diffusion_Chaleur_WC,mon_equation.valeur());
59 const Operateur_Grad& Op_Grad = eq_chal.operateur_gradient_WC();
60 Op_Grad.calculer(Ptot,grad_Ptot); //!< compute grad(P_tot)
61
62 const Zone_dis_base& zone_dis = mon_equation->inconnue().zone_dis_base();
63 const Zone_VF& zone = ref_cast(Zone_VF, zone_dis);
64 assert (zone_dis.que_suis_je() == "Zone_VDF");
65
66// grad should be zero at boundary
67 correct_grad_boundary(zone,grad_Ptot);
68
69// We compute u*grad(P_tot) on each face
70 DoubleTab UgradP(grad_Ptot); //!< field on faces
71 const int n = la_vitesse.dimension_tot(0);
72 assert ( n == zone.nb_faces_tot() && la_vitesse.line_size() == 1);
73 assert ( Ptot.dimension_tot(0) == zone.nb_elem_tot() );
74
75 for (int i=0 ; i <n ; i++) UgradP(i,0) = la_vitesse(i,0) * grad_Ptot(i,0);
76 face_to_elem(zone,UgradP,UgradP_elem);
77}

80void Source_WC_Chaleur_VDF::face_to_elem(const Zone_VF& zone, const DoubleTab& UgradP,DoubleTab& UgradP_elem) const
81{
82 const IntTab& elem_faces = zone.elem_faces();
83 const int nb_face_elem = elem_faces.line_size(), nb_elem_tot= zone.nb_elem_tot();
84 assert (UgradP_elem.dimension_tot(0) == nb_elem_tot && UgradP.dimension_tot(0) == zone.nb_faces_tot());
85
86 UgradP_elem = 0.;
87 for (int ele=0; ele<nb_elem_tot; ele++)
88 for (int s=0; s<nb_face_elem; s++) UgradP_elem(ele,0) += UgradP(elem_faces(ele,s),0);
89
90 UgradP_elem *= 0.5;
91}

93// marche bien mais pas bon pour le vef ... P en vef est partout, grad sur les faces mais pas bon pour nous
94void Source_WC_Chaleur_VDF::compute_interpolate_gradP_old(DoubleTab& UgradP_elem, const DoubleTab& Ptot) const
95{
96// compute the grad
97 const Navier_Stokes_WC& eqHyd = ref_cast(Navier_Stokes_WC,mon_equation->probleme().equation(0));
98 const DoubleTab& la_vitesse = eqHyd.vitesse().valeurs();
99 DoubleTab grad_Ptot(eqHyd.grad_P().valeurs()); //!< initialise avec grad(P) car face
100 const Operateur_Grad& gradient = eqHyd.operateur_gradient(); //!< recuperer op grad de NS
101 gradient.calculer(Ptot,grad_Ptot); //!< compute grad(P_tot)
102
103// XXX : very important : sinon we have values * V !
104 const Solveur_Masse& solv_mass = eqHyd.solv_masse();
105 solv_mass.appliquer(grad_Ptot);
106
107 /*
108 * XXX XXX XXX
109 * READ CAREFULLY : We use the grad operator from Navier_Stokes_std
110 * => created for the pressure normally
111 * There is a special treatement for Neumann bc (Neumann_sortie_libre)
112 * because we use explicitly the prescribed pressure.
113 * So if we calculate the grad of Ptot or rho for example, we will use
114 * P_ext defined in data file which is wrong !
115 *
116 * Now we do the following : we assume open bd as a wall => null gradient
117 */
118
119// We use that of NS because we test the CL too (attention mon_equation is Chaleur...)
120 const Zone_dis_base& zone_dis = eqHyd.inconnue().zone_dis_base();
121 const Zone_VF& zone = ref_cast(Zone_VF, zone_dis);
122 const Zone_Cl_dis& zone_cl = eqHyd.zone_Cl_dis();
123 assert (zone_dis.que_suis_je() == "Zone_VDF");
124
125 for (int n_bord=0; n_bord<zone.nb_front_Cl(); n_bord++)
126 {
127 const Cond_lim& la_cl = zone_cl.les_conditions_limites(n_bord);
128// corrige si Neumann_sortie_libre
129 if ( sub_type(Neumann_sortie_libre,la_cl.valeur()) )
130 {
131// recuperer face et remplace gradient par 0
132 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
133 const int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
134
135 for (int num_face=ndeb; num_face<nfin; num_face++) grad_Ptot(num_face,0) = 0.;
136 }
137 }
138
139// We compute u*grad(P_tot) on each face
140 DoubleTab UgradP(grad_Ptot); //!< field on faces
141 const int n = la_vitesse.dimension(0);
142 assert ( n == zone.nb_faces() );
143 assert ( Ptot.dimension(0) == zone.nb_elem() );
144 for (int i=0 ; i <n ; i++) UgradP(i,0) = la_vitesse(i,0) * grad_Ptot(i,0);
145 face_to_elem(zone,UgradP,UgradP_elem);
146}
