38// In the file fich, go up to the identifier id and return the time found
39double avancer_fichier(Entree& fich, const Nom& id, const int read_header)
40{
41 /* Surcharge par:
42 double LecFicDiffuse::seek(const Nom& id)
43 {
44 double time;
45 diffuse_=0;
46 if (Process::je_suis_maitre()) time = Entree::seek(id);
47 diffuse_=1;
48 envoyer_broadcast(temps, 0);
49 return time;
50 }
51// Possible evolution:
52// avancer_fichier(fich,id) -> fich.avancer_fichier(id);
53// Implementation of double Entree::seek(const Nom& id):
54// diffuse_ redescend dans LecFicDiffuseBase
55 */
56 double temps=0;
57 fich.set_diffuse(0); //!< Entree::set_diffuse() set to 1, only Lec_Diffuse_base::set_diffuse() can set to 0
58 if (fich.get_diffuse() || Process::je_suis_maitre())
59 {
60 Cerr << "Looking for the tag " << id << " in the file." << finl;
61 Nom ident_lu;
62 Nom type_lu;
63 int digits=12; //!< Number of digits to write the time
64 fich >> ident_lu;
65
66 if (read_header)
67 {
68 if (ident_lu!="format_sauvegarde:")
69 {
70 digits=8; //!< Old time format
71 }
72 else
73 {
74// New format
75 int format;
76 fich >> format;
77 fich >> ident_lu;
78 }
79 }
80
81 fich >> type_lu;
82 while (Motcle(ident_lu) != Motcle(id))
83 {
84 DerObjU un_objet;
85 un_objet.typer(type_lu);
86 Cerr << "-> Passing object " << ident_lu << " ... " << flush;
87 un_objet.reprendre(fich);
88 Cerr << "OK" << finl;
89 temps = extract_time_from_string(ident_lu, digits);
90 if (!fich.get_istream()) erreur(id);
91 fich >> ident_lu;
92 if (id!=(Nom)"fin" && ident_lu == (Nom)"fin") erreur(id);
93 fich >> type_lu;
94 }
95 Cerr << "-> Find object " << id << " and reading data in it ..." << finl;
96 }
97 envoyer_broadcast(temps, 0);
98 fich.set_diffuse(1);
99 return temps;
100}
