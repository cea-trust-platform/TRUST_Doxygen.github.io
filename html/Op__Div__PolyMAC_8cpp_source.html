63void Op_Div_PolyMAC::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
64{
65 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
66 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur());
67 const DoubleTab& inco = ch.valeurs(), &press = ref_cast(Navier_Stokes_std, equation()).pression().valeurs();
68 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins(), &fcl = ch.fcl();
69 int i, j, e, f, ne_tot = zone.nb_elem_tot();
70
71 Matrice_Morse *matv = matrices.count("vitesse") ? matrices["vitesse"] : NULL, *matp = matrices.count("pression") ? matrices["pression"] : NULL, matv2, matp2;
72 IntTab sten_v(0,2), sten_p(0, 2);
73 DoubleTab w2; //matrice w2 aux elements (la meme que dans Op_Grad et Assembleur_P)
74 sten_v.set_smart_resize(1), sten_p.set_smart_resize(1), w2.set_smart_resize(1);
75
76 for (f = 0; matv && f < zone.nb_faces(); f++) /* dependance en v : divergence par elem + v = v_imp aux faces de Dirichlet */
77 if (fcl(f, 0) > 1) sten_v.append_line(ne_tot + f, f); /* v impose par CLs : contribution a l'equation v = v_imp a la paroi */
78 else for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
79 if (e < zone.nb_elem()) sten_v.append_line(e, f); /* v calcule : contribution a la divergence aux elems */
80
81 for (e = 0; matp && e < zone.nb_elem_tot(); e++)
82 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(0); i++) /* dependance en p : equation sur p_f */
83 if (fcl(f = e_f(e, i), 0) == 1) sten_p.append_line(ne_tot + f, ne_tot + f); /* aux faces de Neumann : p_f = p_imp */
84 else if (!fcl(f, 0))
85 for (sten_p.append_line(ne_tot + f, e), j = 0; j < w2.dimension(1); j++) /* aux faces internes : egalite des deux gradients */
86 if (w2(i, j, 0)) sten_p.append_line(ne_tot + f, ne_tot + e_f(e, j));
87 if (matp)
88 for (e = 0; e < zone.nb_elem(); e++) sten_p.append_line(e, e); //diagonale du vide!
89
90 if (matv) tableau_trier_retirer_doublons(sten_v), Matrix_tools::allocate_morse_matrix(press.size_totale(), inco.size_totale(), sten_v, matv2);
91 if (matp) tableau_trier_retirer_doublons(sten_p), Matrix_tools::allocate_morse_matrix(press.size_totale(), press.size_totale(), sten_p, matp2);
92 if (matv) matv->nb_colonnes() ? *matv += matv2 : *matv = matv2;
93 if (matp) matp->nb_colonnes() ? *matp += matp2 : *matp = matp2;
94}

96void Op_Div_PolyMAC::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
97{
98 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
99 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur());
100 const Conds_lim& cls = la_zcl_PolyMAC->les_conditions_limites();
101 const DoubleTab& inco = ch.valeurs(), &press = ref_cast(Navier_Stokes_std, equation()).pression().valeurs(), &nf = zone.face_normales();
102 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins(), &fcl = ch.fcl();
103 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face();
104 int i, j, e, f, fb, ne_tot = zone.nb_elem_tot(), d, D = dimension;
105 Matrice_Morse *matv = matrices.count("vitesse") ? matrices["vitesse"] : NULL, *matp = matrices.count("pression") ? matrices["pression"] : NULL, matv2, matp2;
106
107 DoubleTrav w2; //matrice w2 aux elements (la meme que dans Op_Grad et Assembleur_P)
108 w2.set_smart_resize(1);
109
110 for (f = 0; f < zone.nb_faces(); f++) /* divergence aux elements + equations aux bords */
111 {
112 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
113 if (e < zone.nb_elem()) /* divergence aux elems */
114 {
115 secmem(e) -= (i ? 1 : -1) * fs(f) * pf(f) * inco(f);
116 if (fcl(f, 0) < 2 && matv) (*matv)(e, f) += (i ? 1 : -1) * fs(f) * pf(f);
117 }
118 /* equations v = v_imp ou p = p_imp aux faces de bord */
119 if (fcl(f, 0) > 1)
120 {
121 if (fcl(f, 0) == 3)
122 for (d = 0; d < D; d++) secmem(ne_tot + f) += nf(f, d) * pf(f) * ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), d);
123 secmem(ne_tot + f) -= fs(f) * pf(f) * inco(f);
124 if (matv) (*matv)(ne_tot + f, f) += fs(f) * pf(f);
125 }
126 else if (fcl(f, 0) == 1)
127 {
128 secmem(ne_tot + f) += ref_cast(Neumann, cls[fcl(f, 1)].valeur()).flux_impose(fcl(f, 2), 0) - press(ne_tot + f);
129 if (matp) (*matp)(ne_tot + f, ne_tot + f) += 1;
130 }
131 }
132
133 /* equations aux faces internes : egalite des gradients */
134 for (e = 0; e < zone.nb_elem_tot(); e++)
135 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(0); i++)
136 if ((f = e_f(e, i)) < zone.nb_faces() && !fcl(f, 0))
137 {
138 double coeff_e = 0;
139 for (j = 0; j < w2.dimension(1); j++)
140 if (w2(i, j, 0))
141 {
142 fb = e_f(e, j);
143 secmem(ne_tot + f) -= pf(f) * w2(i, j, 0) * (press(ne_tot + fb) - press(e));
144 if (matp) (*matp)(ne_tot + f, ne_tot + fb) += pf(f) * w2(i, j, 0), coeff_e += pf(f) * w2(i, j, 0);
145 }
146 if (matp) (*matp)(ne_tot + f, e) -= coeff_e;
147 }
148}

150DoubleTab& Op_Div_PolyMAC::ajouter(const DoubleTab& vit, DoubleTab& div) const
151{
152 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
153 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face();
154 const DoubleTab& nf = zone.face_normales();
155 const Conds_lim& cls = la_zcl_PolyMAC->les_conditions_limites();
156 const IntTab& f_e = zone.face_voisins(), &fcl = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur()).fcl();
157 int i, e, f, ne_tot = zone.nb_elem_tot(), d, D = dimension, has_f = div.dimension_tot(0) > ne_tot;
158
159 DoubleTab& tab_flux_bords = flux_bords_;
160 tab_flux_bords.resize(zone.nb_faces_bord(),1);
161 tab_flux_bords=0;
162
163 for (f = 0; f < zone.nb_faces(); f++)
164 {
165 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
166 if (e < zone.nb_elem()) /* aux elements */
167 div(e) += (i ? -1 : 1) * fs(f) * pf(f) * vit(f);
168 if (f >= zone.premiere_face_int()) continue;
169 /* si "div" a des valeurs aux faces : bilan de masse avec la CL imposee aux faces de bord de Dirichlet */
170 if (has_f && fcl(f, 0) != 1) div(ne_tot + f) -= fs(f) * pf(f) * vit(f);
171 if (has_f && fcl(f, 0) == 3)
172 for (d = 0; d < D; d++) div(ne_tot + f) += nf(f, d) * pf(f) * ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), d);
173 /* a toutes les faces de bord : contribution a tab_flux_bords */
174 tab_flux_bords(f) = fs(f) * pf(f) * vit(f); //flux aux bords
175 }
176
177 div.echange_espace_virtuel();
178
179 return div;
180}

189int Op_Div_PolyMAC::impr(Sortie& os) const
190{
191
192 const int impr_bord=(la_zone_PolyMAC->zone().Bords_a_imprimer().est_vide() ? 0:1);
193 SFichier Flux_div;
194 ouvrir_fichier(Flux_div,"",je_suis_maitre());
195 EcrFicPartage Flux_face;
196 ouvrir_fichier_partage(Flux_face,"",impr_bord);
197 const Schema_Temps_base& sch = equation().probleme().schema_temps();
198 double temps = sch.temps_courant();
199 if (je_suis_maitre()) Flux_div.add_col(temps);
200
201 int nb_compo=flux_bords_.dimension(1);
202// On parcours les frontieres pour sommer les flux par frontiere dans le tableau flux_bord
203 DoubleVect flux_bord(nb_compo);
204 DoubleVect bilan(nb_compo);
205 bilan = 0;
206 for (int num_cl=0; num_cl<la_zone_PolyMAC->nb_front_Cl(); num_cl++)
207 {
208 flux_bord=0;
209 const Cond_lim& la_cl = la_zcl_PolyMAC->les_conditions_limites(num_cl);
210 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
211 int ndeb = frontiere_dis.num_premiere_face();
212 int nfin = ndeb + frontiere_dis.nb_faces();
213 for (int face=ndeb; face<nfin; face++)
214 for(int k=0; k<nb_compo; k++)
215 flux_bord(k)+=flux_bords_(face, k);
216 for(int k=0; k<nb_compo; k++)
217 flux_bord(k)=Process::mp_sum(flux_bord(k));
218
219 if(je_suis_maitre())
220 {
221 for(int k=0; k<nb_compo; k++)
222 {
223// Ajout pour impression sur fichiers separes
224 Flux_div.add_col(flux_bord(k));
225 bilan(k)+=flux_bord(k);
226 }
227 }
228 }
229
230 if(je_suis_maitre())
231 {
232 for(int k=0; k<nb_compo; k++)
233 {
234 Flux_div.add_col(bilan(k));
235 }
236 Flux_div << finl;
237 }
238
239 for (int num_cl=0; num_cl<la_zone_PolyMAC->nb_front_Cl(); num_cl++)
240 {
241 const Frontiere_dis_base& la_fr = la_zcl_PolyMAC->les_conditions_limites(num_cl).frontiere_dis();
242 const Cond_lim& la_cl = la_zcl_PolyMAC->les_conditions_limites(num_cl);
243 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
244 int ndeb = frontiere_dis.num_premiere_face();
245 int nfin = ndeb + frontiere_dis.nb_faces();
246 if (la_zone_PolyMAC->zone().Bords_a_imprimer().contient(la_fr.le_nom()))
247 {
248 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps " << temps << " : " << finl;
249 for (int face=ndeb; face<nfin; face++)
250 {
251 if (dimension==2)
252 Flux_face << "# Face a x= " << la_zone_PolyMAC->xv(face,0) << " y= " << la_zone_PolyMAC->xv(face,1) << " flux=" ;
253 else if (dimension==3)
254 Flux_face << "# Face a x= " << la_zone_PolyMAC->xv(face,0) << " y= " << la_zone_PolyMAC->xv(face,1) << " z= " << la_zone_PolyMAC->xv(face,2) << " flux=" ;
255 for(int k=0; k<nb_compo; k++)
256 Flux_face << " " << flux_bords_(face, k);
257 Flux_face << finl;
258 }
259 Flux_face.syncfile();
260 }
261 }
262
263 return 1;
264}
