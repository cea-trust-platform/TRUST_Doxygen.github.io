118inline double Op_Diff_VEF_Face_Penalise::longueur(const int Face) const
119{
120 double x_sommet1,x_sommet2;
121 double y_sommet1,y_sommet2;
122
123 int sommet1 = zone_vef().face_sommets(Face,0);
124 int sommet2 = zone_vef().face_sommets(Face,1);
125
126 x_sommet1 = domaine().coord(sommet1,0);
127 y_sommet1 = domaine().coord(sommet1,1);
128
129 x_sommet2 = domaine().coord(sommet2,0);
130 y_sommet2 = domaine().coord(sommet2,1);
131
132 return sqrt( pow(x_sommet1 - x_sommet2,2) + pow(y_sommet1 - y_sommet2,2) );
133}

135inline double Op_Diff_VEF_Face_Penalise::diametre(const int Element) const
136{
137 int face1,face2,face3;
138 double longueur_face1,longueur_face2,longueur_face3;
139 double longueur_max,diametre_element;
140
141 face1 = zone_vef().elem_faces(Element,0);
142 face2 = zone_vef().elem_faces(Element,1);
143 face3 = zone_vef().elem_faces(Element,2);
144
145 longueur_face1 = longueur(face1);
146 longueur_face2 = longueur(face2);
147 longueur_face3 = longueur(face3);
148
149 longueur_max = longueur_face1 >= longueur_face2 ? longueur_face1 :
150 longueur_face2;
151 diametre_element = longueur_face3 >= longueur_max ? longueur_face3 :
152 longueur_max;
153
154 return diametre_element;
155}
