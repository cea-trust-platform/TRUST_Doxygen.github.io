27//! class Eval_Dift_VDF_var<br>Cette classe represente un evaluateur de flux diffusif<br>total (diffusion laminaire et turbulente)<br>avec une diffusivite variable en espace.
28/** \sa Eval_Diff_VDF_var */
29class Eval_Dift_VDF_var : public Eval_Diff_VDF_var, public Eval_Turbulence
30{
31public:
32 inline void mettre_a_jour() override;
33 inline void associer_diff_turb(const Champ_Fonc& );
34 inline const Champ_Fonc& diffusivite_turbulente() const { return diffusivite_turbulente_.valeur(); }
35
36// Overloaded methods used by the flux computation in template class:
37 inline double nu_1_impl(int i, int compo) const
38 {
39 const double nu_lam = Eval_Diff_VDF_var::nu_1_impl(i,compo);
40 const double nu_turb = dv_diffusivite_turbulente(i);
41 return nu_lam+nu_turb;
42 }
43
44 inline double nu_2_impl(int i, int compo) const
45 {
46 return Eval_Diff_VDF_var::nu_2_impl(i,compo);
47 }
48
49 inline double nu_1_impl_face(int i, int j, int compo) const
50 {
51 return 0.5*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j));
52 }
53
54 inline double nu_2_impl_face(int i, int j, int k, int l, int compo) const
55 {
56 return 0.25*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j)+
57 dv_diffusivite_turbulente(k)+dv_diffusivite_turbulente(l));
58 }
59
60 inline double nu_lam_impl_face(int i, int j, int k, int l, int compo) const
61 {
62 return Eval_Diff_VDF_var::nu_2_impl_face(i,j,k,l,compo);
63 }
64
65 inline double nu_lam_impl_face2(int i, int j, int compo) const
66 {
67 return Eval_Diff_VDF_var::nu_1_impl_face(i,j,compo);
68 }
69
70 inline double nu_t_impl(int i, int compo) const
71 {
72 return dv_diffusivite_turbulente(i);
73 }
74
75 inline double compute_heq_impl(double d0, int i, double d1, int j, int compo) const
76 {
77 const double heq_lam = Eval_Diff_VDF_var::compute_heq_impl(d0, i, d1, j, compo);
78 const double heq_turb= 0.5*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j))/(d1+d0);
79 return heq_lam + heq_turb;
80 }
81
82 inline double get_equivalent_distance(int boundary_index,int local_face) const override
83 {
84 return equivalent_distance[boundary_index](local_face);
85 }
86
87protected:
88 REF(Champ_Fonc) diffusivite_turbulente_;
89 DoubleVect dv_diffusivite_turbulente;
90};
