31Entree& PrecondA::readOn(Entree& is )
32{
33 Motcle accolade_ouverte("{");
34 Motcle accolade_fermee("}");
35 Motcles les_parametres(6);
36 {
37 les_parametres[0] = "precond0";
38 les_parametres[1] = "precond1";
39 les_parametres[2] = "preconda";
40 les_parametres[3] = "alpha_0";
41 les_parametres[4] = "alpha_1";
42 les_parametres[5] = "alpha_a";
43 int rang;
44
45 Motcle motlu;
46 is >> motlu;
47 if (motlu != accolade_ouverte)
48 {
49 Cerr << "Error when reading the parameters of the conjugate gradient PrecondA " << finl;
50 Cerr << "One expected : " << accolade_ouverte << finl;
51 exit();
52 }
53 is >> motlu;
54 while (motlu != accolade_fermee)
55 {
56 rang = les_parametres.search(motlu);
57 switch(rang)
58 {
59 case 0:
60 {
61 is >> le_precond_0;
62 break;
63 }
64 case 1:
65 {
66 is >> le_precond_1;
67 break;
68 }
69 case 2:
70 {
71 is >> le_precond_a;
72 break;
73 }
74 case 3:
75 {
76 is >> alpha_0;
77 break;
78 }
79 case 4:
80 {
81 is >> alpha_1;
82 break;
83 }
84 case 5:
85 {
86 is >> alpha_a;
87 break;
88 }
89 default :
90 {
91 Cerr << "Error when reading the parameters of the preconditioning PrecondA " << finl;
92 Cerr << "On expected " << les_parametres << " instead of " << motlu << finl;
93 exit();
94 }
95 }
96 is >> motlu;
97 }
98 }
99 return is;
100}

113void PrecondA::prepare_(const Matrice_Base& m, const DoubleVect& v)
114{
115 Init_Status status = get_status();
116 const Matrice_Bloc_Sym& matbloc = ref_cast(Matrice_Bloc_Sym, m);
117 ConstDoubleTab_parts parts(v);
118 prepare_precond(le_precond_0, matbloc.get_bloc(0,0), parts[0], status);
119 prepare_precond(le_precond_1, matbloc.get_bloc(1,1), parts[1], status);
120 if (parts.size() > 2)
121 prepare_precond(le_precond_a, matbloc.get_bloc(2,2), parts[2], status);
122 Precond_base::prepare_(m, v);
123}

127//! Calcule la solution du systeme lineaire: A * solution = b<br>avec la methode de relaxation PrecondA.
128int PrecondA::preconditionner_(const Matrice_Base& matrice,
129 const DoubleVect& b,
130 DoubleVect& solution)
131{
132
133 double pscb = mp_prodscal(b, b);
134
135 if(pscb<1.e-24)
136 {
137 solution=b;
138 return 1;
139 }
140// On calcule solution = inverse(C)*b avec:
141// inverse(C) = inverse((1/w D - E)) *(2-w/w D) * inverse((1/wD -E)t)
142// D :partie diagonale des blocs de la matrice
143// E :partie triangulaire inferieure des blocs de la matrice
144 const Matrice_Bloc_Sym& matbloc=ref_cast(Matrice_Bloc_Sym, matrice);
145 if (matbloc.nb_bloc_lignes()<2
146 || matbloc.nb_bloc_colonnes()<2
147 || matbloc.nb_bloc_lignes()!=matbloc.nb_bloc_colonnes())
148 {
149 Cerr << "The preconditioning SSOR_BLOC can be used only for symetric block matrixes" << finl;
150 Cerr << "containing at least 2*2 blocs. For example, when considering the following VEF discretizations :"<< finl;
151 Cerr << "P0+P1 or P0+Pa or P0+P1+Pa." << finl;
152 exit();
153 }
154 int nblocs=matbloc.nb_bloc_lignes();
155 const Matrice_Base& A00=matbloc.get_bloc(0,0);
156 const Matrice_Base& A01=matbloc.get_bloc(0,1);
157 const Matrice_Base& A11=matbloc.get_bloc(1,1);
158 const Matrice_Base& A0a=matbloc.get_bloc(0,nblocs-1);
159 const Matrice_Base& A1a=matbloc.get_bloc(1,nblocs-1);
160 const Matrice_Base& Aaa=matbloc.get_bloc(nblocs-1,nblocs-1);
161 DoubleTab_parts parties(solution);
162 DoubleVect& X0 = parties[0];
163 DoubleVect& X1 = parties[1];
164 DoubleVect& Xa = parties[parties.size() - 1];
165
166 DoubleVect F(b);
167 DoubleTab_parts parties_f(F);
168 DoubleVect& Y0 = parties_f[0];
169 DoubleVect& Y1 = parties_f[1];
170 DoubleVect& Ya = parties_f[parties_f.size() - 1];
171
172#ifdef TRACE
173 Cerr << finl << "Y0 " << mp_norme_vect(Y0) << finl;
174 Cerr << finl << "Y1 " << mp_norme_vect(Y1) << finl;
175 if(nblocs == 3)
176 Cerr << finl << "Ya " << mp_norme_vect(Ya) << finl;
177 matrice.multvect(b, F);
178 double psc=mp_prodscal(F,b);
179 assert(psc>0);
180 double pscF=mp_prodscal(F,F);
181 Cout << finl <<"cos(angle) no prec " << psc*psc/(pscF*pscb) << finl;
182 Cout << finl <<"(F,b) " << psc << finl;
183 Cout << finl <<"(b,b) " << pscb << finl;
184 Cout << finl <<"(F,F) " << pscF << finl;
185 F.ajoute(-psc/pscb,b,VECT_ALL_ITEMS);
186
187 Cout << finl <<"(F0,F0) " << mp_prodscal(Y0,Y0) << finl;
188 Cout << finl <<"(F1,F1) " << mp_prodscal(Y1,Y1) << finl;
189 if(nblocs == 3)
190 Cout << finl <<"(Fa,Fa) " << mp_prodscal(Ya,Ya) << finl;
191 F=b;
192#endif
193
194// Descente :
195
196 le_precond_1.preconditionner(A11,Y1,X1);
197 operator_multiply(X1, -alpha_1, VECT_ALL_ITEMS);
198 A11.ajouter_multvect(X1, Y1);
199 A01.ajouter_multvect(X1, Y0);
200 if(nblocs == 3)
201 A1a.ajouter_multvectT(X1, Ya);
202 operator_multiply(X1, -1, VECT_ALL_ITEMS);
203 le_precond_0.preconditionner(A00,Y0,X0);
204 operator_multiply(X0, -alpha_0, VECT_ALL_ITEMS);
205 A00.ajouter_multvect(X0, Y0);
206 A01.ajouter_multvectT(X0, Y1);
207 if(nblocs == 3)
208 A0a.ajouter_multvectT(X0, Ya);
209 operator_multiply(X0, -1, VECT_ALL_ITEMS);
210 if(nblocs == 3)
211 {
212 le_precond_a.preconditionner(Aaa,Ya,Xa);
213 operator_multiply(Xa, -alpha_a, VECT_ALL_ITEMS);
214 Aaa.ajouter_multvect(Xa, Ya);
215 A1a.ajouter_multvect(Xa, Y1);
216 A0a.ajouter_multvect(Xa, Y0);
217 operator_multiply(Xa, -1, VECT_ALL_ITEMS);
218 }
219// X contient S,
220 F=solution;
221// Diagonale :
222
223 A11.multvect(Y1,X1);
224 operator_multiply(X1, 1./alpha_1, VECT_ALL_ITEMS);
225 A00.multvect(Y0,X0);
226 operator_multiply(X0, 1./alpha_0, VECT_ALL_ITEMS);
227 if(nblocs == 3)
228 {
229 Aaa.multvect(Ya,Xa);
230 operator_multiply(Xa, 1./alpha_a, VECT_ALL_ITEMS);
231 }
232// X contient T,
233 F=solution;
234
235// Remontee :
236
237 if(nblocs == 3)
238 {
239 le_precond_a.preconditionner(Aaa,Ya,Xa);
240 operator_multiply(Xa, -alpha_a, VECT_ALL_ITEMS);
241 Aaa.ajouter_multvect(Xa, Ya);
242 A1a.ajouter_multvect(Xa, Y1);
243 A0a.ajouter_multvect(Xa, Y0);
244 operator_multiply(Xa, -1., VECT_ALL_ITEMS);
245 }
246 le_precond_0.preconditionner(A00,Y0,X0);
247 operator_multiply(X0, -alpha_0, VECT_ALL_ITEMS);
248 A00.ajouter_multvect(X0, Y0);
249 A01.ajouter_multvectT(X0, Y1);
250 if(nblocs == 3)
251 A0a.ajouter_multvectT(X0, Ya);
252 operator_multiply(X0, -1., VECT_ALL_ITEMS);
253 le_precond_1.preconditionner(A11,Y1,X1);
254 operator_multiply(X1, -alpha_1, VECT_ALL_ITEMS);
255 A11.ajouter_multvect(X1, Y1);
256 A01.ajouter_multvect(X1, Y0);
257 if(nblocs == 3)
258 A1a.ajouter_multvectT(X1, Ya);
259 operator_multiply(X1, -1., VECT_ALL_ITEMS);
260#ifdef TRACE
261 matrice.multvect(solution, F);
262 psc=mp_prodscal(F,b);
263 assert(psc>0);
264 pscF=mp_prodscal(F,F);
265 Cout << finl <<"cos(angle) " << psc*psc/(pscF*pscb) << finl;
266 Cout << finl <<"(F,b) " << psc << finl;
267 Cout << finl <<"(b,b) " << pscb << finl;
268 Cout << finl <<"(F,F) " << pscF << finl;
269 F.ajoute(-psc/pscb,b);
270
271 Cout << finl <<"(F0,F0) " << mp_prodscal(Y0,Y0) << finl;
272 Cout << finl <<"(F1,F1) " << mp_prodscal(Y1,Y1) << finl;
273 if(nblocs == 3)
274 Cout << finl <<"(Fa,Fa) " << mp_prodscal(Ya,Ya) << finl;
275#endif
276
277 return 1;
278}
