112int Mod_turb_hyd_base::lire_motcle_non_standard(const Motcle& mot, Entree& is)
113{
114 Motcle motlu;
115 if (mot=="turbulence_paroi")
116 {
117 loipar.associer_modele(*this);
118 is >> loipar;
119 is >> loipar.valeur();
120 return 1;
121 }
122 else if (loipar.valeur().que_suis_je()!="negligeable_VDF" && loipar.valeur().que_suis_je()!="negligeable_VEF" && !loipar.valeur().que_suis_je().debute_par("negligeable_PolyMAC"))
123 {
124 if (mot=="dt_impr_ustar")
125 {
126// nom_fichier_=Objet_U::nom_du_cas()+"_"+equation().probleme().le_nom()+"_ustar";
127 is >> dt_impr_ustar;
128 }
129 else if (mot=="dt_impr_ustar_mean_only")
130 {
131 Nom accolade_ouverte="{";
132 Nom accolade_fermee="}";
133 nom_fichier_=Objet_U::nom_du_cas()+"_"+equation().probleme().le_nom()+"_ustar_mean_only";
134 Domaine& dom=equation().probleme().domaine();
135 Zone& zone=dom.zone(0);
136 LIST(Nom) nlistbord_dom; //!< liste stockant tous les noms de frontiere du domaine
137 int nbfr=zone.nb_front_Cl();
138 for (int b=0; b<nbfr; b++)
139 {
140 Frontiere& org=zone.frontiere(b);
141 nlistbord_dom.add(org.le_nom());
142 }
143 is >> motlu;
144 if ( motlu != accolade_ouverte )
145 {
146 Cerr << motlu << " is not a keyword understood by " << que_suis_je() << " in lire_motcle_non_standard"<< finl;
147 Cerr << "A specification of kind : dt_impr_ustar_mean_only { dt_impr periode [boundaries nb_boundaries boundary_name1 boundary_name2 ... ] } was expected."<<finl;
148 exit();
149 }
150 is >> motlu;
151 if (motlu!="dt_impr")
152 {
153 Cerr<<"We expected dt_impr..."<<finl;
154 exit();
155 }
156 is >> dt_impr_ustar_mean_only;
157
158 is >> motlu; //!< boundaries ou accolade_fermee ou pasbon
159 if ( motlu != accolade_fermee )
160 {
161 if ( motlu == "boundaries" )
162 {
163 boundaries_=1;
164 int nb_bords=0;
165 Nom nom_bord_lu;
166
167// read boundaries number
168 is >> nb_bords;
169 if ( nb_bords != 0 )
170 {
171// read boundaries
172 for ( int i=0; i<nb_bords; i++ )
173 {
174 is >> nom_bord_lu;
175 boundaries_list.add(Nom(nom_bord_lu));
176// verif nom bords
177 if (!nlistbord_dom.contient(boundaries_list[i]))
178 {
179 Cerr << "Problem in the dt_impr_ustar_mean_only instruction:" << finl;
180 Cerr << "The boundary named '" << boundaries_list[i] << "' is not a boundary of the domain " << dom.le_nom() << "." << finl;
181 exit();
182 }
183 }
184 }
185// lecture accolade fermee
186 is >> motlu;
187 if ( motlu != accolade_fermee )
188 {
189 Cerr << "Problem in the dt_impr_ustar_mean_only instruction:" << finl;
190 Cerr << "TRUST wants to read a '" << accolade_fermee << "' but find '" << motlu << "'!!" << finl;
191 exit();
192 }
193 }
194 else
195 {
196 Cerr << motlu << " is not a keyword understood by " << que_suis_je() << " in lire_motcle_non_standard"<< finl;
197 Cerr << "A specification of kind : dt_impr_ustar_mean_only { dt_impr periode [boundaries nb_boundaries boundary_name1 boundary_name2 ... ] } was expected."<<finl;
198 exit();
199 }
200 }
201 } //!< fin dt_impr_ustar_mean_only
202 else
203 {
204 Cerr << "Please remove dt_impr_ustar option if the wall law is of Negligeable type." << finl;
205 exit();
206 }
207 } //!< fin loi paroi negligeable
208
209 else
210 {
211 Cerr << mot << " is not a keyword understood by " << que_suis_je() << " in lire_motcle_non_standard"<< finl;
212 exit();
213 }
214 return -1;
215}

416 {
417 lire_distance_paroi();
418 contient_distance_paroi = true;
419 break;
420 }
421 }
422 }
423 ++curseur;
424 }
425 loipar.imprimer_premiere_ligne_ustar(boundaries_, boundaries_list, nom_fichier_);
426 return res;

503int Mod_turb_hyd_base::limpr_ustar(double temps_courant, double temps_prec, double dt, double dt_ustar) const
504{
505 const Schema_Temps_base& sch = mon_equation->schema_temps();
506 if (sch.nb_pas_dt()==0)
507 return 0;
508 if (dt_ustar<=dt || (( sch.temps_cpu_max_atteint() || (!get_disable_stop() && sch.stop_lu()) || sch.temps_final_atteint() || sch.nb_pas_dt_max_atteint() || sch.nb_pas_dt()==1 || sch.stationnaire_atteint()) && !est_egal(dt_ustar,1.e20) ) )
509 return 1;
510 else
511 {
512// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
513 double i, j, epsilon = 1.e-8;
514 modf(temps_courant/dt_ustar + epsilon, &i);
515 modf(temps_prec/dt_ustar + epsilon, &j);
516 return ( i>j );
517 }
518}

607void Mod_turb_hyd_base::limiter_viscosite_turbulente()
608{
609// On initial
610 int size = viscosite_turbulente().valeurs().size();
611 DoubleTab& visco_turb = la_viscosite_turbulente.valeurs();
612 if (borne_visco_turb.size()==0)
613 borne_visco_turb=visco_turb; //.resize(size);
614 borne_visco_turb = XNUTM;
615 if (calcul_borne_locale_visco_turb_ && (equation().schema_temps().nb_pas_dt() != 0 || equation().probleme().reprise_effectuee()))
616 {
617// On recalcule les bornes de la viscosite turbulente apres le premier pas de temps
618 const Operateur_Diff_base& op_diff = ref_cast(Operateur_Diff_base,equation().operateur(0).l_op_base());
