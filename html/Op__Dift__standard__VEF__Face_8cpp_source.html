149void Op_Dift_standard_VEF_Face::calcul_divergence(DoubleTab& dif,
150 const DoubleTab& grad,
151 const DoubleTab& gradt,
152 const DoubleTab& nu,
153 const DoubleTab& nu_turb,
154 const Zone_Cl_VEF& zone_Cl_VEF,
155 const Zone_VEF& zone_VEF,
156 DoubleTab& tab_flux_bords,
157 int nbr_comp) const
158{
159
160 const IntTab& face_voisins = zone_VEF.face_voisins();
161 const DoubleTab& face_normale = zone_VEF.face_normales();
162// const DoubleVect& vol = zone_VEF.volumes();
163 int nb_faces = zone_VEF.nb_faces();
164// int face_int = zone_VEF.premiere_face_int();
165 int num_cl;
166
167 int num_face0,elem10, elem2;
168 int i,j;
169 double nu1,nu2;
170 double nu1t,nu2t;
171 double flux;
172 double c1=0., c2=0., c3=0., c4=0.;
173
174 if(nu_lu) c1=1.;
175 if(nut_lu) c2=1.;
176 if(nu_transp_lu) c3=1.;
177 if(nut_transp_lu) c4=1.;
178
179// Traitement des bords
180 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence dif 0 ",dif);
181
182 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
183 int nb_cl=les_cl.size();
184 for (num_cl=0; num_cl<nb_cl; num_cl++)
185 {
186 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(num_cl);
187 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
188// const IntTab& elem_faces = zone_VEF.elem_faces();
189 int nb_faces_b=le_bord.nb_faces();
190 int num1 = le_bord.num_premiere_face();
191 int num2 = num1 + nb_faces_b;
192
193 if (sub_type(Periodique,la_cl.valeur()))
194 {
195 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
196 IntVect fait(nb_faces_b);
197 fait = 0;
198 for (num_face0=num1; num_face0<num2; num_face0++)
199 {
200 int face_associee=la_cl_perio.face_associee(num_face0-num1);
201 if (fait(num_face0-num1) == 0)
202 {
203 fait(num_face0-num1) = 1;
204 fait(face_associee) = 1;
205 elem10 = face_voisins(num_face0,0);
206 elem2 = face_voisins(num_face0,1);
207 nu1 = c1*nu[elem10]+c2*nu_turb[elem10];
208 nu2 = c1*nu[elem2]+c2*nu_turb[elem2];
209 nu1t= c3*nu[elem10]+c4*nu_turb[elem10];
210 nu2t= c3*nu[elem2]+c4*nu_turb[elem2];
211
212 for (i=0; i<nbr_comp; i++)
213 for (j=0; j<nbr_comp; j++)
214 {
215 dif(num_face0,i) -= face_normale(num_face0,j)*
216 (nu1*grad(elem10,i,j)+nu1t*gradt(elem10,j,i)
217 -(nu2*grad(elem2,i,j)+nu2t*gradt(elem2,j,i)));
218// Cerr << num_face << " " << i << dif(num_face,i) << finl;
219 }
220
221 for (i=0; i<nbr_comp; i++)
222 dif(face_associee+num1,i)=dif(num_face0,i);
223
224// dif(num_face,num_comp) -= 0.5*ori*face_normale(num_face,num_comp2)*(2*nu*Sij(elem,num_comp,num_comp2));
225// Patrick : le coefficient 0.5 dans l'ancienne formulation est incorrect : les faces periodiques ne sont traitees qu'UNE fois
226 }
227 } //!< Fin de la boucle sur les faces
228 } //!< Fin du cas periodique
229 else if ( (sub_type(Dirichlet,la_cl.valeur()))
230 || (sub_type(Dirichlet_homogene,la_cl.valeur()))
231 )
232 {
233 if (sub_type(Dirichlet_paroi_fixe,la_cl.valeur()) ||
234 sub_type(Dirichlet_paroi_defilante,la_cl.valeur()) )
235 {
236 for (num_face0=num1; num_face0<num2; num_face0++)
237 {
238 int elem1=face_voisins(num_face0,0);
239 nu1 = c1*nu[elem1]+c2*nu_turb[elem1];
240 nu1t= c3*nu[elem1]+c4*nu_turb[elem1];
241
242 for (i=0; i<nbr_comp; i++)
243 {
244 flux=0.;
245 for (j=0; j<nbr_comp; j++)
246 flux +=face_normale(num_face0,j)*(nu1*grad(elem1,i,j)+nu1t*gradt(elem1,j,i));
247
248 dif(num_face0,i) = 0.; //!< PQ : valable en regime turbulent d'apres profil lineaire de la vitesse dans la sous couche visqueuse
249 tab_flux_bords(num_face0,i) -= flux;
250 }
251 } //!< Fin de la boucle sur les faces
252 } //!< Fin du cas paroi fixe ou defilante
253 else //On est en Dirichlet ou Dirichlet homogene : Boucle speciale car les volumes de controle sont modifies
254 {
255// Cerr << "Si on passe la , on est mal!!! : Ce n'est pas code!!!" << finl;
256 } //!< Fin du traitement des CL de type Dirichlet ou Dirichlet homogene hors paroi_fixe paroi_defilante
257 } //!< Fin du traitement des CL de type Dirichlet ou Dirichlet homogene
258 else //!< Pour les autres conditions aux limites
259 {
260 for (int num_face=num1; num_face<num2; num_face++)
261 {
262 int elem1=face_voisins(num_face,0);
263 nu1 = c1*nu[elem1]+c2*nu_turb[elem1];
264 nu1t= c3*nu[elem1]+c4*nu_turb[elem1];
265
266 for (i=0; i<nbr_comp; i++)
267 {
268 flux=0.;
269 for (j=0; j<nbr_comp; j++)
270 flux +=face_normale(num_face,j)*(nu1*grad(elem1,i,j)+nu1t*gradt(elem1,j,i));
271
272 dif(num_face,i) = 0.; //!< PQ : en attendant de faire mieux
273 tab_flux_bords(num_face,i) -= flux;
274 }
275
276 }
277 }
278 } //!< Fin du traitement des CL
279
280 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence dif 1 ",dif);
281 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence nu 1 ",nu);
282 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence nu_turb 1 ",nu_turb);
283 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence grad 1 ",grad);
284 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence gradt 1 ",gradt);
285
286
287//
288
289 for (num_face0=zone_VEF.premiere_face_int(); num_face0<nb_faces; num_face0++)
290 {
291 elem10 = face_voisins(num_face0,0);
292 elem2 = face_voisins(num_face0,1);
293 nu1 = c1*nu[elem10]+c2*nu_turb[elem10];
294 nu2 = c1*nu[elem2]+c2*nu_turb[elem2];
295 nu1t= c3*nu[elem10]+c4*nu_turb[elem10];
296 nu2t= c3*nu[elem2]+c4*nu_turb[elem2];
297
298 for (i=0; i<nbr_comp; i++)
299 for (j=0; j<nbr_comp; j++)
300 dif(num_face0,i) -= face_normale(num_face0,j)*
301 (nu1*grad(elem10,i,j)+nu1t*gradt(elem10,j,i)
302 -(nu2*grad(elem2,i,j)+nu2t*gradt(elem2,j,i)));
303
304// (nu1*grad(elem1,i,j)+nu1t*gradt(elem1,j,i)-(i==j)*dst*nu1*div(elem1)
305// -(nu2*grad(elem2,i,j)+nu2t*gradt(elem2,j,i)-(i==j)*dst*nu1*div(elem2)));
306
307// dif(num_face,num_comp) -= ori*face_normale(num_face,num_comp2)*(2*nu*Sij(elem,num_comp,num_comp2));
308 } //!< Fin de la boucle sur les faces internes
309
310 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence dif 2 ",dif);
311
312 dif.echange_espace_virtuel();
313 Debog::verifier("Op_Dift_standard_VEF_Face::calcul_divergence dif 3 ",dif);
314

316DoubleTab& Op_Dift_standard_VEF_Face::ajouter(const DoubleTab& inconnue,
317 DoubleTab& resu) const
318{
319 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();

327 modifier_flux(*this);
328 return resu;
329}
