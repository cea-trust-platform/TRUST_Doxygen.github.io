53void Op_Dift_VDF_base::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
54{
55 statistiques().begin_count(diffusion_counter_);
56 const std::string& nom_inco = equation().inconnue().le_nom().getString();
57 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
58 const DoubleTab& inco = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : equation().inconnue().valeurs();
59
60 if(mat) iter.ajouter_contribution(inco, *mat);
61 iter.ajouter(inco, secmem);
62 if (equation().domaine_application() == Motcle("Hydraulique"))
63// On est dans le cas des equations de Navier_Stokes
64 {
65// Ajout du terme supplementaire en V/(R*R) dans le cas des coordonnees axisymetriques
66 if(Objet_U::bidim_axi == 1)
67 {
68 const Zone_VDF& zvdf=iter.zone();
69 const DoubleTab& xv=zvdf.xv();
70 const IntVect& ori=zvdf.orientation();
71 const IntTab& face_voisins=zvdf.face_voisins();
72 const DoubleVect& volumes_entrelaces=zvdf.volumes_entrelaces();
73 int face, nb_faces=zvdf.nb_faces();//, cst;
74 DoubleTrav diffu_tot(zvdf.nb_elem_tot());
75 double db_diffusivite;
76
77// if (equation().que_suis_je() == "Navier_Stokes_Turbulent")
78 const RefObjU& modele_turbulence = equation().get_modele(TURBULENCE);
79 if (sub_type(Mod_turb_hyd_base,modele_turbulence.valeur()))
80 {
81 const Eval_Diff_VDF& eval=dynamic_cast<const Eval_Diff_VDF&> (iter->evaluateur());
82 const Champ_base& ch=eval.get_diffusivite();
83 const DoubleVect& tab_diffusivite=ch.valeurs();
84 int size=diffu_tot.size_totale();
85 if (tab_diffusivite.size() == 1)
86 {
87 db_diffusivite = tab_diffusivite[0];
88 const Eval_Dift_VDF_const& eval_dift = static_cast<const Eval_Dift_VDF_const&> (eval);
89 const Champ_Fonc& ch_diff_turb = eval_dift.diffusivite_turbulente();
90 const DoubleVect& diffusivite_turb = ch_diff_turb.valeurs();
91 for (int i=0; i<size; i++)
92 diffu_tot[i] = db_diffusivite+diffusivite_turb[i];
93 }
94 else
95 {
96 const Eval_Dift_VDF_var& eval_dift = static_cast<const Eval_Dift_VDF_var&> (eval);
97 const Champ_Fonc& ch_diff_turb = eval_dift.diffusivite_turbulente();
98 const DoubleVect& diffusivite_turb = ch_diff_turb.valeurs();
99 for (int i=0; i<size; i++)
100 diffu_tot[i] = tab_diffusivite[i]+diffusivite_turb[i];
101 }
102 }
103 else
104 {
105 Cerr << "Method Op_Dift_VDF_base::ajouter_blocs" << finl;
106 Cerr << "The type "<<equation().que_suis_je()<<" of the equation associated "<< finl;
107 Cerr << "with the current operator "<< que_suis_je() << "is not coherent."<<finl;
108 Cerr << "It must be sub typing of Navier_Stokes_Turbulent" << finl;
109 exit();
110 }
111
112 for (face=0; face<nb_faces; face++)
113 if(ori(face)==0)
114 {
115 int elem1=face_voisins(face,0);
116 int elem2=face_voisins(face,1);
117 if(elem1==-1)
118 db_diffusivite=diffu_tot(elem2);
119 else if (elem2==-1)
120 db_diffusivite=diffu_tot(elem1);
121 else
122 db_diffusivite=0.5*(diffu_tot(elem2)+diffu_tot(elem1));
123 double r= xv(face,0);
124 if (r >= 1.e-24)
125 {
126 if (mat) (*mat)(face, face) += db_diffusivite * volumes_entrelaces(face) / (r * r);
127 secmem(face) -= inco(face) * db_diffusivite * volumes_entrelaces(face) / (r * r);
128 }
129 }
130 }
131 }
132 statistiques().end_count(diffusion_counter_);
133
134}

138void Op_Dift_VDF_base::contribuer_au_second_membre(DoubleTab& resu) const
139{
140 iter.contribuer_au_second_membre(resu);
141 if (equation().domaine_application() == Motcle("Hydraulique"))
142// On est dans le cas des equations de Navier_Stokes
143 {
144// Ajout du terme supplementaire en V/(R*R) dans le cas des coordonnees axisymetriques
145 if(Objet_U::bidim_axi == 1)
146 {
147 const Zone_VDF& zvdf=iter.zone();
148 const Zone_Cl_VDF& zclvdf = iter.zone_Cl();
149 const DoubleTab& xv=zvdf.xv();
150 const IntVect& ori=zvdf.orientation();
151 const IntTab& face_voisins=zvdf.face_voisins();
152 const DoubleVect& volumes_entrelaces=zvdf.volumes_entrelaces();
153// int face, nb_faces=zvdf.nb_faces(), cst;
154 DoubleTrav diffu_tot(zvdf.nb_elem_tot());
155 double db_diffusivite;
156 if (equation().que_suis_je() == "Navier_Stokes_Turbulent")
157 {
158 const Eval_Diff_VDF& eval=dynamic_cast<const Eval_Diff_VDF&> (iter->evaluateur());
159 const Champ_base& ch=eval.get_diffusivite();
160 const DoubleVect& tab_diffusivite=ch.valeurs();
161 if (tab_diffusivite.size() == 1)
162 {
163 db_diffusivite = tab_diffusivite[0];
164 const Eval_Dift_VDF_const& eval_dift = static_cast<const Eval_Dift_VDF_const&> (eval);
165 const Champ_Fonc& ch_diff_turb = eval_dift.diffusivite_turbulente();
166 const DoubleVect& diffusivite_turb = ch_diff_turb.valeurs();
167 for (int i=0; i<diffusivite_turb.size(); i++)
168 diffu_tot[i] = db_diffusivite+diffusivite_turb[i];
169 }
170 else
171 {
172 const Eval_Dift_VDF_var& eval_dift = static_cast<const Eval_Dift_VDF_var&> (eval);
173 const Champ_Fonc& ch_diff_turb = eval_dift.diffusivite_turbulente();
174 const DoubleVect& diffusivite_turb = ch_diff_turb.valeurs();
175 for (int i=0; i<diffusivite_turb.size(); i++)
176 diffu_tot[i] = tab_diffusivite[i]+diffusivite_turb[i];
177 }
178 }
179 else
180 {
181 Cerr << "Probleme dans Op_Dift_VDF_base::ajouter avec le type de l'equation" << finl;
182 Cerr << "on n'a pas prevu d'autre cas que Navier_Stokes_Turbulent" << finl;
183 exit();
184 }
185// Boucle sur les conditions limites pour traiter les faces de bord
186
187 int ndeb,nfin,num_face;
188
189 for (int n_bord=0; n_bord<zvdf.nb_front_Cl(); n_bord++)
190 {
191 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
192
193 if (sub_type(Dirichlet,la_cl.valeur()))
194 {
195 const Dirichlet& la_cl_diri = ref_cast(Dirichlet,la_cl.valeur());
196 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
197 ndeb = le_bord.num_premiere_face();
198 nfin = ndeb + le_bord.nb_faces();
199 double r_2,coef;
200 for (num_face=ndeb; num_face<nfin; num_face++)
201 {
202 if (ori(num_face) == 0)
203 {
204 int elem1=face_voisins(num_face,0);
205 int elem2=face_voisins(num_face,1);
206 if(elem1==-1) db_diffusivite = diffu_tot(elem2);
207 else
208 {
209 assert (elem2==-1);
210 db_diffusivite=diffu_tot(elem1);
211 }
212 r_2 = xv(num_face,0)*xv(num_face,0);
213 coef = volumes_entrelaces(num_face)/r_2;
214// resu(num_face) -=0.;
215 resu(num_face) -= (la_cl_diri.val_imp(num_face-ndeb)*coef)*db_diffusivite;
216 }
217 }
218 }
219 }
220 }
221 }
