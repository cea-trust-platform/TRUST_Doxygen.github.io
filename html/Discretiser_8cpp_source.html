90Entree& Discretiser::interpreter(Entree& is)
91{
92saisie :
93 Nom nom1, nom2;
94 is >> nom1 >> nom2;
95 Objet_U& ob1=objet(nom1);
96 Objet_U& ob2=objet(nom2);
97
98 if (is_deja_appele==0)
99 is_deja_appele = 1;
100 else
101 {
102 Cerr<<"Be careful, you call several times the interpreter Discretiser"<<finl;
103 Cerr<<"there is probably an error in your data set"<<finl;
104// exit();
105 }
106
107 if( !( sub_type(Discretisation_base, ob2) ) )
108 {
109 Cerr << nom2 << "is not a recognized discretization" << finl;
110 exit();
111 }
112 Discretisation_base& typ=ref_cast(Discretisation_base, ob2);
113 Cerr << "The chosen discretization is of type " << typ.que_suis_je() << finl;
114 if (typ.que_suis_je()=="VEF")
115 {
116 Cerr << "Since 1.6 version, the keyword VEF is obsolete." << finl;
117 Cerr << "Use keyword VEFPreP1B and P0 support to match the old VEF keyword." << finl;
118 exit();
119 }
120// Test on bidim_axi
121 if ( dimension == 2 && !(typ.que_suis_je()=="VDF") && bidim_axi == 1 )
122 {
123 Cerr << "The feature 'bidim_axi' is only available for 2D calculation with VDF discretisation." << finl;
124 exit();
125 }
126//
127 if( sub_type(Probleme_base, ob1) )
128 {
129 Probleme_base& pb=ref_cast(Probleme_base,ob1);
130 if(!pb.schema_temps().lu())
131 {
132 Cerr << "it must have read the time scheme before to discretize" << finl;
133 exit();
134 }
135 Cerr << "The chosen time scheme is of type " << pb.schema_temps().que_suis_je() << finl;
136 Cerr << "We treat the problem " << pb.le_nom() << " of type " << pb.que_suis_je() << finl;
137 Cerr << "Discretization of " << pb.le_nom() << " in progress..." << finl;
138 pb.discretiser(typ);
139 }
140 else if( sub_type(Probleme_Couple, ob1) )
141 {
142 Probleme_Couple& pbc=ref_cast(Probleme_Couple,ob1);
143 if(!pbc.schema_temps().lu())
144 {
145 Cerr << "it must have read the time scheme before to discretize" << finl;
146 exit();
147 }
148 Cerr << "The chosen time scheme is of type " << pbc.schema_temps().que_suis_je() << finl;
149 Cerr << "We treat the problem " << pbc.le_nom() << " of type " << pbc.que_suis_je() << finl;
150 Cerr << "It corresponds to the coupling of the following problems: " << finl;
151 for(int i=0; i< pbc.nb_problemes(); i++)
152 Cerr << " " << pbc.probleme(i).le_nom() << " of type " << pbc.probleme(i).que_suis_je() << finl;
153 Cerr << "Discretization of " << pbc.le_nom() << " in progress..." << finl;
154 pbc.discretiser(typ);
155 }
156 else
157 {
158 Cerr << "It is not known discretize a : " << ob1.que_suis_je() << finl;
159 Cerr << "return at the seizure of the arguments" << finl;
160 goto saisie;
161 }
162 return is;
163}
