133//! Methode appelee par Zone_VF::discretiser apres la creation<br>des faces reelles.<br>On reordonne les faces de sorte a placer les faces "non standard"<br>au debut de la liste des faces. Les faces non standard sont celles<br>dont les volumes de controles sont modifies par les conditions aux<br>limites.
134void Zone_Poly_base::reordonner(Faces& les_faces)
135{
136 if (Process::je_suis_maitre())
137 Cerr << "Zone_Poly_base::reordonner faces " << finl;
138
139// Construction de rang_elem_non_std_ :
140// C'est un vecteur indexe par les elements du domaine.
141// size() = nb_elem()
142// size_tot() = nb_elem_tot()
143// Valeurs dans le tableau :
144// rang_elem_non_std_[i] = -1 si l'element i est standard,
145// sinon
146// rang_elem_non_std_[i] = j, ou j est l'indice de l'element dans
147// les tableaux indexes par les elements non standards (par exemple
148// le tableau Zone_Cl_Poly_base::type_elem_Cl_).
149// Un element est non standard s'il est voisin d'une face frontiere.
150 {
151 const Domaine& dom = zone().domaine();
152 const int nb_elements = nb_elem();
153 const int nb_faces_front = zone().nb_faces_frontiere();
154 dom.creer_tableau_elements(rang_elem_non_std_);
155// rang_elem_non_std_.resize(nb_elements);
156// Scatter::creer_tableau_distribue(dom, Joint::ELEMENT, rang_elem_non_std_);
157 rang_elem_non_std_ = -1;
158 int nb_elems_non_std = 0;
159// D'abord on marque les elements non standards avec rang_elem_non_std_[i] = 0
160 for (int i_face = 0; i_face < nb_faces_front; i_face++)
161 {
162 const int elem = les_faces.voisin(i_face, 0);
163 if (rang_elem_non_std_[elem] < 0)
164 {
165 rang_elem_non_std_[elem] = 0;
166 nb_elems_non_std++;
167 }
168 }
169 nb_elem_std_ = nb_elements - nb_elems_non_std;
170 rang_elem_non_std_.echange_espace_virtuel();
171 int count = 0;
172 const int size_tot = rang_elem_non_std_.size_totale();
173// On remplace le marqueur "0" par l'indice j.
174 for (int elem = 0; elem < size_tot; elem++)
175 {
176 if (rang_elem_non_std_[elem] == 0)
177 {
178 rang_elem_non_std_[elem] = count;
179 count++;
180 }
181 }
182 }
183
184// Construction du tableau de renumerotation des faces. Ce tableau,
185// une fois trie dans l'ordre croissant donne l'ordre des faces dans
186// la zone_VF. La cle de tri est construite de sorte a pouvoir retrouver
187// l'indice de la face a partir de la cle par la formule :
188// indice_face = cle % nb_faces
189 const int nbfaces = les_faces.nb_faces();
190 ArrOfInt sort_key(nbfaces);
191 {
192 nb_faces_std_ =0;
193 const int nb_faces_front = zone().nb_faces_frontiere();
194// Attention : face_voisins_ n'est pas encore initialise, il
195// faut passer par les_faces.voisins() :
196 const IntTab& facevoisins = les_faces.voisins();
197// On place en premier les faces de bord:
198 int i_face;
199 for (i_face = 0; i_face < nbfaces; i_face++)
200 {
201 int key = -1;
202 if (i_face < nb_faces_front)
203 {
204// Si la face est au bord, elle doit etre placee au debut
205// (en fait elle ne doit pas etre renumerotee)
206 key = i_face;
207 }
208 else
209 {
210 const int elem0 = facevoisins(i_face, 0);
211 const int elem1 = facevoisins(i_face, 1);
212// Ces faces ont toujours deux voisins.
213 assert(elem0 >= 0 && elem1 >= 0);
214 if (rang_elem_non_std_[elem0] >= 0 || rang_elem_non_std_[elem1] >= 0)
215 {
216// Si la face est voisine d'un element non standard, elle
217// doit etre classee juste apres les faces de bord:
218 key = i_face;
219 }
220 else
221 {
222// Face standard : a la fin du tableau
223 key = i_face + nbfaces;
224 nb_faces_std_++;
225 }
226 }
227 sort_key[i_face] = key;
228 }
229 sort_key.ordonne_array();
230
231// On transforme a nouveau la cle en numero de face:
232 for (i_face = 0; i_face < nbfaces; i_face++)
233 {
234 const int key = sort_key[i_face] % nbfaces;
235 sort_key[i_face] = key;
236 }
237 }
238// On reordonne les faces:
239 {
240 IntTab& faces_sommets = les_faces.les_sommets();
241 IntTab old_tab(faces_sommets);
242 const int nb_som_faces = faces_sommets.dimension(1);
243 for (int i = 0; i < nbfaces; i++)
244 {
245 const int old_i = sort_key[i];
246 for (int j = 0; j < nb_som_faces; j++)
247 faces_sommets(i, j) = old_tab(old_i, j);
248 }
249 }
250
251 {
252 IntTab& faces_voisins = les_faces.voisins();
253 IntTab old_tab(faces_voisins);
254 for (int i = 0; i < nbfaces; i++)
255 {
256 const int old_i = sort_key[i];
257 faces_voisins(i, 0) = old_tab(old_i, 0);
258 faces_voisins(i, 1) = old_tab(old_i, 1);
259 }
260 }
261
262// Calcul de la table inversee: reverse_index[ancien_numero] = nouveau numero
263 ArrOfInt reverse_index(nbfaces);
264 {
265 for (int i = 0; i < nbfaces; i++)
266 {
267 const int j = sort_key[i];
268 reverse_index[j] = i;
269 }
270 }
271// Renumerotation de elem_faces:
272 {
273// Nombre d'indices de faces dans le tableau
274 const int nb_items = elem_faces_.size();
275 ArrOfInt& array = elem_faces_;
276 for (int i = 0; i < nb_items; i++)
277 {
278 const int old = array[i];
279 if (old<0)
280 array[i] = -1;
281 else
282 array[i] = reverse_index[old];
283 }
284 }
285// Mise a jour des indices des faces de joint:
286 {
287 Joints& joints = zone().faces_joint();
288 const int nbjoints = joints.size();
289 for (int i_joint = 0; i_joint < nbjoints; i_joint++)
290 {
291 Joint& un_joint = joints[i_joint];
292 ArrOfInt& indices_faces = un_joint.set_joint_item(Joint::FACE).set_items_communs();
293 const int nbfaces2 = indices_faces.size_array();
294 assert(nbfaces2 == un_joint.nb_faces()); //!< renum_items_communs rempli ?
295 for (int i = 0; i < nbfaces2; i++)
296 {
297 const int old = indices_faces[i]; //!< ancien indice local
298 indices_faces[i] = reverse_index[old];
299 }
300// Les faces de joint ne sont plus consecutives dans le
301// tableau: num_premiere_face n'a plus ne sens
302 un_joint.fixer_num_premiere_face(-1);
303 }
304 }

322void Zone_Poly_base::discretiser()
323{
324
325 Zone& zone_geom=zone();
326 typer_elem(zone_geom);
327 Elem_geom_base& elem_geom = zone_geom.type_elem().valeur();
328 Zone_VF::discretiser();
329
330// Correction du tableau facevoisins:
331// A l'issue de Zone_VF::discretiser(), les elements voisins 0 et 1 d'une
332// face sont les memes sur tous les processeurs qui possedent la face.
333// Si la face est virtuelle et qu'un des deux elements voisins n'est
334// pas connu (il n'est pas dans l'epaisseur du joint), l'element voisin
335// vaut -1. Cela peut etre un voisin 0 ou un voisin 1.
336// On corrige les faces virtuelles pour que, si un element voisin n'est
337// pas connu, alors il est voisin1. Le voisin0 est donc toujours valide.
338 {
339 IntTab& face_vois = face_voisins();
340 const int debut = nb_faces();
341 const int fin = nb_faces_tot();
342 for (int i = debut; i < fin; i++)
343 {
344 if (face_voisins(i, 0) == -1)
345 {
346 face_vois(i, 0) = face_vois(i, 1);
347 face_vois(i, 1) = -1;
348 }
349 }
350 }
351
352// Verification de la coherence entre l'element geometrique et
353// l'elemnt de discretisation
354
355
356 if (sub_type(Segment_poly,type_elem_.valeur()))
357 {
358 if (!sub_type(Segment,elem_geom))
359 {
360 Cerr << " The type of the element " << elem_geom.que_suis_je() << " is incorrect" << finl;
361 Process::exit();
362 }
363 }
364 else if (sub_type(Tri_poly,type_elem_.valeur()))
365 {
366 if (!sub_type(Triangle,elem_geom))
367 {
368 Cerr << " The type of the element " << elem_geom.que_suis_je() << " is incorrect" << finl;
369 Cerr << " Only the Triangle type is compatible with the PolyMAC_P0 discretisation in dimension 2" << finl;
370 Cerr << " You must triangulate the domain when using the TRUST mesher" ;
371 Cerr << " This can be done by adding : Trianguler nom_dom" << finl;
372 Process::exit();
373 }
374 }
375 else if (sub_type(Tetra_poly,type_elem_.valeur()))
376 {
377 if (!sub_type(Tetraedre,elem_geom))
378 {
379 Cerr << " The type of the element " << elem_geom.que_suis_je() << " is incorrect" << finl;
380 Cerr << " Only the Tetrahedral type is compatible with the PolyMAC_P0 discretisation in dimension 3" << finl;
381 Cerr << " You must tetrahedrize the domain when using the TRUST mesher" ;
382 Cerr << " This can be done by adding : Tetraedriser nom_dom" << finl;
383 Process::exit();
384 }
385 }
386
387 else if (sub_type(Quadri_poly,type_elem_.valeur()))
388 {
389
390 if (!sub_type(Quadrangle_VEF,elem_geom))
391 {
392 Cerr << " The type of the element " << elem_geom.que_suis_je() << " is incorrect" << finl;
393 Process::exit();
394 }
395 }
396 else if (sub_type(Hexa_poly,type_elem_.valeur()))
397 {
398
399 if (!sub_type(Hexaedre_VEF,elem_geom))
400 {
401 Cerr << " The type of the element " << elem_geom.que_suis_je() << " is incorrect" << finl;
402 Process::exit();
403 }
404 }
405
406 int num_face;
407
408
409// On remplit le tableau face_normales_;
410// Attention : le tableau face_voisins n'est pas exactement un
411// tableau distribue. Une face n'a pas ses deux voisins dans le
412// meme ordre sur tous les processeurs qui possedent la face.
413// Donc la normale a la face peut changer de direction d'un
414// processeur a l'autre, y compris pour les faces de joint.
415 {
416 const int n = nb_faces();
417 face_normales_.resize(n, dimension);
418// const Domaine & dom = zone().domaine();
419// Scatter::creer_tableau_distribue(dom, Joint::FACE, face_normales_);
420 creer_tableau_faces(face_normales_);
421 const IntTab& face_som = face_sommets();
422 IntTab& face_vois = face_voisins();
423 const IntTab& elem_face = elem_faces();
424 const int n_tot = nb_faces_tot();
425 for (num_face = 0; num_face < n_tot; num_face++)
426 {
427 type_elem_.normale(num_face,
428 face_normales_,
429 face_som, face_vois, elem_face, zone_geom);
430 }
431
432 DoubleTab old(face_normales_);
433 face_normales_.echange_espace_virtuel();
434 for (num_face = 0; num_face < n_tot; num_face++)
435 {
436 int id=1;
437 for (int d=0; d<dimension; d++)
438 if (!est_egal(old(num_face,d),face_normales_(num_face,d)))
439 {
440 id=0;
441 if (!est_egal(old(num_face,d),-face_normales_(num_face,d)))
442 {
443 Cerr<<"pb in faces_normales" <<finl;
444 Process::exit();
445 }
446 }
447 if (id==0)
448 {
449// on a change le sens de la normale, on inverse elem1 elem2
450 std::swap(face_vois(num_face,0),face_vois(num_face,1));
451 }
452 }
453 }
454
455 /* recalcul de xv pour avoir les vrais CG des faces */
456 const DoubleTab& coords = zone().domaine().coord_sommets();
457 for (int face = 0; dimension == 3 && face < nb_faces(); face++)
458 {
459 double xs[3] = { 0, }, S = 0;
460 for (int k = 0; k < face_sommets_.dimension(1); k++)
461 if (face_sommets_(face, k) >= 0)
462 {
463 int s0 = face_sommets_(face, 0), s1 = face_sommets_(face, k),
464 s2 = k + 1 < face_sommets_.dimension(1) && face_sommets_(face, k + 1) >= 0 ? face_sommets_(face, k +1) : s0;
465 double s = 0;
466 for (int r = 0; r < 3; r++)
467 for (int i = 0; i < 2; i++)
468 s += (i ? -1 : 1) * face_normales_(face, r) * (coords(s2, (r + 1 + i) % 3) - coords(s0, (r + 1 + i) % 3))
469 * (coords(s1, (r + 1 + !i) % 3) - coords(s0, (r + 1 + !i) % 3));
470 for (int r = 0; r < 3; r++) xs[r] += s * (coords(s0, r) + coords(s1, r) + coords(s2, r)) / 3;
471 S += s;
472 }
473 if (S == 0 && sub_type(Hexa_poly,type_elem_.valeur()))
474 {
475 Cerr << "===============================" << finl;
476 Cerr << "Error in your mesh for " << que_suis_je() << "!" << finl;
477 Cerr << "Add this keyword before discretization in your data file to create polyedras:" << finl;
478 Cerr << "Polyedriser " << zone().domaine().le_nom() << finl;
479 Process::exit();
480 }
481 for (int r = 0; r < 3; r++) xv_(face, r) = xs[r] / S;
482 }
483 xv_.echange_espace_virtuel();
484
485 detecter_faces_non_planes();
486
487// volumes_entrelaces_ et volumes_entrelaces_dir : par projection de l'amont/aval sur la normale a la face
488 creer_tableau_faces(volumes_entrelaces_);
489 volumes_entrelaces_dir_.resize(0, 2), creer_tableau_faces(volumes_entrelaces_dir_);
490 for (int f = 0, i, e; f < nb_faces(); f++)
491 for (i = 0; i < 2 && (e = face_voisins_(f, i)) >= 0; volumes_entrelaces_(f) += volumes_entrelaces_dir_(f, i), i++)
492 volumes_entrelaces_dir_(f, i) = std::fabs(dot(&xp_(e, 0), &face_normales_(f, 0), &xv_(f, 0)));
493 volumes_entrelaces_.echange_espace_virtuel(), volumes_entrelaces_dir_.echange_espace_virtuel();
494
495 Zone_VF::calculer_porosites();
496 Zone_VF::calculer_diametres_hydrauliques();
497// calculer_h_carre();
498 /* ordre canonique dans elem_faces_ */
499 std::map<std::array<double, 3>, int> xv_fsa;
500 for (int e = 0, i, j, f; e < nb_elem_tot(); e++)
501 {
502 for (i = 0, j = 0, xv_fsa.clear(); i < elem_faces_.dimension(1) && (f = elem_faces_(e, i)) >= 0; i++)
503 xv_fsa[ {{ xv_(f, 0), xv_(f, 1), dimension < 3 ? 0 : xv_(f, 2) }}] = f;
504 for (auto &&c_f : xv_fsa) elem_faces_(e, j) = c_f.second, j++;
505 }
506}

508void Zone_Poly_base::detecter_faces_non_planes() const
509{
510 const IntTab& f_e = face_voisins(), &f_s = face_sommets_;
511 const DoubleTab& xs = zone().domaine().coord_sommets(), &nf = face_normales_;
512 const DoubleVect& fs = face_surfaces();
513 int i, j, f, s, rk = Process::me(), np = Process::nproc();
514 double sin2;
515 ArrOfDouble val(np); //sur chaque proc : { cos^2 max, indice de face, indices d'elements }
516 IntTab face(np), elem1(np), elem2(np);
517
518// sur chaque proc : on cherche l'angle le plus grand entre un sommet et le plan de sa face
519 for (f = 0; f < nb_faces(); f++)
520 for (i = 0; i < f_s.dimension(1) && (s = f_s(f, i)) >= 0; i++)
521 if ((sin2 = std::pow(dot(&xs(s, 0), &nf(f, 0), &xv_(f, 0)) / fs(f), 2) / dot(&xs(s, 0), &xs(s, 0), &xv_(f, 0), &xv_(f, 0))) > val[rk])
522 val[rk] = sin2, face(rk) = f, elem1(rk) = f_e(f, 0), elem2(rk) = f_e(f, 1);
523 envoyer_all_to_all(val, val), envoyer_all_to_all(face, face), envoyer_all_to_all(elem1, elem1), envoyer_all_to_all(elem2, elem2);
524
525 for (i = j = sin2 = 0; i < Process::nproc(); i++)
526 if (val[i] > sin2) sin2 = val[i], j = i;
527 double theta = asin(sqrt(sin2)) * 180 / M_PI;
528 Cerr << "Zone_Poly_base : angle sommet/face max " << theta << " deg (proc " << j << " , face ";
529 Cerr << face(j) << " , elems " << elem1(j) << " / " << elem2(j) << " )" << finl;
530 if (theta > 1) Cerr << "Zone_Poly_base : non-planar face detected ! Please fix your mesh or call 911 ..." << finl, Process::exit();
531}

533void Zone_Poly_base::calculer_h_carre()
534{
535// Calcul de h_carre
536 h_carre = 1.e30;
537 h_carre_.resize(nb_faces());
538// Calcul des surfaces
539 const DoubleVect& surfaces=face_surfaces();
540 const int nb_faces_elem=zone().nb_faces_elem();
541 const int nbe=nb_elem();
542 for (int num_elem=0; num_elem<nbe; num_elem++)
543 {
544 double surf_max = 0;
545 for (int i=0; i<nb_faces_elem; i++)
546 {
547 double surf = surfaces(elem_faces(num_elem,i));
548 surf_max = (surf > surf_max)? surf : surf_max;
549 }
550 double vol = volumes(num_elem)/surf_max;
551 vol *= vol;
552 h_carre_(num_elem) = vol;
553 h_carre = ( vol < h_carre )? vol : h_carre;
554 }
