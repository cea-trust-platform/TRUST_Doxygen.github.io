30enum class Ordre_RK { UN , RATIO_DEUX , DEUX_CLASSIQUE , DEUX_WILLIAMSON , TROIS_CLASSIQUE , TROIS_WILLIAMSON , QUATRE_CLASSIQUE , QUATRE_CLASSIQUE_3_8 , QUATRE_WILLIAMSON };

32template <Ordre_RK _ORDRE_ >
33class TRUSTSchema_RK : public Schema_Temps_base
34{
35// Renvoie le nombre de valeurs temporelles a conserver. Ici : n et n+1, donc 2.
36 int nb_valeurs_temporelles() const override { return 2 ; }
37
38// Renvoie le nombre de valeurs temporelles futures. Ici : n+1, donc 1.
39 int nb_valeurs_futures() const override { return 1 ; }
40
41// Renvoie le le temps a la i-eme valeur future. Ici : t(n+1)
42 double temps_futur(int i) const override
43 {
44 assert(i == 1);
45 return temps_courant() + pas_de_temps();
46 }
47
48// Renvoie le le temps le temps que doivent rendre les champs a l'appel de valeurs(). Ici : t(n+1)
49 double temps_defaut() const override { return temps_courant() + pas_de_temps(); }
50
51// a surcharger si utile
52 void completer() override { /* Do nothing */ }

81 static constexpr std::array<ARR2, 2> BUTCHER_3 = { { { 1. / 2., 0. }, { -1., 2. } } }; /* RK3 */
82 static constexpr std::array<ARR3, 3> BUTCHER_4 = { { { 1. / 2., 0. , 0.}, { 0., 1. / 2., 0.}, { 0. , 0., 1. } } }; /* RK4 */
83 static constexpr std::array<ARR3, 3> BUTCHER_4_3_8 = { { { 1. / 3., 0. , 0.}, { -1. / 3., 1., 0.}, { 1. , -1., 1. } } }; /* RK4 3/8 rule */
84
85 static constexpr std::array<ARR4, 4> BUTCHER_TAB = { {

87 { 1. / 6., 2. / 3., 1. / 6., 0. }, /* RK3 classique*/
88 { 1. / 6., 1. / 3., 1. / 3., 1. / 6. }, /* RK4 classique*/
