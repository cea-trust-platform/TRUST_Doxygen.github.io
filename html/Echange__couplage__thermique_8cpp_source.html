114void Echange_couplage_thermique::get_dRho_dCp(int i, double& d_rho, double& d_Cp) const
115{
116 const Milieu_base& mil=ma_zone_cl_dis->equation().milieu();
117 Nom nom_pb=ma_zone_cl_dis->equation().probleme().que_suis_je();
118 const Champ_base& rho=mil.masse_volumique().valeur();
119 if ((nom_pb.debute_par("Probleme_Interface")|| nom_pb==Nom("Probleme_Thermo_Front_Tracking"))||!sub_type(Champ_Uniforme,rho)
120 ||nom_pb==Nom("Pb_Conduction") || nom_pb.debute_par("Pb_Conduction_Combustible"))
121 {
122// Pour le front tracking, QC, et conduc_variable, on ne divise pas par Rho*Cp:
123 d_rho=1.;
124 d_Cp=1.;
125 }
126 else
127 {
128 const Champ_Don& Cp =mil.capacite_calorifique();
129
130 if (sub_type(Champ_Uniforme,rho))
131 {
132 d_rho= rho(0,0);
133 }
134 else
135 {
136 d_rho= rho(i);
137 }
138 if (sub_type(Champ_Uniforme,Cp.valeur()))
139 {
140 d_Cp= Cp(0,0);
141 }
142 else
143 {
144 d_Cp= Cp.valeur()(i);
145 }
146 }
147}

149double Echange_couplage_thermique::champ_exterieur(int i, const Champ_front& champ_ext) const
150{
151 if (ma_zone_cl_dis->equation().que_suis_je()!="Convection_Diffusion_Concentration")
152 {
153 double d_rho, d_Cp;
154 get_dRho_dCp(i, d_rho, d_Cp);
155
156 if (champ_ext.valeurs().size()==1)
157 return champ_ext(0,0)/(d_rho*d_Cp);
158 else if (champ_ext.valeurs().dimension(1)==1)
159 return champ_ext(i,0)/(d_rho*d_Cp);
160 else
161 Cerr << "Echange_couplage_thermique::flux_impose erreur" << finl;
162 exit();
163 return 0.;
164 }
165 else
166 {
167 if (champ_ext.valeurs().size()==1)
168 return champ_ext(0,0);
169 else if (champ_ext.valeurs().dimension(1)==1)
170 return champ_ext(i,0);
171 else
172 Cerr << "Echange_couplage_thermique::flux_impose erreur" << finl;
173 exit();
174 return 0.;
175
176 }
177}

179double Echange_couplage_thermique::champ_exterieur(int i,int j, const Champ_front& champ_ext) const
180{
181 if (ma_zone_cl_dis->equation().que_suis_je()!="Convection_Diffusion_Concentration")
182 {
183 double d_rho, d_Cp;
184 get_dRho_dCp(i, d_rho, d_Cp);
185 if (champ_ext.valeurs().dimension(0)==1)
186 return champ_ext(0,j)/(d_rho*d_Cp);
187 else
188 return champ_ext(i,j)/(d_rho*d_Cp);
189 }
190 else
191 {
192 if (champ_ext.valeurs().dimension(0)==1)
193 return champ_ext(0,j);
194 else
195 return champ_ext(i,j);
196 }
197}
