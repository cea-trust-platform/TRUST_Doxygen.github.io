27// Description :
28// Cherche dans le probleme l'equation scalaire
29void Terme_Boussinesq_base::associer_pb(const Probleme_base& pb)
30{
31 int ok=0;
32 int n_eq=pb.nombre_d_equations();
33 for (int eq=0; eq<n_eq; eq++)
34 {
35 const Equation_base& eqn=pb.equation(eq);
36 if (eqn.inconnue().le_nom()==NomScalaire_)
37 {
38 ok=1;
39 equation_scalaire_ = ref_cast(Convection_Diffusion_std,eqn);
40 }
41 }
42 if (ok==0)
43 {
44 Cerr << "Error. The Boussinesq source term can't be defined on a problem of kind " << pb.que_suis_je() << finl;
45 Process::exit();
46 }
47
48// Fill la_gravite_ and beta_ attributes
49// Elie Saikali : compat with pb_multiphase
50 const Fluide_base& fluide = equation().milieu().que_suis_je() == "Milieu_composite" ? ref_cast(Milieu_composite,equation().milieu()).get_medium_for_incompressible() :
51 ref_cast(Fluide_base, equation().milieu());
52
53 la_gravite_ = fluide.gravite();
54
55// this variable indicates if the beta field is valid or not
56 int valid_beta_field = 0; //!< by default it is invalid
57 Nom beta_field_name="??"; //!< only used for the error message
58
59 if (NomScalaire_=="temperature")
60 {
61 valid_beta_field = fluide.beta_t( ).non_nul( );
62 beta_field_name = "thermal expansion value (beta_th)";
63 beta_=fluide.beta_t();
64 }
65 else if (NomScalaire_=="concentration")
66 {
67 valid_beta_field = fluide.beta_c( ).non_nul( );
68 beta_field_name = "volume expansion coefficient values in concentration (beta_co)";
69 beta_=fluide.beta_c();
70 }
71
72 if( ! valid_beta_field )
73 {
74 Cerr << "Error. Boussinesq source term is not able to access to the "<<beta_field_name<<" associated to the fluid."<<finl;
75 Cerr << "Please check your data file. " << finl;
76 Cerr << "Aborting..."<<finl;
77 Process::abort( );
78 }
79
80}

96//! Lit les parametres du terme source a partir d'un flot d'entree.
97Entree& Terme_Boussinesq_base::lire_donnees(Entree& is)
98{
99 Cerr << "Reading Boussinesq source term parameters." << finl;
100 Motcle accolade_ouverte("{");
101 Motcle accolade_fermee("}");
102 Motcle motlu;
103 verif_=1;
104
105 Motcles les_mots(3);
106 {
107 les_mots[0]="T0";
108 les_mots[1]="C0";
109 les_mots[2]="verif_boussinesq";
110 }
111
112 is >> motlu;
113 if (motlu != accolade_ouverte)
114 {
115 Cerr << "Error: { is waited instead of " << motlu << finl;
116 Process::exit();
117 }
118 is >> motlu;
119
120 while (motlu!= accolade_fermee)
121 {
122 int rang = les_mots.search(motlu);
123 switch(rang)
124 {
125 case 0 :
126 {
127// Temperature
128 NomScalaire_="temperature";
129 fct_Scalaire0_.dimensionner(1);
130 Scalaire0_.resize(1);
131 read(is, fct_Scalaire0_[0]);
132 break;
133 }
134 case 1 :
135 {
136// Concentration
137 NomScalaire_="concentration";
138 int dim;
139 is >> dim;
140 fct_Scalaire0_.dimensionner(dim);
141 Scalaire0_.resize(dim);
142 for (int i=0; i<dim; i++)
143 read(is, fct_Scalaire0_[i]);
144 break;
145 }
146 case 2 :
147 {
148 is >> verif_;
149 break;
150 }
151 default :
152 {
153 Cerr << "Error: " << motlu << " is not recognized." << finl;
154 Cerr << "Possible keywords:" << finl;
155 Cerr << les_mots << finl;
156 Process::exit();
157 }
158
159 }
160 is >> motlu;
161 }
162 return is;
163}
