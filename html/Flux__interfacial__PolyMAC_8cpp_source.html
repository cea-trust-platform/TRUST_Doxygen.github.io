44void Flux_interfacial_PolyMAC::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
45{
46 const Champ_Elem_PolyMAC& ch = ref_cast(Champ_Elem_PolyMAC, equation().inconnue().valeur());
47 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, equation().zone_dis().valeur());
48 const DoubleTab& inco = ch.valeurs();
49
50 /* on doit pouvoir ajouter / soustraire les equations entre composantes */
51 int i, j, e, n, N = inco.line_size();
52 if (N == 1) return;
53 std::set<int> idx;
54 for (auto &&n_m : matrices)
55 if (n_m.first.find("/") == std::string::npos) /* pour ignorer les inconnues venant d'autres problemes */
56 {
57 Matrice_Morse& mat = *n_m.second, mat2;
58 const DoubleTab& dep = equation().probleme().get_champ(n_m.first).valeurs();
59 int m, nc = dep.dimension_tot(0), M = dep.line_size();
60 IntTrav sten(0, 2);
61 sten.set_smart_resize(1);
62 if (n_m.first == "temperature" || n_m.first == "pression" || n_m.first == "alpha" || n_m.first == "interfacial_area" ) /* temperature/pression: dependance locale */
63 for (e = 0; e < zone.nb_elem(); e++)
64 for (n = 0; n < N; n++)
65 for (m = 0; m < M; m++) sten.append_line(N * e + n, M * e + m);
66 else if (mat.nb_colonnes())
67 for (e = 0; e < zone.nb_elem(); e++) /* autres variables: on peut melanger les composantes*/
68 {
69 for (idx.clear(), n = 0, i = N * e; n < N; n++, i++)
70 for (j = mat.get_tab1()(i) - 1; j < mat.get_tab1()(i + 1) - 1; j++)
71 idx.insert(mat.get_tab2()(j) - 1); //idx : ensemble des colonnes dont depend au moins une ligne des N composantes en e
72 for (n = 0, i = N * e; n < N; n++, i++)
73 for (auto &&x : idx) sten.append_line(i, x); //ajout de cette depedance a toutes les lignes
74 }
75 else continue;
76 Matrix_tools::allocate_morse_matrix(N * zone.nb_elem_tot(), M * nc, sten, mat2);
77 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
78 }

129// Matrice_Morse *mat = matrices.count(ch.le_nom().getString()) ? matrices.at(ch.le_nom().getString()) : NULL;
130 const Milieu_composite& milc = ref_cast(Milieu_composite, equation().milieu());
131 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, equation().zone_dis().valeur());
132 const DoubleVect& pe = zone.porosite_elem(), &ve = zone.volumes();
133 const tabs_t& der_h = ref_cast(Champ_Inc_base, milc.enthalpie()).derivees();
134 const Champ_base& ch_rho = milc.masse_volumique();

137 *pch_rho = sub_type(Champ_Inc_base, ch_rho) ? &ref_cast(Champ_Inc_base, ch_rho) : NULL;
138 const DoubleTab& inco = ch.valeurs(), &alpha = ch_alpha.valeurs(), &press = ch_p.valeurs(), &temp = ch_temp.valeurs(), &pvit = ch_vit.passe(),
139 &h = milc.enthalpie().valeurs(), *dP_h = der_h.count("pression") ? &der_h.at("pression") : NULL, *dT_h = der_h.count("temperature") ? &der_h.at("temperature") : NULL,

143 *Mt = matrices.count("temperature") ? matrices.at("temperature") : NULL,
144 *Ma = matrices.count("alpha") ? matrices.at("alpha") : NULL,

147 int i, j, col, e, d, D = dimension, k, l, n, N = inco.line_size(), nf_tot = zone.nb_faces_tot(),
148 cL = lambda.dimension_tot(0) == 1, cM = mu.dimension_tot(0) == 1, cR = rho.dimension_tot(0) == 1, cCp = Cp.dimension_tot(0) == 1, is_therm;
149 const Flux_interfacial_base& correlation_fi = ref_cast(Flux_interfacial_base, correlation_.valeur().valeur());
150 const Changement_phase_base *correlation_G = pbm.has_correlation("changement_phase") ? &ref_cast(Changement_phase_base, pbm.get_correlation("changement_phase").valeur()) : NULL;
151 double dt = equation().schema_temps().pas_de_temps(), alpha_min = 1.e-6;
152
153 /* limiteur de changement de phase : on limite gamma pour eviter d'avoir alpha_k < 0 dans une phase */
154 /* pour cela, on assemble l'equation de masse sans changement de phase */
155 DoubleTrav sec_m(alpha); //residus
156 std::map<std::string, Matrice_Morse> mat_m_stockees;
157 matrices_t mat_m; //derivees
158 for (auto &&n_m : matrices)
159 if (n_m.first.find("/") == std::string::npos) /* pour ignorer les inconnues venant d'autres problemes */
160 {
161 Matrice_Morse& dst = mat_m_stockees[n_m.first], &src = *n_m.second;
162 dst.get_set_tab1().ref_array(src.get_set_tab1()); //!< memes stencils que celui de l'equation courante
163 dst.get_set_tab2().ref_array(src.get_set_tab2());
164 dst.set_nb_columns(src.nb_colonnes());
165 dst.get_set_coeff().resize(src.get_set_coeff().size()); //coeffs nuls
166 mat_m[n_m.first] = &dst;
167 }
168 const Masse_Multiphase& eq_m = pbm.eq_masse;
169 for (i = 0; i < eq_m.nombre_d_operateurs(); i++) /* tous les operateurs */
170 eq_m.operateur(i).l_op_base().ajouter_blocs(mat_m, sec_m, semi_impl);
171 for (i = 0; i < eq_m.sources().size(); i++)
172 if (!sub_type(Flux_interfacial_PolyMAC, eq_m.sources()(i).valeur())) /* toutes les sources sauf le flux interfacial */
173 eq_m.sources()(i).valeur().ajouter_blocs(mat_m, sec_m, semi_impl);
174 std::vector<std::array<Matrice_Morse *, 2>> vec_m; //vecteur "matrice source, matrice de destination"
175 for (auto &&n_m : matrices)
176 if (n_m.first.find("/") == std::string::npos && mat_m[n_m.first]->get_tab1().size() > 1) vec_m.push_back({{ mat_m[n_m.first], n_m.second }});

178 /* elements */
179// coefficients et plein de derivees...
180 DoubleTrav hi(N, N), dT_hi(N, N, N), da_hi(N, N, N), dP_hi(N, N), dT_phi(N), da_phi(N), dT_G(N), da_G(N), nv(N, N);
181 for (e = 0; e < zone.nb_elem(); e++)
182 {
183// double vol = pe(e) * ve(e), x, G = 0, dv_min = 0.1, dh = zone.diametre_hydraulique_elem()(e, 0), dP_G = 0.; // E. Saikali : initialise dP_G ici sinon fuite memoire ...
184// PL ToDo: Uniformiser diametre_hydraulique_elem (vecteur) pour C3D et F5
185 double vol = pe(e) * ve(e), x, G = 0, dv_min = 0.1, dh = zone.diametre_hydraulique_elem()(e), dP_G = 0.; //!< E. Saikali : initialise dP_G ici sinon fuite memoire ...
186 for (nv = 0, d = 0; d < D; d++)
187 for (n = 0; n < N; n++)
188 for (k = 0 ; k<N ; k++) nv(n, k) += std::pow( pvit(nf_tot + D * e + d, n) - ((n!=k) ? pvit(nf_tot + D * e + d, k) : 0) , 2); //!< nv(n,n) = ||v(n)||, nv(n, k!=n) = ||v(n)-v(k)||
189 for (n = 0; n < N; n++)
190 for (k = 0 ; k<N ; k++) nv(n, k) = std::max(sqrt(nv(n, k)), dv_min);
191// coeffs d'echange vers l'interface (explicites)
192 correlation_fi.coeffs(dh, &alpha(e, 0), &temp(e, 0), press(e, 0), &nv(0, 0),
193 &lambda(!cL * e, 0), &mu(!cM * e, 0), &rho(!cR * e, 0), &Cp(!cCp * e, 0), e, hi, dT_hi, da_hi, dP_hi);
194
195 for (k = 0; k < N; k++)
196 for (l = k + 1; l < N; l++)
197 if (milc.has_saturation(k, l)) //flux phase k <-> phase l si saturation
198 {
199 Saturation_base& sat = milc.get_saturation(k, l);
200 if (correlation_G) /* taux de changement de phase par une correlation */
201 G = correlation_G->calculer(k, l, dh, &alpha(e, 0), &temp(e, 0), press(e, 0), &nv(0), &lambda(!cL * e, 0), &mu(!cM * e, 0), &rho(!cM * e, 0), &Cp(!cCp * e, 0),
202 sat, dT_G, da_G, dP_G);
203 /* limite thermique */
204 double Tk = temp(e, k), Tl = temp(e, l), p = press(e, 0), Ts = sat.Tsat(p), dP_Ts = sat.dP_Tsat(p), //temperature de chaque cote + Tsat + derivees
205 phi = hi(k, l) * (Tk - Ts) + hi(l, k) * (Tl - Ts) + qi(e, k, l) / vol, L = (phi < 0 ? h(e, l) : sat.Hvs(p)) - (phi > 0 ? h(e, k) : sat.Hls(p));
206 if ((is_therm = !correlation_G || std::fabs(G) > std::fabs(phi / L))) G = phi / L;
207 /* enthalpies des phases (dans le sens correspondant au mode choisi pour G) */
208
209 /* G est-il limite par l'evanescence cote k ou l ? */
210 int n_lim = G > 0 ? k : l, sgn = G > 0 ? 1 : -1; //phase sortante
211 double Glim = sec_m(e, n_lim) / vol + p_ar(e, n_lim) / dt; //changement de phase max acceptable par cette phase
212 if (std::fabs(G) < Glim) n_lim = -1; //G ne rend pas la phase evanescente -> pas de limitation (n_lim = -2)
213 else G = (G > 0 ? 1 : -1) * Glim;//la phase serait evanescente a cause de G -> on le bloque a G_lim (n_lim = k / l)
214
215 double hk = G > 0 ? h(e, k) : sat.Hls(p), dTk_hk = G > 0 && dT_h ? (*dT_h)(e, k) : 0, dP_hk = G > 0 ? (dP_h ? (*dP_h)(e, k) : 0) : sat.dP_Hls(p),
216 hl = G < 0 ? h(e, l) : sat.Hvs(p), dTl_hl = G < 0 && dT_h ? (*dT_h)(e, l) : 0, dP_hl = G < 0 ? (dP_h ? (*dP_h)(e, l) : 0) : sat.dP_Hvs(p);

218 {
219 double dP_phi = dP_hi(k, l) * (Tk - Ts) + dP_hi(l, k) * (Tl - Ts) - (hi(k, l) + hi(l, k)) * dP_Ts + dpqi(e, k, l) / vol,
220 dTk_L = -dTk_hk, dTl_L = dTl_hl, dP_L = dP_hl - dP_hk;
221 dP_G = (dP_phi - G * dP_L) / L;
222 for (n = 0; n < N; n++) dT_G(n) = ((n == k) * (hi(k, l) - G * dTk_L) + (n == l) * (hi(l, k) - G * dTl_L) + dT_hi(k, l, n) * (Tk - Ts) + dT_hi(l, k, n) * (Tl - Ts) + dTqi(e, k, l, n)) / L;
223 for (n = 0; n < N; n++) da_G(n) = (da_hi(k, l, n) * (Tk - Ts) + da_hi(l, k, n) * (Tl - Ts) + daqi(e, k, l, n)) / L;
224 }
225
226
227 if (sub_type(Masse_Multiphase, equation())) //eq de masse -> changement de phase
228 {
229 for (i = 0; i < 2; i++) secmem(e, i ? l : k) -= vol * (i ? -1 : 1) * G;
230 if (n_lim < 0) /* G par limite thermique */
231 {
232 if (Ma)
233 for (i = 0; i < 2; i++)
234 for (n = 0; n < N; n++) //derivees en alpha
235 (*Ma)(N * e + (i ? l : k), N * e + n) += vol * (i ? -1 : 1) * da_G(n);
236 if (Mt)
237 for (i = 0; i < 2; i++)

240 if (Mp)
241 for (i = 0; i < 2; i++) //derivees en p

243 }
244 else for (auto &s_d : vec_m) /* G par evanescence */
245 for (j = s_d[0]->get_tab1()(N * e + n_lim) - 1; j < s_d[0]->get_tab1()(N * e + n_lim + 1) - 1; j++)

264 if (Mp)
265 for (i = 0; i < 2; i++) //derivees en p
266 (*Mp)(N * e + (i ? l : k), e) += vol * (i ? -1 : 1) * (s_c * (dP_hi(n_c, n_d) * (Tc - Ts) - hi(n_c, n_d) * dP_Ts) + (n_lim < 0) * dP_G * hc + G * dP_hc) - (i != c) * dpqi(e, k, l);
267 if (n_lim >= 0)
268 for (auto &s_d : vec_m) /* derivees de G dans le cas evanescent */
269 for (j = s_d[0]->get_tab1()(N * e + n_lim) - 1; j < s_d[0]->get_tab1()(N * e + n_lim + 1) - 1; j++)
270 for (col = s_d[0]->get_tab2()(j) - 1, x = -s_d[0]->get_coeff()(j), i = 0; i < 2; i++)
271 (*s_d[1])(N * e + (i ? l : k), col) += (i ? -1 : 1) * hc * sgn * x;
272 }
273 else if (sub_type(Aire_interfaciale, equation())) //eq d'aire interfaciale ; looks like the mass equation ; not a conservation equation !
274 if (k==0) //!< k est la phase porteuse
275 if (alpha(e, l) > alpha_min) //!< if the phase l is present
276 {
277 secmem(e, l) += vol * 2./3. * inco(e, l) / (alpha(e, l) * (pch_rho ? (*pch_rho).valeurs()(e, l) : rho(e, l))) * G ;
278 if (n_lim < 0) /* G par limite thermique */
279 {
280 if (Ma) //derivees en alpha
281 {
282 (*Ma)(N * e + l , N * e + l) -= vol * 2./3. * inco(e, l) / (-alpha(e, l)*alpha(e, l) * (pch_rho ? (*pch_rho).valeurs()(e, l) : rho(e, l))) * G;
283 for (n = 0; n < N; n++) (*Ma)(N * e + l , N * e + n) -=
284 vol * 2./3. * inco(e, l) / (alpha(e, l) * (pch_rho ? (*pch_rho).valeurs()(e, l) : rho(e, l))) * da_G(n);
285 }
286 if (Mt) //derivees en T
287 {
288 if (pch_rho) (*Mt)(N * e + l , N * e + l) -=
289 vol * 2./3. * inco(e, l) / alpha(e, l)*-pch_rho->derivees().at("temperature")(e, l)/((*pch_rho).valeurs()(e, l)*(*pch_rho).valeurs()(e, l)) * G;

291 vol * 2./3. * inco(e, l) / (alpha(e, l) * (pch_rho ? (*pch_rho).valeurs()(e, l) : rho(e, l))) * dT_G(n);
292 }
293 if (Mp) //derivees en p
294 {
