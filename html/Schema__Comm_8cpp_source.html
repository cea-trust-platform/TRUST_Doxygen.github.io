171static void exchange_data(const ArrOfInt& send_list,
172 const ArrOfInt& send_size,
173 const char * const * const send_buffers,
174 const ArrOfInt& recv_list,
175 const ArrOfInt& recv_size,
176 char * const * const recv_buffers,
177 const Comm_Group& group,
178 bool use_all_to_all)
179{
180// GF je rajoute le ifdef pour les versions sans MPI comm_group_mpi n'existep pas
181#ifdef MPI_
182 if (!use_all_to_all || !sub_type(Comm_Group_MPI, group))

186 if (1)
187#endif
188 {
189 group.send_recv_start(send_list, send_size, send_buffers,
190 recv_list, recv_size, recv_buffers);
191 group.send_recv_finish();
192 }
193 else
194 {
195#ifdef MPI_
196 const int n = group.nproc();
197 const int nsend = send_list.size_array();
198 const int nrecv = recv_list.size_array();
199// Pack all send data in a single buffer:
200 ArrOfInt a_send_size(n);
201 ArrOfInt a_send_offset(n);
202 ArrOfInt a_recv_size(n);
203 ArrOfInt a_recv_offset(n);
204 int i, offset;
205// Compute send_size array
206 for (i = 0; i < nsend; i++)
207 a_send_size[send_list[i]] = send_size[i];
208// Compute send_offset
209 for (i = 0, offset = 0; i < n; i++)
210 {
211 a_send_offset[i] = offset;
212 offset += a_send_size[i];
213 }
214 const int buf_size_send = offset;
215// Compute recv_size array
216 for (i = 0; i < nrecv; i++)
217 a_recv_size[recv_list[i]] = recv_size[i];
218// Compute recv_offset
219 for (i = 0, offset = 0; i < n; i++)
220 {
221 a_recv_offset[i] = offset;
222 offset += a_recv_size[i];
223 }
224 const int buf_size_recv = offset;
225// Allocate contiguous send and recv buffer:
226 char *send_buffer = (char *) malloc(buf_size_send);
227 char *recv_buffer = (char *) malloc(buf_size_recv);
228// Copy send data to send buffer
229 for (i = 0; i < nsend; i++)
230 memcpy(send_buffer + a_send_offset[send_list[i]], //!< dest
231 send_buffers[i], //!< source
232 send_size[i]); //!< size Exchange data
233//
234 ref_cast(Comm_Group_MPI, group).all_to_allv(send_buffer, a_send_size.addr(), a_send_offset.addr(),
235 recv_buffer, a_recv_size.addr(), a_recv_offset.addr());
236// Copy recv data to recv_buffers
237 for (i = 0; i < nrecv; i++)
238 memcpy(recv_buffers[i], //!< dest
239 recv_buffer + a_recv_offset[recv_list[i]], //!< source
240 recv_size[i]); //!< size [ABN] arrruhhhhhmmmmmm:

242 free(send_buffer);

252//! Transmet la taille des messages a envoyer aux processeurs qui<br>vont les recevoir. La taille est le nombre de bytes des obuffers.<br>send_pe_list et recv_pe_list doivent etre initialises.<br>Le schema doit etre dans l'etat WRITING.
253void Schema_Comm::echange_taille(const ArrOfInt& send_size,
254 ArrOfInt& recv_size) const
255{
256 static ArrOfInt send_sz;
257 static ArrOfInt recv_sz;
258 send_sz.set_smart_resize(1);
259 recv_sz.set_smart_resize(1);
260
261 assert(status_ == WRITING && ref_group_.non_nul());
262 const Comm_Group& group = ref_group_.valeur();
263 assert(&group == &PE_Groups::current_group());
264
265// On verifie que tous les membres du groupe executent ceci.
266// Si ca plante ici, c'est que tous les membres declares ne sont
267// pas en train de faire la barriere.
268 if (group.check_enabled()) group.barrier(ECHANGE_MESSAGES_COMM_TAG);
269
270 const int n_send = send_pe_list_.size_array();
271 const int n_recv = recv_pe_list_.size_array();
272
273 recv_size.resize_array(n_recv);
274
275 const char ** send_buffers = new const char* [n_send];
276 char ** recv_buffers = new char* [n_recv];
277
278 send_sz.resize_array(n_send); //!< Taille d'un int (on echange une taille)
279 send_sz = sizeof(int);
280 int i;
281 for (i = 0; i < n_send; i++)
282 send_buffers[i] = (char*) (& send_size[i]);
283
284 recv_sz.resize_array(n_recv);
285 recv_sz = sizeof(int);
286 recv_size.resize_array(n_recv);
287 for (i = 0; i < n_recv; i++)
288 recv_buffers[i] = (char*) (& recv_size[i]);
289
290 exchange_data(send_pe_list_, send_sz, send_buffers,
291 recv_pe_list_, recv_sz, recv_buffers,
292 group,
293 use_all_to_allv_);
294
295 delete[] recv_buffers;
296 delete[] send_buffers;
297}

302//! Cette methode lance l'echange de donnees entre tous les processeurs.<br>La taille des messages recus doit etre deja connue.<br>Le schema passe de WRITING a EXCHANGED.
303void Schema_Comm::echange_messages(const ArrOfInt& send_size,
304 const ArrOfInt& recv_size) const
305{
306 assert(status_ == WRITING && ref_group_.non_nul());
307 const Comm_Group& group = ref_group_.valeur();
308 assert(&group == &PE_Groups::current_group());
309
310// On verifie que tous les membres du groupe executent ceci.
311// Si ca plante ici, c'est que tous les membres declares ne sont
312// pas en train de faire la barriere.
313 if (group.check_enabled()) group.barrier(ECHANGE_MESSAGES_COMM_TAG);
314
315 const int n_send = send_pe_list_.size_array();
316 const int n_recv = recv_pe_list_.size_array();
317 const char ** send_buffers = new const char* [n_send];
318 char ** recv_buffers = new char* [n_recv];
319
320 int i;
321 for (i = 0; i < n_recv; i++)
322 {
323 int pe = recv_pe_list_[i];
324 int size = recv_size[i];
325 InputCommBuffer& buf = ebuffer(pe);
326 recv_buffers[i] = buf.reserve_buffer(size);
327 }
328 for (i = 0; i < n_send; i++)
329 {
330 int pe = send_pe_list_[i];
331 OutputCommBuffer& buf = obuffer(pe);
332// On verifie que la taille des messages en emission est la meme que
333// celle enregistree dans send_size_, ce qui garantit que la taille
334// en reception est juste elle-aussi.
335 assert(send_size[i] == buf.get_buffer_size());
336 send_buffers[i] = buf.get_buffer();
337 }
338
339 exchange_data(send_pe_list_, send_size, send_buffers,
340 recv_pe_list_, recv_size, recv_buffers,
341 group,
342 use_all_to_allv_);
343
344 delete[] recv_buffers;
345 delete[] send_buffers;
346
347// Creation des input streams a partir des buffers recus
348 for (i = 0; i < n_recv; i++)
349 {
350 int pe = recv_pe_list_[i];
351 InputCommBuffer& buf = ebuffer(pe);
352 buf.create_stream();
353 }
354
355// Cas particulier des messages envoyes a moi-meme:
356 if (me_to_me_)
357 {
358 int pe = Process::me();
359 InputCommBuffer& buf = ebuffer(pe);
360 OutputCommBuffer& obuf = obuffer(pe);
361 buf.create_stream_from_output_stream(obuf);
362 }
363
364 status_ = EXCHANGED;
365}

421//! Vide les buffers et libere les ressources: on<br>a fini de lire les donnees recues dans les buffers.<br>Le schema passe de EXCHANGED a RESET
422void Schema_Comm::end_comm() const
423{
424 assert(status_ == EXCHANGED && ref_group_.non_nul());
425 const Comm_Group& group = ref_group_.valeur();
426 assert(&group == &PE_Groups::current_group());
427
428// On verifie que tous les membres du groupe executent ceci.
429// Si ca plante ici, c'est que tous les membres declares ne sont
430// pas en train de faire la barriere.
431 if (group.check_enabled()) group.barrier(END_COMM_TAG);
432
433 int i, n;
434 n = send_pe_list_.size_array();
435 for (i = 0; i < n; i++)
436 {
437 int pe = send_pe_list_[i];
438 obuffer(pe).clear();
439 }
440 n = recv_pe_list_.size_array();
441 for (i = 0; i < n; i++)
442 {
443 int pe = recv_pe_list_[i];
444 ebuffer(pe).clear();
445 }
446
447 if (me_to_me_)
448 {
449 int pe = Process::me();
450 obuffer(pe).clear();
451 ebuffer(pe).clear();
452 }
453
454 status_ = RESET;
455}

537// * Statut == RESET
538// * Cette fonction doit etre appelee simultanement sur tous les membres du groupe.
539//
540//! Verifie que les send/recv_pe_list verifient la propriete<br>"tu m'ecoutes quand je te parle"
541void Schema_Comm::check_send_recv_pe_list() const
542{
543 assert(status_ == RESET);
544// On verifie que les indices de processeurs sont dans le groupe
545 int fail1 = 0;
546 const int np = Process::nproc();
547 const int n1 = send_pe_list_.size_array();
548 int i;
549 for (i = 0; i < n1; i++)
550 if (send_pe_list_[i] < 0 || send_pe_list_[i] >= np)
551 fail1 = 1;
552 const int n2 = recv_pe_list_.size_array();
553 for (i = 0; i < n2; i++)
554 if (recv_pe_list_[i] < 0 || recv_pe_list_[i] >= np)
555 fail1 = 1;
556 int fail2 = 0;
557 ArrOfInt recv_list;
558 if (!fail1)
559 {
560 reverse_send_recv_pe_list(send_pe_list_, recv_list);
561// Le tableau recv_pe_list_ n'est pas forcement trie alors que recv_list l'est toujours
562// On trie avant de comparer
563 ArrOfInt copie(recv_pe_list_);
564 copie.ordonne_array();
565 fail2 = !(recv_list == copie);
566 }
567 if (Process::mp_sum(fail1+fail2))
568 {
569 if (Process::je_suis_maitre())
570 Cerr << "Error in Schema_Comm::check_send_recv_pe_list(), see .log files" << finl;
571 Process::Journal() << "Error in Schema_Comm::check_send_recv_pe_list() :\n"
572 << "send_list:\n" << send_pe_list_
573 << "recv_list:\n" << recv_pe_list_;
574 if (fail1)
575 Process::Journal() << "processor ranks not in current group: current group size = " << np;
576 else if (fail2)
577 Process::Journal() << "recv_list should be this one:\n" << recv_list << finl;
578 else
579 Process::Journal() << "OK on this processor" << finl;
580 Process::barrier();
581 Process::exit();
582 }
