169//! Associe un milieu au probleme,<br>Si le milieu est de type<br>- Fluide_Incompressible, il sera associe a l'equation de l'hydraulique<br>- Constituant, il sera associe a l'equation de convection-diffusion<br>Un autre type de milieu provoque une erreur
170void Pb_Hydraulique_Concentration::associer_milieu_base(const Milieu_base& mil)
171{
172 if ( sub_type(Fluide_Incompressible,mil) )
173 eq_hydraulique.associer_milieu_base(mil);
174 else if ( sub_type(Constituant,mil) )
175 eq_concentration.associer_milieu_base(mil);
176 else
177 {
178 Cerr << "Un milieu de type " << mil.que_suis_je() << " ne peut etre associe a " << finl;
179 Cerr << "un probleme de type Pb_Hydraulique_Concentration " << finl;
180 exit();
181 }
182}
