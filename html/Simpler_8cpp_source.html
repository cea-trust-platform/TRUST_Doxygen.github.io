41// DEscription:
42// calcul D correction_en_vitesse= E current + resu )
43// avec D diagonale de la matrice et E = D-matrice =-(matrice-D)
44int inverser_par_diagonale(const Matrice_Morse& matrice,const DoubleTrav& resu,const DoubleTab& current,DoubleTrav& correction_en_vitesse)
45{
46 const IntVect& tab1 = matrice.get_tab1();
47 const IntVect& tab2 = matrice.get_tab2();
48 const DoubleVect& coeff = matrice.get_coeff();
49
50 int deux_entrees=0;
51 int nb_comp=1;
52 if (resu.nb_dim()==2)
53 {
54 deux_entrees=1;
55 nb_comp=resu.dimension(1);
56 }
57 int nb_ligne_reel=correction_en_vitesse.dimension(0);
58// Determine U^
59 if (deux_entrees==0)
60 {
61 for(int i=0; i<nb_ligne_reel; i++)
62 {
63 for(int j=0; j<nb_comp; j++)
64 {
65 double t=resu(i);
66 for (int k=tab1(i*nb_comp+j); k<tab1(i*nb_comp+j+1)-1; k++)
67 t -= coeff(k)*current(tab2(k)-1);
68
69 correction_en_vitesse(i) = t/matrice(i*nb_comp+j,i*nb_comp+j);
70 }
71 }
72// verif
73// test
74 if (0)
75 for(int i=0; i<nb_ligne_reel; i++)
76 {
77 for (int j=0; j<nb_comp; j++)
78 {
79 int ii=i*nb_comp+j;
80 double tmp=resu(i);
81 for (int k = 0; k<matrice.nb_colonnes(); k++)
82 if ((k/nb_comp)!=i)
83 {
84 if (matrice(ii,k))
85 tmp -= matrice(ii,k)*current((k/nb_comp));
86 }
87
88 double test=tmp/matrice(ii,ii);
89 if (!est_egal(test,correction_en_vitesse(i),1e-5))
90 {
91 Cerr<<i<<" "<<j<<" "<<resu(i)<<finl;
92 Cerr<<test<<" ici "<<correction_en_vitesse(i)-test<<finl;
93 return -1;
94 }
95 }
96 }
97 }
98
99 else
100 {
101 for(int i=0; i<nb_ligne_reel; i++)
102 {
103 for(int j=0; j<nb_comp; j++)
104 {
105 double t=resu(i,j);
106 for (int k=tab1(i*nb_comp+j); k<tab1(i*nb_comp+j+1)-1; k++)
107 {
108 int i1 = (tab2(k)-1)/nb_comp;
109 int j1 = (tab2(k)-1)-i1*nb_comp;
110 t -= coeff(k)*current(i1,j1);
111 }
112 const double diagonal_coefficient = matrice(i*nb_comp+j,i*nb_comp+j);
113 assert(diagonal_coefficient!=0);
114 correction_en_vitesse(i,j) = t/diagonal_coefficient;
115 }
116 }
117// test
118 if (0)
119 {
120 Cerr<<"milieu"<<finl;
121 DoubleTrav corr2(correction_en_vitesse);
122 matrice.multvect(current,corr2);
123 for(int i=0; i<nb_ligne_reel; i++)
124 {
125 for (int j=0; j<nb_comp; j++)
126 {
127 int ii = i*nb_comp+j;
128 corr2(i,j) = (resu(i,j)-corr2(i,j)+matrice(ii,ii)*current(i,j))/matrice(ii,ii);
129 double test = corr2(i,j);
130 if (!est_egal(test,correction_en_vitesse(i,j)))
131 {
132 Cerr<<i<<" "<<j<<" "<<resu(i,j)<<finl;
133 Cerr<<test<<" ici "<<correction_en_vitesse(i,j)<<" "<<current(i,j)<<" "<<matrice(ii,ii) <<finl;
134 return -1;
135 }
136 }
137 }
138 Cerr<<"fin"<<finl;
139 }
140 }
141 correction_en_vitesse.echange_espace_virtuel();
142 return 0;
143}

146// Entree : Uk-1 ; Pk-1
147// Sortie Uk ; Pk

150void Simpler::iterer_NS(Equation_base& eqn,DoubleTab& current,DoubleTab& pression,double dt,Matrice_Morse& matrice,double seuil_resol,DoubleTrav& secmem,int nb_ite,int& converge, int& ok)
151{
152 if (eqn.probleme().is_dilatable())
153 {
154 Cerr<<" Simpler cannot be used with a quasi-compressible fluid."<<finl;
155 Cerr<<__FILE__<<(int)__LINE__<<" non code" <<finl;
156 exit();
157 }
158
159 Parametre_implicite& param = get_and_set_parametre_implicite(eqn);
160 SolveurSys& le_solveur_ = param.solveur();
161
162 Navier_Stokes_std& eqnNS = ref_cast(Navier_Stokes_std,eqn);
163 DoubleTrav gradP(current);
164 DoubleTrav correction_en_pression(pression);
165 DoubleTrav correction_en_vitesse(current);
166 DoubleTrav resu(current);
167
168// int deux_entrees = 0;
169// if (current.nb_dim()==2) deux_entrees = 1;
170 Operateur_Grad& gradient = eqnNS.operateur_gradient();
171 Operateur_Div& divergence = eqnNS.operateur_divergence();
172
173// int nb_comp = 1;
174// int nb_dim = current.nb_dim();
175
176// if (nb_dim==2)
177// nb_comp = current.dimension(1);
178
179// Construction de matrice et resu
180// matrice = A[Uk-1] = M/dt + CONV +DIFF
181// resu = A[Uk-1]Uk-1 -(A[Uk-1]Uk-1-Ss) + Sv -BtPk-1
182 gradient.calculer(pression,gradP);
183 if (eqnNS.has_interface_blocs()) /* si assembler_blocs est disponible */
184 eqnNS.assembler_blocs_avec_inertie({{ "vitesse", &matrice }}, resu);
185 else
186 {
187 resu -= gradP;
188 eqnNS.assembler_avec_inertie(matrice,current,resu);
189 }
190
191 le_solveur_.valeur().reinit();
192
193// Resolution du systeme : D[Uk-1]UPk = E[Uk-1]Uk-1 + Sv + Ss -BtPk-1 + (M/dt)Uk-1
194// matrice = A[Uk-1] = D - E avec D partie diagonale de A
195// correction_en_vitesse = UPk ; current = Uk-1 ; resu = Sv + Ss -BtPk-1
196 int status = inverser_par_diagonale(matrice,resu,current,correction_en_vitesse);
197 if (status!=0) exit();
198 Debog::verifier("Simpler::iterer_NS correction_en_vitesse",correction_en_vitesse);
199
200// Construction de matrice_en_pression_2 = BD-1Bt[Uk-1]
201 Matrice& matrice_en_pression_2 = eqnNS.matrice_pression();
202 assembler_matrice_pression_implicite(eqnNS,matrice,matrice_en_pression_2);
203 SolveurSys& solveur_pression_ = eqnNS.solveur_pression();
204 solveur_pression_.valeur().reinit();
205
206// Calcul de BUPk
207 divergence.calculer(correction_en_vitesse,secmem);
208 secmem *= -1;
209 secmem.echange_espace_virtuel();
210 if (nb_ite==1)
211 eqnNS.assembleur_pression().valeur().modifier_secmem(secmem);
212
213// Resolution du systeme (BD-1Bt)P*_k = BUPk
214// correction_en_pression = P*_k ; secmem = BUPk
215 solveur_pression_.resoudre_systeme(matrice_en_pression_2.valeur(),
216 secmem,correction_en_pression);
217
218// Calcul de Pk = Pk-1 + P*_k
219 operator_add(pression, correction_en_pression, VECT_ALL_ITEMS);
220
221 eqnNS.assembleur_pression().valeur().modifier_solution(pression);
222
223// Calcul de Bt P*_k et ajustement de resu a -Bt Pk
224 gradient.valeur().multvect(correction_en_pression,gradP);
225 resu -= gradP;
226 resu.echange_espace_virtuel();
227 Debog::verifier("Simpler::iterer_NS resu",resu);
228
229// Resolution du systeme : A[Uk-1]U*_k = -BtPk + Sv + Ss + (M/dt)Uk-1
230// current = U*_k ; resu = A[Uk-1]Uk-1 - (A[Uk-1]Uk-1-Ss) + Sv -BtPk + (M/dt)Uk-1
231 le_solveur_.resoudre_systeme(matrice,resu,current);
232
233// Calcul de BU*_k
234 divergence.calculer(current,secmem);
235 secmem *= -1;
236 secmem.echange_espace_virtuel();
237 Debog::verifier("Simpler::iterer_NS secmem",secmem);
238
239// Resolution du systeme (BD-1Bt)P'k = BU*_k
240// correction_en_pression = P'k ; secmem = BU*_k
241 correction_en_pression = 0;
242 solveur_pression_.resoudre_systeme(matrice_en_pression_2.valeur(),
243 secmem,correction_en_pression);
244
245// Resolution du systeme D[Uk-1](Uk-U*_k) = -BtP'k
246// correction_en_vitesse = U'k = Uk-U*_k ; correction_en_pression = P'k
247 calculer_correction_en_vitesse(correction_en_pression,gradP,correction_en_vitesse,matrice,gradient);
248
249 if (0)
250 {
251 Cerr<<" rr "<<mp_max_abs_vect(secmem)<<finl;
252// Cerr<<secmem<<finl;
253 secmem*=-1;
254 divergence.ajouter(correction_en_vitesse, secmem);
255 Cerr<<" rr2 "<<mp_max_abs_vect(secmem)<<finl;
256
257 }
258
259// Calcul de Uk = U*_k + U'k
260 current += correction_en_vitesse;
261 current.echange_espace_virtuel();
262 Debog::verifier("Simpler::iterer_NS current",current);
263
264 if (0)
265 {
266 divergence.calculer(current, secmem);
267 Cerr<<" apresdiv "<<mp_max_abs_vect(secmem)<<finl;;
268 }
269
270}
