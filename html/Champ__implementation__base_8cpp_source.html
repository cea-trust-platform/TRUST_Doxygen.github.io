58const Zone_VF& Champ_implementation_base::get_zone_dis(void) const
59{
60 const Champ_base& ch_base = le_champ();
61
62 if(sub_type(Champ_Inc_base, ch_base))
63 {
64 const Champ_Inc_base& ch_inc = ref_cast(Champ_Inc_base, ch_base);
65 return ref_cast(Zone_VF,ch_inc.zone_dis_base());
66 }
67 else if(sub_type(Champ_Fonc_base, ch_base))

73 {
74 Cerr << le_champ().que_suis_je() << "do not know refer a Zone_VF" << finl;
