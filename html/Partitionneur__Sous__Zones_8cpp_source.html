22// XD partitionneur_sous_zones partitionneur_deriv sous_zones -1 This algorithm will create one part for each specified subzone/domain. All elements contained in the first subzone/domain are put in the first part, all remaining elements contained in the second subzone/domain in the second part, etc... NL2 If all elements of the current domain are contained in the specified subzones/domain, then N parts are created, otherwise, a supplemental part is created with the remaining elements. NL2 If no subzone is specified, all subzones defined in the domain are used to split the mesh.

87void Partitionneur_Sous_Zones::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
88{
89 assert(ref_domaine_.non_nul());
90 const Domaine& dom = ref_domaine_.valeur();
91 const Zone& zone = dom.zone(0);
92 const int nb_elem = zone.nb_elem_tot();
93 elem_part.resize(nb_elem);
94 elem_part = -1;
95 if (noms_domaines_.size()!=0 && noms_sous_zones_.size())
96 {
97 Cerr << "Can't mix sous_zones and domaines yet in sous_zones partitionner." << finl;
98 Process::exit();
99 }
100// Numero de processeur en cours d'attribution:
101 int pe = 0;
102 int count = 0;
103 if (noms_domaines_.size()!=0)
104 {
105 for (int i=0; i<noms_domaines_.size(); i++)
106 {
107 const Nom& nom_domaine = noms_domaines_[i];
108 if (!interprete().objet_existant(nom_domaine) || !sub_type(Domaine, interprete().objet(nom_domaine)))
109 {
110// Contrairement au sous zone, le domaine doit exister pour le moment
111 Cerr << "Domain " << nom_domaine << " is not existing." << finl;
112 Process::exit();
113// ToDo eventuellement liste de zones dans des fichiers ex: DOM_0000.Zones, DOM_0001.Zones
114 }
115 const Domaine& domaine = ref_cast(Domaine, interprete().objet(nom_domaine));
116 DoubleTab domaine_xp;
117 domaine.zone(0).calculer_centres_gravite(domaine_xp);
118 IntVect dom_cells_containing_domaine_cells;
119 dom.zone(0).chercher_elements(domaine_xp, dom_cells_containing_domaine_cells);
120 Cerr << " Allocation of elements of the domain " << domaine.le_nom() << " to the processor " << pe << finl;
121 for (int cell = 0; cell < dom_cells_containing_domaine_cells.size(); cell++)
122 {
123 int elem = dom_cells_containing_domaine_cells[cell];
124 if (elem_part[elem] < 0)
125 {
126 count++;
127 elem_part[elem] = pe;
128 }
129 }
130 Cerr << " Number of elements attributed to the processor " << pe << " : " << count << finl;
131 pe++;
132 }
133 }
134 else
135 {
136 const int nb_sous_zones = dom.nb_ss_zones();
137 const int toutes_sous_zones = (noms_sous_zones_.size() == 0);
138 if (toutes_sous_zones)
139 Cerr << " No subarea specified, we use all existing subareas." << finl;
140 for (int i_sous_zone = 0; i_sous_zone < nb_sous_zones; i_sous_zone++)
141 {
142 const Sous_Zone& sous_zone = dom.ss_zone(i_sous_zone);
143 const Nom& nom = sous_zone.le_nom();
144 bool sous_zone_trouvee = noms_sous_zones_.contient_(nom);
145 if (!toutes_sous_zones && !sous_zone_trouvee)
146 continue;
147
148 Cerr << " Allocation of elements of the subarea " << nom << " to the processor " << pe << finl;
149 count = 0;
150 const int nb_elem_ssz = sous_zone.nb_elem_tot();
151 for (int i = 0; i < nb_elem_ssz; i++)
152 {
153 const int elem = sous_zone[i];
154 if (elem_part[elem] < 0)
155 {
156 count++;
157 elem_part[elem] = pe;
158 }
159 }
160 Cerr << " Number of elements attributed to the processor " << pe << " : " << count << finl;
161 pe++;
162 }
163 }
164 Cerr << " Allocation of the remaining elements to the processor " << pe << " : ";
165 count = 0;
166 for (int elem = 0; elem < nb_elem; elem++)
167 {
168 if (elem_part[elem] < 0)
169 {
170 count++;
171 elem_part[elem] = pe;
172 }
173 }
174 Cerr << count << " elements." << finl;
175 /*
176 if (count==0)
177 {
178 Cerr << "Error! The last zone will be empty with 0 elements. It is surely not what you want." << finl;
179 Cerr << "Check, your partition strategy." << finl;
180 Process::exit();
181 } */
182 if (liste_bords_periodiques_.size() > 0)
183 corriger_bords_avec_liste(dom, liste_bords_periodiques_, 0, elem_part);
