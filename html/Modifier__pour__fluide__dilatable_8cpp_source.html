29// Modif B.M: on ne remplit que la partie reelle du tableau.
30void multiplier_diviser_rho(DoubleVect& tab,const Fluide_Dilatable_base& le_fluide,int diviser)
31{
32 const Zone_VF& zvf = ref_cast(Zone_VF,le_fluide.vitesse().valeur().zone_dis_base());
33// Descripteurs des tableaux aux elements et aux faces:
34 const Zone& zone = zvf.zone();
35 const MD_Vector& md_elem = zone.les_elems().get_md_vector();
36 const MD_Vector& md_faces = zvf.md_vector_faces();
37 const MD_Vector& md_faces_bord = zvf.md_vector_faces_bord();
38
39 if (tab.get_md_vector() == md_faces_bord)
40 {
41 const DoubleTab& rho = ref_cast(Fluide_Dilatable_base,le_fluide).rho_discvit();
42 DoubleVect rho_bord;
43// B.M. je cree une copie sinon il faut truander les tests sur les tailles dans multiply_any_shape
44// ou creer un DoubleTab qui pointe sur rho...
45 zvf.creer_tableau_faces_bord(rho_bord,Array_base::NOCOPY_NOINIT);
46 rho_bord.inject_array(rho,rho_bord.size());
47 multiplier_ou_diviser(tab,rho_bord,diviser);
48 return;
49 }
50
51 if (tab.get_md_vector() == md_faces)
52 {
53 const DoubleTab& rho_faces = ref_cast(Fluide_Dilatable_base,le_fluide).rho_discvit();
54 multiplier_ou_diviser(tab,rho_faces,diviser);
55 return;
56 }
57
58 const DoubleTab& rho = le_fluide.masse_volumique().valeurs();
59 if (tab.get_md_vector() == md_elem && rho.get_md_vector() == md_elem)
60 {
61 multiplier_ou_diviser(tab,rho,diviser);
62 return;
63 }
64
65 if (tab.get_md_vector() == md_elem && rho.get_md_vector() == md_faces)
66 {
67// Il faut calculer rho aux elements
68 const DoubleTab& tab_rho = le_fluide.masse_volumique().valeurs();
69 DoubleVect rho_elem;
70 zone.creer_tableau_elements(rho_elem, Array_base::NOCOPY_NOINIT);
71 const int nb_elem_tot = zone.nb_elem_tot();
72 const IntTab& elem_faces = zvf.elem_faces();
73 const int nfe = elem_faces.dimension(1);
74 const double facteur = 1. / nfe;
75 for (int elem = 0; elem < nb_elem_tot; elem++)
76 {
77 double x = 0.;
78 for (int face=0; face < nfe; face++)
79 {
80 int f = elem_faces(elem, face);
81 x += tab_rho[f];
82 }
83 rho_elem[elem] = x * facteur;
84 }
85 multiplier_ou_diviser(tab,rho_elem,diviser);
86 return;
87 }
88
89 Cerr << "Error in Modifier_pour_fluide_dilatable.cpp: multiplier_rho. Invalid discretization of tab and rho." << finl;
90 Process::exit();
91}
