56/*
57 * Execute a system command and get output.
58 */
59int exec_cmd_and_get_output(const char* cmd, std::string& result)
60{
61 std::array<char, 128> buffer;
62 std::unique_ptr<FILE, decltype(&pclose)> pipe(popen(cmd, "r"), pclose);
63 if (!pipe)
64 {
65 return 0; //!< popen failed - unable to execute system command
66 }
67 while (fgets(buffer.data(), buffer.size(), pipe.get()) != nullptr)
68 {
69 result += buffer.data();
70 }
71 return 1;
72}

74/*
75 * Custom error handler printing out in the journal the backtrace of the exception.
76 */
77void crit_err_hdlr(True_int sig_num, siginfo_t * info, void * ucontext)
78{
79 std::cerr << "Error handler triggered!! See Journal - process ID: " << Process::me() << std::endl;
80 Process::Journal() << "signal " << (int)sig_num
81 << " (" << strsignal(sig_num) << "), address is "
82 << long(info->si_addr) << finl; //" from "
83
84 void * array[100];
85 int size = backtrace(array, 100);
86
87 Process::Journal() << __FUNCTION__ << " backtrace returned " << size << " frames\n" << finl;
88
89// Extract symbols from backtrace
90 char ** messages = backtrace_symbols(array, size);
91
92// Skip first stack frame (points here) and print out backtrace
93 for (int i = 1; i < size && messages != NULL; ++i)
94 {
95 Process::Journal() << "[proc " << Process::me() << "]: (" << i << ") " << messages[i] << finl;
96// Find first occurence of '(' or ' ' in message[i] and assume
97// everything before that is the file name (don't go beyond 0 though, string terminator)
98 size_t p = 0;
99 while(messages[i][p] != '(' && messages[i][p] != ' '
100 && messages[i][p] != 0)
101 ++p;
102
103// Call 'addr2line' system utility to extract line number in the source code:
104 char syscom[256];
105 sprintf(syscom, "addr2line %p --functions -e %.*s | tr '\n' ' ' | c++filt", array[i], (True_int)p, messages[i]);
106 std::string output;
107 int ret = exec_cmd_and_get_output(syscom, output);
108 if(!ret)
109 Process::Journal() << "(Unable to execute 'addr2line' to get line number in source ...)" << finl;
110 else
111 Process::Journal() << "[proc " << Process::me() << "]: (" << i << ") ----> "
112 << output << finl;
113 }
114 Process::Journal() << finl;
115 free(messages);
116
117 Process::exit();
118}

120/*
121 * Install error handlers catching SIGABRT and SIGFPE
122 */
123void install_handlers()
124{
125 struct sigaction sigact;
126 sigemptyset(&sigact.sa_mask);
127 sigact.sa_sigaction = crit_err_hdlr;
128 sigact.sa_flags = SA_RESTART | SA_SIGINFO;
129
130 if (sigaction(SIGABRT, &sigact, (struct sigaction *)NULL) != 0)
131 {
132 std::cerr << "FATAL ERROR setting handler for signal " << SIGABRT
133 << " (" << strsignal(SIGABRT) << ")" << std::endl;
134 Process::exit();
135 }
136
137 if (sigaction(SIGFPE, &sigact, (struct sigaction *)NULL) != 0)
138 {
139 std::cerr << "FATAL ERROR setting handler for signal " << SIGFPE
140 << " (" << strsignal(SIGFPE) << ")" << std::endl;
141 Process::exit();
142 }
143 std::cerr << "Custom error handlers correctly installed. SIGFPE and SIGABRT redirected." << std::endl;
144}
