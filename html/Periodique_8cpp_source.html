65void Periodique::completer()
66{
67 Frontiere& frontiere = frontiere_dis().frontiere();
68 Cerr << "Initialization for periodic on " << frontiere.le_nom() << finl;
69
70// Recherche de la direction de periodicite:
71 ArrOfDouble erreur;
72 int ok = Reordonner_faces_periodiques::check_faces_periodiques(frontiere,
73 direction_perio_,
74 erreur,
75 1 /* verbose */);
76 if (!ok)
77 exit();
78
79 distance_ = norme_array(direction_perio_);
80 int i;
81 const int dim = direction_perio_.size_array();
82 direction_xyz_ = -2;
83 for (i = 0; i < dim; i++)
84 {
85 if (std::fabs(direction_perio_[i]) > precision_geom)
86 {
87 if (direction_xyz_ == -2)
88 direction_xyz_ = i;
89 else
90// Deuxieme coordonnee non nulle, vecteur direction n'est pas aligne sur un axe
91 direction_xyz_ = -1;
92 }
93 }
94 if (direction_xyz_ < 0)
95 Cerr << "Periodic direction not aligned on an axis" << finl;
96 else
97 Cerr << "Periodic direction aligned on the axis " << direction_xyz_ << finl;
98
99 const Zone& zone = frontiere.zone();
100
101// Creation d'un tableau d'indices parallele sur toutes les faces frontieres
102 IntTab tab_face_associee;
103 const Zone_VF& zonevf = ref_cast(Zone_VF, zone_Cl_dis().zone_dis().valeur());
104 zonevf.creer_tableau_faces_bord(tab_face_associee, Array_base::NOCOPY_NOINIT);
105 tab_face_associee = -1;
106
107// Nombre de faces virtuelles de cette frontiere:
108 const int nb_faces = frontiere.nb_faces();
109 const int nb_faces_2_ = nb_faces / 2;
110 const ArrOfInt& faces_virt = frontiere.get_faces_virt();
111 const int nb_faces_virt = faces_virt.size_array();
112
113// On remplit la partie reele du tableau face_associee pour la frontiere qui nous interesse:
114// et la partie reele du tableau "associee"
115 const int i_premiere_face = frontiere.num_premiere_face();
116 for (i = 0; i < nb_faces_2_; i++)
117 {
118 const int i1 = i_premiere_face + i;
119 const int i2 = i_premiere_face + i + nb_faces_2_;
120 tab_face_associee[i1] = i2;
121 tab_face_associee[i2] = i1;
122 }
123 const MD_Vector& md_faces_front = tab_face_associee.get_md_vector();
124// On echange espace virtuel avec traduction des indices:
125 Scatter::construire_espace_virtuel_traduction(md_faces_front,
126 md_faces_front,
127 tab_face_associee,
128 1 /* erreurs fatales */);
129// Tableau qui donne pour chaque face virtuelle de la zone, -1 si ce n'est pas une
130// face frontiere, sinon son indice dans les les frontieres.
131 const ArrOfInt& ind_faces_virt_bord = zone.ind_faces_virt_bord();
132// Creation d'un tableau qui donne, pour chaque face virtuelle des frontieres (toutes frontieres)
133// l'indice de la face dans la frontiere periodique courante (-1 sinon)
134 const int nb_faces_front_tot = tab_face_associee.size_totale();
135 ArrOfInt index(nb_faces_front_tot);
136 index= -2;
137 const int nb_faces_zone = zone_Cl_dis().zone_dis().valeur().face_sommets().dimension(0);
138 for (i = 0; i < nb_faces_virt; i++)
139 {
140 const int face_zone = frontiere.face_virt(i); //!< Indice d'une face de la zone
141 const int face_front = ind_faces_virt_bord[face_zone - nb_faces_zone]; //!< Indice dans les frontieres
142 index[face_front] = nb_faces + i;
143 }
144
145// Remplissage du tableau "face_front_associee_"
146 face_front_associee_.resize_array(nb_faces + nb_faces_virt);
147 for (i = 0; i < nb_faces + nb_faces_virt; i++)
148 {
149 int resu = -1;
150 if (i < nb_faces_2_)
151 resu = i + nb_faces_2_;
152 else if (i < nb_faces)
153 resu = i - nb_faces_2_;
154 else
155 {
156// Face virtuelle :
157 const int face_zone = frontiere.face_virt(i - nb_faces); //!< Indice de la face de la zone
158 const int face_front = ind_faces_virt_bord[face_zone - nb_faces_zone]; //!< Indice dans les frontieres Indice de la face associee dans le tableau face_associee:
159//
160 const int face_front_associee = tab_face_associee[face_front];
161 if (face_front_associee >= 0)
162 {
163// Indice de la face associee dans la frontiere:
164 const int face_asso = index[face_front_associee];
165 assert(face_asso >= 0);
166 resu = face_asso;
167 }
168 else
169 {
170// La face virtuelle associee n'est pas dans le domaine
171 Cerr << "Error in Periodique::completer()" << finl;
172 exit();
173 }
174 }
175 face_front_associee_[i] = resu;
176 }
177}
