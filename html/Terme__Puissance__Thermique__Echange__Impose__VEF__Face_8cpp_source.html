50void Terme_Puissance_Thermique_Echange_Impose_VEF_Face::mettre_a_jour(double temps)
51{
52 int nb_faces = la_zone_VEF.valeur().nb_faces();
53 const Zone_VEF& zone = ref_cast(Zone_VEF, la_zone_VEF.valeur());
54 const IntTab& face_voisins = zone.face_voisins();
55 const DoubleVect& volumes_entrelaces = zone.volumes_entrelaces();
56 const DoubleTab& himp = himp_.valeur().valeurs();
57 const DoubleTab& Text = Text_.valeur().valeurs();
58 const DoubleTab& T = equation().inconnue().valeurs();
59 const DoubleVect& volumes_elements = zone.volumes();
60
61 bilan()(0) = 0;
62
63 for (int num_face = 0; num_face < nb_faces; num_face++)
64 {
65 const double vol = volumes_entrelaces(num_face);
66
67
68 const int elem0 = face_voisins(num_face,0);
69 const int elem1 = face_voisins(num_face,1);
70 double h0 = 0, h1 = 0, text0 = 0, text1 = 0, vol0 = 0, vol1 = 0;
71 if (elem0 >= 0)
72 {
73 h0 = himp.addr()[elem0 * (himp.dimension(0) > 1)];
74 text0 = Text.addr()[elem0 * (Text.dimension(0) > 1)];
75 vol0 = volumes_elements(elem0);
76 }
77 if (elem1 >= 0)
78 {
79 h1 = himp.addr()[elem1 * (himp.dimension(0) > 1)];
80 text1 = Text.addr()[elem1 * (Text.dimension(0) > 1)];
81 vol1 = volumes_elements(elem1);
82 }
83 double hm = (h0 * vol0 + h1 * vol1) / (vol0 + vol1);
84 double htextm = (h0 * text0 * vol0 + h1 * text1 * vol1) / (vol0 + vol1);
85
86 double c = (zone.faces_doubles()[num_face]==1) ? 0.5 : 1 ;
87 bilan()(0) -= c * (hm*T(num_face)-htextm) * vol;
88
89 }
90 himp_.mettre_a_jour(temps);
91 Text_.mettre_a_jour(temps);
92}
