104//! Genere les faces<br>construits les frontieres
105void Zone_VF::discretiser()
106{
107 Cerr << "<<<<<< Discretization VF >>>>>>>>>>" << finl;
108
109 Zone& lazone=zone();
110 histogramme_angle(lazone.domaine(),Cerr);
111 Faces* les_faces_ptr=creer_faces();
112 Faces& les_faces= *les_faces_ptr;
113 {
114 {
115 Type_Face type_face = zone().type_elem().type_face(0);
116 les_faces.typer(type_face);
117 les_faces.associer_zone(zone());
118
119 Static_Int_Lists connectivite_som_elem;
120 const int nb_sommets_tot = zone().domaine().nb_som_tot();
121 const IntTab& elements = zone().les_elems();
122
123 construire_connectivite_som_elem(nb_sommets_tot,
124 elements,
125 connectivite_som_elem,
126 1 /* include virtual elements */);
127
128 Faces_builder faces_builder;
129 faces_builder.creer_faces_reeles(zone(),
130 connectivite_som_elem,
131 les_faces,
132 elem_faces_);
133 }
134
135 reordonner(les_faces);
136
137// Les faces sont dans l'ordre definitif, on peut remplir
138// renum_items_communs des faces:
139 Scatter::calculer_renum_items_communs(lazone.faces_joint(), Joint::FACE);
140
141// Remplissage de face_voisins des frontieres:
142// a factoriser avec DomaineCut.cpp
143 {
144 const IntTab& facevoisins = les_faces.voisins();
145 const int nb_frontieres = lazone.nb_front_Cl();
146 for (int i_frontiere = 0; i_frontiere < nb_frontieres; i_frontiere++)
147 {
148 Frontiere& fr = lazone.frontiere(i_frontiere);
149 IntTab& mes_face_voisins = fr.faces().voisins();
150 const int nbfaces = fr.nb_faces();
151 const int premiere_face = fr.num_premiere_face();
152 mes_face_voisins.resize(nbfaces, 2);
153 for (int i = 0; i < nbfaces; i++)
154 {
155 mes_face_voisins(i, 0) = facevoisins(premiere_face + i, 0);
156 mes_face_voisins(i, 1) = facevoisins(premiere_face + i, 1);
157 }
158 }
159 }
160
161 Scatter::calculer_espace_distant_faces(zone(), les_faces.nb_faces(), elem_faces_);
162// Apres la methode suivante, on aura le droit d'utiliser creer_tableau_faces() :
163 Scatter::construire_md_vector(zone().domaine(), les_faces.nb_faces(), Joint::FACE, md_vector_faces_);
164
165
166 const MD_Vector& md_vect_sommets = zone().domaine().les_sommets().get_md_vector();
167 const MD_Vector& md_vect_elements = zone().les_elems().get_md_vector();
168// Construction de l'espace virtuel du tableau face_sommets
169 Scatter::construire_espace_virtuel_traduction(md_vector_faces_, md_vect_sommets, les_faces.les_sommets());
170
171// Idem pour face_voisins_
172// Certaines faces des elements virtuels les plus externes n'ont pas d'element local voisin => erreurs non fatales
173 Scatter::construire_espace_virtuel_traduction(md_vector_faces_, md_vect_elements, les_faces.voisins(), 0 /* error not fatal */);
174
175// Idem pour elem_faces_
176 Scatter::construire_espace_virtuel_traduction(md_vect_elements, md_vector_faces_, elem_faces_);
177
178 lazone.init_faces_virt_bord(md_vector_faces_, md_vector_faces_front_);
179
180// Calcul des surfaces:
181 les_faces.calculer_surfaces(face_surfaces_);
182// Calcul de la surface des faces virtuelles
183 MD_Vector md_nul;
184 creer_tableau_faces(face_surfaces_);
185 face_surfaces_.echange_espace_virtuel();
186 face_surfaces_.set_md_vector(md_nul); //!< Detache la structure parallele
187
188// Changement a la v1.5.7 beta: xv_ a maintenant un descripteur parallele: dimension(0)=nb_faces
189 les_faces.calculer_centres_gravite(xv_);
190
191 face_voisins().ref(les_faces.voisins());
192 face_sommets().ref(les_faces.les_sommets());
193
194// Calcul des volumes
195 lazone.calculer_volumes(volumes_, inverse_volumes_);
196 }
197 {
198 int i=0, derniere=lazone.nb_bords();
199 les_bords_.dimensionner(zone().nb_front_Cl());
200 for(; i<derniere; i++)
201 {
202 les_bords_[i].associer_frontiere(lazone.bord(i));
203 les_bords_[i].associer_Zone_dis(*this);
204 }
205 int decal=derniere;
206 derniere+=zone().nb_raccords();
207 for(; i<derniere; i++)
208 {
209 int j=i-decal;
210 les_bords_[i].associer_frontiere(lazone.raccord(j).valeur());
211 les_bords_[i].associer_Zone_dis(*this);
212 }
213 decal=derniere;
214 derniere+=zone().nb_frontieres_internes();
215 for(; i<derniere; i++)
216 {
217 int j=i-decal;
218 les_bords_[i].associer_frontiere(lazone.faces_interne(j));
219 les_bords_[i].associer_Zone_dis(*this);
220 }
221 }
222// Centre de gravite des elements (tableau xp_)
223 lazone.calculer_centres_gravite(xp_);
224// Centre de gravite de la zone
225 ArrOfDouble c(dimension);
226 lazone.calculer_mon_centre_de_gravite(c);
227
228// On cree les domaines frontieres
229 lazone.domaine().creer_mes_domaines_frontieres(*this);
230
231// Attention : ca ne remplit pas elem_faces, ca fait
232// creer_faces_virtuelles_non_std() en VEF :
233 remplir_elem_faces();
234
235 delete les_faces_ptr;
236
237// Fill in the table face_numero_bord
238 remplir_face_numero_bord();
239
240//
241// On imprime des infos
242//
243 lazone.domaine().imprimer(); //!< Extremas du domaine et volumes
244 infobord(); //!< Aires des bords
245 info_elem_som(); //!< Nombre elements et sommets
246 Cerr << "<<<<<< End of Discretization VF >>>>>>>>>>" << finl;
247}

267void Zone_VF::modifier_pour_Cl(const Conds_lim& conds_lim)
268{
269 Cerr << " Zone_VF::modifier_pour_Cl" << finl;
270 int nb_cond_lim=conds_lim.size();
271 int i=0;
272 for (; i<nb_cond_lim; i++)
273 {
274// for( cl ...
275 const Cond_lim_base& cl=conds_lim[i].valeur();
276 if (sub_type(Periodique, cl))
277 {
278// if (perio ...
279// Modifications du tableau face_voisins
280// Une face de type periodicite doit avoir les memes voisins que
281// sa face associee
282// face_voisins(face,j) = face_voisins(face_associee(face),j) j=1,2
283//
284
285 const Periodique& la_cl_period = ref_cast(Periodique,cl);
286 const Front_VF& frontieredis = ref_cast(Front_VF,la_cl_period.frontiere_dis());
287
288 int nb_face_bord = frontieredis.nb_faces();
289 int ndeb =0;
290 int nfin = frontieredis.nb_faces_tot();
291#ifndef NDEBUG
292 int num_premiere_face = frontieredis.num_premiere_face();
293 int num_derniere_face = num_premiere_face+nfin;
294#endif
295 int face;
296
297 for (int ind_face=ndeb; ind_face<nfin; ind_face++)
298 {
299 face = frontieredis.num_face(ind_face);
300 int faassociee = la_cl_period.face_associee(ind_face);
301 faassociee = frontieredis.num_face(faassociee);
302 if (ind_face<nb_face_bord)
303 {
304 assert(faassociee>=num_premiere_face);
305 assert(faassociee<num_derniere_face);
306 }
307
308 if (face_voisins_(face,0) == -1)
309 face_voisins_(face,0) = face_voisins_(faassociee,0);
310 else if (face_voisins_(face,1) == -1)
311 face_voisins_(face,1) = face_voisins_(faassociee,1);
312 }
313 }
314 }

322 marquer_faces_double_contrib(conds_lim);

327// Methode pour la construction du tableau num_fac_loc_
328void Zone_VF::construire_num_fac_loc()
329{
330 const int size = nb_faces_tot();
331 num_fac_loc_.resize(size,2);
332 num_fac_loc_=-1;
333 for(int face=0; face<size; face++)
334 for (int voisin=0; voisin<2; voisin++)
335 {
336 int elem = face_voisins_(face,voisin);
337 if(elem!=-1)
338 {
339 int face_loc = numero_face_local(face,elem);
340 if (face_loc != -1) num_fac_loc_(face,voisin) = face_loc;
341 else
342 {
343// Cas periodique
344 int autre_voisin = 1 - voisin;
345 int elem2 = face_voisins(face,autre_voisin);
346 int nb_faces_elem = elem_faces_.dimension(1);
347 for (face_loc=0; face_loc<nb_faces_elem; face_loc++)
348 {
349 const int face_glob = elem_faces(elem,face_loc);
350 const int elem_voisin1 = face_voisins(face_glob,0);
351 const int elem_voisin2 = face_voisins(face_glob,1);
352 if (elem_voisin1==elem2 || elem_voisin2==elem2)
353 {
354 num_fac_loc_(face,voisin) = face_loc;
355 break;
356 }
357 }
358 }
359 }
360 }
361}

363// Renvoie le numero local de face a partir des numeros globaux face et elem
364// Utiliser si possible Zone_VF::num_fac_loc(face,voisin) car la methode est
365// optimisee, celle ci non.

457//
458//! Compute the normalized boundary outward vector associated to the face global_face_number<br>and eventually scaled by scale_factor
459DoubleTab Zone_VF::normalized_boundaries_outward_vector(int global_face_number, double scale_factor) const
460{
461 const IntTab& neighbor_faces = face_voisins();
462 DoubleTab normal_vector(dimension); //result
463 for (int d = 0; d < dimension; d++)
464 normal_vector(d) = face_normales(global_face_number, d) / face_surfaces(global_face_number);
465
466// now we have a normalized normal vector
467// but it remains to orient it correctly (outwards)
468// for this we build a vector that links middle of the face to the center of the element
469// and then we do the inner product between this new vector and normal vector
470// So, if the result is positive, we change the sign of normal vector
471// else we do nothing
472 int neighbor_elem=neighbor_faces(global_face_number,0);
473 if( neighbor_elem == -1 )
474 neighbor_elem = neighbor_faces(global_face_number,1);
475 DoubleTab vector_face_elem(dimension);
476 double inner_product=0;
477 for(int i=0; i<dimension; i++)
478 {
479// ith component of the vector
480 double vec_face_elem = xp(neighbor_elem,i)-xv(global_face_number,i);
481 inner_product+=vec_face_elem*normal_vector(i);
482 }
483
484 if( inner_product > 0 )
485 {
486 for(int i=0; i<dimension; i++)
487 {
488 normal_vector(i)*=-1;
489 }
490 }
491
492// now the normal vector is well oriented (outwards)
493// we can multiply it by a given scale factor
494 for(int i=0; i<dimension; i++)
495 {
496 normal_vector(i)*=scale_factor;
497 }
498
499 return normal_vector;
500}

502void Zone_VF::marquer_faces_double_contrib(const Conds_lim& conds_lim)
503{
504 Journal() << " Zone_VF::marquer_faces_double_contrib" << finl;
505 faces_doubles_.resize_array(nb_faces());
506// marquage des faces periodiques
507//
508
509 int nb_cond_lim=conds_lim.size();
510
511 for (int i=0; i<nb_cond_lim; i++)
512 {
513 const Cond_lim_base& cl=conds_lim[i].valeur();
514 if (sub_type(Periodique, cl))
515 {
516 const Periodique& la_cl_period = ref_cast(Periodique,cl);
517 const Front_VF& frontieredis = ref_cast(Front_VF,la_cl_period.frontiere_dis());
518 int ndeb = frontieredis.num_premiere_face();
519 int nfin = ndeb + frontieredis.nb_faces();
520
521 for (int face=ndeb; face<nfin; face++)
522 {
523 faces_doubles_[face]=1;
524 }
525 }
526 }
527
528// marquage des faces items communs
529//
530 int nbjoints=nb_joints();
531
532 for(int njoint=0; njoint<nbjoints; njoint++)
533 {
534 const Joint& joint_temp = joint(njoint);
535 const IntTab& indices_faces_joint = joint_temp.joint_item(Joint::FACE).renum_items_communs();
536 const int nbfaces = indices_faces_joint.dimension(0);
537 for (int j = 0; j < nbfaces; j++)
538 {
539// Pour acceder au face de joint, il existe desormais un tableau dedie renum_items_communs()
540// car l'ancienne facon d'acceder a ces faces etait de passer par
541// joint.num_premiere_face() et joint.nb_faces() ce qui n'est plus valable !
542// En effet, la numerotation des faces de joint n'est plus continue desormais !
543// joint.num_premiere_face() retourne -1 desormais pour detecter les anciens codages.
544 int face_de_joint = indices_faces_joint(j, 1);
545 faces_doubles_[face_de_joint] = 1;
546 }
547 }

554void Zone_VF::infobord()
555{
556 Cerr << "==============================================" << finl;
557 Cerr << "The boundary areas of the domain " << zone().domaine().le_nom() << " are:" << finl;
558 DoubleVect surfaces;
559
560// Raccords
561 Raccords& raccords=zone().faces_raccord();
562 for (int i=0; i<raccords.nb_raccords(); i++)
563 {
564 Faces& faces=raccords(i).valeur().faces();
565 faces.associer_zone(zone());
566 faces.calculer_surfaces(surfaces);
567 double s=0;
568 for (int j=0; j<faces.nb_faces(); j++)
569 s=s+surfaces(j);
570 s=mp_sum(s);
571 raccords(i).valeur().set_aire(s);
572 Cerr<<"Area of "<<raccords(i).le_nom()<<" \t= "<<s<<finl;
573 }
574
575// Bords
576 Bords& bords=zone().faces_bord();
577 for (int i=0; i<bords.nb_bords(); i++)
578 {
579 Faces& faces=bords(i).faces();
580 faces.associer_zone(zone());
581 faces.calculer_surfaces(surfaces);
582 double s=0;
583 for (int j=0; j<faces.nb_faces(); j++)
584 s=s+surfaces(j);
585 s=mp_sum(s);
586 bords(i).set_aire(s);
587 Cerr<<"Area of "<<bords(i).le_nom()<<" \t= "<<s<<finl;
588 }
589 Cerr << "==============================================" << finl;
590}

592void Zone_VF::info_elem_som()
593{
594 const int nbelem = zone().les_elems().get_md_vector().valeur().nb_items_seq_tot();
595 const int nbsom = zone().domaine().les_sommets().get_md_vector().valeur().nb_items_seq_tot();
596 const int nbfaces = face_voisins().get_md_vector().valeur().nb_items_seq_tot();
597 Cerr<<"Calculation of elements and nodes on " << zone().domaine().le_nom() << " :" << finl;
598 Cerr<<"Total number of elements = "<<nbelem<<finl;
599 Cerr<<"Total number of nodes = "<<nbsom<<finl;
600 Cerr<<"Total number of faces = "<<nbfaces<<finl;
601 Raccords& raccords=zone().faces_raccord();
602 for (int i=0; i<raccords.nb_raccords(); i++)
603 {
