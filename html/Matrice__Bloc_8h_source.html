24/*----------------------------------------------------------------------------*/
25/* */
26/* classe Matrice_Bloc : stockage des blocs d`une matrice bloc A(N,M) */
27/* */
28/* le stockage est effectue ligne par ligne */
29/* dans un vecteur de type VECT(Matrice). */
30/* On utilise: */
31/* blocs_ = vecteur de matrices Aij */
32/* N_ = 1ere dim de blocs_ */
33/* M_ = 2eme dim de blocs_ */
34/* nb_blocs_ = nb total de blocs (= N_ * M_) */
35/* */
36/* Forme matricielle: */
37/* */
38/* [A11 A12 ... A1M] */
39/* [A21 A22 ... A2M] */
40/* A = [... ... ... ...] blocs_=[A11,A12,...,A1M,A21,A22,...,A2M, */
41/* [AN1 AN2 ... ANM] ,...,AN1,AN2,...,ANM] */
42/* */
43/* */
44/*----------------------------------------------------------------------------*/
