73DoubleTab& Terme_Source_Canal_perio_VEF_P1NC::ajouter(DoubleTab& resu) const
74{
75 const Zone_VF& zone_VF = la_zone_VEF.valeur();
76 const DoubleVect& volumes_entrelaces = zone_VF.volumes_entrelaces();
77 const DoubleVect& volumes_entrelaces_Cl = la_zone_Cl_VEF.valeur().volumes_entrelaces_Cl();
78 int premiere_face_std=la_zone_VEF.valeur().premiere_face_std() ;
79 const DoubleVect& porosite_face=zone_VF.porosite_face();
80 ArrOfDouble s;
81 s = source();
82 if (s.size_array()==dimension)
83 {
84// Case Navier Stokes, s is uniform
85 for (int num_face = 0 ; num_face<premiere_face_std; num_face++)
86 {
87 double vol = volumes_entrelaces_Cl(num_face)*porosite_face(num_face);
88 for (int i=0; i<dimension; i++)
89 resu(num_face,i)+= s[i]*vol;
90 }
91 int nb_faces = zone_VF.nb_faces();
92 for (int num_face = premiere_face_std; num_face<nb_faces; num_face++)
93 {
94 double vol = volumes_entrelaces(num_face)*porosite_face(num_face);
95 for (int i=0; i<dimension; i++)
96 resu(num_face,i)+= s[i]*vol;
97 }
98 }
99 else
100 {
101// Case Energy, s is non uniform
102 bilan_=0;
103 const ArrOfInt& fd=zone_VF.faces_doubles();
104 for (int num_face = 0 ; num_face<premiere_face_std; num_face++)
105 {
106 double vol = volumes_entrelaces_Cl(num_face)*porosite_face(num_face);
107 double contrib = s[num_face]*vol;
108 resu(num_face)+= contrib;
109 bilan_(0)+= contrib*(1-0.5*fd[num_face]);
110 }
111 int nb_faces = zone_VF.nb_faces();
112 for (int num_face = premiere_face_std; num_face<nb_faces; num_face++)
113 {
114 double vol = volumes_entrelaces(num_face)*porosite_face(num_face);
115 double contrib = s[num_face]*vol;
116 resu(num_face)+= contrib;
117 bilan_(0)+= contrib*(1-0.5*fd[num_face]);
118 }
119 }
120 return resu;
121}

123void Terme_Source_Canal_perio_VEF_P1NC::calculer_debit(double& debit_e) const
124{
125 const Zone_VF& zone_VF = la_zone_VEF.valeur();
126 const Zone_Cl_dis_base& zone_Cl_dis = la_zone_Cl_VEF.valeur();
127 const Champ_Inc_base& velocity = (sub_type(Convection_Diffusion_std,equation()) ? ref_cast(Convection_Diffusion_std,equation()).vitesse_transportante() : equation().inconnue());
128 const DoubleVect& porosite_face=zone_VF.porosite_face();
129// Check we have really velocity:
130 assert(velocity.le_nom()=="vitesse");
131 const DoubleTab& vitesse = velocity.valeurs();
132 int nb_bords = zone_VF.nb_front_Cl();
133 for (int n_bord=0; n_bord<nb_bords; n_bord++)
134 {
135 const Cond_lim& la_cl = zone_Cl_dis.les_conditions_limites(n_bord);
136 if (sub_type(Periodique,la_cl.valeur()))
137 {
138 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
139 if (bord_periodique_ == le_bord.le_nom())
140 {
141 debit_e=0.;
142// const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
143 int ndeb = le_bord.num_premiere_face();
144 int nfin = ndeb + le_bord.nb_faces()/2;
145 int axe = direction_ecoulement_;
146 if (equation().probleme().is_dilatable()==1)
147 {
148// Si l'on est en Quasi Compressible, il faut conserver
149// le debit massique et non pas le debit volumique.
150// C'est pour cela que dans le cas QC, on multiplie les vecteurs vitesse
151// par la masse volumique discretisee aux faces pour que lorsqu'on integre sur la surface,
152// on obtienne bien un debit massique et non pas un debit volumique.
153 const DoubleTab& tab_rho_face = ref_cast(Fluide_Dilatable_base,equation().milieu()).rho_discvit();
154
155 if (axe>=0)
156 for (int num_face=ndeb; num_face<nfin; num_face++)
157 {
158 double debit_face = porosite_face(num_face)*vitesse(num_face,axe) * std::fabs(zone_VF.face_normales(num_face,axe));
159 debit_e += tab_rho_face[num_face] * debit_face;
160 }
161 else
162 for (int num_face=ndeb; num_face<nfin; num_face++)
163 {
164 double debit_face=0;
165 for (int i=0; i<dimension; i++)
166 debit_face += porosite_face(num_face)*vitesse(num_face,i) * zone_VF.face_normales(num_face,i);
167 debit_e += tab_rho_face[num_face] * debit_face;
168 }
169 }
170 else
171 {
172 if (axe>=0)
173 for (int num_face=ndeb; num_face<nfin; num_face++)
174 {
175 double debit_face = porosite_face(num_face)*vitesse(num_face,axe) * std::fabs(zone_VF.face_normales(num_face,axe));
176 debit_e += debit_face;
177 }
178 else
179 for (int num_face=ndeb; num_face<nfin; num_face++)
180 {
181 double debit_face=0;
182 for (int i=0; i<dimension; i++)
183 debit_face += porosite_face(num_face)*vitesse(num_face,i) * zone_VF.face_normales(num_face,i);
184 debit_e += debit_face;
185 }
186 }
187 }
188 }
189 }
190 debit_e = mp_sum(debit_e);
191}
