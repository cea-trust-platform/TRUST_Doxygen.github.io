77//! Reordonne les sommets du segment.
78void Segment::reordonner()
79{
80 Zone& zone=ma_zone.valeur();
81 IntTab& elem=zone.les_elems();
82 const Domaine& dom=zone.domaine();
83 ArrOfInt S(2);
84 int i;
85 const int nb_elem=zone.nb_elem();
86 for (int num_poly=0; num_poly<nb_elem; num_poly++)
87 {
88 for(i=0; i<2; i++)
89 S[i] = elem(num_poly,i);
90
91 if(sup_strict(dom.coord(S[0], 0),dom.coord(S[1], 0)))
92 {
93 int tmp=S[0];
94 S[0]=S[1];
95 S[1]=tmp;
96 }
97
98 for(i=0; i<2; i++)
99 elem(num_poly, i)=S[i];
100 }
101}

147//! Renvoie 1 si l'element ielem de la zone associee a<br>l'element geometrique contient le point<br>de coordonnees specifiees par le parametre "pos".<br>Renvoie 0 sinon.
148/** \return int<br>
149Signification: 1 si le point de coordonnees specifiees<br>
150appartient a l'element "element"<br>
1510 sinon<br>
152Contraintes: */
153int Segment::contient(const ArrOfDouble& pos, int element ) const
154{
155 assert(pos.size_array()==dimension);
156
157 const Zone& zone=ma_zone.valeur();
158 const Domaine& dom=zone.domaine();
159 const IntTab& elem=zone.les_elems();
160// Test whether OM = a.O1 with O and 1 the extreme points of the seg and M the point to be tested
161 double autre_a = 0;
162 for (int d=0; d<dimension; d++)
163 {
164 double O1 = dom.coord(elem(element,1), d) - dom.coord(elem(element,0), d);
165 double OM = pos[d] - dom.coord(elem(element,0), d);
166 if (!est_egal(O1,0))
167 {
168 double a = OM/O1;
169// M is outside O1?
170 if (a<-Objet_U::precision_geom || a>1+Objet_U::precision_geom) return 0;
171// a is not the same as for another dimension, <=> not aligned
172 if (autre_a>0 && !est_egal(a, autre_a)) return 0;
173 autre_a = a;
174 }
175 else if (!est_egal(OM,0)) return 0; //!< M is not along O1
176 }
177 return 1;
178}

230//! Calcule les volumes des elements de la zone associee.
231void Segment::calculer_volumes(DoubleVect& volumes) const
232{
233 const Zone& zone=ma_zone.valeur();
234 const Domaine& dom=zone.domaine();
235 double dx;
236 int S1,S2;
237
238 int size_tot = zone.nb_elem_tot();
239 assert(volumes.size_totale()==size_tot);
240 for (int num_poly=0; num_poly<size_tot; num_poly++)
241 {
242 S1 = zone.sommet_elem(num_poly,0);
243 S2 = zone.sommet_elem(num_poly,1);
244 dx=0;
245 for (int i=0; i<dimension; i++)
246 {
247 double d = dom.coord(S2,i) - dom.coord(S1,i);
248 dx+=d*d;
249 }
250
251 volumes[num_poly]= sqrt(dx);
252 }
253}
