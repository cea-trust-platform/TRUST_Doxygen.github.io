47double Probleme_Couple::computeTimeStep(bool& stop) const
48{
49 double dt_=DMAXFLOAT;
50
51// On prend le dt min.
52// On stoppe si l'un des problemes veut stopper.
53 for (int i=0; i<nb_problemes(); i++)
54 {
55 double dt1=probleme(i).computeTimeStep(stop);
56 if (stop)
57 return 0;
58 if (dt1<dt_)
59 dt_=dt1;
60 }
61 return dt_;
62}

94bool Probleme_Couple::iterateTimeStep(bool& converged)
95{
96 bool ok=true;
97 converged=true;
98
99 int debut_gr=0;
100 int fin_gr=0;
101 int gr=0;
102 while(ok && fin_gr<nb_problemes())
103 {
104
105 if (gr<groupes.size_array())
106 fin_gr += groupes[gr++];
107 else
108 fin_gr=nb_problemes();
109
110 for(int i=debut_gr; ok && i<fin_gr; i++)
111 ok &= probleme(i).updateGivenFields();
112
113 for(int i=debut_gr; ok && i<fin_gr; i++)
114 {
115 bool cv;
116 ok &= probleme(i).iterateTimeStep(cv);
117 converged = converged && cv;
118 }
119
120 debut_gr=fin_gr;
121 }
122
123 return ok;
124}

136Entree& Probleme_Couple::readOn(Entree& is)
137{
138
139 Cerr << "Reading of Probleme_Couple " << le_nom() << finl;
140
141 Motcle motlu;
142 is >> motlu;
143 if (motlu != Motcle("{"))
144 {
145 Cerr << "We expected { to start to read the Probleme_Couple" << finl;
146 exit();
147 }
148
149 is >> motlu;
150 while (motlu!=Motcle("}")) //!< fin du readOn
151 {
152
153 if (motlu != Motcle("groupes"))
154 {
155 Cerr << "The keyword " << motlu << " is not understood" << finl;
156 exit();
157 }
158
159 if (nb_problemes())
160 {
161 Cerr << "We can associate problems to Probleme_Couple" << finl;
162 Cerr << "* either by \"associer prob_couple pb\" (in which case they are all in the same group)" << finl;
163 Cerr << "* either by the keyword \"groupes\" while reading the object Probleme_Couple" << finl;
164 Cerr << "but not both!" << finl;
165 }
166 assert(nb_problemes()==0);
167
168 LIST(LIST(Nom)) les_noms;
169 is >> les_noms;
170
171 groupes.resize_array(les_noms.size());
172 for (int i=0; i<les_noms.size(); i++)
173 {
174 groupes[i]=les_noms[i].size();
175 for (int j=0; j<les_noms[i].size(); j++)
176 {
177 Nom nom_pb=les_noms[i][j];
178 Objet_U& ob=Interprete::objet(nom_pb);
179 Probleme_base& pb=ref_cast(Probleme_base,ob);
180 ajouter(pb);
181 }
182 }
183
184 is >> motlu;
185 }
186
187 return is;
188}
