158double Echange_interne_impose::calcul_h_imp(const double h_gap, const double invLambda) const
159{
160 return 1./(1./h_gap + invLambda); //!< harmonic mean
161}

163// Compute e/lambda - this will be used:
164// - in calcul_h_imp to adjust for the half cell in VDF
165// - in Echange_interne_vargap for the computation of the h_gap correlation.
166void Echange_interne_impose::update_inv_lambda()
167{
168 bool isVDF = sub_type(Zone_VDF, frontiere_dis().zone_dis());
169 const Front_VF& fvf = ref_cast(Front_VF, frontiere_dis());
170
171 inv_lambda_ = 0.0;
172
173 if (isVDF)
174 {
175 DoubleTab& lambdas = lambda_ref_->valeurs();
176
177 const Zone_VDF& zon = ref_cast(Zone_VDF, frontiere_dis().zone_dis());
178 for (int numf=0; numf < fvf.nb_faces(); numf++)
179 {
180 int facei = fvf.num_face(numf);
181 int vois0 = zon.face_voisins(facei, 0);
182 int elem = (vois0 != -1) ? vois0 : zon.face_voisins(facei, 1);
183 inv_lambda_(numf) = zon.dist_norm_bord(facei)/lambdas(elem, 0);
184 }
185 }
186}

188void Echange_interne_impose::mettre_a_jour(double tps)
189{
190 Echange_externe_impose::mettre_a_jour(tps);
191 h_gap_.mettre_a_jour(tps);
192
193 update_inv_lambda();
194
195 bool isVDF = sub_type(Zone_VDF, frontiere_dis().zone_dis());
196
197 DoubleTab& h_impose = h_imp_.valeurs();
198 DoubleTab& h_gap = h_gap_.valeurs();
199
200// If VDF, fix h_imp to take into account half cell on the other side of the wall:
201 if (isVDF)
202 {
203 const Champ_front_calc_interne& t_ext = ref_cast(Champ_front_calc_interne, T_ext().valeur());
204 const IntTab& face_map = t_ext.face_map();
205 const Front_VF& fvf = ref_cast(Front_VF, frontiere_dis());
206
207 for (int numf=0; numf < fvf.nb_faces(); numf++)
208 {
209 int opp_face = face_map(numf); //!< opposite face
210 if (sub_type(Champ_front_uniforme, h_gap_.valeur()))
211 h_impose(numf,0) = calcul_h_imp(h_gap(0,0), inv_lambda_(opp_face)); //!< inv lambda from the **opposite** side
212 else
213 h_impose(numf,0) = calcul_h_imp(h_gap(numf,0), inv_lambda_(opp_face));
214 }
215 }
216 else
217 {
218 if (sub_type(Champ_front_uniforme, h_gap_.valeur()))
219 h_impose = h_gap(0,0);
220 else
221 h_impose = h_gap;
222 }
223}
