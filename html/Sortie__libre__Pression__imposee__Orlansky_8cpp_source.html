99void Sortie_libre_Pression_imposee_Orlansky::mettre_a_jour(double temps)
100{
101// Cerr << "Sortie_libre_Pression_imposee_Orlansky::mettre_a_jour("
102// <<temps<<")"<<finl;
103
104 Cond_lim_base::mettre_a_jour(temps);
105
106 assert(pression_interne.non_nul());
107 const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
108 int ndeb = le_bord.num_premiere_face();
109 int nb_faces_loc = le_bord.nb_faces();
110
111 DoubleTab& pre_bord=le_champ_front.valeurs();
112 DoubleTab& vit_ext=le_champ_ext.valeurs();
113 const Zone_VDF& zvdf=ref_cast(Zone_VDF,
114 pression_interne->zone_dis_base());
115 const DoubleTab& pre=pression_interne->valeurs();
116 const DoubleTab& vitesse=vitesse_interne->valeurs();
117
118 int face, compo;
119
120 for (face=ndeb; face<ndeb+nb_faces_loc; face++)
121 {
122 int i=face-ndeb;
123
124 int ori = zvdf.orientation(face);
125
126 pression_temps_moins_deux(i)=pression_temps_moins_un(i);
127 pression_temps_moins_un(i)=pre_bord(i);
128 pression_moins_un_temps_moins_deux(i)=
129 pression_moins_un_temps_moins_un(i);
130 pression_moins_un_temps_moins_un(i)=pression_moins_un(i);
131 pression_moins_deux_temps_moins_un(i)=pression_moins_deux(i);
132
133 int elem_un=zvdf.face_voisins(face,0);
134 if (elem_un < 0)
135 elem_un=zvdf.face_voisins(face,1);
136 int face_moins_un = zvdf.elem_faces(elem_un, ori);
137 if (face_moins_un==face)
138 face_moins_un = zvdf.elem_faces(elem_un, ori+dimension);
139 pression_moins_un[i] = pre[elem_un];
140
141 int elem_deux=zvdf.face_voisins(face_moins_un,0);
142 if (elem_deux == elem_un)
143 elem_deux=zvdf.face_voisins(face_moins_un,1);
144 pression_moins_deux[i] = pre[elem_deux];
145
146 double pre_m_un_t_m_deux=pression_moins_un_temps_moins_deux(i);
147 double pre_m_deux_t_m_un=pression_moins_deux_temps_moins_un(i);
148 double pre_m_un=pression_moins_un(i);
149
150 if( pre_m_un_t_m_deux==pre_m_un )
151 VPhiP(i)=0;
152 else
153 VPhiP(i)=(pre_m_un_t_m_deux-pre_m_un)/
154 (pre_m_un+pre_m_un_t_m_deux-2*pre_m_deux_t_m_un);
155 if(VPhiP(i)<=1.e-24) VPhiP(i)=0.0;
156 if(VPhiP(i)>1.) VPhiP(i)=1.0;
157 assert(VPhiP(i)<1.e12);
158
159 pre_bord[i] = (1-VPhiP(i))/(1+VPhiP(i))*pression_temps_moins_un(i)
160 + (2*VPhiP(i)/(1+VPhiP(i)))*pression_moins_un(i);
161
162 }
163 Debog::verifier_bord("Orlansky::mettre_a_jour() : pre bord : " , pre_bord,ndeb );
164// Debog::verifier_bord("Orlansky::mettre_a_jour() : VPhiP : " , VPhiP, ndeb );
165// Debog::verifier_bord("Orlansky::mettre_a_jour() : VPhiV avant : " , VPhiV ,ndeb);
166
167// Debog::verifier_bord("Orlansky::mettre_a_jour() : vitesse_moins_un_temps_moins_un : " , vitesse_moins_un_temps_moins_un, ndeb);
168// Debog::verifier_bord("Orlansky::mettre_a_jour() : vitesse_moins_deux : " , vitesse_moins_deux, ndeb);
169// Debog::verifier_bord("Orlansky::mettre_a_jour() : vitesse_moins_un : " , vitesse_moins_un, ndeb);
170
171
172 for(compo=0; compo<dimension; compo++)
173 for (face=ndeb; face<ndeb+nb_faces_loc; face++)
174 {
175 int i=face-ndeb;
176
177 int ori = zvdf.orientation(face);
178
179 vitesse_temps_moins_deux(i, compo)=vitesse_temps_moins_un(i, compo);
180 vitesse_temps_moins_un(i, compo)=vit_ext(i, compo);
181 vitesse_moins_un_temps_moins_deux(i, compo)=vitesse_moins_un_temps_moins_un(i, compo);
182
183 vitesse_moins_un_temps_moins_un(i, compo)=vitesse_moins_un(i, compo);
184 vitesse_moins_deux_temps_moins_un(i, compo)=vitesse_moins_deux(i, compo);
185
186 int elem_un=zvdf.face_voisins(face,0);
187 if (elem_un < 0)
188 elem_un=zvdf.face_voisins(face,1);
189 int face_moins_un = zvdf.elem_faces(elem_un, ori);
190 if (face_moins_un==face)
191 face_moins_un = zvdf.elem_faces(elem_un, ori+dimension);
192 double vit=0.5*(vitesse(zvdf.elem_faces(elem_un, compo))
193 +vitesse(zvdf.elem_faces(elem_un, compo+dimension)));
194
195 vitesse_moins_un(i, compo) = vit;
196
197 int elem_deux=zvdf.face_voisins(face_moins_un,0);
198 if (elem_deux == elem_un)
199 elem_deux=zvdf.face_voisins(face_moins_un,1);
200 vit=0.5*(vitesse(zvdf.elem_faces(elem_deux, compo))
201 +vitesse(zvdf.elem_faces(elem_deux, compo+dimension)));
202 vitesse_moins_deux(i, compo) = vit;
203
204 double pre_m_un_t_m_deux=vitesse_moins_un_temps_moins_deux(i, compo);
205 double pre_m_deux_t_m_un=vitesse_moins_deux_temps_moins_un(i, compo);
206 double pre_m_un=vitesse_moins_un(i, compo);
207
208 if( pre_m_un_t_m_deux==pre_m_un )
209 VPhiV(i, compo)=0;
210 else
211 VPhiV(i, compo)=(pre_m_un_t_m_deux-pre_m_un)/
212 (pre_m_un+pre_m_un_t_m_deux-2*pre_m_deux_t_m_un);
213 if(VPhiV(i, compo)<=1.e-24) VPhiV(i, compo)=0.0;
214 if(VPhiV(i, compo)>1.) VPhiV(i, compo)=1.0;
215 assert(VPhiV(i, compo)<1.e12);
216
217 vit_ext(i, compo) = (1-VPhiV(i, compo))/(1+VPhiV(i, compo))*vitesse_temps_moins_un(i, compo)
218 + (2*VPhiV(i, compo)/(1+VPhiV(i, compo)))*vitesse_moins_un(i, compo);
219
220 }
221 Debog::verifier_bord("Orlansky::mettre_a_jour() : vit_ext : " , vit_ext,ndeb);
222
223// Debog::verifier_bord("Orlansky::mettre_a_jour() : vitesse_moins_un : " ,vitesse_moins_un ,ndeb);
224// Debog::verifier_bord("Orlansky::mettre_a_jour() : vitesse_moins_deux : " ,vitesse_moins_deux ,ndeb);
225 Debog::verifier_bord("Orlansky::mettre_a_jour() : VPhiV : " , VPhiV ,ndeb);
226}
