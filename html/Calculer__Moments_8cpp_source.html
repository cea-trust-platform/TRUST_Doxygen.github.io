34Entree& Calculer_Moments::interpreter_(Entree& is)
35{
36 Motcle nom_var;
37 associer_domaine(is);
38 c.resize_array(dimension);
39 Domaine& dom=domaine();
40 is >> nom_var;
41 Cerr << nom_var << finl;
42 dom.zone(0).Moments_a_imprimer()=1;
43 if(nom_var=="calcul")
44 {
45 Cerr << "Calculation of center of gravity " << nom_var<< finl;
46 dom.zone(0).calculer_mon_centre_de_gravite(c);
47 }
48 else if (nom_var=="centre_de_gravite")
49 {
50 Cerr << "Center of gravity imposed, of dimension " << dimension << finl;
51 is >> c[0];
52 is >> c[1];
53 if (dimension==3)
54 is >> c[2];
55 dom.zone(0).exporter_mon_centre_de_gravite(c);
56 }
57 else
58 {
59 Cerr << nom_var << " is not recognized." << finl;
60 exit();
61 }
62 return is;
63}
