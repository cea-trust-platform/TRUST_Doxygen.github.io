40char * InputCommBuffer::reserve_buffer(int bufsize)
41{
42 assert(stream_ == 0);
43 size_ = bufsize;
44 if (size_ > memorysize_)
45 {
46 delete[] buffer_;
47 memorysize_ = size_ * 2;

53void InputCommBuffer::create_stream()
54{
55 assert(stream_ == 0);
56 assert(size_ >= 0);
57// (Voir documentation de istrstream) :
58// Si size_ > 0 istrstream(buf, size_) cree un buffer de taille size_.
59// Si size_ = 0, istrstream(buf, size_) suppose que buf est une chaine
60// de caracteres terminee par zero. Pour creer un stream de taille nulle,
61// il faut donc faire ceci:
62 if (size_ == 0)
63 buffer_[0] = 0;
64 stream_ = new istringstream(std::string(buffer_, size_));
65 set_istream(stream_);
66}

68void InputCommBuffer::create_stream_from_output_stream(OutputCommBuffer& output_buf)
69{
70 assert(stream_ == 0);
71 const char nullchar = 0;
72 const char * buffer = output_buf.get_buffer();
73 const int size = output_buf.get_buffer_size();
74// (Voir documentation de istrstream) :
75// Si size_ > 0 istrstream(buf, size_) cree un buffer de taille size_.
76// Si size_ = 0, istrstream(buf, size_) suppose que buf est une chaine
77// de caracteres terminee par zero. Pour creer un stream de taille nulle,
78// il faut donc faire ceci:
79 if (size == 0)
80 buffer = &nullchar;
81 stream_ = new istringstream(std::string(buffer, size));
82 set_istream(stream_);
83}
