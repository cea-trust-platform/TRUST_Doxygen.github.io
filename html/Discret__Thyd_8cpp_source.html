342void Discret_Thyd::concentration(const Schema_Temps_base& sch,
343 Zone_dis& z, Champ_Inc& ch,
344 int nb_constituants, const Nom nom_champ) const
345{
346 Cerr << "Concentration discretization " << finl;
347 discretiser_champ("temperature",z.valeur(),nom_champ,"%",nb_constituants,sch.nb_valeurs_temporelles(),sch.temps_courant(),ch);
348 ch->nommer(nom_champ);
349 if (nb_constituants > 1)
350 {
351 ch->fixer_nature_du_champ(multi_scalaire);
352 Noms noms(nb_constituants);
353 for (int i=0; i<nb_constituants; i++)
354 {
355 noms[i] = nom_champ;
356 Nom param(i);
357 noms[i] += param;
358 ch->fixer_nom_compo(i, noms[i]);
359 }
360 }
361}
