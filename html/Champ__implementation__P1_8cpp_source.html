20double Champ_implementation_P1::form_function(const ArrOfDouble& position, const IntTab& les_elems, const DoubleTab& nodes, ArrOfInt& index, int cell, int ddl) const
21{
22 int nb_nodes_per_cell = les_elems.dimension(1);
23 assert(ddl < nb_nodes_per_cell);
24
25 for (int i=0; i<nb_nodes_per_cell; i++)
26 index[i] = les_elems(cell,(i+ddl)%nb_nodes_per_cell);
27
28 if (nb_nodes_per_cell==2)
29 {
30 double num=0.;
31 double den=0.;
32 for (int d=0; d<Objet_U::dimension; d++)
33 {
34 num+=(position[d]-nodes(index[0],d))*(position[d]-nodes(index[0],d));
35 den+=(nodes(index[1],d)-nodes(index[0],d))*(nodes(index[1],d)-nodes(index[0],d));
36 }
37 double res= 1.-sqrt(num/den);
38// Cerr<<"ii "<<res<<" "<<index[0]<<" "<<ddl<<" "<<(ddl)%nb_nodes_per_cell<<finl;
39 return res;
40 }
41 double num = 0.;
42 double den = 0.;
43
44 if (Objet_U::dimension == 2)
45 {
46 den = (nodes(index[2],0) - nodes(index[1],0)) * (nodes(index[0],1) - nodes(index[1],1))
47 - (nodes(index[2],1) - nodes(index[1],1)) * (nodes(index[0],0) - nodes(index[1],0));
48
49 num = (nodes(index[2],0) - nodes(index[1],0)) * (position[1] - nodes(index[1],1))
50 - (nodes(index[2],1) - nodes(index[1],1)) * (position[0] - nodes(index[1],0));
51 }
52 else if (Objet_U::dimension == 3)
53 {
54 double xp = (nodes(index[2],1) - nodes(index[1],1)) * (nodes(index[0],2) - nodes(index[1],2))
55 - (nodes(index[2],2) - nodes(index[1],2)) * (nodes(index[0],1) - nodes(index[1],1));
56
57 double yp = (nodes(index[2],2) - nodes(index[1],2)) * (nodes(index[0],0) - nodes(index[1],0))
58 - (nodes(index[2],0) - nodes(index[1],0)) * (nodes(index[0],2) - nodes(index[1],2));
59
60 double zp = (nodes(index[2],0) - nodes(index[1],0)) * (nodes(index[0],1) - nodes(index[1],1))
61 - (nodes(index[2],1) - nodes(index[1],1)) * (nodes(index[0],0) - nodes(index[1],0));
62
63 den = xp * (nodes(index[3],0) - nodes(index[1],0))
64 + yp * (nodes(index[3],1) - nodes(index[1],1))
65 + zp * (nodes(index[3],2) - nodes(index[1],2));
66
67 xp = (nodes(index[2],1) - nodes(index[1],1)) * (position[2] - nodes(index[1],2))
68 - (nodes(index[2],2) - nodes(index[1],2)) * (position[1] - nodes(index[1],1));
69
70 yp = (nodes(index[2],2) - nodes(index[1],2)) * (position[0] - nodes(index[1],0))
71 - (nodes(index[2],0) - nodes(index[1],0)) * (position[2] - nodes(index[1],2));
72
73 zp = (nodes(index[2],0) - nodes(index[1],0)) * (position[1] - nodes(index[1],1))
74 - (nodes(index[2],1) - nodes(index[1],1)) * (position[0] - nodes(index[1],0));
75
76 num = xp * (nodes(index[3],0) - nodes(index[1],0))
77 + yp * (nodes(index[3],1) - nodes(index[1],1))
78 + zp * (nodes(index[3],2) - nodes(index[1],2));
79 }
80 else
81 {
82 Cerr << "Error in Champ_implementation_P1::form_function : Invalid dimension" << finl;
83 Process::exit();
84 }
85
86 assert(den != 0.);
87 double result = num/den;
88
89 if ((result < -Objet_U::precision_geom) || (result > 1.+Objet_U::precision_geom))
90 {
91 Cerr << "WARNING: The barycentric coordinate of point :" << finl;
92 Cerr << "x= " << position[0] << " y=" << position[1];
93 if (Objet_U::dimension == 3)
94 {
95 Cerr << " z=" << position[3];
96 }
97 Cerr << finl;
98 Cerr << "is not between 0 and 1 : " << result << finl;
99 Cerr << "On the element " << cell << " of the processor " << Process::me() << finl;
100
101#ifndef NDEBUG
102 Process::exit();
103#endif
104
105 }
106
107 return result;
108}
109void Champ_implementation_P1::value_interpolation(const DoubleTab& positions, const ArrOfInt& cells, const DoubleTab& values, DoubleTab& resu, int ncomp) const
110{
111 const Zone& zone = get_zone_geom();
112 const IntTab& les_elems = zone.les_elems();
113 const DoubleTab& nodes = zone.domaine().les_sommets();
114 const int nb_nodes_per_cell = zone.nb_som_elem(), N = resu.line_size();
115 ArrOfInt index(nb_nodes_per_cell);
116 ArrOfDouble position(Objet_U::dimension);
117 resu = 0;
118 for (int ic=0; ic<cells.size_array(); ic++)
119 {
120 int cell = cells[ic];
121 if (cell<0) continue;
122 for (int k=0; k<Objet_U::dimension; k++)
123 position[k] = positions(ic,k);
124
125 assert(cell >= 0);
126 assert(cell < les_elems.dimension_tot(0));
127 if (ncomp != -1)
128 {
129 for (int j = 0; j < nb_nodes_per_cell; j++)
130 {
131 int node = les_elems(cell, j);
132 resu(ic) += values(node, ncomp) * form_function(position,les_elems,nodes,index,cell,j);
133 }
134 }
135 else
136 {
137
138 assert(values.line_size() == N);
139 for (int j = 0; j < nb_nodes_per_cell; j++)
140 {
141 double weight = form_function(position,les_elems,nodes,index,cell,j);
142 int node = les_elems(cell, j);
143 for (int k = 0; k < N; k++)
144 resu(ic, k) += values(node, k) * weight;
145 }
146 }
147 }
148}

157//! Initialise le tableau de valeurs aux sommets du domaine dom a partir de valeurs lues dans "input".<br>(on dimensionne, associe la structure parallele et remplit les valeurs reelles et virtuelles)<br>Le fichier doit avoir le format suivant (n est le nombre de valeurs nodales<br>stockees, x, y, z sont les coordonnees des sommets, compo1... sont les valeurs<br>des composantes.<br>n (int)<br>x y [z] compo1 [compo2 [compo3 ... ]] (type double)
158void Champ_implementation_P1::init_from_file(DoubleTab& val, const Domaine& dom,
159 int nb_comp, double tolerance, Entree& input)
160{
161 val.resize(0, nb_comp);
162 dom.creer_tableau_sommets(val, Array_base::NOCOPY_NOINIT);
163
164// Construction d'un octree avec les sommets du domaine:
165 const DoubleTab& coord = dom.coord_sommets();
166 Octree_Double octree;
167 octree.build_nodes(coord, 0 /* do not include virtual nodes */);
168
169// Lecture des valeurs dans input
170 int nb_val_lues;
171 input >> nb_val_lues;
172 const int dim = coord.dimension(1);
173 const int tmp_size = dim + nb_comp;
174 ArrOfDouble tmp(tmp_size);
175 ArrOfDouble node_coord; //!< points to the dim first elements of "tmp" (coordinates of the node)
176 node_coord.ref_array(tmp, 0 /* start index */, dim /* size */);
177 ArrOfInt node_list;
178 node_list.set_smart_resize(1);
179
180 ArrOfInt count(dom.nb_som()); //!< number of times this coordinate has been found
181
182 for (int i_val = 0; i_val < nb_val_lues; i_val++)
183 {
184 input.get(tmp.addr(), tmp_size);
185// first call returns more points (some might be at a larger distance)
186 octree.search_elements_box(node_coord, tolerance, node_list);
187 octree.search_nodes_close_to(node_coord, coord, node_list, tolerance);
188 const int n = node_list.size_array();
189 if (n > 1)
190 {
191 Cerr << "Error in Champ_som_lu::readOn: point " << node_coord
192 << " corresponds to " << node_list.size_array()
193 << " nodes in the geometry within the specified tolerance" << finl;
194 Process::exit();
195 }
196 if (n == 1)
197 {
198 const int node_index = node_list[0];
199 count[node_index]++;
200 for (int i = 0; i < nb_comp; i++)
201 val(node_index, i) = tmp[dim + i];
202 }
203 else
204 {
205// Ce sommet n'est pas sur ce processeur...
206 }
207 }
208
209 if (dom.nb_som() > 0 && min_array(count) == 0)
210 {
211 Cerr << "Error in Champ_som_lu::readOn: some coordinates were not found in the file" << finl;
212 Process::exit();
213 }
214 val.echange_espace_virtuel();
215}

267/** \return double<br>

276{

285 Cerr<<"The number of nodes by element " <<nb_som_elem<<" does not correspond to a treated situation in the coord_barycentrique_P1 function."<<finl;
286 Process::exit();
287 return 0.;
288}
