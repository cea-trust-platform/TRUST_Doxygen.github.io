91double Schema_Temps_base::computeTimeStep(bool& is_stop) const
92{
93// reevaluation de dt_max_ si fonction du temps
94 if (dt_max_str_ != Nom())
95 {
96 dt_max_fn_.setVar(0, temps_courant());
97 dt_max_ = dt_max_fn_.eval();
98 }
99
100 is_stop=false;
101// Correction en premier du pas de temps
102 double dt = dt_stab_;
103 if (!corriger_dt_calcule(dt)) //!< Change le contenu de dt
104 is_stop=true;
105 dt = std::min(dt, dt_failed_ / sqrt(2)); //pour provoquer une baisse de dt en cas d'echec a la resolution precedente
106 if (temps_courant_ > temps_precedent_)
107 dt = std::min(dt, (temps_courant_ - temps_precedent_) * dt_gf_); //pour ne pas remonter dt trop vite (comme facsec)
108
109// Mise a jour immediate de l'attribut dt_ afin que pas_de_temps()
110// soit a jour tout le temps (en particulier au moment du postraitement)
111// Ce n'etait pas le cas pour les versions <= 1.6.3 et les champs dependant
112// de dt n'etaient pas juste si facsec<>1
113// dt_ = dt;
114
115 if (this->stop())
116 {
117 is_stop=true;
118 return 0;
119 }
120 else
121 return dt;
122}

178int Schema_Temps_base::limpr() const
179{
180 if (dt_impr_<=0)
181 return 0; //!< Negative value for dt_impr : we never print
182 if (nb_pas_dt_==0)
183 return 0; //!< No solve cause 0 time step : we print nothing
184 if (dt_impr_<=dt_)
185 return 1; //!< We print every time step if dt_impr is lower than dt
186 if (nb_pas_dt_<=1)
187 return 1; //!< We print the first time step
188 if (tmax_<=temps_courant_ || nb_pas_dt_max_<=nb_pas_dt_ || stationnaires_atteints_)
189 return 1; //!< We print the last time step
190
191// 26/01/2010 : On utilise desormais la fonction "modf(operation , & partie entiere)" de math.h
192// Cette fonction decompose le resultat de "operation" en une partie entiere et une partie decimale.
193// Ainsi, on ne transforme plus un double en int avec tout les risques que cela comporte.
194// ex : modf(9/7,&i) donne 1.000000 pour la partie entiere et 0.285714 pour le partie decimale
195//
196// epsilon permet d'assurer que le resultat de l'operation est independant de la precision machine :
197// ex : 0.99999999/1.00000000 peut donner 0.99999999 ou 1.00000000 suivant les machines
198// alors que
199// ex : 0.99999999/1.00000000 + 1.e-8 donne tjrs 1.00000000.
200 double i, j, epsilon = 1.e-8;
201 modf(temps_courant_/dt_impr_ + epsilon, &i);
202 modf(temps_precedent_/dt_impr_ + epsilon, &j);
203 return ( i>j );
204}

206void Schema_Temps_base::validateTimeStep()
207{
208 statistiques().begin_count(mettre_a_jour_counter_);
209// Update the problem:
210 Probleme_base& problem=pb_base();
211 problem.mettre_a_jour(temps_courant_+dt_);
212 if (je_suis_maitre())
213 {
214 if (!disable_dt_ev())
215 {
216 if (schema_impr())
217 dt_ev_ << finl << temps_courant_ << "\t " << dt_ << "\t " << facsec_ <<"\t " << residu_ << "\t " << dt_stab_ << "\t ";
218 for (int i=0; i<pb_base().nombre_d_equations(); i++)
219 pb_base().equation(i).imprime_residu(dt_ev_);
220 }
221
222// Impression du temps CPU estime restant
223 if (schema_impr())
224 {
225 if ((residu_>0)&&(residu_old_slope_>0))
226 cumul_slope_+=(log(residu_)-log(residu_old_slope_))/dt_;
227 residu_old_slope_=residu_;
228 }
229 if (schema_impr()&&(nb_pas_dt()>0 && pas_de_temps()>0))
230 {
231// On calcule le temps CPU moyen par pas de temps, inconvenient il peut varier fortement au cours du temps si divergence du calcul ou au contraire acceleration
232// Mais Statistiques ne permet pas d'avoir le temps CPU du dernier pas de temps (last_time appele ici renverrait le temps CPU depuis le debut du pas de temps)
233// double cpu_per_timestep = statistiques().last_time(temps_total_execution_counter_) / nb_pas_dt();
234 double nb_pas_selon_tmax = (temps_max()-temps_courant()) / pas_de_temps();
235 double nb_pas_selon_nb_pas_dt_max = nb_pas_dt_max() - nb_pas_dt();
236 double nb_pas_avant_fin= std::min(nb_pas_selon_tmax,nb_pas_selon_nb_pas_dt_max);
237// double seconds_to_finish = nb_pas_avant_fin * cpu_per_timestep;
238 double dpercent=(1.-nb_pas_avant_fin/(nb_pas_avant_fin+ nb_pas_dt())); //!< marche meme si c'est ltemps max qui limite si la pente est >0 on diverge ....
239//
240 if ((seuil_statio_>0)&&(cumul_slope_<-1e-20) && seuil_statio_ < residu_) //!< dans un pb_couple on peut avoir (seuil_statio_ > residu) pour un des problemes
241 {
242 double distance= (-log(residu_+1e-20)+log(seuil_statio_))/(cumul_slope_)* nb_pas_dt();
243
244// Cerr<<distance<<" DDDDDD"<<finl;
245 double dpercent2=temps_courant()/(temps_courant()+distance);
246 dpercent=std::max(dpercent,dpercent2);
247 }
248 int percent=int(dpercent*100);
249
250 if (limpr() )
251 {
252 double seconds_to_finish = statistiques().last_time(temps_total_execution_counter_)*(1.-dpercent)/dpercent;
253 int integer_limit=(int)(pow(2.0,(double)((sizeof(True_int)*8)-1))-1);
254 if (seconds_to_finish<integer_limit)
255 {
256 int h = int(seconds_to_finish/3600);
257 int mn = int((seconds_to_finish-3600*h)/60);
258 int s = int(seconds_to_finish-3600*h-60*mn);
259 Cout << finl << "Estimated CPU time to finish the run (according to " << (nb_pas_selon_tmax<nb_pas_selon_nb_pas_dt_max?"tmax":"nb_pas_dt_max") << " value) : ";
260 if (seconds_to_finish<1)
261 Cout << seconds_to_finish << " s";
262 else
263 Cout << h << "h" << mn << "mn" << s << "s";
264
265 Cout << ". Progress: "<<(percent)<< finl;
266 }
267 }
268 if (!disable_progress() && Process::je_suis_maitre())
269 progress_<< (percent)<< finl;
270 }
271 }
272// Update time scheme:
273 mettre_a_jour();
274 statistiques().end_count(mettre_a_jour_counter_);
275 dt_failed_ = DBL_MAX;
