262//! Cas gaz Reel : doit recalculer l'enthalpie a partir de la pression et la temperature
263double Loi_Etat_rhoT_GR_QC::calculer_H(double Pth_, double T_) const
264{
265 double res=0;
266 if (R==-1)
267 {
268// il faut resoudre c0-T_ + c1.H + c2.H^2 +...=0
269 int i,j, it,max_iter = 1000;
270 double eps = 1.e-6;
271 DoubleVect coef(PolyT_.line_size());
272 coef = 0;
273
274 for (i=0 ; i<PolyT_.dimension(0) ; i++)
275 for (j=0 ; j<PolyT_.line_size() ; j++)
276 coef(j) += PolyT_(i,j) * pow(Pth_,i);
277
278 coef(0) -= T_;
279 double dH, H = -coef(0)/coef(1), num=coef(0), den=0;
280 for (j=1 ; j<PolyT_.line_size() ; j++)
281 {
282 num += coef(j) * pow(H,j);
283 den += j*coef(j) * pow(H,j-1);
284 }
285 dH = num/den;
286 H -= dH;
287
288 it=0;
289 while (dH/H>eps && it<max_iter)
290 {
291 num=coef(0);
292 den=0;
293 for (j=1 ; j<PolyT_.line_size() ; j++)
294 {
295 num += coef(j) * pow(H,j);
296 den += j*coef(j) * pow(H,j-1);
297 }
298 dH = num / den;
299 H -= dH;
300 it++;
301 }
302
303 if (dH/H>eps)
304 {
305 Cerr<<"PB Loi_Etat_rhoT_GR_QC::calculer_H: resolution Newton impossible "<<finl;
306 Cerr<<"Pth= "<<Pth_<<" t="<<T_<<" H="<<H<<finl;
307 abort();
308 }
309 res = H;
310 }
311 else
312 {
313// debuggage
314 res = Cp_*T_;
315 }
316 return res;
317}
