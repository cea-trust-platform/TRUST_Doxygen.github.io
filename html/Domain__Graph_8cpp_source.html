27static void construire_connectivite_real_som_virtual_elem(const int nb_sommets,
28 const IntTab& les_elems,
29 Static_Int_Lists& som_elem,
30 const IntTab& elem_virt_pe_num,
31 const ArrOfInt& offsets)
32{
33// Nombre d'elements du domaine
34 const int nb_elem = les_elems.dimension_tot(0);
35 const int local_nb_elem = les_elems.dimension(0);
36// Nombre de sommets par element
37 const int nb_sommets_par_element = les_elems.dimension(1);
38
39// Construction d'un tableau initialise a zero : pour chaque sommet,
40// nombre d'elements voisins de ce sommet
41 ArrOfInt nb_elements_voisins(nb_sommets);
42
43// Premier passage : on calcule le nombre d'elements voisins de chaque
44// sommet pour creer la structure de donnees
45 int elem, i;
46
47// real elements
48 for (elem = 0; elem < nb_elem; elem++)
49 {
50 for (i = 0; i < nb_sommets_par_element; i++)
51 {
52 int sommet = les_elems(elem, i);
53 if(sommet >= nb_sommets) continue; //skipping virtual node
54// GF cas des polyedres
55 if (sommet==-1) break;
56 nb_elements_voisins[sommet]++;
57 }
58 }
59
60 som_elem.set_list_sizes(nb_elements_voisins);
61
62// On reutilise le tableau pour stocker le nombre d'elements dans
63// chaque liste pendant qu'on la remplit
64 nb_elements_voisins = 0;
65
66// Remplissage du tableau des elements voisins.
67 for (elem = 0; elem < nb_elem; elem++)
68 {
69 for (i = 0; i < nb_sommets_par_element; i++)
70 {
71 int sommet = les_elems(elem, i);
72 if(sommet >= nb_sommets) continue;
73// GF cas des polyedres
74 if (sommet==-1) break;
75 int n = (nb_elements_voisins[sommet])++;
76
77 int elem_num_global = -1;
78 if(elem>= local_nb_elem)
79 {
80 int proc_of_elem = elem_virt_pe_num(elem-local_nb_elem, 0);
81 int elem_number_on_local_proc = elem_virt_pe_num(elem-local_nb_elem, 1);
82 elem_num_global = elem_number_on_local_proc + offsets[proc_of_elem];
83 }
84 else
85 elem_num_global = elem + offsets[Process::me()];
86
87 som_elem.set_value(sommet, n, elem_num_global);
88 }
89 }
90
91// Tri de toutes les listes dans l'ordre croissant
92 som_elem.trier_liste(-1);
93}

95void Domain_Graph::construire_graph_from_segment(const Domaine& dom,
96 const int use_weights)
97{
98 const IntTab& liaisons = dom.zone(0).les_elems();
99
100// ****************************************************************
101// PREMIERE ETAPE: calcul du nombre de vertex et edges du graph:
102 int nb_edges = liaisons.size_array(); //!< 2 liens par liaison
103 int nb_elem=liaisons.local_max_vect()+1; //!< mouif
104 nvtxs = nb_elem;
105 xadj.resize_array(nb_elem+1);
106 vwgts.resize_array(0);
107 if (! use_weights)
108 {
109 weightflag = 0;
110 ewgts.resize_array(0);
111 }
112 else
113 {
114 abort();
115 weightflag = 1;
116 ewgts.resize_array(nb_edges);
117 }
118
119// on construit connectivite item item
120 IntTab stencyl(0,2);
121 stencyl.set_smart_resize(1);
122 int size=0;
123 int nbl=liaisons.dimension(0);
124 for (int i=0; i<nbl; i++)
125 {
126 stencyl.resize(size+2,2);
127 int n1=liaisons(i,0);
128 int n2=liaisons(i,1);
129 {
130 stencyl(size,0)=n1;
131 stencyl(size,1)=n2;
132
133 size++;
134 stencyl(size,0)=n2;
135 stencyl(size,1)=n1;
136
137 size++;
138 }
139 }
140 tableau_trier_retirer_doublons(stencyl);
141 Matrice_Morse A;
142 Matrix_tools::allocate_morse_matrix(nb_elem,nb_elem,stencyl,A);
143
144 nb_edges=A.get_tab2().size_array(); //!< des liens peuvent etre doubles
145 nedges = nb_edges;
146 adjncy.resize_array(nb_edges);
147//
148 assert(A.get_tab1().size_array()==nvtxs+1);
149 for (int c=0; c<nvtxs+1; c++)
150 {
151 xadj[c]=A.get_tab1()(c)-1;
152 }
153 for (int c=0; c<nedges; c++)
154 {
155 adjncy[c]=A.get_tab2()(c)-1;
156 }
157
158}

160// Si use_weights, on pondere les liens entre les elements periodiques
161// pour les forcer a etre sur le meme processeur. Cela diminue le nombre
162// de corrections a faire ensuite (voir (***))
163void Domain_Graph::construire_graph_elem_elem(const Domaine& dom,
164 const Noms& liste_bords_periodiques,
165 const int use_weights,
166 Static_Int_Lists& graph_elements_perio)
167{
168 Static_Int_Lists som_elem;
169 const Zone& zone = dom.zone(0);
170 const Elem_geom_base& type_elem = zone.type_elem().valeur();
171 IntTab faces_element_reference;
172 const int is_regular =
173 type_elem.get_tab_faces_sommets_locaux(faces_element_reference);
174 if (! is_regular)
175 {
176 Cerr << "Error in Domain_Graph::construire_graph_elem_elem\n"
177 << " The type of element is not supported" << finl;
178 ref_cast(Poly_geom_base,type_elem).get_tab_faces_sommets_locaux(faces_element_reference,0);
179 }
180 int nb_faces_par_element = faces_element_reference.dimension(0);
181 const int nb_sommets_par_face = faces_element_reference.dimension(1);
182
183 const IntTab& elem_som = zone.les_elems();
184 const int nb_elem = zone.nb_elem();
185
186 ArrOfInt offsets(Process::nproc());
187 offsets = mppartial_sum(nb_elem);
188 envoyer_all_to_all(offsets, offsets);
189 int my_offset = offsets[Process::me()];
190
191 Cerr << " Construction of the som_elem connectivity" << finl;
192 if(Process::nproc() > 1)
193 {
194// what we're doing here is not equivalent to construire_connectivite_som_elem with virtual elements set to 1
195// in the latter, we also build the connectivity for virtual nodes
196// here, we add the connectivity of real nodes only with virtual elements
197// + we want som_elem to contain global numerotation for elements
198 IntTab elem_virt_pe_num;
199 zone.construire_elem_virt_pe_num(elem_virt_pe_num);
200 construire_connectivite_real_som_virtual_elem(dom.nb_som(),
201 elem_som,
202 som_elem,
203 elem_virt_pe_num,
204 offsets);
205
206 }
207 else
208 construire_connectivite_som_elem(dom.nb_som(),
209 elem_som,
210 som_elem,
211 0 /* ne pas inclure les elements virtuels */);
212
213
214 int nb_connexions_perio = 0;
215 if (liste_bords_periodiques.size() > 0)
216 {
217 Cerr << " Construction of graph connectivity for periodic boundaries" << finl;
218 nb_connexions_perio = Partitionneur_base::calculer_graphe_connexions_periodiques(dom.zone(0),
219 liste_bords_periodiques,
220 som_elem,
221 my_offset,
222 graph_elements_perio);
223 }
224
225// ****************************************************************
226// PREMIERE ETAPE: calcul du nombre de vertex et edges du graph:
227
228// Nombre total de faces de bord:
229 const int nb_faces_bord = zone.nb_faces_frontiere();
230
231// Chaque element du maillage est un "vertex" du graph.
232// Les "edges" du graph relient chaque element a ses voisins par une face.
233// Il y a autant d'edges que de faces ayant deux voisins, fois 2
234// Formule classique: nb_faces internes = nnn/2 avec :
235 int nnn = nb_elem * nb_faces_par_element - nb_faces_bord + nb_connexions_perio;
236 if (sub_type(Poly_geom_base,zone.type_elem().valeur()))
237 {
238 const Poly_geom_base& poly=ref_cast(Poly_geom_base,zone.type_elem().valeur());
239 nnn= poly.get_somme_nb_faces_elem() - nb_faces_bord + nb_connexions_perio;
240 }
241
242 const int nb_edges = nnn + nb_faces_bord;
243
244 nvtxs = nb_elem + zone.nb_faces_joint(); //each joint face is linked to a virtual element
245 xadj.resize_array(nb_elem+1);
246 xadj = -1;
247 adjncy.resize_array(nb_edges+nb_faces_bord);
248 adjncy = -1;
249 edgegsttab.resize_array(nb_edges+nb_faces_bord);
250 edgegsttab = -1;
251
252 vwgts.resize_array(0); //NULL
253 if (! use_weights)
254 {
255 weightflag = 0;
256 ewgts.resize_array(0); //NULL
257 }
258 else
259 {
260 weightflag = 1;
261 ewgts.resize_array(nb_edges + nb_faces_bord);
262 }
263
264 vtxdist.resize_array(Process::nproc()+1);
265 for(int p = 0; p < Process::nproc(); p++)
266 vtxdist[p] = offsets[p];
267 vtxdist[Process::nproc()] = Process::mp_sum(nb_elem);
268
269 Cerr << " Construction of the elem_elem connectivity" << finl;
270// ***************************************************************
271// DEUXIEME ETAPE: remplissage du graph
272// Algorithme: Pour chaque element, boucle sur les faces de l'element
273// et on ajoute un "edge" entre l'element et ses voisins. On cherche
274// l'element voisin par une face a l'aide de la fonction find_adjacent_elements.
275//
276// Deux tableaux de travail:
277 ArrOfInt une_face(nb_sommets_par_face); //!< Les sommets de la face en cours
278 ArrOfInt voisins; //!< Les elements voisins d'une_face
279 voisins.set_smart_resize(1);
280
281 int error = 0;
282 int edge_count = 0;
283 int i_elem;
284 for (i_elem = 0; i_elem < nb_elem; i_elem++)
285 {
286 xadj[i_elem] = edge_count;
287 int i_face;
288
289 if (!is_regular)
290 {
291 ref_cast(Poly_geom_base,type_elem).get_tab_faces_sommets_locaux(faces_element_reference,i_elem);
292 int nb_faces_elem = faces_element_reference.dimension(0);
293 while ( faces_element_reference(nb_faces_elem-1,0)==-1)
294 nb_faces_elem--;
295 nb_faces_par_element= nb_faces_elem;
296 }
297 for (i_face = 0; i_face < nb_faces_par_element; i_face++)
298 {
299// Construction de cette face de l'element:
300// (indice des sommets de la face dans le domaine)
301 {
302 int i;
303 for (i = 0; i < nb_sommets_par_face; i++)
304 {
305 const int i_som = faces_element_reference(i_face, i);
306 if (i_som<0)
307 une_face[i] = i_som;
308 else
309 {
310 const int sommet = elem_som(i_elem, i_som);
311 une_face[i] = sommet;
312 }
313 }
314 }
315// Recherche des elements voisins de cette face:
316// (indices des elements contenant les sommets de la face)
317 find_adjacent_elements(som_elem, une_face, voisins);
318
319 const int nb_voisins = voisins.size_array();
320 int elem_voisin = -1;
321 switch (nb_voisins)
322 {
323 case 0:
324 {
325// Aucun voisin: erreur interne
326 error = 3;
327 break;
328 }
329 case 1:
330 {
331// Un seul voisin, c'est une face frontiere
332 const int elem = voisins[0];
333 if (elem != i_elem + my_offset)
334 error = 3; //!< l'element i_elem n'est pas voisin: erreur interne
335 else
336 elem_voisin = -1;
337 break;
338 }
339 case 2:
340 {
341// Le cas le plus courant:
342 const int elem0 = voisins[0];
343 const int elem1 = voisins[1];
344 if (elem0 == i_elem + my_offset) //neighbours contain global numerotation
345 elem_voisin = elem1;
346 else if (elem1 == i_elem + my_offset)
347 elem_voisin = elem0;
348 else
349 error = 3; //!< l'element i_elem n'est pas voisin: erreur interne
350 break;
351 }
352 default:
353 {
354// Plus de deux voisins
355 error = 2;
356 }
357 }
358
359 if (error)
360 break;
361
362 if (elem_voisin >= 0)
363 {
364 if (edge_count >= nb_edges)
365 {
366 error = 1;
367 break;
368 }
369 adjncy[edge_count] = elem_voisin;
370
371 if (use_weights)
372 {
373// internal connection have more weight:
374// it's better if the generated partition is not dispatched on several processors
375 if( my_offset <= elem_voisin && elem_voisin < nb_elem+my_offset) //neighbour belongs to me = strong connection
376 ewgts[edge_count] = 4;
377 else
378 ewgts[edge_count] = 1; //!< -1 peut faire des partitions discontinues ou pas equilibrees du tout
379
380 }
381
382 edge_count++;
383 }
384 }
385// Ajout des connexions supplementaires pour les faces periodiques
386 if (nb_connexions_perio > 0)
387 {
388 const int n = graph_elements_perio.get_list_size(i_elem);
389 for (int i = 0; i < n; i++)
390 {
391 const int elem_voisin = graph_elements_perio(i_elem, i);
392 if (edge_count >= nb_edges)
393 {
394 error = 1;
395 break;
396 }
397 adjncy[edge_count] = elem_voisin;
398// Les connexions entre faces periodiques sont fortes:
399// on veut que ces elements soient sur le meme processeur
400// Attention, si on met un poids nettement plus eleve que les autres
401// edges, on degrade la qualite du decoupage !
402 if (use_weights)
403 {
404 ewgts[edge_count] = 4;
405 }
406 edge_count++;
407 }
408 }
409 else if (use_weights)
410 {
411 if(Process::nproc() == 1)
412 {
413 Cerr << "Warning: You specify use_weights option with Metis but you didn't use Periodique keyword" << finl;
414 Cerr << "to define the boundary where periodicity apply." << finl;
415 Cerr << "Either suppress use_weight or add Periodique keyword." << finl;
416 Process::exit();
417 }
418 }
419 if (error > 0)
420 break;
421 }
422 xadj[nb_elem] = edge_count;
423 nedges = edge_count;
424
425 std::map<int,int> global_to_local_index;
426 int cnt=nb_elem;
427 for(int e=0; e<nb_edges + nb_faces_bord; e++)
428 {
429 int vertex = adjncy[e];
430 if( my_offset <= vertex && vertex < nb_elem+my_offset) //neighbour belongs to me
431 {
432 edgegsttab[e] = vertex - my_offset;
433 }
434 else
435 {
436 std::map<int,int>::iterator it = global_to_local_index.find(vertex);
437 if(it != global_to_local_index.end())
438 {
439 edgegsttab[e] = global_to_local_index[vertex];
440
441 }
442 else
443 {
444 global_to_local_index[vertex] =cnt++;
445 edgegsttab[e] = global_to_local_index[vertex];
446 }
447
448 }
449 }
450
451 if (error == 1)
452 {
453 Cerr << "Error in Domaine_Graph::construire_graph_elem_elem\n"
454 << " The number of element-element connections is greater than expected\n"
455 << " The number of boundary faces is wrong.\n"
456 << " You must discretize the domain to check." << finl;
457 Process::exit();
458 }
459 else if (error == 2)
460 {
461 Cerr << "Error in Domain_Graph::construire_graph_elem_elem\n"
462 << " Problem in the mesh: one internal face has more than two neighboring elements\n"
463 << " List of neighboring elements:" << voisins
464 << "\n Nodes of the face:" << une_face;
465 Process::exit();
466 }
467 else if (error)
468 {
469 Cerr << "Internal error in Domain_Graph::construire_graph_elem_elem\n"
470 << " (error " << error << ") Problem in neighborhood algorithms"
471 << finl;
472 Process::exit();
473 }
474
475}
