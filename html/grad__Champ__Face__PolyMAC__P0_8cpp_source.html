107void grad_Champ_Face_PolyMAC_P0::calc_gradfve()
108{
109 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0,zone_vf());
110 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, champ_a_deriver());
111 const IntTab& fcl = fcl_g;
112 const Conds_lim& cls = cls_g;
113 int f, n;
114 int d_U ; //coordonnee de la vitesse
115 int D = dimension;
116 int N = champ_a_deriver().valeurs().line_size(); //nombre phases
117 int ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), nf = zone.nb_faces();
118
119 const DoubleTab& tab_ch = ch.passe();
120 DoubleTab& val = valeurs();
121
122 for (f = 0; f < nf; f++)
123 {
124 for (d_U = 0; d_U < D; d_U++)
125 for (n = 0; n < N; n++) //!< Coordonnees de la vitesse et phase
126 {
127 val(f, d_U + n*D) = 0;
128 for (int j = gradve_d(f); j < gradve_d(f+1) ; j++)
129 {
130 int e = gradve_e(j);
131 int f_bord;
132 if ( e < ne_tot) //contrib d'un element
133 {
134 double val_e = tab_ch(nf_tot+ d_U + e * D, n);
135 val(f, d_U + n*D) += gradve_w(j, n) * val_e;
136 }
137 else if (fcl(f_bord = e - ne_tot, 0) == 3) //contrib d'un bord : seul Dirichlet contribue
138 {
139 double val_f_bord = ref_cast(Dirichlet, cls[fcl(f_bord, 1)].valeur()).val_imp(fcl(f_bord, 2), N * d_U + n);
140 val(f, d_U + n*D) += gradve_w(j, n) * val_f_bord;
141 }
142 }
143 }
144 }

148void grad_Champ_Face_PolyMAC_P0::update_ge()
149{
150 DoubleTab& val = valeurs() ;
151 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0,zone_vf());
152 const DoubleVect& ve = zone.volumes(), &fs = zone.face_surfaces();
153 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins();
154 const DoubleTab& xp = zone.xp(), &xv = zone.xv();
155 int e, f, j, d, D = dimension, n, N = val.line_size(), ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot();
156 double fac;
157
158 for (e = 0; e < ne_tot; e++)
159 {
160 for (d = 0; d < D; d++)
161 for (n = 0; n < N; n++) val(nf_tot + D * e + d, n) = 0;
162 for (j = 0; j < e_f.dimension(1) && (f = e_f(e, j)) >= 0; j++)
163 for (fac = (e == f_e(f, 0) ? 1 : -1) * fs(f) / ve(e), d = 0; d < D; d++)
164 for (n = 0; n < N; n++)
165 val(nf_tot + D * e + d, n) += fac * (xv(f, d) - xp(e, d)) * val(f, n);
166 }
167}
