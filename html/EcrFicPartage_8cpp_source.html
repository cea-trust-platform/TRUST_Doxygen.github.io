63//! Ouvre le fichier avec les parametres mode et prot donnes<br>Ces parametres sont les parametres de la methode open standard
64int EcrFicPartage::ouvrir(const char* name,IOS_OPEN_MODE mode)
65{
66// Verification sanitaire : tous les processeurs sont la ?
67 barrier();
68
69 int ok = 1;
70 if(je_suis_maitre())
71 {
72#ifdef FILESYSTEM_NON_GLOBAL
73 nom_fic_ = pwd();
74 nom_fic_ += "/";
75 nom_fic_ += name;
76#else
77 nom_fic_ = name;
78#endif
79// Seul le maitre ouvre le fichier
80 ok = SFichier::ouvrir((const char *)nom_fic_, mode);
81 }
82 syncfile();
83
84// Modif B.Math. 22/09/2004: tous les processeurs passent par le buffer,
85// y compris le maitre.
86 get_obuffer().new_buffer();
87 return ok;
88}

102void EcrFicPartage::close()
103{
104// Verification sanitaire : tout le monde est la ?
105 barrier();
106 const int buflen = get_obuffer().len();
107 if(buflen > 0)
108 {
109 Cerr << "***** WARNING : EcrFicPartage::close() ******* "<<nom_fic_
110 << "\non PE " << me() << " the buffer is not empty\n"
111 << " (Missing syncfile) : one makes a last syncfile" << finl;
112 Cerr<<get_obuffer().str()<<finl;
113 }
114#ifndef NDEBUG
115// Y a-t-il un processeur sur lequel il reste des donnees
116 const int maxbuflen = mp_sum(buflen);
117 if (maxbuflen > 0)
118 syncfile();
119#endif
120 if(je_suis_maitre())
121 SFichier::close();
122#ifndef NDEBUG
123 if (maxbuflen > 0)
124 {
125 Cerr<<"It missed a syncfile somewhere"<<finl;
126 Cerr<<"Indeed, maxbuflen =" << maxbuflen << finl;
127 Cerr<<"GF prefers to stop the calculation to correct "<<finl;
128 exit();
129 }
130#endif
131}

181//! Provoque l'ecriture sur disque des donnees accumulees sur les differents processeurs<br>depuis le dernier appel a syncfile().<br>Les donnees sont ecrites dans l'ordre croissant des processeurs.<br>Cette fonction doit etre appelee le meme nombre de fois sur tous les processeurs !<br>Exemple:<br>processeur 0: processeur 1:<br>file << "pe0 : 1" << finl; file << "pe1 : 1" << finl;<br>file << "pe0 : 2" << finl; file << "pe1 : 2" << finl;<br>file.syncfile(); file.syncfile();<br>file << "pe0 : 3" << finl; file << "pe1 : 3" << finl;<br>file << "pe0 : 4" << finl;<br>file.syncfile(); file.syncfile();<br>file << "pe0 : end" << finl; // le processeur 1 n'ecrit pas de donnees<br>file.syncfile(); file.syncfile();<br>Contenu du fichier :<br>pe0 : 1<br>pe0 : 2<br>pe1 : 1<br>pe1 : 2<br>pe0 : 3<br>pe0 : 4<br>pe1 : 3<br>pe0 : end
182Sortie& EcrFicPartage::syncfile()
183{
184// En mode ascii, les donnees sont converties en ascii lors de l'ecriture dans
185// le buffer. Une deuxieme conversion a lieu lors de l'ecriture sur disque
186// (selon le systeme d'exploitation, '\n' est code differemment par exemple)
187// Donc, en ascii, on utilisera
188// file << buffer;
189// et en binaire
190// file.write(buffer, size);
191// Or file << buffer determine la longueur du buffer en cherchant de caractere '\0'
192// Par consequent, en ascii, il faut que le buffer finisse par un caractere '\0'
193// que l'on ajoute ici:
194
195 if (! bin_)
196 {
197 get_obuffer().put_null_char();
198 }
199
200 const Comm_Group& group = PE_Groups::current_group();
201 if(je_suis_maitre())
202 {
203 int p;
204 const int nb_proc = nproc();
205 for(p=0; p<nb_proc; p++)
206 {
207 const char * buffer_data = 0;
208 char * allocated_buffer = 0;
209 int buf_size;
210
211// On recupere les donnees du processeur p, soit directement (p==me()),
212// soit par communication :
213 if (p == me())
214 {
215// Ecriture de mes propres donnees : je les prends dans le buffer.
216// Ce pointeur peut etre nul:
217 buffer_data = get_obuffer().str();
218 buf_size = get_obuffer().len();
219 }
220 else
221 {
222// Ecriture des donnees d'un autre processeur, je les recupere.
223 int dummy = 0;
224 envoyer(dummy, p, 100); //!< Faire coucou au processeur p pour qu'il envoie ses donnees
225 recevoir(buf_size, p, 100);
226 if (buf_size > 0)
227 {
228 buffer_data = allocated_buffer = new char[buf_size];
229 group.recv(p, allocated_buffer, buf_size, 100);
230 }
231 }
232// Ecriture dans le fichier disque
233 if (buf_size > 0)
234 {
235 assert(buffer_data);
236 ostream& os = get_ostream();
237 if (bin_)
238 {
239// Ecriture binaire sans conversion :
240 statistiques().begin_count(IO_EcrireFicPartageBin_counter_);
241 os.write(buffer_data, buf_size);
242 statistiques().end_count(IO_EcrireFicPartageBin_counter_, buf_size);
243 }
244 else
245 {
246// On verifie que le buffer finit bien par un caractere 0 :
247 assert(buffer_data[buf_size-1] == 0);
248// Ecriture de buffer_data comme une chaine
249// (conversion des \n sur certains systemes, etc...)
250 os << buffer_data;
251 }
252 }
253 if (allocated_buffer)
254 delete[] allocated_buffer;
255 }
256// Force a tout ecrire sur le disque tout de suite:
257// (appel a la fonction flush de bas niveau, pas celle de trio).
258 get_ostream().flush();
259 }
260 else
261 {
262// Envoi du buffer au processeur maitre:
263// On attend qu'il demande les donnees pour ne pas engorger le reseau :
264// (sinon tous les processeurs envoient simultanement leurs donnees)
265 int dummy;
266 recevoir(dummy, 0, 100);
267 int buf_size = get_obuffer().len();
268 envoyer(buf_size, 0, 100);
269// Si la taille est non nulle, on envoie le buffer :
270 if (buf_size > 0)
271 {
272 const char * buffer_data = get_obuffer().str();
273 group.send(0, buffer_data, buf_size, 100);
274 }
275 }
276
277// On vide le buffer :
278 get_obuffer().new_buffer();
279 return *this;
280}
