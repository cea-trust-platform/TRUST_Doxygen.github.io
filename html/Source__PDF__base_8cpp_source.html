37Entree& Source_PDF_base::readOn(Entree& s)
38{
39 Param param(que_suis_je());
40 param.ajouter("aire", &champ_aire_lu_,Param::REQUIRED); //!< XD_ADD_P field_base volumic field: a boolean for the cell (0 or 1) indicating if the obstacle is in the cell
41 param.ajouter("rotation", &champ_rotation_lu_,Param::REQUIRED); //!< XD_ADD_P field_base volumic field with 9 components representing the change of basis on cells (local to global). Used for rotating cases for example.
42 param.ajouter_flag("transpose_rotation", &transpose_rotation_,Param::OPTIONAL); //!< XD_ADD_P rien whether to transpose the basis change matrix.
43 param.ajouter("modele",&modele_lu_,Param::REQUIRED); //!< XD_ADD_P bloc_pdf_model model used for the Penalized Direct Forcing
44 temps_relax_ = modele_lu_.temps_relax_;
45 echelle_relax_ = modele_lu_.echelle_relax_;
46 param.ajouter("interpolation",&interpolation_lue_,Param::OPTIONAL); //!< XD_ADD_P interpolation_ibm_base interpolation method
47
48 param.lire_avec_accolades(s);
49 interpolation_bool_ = false;
50 if ((&interpolation_lue_)->non_nul())
51 {
52 if (!(interpolation_lue_.valeur().que_suis_je() == "Interpolation_IBM_aucune"))
53 {
54 interpolation_bool_ = true;
55 }
56 }
57 return s;
58}

102double Source_PDF_base::fonct_coeff(const double rho_m, const double aire, const double dt) const
103{
104 double val_coeff = 0.;
105 if (aire<=0.)
106 {
107 return val_coeff;
108 }
109
110 double inv_dt = 1./dt;
111// Cerr<<"dt pour Source_PDF_base::fonct_coeff : "<< dt <<finl;
112 val_coeff = rho_m * inv_dt;
113 const double tps_cour = equation().probleme().schema_temps().temps_courant();
114 if (temps_relax_ != 1.0e+12)
115 {
116// double bco = 100.*(1.-exp(-0.1/echelle_relax_)); // tps_cour_intersection = 0.1 * temps_relax_
117 double deltrel = tps_cour - temps_relax_;
118 double coeff_relax = (1.+tanh(deltrel/(echelle_relax_ * temps_relax_)))/2.;
119// double coeff_relax = min (1., std::max(1.-bco*deltrel*deltrel/temps_relax_/temps_relax_, exp(deltrel/(echelle_relax_ * temps_relax_))));
120// double coeff_relax = 1.0 - exp(-tps_cour / temps_relax_);
121// double coeff_relax = tanh(tps_cour / temps_relax_);
122 val_coeff *= coeff_relax ;
123// Cerr<< "coeff. relax. pour PDF = "<<coeff_relax<<" val_coeff = "<<val_coeff <<endl;
124 }
125 return val_coeff;
126}

309void Source_PDF_base::ouvrir_fichier(SFichier& os, const Nom& type, const int flag=1) const
310{
311// flag nul on n'ouvre pas le fichier
312 if (flag==0)
313 return ;
314
315 const Probleme_base& pb=equation().probleme();
316 const Schema_Temps_base& sch=pb.schema_temps();
317 const int precision=sch.precision_impr();
318 Nom nomfichier(out_);
319 if (type!="") nomfichier+=(Nom)"_"+type;
320 nomfichier+=".out";
321
322// On cree le fichier a la premiere impression avec l'en tete
323 if (sch.nb_impr()==1 && !pb.reprise_effectuee())
324 {
325 os.ouvrir(nomfichier);
326 SFichier& fic=os;
327 Nom espace="\t\t";
328 fic << (Nom)"# Printing of the source term "+que_suis_je()+" of the equation "+equation().que_suis_je()+" of the problem "+equation().probleme().le_nom() << finl;
329 fic << "# " << description() << finl;
330 fic << "# Time" << espace << "Fx" << espace << "Fy" << espace << "Fz";
331 fic << finl;
332 }
333// Sinon on l'ouvre
334 else
335 {
336 os.ouvrir(nomfichier,ios::app);
337 }
338 os.precision(precision);
339 os.setf(ios::scientific);
340}
