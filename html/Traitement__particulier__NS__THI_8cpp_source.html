64Entree& Traitement_particulier_NS_THI::lire(Entree& is)
65{
66// valeurs par defaut
67 init = 0;
68 fac_init = 0;
69 oui_transf = 0;
70 oui_calc_spectre = 0;
71// FIN valeurs par defaut
72 Motcle accouverte = "{" , accfermee = "}" ;
73 Motcle valec="val_Ec";
74 Motcle facon="facon_init";
75 Motcle motbidon, motlu;
76 is >> motbidon ;
77 if (motbidon == accouverte)
78 {
79 Motcles les_mots(2);
80 {
81 les_mots[0] = "init_Ec";
82 les_mots[1] = "calc_spectre";
83 }
84 {
85 is >> motlu;
86 while(motlu != accfermee)
87 {
88 int rang=les_mots.search(motlu);
89 switch(rang)
90 {
91 case 0 :
92 {
93 is >> init; //!< init_Ec = 1
94 if (init!=0)
95 {
96 is >> motlu;
97 if (motlu==valec)
98 {
99 is >> Ec_init; //!< on lit la valeur de Ec_init
100 Cerr << "Avec Initialisation de l Energie Cinetique Ec_init= " << Ec_init << finl;
101 is >> motlu;
102 if (motlu==facon)
103 {
104 is >> fac_init; //!< Sur quelle valeur se base l initialisation??
105 Cerr << "Avec Initialisation de l Energie Cinetique sur Ecspat (init_fac==0) ou Ecspec (init_fac==1) : fac_init=" << fac_init << finl;
106 }
107 }
108 else
109 {
110 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI_VDF" << finl;
111 Cerr << "Le seul mot cle possible ici est : val_Ec" << finl;
112 Cerr << "Vous avez lu :" << motlu << finl;
113 exit();
114 }
115 }
116 break;
117 }
118 case 1 :
119 {
120 is >> oui_calc_spectre;
121 if (oui_calc_spectre!=0) Cerr << "Calcul des spectres." << finl;
122 break;
123 }
124
125 default :
126 {
127 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI";
128 Cerr << "Les mots cles possibles sont : init_Ec, calc_spectre, { et }" << finl;
129 Cerr << "Vous avez lu :" << motlu << finl;
130 exit();
131 break;
132 }
133 }
134 is >> motlu;
135 }
136 is >> motlu;
137 if (motlu != accfermee)
138 {
139 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI";
140 Cerr << "On attendait une }" << finl;
141 exit();
142 }
143 }
144 }
145 else
146 {
147 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI";
148 Cerr << "On attendait une {" << finl;
149 exit();
150 }
151 return is;
152}

183int& Traitement_particulier_NS_THI::calcul_nb_som_dir(const Zone& zone)
184{
185 const char* methode_actuelle="Traitement_particulier_NS_THI::calcul_nb_som_dir";
186
187// Sert a calculer le nombre de sommet commun en parallele
188 ArrOfBit unused;
189 int nb_som = MD_Vector_tools::get_sequential_items_flags(zone.domaine().les_sommets().get_md_vector(), unused);
190// Somme sur tous les processeurs
191 nb_som = mp_sum(nb_som);
192
193 double nb=pow(nb_som*1.,1./3.);
194 nb_som_dir = (int)(nb);
195 if (nb_som_dir*nb_som_dir*nb_som_dir != nb_som)
196 {
197 nb_som_dir=nb_som_dir+1;
198 if (nb_som_dir*nb_som_dir*nb_som_dir != nb_som)
199 msg_erreur_maillage(methode_actuelle);
200 }
201 nb_som_dir=nb_som_dir-1;
202 return nb_som_dir;
203}
