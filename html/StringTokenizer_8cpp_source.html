100int StringTokenizer::check_GRP()
101{
102 const char* ch = str.c_str();
103 int nb_o=0;
104 int nb_f=0;
105 int sz = strlen(ch);
106 for (int i=0; i<sz; i++)
107 {
108 if (*(ch+i)=='(')
109 nb_o++;
110 else if (*(ch+i)==')')
111 nb_f++;
112 }
113 return (nb_o==nb_f);
114}

117int StringTokenizer::nextToken(void)
118{
119 char *tmp;
120 int type_sep, length;
121 tmp=find_sep(reste, type_sep, length);
122 if (tmp == NULL)
123 {
124// Cout << "Fin ? " << reste << finl;
125 if (reste[0] == '\0')
126 {
127 type = EOS;
128 }
129 else
130 {
131 if ( ((reste[0] >= '0') && ( reste[0] <= '9')) || (reste[0] == '.') )
132 {
133// ajout OC pour provoquer une erreur avec 2x
134 int ind=strlen(reste)-1;
135 if ( ((reste[ind] < '0') || (reste[ind] > '9')) && (reste[ind] != '.') )
136 {
137 Cerr << "The syntax " << reste << " is not allowed." << finl;
138 Process::exit();
139 }
140 type = NUMBER;
141 stringstream stream;
142 stream << reste ;
143 stream >> nval;
144 if (!stream.eof() )
145 {
146 Cerr<<"Error conversion "<<reste<<" in number"<<finl;
147 Process::exit();
148 }
149 }
150 else
151 {
152 type=STRING;
153 sval = reste;
154 for (auto & c: sval) c = toupper(c);
155 }
156 while ((*reste++) != '\0') ;
157 reste--;
158 }
159 }
160 else if (tmp==reste)
161 {
162 type = type_sep;
163 reste=reste+length;
164 }
165 else
166 {
167 Char_ptr token;
168 token.allocate(tmp-reste);
169
170 char* tok = token.getChar();;
171 int j=0;
172 for (int i=0; i<tmp-reste; i++)
173 {
174 if ( reste[i] != ' ' ) tok[j++] = reste[i];
175 }
176 tok[j] = '\0';
177 if (((tok[0] >= '0') && ( tok[0] <= '9')) || (tok[0] == '.') )
178 {
179 int ind;
180 double nval_tmp;
181
182 type = NUMBER;
183 stringstream stream;
184 ind = tmp-reste-1;
185 reste = tmp;
186 if (tok[ind] == 'e' || tok[ind] == 'E' )
187 {
188 tok[ind]=' ';
189 stream << tok ;
190 stream >> nval_tmp;
191 char c;
192 stream >> c;
193
194 if (!stream.eof() )
195 {
196 Cerr<<"Error conversion "<<tok<<" in number"<<finl;
197 Process::exit();
198 }
199 nextToken();
200 if (type == SUBTRACT)
201 {
202 nextToken();
203 if (type != NUMBER)
204 {
205 Cerr << "Error while interpreting the string " << str << finl;
206 Process::exit();
207 }
208 nval_tmp*=pow(10,-nval);
209 }
210 else if (type == ADD)
211 {
212 nextToken();
213 if (type != NUMBER)
214 {
215 Cerr << "Error while interpreting the string " << str << finl;
216 Process::exit();
217 }
218 nval_tmp*=pow(10.,nval);
219 }
220 else if (type == NUMBER)
221 {
222// GF suivant les versions on a soit
223 nval_tmp*=pow(10.,nval);
224// soit
225// nval_tmp = nval;
226// J'interdis pour l'instant
227// Cerr << "Possible Erreur en interpretant la chaine " << str << finl;
228// Process::exit();
229 }
230 else
231 {
232 Cerr << "Error while interpreting the string " << str << finl;
233 Process::exit();
234 }
235 }
236 else if ( ((tok[ind] < '0') || (tok[ind] > '9')) && (tok[ind] != '.') )
237 {
238 Cerr << " The syntax " << tok << " is not allowed." << finl;
239 Process::exit();
240 throw;
241 }
242 else
243 {
244 stream << tok ;
245 stream >> nval_tmp;
246 if (!stream.eof() )
247 {
248
249 Cerr<<"Error conversion "<<tok<<" in number"<<finl;
250 Process::exit();
251 }
252 }
253 nval = nval_tmp;
254 }
255 else
256 {
257 type=STRING;
258 sval = tok;
259 for (auto & c: sval) c = toupper(c);
260 reste=tmp;
261 }
262// delete[] tok;
263 }
264 return type;
265}

271void StringTokenizer::init_keyword_op()
272{
273 op_sep = new char*[nb_op];
274// on encadre le mot-cle de chaque operateur par des "_" :
275// LT => _LT_ etc...
276 for (int i=0; i<nb_op; i++)
277 {
278 const char* blanc="_";
279 op_sep[i] = new char[strlen(keyword_op[i])+3];
280 strcpy(op_sep[i],blanc);
281 strcat(op_sep[i], keyword_op[i]);
282 strcat(op_sep[i], blanc);
283 }
284}

287/**
288 * Cherche dans la chaine "ch" la prochaine apparition d'un separateur.
289 * Renvoie dans le parametre "type_sep", le type de separateur ((, ), +, - etc...)
290 * Renvoie egalement la longueur de la chaine correspondante au separateur trouve
291 */
292char* StringTokenizer::find_sep(char* ch, int& type_sep, int& length)
293{
294 char * trouve=NULL;
295 char * trouve_tmp;
296 type_sep=-1;
297 trouve_tmp = strstr(ch, "(");
298 int pos=100000;
299 if (trouve_tmp != NULL)
300 {
301 pos = trouve_tmp-ch;
302 type_sep=GRP;
303 length=1;
304 trouve=trouve_tmp;
305 }
306 trouve_tmp = strstr(ch, ")");
307 if ((trouve_tmp != NULL) && (trouve_tmp-ch<pos))
308 {
309 pos = trouve_tmp-ch;
310 type_sep=ENDGRP;
311 length=1;
312 trouve=trouve_tmp;
313 }
314 for (int i=0; i<StringTokenizer::nb_op; i++)
315 {
316 trouve_tmp = strstr(ch, op_sep[i]);
317 if ((trouve_tmp != NULL) && (trouve_tmp-ch<pos))
318 {
319 pos = trouve_tmp-ch;
320 type_sep=i;
321 length=strlen(op_sep[i]);
322 trouve=trouve_tmp;
323 }
324 }
325 for (int i=0; i<StringTokenizer::nb_op_bis; i++)
326 {
327 trouve_tmp = strstr(ch, keyword_op_bis[i]);
328 if ((trouve_tmp != NULL) && (trouve_tmp-ch<pos))
329 {
330 pos = trouve_tmp-ch;
331 type_sep=i;
332 length=strlen(keyword_op_bis[i]);
333 trouve=trouve_tmp;
334 }
335 }
336
337 return trouve;
338}
