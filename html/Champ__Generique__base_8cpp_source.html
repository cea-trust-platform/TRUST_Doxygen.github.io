306const Champ_Generique_base& Champ_Generique_base::get_champ_post(const Motcle& nom) const
307{
308 REF(Champ_Generique_base) ref_champ;
309
310 Motcle nom_champ;
311 const Noms nom_champ_post = get_property("nom");
312 nom_champ = Motcle(nom_champ_post[0]);
313
314 if (nom_champ==nom)
315 {
316 ref_champ = *this;
317 ref_champ->fixer_identifiant_appel(nom);
318 return ref_champ;
319 }
320 const Noms syno = get_property("synonyms");
321 int nb_syno = syno.size();
322 for (int i=0; i<nb_syno; i++)
323 {
324 nom_champ = Motcle(syno[i]);
325 if (nom_champ==nom)
326 {
327 ref_champ = *this;
328 ref_champ->fixer_identifiant_appel(nom);
329 return ref_champ;
330 }
331 }
332
333 {
334 const Noms composantes = get_property("composantes");
335 int nb_composantes = composantes.size();
336
337 for (int i=0; i<nb_composantes; i++)
338 {
339 nom_champ = Motcle(composantes[i]);
340 if (nom_champ==nom)
341 {
342 ref_champ = *this;
343 ref_champ->fixer_identifiant_appel(nom);
344 return ref_champ;
345 }
346 }
347 }
348
349 throw Champs_compris_erreur();
350 return ref_champ;
351}

353int Champ_Generique_base::comprend_champ_post(const Motcle& identifiant) const
354{
355 Motcle nom_champ;
356 const Noms nom_champ_post = get_property("nom");
357 nom_champ = Motcle(nom_champ_post[0]);
358
359 if (nom_champ==identifiant)
360 return 1;
361 else
362 {
363 const Noms composantes = get_property("composantes");
364 int nb_composantes = composantes.size();
365
366 for (int i=0; i<nb_composantes; i++)
367 {
368 nom_champ = Motcle(composantes[i]);
369 if (nom_champ==identifiant)
370 return 1;
371 }
372 }
373 return 0;
374}
