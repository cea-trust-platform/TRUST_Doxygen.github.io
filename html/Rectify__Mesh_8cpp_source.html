125void Rectify_Mesh::check_cell_orientation_2D(void)
126{
127 const DoubleTab& nodes = domaine().les_sommets();
128 IntTab& cells = domaine().zone(0).les_elems();
129
130 const int nb_cells = cells.dimension(0);
131
132 int nb_swaps = 0;
133
134 for (int c=0; c<nb_cells; ++c)
135 {
136 const int n0 = cells(c,0);
137 const int n1 = cells(c,1);
138 const int n2 = cells(c,2);
139
140 const double x0 = nodes(n0,0);
141 const double x1 = nodes(n1,0);
142 const double x2 = nodes(n2,0);
143
144 const double y0 = nodes(n0,1);
145 const double y1 = nodes(n1,1);
146 const double y2 = nodes(n2,1);
147
148 const double area = (x1-x0)*(y2-y0) - (x2-x0)*(y1-y0);
149 if ( area < 0.)
150 {
151 swap_nodes(cells,c,1,2);
152 ++nb_swaps;
153 }
154 }
155
156 Cerr << "************************************************" << finl;
157 Cerr << "Rectify_Mesh::check_cell_orientation_2D report :" << finl;
158 Cerr << " " << nb_swaps << " swap" << ((nb_swaps ==1) ? "" : "s") << " were performed" << finl;
159 Cerr << "************************************************" << finl;
160}

162void Rectify_Mesh::check_cell_orientation_3D(void)
163{
164 const DoubleTab& nodes = domaine().les_sommets();
165 IntTab& cells = domaine().zone(0).les_elems();
166
167 const int nb_cells = cells.dimension(0);
168
169 int nb_swaps = 0;
170
171 for (int c=0; c<nb_cells; ++c)
172 {
173 const int n0 = cells(c,0);
174 const int n1 = cells(c,1);
175 const int n2 = cells(c,2);
176 const int n3 = cells(c,3);
177
178 const double x0 = nodes(n0,0);
179 const double x1 = nodes(n1,0);
180 const double x2 = nodes(n2,0);
181 const double x3 = nodes(n3,0);
182
183 const double y0 = nodes(n0,1);
184 const double y1 = nodes(n1,1);
185 const double y2 = nodes(n2,1);
186 const double y3 = nodes(n3,1);
187
188 const double z0 = nodes(n0,2);
189 const double z1 = nodes(n1,2);
190 const double z2 = nodes(n2,2);
191 const double z3 = nodes(n3,2);
192
193 const double volume = (x1-x0) * (y2-y0) * (z3-z0)
194 + (x2-x0) * (y3-y0) * (z1-z0)
195 + (x3-x0) * (y1-y0) * (z2-z0)
196 - (z1-z0) * (y2-y0) * (x3-x0)
197 - (x2-x0) * (z3-z0) * (y1-y0)
198 - (y3-y0) * (x1-x0) * (z2-z0);
199 if (volume < 0.)
200 {
201 swap_nodes(cells,c,2,3);
202 ++nb_swaps;
203 }
204 }
205
206 Cerr << "************************************************" << finl;
207 Cerr << "Rectify_Mesh::check_cell_orientation_3D report :" << finl;
208 Cerr << " " << nb_swaps << " swap" << ((nb_swaps ==1) ? "" : "s") << " were performed" << finl;
209 Cerr << "************************************************" << finl;
210
211}

213void Rectify_Mesh::check_cell_enumeration_3D(void)
214{
215 const DoubleTab& nodes = domaine().les_sommets();
216 IntTab& cells = domaine().zone(0).les_elems();
217
218 const int nb_cells = cells.dimension(0);
219
220 ArrOfDouble lengths(3);
221
222 int nb_swaps = 0;
223
224 for (int c=0; c<nb_cells; ++c)
225 {
226 const int n0 = cells(c,0);
227 const int n1 = cells(c,1);
228 const int n2 = cells(c,2);
229 const int n3 = cells(c,3);
230
231 const double x0 = nodes(n0,0);
232 const double x1 = nodes(n1,0);
233 const double x2 = nodes(n2,0);
234 const double x3 = nodes(n3,0);
235
236 const double y0 = nodes(n0,1);
237 const double y1 = nodes(n1,1);
238 const double y2 = nodes(n2,1);
239 const double y3 = nodes(n3,1);
240
241 const double z0 = nodes(n0,2);
242 const double z1 = nodes(n1,2);
243 const double z2 = nodes(n2,2);
244 const double z3 = nodes(n3,2);
245
246 lengths[0] = ((x0+x1)-(x2+x3))*((x0+x1)-(x2+x3))
247 + ((y0+y1)-(y2+y3))*((y0+y1)-(y2+y3))
248 + ((z0+z1)-(z2+z3))*((z0+z1)-(z2+z3));
249
250 lengths[1] = ((x0+x2)-(x1+x3))*((x0+x2)-(x1+x3))
251 + ((y0+y2)-(y1+y3))*((y0+y2)-(y1+y3))
252 + ((z0+z2)-(z1+z3))*((z0+z2)-(z1+z3));
253
254 lengths[2] = ((x0+x3)-(x1+x2))*((x0+x3)-(x1+x2))
255 + ((y0+y3)-(y1+y2))*((y0+y3)-(y1+y2))
256 + ((z0+z3)-(z1+z2))*((z0+z3)-(z1+z2));
257
258 const int id = imin_array(lengths);
259// const int id = imax_array(lengths);
260
261 if ( (id == 0) && (lengths[0] < lengths[1]) )
262 {
263 swap_nodes(cells,c,1,2);
264 swap_nodes(cells,c,1,3);
265 ++nb_swaps;
266 }
267
268 if (id == 1)
269 {
270// nothing to do
271 }
272
273 if ( (id == 2) && (lengths[2] < lengths[1]) )
274 {
275 swap_nodes(cells,c,2,3);
276 swap_nodes(cells,c,1,3);
277 ++nb_swaps;
278 }
279 }
280
281 Cerr << "************************************************" << finl;
282 Cerr << "Rectify_Mesh::check_cell_enumeration_3D report :" << finl;
283 Cerr << " " << nb_swaps << " swap" << ((nb_swaps ==1) ? "" : "s") << " were performed" << finl;
284 Cerr << "************************************************" << finl;
285
286}

288void Rectify_Mesh::swap_nodes(IntTab& cells, int cell, int node0, int node1)
289{
290 const int tmp = cells(cell,node0);

292 cells(cell,node1) = tmp;
