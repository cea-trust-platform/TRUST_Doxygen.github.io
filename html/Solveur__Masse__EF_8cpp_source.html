53DoubleTab& Solveur_Masse_EF::appliquer_impl(DoubleTab& sm) const
54{
55 Debog::verifier("Solveur_Masse_EF::appliquer deb, sm=",sm);
56 const Zone_Cl_EF& zone_Cl_EF = ref_cast(Zone_Cl_EF,la_zone_Cl_EF.valeur());
57 const Zone_EF& zone_EF = la_zone_EF.valeur();
58 const DoubleVect& volumes_sommets_thilde = zone_EF.volumes_sommets_thilde();
59
60// const DoubleVect& porosite_sommet = zone_EF.porosite_sommet();
61 int nfa = zone_EF.nb_som();
62 int nbsom=zone_EF.nb_som();
63 int face;
64 if (nfa != sm.dimension(0))
65 {
66 Cerr << "erreur dans Solveur_Masse_EF : ";
67 Cerr << "nombre de faces : " << nfa
68 << " taille du second membre : " << sm.dimension(0) << finl;
69 exit();
70 }
71 const IntTab& faces_sommets=zone_EF.face_sommets();
72 int nb_som_face=faces_sommets.dimension(1);
73 if (sm.nb_dim() == 1)
74 {
75// On traite les faces standard qui ne portent pas de conditions aux limites
76
77 for (int som=0; som<nbsom; som++)
78 {
79 sm(som) /= (volumes_sommets_thilde(som));
80
81 }
82
83// On traite les faces non standard
84// les faces des bord sont des faces non standard susceptibles de porter des C.L
85// les faces internes non standard ne portent pas de C.L
86
87// On traite les conditions aux limites
88 int nb_cl=zone_Cl_EF.nb_cond_lim();
89 for (int n_bord=0; n_bord<nb_cl; n_bord++)
90 {
91
92 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
93 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
94 int num1 = le_bord.num_premiere_face();
95 int num2 = num1 + le_bord.nb_faces();
96
97 if ( (sub_type(Dirichlet,la_cl.valeur()))
98 ||
99 (sub_type(Dirichlet_homogene,la_cl.valeur()))
100 )
101// Pour les faces de Dirichlet on met sm a 0
102 for (face=num1; face<num2; face++)
103 for (int s=0; s<nb_som_face; s++)
104 {
105 int som=faces_sommets(face,s);
106 sm(som) =0;
107 }
108 }
109 }
110
111
112 else
113 {
114 int nbcomp = sm.dimension(1);
115// On traite les faces standard qui ne portent pas de conditions aux limites
116
117 for (int som=0; som<nbsom; som++)
118 {
119 for (int comp=0; comp<nbcomp; comp++)
120 sm(som,comp) /= (volumes_sommets_thilde(som));
121
122 }
123
124// On traite les faces non standard
125// les faces des bord sont des faces non standard susceptibles de porter des C.L
126// les faces internes non standard ne portent pas de C.L
127
128// On traite les conditions aux limites
129 if (zone_Cl_EF.equation().inconnue()->nature_du_champ()==vectoriel)
130 zone_Cl_EF.imposer_symetrie(sm);
131 int nb_cl=zone_Cl_EF.nb_cond_lim();
132 for (int n_bord=0; n_bord<nb_cl; n_bord++)
133 {
134
135 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
136 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
137 int num1 = le_bord.num_premiere_face();
138 int num2 = num1 + le_bord.nb_faces();
139
140 if ( (sub_type(Dirichlet,la_cl.valeur()))
141 ||
142 (sub_type(Dirichlet_homogene,la_cl.valeur()))
143 )
144// Pour les faces de Dirichlet on met sm a 0
145 for (face=num1; face<num2; face++)
146 for (int s=0; s<nb_som_face; s++)
147 {
148 int som=faces_sommets(face,s);
149 for (int comp=0; comp<nbcomp; comp++)
150 sm(som,comp) =0;
151 }
152 }
153 }
154
155
156 sm.echange_espace_virtuel();
157 Debog::verifier("Solveur_Masse_EF::appliquer, sm=",sm);
158 return sm;
159}

184 inco.fixer_nature_du_champ(vectoriel);
