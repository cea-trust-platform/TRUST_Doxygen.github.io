222void Loi_Etat_base::calculer_lambda()
223{
224 const Champ_Don& mu = le_fluide->viscosite_dynamique();
225 const DoubleTab& tab_Cp = le_fluide->capacite_calorifique().valeurs();
226 const DoubleTab& tab_mu = mu.valeurs();
227 Champ_Don& lambda = le_fluide->conductivite();
228 DoubleTab& tab_lambda = lambda.valeurs();
229
230 int i, n = tab_lambda.size();
231// La conductivite est soit un champ uniforme soit calculee a partir de la viscosite dynamique et du Pr

238 }

240 {
241 for (i=0 ; i<n ; i++) tab_lambda(i,0) = tab_mu(i,0) * tab_Cp(i,0) / Pr_;

260//! Calcule la viscosite cinematique
261void Loi_Etat_base::calculer_nu()
262{
263 const Champ_Don& mu = le_fluide->viscosite_dynamique();
264 const DoubleTab& tab_rho = le_fluide->masse_volumique().valeurs();
265 const DoubleTab& tab_mu = mu.valeurs();
266 Champ_Don& nu= le_fluide->viscosite_cinematique ();
267 DoubleTab& tab_nu = nu.valeurs();
268
269 int i, n = tab_nu.size(), isVDF=0;
270 if (nu.valeur().que_suis_je()=="Champ_Fonc_P0_VDF") isVDF = 1;
271
272 if (isVDF)
273 {
274 if (sub_type(Champ_Uniforme,mu.valeur()))
275 {
276 double mu0 = tab_mu(0,0);
277 for (i=0 ; i<n ; i++) tab_nu(i,0) = mu0 / tab_rho(i,0);
278 }
279 else
280 {
281 for (i=0 ; i<n ; i++) tab_nu(i,0) = tab_mu(i,0) / tab_rho(i,0);
282 }
283 }
284 else //!< VEF
285 {
286 const IntTab& elem_faces=ref_cast(Zone_VF,le_fluide->vitesse().zone_dis_base()).elem_faces();
287 double rhoelem;
288 int face, nfe = elem_faces.line_size();
289 if (sub_type(Champ_Uniforme,mu.valeur()))
290 {
291 double mu0 = tab_mu(0,0);
292 for (i=0 ; i<n ; i++)
293 {
294 rhoelem=0;
295 for (face=0; face<nfe; face++) rhoelem += tab_rho(elem_faces(i,face),0);
296 rhoelem /= nfe;
297 tab_nu(i,0) = mu0 / rhoelem;
298 }
299 }
300 else
301 {
302 for (i=0 ; i<n ; i++)
303 {
304 rhoelem=0;
305 for (face=0; face<nfe; face++) rhoelem += tab_rho(elem_faces(i,face),0);
306 rhoelem /= nfe;
307 tab_nu(i,0) = tab_mu(i,0) / rhoelem;
308 }
309 }
310 }
311 tab_nu.echange_espace_virtuel();
312 Debog::verifier("Loi_Etat_base::calculer_nu tab_nu",tab_nu);
313}

328//! Calcule la diffusivite
329void Loi_Etat_base::calculer_alpha()
330{
331 const Champ_Don& lambda = le_fluide->conductivite();
332 const DoubleTab& tab_lambda = lambda.valeurs();
333 const DoubleTab& tab_Cp = le_fluide->capacite_calorifique().valeurs();
334 const DoubleTab& tab_rho = le_fluide->masse_volumique().valeurs();
335 DoubleTab& tab_alpha = le_fluide->diffusivite().valeurs();
336
337 int i, n = tab_alpha.size();
338 if (sub_type(Champ_Uniforme,lambda.valeur()))
339 {
340 double lambda0 = tab_lambda(0,0);
341 for (i=0 ; i<n ; i++) tab_alpha(i,0) = lambda0 / (tab_rho(i,0)*tab_Cp(i,0));
342 }
343 else
344 {
345 for (i=0 ; i<n ; i++) tab_alpha(i,0) = tab_lambda(i,0) / (tab_rho(i,0)*tab_Cp(i,0));
346 }
347 tab_alpha.echange_espace_virtuel();
348 Debog::verifier("Loi_Etat_base::calculer_alpha alpha",tab_alpha);
349}

405//! Recalcule la masse volumique
406void Loi_Etat_base::calculer_masse_volumique()
407{
408 const DoubleTab& tab_ICh = le_fluide->inco_chaleur().valeurs();
409 DoubleTab& tab_rho = le_fluide->masse_volumique().valeurs();
410 double Pth = le_fluide->pression_th();
411 int n=tab_rho.size();
412 for (int som=0 ; som<n ; som++)
413 {
414 tab_rho_np1(som) = calculer_masse_volumique(Pth,tab_ICh(som,0));
415 tab_rho(som,0) = 0.5 * ( tab_rho_n(som) + tab_rho_np1(som) );
416 }
417 tab_rho.echange_espace_virtuel();
418 tab_rho_np1.echange_espace_virtuel();
419 le_fluide->calculer_rho_face(tab_rho_np1);
420}
