36void Champ_Generique_Ecart_Type::completer(const Postraitement_base& post)
37{
38 Champ_Gen_de_Champs_Gen::completer(post);
39 const Probleme_base& Pb = get_ref_pb_base();
40 const Champ_Generique_base& ch_integre = get_source(0);
41 const Zone_dis_base& zdis = get_ref_zone_dis_base();
42
43 Op_Ecart_Type_.associer(zdis,ch_integre,tstat_deb_,tstat_fin_);
44
45 Noms nom = get_source(0).get_property("nom");
46 Motcle nom_champ = nom[0];
47 Motcle nom_champ_moyenne("Moyenne_");
48 nom_champ_moyenne += nom_champ;
49 nom_champ_moyenne += "_";
50 nom_champ_moyenne += Nom(tstat_deb_,"%e");
51 nom_champ_moyenne += "_";
52 nom_champ_moyenne += Nom(tstat_fin_,"%e");
53
54 const Postraitement& mon_post = ref_cast(Postraitement,post);
55 if (mon_post.comprend_champ_post(nom_champ_moyenne))
56 {
57 const Champ_Generique_base& champ_stat_base = mon_post.get_champ_post(nom_champ_moyenne);
58 const Operateur_Statistique_tps_base& operateur = ref_cast(Champ_Generique_Moyenne,champ_stat_base).Operateur_Statistique();
59 const Op_Moyenne& op_moyenne = ref_cast(Op_Moyenne,operateur);
60 Op_Ecart_Type_.associer_op_stat(op_moyenne);
61 Op_Ecart_Type_.completer(Pb);
62 }
63 else
64 {
65 Cerr<<"The specification of the post-processing rms field "<<finl;
66 Cerr<<get_property("nom")[0]<<finl;
67 Cerr<<"implies to specify before the corresponding "<<finl;
68 Cerr<<"post-processing statistics average field."<<finl;
69 Cerr<<"Please, modify your data set to specify the post-processing statistics"<<finl;
70 Cerr<<"average field before (instead of after) the rms field "<<finl;
71 Cerr<<get_property("nom")[0]<<finl;
72 exit();
73 }
74}

76const Champ_base& Champ_Generique_Ecart_Type::get_champ(Champ& espace_stockage) const
77{
78 const REF(Champ_Generique_base)& mon_champ = integrale().le_champ();
79 Champ espace_stockage_source;
80 const Champ_base& source = mon_champ->get_champ(espace_stockage_source);
81 Nature_du_champ nature_source = source.nature_du_champ();
82 int nb_comp = source.nb_comp();
83 Champ_Fonc es_tmp;
84 espace_stockage = creer_espace_stockage(nature_source,nb_comp,es_tmp);
85
86 DoubleTab& tab_ecart_type = espace_stockage.valeurs();
87 tab_ecart_type = Op_Ecart_Type_.calculer_valeurs();
88 tab_ecart_type.echange_espace_virtuel();
89 return espace_stockage.valeur();
90}
