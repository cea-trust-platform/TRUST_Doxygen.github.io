67void Op_Grad_PolyMAC_Face::completer()
68{
69 Operateur_Grad_base::completer();
70 const Zone_PolyMAC& zone = ref_zone.valeur();
71 /* initialisation des inconnues auxiliaires de la pression... */
72 ref_cast(Champ_Elem_PolyMAC, ref_cast(Navier_Stokes_std, equation()).pression().valeur()).init_auxiliary_variables();
73 /* et de grad P si la vitesse en a */
74 if (equation().inconnue().valeurs().get_md_vector() == zone.mdv_faces_aretes)

79}

81void Op_Grad_PolyMAC_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
82{
83 if (!matrices.count("pression")) return; //rien a faire
84 const Zone_PolyMAC& zone = ref_zone.valeur();
85 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur());
86 const IntTab& e_f = zone.elem_faces(), &fcl = ch.fcl();
87 const DoubleTab& vit = ch.valeurs(), &press = ref_cast(Navier_Stokes_std, equation()).pression().valeurs();
88 int i, j, e, f, fb, ne_tot = zone.nb_elem_tot(), n, N = vit.line_size(), m, M = press.line_size();
89 Matrice_Morse *mat = matrices["pression"], mat2;
90 IntTrav sten(0, 2);
91 DoubleTrav w2;
92 sten.set_smart_resize(1), w2.set_smart_resize(1);
93 for (e = 0; e < ne_tot; e++)
94 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(0); i++)
95 if ((f = e_f(e, i)) < zone.nb_faces()) /* faces reelles seulement */
96 {
97 for (n = 0, m = 0; n < N; n++, m += (M > 1)) sten.append_line(N * f + n, M * e + m); /* bloc (face, elem )*/
98 for (j = 0; j < w2.dimension(1); j++)
99 if (fcl(fb = e_f(e, j), 0) != 1 && w2(i, j, 0)) /* bloc (face, face) */
100 for (n = 0, m = 0; n < N; n++, m += (M > 1)) sten.append_line(N * f + n, M * (ne_tot + fb) + m);
101 }
102
103 /* allocation / remplissage */
104 tableau_trier_retirer_doublons(sten);
105 Matrix_tools::allocate_morse_matrix(vit.size_totale(), press.size_totale(), sten, mat2);
106 mat->nb_colonnes() ? *mat += mat2 : *mat = mat2;
107}

109void Op_Grad_PolyMAC_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
110{
111 const Zone_PolyMAC& zone = ref_zone.valeur();
112 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces(), &fcl = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur()).fcl();
113 const DoubleTab& vfd = zone.volumes_entrelaces_dir(), &press = semi_impl.count("pression") ? semi_impl.at("pression") : ref_cast(Navier_Stokes_std, equation()).pression().valeurs(),
114 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
115 const DoubleVect& fs = zone.face_surfaces(), &vf = zone.volumes_entrelaces(), &pe = zone.porosite_elem();
116 int i, j, e, eb, f, fb, ne_tot = zone.nb_elem_tot(), n, N = secmem.line_size(), m, M = press.line_size();
117
118 Matrice_Morse *mat = !semi_impl.count("pression") && matrices.count("pression") ? matrices.at("pression") : NULL;
119 DoubleTrav w2, alpha(N), coeff_e(N); //matrice W2 dans chaque element, taux de vide a la face
120 w2.set_smart_resize(1);
121
122 for (e = 0; e < ne_tot; e++)
123 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(0); i++)
124 if ((f = e_f(e, i)) < zone.nb_faces())
125 {
126 /* taux de vide a la face (identique a celui de Masse_PolyMAC_Face) */
127 double prefac = (e == f_e(f, 0) ? 1 : -1) * pe(e) * vfd(f, e != f_e(f, 0)) / fs(f); /* ponderation pour elimner p_f si on est en TPFA */
128 for (alpha = 0, j = 0; j < 2 && (eb = f_e(f, j)) >= 0; j++)
129 for (n = 0; n < N; n++) alpha(n) += vfd(f, j) * (alp ? (*alp)(eb, n) : 1) / vf(f);
130 for (coeff_e = 0, j = 0; j < w2.dimension(1); j++)
131 if (w2(i, j, 0))
132 for (fb = e_f(e, j), n = 0, m = 0; n < N; n++, m += (M > 1))
133 {
134 double fac = alpha(n) * w2(i, j, 0) * prefac;
135 secmem(f, n) -= fac * (press(ne_tot + fb, m) - press(e, m));
136 if (mat && fcl(fb, 0) != 1) (*mat)(N * f + n, M * (ne_tot + fb) + m) += fac; /* bloc (face, face) */
137 coeff_e(n) += fac;
138 }
139 if (mat)
140 for (n = 0, m = 0; n < N; n++, m += (M > 1)) (*mat)(N * f + n, M * e + m) -= coeff_e(n); /* bloc (face, elem) */
141 }
142}
