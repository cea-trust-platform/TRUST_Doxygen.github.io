83//! Creation de la liste des sommets du sous-domaine "partie".<br>C'est l'ensemble des sommets des elements appartenant a ce sous-domaine.<br>On ne traite que les elements reels.<br>Parametre: nb_sommets<br>Signification: nombre de sommets de la zone globale<br>Parametre: les_elems<br>Signification tableau des elements de la zone globale (pour chaque element, numeros de ses sommets)<br>Parametre: elem_part<br>Signification: tableau de decoupage (pour chaque element i du domaine global,<br>elem_part[i] est le numero du sous-domaine auquel il est affecte)<br>Parametre: partie<br>Signification: le numero du sous-domaine a construire<br>Parametre: liste_sommets<br>Signification: en sortie : liste des sommets du sous-domaine: liste_sommets[i] est<br>l'indice dans la zone_globale du i-ieme sommet du sous-domaine.<br>Les indices sont classes dans l'ordre croissant.<br>Parametre: liste_inverse_sommets<br>Signification: en sortie : on lui donne la taille nb_sommets et on l'initialise.<br>liste_inverse_sommet[i] est l'indice du sommet dans le sous-domaine<br>ou -1 si le sommet i n'est pas dans le sous-domaine)
84static void construire_liste_sommets_sousdomaine(const int nb_sommets,
85 const IntTab& les_elems,
86 const ArrOfInt& liste_elements,
87 const int i_part,
88 const Static_Int_Lists *som_raccord,
89 ArrOfInt& liste_sommets,
90 ArrOfInt& liste_inverse_sommets)
91{
92 const int nb_elem_part = liste_elements.size_array();
93 const int nb_sommets_par_element = les_elems.dimension(1);
94
95// Algorithme : on parcourt les elements: pour les elements de la partie,
96// on marque les sommets de l'element par un drapeau.
97// Puis on fait une boucle sur les sommets, et ceux dont le drapeau
98// est mis sont mis dans la liste.
99
100// D'abord, on compte les sommets de la partie et on
101// remplit drapeau_sommet
102 ArrOfBit drapeau_sommet(nb_sommets);
103
104 drapeau_sommet = 0;
105// Nombre de sommets de la partie "part"
106 int nb_sommets_part = 0;
107 for (int i_elem = 0; i_elem < nb_elem_part; i_elem++)
108 {
109 const int elem = liste_elements[i_elem];
110 for (int j = 0; j < nb_sommets_par_element; j++)
111 {
112 int sommet = les_elems(elem, j);
113 if (sommet>-1)
114 {
115 int bit = drapeau_sommet.testsetbit(sommet);
116// Si le drapeau n'etait pas mis, cela fait un sommet de plus
117 if (! bit)
118 nb_sommets_part++;
119 }
120 }
121 }
122// sommets a ajouter a cause de som_raccord
123 if (som_raccord)
124 for (int s = 0; s < som_raccord->get_nb_lists(); s++)
125 for (int i = 0; i < som_raccord->get_list_size(s); i++)
126 if ((*som_raccord)(s, i) == i_part && !drapeau_sommet.testsetbit(s)) //le sommet est demande par ce proc
127 nb_sommets_part++; //si on ne l'a pas deja, on le rajoute
128
129// Remplissage de liste_sommets et liste_inverse_sommets
130 liste_sommets.resize_array(0); //!< On oublie les valeurs precedentes
131 liste_sommets.resize_array(nb_sommets_part);
132
133 liste_inverse_sommets.resize_array(0); //!< On oublie les valeurs precedentes
134 liste_inverse_sommets.resize_array(nb_sommets,Array_base::NOCOPY_NOINIT);
135 liste_inverse_sommets = -1;
136
137 int n = 0;
138 for (int i = 0; i < nb_sommets; i++)
139 {
140 if (drapeau_sommet[i])
141 {
142 liste_sommets[n] = i;
143 liste_inverse_sommets[i] = n;
144 n++;
145 }
146 }
147}

149// Remplissage du tableau des coordonnees des sommets d'une partie
150// a partir des coordonnees des sommets du domaine complet (sommets_glob)
151// et de la liste des sommets de la partie (liste_sommets)
152// On cree sommets_loc comme suit:
153// sommets_loc.dimension(0) = liste_sommets.size_array()
154// sommets_loc.dimension(1) = sommets_glob.dimension(1)
155//
156// Parametre: sommets_glob
157// Signification: les coordonnees des sommets du domaine global
158// Parametre: liste_sommets
159// Signification: les indices des sommets de sommets_glob a copier
160// dans sommets_loc.
161static void remplir_coordsommets_sous_domaine(const DoubleTab& sommets_glob,
162 const ArrOfInt& liste_sommets,
163 DoubleTab& sommets_loc)
164{
165 const int nb_som = liste_sommets.size_array();
166 const int dim = sommets_glob.dimension(1);
167 sommets_loc.resize(nb_som, dim);
168
169 int i;
170 for (i = 0; i < nb_som; i++)
171 {
172 int indice = liste_sommets[i];
173 int j;
174 for (j = 0; j < dim; j++)
175 sommets_loc(i, j) = sommets_glob(indice, j);
176 }
177}

179// Construction du tableau elems des elements de la partie "part".
180// elems(i, j) sera le nouveau numero du sommet j de l'element i
181// dans la partie part. On utilise la liste_inverse pour obtenir les
182// nouveaux numeros des sommets.
183// Les elements du domaine local sont crees dans l'ordre croissant de
184// leur indice dans la zone globale.
185
186void construire_elems_sous_domaine(const IntTab& elems_zone_globale,
187 const ArrOfInt& liste_elements,
188 const ArrOfInt& liste_inverse_sommets,
189 IntTab& elems_zone_locale,
190 ArrOfInt& liste_inverse_elements)
191{
192 const int nb_elem_tot = elems_zone_globale.dimension_tot(0);
193 const int nb_sommets_par_element = elems_zone_globale.dimension(1);
194
195 liste_inverse_elements.resize(nb_elem_tot,Array_base::NOCOPY_NOINIT);
196 liste_inverse_elements = -1;
197
198// Premier passage: comptage du nombre d'elements de la partie
199 const int nb_elem_part = liste_elements.size_array();
200 elems_zone_locale.resize(nb_elem_part, nb_sommets_par_element);
201
202// Deuxieme passage: remplissage du tableau
203 for (int i_elem = 0; i_elem < nb_elem_part; i_elem++)
204 {
205 int elem = liste_elements[i_elem];
206
207// Nouveau numero de l'element dans la partie
208 liste_inverse_elements[elem] = i_elem;
209// Copie de l'element avec traduction du numero des sommets en
210// numero local dans le sous-domaine
211 for (int i = 0; i < nb_sommets_par_element; i++)
212 {
213 int sommet = elems_zone_globale(elem, i);
214 if (sommet<0)
215 elems_zone_locale(i_elem, i) =sommet;
216 else
217 {
218 int new_num = liste_inverse_sommets[sommet];
219 assert(new_num >= 0);
220 elems_zone_locale(i_elem, i) = new_num;
221 }
222 }
223 }
224}

238//! Pour une liste de "faces" de la zone globale, compter le nombre de<br>faces incluses dans la partie "part" et les copier dans la structure<br>faces_partie en remplacant les numeros de sommets par les numeros locaux<br>dans le sous-domaine.<br>L'ordre des faces est conserve (si une face apparait avant une autre dans<br>la liste du domaine complet et si elles sont toutes les deux dans la sous-partie,<br>alors leur ordre est conserve). C'est important pour le periodique<br>(hypothese qu'il y a correspondance entre la face i et la face i+n/2 du bord<br>periodique).<br>Attention: la condition pour qu'une face soit incluse est "la face appartient<br>a un element de la partie voisine". La condition "les sommets des faces sont<br>des sommets de joint" n'est PAS suffisante.
239void construire_liste_faces_sous_domaine(const ArrOfInt& elements_voisins,
240 const IntVect& elem_part,
241 const int partie,
242 const IntTab& faces_sommets,
243 const ArrOfInt& liste_inverse_sommets,
244 IntTab& faces_sommets_partie)
245{
246 const int nb_faces = faces_sommets.dimension(0);
247 const int nb_sommets_par_face = faces_sommets.dimension(1);
248 int i;
249
250 assert(elements_voisins.size_array() == nb_faces);
251// La liste des faces du tableau faces_sommets a inclure dans le sous-domaine
252 ArrOfInt liste_faces;
253 liste_faces.set_smart_resize(1); //!< Pour faire append_array...
254
255// Premier passage : on cherche les faces a inclure
256 {
257 for (i = 0; i < nb_faces; i++)
258 {
259 int elem = elements_voisins[i];
260 const int part = elem_part[elem];
261 if (part == partie)
262 liste_faces.append_array(i);
263 }
264 }
265
266 const int nb_faces_part = liste_faces.size_array();
267 faces_sommets_partie.resize(nb_faces_part, nb_sommets_par_face);
268
269// Deuxieme passage : stockage des faces et conversion des numeros
270// de sommets des faces en numeros locaux dans le sous-domaine.
271 for (i = 0; i < nb_faces_part; i++)
272 {
273 const int i_face = liste_faces[i];
274 int j;
275 for (j = 0; j < nb_sommets_par_face; j++)
276 {
277 int sommet = faces_sommets(i_face, j);
278 if (sommet>-1)
279 {
280 int new_num = liste_inverse_sommets[sommet];
281 assert(new_num >= 0);
282 faces_sommets_partie(i, j) = new_num;
283 }
284 else
285 faces_sommets_partie(i, j) = -1;
286 }
287 }
288}

425//! A partir d'une liste de sommets de depart (liste_sommets_depart), on<br>parcourt les elements voisins de ces sommets (si epaisseur <= 1),<br>puis les voisins de ces elements (voisins par un sommet de l'element) si epaisseur <= 2,<br>puis les voisins des voisins si epaisseur <= 3, etc<br>Les elements appartenant a la "partie_a_ignorer" ne sont pas parcourus.<br>Les indices des elements parcourus sont ranges dans liste_elements_trouves.<br>Cette methode a ete codee pour construire_elements_distants_ssdom()
426static void parcourir_epaisseurs_elements(const IntTab& elements,
427 const Static_Int_Lists& elem_som,
428 const IntVect& elem_part,
429 ArrOfInt liste_sommets_depart, /* par valeur car on va la modifier */
430 const int partie_a_ignorer, /* ne pas parcourir les elems de cette partie */
431 const int epaisseur,
432 ArrOfInt& liste_elements_trouves)
433{
434 const int nb_sommets = elem_som.get_nb_lists();
435 const int nb_elements = elements.dimension_tot(0);
436 const int nb_som_elem = elements.dimension(1);
437
438 liste_elements_trouves.set_smart_resize(1);
439 liste_elements_trouves.resize_array(0);
440
441 ArrOfBit sommets_parcourus(nb_sommets);
442 ArrOfBit elements_parcourus(nb_elements);
443 ArrOfInt new_liste;
444 liste_sommets_depart.set_smart_resize(1);
445 new_liste.set_smart_resize(1);
446 sommets_parcourus = 0;
447 elements_parcourus = 0;
448
449 {
450// Marquage des sommets de depart et construction d'une liste de sommets uniques
451 const int sz_liste = liste_sommets_depart.size_array();
452 for (int i = 0; i < sz_liste; i++)
453 {
454 const int sommet = liste_sommets_depart[i];
455 if (sommet>-1)
456 if (!sommets_parcourus.testsetbit(sommet))
457 new_liste.append_array(sommet);
458 }
459 }
460
461// Boucle sur les epaisseurs successives d'elements autour des sommets de joint.
462 for (int ep = 1; ep <= epaisseur; ep++)
463 {
464 liste_sommets_depart = new_liste;
465 new_liste.resize_array(0);
466 const int sz_liste = liste_sommets_depart.size_array();
467 for (int i_sommet = 0; i_sommet < sz_liste; i_sommet++)
468 {
469 const int sommet = liste_sommets_depart[i_sommet];
470// Parcours des elements voisins de ce sommet :
471 const int nb_elems_voisins = elem_som.get_list_size(sommet);
472 for (int i_elem = 0; i_elem < nb_elems_voisins; i_elem++)
473 {
474 const int elem = elem_som(sommet, i_elem);
475 if (elements_parcourus[elem])
476 continue; //!< Cet element a deja ete visite Numero de zone de l'element voisin:
477//
478 const int part = elem_part[elem];
479 if (part == partie_a_ignorer)
480 continue; //!< Ne traiter que les elements des autres parties
481
482 liste_elements_trouves.append_array(elem);
483 elements_parcourus.setbit(elem);
484
485// Ajout a la prochaine liste de sommets a traiter les sommets de cet element:
486 for (int i = 0; i < nb_som_elem; i++)
487 {
488 const int sommet2 = elements(elem, i);
489 if (sommet2>-1)
490 {
491 if (sommets_parcourus.testsetbit(sommet2) == 0)
492 new_liste.append_array(sommet2);
493 }
494 }
495 }
496 }
497 }
498}

523// (elements des autres parties situes dans l'epaisseur de joint en partant
524// des sommets de joint).
525 ArrOfInt elements_virtuels;
526 {
527// Construction d'un tableau contenant tous les sommets de tous les joints
528 ArrOfInt sommets_joint;
529 sommets_joint.set_smart_resize(1);
530 const int nb_joints = zone_partie.nb_joints();
531 for (int i_joint = 0; i_joint < nb_joints; i_joint++)
532 {
533 const ArrOfInt& sommets = zone_partie.joint(i_joint).joint_item(Joint::SOMMET).items_communs();
534 const int n = sommets.size_array();
535 for (int i = 0; i < n; i++)
536 {
537 const int sommet_local = sommets[i];
538 const int sommet_global = liste_sommets[sommet_local];
539 sommets_joint.append_array(sommet_global);
540 }
541 }
542
543 parcourir_epaisseurs_elements(elements, som_elem_, elem_part,
544 sommets_joint, //!< Sommets de depart
545 partie, /* ignorer les elements de la partie locale */
546 epaisseur_joint_,
547 elements_virtuels);
548 }
549 const int nb_elements_virtuels = elements_virtuels.size_array();
550// Construire la liste des numeros de zones voisines et ajouter les joints manquants:
551 ArrOfInt parties_voisines;
552 {
553 parties_voisines.set_smart_resize(1);
554 for (int i = 0; i < nb_elements_virtuels; i++)
555 {
556 const int elem = elements_virtuels[i];
557 const int part = elem_part[elem];
558 parties_voisines.append_array(part);
559 }
560 array_trier_retirer_doublons(parties_voisines);
561 ajouter_joints(zone_partie, parties_voisines, epaisseur_joint_);
562 }
563
564// Pour chaque partie voisine, chercher les elements distants:
565 const int nb_parties_voisines = parties_voisines.size_array();
566 ArrOfInt sommets_depart;
567 sommets_depart.set_smart_resize(1);
568 for (int i_part = 0; i_part < nb_parties_voisines; i_part++)
569 {
570 const int partie_voisine = parties_voisines[i_part];
571 sommets_depart.resize_array(0);
572// Liste des sommets des elements virtuels de la partie voisine:
573 for (int i_elem = 0; i_elem < nb_elements_virtuels; i_elem++)
574 {
575 const int elem = elements_virtuels[i_elem];
576 if (elem_part(elem) == partie_voisine)
577 {
578 for (int i = 0; i < nb_som_elem; i++)
579 {
580 const int som = elements(elem, i);
581 sommets_depart.append_array(som);
582 }
583 }
584 }
585
586// Recherche des elements voisins de cette liste:
587 ArrOfInt elements_distants;
588 parcourir_epaisseurs_elements(elements, som_elem_, elem_part,
589 sommets_depart,
590 partie_voisine, /* ignorer les elements de la partie voisine */
591 epaisseur_joint_,
592 elements_distants);
593
594// Retirer de la liste les elements qui ne sont pas dans la partie locale:
595 {
596 int count = 0;
597 const int n = elements_distants.size_array();
598 for (int i = 0; i < n; i++)
599 {
600 const int elem = elements_distants[i];
601 if (elem_part[elem] == partie && elem < elements.dimension(0) )
602 elements_distants[count++] = elem;
603 }
604 elements_distants.resize_array(count);
605 }
606// Traduire les indices des elements en indices locaux sur le sous-domaine.
607 {
608 const int n = elements_distants.size_array();
609 for (int i = 0; i < n; i++)
610 {
611 const int elem = elements_distants[i];
612 const int renum = liste_inverse_elements[elem];
613 elements_distants[i] = renum;
614 }
615 }
616// Trier les elements dans l'ordre croissant des indices
617 elements_distants.ordonne_array();
618// Stocker la liste des elements distants dans le joint
619 Joint& joint = zone_partie.joint_of_pe(partie_voisine);
620 joint.set_joint_item(Joint::ELEMENT).set_items_distants() = elements_distants;
621 }
622}

635//! Creation des joints et construction des listes et des<br>tableaux de sommets joints pour tous les joints de<br>la partie part. Les sommets du joint avec le "PEvoisin" sont<br>les sommets de la partie "part" qui sont aussi un sommet d'un<br>element appartenant au PEvoisin.<br>Les joints apparaissent dans la liste dans l'ordre croissant des PEs.<br>Note: Les faces que l'on va creer ensuite utilisent forcement<br>des sommets que l'on va trouver ici.<br>Propriete : Les sommets du joint sont classes dans l'ordre croissant<br>de leur numero local, donc dans l'ordre croissant de<br>leur numero global (voir Remplir_Numeros_Sommets)
636
637void DomaineCutter::construire_sommets_joints_ssdom(const ArrOfInt& liste_sommets,
638 const ArrOfInt& liste_inverse_sommets,
639 const int partie,
640 const Static_Int_Lists* som_raccord,
641 Zone& zone_partie) const
642{
643 Joints& les_joints = zone_partie.faces_joint();
644
645 const int parts = nb_parties_;
646 const IntVect& elem_part = ref_elem_part_.valeur();
647 const Static_Int_Lists& som_elem = som_elem_;
648
649// Liste de sommets de joint (toutes parties confondues, un sommet peut
650// apparaitre plusieurs fois dans le tableau, une fois par pe voisin au maximum)
651// C'est le numero global du sommet
652 ArrsOfInt joints_sommets(parts);
653 {
654// Pour append_array :
655 for (int i = 0; i < parts; i++)
656 joints_sommets[i].set_smart_resize(1);
657 }
658
659// Algorithme : boucle sur les sommets de la partie.
660// Pour chaque sommet du sous-domaine "p", on cherche les PEs auxquels appartiennent
661// les elements voisins du sommet, et, s'il y a un PE voisin different de "p",
662// on ajoute ce sommet aux joints avec ces PEs.
663
664 const int nb_sommets = liste_sommets.size_array();
665
666 int i_sommet;
667// i = indice local du sommet dans le sous-domaine
668 for (i_sommet = 0; i_sommet < nb_sommets; i_sommet++)
669 {
670// Numero global du sommet
671 int sommet = liste_sommets[i_sommet];
672// Boucle sur les elements voisins du sommet i
673 int nb_elements_voisins = som_elem.get_list_size(sommet);
674 int i;
675 for (i = 0; i < nb_elements_voisins; i++)
676 {
677 int elem = som_elem(sommet, i);
678 int PEvoisin = elem_part[elem];
679// Faut-il ajouter ce sommet aux sommets du joint avec le PEvoisin ?
680 if (PEvoisin != partie)
681 joints_sommets[PEvoisin].append_array(sommet);
682 }
683// boucle sur les procs connectes au sommet par un raccord
684 if (som_raccord)
685 for (i = 0; i < som_raccord->get_list_size(sommet); i++)
686 if ((*som_raccord)(sommet, i) != partie) joints_sommets[(*som_raccord)(sommet, i)].append_array(sommet);
687 }
688
689// Creation des joints : dans l'ordre croissant du PE voisin
690 int PEvoisin;
691 for (PEvoisin = 0; PEvoisin < parts; PEvoisin++)
692 {
693 ArrOfInt& sommets = joints_sommets[PEvoisin];
694 if (sommets.size_array() > 0)
695 {
696// Trier les sommets de joint par ordre croissant de l'indice global
697// (donc toutes les listes sur les processeurs voisins seront ordonnees
698// de la meme facon) et retirer les doublons
699 array_trier_retirer_doublons(sommets);
700 const int nb_sommets2 = sommets.size_array();
701
702// On cree un nouveau joint
703 Joint& joint = les_joints.add(Joint());
704 Nom nom_joint("Joint_");
705 Nom nom_numero(PEvoisin);
706 nom_joint+=nom_numero;
707 joint.nommer(nom_joint);
708 joint.affecte_epaisseur(epaisseur_joint_);
709 joint.associer_zone(zone_partie);
710 joint.affecte_PEvoisin(PEvoisin);
711 ArrOfInt& sommets_locaux = joint.set_joint_item(Joint::SOMMET).set_items_communs();
712 sommets_locaux.resize_array(nb_sommets2);
713// Remplissage du tableau (transformation en indice local de sommet)
714 for (int i = 0; i < nb_sommets2; i++)
715 {
716 const int indice_global = sommets[i];
717 const int indice_local = liste_inverse_sommets[indice_global];
718
719 assert(indice_local >= 0);
720 sommets_locaux[i] = indice_local;
721 }
722 }
723 }
724
725}

727// Recherche des faces de joints (faces adjacentes a deux elements appartenant
728// a deux processeurs differents).
729// On suppose que les sommets des joints ont ete construits.
730
731void DomaineCutter::construire_faces_joints_ssdom(const int partie,
732 const DomaineCutter_Correspondance& correspondance,
733 Zone& zone_partie) const
734{
735 const int nb_sommets_ssdom = zone_partie.nb_som();
736 const int nb_elem_ssdom = zone_partie.nb_elem();
737 const ArrOfInt& liste_sommets = correspondance.get_liste_sommets();
738 const ArrOfInt& liste_inverse_elements = correspondance.get_liste_inverse_elements();
739 const Static_Int_Lists& som_elem = som_elem_;
740
741// Premiere etape: reperage des "elements de joint" (elements de la "partie"
742// possedant un sommet de joint), et marquage des sommets de joint
743 ArrOfInt liste_elements_joint;
744 ArrOfBit drapeaux_sommets_joints(nb_sommets_ssdom);
745 {
746// Pour l'instant, on utilise les joints en const...
747 const Joints& joints_partie = zone_partie.faces_joint();
748
749 drapeaux_sommets_joints = 0;
750 liste_elements_joint.set_smart_resize(1);
751
752// Marqueurs: pour chaque element du sous-domaine, est-il deja dans la liste ?
753 ArrOfBit drapeaux_elements(nb_elem_ssdom);
754 drapeaux_elements = 0;
755// Boucle sur tous les sommets de joint
756 const int nb_joints = joints_partie.size();
757 int i_joint;
758 for (i_joint = 0; i_joint < nb_joints; i_joint++)
759 {
760 const Joint& joint = joints_partie[i_joint];
761 const ArrOfInt& sommets_du_joint = joint.joint_item(Joint::SOMMET).items_communs();
762 const int n = sommets_du_joint.size_array();
763 int i;
764 for (i = 0; i < n; i++)
765 {
766 const int i_sommet_local = sommets_du_joint[i];
767 const int i_sommet_global = liste_sommets[i_sommet_local];
768 const int nb_elem_voisins = som_elem.get_list_size(i_sommet_global);
769// Marquage du sommet de joint dans le sous-domaine
770 drapeaux_sommets_joints.setbit(i_sommet_local);
771 int j;
772 for (j = 0; j < nb_elem_voisins; j++)
773 {
774 const int i_elem_global = som_elem(i_sommet_global, j);
775 const int i_elem_local = liste_inverse_elements[i_elem_global];
776// On ajoute l'element a la liste d'elements joint s'il est dans ma partie
777// et s'il n'est pas encore dans la liste.
778 if (i_elem_local >= 0)
779 if (! drapeaux_elements.testsetbit(i_elem_local))
780 liste_elements_joint.append_array(i_elem_global);
781 }
782 }
783 }
784 }
785
786// Deuxieme etape: pour chaque element de joint, construction de ses
787// faces et on regarde si la face a un element voisin dans une autre partie.
788// Si oui, c'est une face de joint
789
790// Tableau des indices des sommets de toutes les faces de joint
791// (tous joints confondus, indices de sommets dans le sous-domaine)
792 IntTab faces_joints;
793// Pour chaque face de joint, numero du sous-domaine voisin par cette face
794 ArrOfInt faces_pe_voisins;
795// Pour chaque processeur, nombre de faces de joint avec ce proc.
796// (on initialise a zero)
797 ArrOfInt nb_faces_joints(nb_parties_);
798 {
799 faces_joints.set_smart_resize(1);
800 faces_pe_voisins.set_smart_resize(1);
801
802// Indices des sommets des faces sur l'element de reference
803 IntTab faces_element_reference;
804 const Elem_geom_base& type_elem = zone_partie.type_elem().valeur();
805 int is_regular = type_elem.get_tab_faces_sommets_locaux(faces_element_reference);
806// On ne sait pas traiter les elements non reguliers
807 if (!is_regular)
808 {
809// Cerr << "DomaineCutter::faces_joints Error: non-regular elements not supported" << finl;
810 ref_cast(Poly_geom_base,type_elem).get_tab_faces_sommets_locaux(faces_element_reference,0);
811// exit();
812 }
813
814 int nb_faces_elem = faces_element_reference.dimension(0);
815 const int nb_sommets_par_face = faces_element_reference.dimension(1);
816 faces_joints.resize(0, nb_sommets_par_face); //!< Voir *suite*
817 const ArrOfInt& liste_inverse_sommets = correspondance.get_liste_inverse_sommets();
818 const IntVect& elem_part = ref_elem_part_.valeur();
819// Sommets des elements du maillage global
820 const Domaine& domaine = ref_domaine_.valeur();
821 const IntTab& elem_som = domaine.zone(0).les_elems();
822
823 ArrOfInt une_face(nb_sommets_par_face);
824 ArrOfInt elements_voisins;
825 elements_voisins.set_smart_resize(1);
826
827// Boucle sur les elements de joint
828 const int nb_elem_joints = liste_elements_joint.size_array();
829 int i_elem_joint;
830 for (i_elem_joint = 0; i_elem_joint < nb_elem_joints; i_elem_joint++)
831 {
832// Indice de l'element dans le domaine global
833 const int i_elem_global = liste_elements_joint[i_elem_joint];
834// Boucle sur les faces de l'element
835 int i_face;
836 if (!is_regular)
837 {
838// global ou non ????????
839 ref_cast(Poly_geom_base,type_elem).get_tab_faces_sommets_locaux(faces_element_reference,liste_inverse_elements[i_elem_global]);
840 nb_faces_elem = faces_element_reference.dimension(0);
841 while ( faces_element_reference(nb_faces_elem-1,0)==-1)
842 nb_faces_elem--;
843 }
844 for (i_face = 0; i_face < nb_faces_elem; i_face++)
845 {
846// Construction de la face
847 int i;
848 int face_ok = 1;
849 for (i = 0; i < nb_sommets_par_face; i++)
850 {
851 const int i_ref = faces_element_reference(i_face, i);
852 if (i_ref<0)
853 {
854 une_face[i] = i_ref;
855 }
856 else
857 {
858 const int i_som = elem_som(i_elem_global, i_ref);
859 une_face[i] = i_som;
860// Indice du sommet dans le sous-domaine
861
862 if (i_som>-1)
863 {
864 const int i_som_local = liste_inverse_sommets[i_som];
865 if(i_som_local>0)
866 {
867// Le sommet est-il sur un joint ?
868 if (drapeaux_sommets_joints[i_som_local] == 0)
869 face_ok = 0; //!< Non => cette face n'est pas sur un joint
870 }
871 }
872 }
873 }
874// Premier test pour eliminer tout de suite la face
875// si tous ses sommets ne sont pas des sommets de joint.
876 if (face_ok)
877 {
878// Recherche des elements voisins de cette face
879 find_adjacent_elements(som_elem, une_face, elements_voisins);
880// Recherche d'un element voisin qui n'est pas dans ma partie
881 const int nb_elem_voisins = elements_voisins.size_array();
882 int PEvoisin = -1;
883 for (i = 0; i < nb_elem_voisins; i++)
884 {
885 const int i_elem_glob = elements_voisins[i];
886 PEvoisin = elem_part[i_elem_glob];
887 if (PEvoisin != partie)
888 break;
889 }
890 if (i < nb_elem_voisins)
891 {
892// J'ai une face de joint, je l'ajoute au tableau
893 faces_pe_voisins.append_array(PEvoisin);
894 const int n = faces_joints.dimension(0);
895 faces_joints.resize(n+1, nb_sommets_par_face);
896 for (i = 0; i < nb_sommets_par_face; i++)
897 {
898 const int i_som_global = une_face[i];
899 if (i_som_global>-1)
900 {
901 const int i_som_local = liste_inverse_sommets[i_som_global];
902 faces_joints(n, i) = i_som_local;
903 }
904 else
905 faces_joints(n, i) = -1;
906 }
907 nb_faces_joints[PEvoisin]++;
908 }
909 }
910 }
911 }
912 }
913
914// Troisieme etape: ajout des faces dans les joints
915 {
916// Le type des faces de joint
917 const Zone& zone_globale = ref_domaine_.valeur().zone(0);
918 const Type_Face& type_face_joint = zone_globale.type_elem().type_face();
919// On va modifier les joints:
920 Joints& joints_partie = zone_partie.faces_joint();
921 const int nb_joints = joints_partie.size();
922 int i_joint;
923
924 const int nb_faces_joints_tot = faces_joints.dimension(0);
925// *suite* : tableau de dimension 2 meme s'il n'y a pas de faces
926 const int nb_sommets_par_face = faces_joints.dimension(1);
927
928 for (i_joint = 0; i_joint < nb_joints; i_joint++)
929 {
930// Typage des faces et allocation memoire
931 Joint& joint = joints_partie[i_joint];
932 joint.typer_faces(type_face_joint);
933 joint.faces().les_sommets().resize(0, nb_sommets_par_face);
934 const int PE_voisin = joint.PEvoisin();
935 const int nb_faces = nb_faces_joints[PE_voisin];
936 joint.dimensionner(nb_faces);
937// Remplissage du tableau de faces
938 IntTab& faces = joint.faces().les_sommets();
939 int i;
940 int k = 0;
941 for (i = 0; i < nb_faces_joints_tot; i++)
942 {
943 const int face_pe = faces_pe_voisins[i];
944 if (face_pe == PE_voisin)
945 {
946 int j;
947 for (j = 0; j < nb_sommets_par_face; j++)
948 faces(k, j) = faces_joints(i,j);
949 k++;
950 }
951 }
952 }
953 }
