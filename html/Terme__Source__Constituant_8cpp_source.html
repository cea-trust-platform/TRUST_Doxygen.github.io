50void Terme_Source_Constituant::ouvrir_fichier(const Equation_base& eq, const Nom& out, const Nom& qsj, const Nom& description, SFichier& os,const Nom& type, const int flag) const
51{
52// flag nul on n'ouvre pas le fichier
53 if (flag==0)
54 return ;
55
56 const Probleme_base& pb = eq.probleme();
57 const Schema_Temps_base& sch = pb.schema_temps();
58
59 Nom nomfichier(out);
60 if (type!="") nomfichier+=(Nom)"_"+type;
61 nomfichier+=".out";
62
63// On cree le fichier a la premiere impression avec l'en tete
64 const int wcol = std::max(colw_, sch.wcol());
65 os.set_col_width(wcol);
66 if (sch.nb_impr()==1 && !pb.reprise_effectuee())
67 {
68 const int gnuplot_header = sch.gnuplot_header();
69 os.ouvrir(nomfichier);
70 SFichier& fic=os;
71 fic << (Nom)"# Printing of the source term " + qsj + " of the equation "+ eq.que_suis_je() +" of the problem "+ eq.probleme().le_nom() << finl;
72 fic << "# " << description << finl;
73 if (!gnuplot_header) fic << "#";
74 os.set_col_width(wcol - !gnuplot_header);
75 fic.add_col("Time");
76 os.set_col_width(wcol);
77 for (int i = 0; i < eq.inconnue()->nb_comp(); ++i)
78 fic.add_col(eq.inconnue()->noms_compo()[i].getChar());
79 fic << finl;
80 }
81// Sinon on l'ouvre
82 else
83 {
84 os.ouvrir(nomfichier,ios::app);
85 }
86 os.precision(sch.precision_impr());
87 os.setf(ios::scientific);
88}
