42//! Lit le terme de puissance thermique a partir<br>d'un flot d'entree et d une equation pour eventuellement<br>acceder a la discretisation afin de typer la_puissance.<br>Lit uniquement un champ donne representant la_puissance.
43void Terme_Puissance_Thermique::lire_donnees(Entree& is,const Equation_base& eqn)
44{
45 Cerr << "Lecture du Champ Puissance" << finl;
46
47 Motcle type;
48 is >> type;
49 if (type=="Valeur_totale_sur_volume")
50 {
51 const Zone_dis_base& zdis = eqn.zone_dis().valeur();
52 const Nom& type_zdis = zdis.que_suis_je();
53 if (type_zdis.debute_par("Zone_VDF"))
54 type += "_VDF";
55 else if (type_zdis.debute_par("Zone_VEF"))
56 type += "_VEF";
57 }
58 la_puissance_lu.typer(type);
59 Champ_Don_base& ch_puissance_lu = ref_cast(Champ_Don_base,la_puissance_lu.valeur());
60 is >> ch_puissance_lu;
61 const int nb_comp = ch_puissance_lu.nb_comp();
62
63 eqn.probleme().discretisation().discretiser_champ("CHAMP_ELEM", eqn.zone_dis(), "pp", "1",nb_comp,0., la_puissance);
64 la_puissance_lu->fixer_nb_comp(nb_comp);
65 if (ch_puissance_lu.le_nom()=="anonyme") ch_puissance_lu.nommer("Puissance_volumique");
66
67 for (int n = 0; n < nb_comp; n++) la_puissance_lu->fixer_nom_compo(n, ch_puissance_lu.le_nom() + (nb_comp > 1 ? Nom(n) :""));
68 for (int n = 0; n < nb_comp; n++) la_puissance->fixer_nom_compo(n, ch_puissance_lu.le_nom() + (nb_comp > 1 ? Nom(n) :""));
69// PL: Il faut faire nommer_completer_champ_physique les 2 champs (plantage sinon pour une puissance de type Champ_fonc_tabule)
70 eqn.discretisation().nommer_completer_champ_physique(eqn.zone_dis(),ch_puissance_lu.le_nom(),"W/m3",la_puissance_lu,eqn.probleme());
71 eqn.discretisation().nommer_completer_champ_physique(eqn.zone_dis(),ch_puissance_lu.le_nom(),"W/m3",la_puissance,eqn.probleme());
72 la_puissance.valeur().valeurs() = 0;
73 la_puissance.valeur().affecter(ch_puissance_lu);
74}

77{
