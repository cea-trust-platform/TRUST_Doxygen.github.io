78void Schema_Adams_Bashforth_base::modifier_second_membre(const Equation_base& eqn, DoubleTab& secmem)
79{
80 if (nb_pas_dt()> nb_pas_dt_seuil())
81 {
82
83 const Equation_base& eqn_bis = ref_cast(Equation_base,eqn);
84 int i = 0;
85 int offset = 0;
86 double time_step = pas_de_temps(); //tn+1-tn
87
88 DoubleTab times(nb_valeurs_temporelles());
89
90// Building of a time-in-ascending-order table
91 assert(nb_valeurs_passees()+2 == nb_valeurs_temporelles());
92
93 for (i=0; i<nb_valeurs_passees(); ++i)
94 {
95 offset = nb_valeurs_passees()-i;
96 times[i] = eqn_bis.inconnue().valeur().recuperer_temps_passe(offset); //past
97 }
98 times[nb_valeurs_passees()] = eqn_bis.inconnue().valeur().recuperer_temps_futur(0); //present
99 times[nb_valeurs_passees()+1] = eqn_bis.inconnue().valeur().recuperer_temps_futur(1); //future
100
101 compute_coefficients(time_step,times);
102
103// Modification due to present time and past times
104 secmem*=coefficients()[nb_valeurs_passees()]; //present time
105
106 DoubleTrav dudt(eqn.inconnue().valeurs());
107 DoubleTrav tmp(dudt);
108// Compute sum(coeff*dudt)
109 for (i=0; i<nb_valeurs_passees(); ++i)
110 {
111 offset = nb_valeurs_passees()-i;
112 tmp = eqn.derivee_en_temps().passe(offset);
113 tmp *= coefficients()[i];
114 dudt += tmp; //past time
115 }
116// secmem+=Mdu/dt
117 eqn.solv_masse().ajouter_masse(1.,secmem,dudt,0);
118 }
119}
