97Champ_base& Champ_Fonc_Face::affecter_(const Champ_base& ch)
98{
99 const DoubleTab& v = ch.valeurs();
100 DoubleTab& val = valeurs();
101 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
102 int nb_faces = zone_VDF.nb_faces();
103 const IntVect& orientation = zone_VDF.orientation();
104 int ori, n0, n1;
105
106 if (sub_type(Champ_Uniforme,ch))
107 {
108 for (int num_face=0; num_face<nb_faces; num_face++)
109 val(num_face) = v(0,orientation(num_face));
110 }
111 else if ( (sub_type(Champ_Uniforme_Morceaux,ch))
112 || (sub_type(Champ_Don_lu,ch)) )
113 {
114 int ndeb_int = zone_VDF.premiere_face_int();
115 const IntTab& face_voisins = zone_VDF.face_voisins();
116 int num_face;
117
118 for (num_face=0; num_face<ndeb_int; num_face++)
119 {
120 ori = orientation(num_face);
121 n0 = face_voisins(num_face,0);
122 if (n0 != -1)
123 val(num_face) = v(n0,ori);
124 else
125 val(num_face) = v(face_voisins(num_face,1),ori);
126 }
127
128 for (num_face=ndeb_int; num_face<nb_faces; num_face++)
129 {
130 ori = orientation(num_face);
131 n0 = face_voisins(num_face,0);
132 n1 = face_voisins(num_face,1);
133 val(num_face) = 0.5*(v(n0,ori)+v(n1,ori));
134 }
135 }
136 else
137 {
138// int ndeb_int = zone_VDF.premiere_face_int();
139// const IntTab& face_voisins = zone_VDF.face_voisins();
140 DoubleTab positionX(zone_VDF.nb_faces_X() ,dimension);
141 DoubleVect U(zone_VDF.nb_faces_X());
142 DoubleTab positionY(zone_VDF.nb_faces_Y() ,dimension);
143 DoubleVect V(zone_VDF.nb_faces_Y());
144 DoubleTab positionZ(zone_VDF.nb_faces_Z() ,dimension);
145 DoubleVect W(zone_VDF.nb_faces_Z());
146 const DoubleTab& xv=zone_VDF.xv();
147 int nbx=0;
148 int nby=0;
149 int nbz=0;
150 int num_face, k;
151
152 for (num_face=0; num_face<nb_faces; num_face++)
153 {
154 ori = orientation(num_face);
155 switch(ori)
156 {
157 case 0:
158 for(k=0; k<dimension; k++)
159 positionX(nbx, k)=xv(num_face, k);
160 nbx++;
161 break;
162 case 1:
163 for(k=0; k<dimension; k++)
164 positionY(nby, k)=xv(num_face, k);
165 nby++;
166 break;
167 case 2:
168 for(k=0; k<dimension; k++)
169 positionZ(nbz, k)=xv(num_face, k);
170 nbz++;
171 break;
172 }
173 }
174
175 ch.valeur_aux_compo(positionX, U, 0);
176 ch.valeur_aux_compo(positionY, V, 1);
177 if(dimension == 3)
178 ch.valeur_aux_compo(positionZ, W, 2);
179 nbx=nby=nbz=0;
180 for (num_face=0; num_face<nb_faces; num_face++)
181 {
182 ori = orientation(num_face);
183 switch(ori)
184 {
185 case 0:
186 val(num_face) = U(nbx++);
187 break;
188 case 1:
189 val(num_face) = V(nby++);
190 break;
191 case 2:
192 val(num_face) = W(nbz++);
193 break;
194 }
195 }
196 }
197 return *this;
198}
