50 /* elimination par blocs d'un systeme lineaire */
51// entree : ordre -> groupes de (variables, indice de bloc) a eliminer successivement : par ex. { { {"vitesse", 0 } }, { {"alpha", 0 }, {"temperature", 0 } } }
52// inco_p -> nom de l'inconnue principale (ex. : "pression")
53// mats, sec : matrices / seconds membres du systeme lineaire mats.d{incos} = {secs}
54//
55//
56// sortie : A_p / b_p t.q. d{inco} = A_p.d{inco_p} + b_p
57//
58// contraintes : - les inconnues du meme bloc doivent partager un meme MD_Vector
59// - pour chaque bloc { i_1, i_2 }, la matrice { mats[i_j][i_k] } doit etre diagonale par blocs par rapport a ce MD_Vector
60// - hors cette diagonale, les inconnues d'un blocs ne peuvent dependre que des blocs precedents et de inco_p
61

65 /* assemblage d'un systeme en inco_p a partir des expressions d.{inco} : A_p[inco].d{inco_p} + b_p[inco] */
66// entree : - inco_p -> l'inconnue principale
67// - A_p, b_p -> expressions des autres inconnues calculees par eliminer()
68// - mats, sec -> systeme lineaire contenant une equation sur inco_p (second membre dans sec[inco_p], jacobienne dans mats[inco_p])
69//
70// sortie : systeme matrice.d{inco_p} = secmem
71//
72// contraintes : toutes les autres inconnues doivent etre exprimees dans A_p / b_p
