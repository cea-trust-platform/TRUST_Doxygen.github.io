117void Champ_front_t::mettre_a_jour(double temps)
118{
119 int dim=nb_comp();
120 const Frontiere_dis_base& fr_dis=frontiere_dis();
121 int nb_faces=ref_cast(Front_VF, fr_dis).nb_faces();
122 int i,k;
123 DoubleTab& tab=valeurs_au_temps(temps);
124 for( i=0; i<nb_faces; i++)
125 {
126 for( k=0; k<dim; k++)
127 {
128 f_t[k].setVar("t",temps);
129 tab(i,k)=f_t[k].eval();
130 }
131 }
132 tab.echange_espace_virtuel();
133}
