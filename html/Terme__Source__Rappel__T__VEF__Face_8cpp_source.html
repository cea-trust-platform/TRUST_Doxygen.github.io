43Entree& Terme_Source_Rappel_T_VEF_Face::readOn(Entree& is )
44{
45 Motcle mot_lu;
46 Motcle acc_ouverte("{");
47 Motcle acc_fermee("}");
48
49
50
51 alpha_tau=1.;
52 fct_ok=-1;
53
54 Motcles les_mots(3);
55 {
56 les_mots[0]="alpha_tau"; //coeff de relaxation du terme source
57 les_mots[1]="force_rappel"; //!< Expression de la force de rappel
58 les_mots[2]="probleme"; //!< Expression de la force de rappel
59 }
60 is >> mot_lu;
61 if(mot_lu != acc_ouverte)
62 {
63 Cerr << "On attendait { a la place de " << mot_lu
64 << " lors de la lecture des parametres de la loi de paroi " << finl;
65 }
66 is >> mot_lu;
67 while(mot_lu != acc_fermee)
68 {
69 int rang=les_mots.search(mot_lu);
70 switch(rang)
71 {
72 case 0:
73 is >> alpha_tau;
74 break;
75 case 1:
76 {
77 Nom tmp;
78 is >> tmp;
79 force_rappel.setNbVar(dimension);
80 force_rappel.setString(tmp);
81 force_rappel.addVar("x");
82 force_rappel.addVar("y");
83 if (dimension>2)
84 force_rappel.addVar("z");
85 force_rappel.parseString();
86 fct_ok=1;
87 break;
88 }
89 case 2:
90 fct_ok=0;
91 is >> nom_autre_pb >> nom_inco;
92 break;
93 default :
94 {
95 Cerr << mot_lu << " n'est pas un mot compris par Source_Canal_RANS_LES_VEF_P1NC" << finl;
96 Cerr << "Les mots compris sont : " << les_mots << finl;
97 exit();
98 }
99 }
100 is >> mot_lu;
101 }
102
103 return is;

118void Terme_Source_Rappel_T_VEF_Face::completer()
119{
120 Source_base::completer();
121 if (fct_ok == 0)
122 {
123 const Zone_VEF& zone_VEF = ref_cast(Zone_VEF,equation().zone_dis().valeur());
124 const Zone& zone = zone_VEF.zone();
125 const int nb_elem = zone_VEF.nb_elem();
126
127 Objet_U& ob1=Interprete::objet(nom_autre_pb);
128 REF(Probleme_base) pb;
129 if( sub_type(Probleme_base, ob1) )
130 pb = ref_cast(Probleme_base, ob1);
131 else
132 {
133 Cerr << "On ne trouve pas de probleme de nom " << nom_autre_pb << finl;
134 exit();
135 }
136
137 REF(Champ_base) rch;
138 rch = pb->get_champ(nom_inco);
139 l_inconnue=ref_cast(Champ_Inc_base, rch.valeur()) ;
140
141 zone_VEF_autre_pb = ref_cast(Zone_VEF,pb->domaine_dis().zone_dis(0).valeur());
142 const int nb_elem_autre_pb = zone_VEF_autre_pb->nb_elem();
143 const DoubleTab& xp_autre_pb = zone_VEF_autre_pb->xp() ;
144 const DoubleVect& volumes = zone_VEF_autre_pb->volumes();
145
146// Dimensionnement des tabs
147 nb_fin.resize_array(nb_elem);
148 corresp_fin_gros.resize_array(nb_elem_autre_pb);
149 nb_fin=0;
150 corresp_fin_gros=0;
151 vol_fin_tot.resize_array(nb_elem);
152 vol_fin_tot=0.;
153
154// Calcul des connectivites entre les maillages des deux pbs
155 zone.chercher_elements(xp_autre_pb, corresp_fin_gros);
156 for (int i=0; i<nb_elem_autre_pb; i++)
157 {
158 assert(corresp_fin_gros[i]>0);
159 nb_fin[corresp_fin_gros[i]]++;
160 vol_fin_tot[corresp_fin_gros[i]] += volumes(i);
161 }
162 }

168DoubleTab& Terme_Source_Rappel_T_VEF_Face::ajouter(DoubleTab& resu) const
169{
170 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
171 const int nb_faces = zone_VEF.nb_faces();
172 const int premiere_face_std=zone_VEF.premiere_face_std();
173 const int ndeb = zone_VEF.premiere_face_int();
174
175 const int nb_elem = zone_VEF.nb_elem();
176 const DoubleTab& temperature = mon_equation->inconnue().valeurs();
177 const DoubleVect& volumes_entrelaces = zone_VEF.volumes_entrelaces();
178 const DoubleVect& volumes_entrelaces_Cl = la_zone_Cl_VEF->volumes_entrelaces_Cl();
179 const DoubleTab& xv=zone_VEF.xv() ;
180 const double dt = mon_equation->schema_temps().pas_de_temps();
181
182 const DoubleVect& volumes_autre_pb = zone_VEF_autre_pb->volumes();
183
184 double vol;
185 double force=0.;
186
187 if (fct_ok == 1)
188 {
189 for(int num_face = 0 ; num_face<premiere_face_std ; num_face++)
190 {
191 vol = volumes_entrelaces_Cl(num_face);
192
193 for (int i=0; i<dimension; i++)
194 force_rappel.setVar(i,xv(num_face,i));
195
196 force = force_rappel.eval();
197 resu(num_face) += (force-temperature(num_face))/(alpha_tau*dt)*vol;
198 }
199
200 for(int num_face = 0 ; num_face<nb_faces ; num_face++)
201 {
202 vol = volumes_entrelaces(num_face);
203
204 for (int i=0; i<dimension; i++)
205 force_rappel.setVar(i,xv(num_face,i));
206
207 force = force_rappel.eval();
208 resu(num_face) += (force-temperature(num_face))/(alpha_tau*dt)*vol;
209 }
210 }
211 else
212 {
213 const Zone_VEF& zoneVEF_autre_pb = ref_cast(Zone_VEF,l_inconnue->zone_dis_base());
214 const int nb_elem_autre_pb = zoneVEF_autre_pb.nb_elem();
215 const DoubleTab& xp=zoneVEF_autre_pb.xp() ;
216
217 const IntTab& face_voisins=zone_VEF.face_voisins() ;
218 ArrOfDouble val_elem(nb_elem); //!< valeurs sur les elems du pb courant du champ de l'autre pb.
219
220 DoubleTab valeurs(nb_elem_autre_pb,l_inconnue->nb_comp()); //valeurs sur l'autre pb
221 IntTab elems(nb_elem_autre_pb);
222 for (int i=0; i<nb_elem_autre_pb; i++)
223 elems(i)=i;
224 l_inconnue->valeur_aux_elems(xp, elems, valeurs);
225
226 val_elem=0.;
227 for (int i=0; i<nb_elem_autre_pb; i++)
228 {
229 val_elem[corresp_fin_gros[i]] += valeurs(i,0)*volumes_autre_pb(i)/vol_fin_tot[corresp_fin_gros[i]] ; ///nb_fin(corresp_fin_gros(i));
230 }
231
232
233 for(int num_face = 0 ; num_face<ndeb ; num_face++)
234 {
235 vol = volumes_entrelaces_Cl(num_face);
236
237 force = val_elem[face_voisins(num_face,0)];
238 resu(num_face) += (force-temperature(num_face))/(alpha_tau*dt)*vol;
239 }
240 for(int num_face = ndeb ; num_face<premiere_face_std ; num_face++)
241 {
242 vol = volumes_entrelaces_Cl(num_face);
243
244 force = 0.5*(val_elem[face_voisins(num_face,0)]+val_elem[face_voisins(num_face,1)]);
245 if ((force!=0.))//&& (force-temperature(num_face)>0.))
246 resu(num_face) += (force-temperature(num_face))/(alpha_tau*dt)*vol;
247 }
248
249 for(int num_face = premiere_face_std ; num_face<nb_faces ; num_face++)
250 {
251 vol = volumes_entrelaces(num_face);
252
253 if ((val_elem[face_voisins(num_face,0)] !=0.) && (val_elem[face_voisins(num_face,1)] !=0.))
254 force = 0.5*(val_elem[face_voisins(num_face,0)]+val_elem[face_voisins(num_face,1)]);
255 else
256 force = val_elem[face_voisins(num_face,0)]+val_elem[face_voisins(num_face,1)];
257 if ((force!=0.))//&& (force-temperature(num_face)>0.))
258 {
259// Cout << "force = " << force << finl;
260 resu(num_face) += (force-temperature(num_face))/(alpha_tau*dt)*vol;
261 }
262// Cout << "force-temperature(num_face) = " << force-temperature(num_face) << finl;
263 }
264
265
266
267 }
268
269
270 return resu;
271}
