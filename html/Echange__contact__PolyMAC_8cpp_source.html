50void Echange_contact_PolyMAC::init_op() const
51{
52 if (o_diff.non_nul()) return;
53 Champ_front_calc ch;
54 ch.creer(nom_autre_pb_, nom_bord_, nom_champ_);
55 fvf = ref_cast(Front_VF, frontiere_dis()), o_fvf = ref_cast(Front_VF, ch.front_dis()); //frontieres
56 const Equation_base& eqn = zone_Cl_dis().equation(), &o_eqn = ch.equation(); //equations
57 i_fvf = eqn.zone_dis()->rang_frontiere(fvf.le_nom()), i_o_fvf = o_eqn.zone_dis()->rang_frontiere(nom_bord_);
58
59 int i_op = -1, o_i_op = -1, i; //indice de l'operateur de diffusion dans l'autre equation
60 for (i = 0; i < eqn.nombre_d_operateurs(); i++)
61 if (sub_type(Op_Diff_PolyMAC_Elem, eqn.operateur(i).l_op_base())) i_op = i;
62 if (i_op < 0) Process::exit(le_nom() + " : no diffusion operator found in " + eqn.probleme().le_nom() + " !");
63 for (i = 0; i < o_eqn.nombre_d_operateurs(); i++)
64 if (sub_type(Op_Diff_PolyMAC_Elem, o_eqn.operateur(i).l_op_base())) o_i_op = i;
65 if (o_i_op < 0) Process::exit(le_nom() + " : no diffusion operator found in " + o_eqn.probleme().le_nom() + " !");
66 diff = ref_cast(Op_Diff_PolyMAC_Elem, eqn.operateur(i_op).l_op_base());
67 o_diff = ref_cast(Op_Diff_PolyMAC_Elem, o_eqn.operateur(o_i_op).l_op_base());
68}

70/* identification des elements / faces de l'autre cote de la frontiere, avec offsets */
71void Echange_contact_PolyMAC::init_f_dist() const
72{
73 if (f_dist_init_) return; //deja fait
74 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, fvf->zone_dis()), &o_zone = ref_cast(Zone_PolyMAC, o_fvf->zone_dis());
75 const DoubleTab& xv = zone.xv(), &o_xv = o_zone.xv();
76
77 int i, f, o_f, nf_tot = fvf->nb_faces_tot(), o_nf_tot = o_fvf->nb_faces_tot(), d, D = dimension;
78 f_dist.resize(nf_tot);
79
80 DoubleTrav xvf(nf_tot, D), o_xvf(o_nf_tot, D); //positions locales/distantes -> pour calcul de correspondance
81 for (i = 0; i < nf_tot; i++)
82 for (d = 0; d < D; d++) xvf(i, d) = xv( fvf->num_face(i), d);
83 for (i = 0; i < o_nf_tot; i++)
84 for (d = 0; d < D; d++) o_xvf(i, d) = o_xv(o_fvf->num_face(i), d);
85
86#ifdef MEDCOUPLING_
87 MCAuto<DataArrayDouble> fdad(DataArrayDouble::New()), o_fdad(DataArrayDouble::New());
88 fdad->useExternalArrayWithRWAccess(xvf.addr(), nf_tot, D), o_fdad->useExternalArrayWithRWAccess(o_xvf.addr(), o_nf_tot, D);
89// point de o_fdad le plus proche de chaque point de {f,s}dad
90 MCAuto<DataArrayInt> f_idx(nf_tot && o_nf_tot ? o_fdad->findClosestTupleId(fdad) : NULL);
91
92 for (i = 0; i < nf_tot; i++) //remplissage de f_dist : face distante si coincidence, -1 sinon
93 {
94 f = fvf->num_face(i), o_f = o_nf_tot ? o_fvf->num_face(f_idx->getIJ(i, 0)) : -1;
95 double d2 = o_f >= 0 ? zone.dot(&xv(f, 0), &xv(f, 0), &o_xv(o_f, 0), &o_xv(o_f, 0)) : 1e8;
96 if (d2 < 1e-12) f_dist(i) = o_f;
97 else f_dist(i) = -1;
98 if (i < fvf->nb_faces() && d2 >= 1e-12)
99 Process::exit(Nom("Echange_contact_PolyMAC: missing opposite faces detected between ") + fvf->le_nom() + " and " + o_fvf->le_nom()
100 + " ! Have you used Decouper_multi?");
101 }
102
103#else
104 Process::exit("Echange_contact_PolyMAC : MEDCoupling is required!");
105#endif
106 f_dist_init_ = 1;
107}
