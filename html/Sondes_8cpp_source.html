38//! Lit une liste de sondes a partir d'un flot d'entree<br>Format:<br>{<br>[LIRE UNE SONDE AUTANT DE FOIS QUE NECESSAIRE]<br>}
39/** \return Entree\&<br>
40Signification: le flot d'entree modifie<br>
41Contraintes: */
42Entree& Sondes::readOn(Entree& s )
43{
44 assert(mon_post.non_nul());
45
46 Motcle motlu;
47 Motcle accolade_ouverte("{");
48 Motcle accolade_fermee("}");
49
50 s >> motlu;
51
52 if (motlu != accolade_ouverte)
53 {
54 Cerr << "Error while reading the probes in the postprocessing" << finl;
55 Cerr << "We expected { to start to read the probes" << finl;
56 exit();
57 }
58 s >> motlu;
59
60 bool depuisFichier = false;
61 Nom nom_fichier;
62
63 if (motlu == "FICHIER")
64 {
65 Cerr<<"Warning: Sondes { " << motlu << " .... } no more allowed, use Sondes_fichier { " << motlu << " .... }" <<finl;
66 exit();
67 depuisFichier = true;
68 s >> nom_fichier;
69 }
70 else if (motlu == accolade_fermee)
71 {
72 Cerr << "Error while reading the probes in the postprocessing" << finl;
73 Cerr << "You have not defined any probe" << finl;
74 exit();
75 }
76 set_noms_champs_postraitables();
77 if (depuisFichier)
78 {
79 lire_fichier(nom_fichier);
80 s >> motlu;
81 if (motlu != accolade_fermee)
82 {
83 Cerr << "Error while reading the probes in the postprocessing" << finl;
84 Cerr << "We expected } to end to read the probes" << finl;
85 exit();
86 }
87 }
88 else
89 {
90 while (motlu != accolade_fermee)
91 {
92 Sonde une_sonde(motlu);
93 une_sonde.associer_post(mon_post.valeur());
94 s >> une_sonde;
95 add(une_sonde);
96 s >> motlu;
97 }
98 }
99 Cerr << "End of reading probes " << finl;
100 return s;
101}

103void Sondes::lire_fichier(const Nom& nom_fichier)
104{
105 set_noms_champs_postraitables();
106 LecFicDiffuse_JDD f(nom_fichier);
107 Motcle motlu;
108
109 if (!f.good())
110 {
111 Cerr << "Cannot open the file " << nom_fichier << finl;
112 Process::exit();
113 }
114
115 f >> motlu;
116 while (!f.eof())
117 {
118 Sonde une_sonde(motlu);
119 une_sonde.associer_post(mon_post.valeur());
120 f >> une_sonde;
121 add(une_sonde);
122 f >> motlu;
123 }
124 f.close();
125}

203REF(Champ_base) Sondes::get_from_cache(REF(Champ_Generique_base)& mon_champ, const Nom& nom_champ_lu_)
204{
205 mon_champ->fixer_identifiant_appel(nom_champ_lu_);
206 int num = sourceNoms.rang(nom_champ_lu_);
207 if (num < 0)
208 {
209 Champ espace_stockage;
210 const Champ_base& ma_source = ref_cast(Champ_base, mon_champ->get_champ(espace_stockage));
211 sourceList.add(ma_source);
212 espaceStockageList.add(espace_stockage);
213 sourceNoms.add(nom_champ_lu_);
214// Voir Champ_Generique_base pour la definition de l'espace stockage
215 return espace_stockage.non_nul() ? espaceStockageList.dernier().valeur() : ma_source;
216 }
217 else if (espaceStockageList(num).non_nul())
218 return espaceStockageList(num).valeur();
219 else
220 return sourceList(num);
221}
