59double Op_Dift_VEF_Face_Q1::calculer_dt_stab() const
60{
61// Calcul de dt_stab
62// La diffusivite est constante par elements donc
63// il faut calculer dt_diff pour chaque element et
64// dt_stab=Min(dt_diff (K) = h(K)*h(K)/(2*dimension*diffu2_(K)))
65// ou diffu2_ est la somme des 2 diffusivite laminaire et turbulente
66
67 int num_face;
68 double vol,surf;
69 double surf_max=0;
70 double dt_stab=1.e30;
71 double coef;
72 const Zone_VEF& la_zone_VEF = la_zone_vef.valeur();
73 const DoubleVect& volumes = la_zone_VEF.volumes();
74 const IntTab& elem_faces = la_zone_VEF.elem_faces();
75 const DoubleTab& face_normales = la_zone_VEF.face_normales();
76 const Zone& la_zone= la_zone_VEF.zone();
77 int nb_faces_elem = la_zone.nb_faces_elem();
78 double diffu = (diffusivite_.valeur())(0,0);
79 const DoubleVect& diffu_turb=diffusivite_turbulente()->valeurs();
80 double diffu2_;
81 int la_zone_nb_elem=la_zone.nb_elem();
82 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
83 {
84 surf_max = 1.e-30;
85 for (int i=0; i<nb_faces_elem; i++)
86 {
87 num_face = elem_faces(num_elem,i);
88 surf = face_normales(num_face,0)*face_normales(num_face,0);
89 for (int j=1; j<dimension; j++)
90 surf += face_normales(num_face,j)*face_normales(num_face,j);
91 surf_max = (surf > surf_max)? surf : surf_max;
92 }
93 vol = volumes(num_elem);
94 vol *=vol/surf_max;
95 diffu2_ = diffu + diffu_turb[num_elem];
96 coef=vol/(2.*dimension*(diffu2_+DMINFLOAT));
97 dt_stab = (coef < dt_stab )? coef : dt_stab;
98
99 }
100
101 return Process::mp_min(dt_stab);
102}

107// Fonction utile visc
108// mu <Si, Sj> / |K|
109
110static double viscA_Q1(const Zone_VEF& la_zone,int num_face,int num2,int dimension,
111 int num_elem,double diffu)
112{
113 double constante = (diffu)/la_zone.volumes(num_elem);
114 double Valeur=0;
115
116 if(dimension==2)
117 {
118 Valeur=la_zone.face_normales(num_face,0)*
119 la_zone.face_normales(num2,1) -
120 la_zone.face_normales(num_face,1)*la_zone.face_normales(num2,0);
121 Valeur=constante*std::fabs(Valeur);
122 }
123 else
124 {
125 Valeur=(la_zone.face_normales(num_face,1)*la_zone.face_normales(num2,2) -
126 la_zone.face_normales(num_face,2)*la_zone.face_normales(num2,1)) +
127 (la_zone.face_normales(num_face,2)*la_zone.face_normales(num2,0) -
128 la_zone.face_normales(num_face,0)*la_zone.face_normales(num2,2)) +
129 (la_zone.face_normales(num_face,0)*la_zone.face_normales(num2,1) -
130 la_zone.face_normales(num_face,1)*la_zone.face_normales(num2,0));
131 Valeur=constante*std::fabs(Valeur);
132 }
133 return Valeur;
134}

182 d_mu = mu+mu_turb(elem1);
183 for (i=0; i<nb_faces_elem; i++)
184 {

190 resu(j,0)+=0.5*valA*inconnue(num_face,0)*porosite_face(j);
191 resu(j,0)-=0.5*valA*inconnue(j,0)*porosite_face(j);

193 }
194 elem2 = face_voisins(num_face,1);
195 d_mu = mu+mu_turb(elem2);
196 for (i=0; i<nb_faces_elem; i++)
197 {
198 if ( ( (j= elem_faces(elem2,i)) > num_face ) && ( j != fac_asso ) )
199 {
200 valA = viscA_Q1(zone_VEF,num_face,j,dimension,elem2,d_mu);
201 resu(num_face,0)+=valA*inconnue(j,0)*porosite_face(num_face);
202 resu(num_face,0)-=valA*inconnue(num_face,0)*porosite_face(num_face);
203 resu(j,0)+=0.5*valA*inconnue(num_face,0)*porosite_face(j);
204 resu(j,0)-=0.5*valA*inconnue(j,0)*porosite_face(j);
205 }
206 }
207 }
208 }
209 else
210 for (int ind_face=num1; ind_face<num2; ind_face++)
211 {
212 num_face = le_bord.num_face(ind_face);
213 elem1 = face_voisins(num_face,0);
214 d_mu = mu+mu_turb(elem1);
215 for (i=0; i<nb_faces_elem; i++)
216 {
217 if ( (j= elem_faces(elem1,i)) > num_face )
218 {
219 valA = viscA_Q1(zone_VEF,num_face,j,dimension,elem1,d_mu);
220 resu(num_face,0)+=valA*inconnue(j,0)*porosite_face(num_face);
221 resu(num_face,0)-=valA*inconnue(num_face,0)*porosite_face(num_face);
222 resu(j,0)+=valA*inconnue(num_face,0)*porosite_face(j);
223 resu(j,0)-=valA*inconnue(j,0)*porosite_face(j);
224 }
225 }
226 }
227 }
228 else //!< nb_comp > 1
229 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
230 {
231 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres nb_compo> 1 nbords, resu",resu);
232
233 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
234 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
235// const IntTab& elem_faces = zone_VEF.elem_faces();
236 int nb_faces_bord_tot = le_bord.nb_faces_tot();
237 int nb_faces_bord = le_bord.nb_faces();
238 int num1=0;
239 int num2=nb_faces_bord;
240
241 if (sub_type(Periodique,la_cl.valeur()))
242 {
243 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
244 int fac_asso;
245 for (int ind_face=num1; ind_face<num2; ind_face++)
246 {
247 num_face = le_bord.num_face(ind_face);
248 fac_asso = le_bord.num_face(la_cl_perio.face_associee(ind_face));
249 elem1 = face_voisins(num_face,0);
250 d_mu = mu+mu_turb(elem1);
251 for (i=0; i<nb_faces_elem; i++)
252 {
253 if ( ( (j= elem_faces(elem1,i)) > num_face ) && (j != fac_asso ) )
254 {
255 valA = viscA_Q1(zone_VEF,num_face,j,dimension,elem1,d_mu);
256 for (int nc=0; nc<nb_comp; nc++)
257 {
258 resu(num_face,nc)+=valA*inconnue(j,nc)*porosite_face(num_face);
259 resu(num_face,nc)-=valA*inconnue(num_face,nc)*porosite_face(num_face);
260 resu(j,nc)+=0.5*valA*inconnue(num_face,nc)*porosite_face(j);
261 resu(j,nc)-=0.5*valA*inconnue(j,nc)*porosite_face(j);
262 }
263 }
264 }
265 elem2 = face_voisins(num_face,1);
266 d_mu = mu+mu_turb(elem2);
267 for (i=0; i<nb_faces_elem; i++)
268 {
269 if ( ( (j= elem_faces(elem2,i)) > num_face ) && (j!= fac_asso ) )
270 {
271 valA = viscA_Q1(zone_VEF,num_face,j,dimension,elem2,d_mu);
272 for (int nc=0; nc<nb_comp; nc++)
273 {
274 resu(num_face,nc)+=valA*inconnue(j,nc)*porosite_face(num_face);
275 resu(num_face,nc)-=valA*inconnue(num_face,nc)*porosite_face(num_face);
276 resu(j,nc)+=0.5*valA*inconnue(num_face,nc)*porosite_face(j);
277 resu(j,nc)-=0.5*valA*inconnue(j,nc)*porosite_face(j);
278 }
279 }
280 }
281 }
282 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres periodique , resu",resu);
283
284 }
285 else if (sub_type(Dirichlet_paroi_fixe,la_cl.valeur()) ||
286 sub_type(Dirichlet_paroi_defilante,la_cl.valeur()) )
287 {
288 if (le_modele_turbulence.valeur().utiliser_loi_paroi())
289 {
290 if (dimension == 2 )
291 {
292 for (int ind_face=num1; ind_face<num2; ind_face++)
293 {
294 num_face = le_bord.num_face(ind_face);
295 int elem=face_voisins(num_face,0);
296// Calcul du gradient gradU=tau*Ny*N
297// Au bord, n toujours oriente vers l'exterieur
298 n[0]=face_normale(num_face,0);
299 n[1]=face_normale(num_face,1);
300 n/=norme_array(n);
301 double signe=0;
302 Tgrad(0,0)=tau_tan_(num_face,0)*n[1]*n[0];
303 Tgrad(0,1)=tau_tan_(num_face,0)*n[1]*n[1];
304 Tgrad(1,0)=-tau_tan_(num_face,0)*n[0]*n[0];
305 Tgrad(1,1)=-tau_tan_(num_face,0)*n[0]*n[1];
306// On determine le sens du frottement parietal :
307 signe=0;
308 for (i=0; i<nb_faces_elem; i++)
309 {
310 j=elem_faces(elem,i);
311// Calcul du signe de (Utan-Uparoi)*tan :
312 signe+=(inconnue(j,1)-inconnue(num_face,1))*face_normale(num_face,0)
313 -(inconnue(j,0)-inconnue(num_face,0))*face_normale(num_face,1);
314 }
315 if (signe<0) signe=-1.;
316 else signe=1.;
317// On calcule la contribution sur chaque face de l'element :
318 for (i=0; i<nb_faces_elem; i++)
319 {
320 j=elem_faces(elem,i);
321 double ori=1.;
322 if ( zone_VEF.face_voisins(j,0) != elem ) ori=-1;
323 if (j != num_face)
324 for (int nc=0; nc<nb_comp; nc++)
325 resu(j,nc)-=ori*signe*(Tgrad(nc,0)*face_normale(j,0)+Tgrad(nc,1)*face_normale(j,1))*porosite_face(j);
326 }
327 }
328 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres dirichlet dim2, resu",resu);
329 }
330 else if (dimension==3)
331 {
332 for (int ind_face=num1; ind_face<num2; ind_face++)
333 {
334 num_face = le_bord.num_face(ind_face);
335 int elem=face_voisins(num_face,0);
336// Calcul du gradient gradU
337// Au bord, n toujours oriente vers l'exterieur
338 n[0]=face_normale(num_face,0);
339 n[1]=face_normale(num_face,1);
340 n[2]=face_normale(num_face,2);
341 n/=norme_array(n);
342// Difference avec le 2D car ici tau_tan est signe !
343// A tester malgre tout !
344 Tgrad(0,0)=tau_tan_(num_face,0)*n[0];
345 Tgrad(0,1)=tau_tan_(num_face,0)*n[1];
346 Tgrad(0,2)=tau_tan_(num_face,0)*n[2];
347 Tgrad(1,0)=tau_tan_(num_face,1)*n[0];
348 Tgrad(1,1)=tau_tan_(num_face,1)*n[1];
349 Tgrad(1,2)=tau_tan_(num_face,1)*n[2];
350 Tgrad(2,0)=tau_tan_(num_face,2)*n[0];
351 Tgrad(2,1)=tau_tan_(num_face,2)*n[1];
352 Tgrad(2,2)=tau_tan_(num_face,2)*n[2];
353 d_mu = mu+mu_turb(elem);
354// On calcule la contribution sur chaque face de l'element :
355 for (i=0; i<nb_faces_elem; i++)
356 {
357 j=elem_faces(elem,i);
358 double ori=1.;
359 if ( zone_VEF.face_voisins(j,0) != elem ) ori=-1;
360 if (j != num_face)
361 for (int nc=0; nc<nb_comp; nc++)
362 resu(j,nc)-=ori*(Tgrad(nc,0)*face_normale(j,0)
363 +Tgrad(nc,1)*face_normale(j,1)
364 +Tgrad(nc,2)*face_normale(j,2))*porosite_face(j);
365 }
366 }
367 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres dirichlet dim3, resu",resu);
368 }
369 }
370 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres dirichlet (je crois), resu",resu);
371 }
372 else
373 {
374 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter avant else des bords, resu",resu);
375 num1 = 0;
376 num2 = nb_faces_bord_tot;
377 for (int ind_face=num1; ind_face<num2; ind_face++)
378 {
379 num_face = le_bord.num_face(ind_face);
380 int elem=face_voisins(num_face,0);
381 d_mu = mu+mu_turb(elem);
382// Boucle sur les faces :
383 for (int ii=0; ii<nb_faces_elem; ii++)
384 {
385 j= elem_faces(elem,ii);
386 if (((j > num_face) && (ind_face<nb_faces_bord)) || ((j!=num_face) && (ind_face>=nb_faces_bord)))
387 {
388 valA = viscA_Q1(zone_VEF,num_face,j,dimension,elem,d_mu);
389 for (int nc=0; nc<nb_comp; nc++)
390 {
391 resu(num_face,nc)+=valA*inconnue(j,nc)*porosite_face(num_face);
392 resu(num_face,nc)-=valA*inconnue(num_face,nc)*porosite_face(num_face);
393 resu(j,nc)+=valA*inconnue(num_face,nc)*porosite_face(j);
394 resu(j,nc)-=valA*inconnue(j,nc)*porosite_face(j);
395 }
396 }
397 }
398 }
399 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres else des bords, resu",resu);
400 }
401 }
402
403 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres bords, resu",resu);
404
405// On traite les faces internes
406
407 for (num_face=zone_VEF.premiere_face_int(); num_face<n1; num_face++)
408 {
409 for (int kk=0; kk<2; kk++)
410 {
411 elem0 = face_voisins(num_face,kk);
412 d_mu = mu+mu_turb(elem0);
413// On elimine les elements avec CL de paroi (rang>=1)
414 int rang = rang_elem_non_std(elem0);
415 if (rang<1)
416 {
417 for (i=0; i<nb_faces_elem; i++)
418 {
419 if ( (j= elem_faces(elem0,i)) > num_face )
420 {
421 valA = viscA_Q1(zone_VEF,num_face,j,dimension,elem0,d_mu);
422 for (int nc=0; nc<nb_comp; nc++)
423 {
424 resu(num_face,nc)+=valA*inconnue(j,nc)*porosite_face(num_face);
425 resu(num_face,nc)-=valA*inconnue(num_face,nc)*porosite_face(num_face);
426 resu(j,nc)+=valA*inconnue(num_face,nc)*porosite_face(j);
427 resu(j,nc)-=valA*inconnue(j,nc)*porosite_face(j);
428 }
429 }
430 }
431 }
432 }
433 }
434
435 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres interne, resu",resu);
436
437 int nb_comp_bis = resu.line_size();
438 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
439 {
440 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
441 if (sub_type(Neumann_paroi,la_cl.valeur()))
442 {
443 const Neumann_paroi& la_cl_paroi = ref_cast(Neumann_paroi, la_cl.valeur());
444 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
445 int ndeb = le_bord.num_premiere_face();
446 int nfin = ndeb + le_bord.nb_faces();
447 for (int face=ndeb; face<nfin; face++)
448 for (int comp=0; comp<nb_comp_bis; comp++)
449 resu(face,comp) += la_cl_paroi.flux_impose(face-ndeb,comp)*zone_VEF.surface(face)*porosite_face(face);
450 }
451 else if (sub_type(Echange_externe_impose,la_cl.valeur()))
452 {
453 const Echange_externe_impose& la_cl_paroi = ref_cast(Echange_externe_impose, la_cl.valeur());
454 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
455 int ndeb = le_bord.num_premiere_face();
456 int nfin = ndeb + le_bord.nb_faces();
457 for (int face=ndeb; face<nfin; face++)
458 for (int comp=0; comp<nb_comp_bis; comp++)
459 resu(face,comp) += la_cl_paroi.h_imp(face-ndeb,comp)*(la_cl_paroi.T_ext(face-ndeb)-inconnue(face,comp))*zone_VEF.surface(face)*porosite_face(face);
460 }
461 }
462
463 Debog::verifier("Op_Dift_VEF_Face_Q1::ajouter apres fin, resu",resu);
464 modifier_flux(*this);
465 return resu;
466}

648void Op_Dift_VEF_Face_Q1::contribue_au_second_membre(DoubleTab& resu ) const
649{
650 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
651 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
652 const IntTab& face_voisins = zone_VEF.face_voisins();
653 int nb_faces_elem = zone_VEF.zone().nb_faces_elem();
654 int elem,i,j;
655 const int nb_comp = resu.line_size();
656 const DoubleTab& face_normale = zone_VEF.face_normales();
657 DoubleVect n(dimension);
658 DoubleTrav Tgrad(dimension,dimension);
659
660// On traite les faces bord
661 if (nb_comp!=1)
662 {
663 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
664 {
665 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
666 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
667 const IntTab& elem_faces = zone_VEF.elem_faces();
668 int num1 = le_bord.num_premiere_face();
669 int num2 = num1 + le_bord.nb_faces();
670
671 if (sub_type(Dirichlet_paroi_fixe,la_cl.valeur()) ||
672 sub_type(Dirichlet_paroi_defilante,la_cl.valeur()) )
673 {
674 if (dimension == 2 )
675 {
676 for (int num_face=num1; num_face<num2; num_face++)
677 {
678 int elembis=face_voisins(num_face,0);
679// Calcul du gradient gradU=tau*Ny*N
680// Au bord, n toujours oriente vers l'exterieur
681 n[0]=face_normale(num_face,0);
682 n[1]=face_normale(num_face,1);
683 n/=norme_array(n);
684 double signe=0;
685 Tgrad(0,0)=tau_tan_(num_face,0)*n[1]*n[0];
686 Tgrad(0,1)=tau_tan_(num_face,0)*n[1]*n[1];
687 Tgrad(1,0)=-tau_tan_(num_face,0)*n[0]*n[0];
688 Tgrad(1,1)=-tau_tan_(num_face,0)*n[0]*n[1];
689// On determine le sens du frottement parietal :
690 signe=0;
691 for (i=0; i<nb_faces_elem; i++)
692 {
693 j=elem_faces(elembis,i);
694// Calcul du signe de (Utan-Uparoi)*tan :
695 signe+=(inconnue_->valeurs()(j,1)-inconnue_->valeurs()(num_face,1))*face_normale(num_face,0)
696 -(inconnue_->valeurs()(j,0)-inconnue_->valeurs()(num_face,0))*face_normale(num_face,1);
697 }
698 if (signe<0) signe=-1.;
699 else signe=1.;
700// On calcule la contribution sur chaque face de l'element :
701 for (i=0; i<nb_faces_elem; i++)
702 {
703 j=elem_faces(elembis,i);
704 double ori=1.;
705 if ( zone_VEF.face_voisins(j,0) != elembis ) ori=-1;
706 if (j != num_face)
707 for (int nc=0; nc<nb_comp; nc++)
708 resu(j,nc)-=ori*signe*(Tgrad(nc,0)*face_normale(j,0)+Tgrad(nc,1)*face_normale(j,1));
709 }
710 }
711 }
712 else if (dimension==3)
713 {
714 for (int num_face=num1; num_face<num2; num_face++)
715 {
716 elem=face_voisins(num_face,0);
717
718// Calcul du gradient gradU
719// Au bord, n toujours oriente vers l'exterieur
720
721 n[0]=face_normale(num_face,0);
722 n[1]=face_normale(num_face,1);
723 n[2]=face_normale(num_face,2);
724 n/=norme_array(n);
725
726// Difference avec le 2D car ici tau_tan est signe !
727// A tester malgre tout !
728
729 Tgrad(0,0)=tau_tan_(num_face,0)*n[0];
730 Tgrad(0,1)=tau_tan_(num_face,0)*n[1];
731 Tgrad(0,2)=tau_tan_(num_face,0)*n[2];
732 Tgrad(1,0)=tau_tan_(num_face,1)*n[0];
733 Tgrad(1,1)=tau_tan_(num_face,1)*n[1];
734 Tgrad(1,2)=tau_tan_(num_face,1)*n[2];
735 Tgrad(2,0)=tau_tan_(num_face,2)*n[0];
736 Tgrad(2,1)=tau_tan_(num_face,2)*n[1];
737 Tgrad(2,2)=tau_tan_(num_face,2)*n[2];
738
739// On calcule la contribution sur chaque face de l'element :
740 for (i=0; i<nb_faces_elem; i++)
741 {
742 j=elem_faces(elem,i);
743 double ori=1.;
744 if ( zone_VEF.face_voisins(j,0) != elem ) ori=-1;
745 if (j != num_face)
746 for (int nc=0; nc<nb_comp; nc++)
747 resu(j,nc)-=ori*(Tgrad(nc,0)*face_normale(j,0)
748 +Tgrad(nc,1)*face_normale(j,1)
749 +Tgrad(nc,2)*face_normale(j,2));
750 }
751 }
752 }
753 }
754 }
755 }

757 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
758 {
759 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);

761 {
762 const Neumann_paroi& la_cl_paroi = ref_cast(Neumann_paroi, la_cl.valeur());
763 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());

765 int nfin = ndeb + le_bord.nb_faces();

767 for (int comp=0; comp<nb_comp; comp++)
768 resu(face,comp) += la_cl_paroi.flux_impose(face-ndeb,comp)*zone_VEF.surface(face);
769 }
770 else if ( nb_comp == 1 && sub_type(Echange_externe_impose,la_cl.valeur()))
771 {
772 Cerr << "Non code pour Echange_externe_impose" << finl;
773 assert(0);
