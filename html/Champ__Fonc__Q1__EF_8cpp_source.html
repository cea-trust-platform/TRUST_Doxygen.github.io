98int Champ_Fonc_Q1_EF::imprime(Sortie& os, int ncomp) const
99{
100 const Zone_dis_base& zone_dis = zone_dis_base();
101 const Zone& zone = zone_dis.zone();
102 const DoubleTab& coord=zone.domaine().coord_sommets();
103 const int nb_som = zone.domaine().nb_som();
104 const DoubleTab& val = valeurs();
105 int som;
106 os << nb_som << finl;
107 for (som=0; som<nb_som; som++)
108 {
109 if (dimension==3)
110 os << coord(som,0) << " " << coord(som,1) << " " << coord(som,2) << " " ;
111 if (dimension==2)
112 os << coord(som,0) << " " << coord(som,1) << " " ;
113 if (nb_compo_ == 1)
114 os << val(som) << finl;
115 else
116 os << val(som,ncomp) << finl;
117 }
118 os << finl;
119 Cout << "Champ_Fonc_Q1_EF::imprime FIN >>>>>>>>>> " << finl;
120 return 1;
