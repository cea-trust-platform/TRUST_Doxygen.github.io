19//! Ecriture d'une liste sur un flot de sortie les elements separes par des virgules figurent entre des accolades
20template<typename _TYPE_>
21inline Sortie& TRUSTList<_TYPE_>::printOn(Sortie& os) const
22{
23 TRUSTList_Curseur<_TYPE_> curseur(*this);
24 Nom accouverte = "{", accfermee = "}", virgule = ",";
25 os << accouverte << " ";
26 while (curseur)
27 {
28// if(est_dernier()) GF sinon on a une virgule de trop
29 if (curseur.list().est_dernier()) os << curseur.valeur() << " ";
30 else os << curseur.valeur() << " " << virgule << " ";
31 ++curseur;
32 }
33 os << accfermee;
34 return os;
35}

37//! Lecture d'une liste sur un flot d'entree les elements separes par des virgules figurent entre des accolades
38template<typename _TYPE_>
39inline Entree& TRUSTList<_TYPE_>::readOn(Entree& is)
40{
41 Nom accouverte = "{", accfermee = "}", virgule = ",", nom;
42 _TYPE_ t;
43 is >> nom;
44 assert(nom == accouverte);
45 while (nom != accfermee)
46 {
47 is >> t;
48 add(t);
49 is >> nom;
50 assert((nom == accfermee) || (nom == virgule));
51 }
52 return is;

73//! insertion en queue
74template<typename _TYPE_>
75inline TRUSTList<_TYPE_>& TRUSTList<_TYPE_>::add(_TYPE_ value_to_add)
76{
77 if (value_to_add < min_data) min_data = value_to_add;
78 if (value_to_add > max_data) max_data = value_to_add;
79
80 if (TRUSTListElem<_TYPE_>::est_vide())
81 {
82 this->data = value_to_add;
83 this->suivant_ = 0;
84 return *this;
85 }
86 else
87 {
88 if (TRUSTListElem<_TYPE_>::est_dernier())
89 {
90 TRUSTListElem<_TYPE_> *next = new TRUSTListElem<_TYPE_>(value_to_add);
91 this->suivant_ = next;
92 dernier_ = next;
93 }
94 else
95 {
96 dernier().add(value_to_add);
97 dernier_ = &dernier_->suivant();
98 }
99 return *this;
100 }
101}

127template<typename _TYPE_>
128inline int TRUSTList<_TYPE_>::contient(_TYPE_ x) const
129{
130 if (TRUSTListElem<_TYPE_>::est_vide() || x > max_data || x < min_data) return 0;

188template<typename _TYPE_>
189int operator ==(const TRUSTList<_TYPE_>& list1, const TRUSTList<_TYPE_>& list2)
190{
191 int retour = 1;
192 if (list1.data != list2.data) retour = 0;
193 if ((!list1.est_dernier()) && (list2.est_dernier())) retour = 0;
194 if ((list1.est_dernier()) && (!list2.est_dernier())) retour = 0;
195 if ((!list1.est_dernier()) && (!list2.est_dernier())) retour = (*list1.suivant_ == *list2.suivant_);
196 return retour;

199//! Supprime un element contenu dans la liste
200template<typename _TYPE_>
201inline void TRUSTList<_TYPE_>::suppr(_TYPE_ obj)
202{
203 if (TRUSTListElem<_TYPE_>::valeur() == obj)
204 {
205 if (this->suivant_)
206 {
207 TRUSTListElem<_TYPE_> *next = this->suivant_;
208 this->suivant_ = next->suivant_;
209 this->data = next->valeur();
210 next->suivant_ = 0;
211 delete next;
212 }
213 else
214 {
215 this->suivant_ = this;
216 dernier_ = this;
217 }
218 calcule_min_max();
219 return;
220 }
221 TRUSTList_Curseur<_TYPE_> curseur_pre = *this;
222 TRUSTList_Curseur<_TYPE_> curseur = *(this->suivant_);
223
224 while (curseur)
225 {
226 if (curseur.valeur() == obj)
227 {
228 TRUSTListElem<_TYPE_> *next = &curseur_pre.list().suivant();
229 curseur_pre.list().suivant_ = curseur.list().suivant_;
230 if (next->suivant_ == 0) dernier_ = &curseur_pre.list();
231 else next->suivant_ = 0;
232
233 delete next;
234 calcule_min_max();
235 return;
236 }
237 ++curseur;
238 ++curseur_pre;
239 }
240 Cerr << "WARNING during deletion of an element in a list " << finl;
241 Cerr << "One has not found object == : " << obj << finl;
242}

244template<typename _TYPE_>
245inline void TRUSTList<_TYPE_>::calcule_min_max()
246{
247 min_data = std::is_same<_TYPE_,double>::value ? DMAXFLOAT : IMAX;
248 max_data = std::is_same<_TYPE_,double>::value ? -DMAXFLOAT : -IMAX;
249 TRUSTList_Curseur<_TYPE_> curseur = *this;
250 while (curseur)
251 {
252 _TYPE_ la_valeur = curseur.valeur();
253 if (la_valeur < min_data) min_data = la_valeur;
254 if (la_valeur > max_data) max_data = la_valeur;
255 ++curseur;
256 }
257}
