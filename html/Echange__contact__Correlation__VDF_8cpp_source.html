306/**
307 * Calcule le coeff d echange suivant la correlation entree dans le jdd
308 */
309double Echange_contact_Correlation_VDF::calculer_coefficient_echange(int i)
310{
311 double Re,Pr;
312 Re = std::fabs(getQh()*getDh()/getMu(i));
313 Pr = getMu(i)*getCp()/getLambda(i);
314 fct_Nu.setVar("Re",Re);
315 fct_Nu.setVar("Pr",Pr);
316 return fct_Nu.eval()*getLambda(i)/getDh();
317}

319/**
320 * Calcul du terme source de puissance volumique dans l'equation d'energie 1D fluide.
321 */
322void Echange_contact_Correlation_VDF::calculer_Q()
323{
324 const Zone_VDF& ma_zvdf = ref_cast(Zone_VDF,zone_Cl_dis().zone_dis().valeur());
325 const Front_VF& ma_front_vf = ref_cast(Front_VF,frontiere_dis());
326 const int ndeb = ma_front_vf.num_premiere_face();
327 const int nb_faces_bord = ma_front_vf.nb_faces();
328 const DoubleVect& surfaces = ma_zvdf.face_surfaces();
329 const IntTab& face_voisins = ma_zvdf.face_voisins();
330 Champ_Inc_base& inco = zone_Cl_dis().equation().inconnue();
331 DoubleTab& Ts = inco.valeurs();
332 DoubleTab& mon_h= h_imp_->valeurs();
333
334
335 Qvol=0.;
336 for(int iface=0; iface<nb_faces_bord; iface++)
337 {
338 int corresp = correspondance_solide_fluide(iface);
339 int elem = face_voisins(ndeb+iface,0);
340 if(elem == -1)
341 elem = face_voisins(ndeb+iface,1);
342 Qvol(corresp)+=mon_h(iface,0)*(Ts(elem)-T(corresp))*surfaces(ndeb+iface);
343 }
344 for (int i=0; i<N; i++)
345 Qvol(i)/=vol(i);
346}

350void Echange_contact_Correlation_VDF::trier_coord()
351{
352 const Zone_VDF& ma_zvdf = ref_cast(Zone_VDF,zone_Cl_dis().zone_dis().valeur());
353 const DoubleVect& surfaces = ma_zvdf.face_surfaces();
354 const Front_VF& ma_front_vf = ref_cast(Front_VF,frontiere_dis());
355
356 const int ndeb = ma_front_vf.num_premiere_face();
357 const int nb_faces_bord = ma_front_vf.nb_faces();
358
359 IntVect face_triee(nb_faces_bord);
360 const DoubleTab& xv = ma_zvdf.xv();
361 int i, j ,tmp;
362
363// On trie les faces par coordonnees croissante (selon la coordonnee correspondante a la direction du tube)
364 if (nb_faces_bord>1)
365 {
366 for(i=0; i<nb_faces_bord; i++)
367 face_triee(i) = ndeb+i;
368
369 for(i=1; i<nb_faces_bord; i++)
370 {
371 j=i-1;
372 tmp = face_triee(i);
373 while ( j>=0 && xv(face_triee(j),dir)>xv(tmp,dir))
374 {
375 face_triee(j+1) = face_triee(j);
376 j--;
377 }
378 face_triee(j+1) = tmp;
379 }
380 }
381// On determine le nb de mailles 1D dans le fluide fictif
382// et on remplit les correspondances elements solides -> elements fluides
383 correspondance_solide_fluide.resize(nb_faces_bord);
384 N=3; //!< on rajoute les noeuds des bords
385 correspondance_solide_fluide(face_triee(0)-ndeb) = 1;
386 for(i=1; i<nb_faces_bord; i++)
387 {
388 if (xv(face_triee(i),dir) != xv(face_triee(i-1),dir))
389 N++;
390 correspondance_solide_fluide(face_triee(i)-ndeb) = N-2;
391 }
392
393 U.resize(N);
394 T.resize(N);
395 Qvol.resize(N);
396 rho.resize(N);
397 mu.resize(N);
398 lambda.resize(N);
399 vol.resize(N);
400 coord.resize(N);
401 h_correlation.resize(N);
402
403
404
405// calcul des volumes des tranches
406 DoubleVect surf(N);
407 surf=0.;
408 vol=0.;
409 surf(0) = surf(N-1) = 0.; //!< ne sert pas
410 for(int iface=0; iface<nb_faces_bord; iface++)
411 {
412 int corresp = correspondance_solide_fluide(iface);
413 surf(corresp)+=surfaces(ndeb+iface);
414 coord(corresp) = xv(ndeb+iface,dir);
415 }
416 coord(0) = coord(1)-0.5*(coord(2)-coord(1)); //!< le premier point du maillage
417 coord(N-1) = coord(N-2)+0.5*(coord(N-2)-coord(N-3)); //!< le dernier point du maillage
418 for (i=1; i<N-1; i++)
419 {
420 vol(i)=volume(surf(i),diam);
421 }
422 vol(0) = vol(N-1) = 1.; //!< ne sert pas mais evite une division par zero dans le calcul de Qvol.
423}

497/**
498 * Calcule la temperature 1D dans le fluide en resolvant la conservation de l'energie
499 */
500void Echange_contact_Correlation_VDF::calculer_Tfluide()
501{
502 const Equation_base& mon_eqn = zone_Cl_dis().equation();
503 const double dt = mon_eqn.schema_temps().pas_de_temps();
504 DoubleVect ma(N); //!< diagonale
505 DoubleVect mb(N-1); //!< sous-diagonale
506 DoubleVect mc(N-1); //!< sur-diagonale
507 DoubleVect sm(N);
508 const int sgn = (debit>0) ? 1 : -1; //!< schema amont pour le transport
509
510
511 ma(0) = 1.;
512 ma(N-1) = 1.;
513 sm(0) = T_CL0;
514 sm(N-1) = T_CL1;
515 mc(0) = 0.;
516 mb(N-2) = 0.;
517 int i;
518 for (i=1; i<N-1; i++)
519 {
520 const double dtrhoCp = dt/rho(i)/Cp;
521 const double dz1 = coord(i+1)-coord(i);
522 const double dz2 = coord(i)-coord(i-1);
523 const double l1 = 0.5*(lambda(i+1)+lambda(i));
524 const double l2 = 0.5*(lambda(i)+lambda(i-1));
525
526 ma(i)=1+dtrhoCp*(sgn*Cp*debit/dz1+2*(l1/dz1+l2/dz2)/(dz1+dz2));
527
528 sm(i) = T(i)+dtrhoCp*Qvol(i);
529
530 mc(i) = dtrhoCp*(0.5*(1-sgn)*Cp*debit/dz1-2*l1/dz1/(dz1+dz2));
531
532 mb(i-1) = dtrhoCp*(-2*l2/dz2/(dz1+dz2));
533 mb(i-1) += dtrhoCp*(-0.5*(1+sgn)*Cp*debit/dz1);
534 }
535
536
537
538
539 DoubleVect xx(N);
540 xx=0.;
541 Solv_TDMA::resoudre(ma,mb,mc,sm,xx,N);
542 for (i=0; i<N; i++)
543 T(i) = xx(i);
544}
