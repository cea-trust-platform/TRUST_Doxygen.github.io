41//! Cherche le domaine de nom "le_nom_domaine" parmi les objets connus<br>de l'interprete et renvoie une ref a ce domaine.
42const Domaine& Decouper::find_domain(const Nom& le_nom_domaine)
43{
44 const Domaine * domaine_ptr = 0;
45
46 if (objet_existant(le_nom_domaine))
47 {
48 const Objet_U& objet_domaine = objet(le_nom_domaine);
49 if (sub_type(Domaine, objet_domaine))
50 domaine_ptr = & ref_cast(Domaine, objet_domaine);
51 }
52
53 if (domaine_ptr == 0)
54 {
55 Cerr << "Error in Decouper::interpreter\n "
56 << " Syntax:\n Decouper DOMAINNAME { ... }\n ";
57 Cerr << le_nom_domaine << " is not a domain known" << finl;
58 Process::exit();
59 }
60 return *domaine_ptr;
61}

101static void postraiter_decoupage(const Nom& nom_fichier_lata,
102 const Domaine& domaine,
103 const IntVect& elem_part)
104{
105 Cerr << "Postprocessing of the splitting at the lata format: " << nom_fichier_lata << finl;
106 Postraitement_lata::Format format = Postraitement_lata::BINAIRE;
107 Postraitement_lata::ecrire_entete(nom_fichier_lata,
108 "Discretisation_inconnue",
109 format);
110 Postraitement_lata::ecrire_zone(nom_fichier_lata,
111 domaine.zone(0),
112 format);
113
114 Postraitement_lata::ecrire_temps(nom_fichier_lata,
115 0.,
116 format);
117 const int n = elem_part.size_reelle();
118 DoubleTab data(n);
119 for (int i = 0; i < n; i++)
120 data(i) = elem_part[i];
121 Postraitement_lata::ecrire_champ(nom_fichier_lata,
122 Nom("partition"),
123 Postraitement_lata::CHAMP,
124 'I',
125 domaine.le_nom(),
126 "pb",
127 0., /* temps */
128 Postraitement_base::ELEMENTS,
129 data,
130 format);
131
132 Postraitement_lata::finir_lata(nom_fichier_lata);
133}

161// XD attr larg_joint entier larg_joint 1 This keyword specifies the thickness of the virtual ghost zone (data known by one processor though not owned by it). The default value is 1 and is generally correct for all algorithms except the QUICK convection scheme that require a thickness of 2. Any non-zero positive value can be used, but the amount of data to store and exchange between processors grows quickly with the thickness.

This file contains the zone number for each element\'s mesh. Then you can easily permute zone numbers in this file. Then read the new partition to create the .Zones files with the Fichier_Decoupage keyword.

This option might be used to perform coupled parallel computations. Supplemental empty zones from M to N-1 are created. This keyword is used when you want to run a parallel calculation on several domains with for example, 2 processors on a first domain and 10 on the second domain because the first domain is very small compare to second one. You will write Nb_parts 2 and Nb_parts_tot 10 for the first domain and Nb_parts 10 for the second domain.

171Entree& Decouper::interpreter(Entree& is)
172{
173 Cerr << "Decouper : Splitting of a domain" << finl;
174// lecture des parametres
175 lire(is);
176
177// generation de la partition
178 IntVect elem_part;
179 Partitionneur_base& partitionneur = deriv_partitionneur.valeur();
180 partitionneur.declarer_bords_periodiques(liste_bords_periodiques);
181 partitionneur.construire_partition(elem_part,nb_parts_tot);
182
183// ecriture
184 ecrire(elem_part, {});
185
186 Cerr << "End of the interpreter Decouper" << finl;
187 return is;
188}

191Entree& Decouper::lire(Entree& is)
192{
193// Lecture du nom du domaine a decouper:
194 is >> nom_domaine;
195 Cerr << " Domain name to split : " << nom_domaine << finl;
196 ref_domaine = find_domain(nom_domaine);
197// Avant de decouper on imprime des infos
198 ref_domaine->imprimer();
199
200 Param param(que_suis_je());
201 param.ajouter_non_std("partitionneur|partition_tool",(this),Param::REQUIRED);
202 param.ajouter("larg_joint",&epaisseur_joint);
203 param.ajouter_condition("value_of_larg_joint_ge_1","The joint thickness must greater or equal to 1.");
204 param.ajouter("nom_zones|zones_name",&nom_zones_decoup);
205 param.ajouter("ecrire_decoupage",&nom_fichier_decoupage);
206 param.ajouter("ecrire_lata",&nom_fichier_lata);
207 param.ajouter("nb_parts_tot",&nb_parts_tot);
208 param.ajouter("reorder",&reorder);
209 param.ajouter_flag("single_hdf",&format_hdf);
210 param.ajouter("periodique",&liste_bords_periodiques);
211 param.ajouter("print_more_infos",&Decouper::print_more_infos);
212 param.lire_avec_accolades_depuis(is);
213 return is;
214}

217void Decouper::ecrire(IntVect& elem_part, const Static_Int_Lists* som_raccord)
218{
219// Calcul du nombre de parties generees par le partitionneur
220 int nb_parties = 0;
221 if (elem_part.size_array() > 0)
222 nb_parties = max_array(elem_part) + 1;
223 nb_parties = mp_max(nb_parties);
224 Cerr << "The partitioner has generated " << nb_parties << " parts." << finl;
225
226// Prise en compte de la directive nb_parts_tot
227 if (nb_parts_tot >= 0)
228 {
229 if (nb_parties > nb_parts_tot)
230 {
231 Cerr << "Error: nb_parts_tot is less than the number of parts generated by the partitioner."
232 << finl;
233 exit();
234 }
235 Cerr << "Number of parts requested : " << nb_parts_tot
236 << "\nGeneration of " << nb_parts_tot - nb_parties << " empty parts." << finl;
237 nb_parties = nb_parts_tot;
238 }
239// Force un seul fichier .Zones au dela d'un certain nombre de rangs MPI:
240 if (Process::force_single_file(nb_parties, nom_zones_decoup+".Zones"))
241 format_hdf = 1;
242
243 if (nom_fichier_decoupage != "?")
244 ecrire_fichier_decoupage(nom_fichier_decoupage, elem_part, nb_parts_tot);
245
246 if (nom_zones_decoup != "?")
247 {
248 Decouper::ZonesFileOutputType typ;
249 if (format_binaire) typ = BINARY_MULTIPLE;
250 if (format_hdf) typ = HDF5_SINGLE;
251 ecrire_sous_zones(nom_zones_decoup, typ,
252 ref_domaine.valeur(), elem_part, nb_parties, epaisseur_joint, reorder,
253 liste_bords_periodiques, som_raccord);
254 }
255
256 if (nom_fichier_lata != "?")
257 postraiter_decoupage(nom_fichier_lata, ref_domaine.valeur(), elem_part);
258
259 Cout << "\nQuality of partitioning --------------------------------------------" << finl;
260 int total_elem = Process::mp_sum(elem_part.size_reelle());
261 Cout << "\nTotal number of elements = " << total_elem << finl;
262 Cout << "Number of Zones : " << nb_parties << finl;
263
264 if (Decouper::print_more_infos)
265 {
266 DoubleVect A(nb_parties);
267 A = 0;
268 for (int i = 0; i < elem_part.size_reelle(); i++)
269 A(elem_part[i]) = A(elem_part[i]) + 1;
270
271 if(Process::je_suis_maitre())
272 {
273 for(int proc=1; proc<Process::nproc(); proc++)
274 {
275 DoubleVect tmp(nb_parties);
276 tmp = 0;
277 recevoir(tmp, proc, 0, proc+2005);
278
279 for(int i_part=0; i_part<nb_parties; i_part++)
280 A(i_part) += tmp(i_part);
281 }
282
283
284 double mean_element_zone = total_elem/nb_parties;
285 if (mean_element_zone>0)
286 {
287 double load_imbalance = double(local_max_vect(A) / mean_element_zone);
288 Cout << "Number of cells per Zone (min/mean/max) : " << local_min_vect(A) << " / " << mean_element_zone << " / "
289 << local_max_vect(A) << " Load imbalance: " << load_imbalance << "\n" << finl;
290
291 }
292 }
293 else
294 envoyer(A, Process::me(), 0, Process::me()+2005);
295
296
297// we could do it as below, but if nb_parties is big, it would involve a lot of collective communication
298// for (int i = 0; i < nb_parties; i++)
299// A[i] = Process::mp_sum(A[i]);
300
301
302 }
303
304 Cerr << "End of the interpreter Decouper" << finl;
305 if (reorder==0 && nb_parties>128)
306 {
307 Cerr << "Performance tip: You could add \"reorder 1\" option to have less distance between communicating processes on the network." << finl;
308 Cerr << "Add also \"Ecrire_lata filename\" to post-process the partition numeration and see the difference." << finl;
309 }
310}
