26Entree& Multiplicateur_diphasique_homogene::readOn(Entree& is)
27{
28 Param param(que_suis_je());
29 param.ajouter("alpha_min", &alpha_min_);
30 param.ajouter("alpha_max", &alpha_max_);
31 param.lire_avec_accolades_depuis(is);
32
33 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, pb_.valeur()) ? &ref_cast(Pb_Multiphase, pb_.valeur()) : NULL;
34
35 if (!pbm || pbm->nb_phases() == 1) n_l = 0, n_g = -1; //pas un Pb_Multiphase -> monophasique liquide
36 else for (int n = 0; n < pbm->nb_phases(); n++) //recherche de n_l, n_g : phase {liquide,gaz}_continu en priorite
37 if (pbm->nom_phase(n).debute_par("liquide") && (n_l < 0 || pbm->nom_phase(n).finit_par("continu"))) n_l = n;
38 else if (pbm->nom_phase(n).debute_par("gaz") && (n_g < 0 || pbm->nom_phase(n).finit_par("continu"))) n_g = n;
39
40 if (n_l < 0) Process::exit(que_suis_je() + " : main phase not found!");
41
42 return is;
43}

45void Multiplicateur_diphasique_homogene::coefficient(const double *alpha, const double *rho, const double *v, const double *f,
46 const double *mu, const double Dh, const double gamma, const double *Fk,
47 const double Fm, DoubleTab& coeff) const
48{
49 double G = alpha[n_l] * rho[n_l] * std::fabs(v[n_l]) + alpha[n_g] * rho[n_g] * std::fabs(v[n_g]), //debit total
50 x = G ? alpha[n_g] * rho[n_g] * v[n_g] / G : 0, //titre
51 Phi2 = 1 + x * (rho[n_l] / rho[n_g] - 1), //le multiplicateur!
52 frac_g = std::min(std::max((alpha[n_g] - alpha_min_) / (alpha_max_ - alpha_min_), 0.), 1.), frac_l = 1 - frac_g; //fraction appliquee a la vapeur
53 coeff = 0;
54 coeff(n_l, 0) = frac_l * Phi2;
55 coeff(n_g, 0) = frac_g ? frac_g / (alpha[n_g] * alpha[n_g]) : 0; //basculement vers le frottement "vapeur seule"
56}
