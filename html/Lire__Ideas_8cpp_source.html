35Entree& Lire_Ideas::interpreter_(Entree& is)
36{
37 Nom nom_fic;
38 associer_domaine(is);
39 is >> nom_fic;
40 Domaine& dom=domaine();
41//
42 int NDIM=dimension;
43 int NNOEUDS=0;
44 int NELEM=0;
45 int NTETRA=0;
46 int NFACB=0;
47 int NSEG=0;
48 int NGROUPE=0;
49 int CAS=0;
50//
51 Cerr << "First pass to determine the sizes of arrays" << finl;
52//
53 char nom_fic2[81];
54 for (int ii=0; ii<80; ii++)
55 nom_fic2[ii]='\0';
56 strcpy(nom_fic2,nom_fic);
57 F77NAME(LIREPREM) (nom_fic2,
58 &NDIM,
59 &NNOEUDS,
60 &NELEM,
61 &NTETRA,
62 &NFACB,
63 &NSEG,
64 &NGROUPE,
65 &CAS);
66//
67 Cerr << "Number of nodes : " << NNOEUDS << finl;
68 Cerr << "Number of elements : " << NELEM << finl;
69 Cerr << "Number of tetrahedra : " << NTETRA << finl;
70 Cerr << "Number of boundary faces : " << NFACB << finl;
71 Cerr << "Number of boundary segments : " << NSEG << finl;
72 Cerr << "Number of groups : " << NGROUPE << finl;
73//
74 ArrOfInt NGELEM(NGROUPE);
75 ArrOfInt WORK(std::max(NTETRA,NFACB*4));
76//
77// Passage F77 a C++: les indices entre les deux langages sont inverses
78// (methode de stockage differente)
79//
80 IntTab TETRA(4,NTETRA);
81 IntTab FACB(3,NFACB);
82 IntTab GROUP(NFACB,NGROUPE);
83//
84 ArrOfDouble X(NNOEUDS);
85 ArrOfDouble Y(NNOEUDS);
86 ArrOfDouble Z(NNOEUDS);
87 Cerr << "call of LIREIDEAS " << finl;
88//
89 F77NAME(LIREIDEAS)(nom_fic2,
90 &NDIM,
91 X.addr(),
92 Y.addr(),
93 Z.addr(),
94 TETRA.addr(),
95 FACB.addr(),
96 GROUP.addr(),
97 &NNOEUDS,
98 &NELEM,
99 &NTETRA,
100 &NFACB,
101 &NGROUPE,
102 NGELEM.addr(),
103 WORK.addr(),
104 &CAS);
105//
106 Cerr << " Reading completed " << finl;
107//
108 DoubleTab& coord=dom.les_sommets();
109 Zone une_zone;
110 Zone& la_zone=dom.add(une_zone);
111//
112 la_zone.nommer(dom.le_nom());
113 la_zone.associer_domaine(dom);
114 la_zone.type_elem().typer("Tetraedre");
115 la_zone.type_elem().associer_zone(la_zone);
116
117//
118// On commence par transferer les coordonnees
119//
120 coord.resize(NNOEUDS, dimension);
121 {
122 for(int i=0; i<NNOEUDS; i++)
123 {
124 coord(i,0) = (X[i]);
125 coord(i,1) = (Y[i]);
126 coord(i,2) = (Z[i]);
127 }
128 }
129//
130 Cerr << " Completing the assignment of coordinates of nodes" << finl;
131//
132// On continue par les elements (connectivite element -> noeuds)
133//
134 IntTab& les_elems=la_zone.les_elems();
135 les_elems.resize(NTETRA, dimension+1);
136 for(int i=0; i<NTETRA; i++)
137 {
138 for(int j=0; j<dimension+1; j++)
139 {
140//
141// ATTENTION : Inversion des indices entre le tableau Fortran et C++
142//
143 les_elems(i,j)= TETRA(j,i)-1 ;
144 }
145 }
146//
147 Cerr << " Completing the assignment of elements" << finl;
148//
149//
150// Maintenant, on traite les faces de bords (tableau FACB, de dim (NFACB,3))
151// qui se trouvent dans les differents groupes de faces de bords. Chaque
152// groupe correspond a un bord bien defini du domaine (paroi, entree, sortie,
153// symetrie, ...) et chaque groupe porte le nom de ce type de bord (GRNAME)
154//
155// En resume :
156// GRNAME (NGROUPE) : Nom des differents bords
157// GROUP (NNOEUDS,NGROUPE) : Groupe de faces de bord (Numero de la face)
158// NGELEM (NGROUPE) : Nombre de faces de bords pour le groupe NGROUPE
159// FACB (NFACB,3) : Connectivite des faces de bords (les 3 noeuds du triangle)
160//
161//
162//
163// On ouvre le fichier contenant les noms des groupes
164//
165 Nom nom1="nom.groupe";
166 EFichier fic(nom1);
167 fic.set_check_types(1); //!< Remplace UFichier
168 Cerr << "Reading of the file : " << nom1 << finl;
169// On definit les bords comme etant des faces de bords de la zone
170// On fait Zone --> les_bords
171//
172 Bords& les_bords=la_zone.faces_bord();
173//
174// Ici, on associe les bords a la zone (on se refere a la_zone)
175// On fait les_bords --> Zone
176//
177 les_bords.associer_zone(la_zone);
178//
179 Nom nom_bord="Bord";
180//
181// On affecte les bords :
182//
183 for(int k=0; k<NGROUPE; k++)
184 {
185 fic >> nom_bord;
186 Bord& nouveau=les_bords.add(Bord());
187//
188 nouveau.nommer(nom_bord);
189 nouveau.faces().typer(Faces::triangle_3D);
190//
191 nouveau.faces().dimensionner(NGELEM[k]);
192//
193 int num;
194 for (int j=0; j<NGELEM[k]; j++)
195 {
196//
197// ATTENTION : Inversion des indices entre le tableau Fortran et C++
198//
199// GF renumerotation dans lire_ideas.f maintenant...
200//
201// num=GROUP(j,k)-1-NTETRA;
202 num=GROUP(j,k)-1;
203 nouveau.faces().sommet(j,0)= FACB(0,num)-1;
204 nouveau.faces().sommet(j,1)= FACB(1,num)-1;
205 nouveau.faces().sommet(j,2)= FACB(2,num)-1;
206 }
207 nouveau.associer_zone(la_zone);
208 }
209//
210 Cerr << " Completing the assignment of the boundary faces" << finl;
211//
212 Cerr << " Exit of Lire_Ideas " << finl;
213//
214 la_zone.fixer_premieres_faces_frontiere();
215
216 Scatter::init_sequential_domain(dom);
217 return is;
218}
