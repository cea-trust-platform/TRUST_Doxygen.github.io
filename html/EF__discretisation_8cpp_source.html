65//! Discretisation d'un champ pour le EF en fonction d'une directive<br>de discretisation. La directive est un Motcle comme "vitesse", "pression",<br>"temperature", "champ_elem" (cree un champ de type P0), ...<br>Cette methode determine le type du champ a creer en fonction du type d'element<br>et de la directive de discretisation. Elle determine ensuite le nombre de ddl<br>et fixe l'ensemble des parametres du champ (type, nb_compo, nb_ddl, nb_pas_dt,<br>nom(s), unite(s) et nature du champ) et associe la Zone_dis au champ.<br>Voir le code pour avoir la correspondance entre les directives et<br>le type de champ cree.
66void EF_discretisation::discretiser_champ(
67 const Motcle& directive, const Zone_dis_base& z,
68 Nature_du_champ nature,
69 const Noms& noms, const Noms& unites,
70 int nb_comp, int nb_pas_dt, double temps,
71 Champ_Inc& champ, const Nom& sous_type ) const
72{
73 const Zone_EF& zone_EF = ref_cast(Zone_EF, z);
74
75 Motcles motcles(7);
76 motcles[0] = "vitesse"; //!< Choix standard pour la vitesse
77 motcles[1] = "pression"; //!< Choix standard pour la pression
78 motcles[2] = "temperature"; //!< Choix standard pour la temperature
79 motcles[3] = "divergence_vitesse"; //!< Le type de champ obtenu en calculant div v
80 motcles[4] = "gradient_pression"; //!< Le type de champ obtenu en calculant grad P
81 motcles[5] = "champ_elem"; //!< Creer un champ aux elements (de type P0)
82 motcles[6] = "champ_sommets"; //!< Creer un champ aux sommets (type P1)
83
84// Le type de champ de vitesse depend du type d'element :
85 Nom type_champ_vitesse;
86 if (sub_type(Tri_EF, zone_EF.type_elem().valeur())
87 || sub_type(Segment_EF,zone_EF.type_elem().valeur())
88 || sub_type(Tetra_EF,zone_EF.type_elem().valeur()))
89 type_champ_vitesse = "Champ_P1_EF";
90 else if (sub_type(Quadri_EF,zone_EF.type_elem().valeur())
91 || sub_type(Hexa_EF,zone_EF.type_elem().valeur()))
92 type_champ_vitesse = "Champ_Q1_EF";
93 else
94 {
95 Cerr << "EF_discretisation::discretiser_champ :\n L'element geometrique ";
96 Cerr << zone_EF.type_elem().valeur().que_suis_je();
97 Cerr << " n'est pas supporte." << finl;
98 exit();
99 }
100
101 Nom type;
102 int default_nb_comp = 0; //!< Valeur par defaut du nombre de composantes
103 int rang = motcles.search(directive);
104 switch(rang)
105 {
106 case 0:
107 type = type_champ_vitesse;
108 default_nb_comp = dimension;
109 break;
110 case 1:
111 type = "Champ_P0_EF";
112 default_nb_comp = 1;
113 break;
114 case 2:
115 type = type_champ_vitesse;
116 default_nb_comp = 1;
117 break;
118 case 3:
119 type = "Champ_P0_EF";
120 default_nb_comp = 1;
121 break;
122 case 4:
123 type = type_champ_vitesse;
124 default_nb_comp = dimension;
125 break;
126 case 5:
127 type = "Champ_P0_EF";
128 default_nb_comp = 1;
129 break;
130 case 6:
131 type = "Champ_P1_EF";
132 default_nb_comp = 1;
133 break;
134 default:
135 assert(rang < 0);
136 break;
137 }
138
139 if (directive == demande_description)
140 Cerr << "EF_discretisation : " << motcles;
141
142// Si on n'a pas compris la directive (ou si c'est une demande_description)
143// alors on appelle l'ancetre :
144 if (rang < 0)
145 {
146 Discret_Thyd::discretiser_champ(directive, z,
147 nature, noms, unites,
148 nb_comp, nb_pas_dt, temps, champ);
149 return;
150 }
151
152// Calcul du nombre de ddl
153 int nb_ddl = 0;
154 if (type == "Champ_P0_EF")
155 nb_ddl = z.nb_elem();
156 else if (type == type_champ_vitesse)
157 nb_ddl = zone_EF.nb_som();
158 else if (type == "Champ_P1_EF")
159 nb_ddl = zone_EF.nb_som();
160 else
161 assert(0);
162
163 /* if (nature == multi_scalaire) {
164 // Pas encore code
165 Cerr << "Champ multi_scalaire pas code" << finl;
166 assert(0); exit();
167// Si c'est un champ multiscalaire, uh !
168 } else {*/
169 if (nb_comp < 0)
170 nb_comp = default_nb_comp;
171 assert(nb_comp > 0);
172 creer_champ(champ, z, type, noms[0], unites[0], nb_comp, nb_ddl, nb_pas_dt, temps,
173 directive, que_suis_je());
174 if (nature == multi_scalaire)
175 {
176 champ.valeur().fixer_nature_du_champ(nature);
177 champ.valeur().fixer_unites(unites);
178 champ.valeur().fixer_noms_compo(noms);
179 }
180
181}

214//! Idem que EF_discretisation::discretiser_champ(... , Champ_Inc)<br>Traitement commun aux champ_fonc et champ_don.<br>Cette methode est privee (passage d'un Objet_U pas propre vu<br>de l'exterieur ...)
215void EF_discretisation::discretiser_champ_fonc_don(
216 const Motcle& directive, const Zone_dis_base& z,
217 Nature_du_champ nature,
218 const Noms& noms, const Noms& unites,
219 int nb_comp, double temps,
220 Objet_U& champ) const
221{
222// Deux pointeurs pour acceder facilement au champ_don ou au champ_fonc,
223// suivant le type de l'objet champ.
224 Champ_Fonc * champ_fonc = 0;
225 Champ_Don * champ_don = 0;
226 if (sub_type(Champ_Fonc, champ))
227 champ_fonc = & ref_cast(Champ_Fonc, champ);
228 else
229 champ_don = & ref_cast(Champ_Don, champ);
230
231 const Zone_EF& zone_EF = ref_cast(Zone_EF, z);
232
233 Motcles motcles(7);
234 motcles[0] = "pression"; //!< Choix standard pour la pression
235 motcles[1] = "temperature"; //!< Choix standard pour la temperature
236 motcles[2] = "divergence_vitesse"; //!< Le type de champ obtenu en calculant div v
237 motcles[3] = "champ_elem"; //!< Creer un champ aux elements (de type P0)
238 motcles[6] = "champ_sommets"; //!< Creer un champ aux elements (de type P1)
239 motcles[4] = "vitesse"; //!< Choix standard pour la vitesse
240 motcles[5] = "gradient_pression"; //!< Le type de champ obtenu en calculant grad P
241
242// Le type de champ de vitesse depend du type d'element :
243 Nom type_champ_vitesse;
244 {
245 const Elem_EF_base& elem_EF = zone_EF.type_elem().valeur();
246 if (sub_type(Tri_EF, elem_EF) || sub_type(Segment_EF,elem_EF) ||sub_type(Tetra_EF, elem_EF) || sub_type(Point_EF,elem_EF) )
247 type_champ_vitesse = "Champ_Fonc_P1_EF";
248 else if (sub_type(Quadri_EF, elem_EF) || sub_type(Hexa_EF, elem_EF))
249 type_champ_vitesse = "Champ_Fonc_Q1_EF";
250 else
251 {
252 Cerr << "EF_discretisation::discretiser_champ :\n L'element geometrique ";
253 Cerr << elem_EF.que_suis_je();
254 Cerr << " n'est pas supporte." << finl;
255 exit();
256 }
257 }
258 Nom type;
259 int default_nb_comp = 0; //!< Valeur par defaut du nombre de composantes
260 int rang = motcles.search(directive);
261 switch(rang)
262 {
263 case 0:
264 type = "Champ_Fonc_P0_EF";
265 default_nb_comp = 1;
266 break;
267 case 1:
268 type = type_champ_vitesse;
269 default_nb_comp = 1;
270 break;
271 case 2:
272 type = "Champ_Fonc_P0_EF";
273 default_nb_comp = 1;
274 break;
275 case 3:
276 type = "Champ_Fonc_P0_EF";
277 default_nb_comp = 1;
278 break;
279 case 4:
280 type = type_champ_vitesse;
281 default_nb_comp = dimension;
282 break;
283 case 5:
284 type = type_champ_vitesse;
285 default_nb_comp = dimension;
286 break;
287 case 6:
288 type = type_champ_vitesse;
289 default_nb_comp = 1;
290 break;
291 default:
292 assert(rang < 0);
293 break;
294 }
295
296 if (directive == demande_description)
297 Cerr << "EF_discretisation : " << motcles;
298
299// Si on n'a pas compris la directive (ou si c'est une demande_description)
300// alors on appelle l'ancetre :
301 if (rang < 0)
302 {
303 if (champ_fonc)
304 Discret_Thyd::discretiser_champ(directive, z, nature, noms, unites,
305 nb_comp, temps, *champ_fonc);
306 else
307 Discret_Thyd::discretiser_champ(directive, z, nature, noms, unites,
308 nb_comp, temps, *champ_don);
309 return;
310 }
311
312// Calcul du nombre de ddl
313 int nb_ddl = 0;
314 if (type == "Champ_Fonc_P0_EF") nb_ddl = z.nb_elem();
315 else if (type == type_champ_vitesse) nb_ddl = zone_EF.nb_som();
316 else assert(0);
317
318 /* // Si c'est un champ multiscalaire, uh !
319 if (nature == multi_scalaire)
320 {
321 // Pas encore code
322 Cerr << "Champ multi_scalaire pas code" << finl;
323 assert(0);
324 exit();
325 }
326 else */
327 {
328 if (nb_comp < 0)
329 nb_comp = default_nb_comp;
330 assert(nb_comp > 0);
331 if (champ_fonc)
332 creer_champ(*champ_fonc, z, type, noms[0], unites[0], nb_comp, nb_ddl, temps,
333 directive, que_suis_je());
334 else
335 creer_champ(*champ_don, z, type, noms[0], unites[0], nb_comp, nb_ddl, temps,
336 directive, que_suis_je());
337 }
338
339 if ((nature == multi_scalaire) && (champ_fonc))
340 {
341 champ_fonc->valeur().fixer_nature_du_champ(nature);
342 champ_fonc->valeur().fixer_unites(unites);
343 champ_fonc->valeur().fixer_noms_compo(noms);
344 }
345 else if ((nature == multi_scalaire) && (champ_don))
346 {
347 Cerr<<"There is no field of type Champ_Don with a multi_scalaire nature."<<finl;
348 exit();
349 }
350
351}

356 Zone_dis& z, Champ_Fonc& ch) const

542 Cerr<<"fait changer_temps"<<finl;
543 ch_mu.changer_temps(vit.temps());
544 Cerr<<"mu EF est discretise "<<finl;

563void EF_discretisation::critere_Q(const Zone_dis& z,const Zone_Cl_dis& zcl,const Champ_Inc& ch_vitesse, Champ_Fonc& ch) const
564{
565#ifdef dependance
566// On passe la zcl, pour qu'il n y ait qu une methode qqsoit la dsicretisation
