35DoubleVect& Champ_val_tot_sur_vol_VDF::eval_contrib_loc(const Zone_dis_base& zdis,const Zone_Cl_dis_base& zcldis,DoubleVect& vol_glob_pond)
36{
37 const Zone_VF& zvf = ref_cast(Zone_VF,zdis);
38 const int nb_elem = zvf.nb_elem();
39 int size_vol = les_sous_zones.size()+1;
40 vol_glob_pond.resize(size_vol);
41
42 const DoubleVect& vol = zvf.volumes();
43 const DoubleVect& por_elem = zvf.porosite_elem();
44
45 int cpt=1;
46 LIST_CURSEUR(REF(Sous_Zone)) curseur1 = les_sous_zones;
47 while(curseur1)
48 {
49 const Sous_Zone& sz = curseur1.valeur().valeur();
50 int size_sz = sz.nb_elem_tot();
51 int el;
52
53 for (int elem=0; elem<size_sz; elem++)
54 {
55 el = sz(elem);
56// On ne retient que les elements reels
57 if (el<nb_elem)
58 vol_glob_pond(cpt) += vol(el)*por_elem(el);
59 }
60 cpt++;
61 ++curseur1;
62 }
63
64 for (int elem=0; elem<nb_elem; elem++)
65 vol_glob_pond(0) += vol(elem)*por_elem(elem);
66
67 vol_glob_pond(0) = mp_sum(vol_glob_pond(0));
68
69 for (int i=1; i<size_vol; i++)
70 {
71 vol_glob_pond(i) = mp_sum(vol_glob_pond(i));
72 vol_glob_pond(0) -= vol_glob_pond(i);
73 }
74
75 return vol_glob_pond;
76}
