247//! Si force=1, effectue le postraitement sans tenir compte des<br>frequences de postraitement.<br>Le postraitement est mis a jour et les traitements eventuels sur<br>les sondes, champs et statistiques sont effectues.<br>Si force=0, tient compte des frequences d'impression demandees.
248int Probleme_base::postraiter(int force)
249{
250 statistiques().begin_count(postraitement_counter_);
251 Schema_Temps_base& sch = schema_temps();
252 Debog::set_nom_pb_actuel(le_nom());
253 if (sch.nb_pas_dt()!=0)
254 imprimer(Cout);
255 if (force)
256 {
257// Les sources postraitables (Terme_Source_Acceleration) ne sont pas mis a jour
258// pour le temps final et ne font pas partie des champs_crees_ du postraitement
259// qui eux sont mis a jour par les_postraitements.mettre_a_jour.
260// On les met donc a jour ici
261
262 const int nb_pas_dt_max = sch.nb_pas_dt_max();
263 int& indice_nb_pas_dt = sch.set_indice_nb_pas_dt_max_atteint();
264 int& indice_tps_final = sch.set_indice_tps_final_atteint();
265 const double t_init = sch.temps_init();
266 const double t_max = sch.temps_max();
267
268// Test pour eviter de repeter le postraitement a l instant initial
269// Cela evite un plantage dans le cas d un postraitement au format meshtv
270
271 if (!((indice_nb_pas_dt) && (nb_pas_dt_max==0))
272 && (!((indice_tps_final) && (est_egal(t_init,t_max)))))
273 {
274 for (int i=0; i<nombre_d_equations(); i++)
275 equation(i).sources().mettre_a_jour(schema_temps().temps_courant());
276
277 les_postraitements.mettre_a_jour(schema_temps().temps_courant());
278 les_postraitements.postraiter();
279 if (nb_pas_dt_max==0) indice_nb_pas_dt=1;
280 if (est_egal(t_init,t_max)) indice_tps_final = 1;
281 }
282 }
283 else
284 {
285 les_postraitements.traiter_postraitement();
286 }
287 statistiques().end_count(postraitement_counter_);
288 return 1;
289}

432//! 433/** \return Entree\&<br>
434Signification: le flot d'entre modifie<br>
435Contraintes: */
436Entree& Probleme_base::readOn(Entree& is)
437{
438 Cerr << "Reading of the problem " << le_nom() << finl;
439 format_sauv="binaire";

441// Desormais les fichiers de sauvegarde sont nommes par defaut nomducas_nomdupb.sauv

444 nom_fich+= ".sauv";
445 Motcle accolade_ouverte("{");
446 Motcle accolade_fermee("}");
447 Motcle motlu;
448
449 is >> motlu;
450
451 if (motlu != accolade_ouverte)
452 {
453 Cerr << "We expected { to start to read the problem" << finl;
454 exit();
455 }
456
457 lire_equations(is, motlu); //"motlu" contient le premier mot apres la lecture des equations
458
459// Si le postraitement comprend le mot, on en lit un autre...
460 while (les_postraitements.lire_postraitements(is, motlu, *this))
461 {
462 is >> motlu;
463 }
464
465 completer();
466 Cerr << "Step verification of data being read in progress..." << finl;
467 verifier();
468 Cerr << "The read data are coherent" << finl;
469 int resume_last_time = 0;
470 while(1)
471 {
472//
473// Lecture des options de reprise d'un calcul
474//
475 if ((motlu == "reprise") || (motlu == "resume_last_time"))

477 resume_last_time = (motlu == "resume_last_time" ? 1 : 0 );
478// remise a defaut a zero pour pouvoir faire une reprise std apres une reprise xyz
479 EcritureLectureSpecial::mode_lec=0;
480 Motcle format_rep;
481 is >> format_rep;

484 {
485 Cerr << "Restarting calculation... : keyword " << format_rep << " not understood. Waiting for:" << finl

488 }

509 }

512 {
513#ifdef MPI_

521 fic_hdf.read_dataset("/sauv", Process::me(),input_data);
522#endif
523 }
524 else
525 {
526 fic->ouvrir(nomfic);
527 if(fic->fail())
528 {
529 Cerr<<"Error during the opening of the restart file : " <<nomfic<<finl;
530 exit();
531 }
532 }
533
534// Restart from the last time
535 if (resume_last_time)
536 {
537// Resume_last_time is supported with xyz format
538// if (format_rep == "xyz")
539// {
540// Cerr << "Resume_last_time is not supported with xyz format yet." << finl;
541// exit();
542// }
543// Look for the last time and set it to tinit if tinit not set
544 double last_time = -1.;
545 if(format_rep == "single_hdf")
546 {
547#ifdef MPI_
548 last_time = get_last_time(input_data);
549#endif
550 }
551 else
552 last_time = get_last_time(fic);
553// Set the time to restart the calculation
554 schema_temps().set_temps_courant()=last_time;
555// Initialize tinit and current time according last_time
556 if (schema_temps().temps_init()>-DMAXFLOAT)
557 {
558 Cerr<<"tinit was defined in .data file to "<< schema_temps().temps_init()<< ". The value is fixed to "<<last_time<<" accroding to resume_last_time_option" <<finl;
559 }
560 schema_temps().set_temps_init()=last_time;
561 schema_temps().set_temps_precedent()=last_time;
562 Cerr << "==================================================================================================" << finl;
563 Cerr << "In the " << nomfic << " file, we find the last time: " << last_time << " and read the fields." << finl;
564 if( format_rep != "single_hdf")
565 {
566 fic->close();
567 fic->ouvrir(nomfic);
568 }
569 else
570 {
571#ifdef MPI_
572 fic_hdf.read_dataset("/sauv", Process::me(), input_data);
573#endif
574 }
575 }
576// Lecture de la version du format de sauvegarde si c'est une reprise classique
577// Depuis la 1.5.1, on marque le format de sauvegarde en tete des fichiers de sauvegarde
578// afin de pouvoir faire evoluer plus facilement ce format dans le futur
579// En outre avec la 1.5.1, les faces etant numerotees differemment, il est faux
580// de faire une reprise d'un fichier de sauvegarde anterieur et c'est donc un moyen
581// de prevenir les utilisateurs: il leur faudra faire une reprise xyz pour poursuivre
