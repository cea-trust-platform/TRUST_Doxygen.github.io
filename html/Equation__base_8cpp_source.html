314int Equation_base::lire_motcle_non_standard(const Motcle& mot, Entree& is)
315{
316 if (mot=="sources")
317 {
318 lire_sources(is);
319 return 1;
320 }
321 else if (mot=="conditions_limites|boundary_conditions")
322 {
323 lire_cl(is);
324 verif_Cl();
325 return 1;
326 }
327 else if (mot=="conditions_initiales|initial_conditions")
328 {
329 lire_cond_init(is);
330 return 1;
331 }
332 else if ((mot=="ecrire_fichier_xyz_valeur") || (mot=="ecrire_fichier_xyz_valeur_bin"))
333 {
334 if (mot=="ecrire_fichier_xyz_valeur_bin")
335 ecrit_champ_xyz_bin=1;
336 dt_ecrire_fic_xyz.resize_array(nombre_champ_xyz+1);
337 nb_bords_post_xyz.resize(nombre_champ_xyz+1);
338 nb_bords_post_xyz[nombre_champ_xyz]=0;
339 Motcle nom_champ;
340 is >> nom_champ;
341// il faut creer les champs associes au mot clef ecrire_fichier_xyz_valeur ICI
342// car dans methode ecrire_fichier_xyz()
343// seuls les champs definis via equation ou post-traiment sont creer
344 mon_probleme.valeur().creer_champ(nom_champ);
345
346 nom_champ_xyz.add(nom_champ);
347 is >> dt_ecrire_fic_xyz[nombre_champ_xyz];
348 nombre_champ_xyz++;
349 return 1;
350 }
351 else if (mot=="bords")
352 {
353 int nb_bords_post;
354 is >> nb_bords_post;
355 nb_bords_post_xyz[nombre_champ_xyz-1] = nb_bords_post;
356 if (nb_bords_post<=0)
357 {
358 Cerr << " ecrire_fichier_xyz_valeur - bords : the number of boundary on which you want to postprocess must be positive !! " << finl;
359 exit();
360 }
361 Noms noms_bord;
362 noms_bord.dimensionner(nb_bords_post);
363 for (int i=0; i<nb_bords_post; i++)
364 {
365 is >> noms_bord[i];
366 }
367 noms_bord_xyz.add(noms_bord);
368 return 1;
369 }
370 else if (mot=="equation_non_resolue")
371 {
372// differentiate an entry with an expression from an entry with the keyword eq_non_resoue_input_
373// On complete:
374 Motcle expr_equation_non_resolue;
375 is >> expr_equation_non_resolue;
376 equation_non_resolue_.setString(expr_equation_non_resolue);
377 if (expr_equation_non_resolue=="INPUT_INT_VALUE")
378 is >> eq_non_resolue_input_;
379 else
380 equation_non_resolue_.parseString();
381 }
382 else
383 {
384 Cerr << mot << " is not a keyword understood by " << que_suis_je() << " in lire_motcle_non_standard"<< finl;
385 exit();
386 }
387 return -1;
388}

839//! Returns the time derivative of the unknown I of the equation:<br>dI/dt = M-1*(sum(operators(I) + sources))<br>In : derivee contains I (but immediatly set to 0)<br>Out: derivee contains dI/dt
840DoubleTab& Equation_base::derivee_en_temps_inco(DoubleTab& derivee)
841{
842 derivee=0;
843 DoubleTrav secmem(derivee);
844// secmem = sum(operators) + sources + equation specific terms
845 if (schema_temps().diffusion_implicite())
846 {
847// Add convection operator only if equation has one
848 derivee=inconnue().valeurs();
849 if (nombre_d_operateurs()>1)
850 derivee_en_temps_conv(secmem, derivee);
851 }
852 else
853 {
854// Add all explicit operators
855 for(int i=0; i<nombre_d_operateurs(); i++)
856 if(operateur(i).l_op_base().get_decal_temps()!=1)
857 operateur(i).ajouter(secmem);
858 }
859 les_sources.ajouter(secmem);
860 if (calculate_time_derivative())
861 {
862// Store dI/dt(n) = M-1 secmem :
863 derivee_en_temps().valeurs()=secmem;
864 solveur_masse.appliquer(derivee_en_temps().valeurs());
865 schema_temps().modifier_second_membre((*this),secmem); //!< Change secmem for some schemes (eg: Adams_Bashforth)
866 }
867
868 corriger_derivee_expl(secmem); //!< Add specific term for an equation (eg: -gradP for Navier Stokes)
869
870 if (implicite_==0)
871 {
872 solveur_masse.appliquer(secmem); //!< M-1 * secmem
873 if (schema_temps().diffusion_implicite())
874 {
875// Solve: (1/dt + M-1*A)*dI = M-1 * secmem
876// where A is the diffusion
877 Equation_base::Gradient_conjugue_diff_impl(secmem, derivee);
878 }
879 else
880 {
881 derivee = secmem;
882 derivee.echange_espace_virtuel();
883 }
884 corriger_derivee_impl(derivee); //!< Solve specific implicit term for an equation (eg: pressure for Navier Stokes)
885
886 }
887 else if (implicite_>0)
888 {
889// TRUST support notices that this part has never been covered...
890// implicite
891// M dU/dt + AU* = f -BUn;
892// U* = Un+dt dU/dt
893// (M/dt + A) U* = f -BUn + M/dt Un
894//
895 double dt=schema_temps().pas_de_temps();
896 for(int i=0; i<nombre_d_operateurs(); i++)
897 {
898// boucle sur les operateurs
899 Operateur_base& op=operateur(i).l_op_base();
900 if(!op.get_matrice().non_nul())
901 op.set_matrice().typer("Matrice_Morse");
902 if(op.get_decal_temps()==1)
903 {
904// if (op.set_matrice().valeur().nb_lignes()<2)
905 {
906 Matrice_Morse& matrice=ref_cast(Matrice_Morse,op.set_matrice().valeur());
907 op.dimensionner(matrice);
908 sys_invariant_=0;
909 }
910 if(!sys_invariant_)
911 {
912 Matrice_Morse& matrice=ref_cast(Matrice_Morse, op.set_matrice().valeur());
913 op.contribuer_a_avec(inconnue().valeurs(), matrice);
914 solv_masse().ajouter_masse(dt, op.set_matrice().valeur());
915
916 if(
917 (op.get_solveur()->que_suis_je()=="Solv_Cholesky")
918 ||
919 (op.get_solveur()->que_suis_je()=="Solv_GCP")
920 )
921 {
922 Matrice_Morse_Sym new_mat(matrice);
923 new_mat.set_est_definie(1);
924 op.set_matrice()=new_mat;
925 ref_cast_non_const(SolveurSys,op.get_solveur())->reinit();
926 }
927 }
928 }
929 }
930 if(implicite_==1)
931 {
932// Un seul operateur implicite.
933// On suppose que c'est le premier (la diffusion !!)
934 Operateur_base& op=operateur(0).l_op_base();
935 Matrice_Base& matrice=op.set_matrice().valeur();
936// DoubleTrav secmem(derivee);
937 secmem=derivee;
938 solv_masse().ajouter_masse(dt, secmem, inconnue().valeurs());
939 op.contribuer_au_second_membre(secmem );
940 op.set_solveur().resoudre_systeme(matrice,
941 secmem,
942 derivee
943 );
944 solv_masse().corriger_solution(derivee,inconnue().valeurs());
945
946 derivee-=inconnue().valeurs();
947 derivee/=dt;
948
949// Sert uniquement a calculer les flux sur les bords quand la diffusion est implicitee !
950 DoubleTab resu;
951 resu=derivee;
952 operateur(0).calculer(inconnue().valeurs(), resu);
953 }
954 else
955 {
956// plusieurs operateurs implicites ...
957 Cerr << "Must be coded ... " << finl;
958 exit();
959 }
960 }
961 else
962 {
963 Cerr << "Error in Equation_base::derivee_en_temps_inco" << finl;
964 Cerr << "implicite_ = " << implicite_ << " has not been initialized!" << finl;
965 Cerr << "May be " << que_suis_je() << "::completer() method doesn't call Equation_base::completer()" << finl;
966 Cerr << "Contact TRUST support." << finl;
967 Process::exit();
968 }
969 return derivee;
970}
