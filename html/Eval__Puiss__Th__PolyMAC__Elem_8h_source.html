54inline double Eval_Puiss_Th_PolyMAC_Elem::calculer_terme_source(int num_elem) const
55{
56 if (sub_type(Champ_Uniforme,la_puissance.valeur().valeur()))
57 return puissance(0,0)*volumes(num_elem)*porosite_vol(num_elem);
58 else
59 {
60 if (puissance.nb_dim()==1)
61 return puissance(num_elem)*volumes(num_elem)*porosite_vol(num_elem);
62 else
63 return puissance(num_elem,0)*volumes(num_elem)*porosite_vol(num_elem);
64 }
65}

67inline void Eval_Puiss_Th_PolyMAC_Elem::calculer_terme_source(int e, DoubleVect& S) const
68{
69 const int N = S.size(), pc = sub_type(Champ_Uniforme,la_puissance.valeur().valeur());
70 for (int n = 0; n < N; n++) S[n] = puissance(!pc * e, n) * volumes(e) * porosite_vol(e);
71}
