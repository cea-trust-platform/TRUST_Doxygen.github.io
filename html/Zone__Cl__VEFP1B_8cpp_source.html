37void Zone_Cl_VEFP1B::imposer_cond_lim(Champ_Inc& ch,double temps)
38{
39 Zone_Cl_VEF::imposer_cond_lim(ch,temps);
40// dans le cas Navier stokes et si la condition est forte en pression aux sommets on impose la valeur aux sommets
41
42 if (sub_type(Navier_Stokes_std,ch.valeur().equation()))
43 {
44 const Zone_VEF_PreP1b& zone_vef=ref_cast(Zone_VEF_PreP1b,equation().zone_dis().valeur());
45 if ((zone_vef.get_cl_pression_sommet_faible()==0) && (zone_vef.get_alphaS()))
46 {
47 int nps=zone_vef.numero_premier_sommet();
48 DoubleTab& pression=ref_cast(Navier_Stokes_std,ch->equation()).pression().valeurs();
49 int nbsom_tot = zone_vef.nb_som_tot();
50 ArrOfDouble surf_loc(nbsom_tot);
51 int nb_cond_lims = nb_cond_lim();
52// On boucle une premiere fois pour mettre a zero la pression aux sommets
53 for(int i=0; i<nb_cond_lims; i++)
54 {
55 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
56 if (sub_type(Neumann_sortie_libre,la_cl))
57 {
58 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
59 const IntTab& faces = zone_vef.face_sommets();
60 int nbsf = faces.dimension(1);
61 int num2 = le_bord.nb_faces_tot();
62 for (int ind_face=0; ind_face<num2; ind_face++)
63 {
64 int face = le_bord.num_face(ind_face);
65 for(int som=0; som<nbsf; som++)
66 {
67 int som_glob = faces(face,som);
68 pression(nps+som_glob) = 0;
69 }
70 }
71 }
72 }
73
74// On boucle une deuxieme fois pour ajouter la contribution de chaque face
75 for(int i=0; i<nb_cond_lims; i++)
76 {
77 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
78 if (sub_type(Neumann_sortie_libre,la_cl))
79 {
80 const Neumann_sortie_libre& la_sortie_libre = ref_cast(Neumann_sortie_libre, la_cl);
81 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
82 const IntTab& faces = zone_vef.face_sommets();
83 int nbsf = faces.dimension(1);
84 int num2 = le_bord.nb_faces_tot();
85 for (int ind_face=0; ind_face<num2; ind_face++)
86 {
87 int face = le_bord.num_face(ind_face);
88 double P_imp = la_sortie_libre.flux_impose(ind_face);
89 double face_surf = zone_vef.face_surfaces(face);
90 for(int som=0; som<nbsf; som++)
91 {
92 int som_glob = faces(face,som);
93 pression(nps+som_glob) += P_imp*face_surf;
94 surf_loc[som_glob] += face_surf;
95 }
96 }
97 }
98 }
99// On boucle une troisieme fois pour diviser par la surface
100 for(int i=0; i<nb_cond_lims; i++)
101 {
102 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
103 if (sub_type(Neumann_sortie_libre,la_cl))
104 {
105 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
106 const IntTab& faces = zone_vef.face_sommets();
107 int nbsf = faces.dimension(1);
108 int num2 = le_bord.nb_faces_tot();
109 for (int ind_face=0; ind_face<num2; ind_face++)
110 {
111 int face = le_bord.num_face(ind_face);
112 for(int som=0; som<nbsf; som++)
113 {
114 int som_glob = faces(face,som);
115 double& surf = surf_loc[som_glob];
116 if (surf>0)
117 {
118 pression(nps+som_glob) /= surf;
119 surf = -1;
120 }
121 }
122 }
123 }
124 }
125 pression.echange_espace_virtuel();
126 Debog::verifier("pression dans Zone_Cl_VEFP1B::imposer_cond_lim",pression);
127 }
128 }
129}
