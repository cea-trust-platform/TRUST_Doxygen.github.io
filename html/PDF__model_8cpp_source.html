36Entree& PDF_model::readOn(Entree& is)
37{
38// (xdata documentation is in the TRAD_2.org because we need a special bloc_lecture object)
39 type_vitesse_imposee_ = -1; //DEFAULT VALUE = not set
40 Param param(que_suis_je());
41 param.ajouter("eta",&eta_, Param::REQUIRED);
42 temps_relax_=1.0e+12;
43 param.ajouter("temps_relaxation_coefficient_PDF",&temps_relax_,Param::OPTIONAL);
44 echelle_relax_=5.0e-2;
45 param.ajouter("echelle_relaxation_coefficient_PDF",&echelle_relax_,Param::OPTIONAL);
46 param.ajouter_flag("local",&local_);
47 param.ajouter_non_std("vitesse_imposee_data",(this),Param::OPTIONAL);
48 param.ajouter_non_std("vitesse_imposee_fonction",(this),Param::OPTIONAL);
49 param.lire_avec_accolades_depuis(is);
50 if (type_vitesse_imposee_ == -1)
51 {
52 Cerr << "PDF_model: you must specify either 'vitesse_imposee_data' or 'vitesse_imposee_fonction'" << finl;
53 Process::exit();
54 }
55 coefku_ = 1./eta_;
56 return is;
57}

59int PDF_model::lire_motcle_non_standard(const Motcle& un_mot, Entree& is)
60{
61 if (un_mot == "vitesse_imposee_fonction")
62 {
63 if (Process::je_suis_maitre())
64 {
65 Cerr << " PDF_model is reading imposed velocity... " << finl;
66 }
67 type_vitesse_imposee_ = 1;
68 Nom expr_vit_imp;
69 int dim = Objet_U::dimension;
70 parsers_.dimensionner(dim);
71 for (int i = 0; i < dim; i++)
72 {
73 is >> expr_vit_imp;
74 std::string sx(expr_vit_imp);
75 for (auto & c: sx) c = toupper(c);
76 parsers_[i].setString(sx);
77 parsers_[i].setNbVar(4);
78 parsers_[i].addVar("x");
79 parsers_[i].addVar("y");
80 parsers_[i].addVar("z");
81 parsers_[i].addVar("t");
82 parsers_[i].parseString();
83 }
84 }
85 else if (un_mot == "vitesse_imposee_data")
86 {
87 type_vitesse_imposee_ = 0;
88 is >> vitesse_imposee_lu_;
89 }
90 else
91 {
92 Cerr << "PDF_model: token not understood: " << un_mot << finl;
93 Process::exit();
94 }
95 return 1;
96}

98void PDF_model::affecter_vitesse_imposee(Zone_VF& la_zone, const DoubleTab& coords)
99{
100 if (type_vitesse_imposee_ == 1)
101 {
102 int nb_som_tot = la_zone.nb_som_tot();
103
104 DoubleTab& vitesse_imposee_ref = vitesse_imposee_.valeur().valeurs();
105
106 int dim = Objet_U::dimension;
107
108 ArrOfDouble x(dim);
109
110 for (int i = 0; i < nb_som_tot; i++)
111 {
112 for (int j = 0; j < dim; j++)
113 {
114 x[j] = coords(i,j);
115 }
116 for (int j = 0; j < dim; j++)
117 {
118 vitesse_imposee_ref(i,j) = get_vitesse_imposee(x,j);
119 }
120 }
121 }
122 else
123 {
124 Cerr << __FILE__ << ", line " << (int)__LINE__ << " : Unexpected error." << finl;
125 exit();
126 }
127}
