107//! Discretise le domaine.<br>On type et on discretise chaque zone discretisee du domaine discretise<br>grace a la discretisation passee en parametre.
108void Domaine_dis::discretiser(const Nom& type_1)
109{
110 Nom type(type_1);
111 int face_ok=1;
112 type.suffix("NO_FACE_");
113 if (type!=type_1) face_ok=0;
114 if (!le_domaine.non_nul())
115 {
116 Cerr << "The domain is not specified. Please check your data file." << finl;
117 exit();
118 }
119 const Domaine& dom=le_domaine.valeur();
120 i_am_allocator_of_les_zones = domaine().le_nom()+"_"+type; //!< Nom unique
121 if (domaines_dis.find(i_am_allocator_of_les_zones.getChar()) != domaines_dis.end()) //on a deja discretise ce domaine!
122 {
123 *this = domaines_dis[i_am_allocator_of_les_zones.getChar()].valeur();
124 i_am_allocator_of_les_zones = "";
125 return;
126 }
127// Initialisation du tableau renum_som_perio
128 const int nb_som = dom.nb_som();
129 IntTab renum(nb_som);
130 for (int i = 0; i < nb_som; i++)
131 renum[i] = i;
132 le_domaine.valeur().set_renum_som_perio(renum);
133
134// Cree les Zone_dis, les type et leur associe la zone et le domaine
135// correspondants.
136 Zones_dis *zptr = new Zones_dis;
137 les_zones = *zptr;
138 les_zones->dimensionner(dom.nb_zones());
139
140 for(int i=0; i<nombre_de_zones(); i++)
141 {
142 zone_dis(i).typer(type);
143 zone_dis(i).associer_zone(dom.zone(i));
144 if (face_ok)
145 zone_dis(i).discretiser();
146 else
147 {
148 Zone& zone_geom = domaine().zone(0);
149 Zone_VF& zvf = ref_cast(Zone_VF,zone_dis(i).valeur());
150 zvf.typer_elem(zone_geom);
151// Calcul du volume de la zone discretisee
152 zone_geom.calculer_volumes(zvf.volumes(),zvf.inverse_volumes());
153 }
154 zone_dis(i).associer_domaine_dis(*this);
155 }
156
157// Cree les sous_zones_dis, les type, et leur associe
158// les zone_dis et les sous_zone correspondantes.
159 Sous_zones_dis *szptr = new Sous_zones_dis;
160 les_sous_zones_dis = *szptr;
161 les_sous_zones_dis->dimensionner(dom.nb_ss_zones());
162
163 for (int i=0; i<dom.nb_ss_zones(); i++)
164 {
165
166// Cherche la Zone_dis associee
167 const Zone& la_zone=dom.ss_zone(i).zone();
168 int z=0;
169 while(z<nombre_de_zones() && !zone_dis(z).zone().est_egal_a(la_zone))
170 z++;
171 if (z==nombre_de_zones())
172 {
173 Cerr << "In Domaine_dis::discretiser, impossible to find a "
174 << "zone_dis corresponding to sous_zone " << i << finl;
175 exit();
176 }
177 const Zone_dis& la_zone_dis=zone_dis(z);
178 if (face_ok)
179 {
180 if (sub_type(Zone_VF,la_zone_dis.valeur()))
181 {
182 sous_zone_dis(i).typer("Sous_zone_VF");
183 sous_zone_dis(i).associer_sous_zone(dom.ss_zone(i));
184 sous_zone_dis(i).associer_zone_dis(la_zone_dis.valeur());
185 sous_zone_dis(i).discretiser();
186 }
187 else
188 {
189 Cerr << "Domaine_dis::discretiser : there is no sub-areas discretized "
190 << "for discretized areas which not deriving from Zone_VF : "
191 << la_zone_dis.valeur().que_suis_je() << finl;
192 exit();
193 }
194 }
195 }
196// memoization
197 domaines_dis[i_am_allocator_of_les_zones.getChar()] = *this;
198 domaines_dis[Nom(Nom("NO_FACE_") + i_am_allocator_of_les_zones).getChar()] = *this;
199
200}
