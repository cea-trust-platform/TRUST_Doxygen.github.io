153// J'ajoute la methode valeur_a_elem()
154inline DoubleVect& Champ_Fonc_MED::valeur_a_elem(const DoubleVect& position, DoubleVect& tab_valeurs, int le_poly) const
155{
156// Le codage de cette methode passe par valeur_a() et ignore l'entier le_poly
157// car si on utilise le_champ().valeur_a_elem(position,tab_valeurs,le_poly), on
158// a des ecarts sequentiel-parallele pour les champs P0 (et probablement aussi pour
159// les autres champs).
160// Je m'explique cela par le fait qu'en parallele, le_poly designe un numero d'element
161// dans la partition associee a un processeur, alors que le champ designe par le pointeur vrai_champ_
162// est defini sur la totalite du domaine.
163// Pour cette raison, j'utilise valeur_a(), qui localise la position dans la totalite du domaine,
164// et qui retourne la valeur de vrai_champ_ dans cet element.
165 return le_champ().valeur_a(position,tab_valeurs);
166}
