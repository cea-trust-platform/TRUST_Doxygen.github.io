56double Op_Conv_VDF_base::calculer_dt_stab() const
57{
58 const Zone_VDF& zone_VDF = iter.zone();
59 const Zone_Cl_VDF& zone_Cl_VDF = iter.zone_Cl();
60 const IntTab& face_voisins = zone_VDF.face_voisins();
61 const DoubleVect& volumes = zone_VDF.volumes();
62 const DoubleVect& face_surfaces = zone_VDF.face_surfaces();
63 const DoubleVect& vit_associe = vitesse().valeurs();
64 const DoubleVect& vit= (vitesse_pour_pas_de_temps_.non_nul()?vitesse_pour_pas_de_temps_.valeur().valeurs(): vit_associe);
65 DoubleTab fluent;
66// fluent est initialise a zero par defaut:
67 zone_VDF.zone().creer_tableau_elements(fluent);
68
69// Remplissage du tableau fluent
70 double psc;
71 int num1, num2, face, elem1;
72
73// On traite les bords
74 for (int n_bord = 0; n_bord < zone_VDF.nb_front_Cl(); n_bord++)
75 {
76 const Cond_lim& la_cl = zone_Cl_VDF.les_conditions_limites(n_bord);
77 if ( sub_type(Dirichlet_entree_fluide,la_cl.valeur()) || sub_type(Neumann_sortie_libre,la_cl.valeur()) )
78 {
79 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
80 num1 = le_bord.num_premiere_face();
81 num2 = num1 + le_bord.nb_faces();
82 for (face=num1; face<num2; face++)
83 {
84 psc = vit[face]*face_surfaces(face);
85 if ( (elem1 = face_voisins(face,0)) != -1)
86 {
87 if (psc < 0) fluent[elem1] -= psc;
88 }
89 else //!< (elem2 != -1)
90 if (psc > 0) fluent[face_voisins(face,1)] += psc;
91 }
92 }
93 }
94
95// Boucle sur les faces internes pour remplir fluent
96 const int zone_VDF_nb_faces = zone_VDF.nb_faces(), premiere_face = zone_VDF.premiere_face_int();
97 for (face = premiere_face; face < zone_VDF_nb_faces; face++)
98 {
99 psc = vit[face]*face_surfaces(face);
100 eval_fluent(psc,face_voisins(face,0),face_voisins(face,1),fluent);
101 }
102
103// Calcul du pas de temps de stabilite a partir du tableau fluent
104 if (vitesse().le_nom()=="rho_u" && equation().probleme().is_dilatable())
105 diviser_par_rho_si_dilatable(fluent,equation().milieu());
106
107 double dt_stab = 1.e30;
108 int zone_VDF_nb_elem=zone_VDF.nb_elem();
109// dt_stab = min ( 1 / ( |U|/dx + |V|/dy + |W|/dz ) )
110 for (int num_poly=0; num_poly<zone_VDF_nb_elem; num_poly++)
111 {
112 double dt_elem = volumes(num_poly)/(fluent[num_poly]+DMINFLOAT);
113 if (dt_elem<dt_stab) dt_stab = dt_elem;
114 }
115 dt_stab = Process::mp_min(dt_stab);
116
117// astuce pour contourner le type const de la methode
118 Op_Conv_VDF_base& op =ref_cast_non_const(Op_Conv_VDF_base, *this);
119 op.fixer_dt_stab_conv(dt_stab);
120 return dt_stab;
121}

125// This is the equivalent of "Op_Conv_VDF_base :: calculer_dt_stab ()"
126void Op_Conv_VDF_base::calculer_dt_local(DoubleTab& dt_face) const
127{
128 const Zone_VDF& zone_VDF = iter.zone();
129 const Zone_Cl_VDF& zone_Cl_VDF = iter.zone_Cl();
130 const DoubleVect& volumes_entrelaces= zone_VDF.volumes_entrelaces();
131 const DoubleVect& face_surfaces = zone_VDF.face_surfaces();
132// const DoubleVect& vit= vitesse_pour_pas_de_temps_.valeur().valeurs();
133 const DoubleVect& vit=equation().inconnue().valeurs();
134 DoubleTrav fluent(volumes_entrelaces);
135
136// Remplissage du tableau fluent
137// On traite les bords
138 for (int n_bord = 0; n_bord < zone_VDF.nb_front_Cl(); n_bord++)
139 {
140 const Cond_lim& la_cl = zone_Cl_VDF.les_conditions_limites(n_bord);
141
142 if ( sub_type(Dirichlet_entree_fluide,la_cl.valeur()) || sub_type(Neumann_sortie_libre,la_cl.valeur()) )
143 {
144 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
145 const int num1 = le_bord.num_premiere_face(), num2 = num1 + le_bord.nb_faces();
146 for (int face = num1; face < num2; face++)
147 {
148 double value = vit[face]*face_surfaces(face);
149 if (value >0) fluent[face] = value;
150 else fluent[face] -= value;
151 }
152 }
153 }
154
155// Boucle sur les faces internes pour remplir fluent
156 const int zone_VDF_nb_faces = zone_VDF.nb_faces(), premiere_face = zone_VDF.premiere_face_int();
157 for (int face = premiere_face; face < zone_VDF_nb_faces; face++)
158 {
159 const double value = vit[face]*face_surfaces(face);
160 if (value >0) fluent[face] = value;
161 else fluent[face] -= value;
162 }
163
164
165// Calcul du pas de temps de stabilite a partir du tableau fluent
166 if (vitesse().le_nom()=="rho_u" && equation().probleme().is_dilatable())
167 diviser_par_rho_si_dilatable(fluent,equation().milieu());
168
169 dt_face=(volumes_entrelaces);
170// Boucle sur les faces de bords
171 for (int n_bord=0; n_bord<zone_VDF.nb_front_Cl(); n_bord++)
172 {
173 const Cond_lim& la_cl = zone_Cl_VDF.les_conditions_limites(n_bord);
174 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
175 const int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
176 for (int num_face = ndeb; num_face < nfin; num_face++)
177 {
178 if( sup_strict(fluent[num_face], 1.e-16) ) dt_face(num_face)= volumes_entrelaces(num_face)/fluent[num_face];
179 else dt_face(num_face) = -1.;
180 }
181 }
182
183// Boucle sur les faces internes
184 for (int num_face = premiere_face; num_face<zone_VDF_nb_faces; num_face++)
185 {
186 if( sup_strict(fluent[num_face], 1.e-16) ) dt_face(num_face)= volumes_entrelaces(num_face)/fluent[num_face];
187 else dt_face(num_face) = -1.;
188 }
189
190 double max_dt_local= dt_face.mp_max_abs_vect();
191 const int taille = dt_face.size();
192 for(int i = 0; i < taille; i++)
193 if(! sup_strict(dt_face(i), 1.e-16)) dt_face(i) = max_dt_local;
194
195 dt_face.echange_espace_virtuel();
196
197 for (int n_bord = 0; n_bord < zone_VDF.nb_front_Cl(); n_bord++)
198 {
199 const Cond_lim& la_cl = zone_Cl_VDF.les_conditions_limites(n_bord);
200 if (sub_type(Periodique,la_cl.valeur()))
201 {
202 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
203 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
204 const int nb_faces_bord = le_bord.nb_faces();
205 for (int ind_face = 0; ind_face < nb_faces_bord; ind_face++)
206 {
207 int ind_face_associee = la_cl_perio.face_associee(ind_face);
208 int face = le_bord.num_face(ind_face), face_associee = le_bord.num_face(ind_face_associee);
209 if (!est_egal(dt_face(face),dt_face(face_associee),1.e-8))
210 {
211 dt_face(face) = std::min(dt_face(face),dt_face(face_associee));
212 }
213 }
214 }
215 }
216 dt_face.echange_espace_virtuel();
217// dt_conv_locaux=dt_face;
218}
