88// Methode de calcul du rotationnel de la vorticite.
89DoubleTab& Op_Rot_VEFP1B::ajouter(const DoubleTab& vorticite, DoubleTab& rot) const
90{
91 Cerr << "je suis dans OpRot" << finl;
92
93 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
94 const Zone& zone = zone_VEF.zone();
95 const IntTab& face_voisins = zone_VEF.face_voisins();
96
97 int elem0,elem1;
98 int sommet_global,face_opp;
99
100// Quelques variables bien utiles
101 DoubleTab vecteur_normal_elem0(dimension);
102 DoubleTab vecteur_normal_elem1(dimension);
103 DoubleTab vecteur_normal_opp_elem0(dimension);
104 DoubleTab vecteur_normal_opp_elem1(dimension);
105
106 IntList sommets_elem0, sommets_elem1;
107 IntList faces_opp_elem0,faces_opp_elem1;
108
109// On traite les conditions limites:
110// pour l'instant pas de conditions aux limites pour la vorticite.
111// Sortie libre = pas de conditions aux limites ????
112
113// On traite les faces internes i.e. sans les conditions aux limites
114 int premiere_face_int = zone_VEF.premiere_face_int();
115 int nb_faces = zone_VEF.nb_faces();
116
117 if (dimension > 2)
118 {
119 Cerr << "Erreur OpRot: seul le cas 2D est etudie pour le moment" << finl;
120 Process::exit();
121 }
122
123 for (int face=premiere_face_int; face<nb_faces; face++)
124 {
125 int modulo;
126 elem0 = face_voisins(face,0);
127 elem1 = face_voisins(face,1);
128
129// Calcul du rotationnel de la vorticite: partie P0 de l'inconnue
130
131 /* Calcul du vecteur tangent adequat pour la partie P0 de l'inconnue */
132 vecteur_normal_elem0 = vecteur_normal(face,elem0);
133 vecteur_normal_elem1 = vecteur_normal(face,elem1);
134
135 /* Calcul du rotationnel de la vorticite proprement dit */
136 /* Partie P0 */
137 for (int composante=0; composante<dimension; composante++)
138 {
139 modulo = (1+composante)%dimension;
140 rot(face,composante) += pow(-1.,modulo)*
141 ( vorticite(elem0) * vecteur_normal_elem0(modulo)+
142 vorticite(elem1) * vecteur_normal_elem1(modulo));
143
144 }
145
146// Calcul du rotationnel de la vorticite: partie P1 de l'inconnue
147
148 /* On a besoin pour chaque triangle de connaitre la liste */
149 /* des sommets qui en font parti ainsi que les faces */
150 /* opposees a ces sommets */
151 if (!sommets_elem0.est_vide()) sommets_elem0.vide();
152 if (!sommets_elem1.est_vide()) sommets_elem1.vide();
153 if (!faces_opp_elem0.est_vide()) faces_opp_elem0.vide();
154 if (!faces_opp_elem1.est_vide()) faces_opp_elem1.vide();
155
156 for (int i=0; i<zone.nb_som_elem(); i++)
157 {
158 sommets_elem0.add_if_not(zone.sommet_elem(elem0,i));
159 faces_opp_elem0.add_if_not(zone_VEF.elem_faces(elem0,i));
160 sommets_elem1.add_if_not(zone.sommet_elem(elem1,i));
161 faces_opp_elem1.add_if_not(zone_VEF.elem_faces(elem1,i));
162 }
163
164 /* Maintenant, on calcule la contribution au rotationnel */
165 /* pour la partie P1 de l'inconnue */
166 /* ATTENTION: compte tenu de la definition de l'espace */
167 /* vectoriel engendrant omega, il NE FAUT PAS tenir compte */
168 /* du sommet de plus GRAND numero */
169 for (int composante=0; composante<dimension; composante++)
170 {
171 modulo = (1+composante)%dimension;
172
173//
174 /* Partie (w,curl Psi_c e_i) */
175//
176
177 double sommation_sommets=0.;
178// On fait les calculs pour "elem0"
179 for (int i=0; i<sommets_elem0.size(); i++)
180 {
181 sommet_global = sommets_elem0[i];
182
183// Pour eliminer le dernier sommet
184 if (sommet_global != zone.nb_som()-1)
185 sommation_sommets += vorticite(zone.nb_elem()+sommet_global);
186 }
187
188 rot(face,composante) += -pow(-1.,modulo)/(dimension+1)*
189 vecteur_normal_elem0(modulo) * sommation_sommets;
190
191// On fait les calculs pour "elem1"
192 sommation_sommets=0.;
193 for (int i=0; i<sommets_elem1.size(); i++)
194 {
195 sommet_global = sommets_elem1[i];
196
197// Pour eliminer le dernier sommet
198 if (sommet_global != zone.nb_som()-1)
199 sommation_sommets += vorticite(zone.nb_elem()+sommet_global);
200 }
201
202 rot(face,composante) += -pow(-1.,modulo)/(dimension+1)*
203 vecteur_normal_elem1(modulo) * sommation_sommets;
204
205
206//
207 /* Fin de partie (w,curl Psi_c e_i) */
208//
209
210//
211 /* Partie (curl w,Psi_c e_i) */
212//
213
214// On fait les calculs pour "elem0"
215 sommation_sommets=0.;
216 for (int i=0; i<sommets_elem0.size(); i++)
217 {
218 sommet_global = sommets_elem0[i];
219
220// Pour eliminer le dernier sommet
221 if (sommet_global != zone.nb_som()-1)
222 {
223 face_opp = faces_opp_elem0[i];
224 vecteur_normal_opp_elem0 = vecteur_normal(face_opp,elem0);
225
226 sommation_sommets += vorticite(zone.nb_elem()+sommet_global)
227 * vecteur_normal_opp_elem0(modulo);
228 }
229
230 }
231
232 rot(face,composante) += pow(-1.,modulo)/(dimension*(dimension+1))*
233 sommation_sommets;
234
235// On fait les calculs pour "elem1"
236 sommation_sommets=0.;
237 for (int i=0; i<sommets_elem1.size(); i++)
238 {
239 sommet_global = sommets_elem1[i];
240
241// Pour eliminer le dernier sommet
242 if (sommet_global != zone.nb_som()-1)
243 {
244 face_opp = faces_opp_elem1[i];
245 vecteur_normal_opp_elem1 = vecteur_normal(face_opp,elem1);
246
247 sommation_sommets += vorticite(zone.nb_elem()+sommet_global)
248 * vecteur_normal_opp_elem1(modulo);
249 }
250
251 }
252
253// C_h^t
254 rot(face,composante) += pow(-1.,modulo)/(dimension*(dimension+1))*
255 sommation_sommets;
256
257//
258 /* Fin de partie (curl w,Psi_c e_i) */
259//
260
261 } //fin du for sur "composante" pour la partie P1 de l'inconnue
262
263 } //fin du for sur "face"
264
265 return rot;
266
267}

269DoubleTab
270Op_Rot_VEFP1B::vecteur_normal(const int face, const int elem) const
271{
272 assert(dimension == 2);
273
274 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
275 DoubleTab le_vecteur_normal(dimension);
276
277 for (int composante = 0; composante<dimension; composante++)
278
279 le_vecteur_normal(composante) = zone_VEF.face_normales(face,composante)
280 * zone_VEF.oriente_normale(face,elem);
281
282 return le_vecteur_normal;
283}
