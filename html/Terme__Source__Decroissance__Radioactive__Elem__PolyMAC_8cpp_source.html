94void Terme_Source_Decroissance_Radioactive_Elem_PolyMAC::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
95{
96 const Zone_VF& zone = la_zone_PolyMAC.valeur();
97 const DoubleVect& pe = zone.porosite_elem(), &ve = zone.volumes();
98 const DoubleTab& c = equation().inconnue().valeurs();
99 std::string nom_inco = equation().inconnue().le_nom().getString();
100 Matrice_Morse *Mc = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
101 const int N = c.line_size();
102
103 for (int e = 0; e < zone.nb_elem(); e++)
104 for (int l = 0; l < N; l++)
105 {
106 const double fac = pe(e) * ve(e) * lambda[l];
107 secmem(e, l) -= fac * c(e, l);
108 if (Mc) (*Mc)(N * e + l, N * e + l) += fac;
109 }
