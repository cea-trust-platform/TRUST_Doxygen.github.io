90void Milieu_base::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
91{
92 Cerr << "Medium discretization." << finl;
93 Champ_Don& ch_lambda = conductivite();
94 Champ_Don& ch_alpha = diffusivite();
95 Champ_Don& ch_beta_th = beta_t();
96
97 const Zone_dis_base& zone_dis=pb.equation(0).zone_dis();
98// PL: pas le temps de faire plus propre, je fais comme dans Fluide_Incompressible::discretiser
99// pour gerer une conductivite lue dans un fichier MED. Test: Reprise_grossier_fin_VEF
100// ToDo: reecrire ces deux methodes discretiser
101
102// E. Saikali
103// The thermal conductivity and diffusivity fields are considered as
104// multi_scalaire fields, sure if the number of components read
105// in the data file for lambda > 1, i.e: case of anisotropic diffusion !
106
107 int lambda_nb_comp = 0;
108
109 if(ch_lambda.non_nul())
110 {
111// Returns number of components of lambda field
112 lambda_nb_comp = ch_lambda.nb_comp( );
113 if (sub_type(Champ_Fonc_MED,ch_lambda.valeur()))
114 {
115 double temps=ch_lambda.valeur().temps();
116 Cerr<<"Convert Champ_fonc_MED lambda in Champ_Don..."<<finl;
117 Champ_Don ch_lambda_prov;
118 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",lambda_nb_comp,temps,ch_lambda_prov);
119 ch_lambda_prov.affecter_(ch_lambda.valeur());
120 ch_lambda.detach();
121 ch_alpha.detach();
122 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",lambda_nb_comp,temps,ch_lambda);
123 ch_lambda.valeur().valeurs()=ch_lambda_prov.valeur().valeurs();
124 }
125
126 if(lambda_nb_comp >1) //!< Pour anisotrope
127 ch_lambda.valeur().fixer_nature_du_champ(multi_scalaire);
128
129 dis.nommer_completer_champ_physique(zone_dis,"conductivite","W/m/K",ch_lambda.valeur(),pb);
130
131// le vrai nom sera donne plus tard
132 if (sub_type(Champ_Fonc_Tabule,ch_lambda.valeur()))
133 {
134 double temps=ch_lambda.valeur().temps();
135 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",lambda_nb_comp,temps,ch_alpha);
136 }
137 champs_compris_.ajoute_champ(ch_lambda.valeur());
138 }
139 if (!ch_alpha.non_nul()&&(ch_lambda.non_nul()))
140 {
141 double temps=ch_lambda.valeur().temps();
142// ch_alpha (i.e. diffusivite_thermique) will have same component number as ch_lambda
143 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",lambda_nb_comp,temps,ch_alpha);
144 }
145 if (ch_alpha.non_nul())
146 {
147 dis.nommer_completer_champ_physique(zone_dis,"diffusivite_thermique","m2/s",ch_alpha.valeur(),pb);
148 champs_compris_.ajoute_champ(ch_alpha.valeur());
149 }
150 if (ch_beta_th.non_nul())
151 {
152 dis.nommer_completer_champ_physique(zone_dis,"dilatabilite","K-1",ch_beta_th.valeur(),pb);
153 champs_compris_.ajoute_champ(ch_beta_th.valeur());
154 }
155 if (Cp.non_nul())
156 {
157 dis.nommer_completer_champ_physique(zone_dis,"capacite_calorifique","J/kg/K",Cp.valeur(),pb);
158 champs_compris_.ajoute_champ(Cp.valeur());
159 }
160 if (rho.non_nul())
161 {
162 dis.nommer_completer_champ_physique(zone_dis,"masse_volumique","kg/m^3",rho.valeur(),pb);
163 champs_compris_.ajoute_champ(rho.valeur());
164 }
165 if (rho.non_nul() && Cp.non_nul())
166 {
167 assert (rho.nb_comp() == Cp.nb_comp());
168 if(!rho_cp_comme_T_.non_nul())
169 {
170 const double temps = pb.schema_temps().temps_courant();
171// E. Saikali
172// XXX : j'ai remis nb_comp = 1, sinon ca bloque dans Solveur_Masse_base => tab_divide_any_shape
173// parce qu'on a pas line_size % line_size_vx == 0 (cas nb_comp >1 pour rho et cp
174// TODO : FIXME : faut coder un cas generique dans DoubleVect::tab_divide_any_shape... bon courage
175 dis.discretiser_champ("temperature", zone_dis, "rho_cp_comme_T", "J/m^3/K", 1 /* rho.nb_comp() */, temps, rho_cp_comme_T_);
176 dis.discretiser_champ( "champ_elem", zone_dis, "rho_cp_elem", "J/m^3/K", 1 /* rho.nb_comp() */, temps, rho_cp_elem_);
177 }
178 champs_compris_.ajoute_champ(rho_cp_comme_T_.valeur());
179 champs_compris_.ajoute_champ(rho_cp_elem_.valeur());
180 }
181}

332//! Calcul de alpha=lambda/(rho*Cp).<br>Suivant lambda, rho et Cp alpha peut-etre type comme<br>un champ uniforme ou un champ uniforme par morceau.
333void Milieu_base::calculer_alpha()
334{
335 if(lambda.non_nul())
336 {
337 DoubleTab& tabalpha = alpha.valeurs();
338 tabalpha = lambda.valeurs();
339
340// [ABN]: allows variable rho, Cp at this level (will be used by Solide_Milieu_Variable for instance).
341 if (sub_type(Champ_Uniforme,rho.valeur()))
342 tabalpha /= rho.valeurs()(0,0);
343 else
344 tab_divide_any_shape(tabalpha,rho.valeurs());
345
346 if (sub_type(Champ_Uniforme,Cp.valeur()))
347 tabalpha /= Cp.valeurs()(0,0);
348 else
349 tab_divide_any_shape(tabalpha,Cp.valeurs());
350 }
351 else
352 {
353 Cerr << "alpha calculation is not possible since lambda is not known." << finl;
354 }

356void Milieu_base::mettre_a_jour(double temps)
357{
358// Cerr << que_suis_je() << "Milieu_base::mettre_a_jour" << finl;
359 if (rho.non_nul())

361 if (g.non_nul())

363 if (lambda.non_nul())
364 lambda.mettre_a_jour(temps);

367 if (beta_th.non_nul())

369
370 if ( (lambda.non_nul()) && (Cp.non_nul()) && (rho.non_nul()) )
371 {
372 calculer_alpha();
373 alpha.valeur().changer_temps(temps);
374 }
375 if (rho_cp_comme_T_.non_nul())
376 update_rho_cp(temps);

380void Milieu_base::update_rho_cp(double temps)
381{
382 rho_cp_elem_.changer_temps(temps);
383 rho_cp_elem_.valeur().changer_temps(temps);
384 DoubleTab& rho_cp=rho_cp_elem_.valeurs();
385 if (sub_type(Champ_Uniforme,rho.valeur()))
386 rho_cp=rho.valeurs()(0,0);
387 else
388 {
389// AB: rho_cp = rho.valeurs() turns rho_cp into a 2 dimensional array with 1 compo. We want to stay mono-dim:
390 rho_cp = 1.;
391 tab_multiply_any_shape(rho_cp, rho.valeurs());
392 }
393 if (sub_type(Champ_Uniforme,Cp.valeur()))
394 rho_cp*=Cp.valeurs()(0,0);
395 else
396 tab_multiply_any_shape(rho_cp,Cp.valeurs());
397 rho_cp_comme_T_.changer_temps(temps);
398 rho_cp_comme_T_.valeur().changer_temps(temps);
399 const MD_Vector& md_som = rho_cp_elem_.zone_dis_base().domaine_dis().domaine().md_vector_sommets(),
400 &md_faces = ref_cast(Zone_VF,rho_cp_elem_.zone_dis_base()).md_vector_faces();
401 if (rho_cp_comme_T_.valeurs().get_md_vector() == rho_cp_elem_.valeurs().get_md_vector())
402 rho_cp_comme_T_.valeurs() = rho_cp;
403 else if (rho_cp_comme_T_.valeurs().get_md_vector() == md_som)
404 Discretisation_tools::cells_to_nodes(rho_cp_elem_,rho_cp_comme_T_);
405 else if (rho_cp_comme_T_.valeurs().get_md_vector() == md_faces)
406 Discretisation_tools::cells_to_faces(rho_cp_elem_,rho_cp_comme_T_);
407 else
408 {
409 Cerr<< que_suis_je()<<(int)__LINE__<<finl;
410 Process::exit();
411 }
