78//! Fonction principale de l'interprete Extract_2D_from_3D<br>Structure du jeu de donnee (en dimension 3) :<br>Extract_2D_from_3D dom3D bord3D dom2D
79/** \return Entree\&<br>

82Entree& TroisDto2D::interpreter_(Entree& is)
83{
84 if(Objet_U::dimension==2)
85 {
86 Cerr << finl;
87 Cerr << "You can not use " << que_suis_je() << " on a case already 2D" << finl;
88 exit();
89 }
90 dimension=3;
91 int test_axi=0;
92 if (que_suis_je()=="Extract_2Daxi_from_3D")
93 {
94 test_axi=1;
95 Cerr<<"We generate a 2D axi domain"<<finl;
96 }
97 Nom nom_bord, nom_dom2D;
98 associer_domaine(is);
99 is >> nom_bord >> nom_dom2D;
100 Cerr << "Extraction of boundary " << nom_bord
101 << " of domain " << domaine().le_nom()
102 << " --> " << nom_dom2D << finl;
103 {
104// Si l'objet nom_dom2D n'existe pas encore, on
105// ajoute un nouvel objet a l'interprete bloc courant
106 Interprete_bloc& interp = Interprete_bloc::interprete_courant();
107 if (interp.objet_global_existant(nom_dom2D))
108 {
109 if (sub_type(Domaine, interp.objet_global(nom_dom2D)))
110 {
111 Cerr << "Domain " << nom_dom2D
112 << " already exists, writing to this object." << finl;
113 }
114 else
115 {
116 Cerr << "Error : object " << nom_dom2D
117 << " already exists and is not a domain." << finl;
118 exit();
119 }
120 }
121 else
122 {
123 Cerr << "Declaring a new domain with name " << nom_dom2D << finl;
124 DerObjU ob;
125 ob.typer("Domaine");
126 interp.ajouter(nom_dom2D, ob);
127 }
128 }
129 Domaine& dom2D = ref_cast(Domaine, objet(nom_dom2D));
130 const Domaine& dom3D = domaine();
131 const Zone& zone3D=dom3D.zone(0);
132 const Bord& bord3D=zone3D.bord(nom_bord);
133 Scatter::uninit_sequential_domain(dom2D);
134 extraire_2D(dom3D, dom2D, bord3D,nom_bord, test_axi);
135 Scatter::init_sequential_domain(dom2D);
136 return is;

140void TroisDto2D::extraire_2D(const Domaine& dom3D, Domaine& dom2D, const Bord& bord3D, const Nom& nom_bord, int test_axi)
141{
142 const DoubleTab& coord_sommets3D=dom3D.coord_sommets();
143 DoubleTab& coord_sommets2D=dom2D.les_sommets();
144 const Zone& zone3D=dom3D.zone(0);
145 const Faces& faces3D=bord3D.faces();
146 const IntTab& les_faces3D=faces3D.les_sommets();
147 int nb_som3D=coord_sommets3D.dimension(0);
148 Cerr << "number of 3D nodes :" << nb_som3D << finl;
149 int nb_faces3D=les_faces3D.dimension(0);
150 Cerr << "number of 3D faces of " << nom_bord << " :" << nb_faces3D << finl;
151 int nb_som_fac3D=les_faces3D.dimension(1);
152 Cerr << "number of nodes per faces in 3D :" << nb_som_fac3D << finl;
153
154// Construction de renum_som3D2D
155 ArrOfInt renum_som3D2D(nb_som3D);
156 for (int i=0; i<nb_som3D ; i++)
157 renum_som3D2D[i]=-1;
158 int compteur=0;
159 for (int i=0; i<nb_faces3D; i++)
160 for (int j=0; j<nb_som_fac3D; j++)
161 {
162 int& tmp=renum_som3D2D[les_faces3D(i,j)];
163 if(tmp==-1)
164 tmp=compteur++;
165 }
166
167 int nb_som2D=compteur;
168// Construction de renum_som2D3D
169 ArrOfInt renum_som2D3D(nb_som2D);
170
171 for (int i=0; i<nb_som3D ; i++)
172 {
173 int j;
174 if((j=renum_som3D2D[i])!=-1)
175 renum_som2D3D[j]=i;
176 }
177 coord_sommets2D.resize(nb_som2D,2);
178 if (coupe_==2) coord_sommets2D.resize(nb_som2D,3);
179 Zone tmpz;
180 Zone& zone2D=dom2D.add(tmpz);
181 Nom nom_zone("surface");
182 nom_zone+=dom2D.le_nom();
183 zone2D.nommer("surface");
184 if(zone3D.type_elem()->que_suis_je()=="Hexaedre")
185 if (test_axi)
186 zone2D.typer("Rectangle_2D_axi");
187 else
188 zone2D.typer("Rectangle");
189 else if(zone3D.type_elem()->que_suis_je()=="Tetraedre")
190 {
191 zone2D.typer("Triangle");
192 }
193 else if(zone3D.type_elem()->que_suis_je()=="Hexaedre_VEF")
194 {
195 zone2D.typer("Quadrangle");
196 }
197 if (coupe_!=1)
198 {
199// Algorithme general
200// La frontiere doit etre plane (non verifie), parallele a un axe ou pas
201// Attention: les coordonnees du domaine 2D sont dans le nouveau repere
202 double xb,yb,zb;
203 double xc,yc,zc;
204 double ux,uy,uz;
205 double norm;
206 double scalI,scalJ;
207
208 xa=0.;
209 ya=0.;
210 za=0.;
211 xb=0.;
212 yb=0.;
213 zb=0.;
214 xc=0.;
215 yc=0.;
216 zc=0;
217 ux=0.;
218 uy=0.;
219 uz=0.;
220 norm=0.;
221 scalI=0.;
222 scalJ=0.;
223
224// on determine un repere orthonorme (A;IJK) sur le bord a extraire
225 xa = coord_sommets3D(les_faces3D(0,0),0);
226 ya = coord_sommets3D(les_faces3D(0,0),1);
227 za = coord_sommets3D(les_faces3D(0,0),2);
228
229 xb = coord_sommets3D(les_faces3D(0,1),0);
230 yb = coord_sommets3D(les_faces3D(0,1),1);
231 zb = coord_sommets3D(les_faces3D(0,1),2);
232
233 xc = coord_sommets3D(les_faces3D(0,2),0);
234 yc = coord_sommets3D(les_faces3D(0,2),1);
235 zc = coord_sommets3D(les_faces3D(0,2),2);
236
237// calcul de I = AB/norme(AB)
238 Ix=xb-xa;
239 Iy=yb-ya;
240 Iz=zb-za;
241 norm = sqrt(Ix*Ix+Iy*Iy+Iz*Iz);
242 Ix/=norm;
243 Iy/=norm;
244 Iz/=norm;
245
246// calcul de K = AB^AC/norme(AB^AC)
247 ux=xc-xa;
248 uy=yc-ya;
249 uz=zc-za;
250 norm = sqrt(ux*ux+uy*uy+uz*uz);
251 ux/=norm;
252 uy/=norm;
253 uz/=norm;
254 Kx = Iy*uz-uy*Iz;
255 Ky = Iz*ux-Ix*uz;
256 Kz = Ix*uy-Iy*ux;
257 norm = sqrt(Kx*Kx+Ky*Ky+Kz*Kz);
258 Kx/=norm;
259 Ky/=norm;
260 Kz/=norm;
261
262// calcul de J = K^I
263 Jx = Ky*Iz-Iy*Kz;
264 Jy = Kz*Ix-Iz*Kx;
265 Jz = Kx*Iy-Ky*Ix;
266
267// produit scalaire OA.I et OA.J
268 scalI = xa*Ix+ya*Iy+za*Iz;
269 scalJ = xa*Jx+ya*Jy+za*Jz;
270
271// on calcule les coordonnees des points du bord dans le nouveau repere 2D (A;I;J)
272// pour cela changement de repere du repere 3D vers le repere (A;I;J)
273 for(int i=0; i<nb_som2D; i++)
274 {
275 double x = coord_sommets3D(renum_som2D3D[i],0);
276 double y = coord_sommets3D(renum_som2D3D[i],1);
277 double z = coord_sommets3D(renum_som2D3D[i],2);
278 if (coupe_==0)
279 {
280 coord_sommets2D(i,0) = -scalI + x*Ix + y*Iy + z*Iz;
281 coord_sommets2D(i,1) = -scalJ + x*Jx + y*Jy + z*Jz;
282 }
283 else
284 {
285 assert(coupe_==2);
286// on ne ramene pas en 2D
287 coord_sommets2D(i,0)=x;
288 coord_sommets2D(i,1)=y;
289 coord_sommets2D(i,2)=z;
290 }
291// Cout << "coord 1 = " << coord_sommets2D(i,0) << finl;
292// Cout << "coord 2 = " << coord_sommets2D(i,1) << finl;
293// Cout << "coord 3 = " << -xa*Kx-ya*Ky-za*Kz+x*Kx + y*Ky + z*Kz << finl;
294 }
295 }
296 else
297 {
298// Algorithme de coupe utilise par Xprepro en VDF
299// Attention: limite a une frontiere plane parallele a un axe
300// Les coordoonnees du domaine 2D sont conservees
301 int test=0;
302 double precision=DMAXFLOAT;
303 for (test=2; test>-1; test--)
304 {
305 double x1,x2,x3;
306 x1=coord_sommets3D(les_faces3D(0,0),test);
307 x2=coord_sommets3D(les_faces3D(0,1),test);
308 x3=coord_sommets3D(les_faces3D(0,2),test);
309 double tmp=std::fabs(x1-x2);
310 precision=(tmp<precision && tmp>0?tmp:precision);
311 tmp=std::fabs(x3-x2);
312 precision=(tmp<precision && tmp>0?tmp:precision);
313 if (est_egal(x1,x2)&&est_egal(x2,x3)) break;
314 }
315 if (test==-1)
316 {
317 Cerr << "Error into TroisDto2D.cpp..." << finl;
318 Cerr << "May be the " << dom2D.le_nom() << " boundary domain is not plane enough" << finl;
319 Cerr << "Try to use: PrecisionGeom " << 100*precision << " in your data file" << finl;
320 Cerr << "Or contact TRUST support." << finl;
321 exit();
322 }
323 IntVect dir(2);
324 int kk=0;
325 for (int jj=0; jj<3; jj++)
326 {
327 if (test!=jj)
328 {
329 dir[kk]=jj;
330 kk++;
331 }
332 }
333 for(int i=0; i<nb_som2D; i++)
334 for(int j=0; j<2; j++)
335 coord_sommets2D(i,j)=coord_sommets3D(renum_som2D3D[i],dir[j]);
336 }
337
338 zone2D.associer_domaine(dom2D);
339
340 IntTab& les_elems2D=zone2D.les_elems();
341 les_elems2D=les_faces3D;
342 for (int i=0; i< nb_faces3D; i++)
343 for (int j=0; j< nb_som_fac3D; j++)
344 {
345 les_elems2D(i,j)=renum_som3D2D[les_elems2D(i,j)];
346 }
347
348 dimension=2;
349 {
350// On recupere les bords :
351
352 CONST_LIST_CURSEUR(Bord) curseur(zone3D.faces_bord());
353 while(curseur)
354 {
355 const Frontiere& front=curseur.valeur();
356 const Faces& faces3D_front=front.faces();
357 if ( (front.le_nom()!=nom_bord))
358 {
359 Bord tmp;
360 Bord& bord2D=zone2D.faces_bord().add(tmp);
361 bord2D.nommer(front.le_nom());
362 bord2D.associer_zone(zone2D);
363 if (test_axi)
364 bord2D.typer_faces("QUADRILATERE_2D_AXI");
365 else
366 bord2D.typer_faces("segment_2D");
367// creer les faces de bord 2D ici!
368 int compteur2=0;
369 const IntTab& faces_sommets=faces3D_front.les_sommets();
370 int nb_faces=faces_sommets.dimension(0);
371 IntTab& aretes=bord2D.les_sommets_des_faces();
372 IntTab& faces_voisins=bord2D.faces().voisins();
373
374 aretes.resize(nb_faces, 2);
375// Boucle sur les faces du domaine 2D
376 int doublons=0;
377 for(int face=0; face < nb_faces; face++)
378 {
379 int ok=0;
380 int tmpbis;
381 for(int i=0; i<nb_som_fac3D; i++)
382 {
383 tmpbis=renum_som3D2D[faces_sommets(face,i)];
384 if (tmpbis!=-1)
385 {
386 ok++;
387 if(ok==1)
388 aretes(compteur2,0)=tmpbis;
389 if(ok==2)
390 {
391 aretes(compteur2,1)=tmpbis;
392 compteur2++;
393 }
394 if(ok==3)
395 {
396// Si 3 sommets de la face appartienne a la frontiere
397// il y'a un probleme, on ajoute les 3 aretes et on cherchera
398// les doublons qu'il faudra eliminer...
399 doublons=1;
400 aretes(compteur2,0)=aretes(compteur2-1,1);
401 aretes(compteur2,1)=tmpbis;
402 compteur2++;
403 aretes(compteur2,0)=aretes(compteur2-1,1);
404 aretes(compteur2,1)=aretes(compteur2-2,0);
405 compteur2++;
406 }
407 }
408 }
409 }
410 if (doublons)
411 {
412// Recherche des doublons (algo en n^2)
413 for (int i=0; i<compteur2; i++)
414 {
415 int& S10=aretes(i,0);
416 int& S11=aretes(i,1);
417 for (int j=i+1; j<compteur2; j++)
418 {
419 int& S20=aretes(j,0);
420 int& S21=aretes(j,1);
421 if ( (S10==S20 && S11==S21) || (S10==S21 && S11==S20) )
422 {
423 S10=-1;
424 S11=-1;
425 S20=-1;
426 S21=-1;
427 }
428 }
429 }
430// On supprime les doublons
431 for (int i=0; i<compteur2; i++)
432 if (aretes(i,0)==-1)
433 {
434 for (int j=i; j<compteur2-1; j++)
435 {
436 aretes(j,0)=aretes(j+1,0);
437 aretes(j,1)=aretes(j+1,1);
438 }
439 compteur2--;
440 }
441 }
442 aretes.resize(compteur2, 2);
443 faces_voisins.resize(compteur2,2);
444 faces_voisins=-1;
445 if (compteur2==0)
446 {
447 Cerr<<"boundary to eliminate " << front.le_nom()<<finl;
448 zone2D.faces_bord().suppr(bord2D);
449 }
450 else
451 {
452 Cerr<<"boundary maintained " << front.le_nom()<<finl;
453 }
454 }
455 ++curseur;
456 }
457 }
458
459 {
460// On recupere les raccords :
461
462 CONST_LIST_CURSEUR(Raccord) curseur(zone3D.faces_raccord());;;
463 while(curseur)
464 {
465// while(curseur)
466 const Frontiere& front=curseur->valeur();
467 const Faces& faces3Dfront=front.faces();
468 if ( (front.le_nom()!=nom_bord))
469 {
470 Raccord tmpbis;
471 Raccord& bord2D=zone2D.faces_raccord().add(tmpbis);
472 bord2D.typer(curseur->valeur().le_type());
473 bord2D->nommer(front.le_nom());
474 bord2D->associer_zone(zone2D);
475 if (test_axi)
476 bord2D->typer_faces("QUADRILATERE_2D_AXI");
477 else
478 bord2D->typer_faces("segment_2D");
479// creer les faces de bord 2D ici!
480 int compteur2=0;
481 const IntTab& faces_sommets=faces3Dfront.les_sommets();
482 int nb_faces=faces_sommets.dimension(0);
483 IntTab& aretes=bord2D->les_sommets_des_faces();
484 IntTab& faces_voisins=bord2D->faces().voisins();
485
486 aretes.resize(nb_faces, 2);
487 for(int face=0; face < nb_faces; face++)
488 {
489 int ok=0;
490 int tmp;
491 for(int i=0; i<nb_som_fac3D; i++)
492 {
493 tmp=renum_som3D2D[faces_sommets(face,i)];
494 if (tmp!=-1)
495 {
496 ok++;
497 if(ok==1)
498 aretes(compteur2,0)=tmp;
499 else if(ok==2)
500 {
501 aretes(compteur2,1)=tmp;
502 compteur2++;
503 i=nb_som_fac3D;
504 }
505 }
506 }
507 }
508 aretes.resize(compteur2, 2);
509 faces_voisins.resize(compteur2,2);
510 faces_voisins=-1;
511 if (compteur2==0)
512 {
513 Cerr<<"connector to eliminate " << front.le_nom()<<finl;
514 zone2D.faces_raccord().suppr(bord2D);
515 }
516 else
517 {
518 Cerr<<"connector maintained " << front.le_nom()<<finl;
519 }
520 }
521 ++curseur;
522 }
523 }
524
525 if(zone3D.type_elem()->que_suis_je()!="Hexaedre_VEF")
526 if (coupe_!=2)
527 dom2D.reordonner();
528 dimension=3;
529}
