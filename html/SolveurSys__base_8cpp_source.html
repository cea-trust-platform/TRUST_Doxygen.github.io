59void SolveurSys_base::save_matrice_secmem_conditionnel(const Matrice_Base& la_matrice, const DoubleVect& secmem, const DoubleVect& solution, int binaire2 )
60{
61 int binaire=binaire2;
62 if (save_matrice_)
63 {
64// on simule l'ecriture d'une Matrice et non d'une Matrice_Base
65// pour aider a la relecture
66 {
67 EcrFicCollecte sortie;
68 sortie.set_bin(binaire);
69 sortie.ouvrir("Matrice.sa");
70 sortie<<la_matrice.que_suis_je()<<finl;
71 sortie<<la_matrice;
72 }
73// on sauvegarde le Secmem avec les espaces virtuels !!!!!
74 {
75 EcrFicCollecte sortie;
76 sortie.set_bin(binaire);
77 sortie.ouvrir("Secmem.sa");
78 MD_Vector_tools::dump_vector_with_md(secmem, sortie);
79 }
80// on sauvegarde la solution avec les espaces virtuels !!!!!
81 {
82 EcrFicCollecte sortie;
83 sortie.set_bin(binaire);
84 sortie.ouvrir("Solution.sa");
85 MD_Vector_tools::dump_vector_with_md(solution, sortie);
86 }
87 Cout <<"Saving of the matrix, secmem and solution ended."<<finl;
88 }
89}
