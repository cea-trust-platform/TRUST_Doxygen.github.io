56// Description
57// Provides afield with a name, a time interval, components, and no field ownership.
58// The geometry includes all nodes of the domain and all faces of the boundary.
59// WEC : optimization possible here
60void Ch_front_input::getTemplate(TrioField& afield) const
61{
62
63 afield.clear();
64 afield.setName(le_nom().getChar());
65
66 afield._time1=mon_pb->presentTime();
67 afield._time2=mon_pb->futureTime();
68
69 afield._nb_field_components=nb_comp();
70
71 int org=1;
72 if (faces_.dimension(1)==4) org=0;
73 org=0;
74// en mettant org a o1 on revient a l'ancien codage
75
76
77 const DoubleTab& sommets=org==1?mon_pb->domaine().les_sommets():sommets_;
78 const IntTab& faces=org==1?frontiere_dis().frontiere().faces().les_sommets():faces_;
79
80 assert(sommets.nb_dim()==2);
81 assert(faces.nb_dim()==2);
82
83 afield._space_dim=sommets.dimension(1);
84 afield._mesh_dim=afield._space_dim-1;
85 afield._nbnodes=sommets.dimension(0);
86 afield._nodes_per_elem=::mp_max(faces.dimension(1));
87 afield._nb_elems=faces.dimension(0);
88
89// Includes all the nodes, even those not used in connectivity
90// order is x y z x y z x y z
91 affecte_double_avec_doubletab(&afield._coords,sommets);
92
93 affecte_int_avec_inttab(&afield._connectivity,faces);
94}

96// In the case of a mobile mesh (ALE frame) we need to call this function after each displacement of the mesh
97void Ch_front_input::buildSommetsFaces() const
98{
99 const DoubleTab& sommets_org=(mon_pb->domaine().les_sommets());
100 DoubleTab& sommets=sommets_;
101 sommets.resize(sommets_org.dimension_tot(0),sommets_org.dimension(1));
102 sommets.set_smart_resize(1);
103
104 IntTab& faces=faces_;
105 const IntTab& faces_org=(frontiere_dis().frontiere().faces().les_sommets());
106 faces=faces_org;
107 int ntot=0;
108 ArrOfInt marqueur(sommets_org.dimension(0));
109 marqueur=-1;
110 for (int f=0; f<faces.dimension(0); f++)
111 {
112 for (int s=0; s<faces.dimension(1); s++)
113 {
114 int som=faces(f,s);
115 if (som >= 0 && marqueur[som]==-1)
116 {
117
118 marqueur[som]=ntot;
119 for (int dir=0; dir<dimension; dir++)
120 {
121 sommets(ntot,dir)=sommets_org(som,dir);
122 }
123 ntot++;
124 }
125 }
126 }
127 sommets.resize(ntot,dimension);
128
129
130// on refait les faces
131 for (int f=0; f<faces.dimension(0); f++)
132 for (int s=0; s<faces.dimension(1); s++)
133 faces(f,s) = faces_org(f, s) >= 0 ? marqueur[faces_org(f,s)] : -1;
134
135}
