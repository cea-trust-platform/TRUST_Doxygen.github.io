204DoubleTab& Op_Diff_EF::ajouter_new(const DoubleTab& tab_inconnue, DoubleTab& resu) const
205{
206 Cerr<<"NEW"<<finl;
207 remplir_nu(nu_);
208// const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
209// const Zone_EF& zone_EF = la_zone_EF.valeur();
210
211
212 const int N = resu.line_size();
213 ArrOfInt marqueur_neuman;
214 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
215 if(N > 1)
216 remplir_marqueur_sommet_neumann( marqueur_neuman,zone_ef,la_zcl_EF.valeur(),transpose_partout_ );
217
218 const DoubleVect& volumes= zone_ef.volumes();
219
220 const DoubleTab& bij=zone_ef.Bij();
221 const DoubleTab& bij_thilde=zone_ef.Bij_thilde();
222 int nb_elem_tot=zone_ef.zone().nb_elem_tot();
223 int nb_som_elem=zone_ef.zone().nb_som_elem();
224 const IntTab& elems=zone_ef.zone().les_elems() ;
225
226 for (int elem=0; elem<nb_elem_tot; elem++)
227 if (elem_contribue(elem))
228 {
229 double pond=1./volumes(elem);
230 assert(N == dimension || N == 1);
231
232 for (int i1=0; i1<nb_som_elem; i1++)
233 {
234 int glob=elems(elem,i1);
235 int transpose = (marqueur_neuman[glob] == 1 || N == 1) ? 0 : transpose_;
236 for (int i2=0; i2<nb_som_elem; i2++)
237 {
238 int glob2=elems(elem,i2);
239 for (int n = 0; n < N; n++)
240 for (int d = 0; d < dimension; d++)
241 resu(glob, n) -= bij(elem, i1, d) * (bij_thilde(elem, i2, d) * tab_inconnue(glob2, n) + transpose * bij_thilde(elem, i2, n) * tab_inconnue(glob2, d)) * nu_(elem) * pond;
242 }
243 }
244 }
245
246// on ajoute la contribution des bords
247 ajouter_bords(tab_inconnue,resu);
248 return resu;
249}

262inline double& coeff_opt(Matrice_Morse& matrice,int i, int j)
263{
264 const IntVect& tab1_=matrice.get_tab1();
265 const IntVect& tab2_=matrice.get_tab2();
266 int k1=tab1_[i]-1;
267 int k2=tab1_[i+1]-1;
268 int k;
269 for (k=k1; k<k2; k++)
270 if (tab2_[k]-1 == j) return(matrice.get_set_coeff()(k));
271 Cerr << "i ou j ne conviennent pas " << finl;
272 Cerr << "i=" << i << finl;
273 Cerr << "j=" << j << finl;
274 Cerr << "n_lignes=" << matrice.nb_lignes() << finl;
275 Cerr << "n_colonnes=" << matrice.nb_colonnes() << finl;
276 assert( k<tab1_[i]-1);
277 Process::exit();
278 return coeff_opt(matrice,i,j);
279}

282void Op_Diff_EF::ajouter_contribution(const DoubleTab& transporte, Matrice_Morse& matrice ) const
283{
284 if (1)
285 if ((equation().nombre_d_operateurs()>1)&&sub_type(Op_Conv_EF,equation().operateur(1).l_op_base()))
286 ref_cast(Op_Conv_EF,equation().operateur(1).l_op_base()).ajouter_contribution_a_la_diffusion(transporte,matrice);
287
288 if (nouvelle_expression_==1)
289 {
290 ajouter_contribution_new(transporte,matrice);
291 }
292// On remplit le tableau nu car l'assemblage d'une
293// matrice avec ajouter_contribution peut se faire
294// avant le premier pas de temps
295 remplir_nu(nu_);
296
297 const int N = transporte.line_size();
298 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
299 const DoubleVect& volumes_thilde= zone_ef.volumes_thilde();
300 const DoubleVect& volumes= zone_ef.volumes();
301
302 const DoubleTab& bij=zone_ef.Bij();
303 int nb_elem_tot=zone_ef.zone().nb_elem_tot();
304 int nb_som_elem=zone_ef.zone().nb_som_elem();
305 const IntTab& elems=zone_ef.zone().les_elems() ;
306 int nb_som=zone_ef.zone().nb_som();
307
308 ArrOfInt marqueur_neuman;
309 remplir_marqueur_sommet_neumann( marqueur_neuman,zone_ef,la_zcl_EF.valeur(),transpose_partout_ );
310 for (int elem=0; elem<nb_elem_tot; elem++)
311 if (elem_contribue(elem))
312 {
313 double pond=volumes_thilde(elem)/volumes(elem)/volumes(elem);
314
315 for (int i1=0; i1<nb_som_elem; i1++)
316 {
317 int glob=elems(elem,i1);
318
319 int transpose = (marqueur_neuman[glob] == 1 || N == 1) ? 0 : transpose_;
320 if (glob<nb_som)
321 for (int i2=0; i2<nb_som_elem; i2++)
322 {
323 int glob2=elems(elem,i2);
324 double cb=0;
325 for (int b=0; b<dimension; b++)
326 cb+=bij(elem,i1,b)*bij(elem,i2,b);
327 for (int n = 0; n < N; n++)
328 {
329 matrice_coef(glob * N + n, glob2 * N + n) += cb * nu_(elem) * pond;
330 if (transpose)
331 for (int d = 0; d < dimension; d++)
332 matrice_coef(glob * N + n, glob2 * N + d) += bij(elem, i1, d) * bij(elem, i2, n) * nu_(elem) * pond;
333 }
334 }
335 }
336 }
337 if (N == 1) ajouter_contributions_bords(matrice);

400void Op_Diff_EF::ajouter_bords(const DoubleTab& tab_inconnue,DoubleTab& resu, int contrib_interne ) const
401{
402// a mettre dans calculer_flux_bord....
403
404 const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
405 const Zone_EF& zone_EF = la_zone_EF.valeur();
406 flux_bords_=0.;
407// const DoubleTab& tab_inconnue=equation().inconnue().valeurs();
408// on parcourt toutes les faces de bord et on calcule lambda*gradT
409 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
410 const IntTab& face_voisins=zone_ef.face_voisins();
411 const DoubleTab& bij=zone_ef.Bij();
412 int nb_som_elem=zone_ef.zone().nb_som_elem();
413 const IntTab& elems=zone_ef.zone().les_elems() ;
414 const DoubleTab& face_normales=zone_ef.face_normales();
415 const DoubleVect& volumes_thilde= zone_ef.volumes_thilde();
416 const DoubleVect& volumes= zone_ef.volumes();
417
418 const IntTab& face_sommets=zone_ef.face_sommets();
419 int nb_som_face=zone_ef.nb_som_face();
420
421 const int N = resu.line_size();
422
423 if (N > 1)
424 {
425// Cerr<<__PRETTY_FUNCTION__<<" non code pour les vecteurs"<<finl;
426 modifier_flux(*this);
427 return;
428 }
429 int premiere_face_int=zone_ef.premiere_face_int();
430 for (int face=0; face<premiere_face_int; face++)
431 {
432 int elem=face_voisins(face,0);
433 if (elem==-1) face_voisins(face,1);
434
435 double pond= volumes_thilde(elem)/volumes(elem)/volumes(elem);
436
437 for (int i1=0; i1<nb_som_elem; i1++)
438 {
439
440 int glob2=elems(elem,i1);
441 {
442
443 for (int a=0; a<dimension; a++)
444 {
445 flux_bords_(face,0)+=face_normales(face,a)*bij(elem,i1,a)*tab_inconnue(glob2)*nu_(elem)*pond;
446 }
447 }
448 }
449
450 }
451
452// Neumann :
453 int n_bord;
454 int nb_bords=zone_Cl_EF.nb_cond_lim();
455
456 for (n_bord=0; n_bord<nb_bords; n_bord++)
457 {
458 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
459 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
460 int ndeb = le_bord.num_premiere_face();
461 int nfin = ndeb + le_bord.nb_faces();
462
463 if (sub_type(Neumann_paroi,la_cl.valeur()))
464 {
465 const Neumann_paroi& la_cl_paroi = ref_cast(Neumann_paroi, la_cl.valeur());
466 for (int face=ndeb; face<nfin; face++)
467 {
468
469 double flux=la_cl_paroi.flux_impose(face-ndeb)*zone_EF.surface(face);
470 for (int i1=0; i1<nb_som_face; i1++)
471 {
472 int glob2=face_sommets(face,i1);
473 {
474
475 resu[glob2] += flux/nb_som_face;
476 }
477 flux_bords_(face,0) = flux;
478 }
479 }
480 }
481 else if (sub_type(Echange_couplage_thermique, la_cl.valeur()))
482 {
483 const Echange_couplage_thermique& la_cl_paroi = ref_cast(Echange_couplage_thermique, la_cl.valeur());
484 for (int face=ndeb; face<nfin; face++)
485 {
486
487 double h=la_cl_paroi.h_imp(face-ndeb);
488 double Text=la_cl_paroi.T_ext(face-ndeb);
489 double phiext=la_cl_paroi.flux_exterieur_impose(face-ndeb);
490
491 double tm=0;
492 if (contrib_interne)
493 {
494 for (int i1=0; i1<nb_som_face; i1++)
495 {
496 int glob2=face_sommets(face,i1);
497 tm+=tab_inconnue(glob2);
498 }
499 tm/=nb_som_face;
500 }
501 double flux=(phiext+h*(Text-tm))*zone_EF.surface(face);
502 flux_bords_(face,0) = flux;
503 flux/=nb_som_face;
504 for (int i1=0; i1<nb_som_face; i1++)
505 {
506 int glob2=face_sommets(face,i1);
507 resu[glob2] += flux;
508 }
509 }
510 }
511 else if (sub_type(Echange_interne_global_parfait, la_cl.valeur()))
512 {
513 if (contrib_interne)
514 {
515 const Echange_interne_global_parfait& la_cl_paroi = ref_cast(Echange_interne_global_parfait, la_cl.valeur());
516 const Champ_front_calc_interne& Text = ref_cast(Champ_front_calc_interne, la_cl_paroi.T_ext().valeur());
517 const IntTab& fmap = Text.face_map();
518 std::vector<bool> hit(nfin-ndeb);
519 std::fill(hit.begin(), hit.end(), false);
520 for (int face=ndeb; face<nfin; face++)
521 {
522 int opp_face = fmap(face-ndeb)+ndeb; //!< face on the other side of the inner wall:
523
524// STRONG assumption : 1D, only one node per face:
525 int som=face_sommets(face,0);
526 int som_opp=face_sommets(opp_face,0);
527 if (!hit[face-ndeb]) //!< first time we encounter one of the face of the pair
528 {
529// This where we write the heat equation skipping duplicated face: -A.T(i-1) + (A*T(i)+B.T(i)) - B.T(i+2)
530// By default, we just have -A.T(i-1) + A*T(i)
531// Recompute the extra bit of flux coming from the other side:
532// Get opposite element
533 int elem1 = zone_EF.face_voisins(opp_face, 0);
534 int elem_opp = (elem1 != -1) ? elem1 : zone_EF.face_voisins(opp_face, 1);
535// Other face of the opposite element (i.e. face "after" opp_face)
536 int f1 = zone_EF.elem_faces(elem_opp, 0);
537 int face_plus_2 = f1 != opp_face ? f1 : zone_EF.elem_faces(elem_opp, 1); //!< 2 faces per elem max - 1D
538 int som_p2 = face_sommets(face_plus_2, 0);
539
540 double pond = volumes_thilde(elem_opp)/volumes(elem_opp)/volumes(elem_opp); //!< taken from ajouter() ...
541 double B = nu_(elem_opp)*pond;
542 resu[som] -= B*(tab_inconnue[som]-tab_inconnue[som_p2]);
543 }
544 else
545 {

655void Op_Diff_EF::ajouter_contributions_bords(Matrice_Morse& matrice ) const
656{
657 const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
658 const Zone_EF& zone_EF = la_zone_EF.valeur();
659 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
660
661 const IntTab& face_sommets=zone_ef.face_sommets();
662 int nb_som_face=zone_ef.nb_som_face();
663
664 int N = equation().inconnue().valeurs().line_size();
665
666 if (N > 1)
667 {
668// Cerr<<__PRETTY_FUNCTION__<<" non code pour les vecteurs"<<finl;
669 throw;
670 return;
671 }
672
673// Neumann :
674 int n_bord;
675 int nb_bords=zone_Cl_EF.nb_cond_lim();
676
677 for (n_bord=0; n_bord<nb_bords; n_bord++)
678 {
679 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
680 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
681 int ndeb = le_bord.num_premiere_face();
682 int nfin = ndeb + le_bord.nb_faces();
683
684 if (sub_type(Echange_couplage_thermique, la_cl.valeur()))
685 {
686// matrice.imprimer(Cout);
687 const Echange_couplage_thermique& la_cl_paroi = ref_cast(Echange_couplage_thermique, la_cl.valeur());
688 for (int face=ndeb; face<nfin; face++)
689 {
690
691 double h=la_cl_paroi.h_imp(face-ndeb);
692 double dphi_dT=la_cl_paroi.derivee_flux_exterieur_imposee(face-ndeb);
693
694 double tm=1./(nb_som_face*nb_som_face);
695 double flux=(dphi_dT+h)*zone_EF.surface(face)*tm;
696
697 for (int i1=0; i1<nb_som_face; i1++)
698 {
699 int glob2=face_sommets(face,i1);
700 for (int j1=0; j1<nb_som_face; j1++)
701 {
702 int glob1=face_sommets(face,j1);
703 matrice.coef(glob1,glob2) += flux;
704 }
705 }
706 }
707// matrice.imprimer(Cout);exit();
708 }
709 else if (sub_type(Echange_interne_global_parfait, la_cl.valeur()))
710 {
711 const Echange_interne_global_parfait& la_cl_paroi = ref_cast(Echange_interne_global_parfait, la_cl.valeur());
712 const Champ_front_calc_interne& Text = ref_cast(Champ_front_calc_interne, la_cl_paroi.T_ext().valeur());
713 const IntTab& fmap = Text.face_map();
714 std::vector<bool> hit(nfin-ndeb);
715 std::fill(hit.begin(), hit.end(), false);
716 for (int face=ndeb; face<nfin; face++)
717 {
718// face on the other side of the inner wall:
719 int opp_face = fmap(face-ndeb)+ndeb;
720// element attached to it
721 int elem1 = zone_EF.face_voisins(opp_face, 0);
722 int elem_opp = (elem1 != -1) ? elem1 : zone_EF.face_voisins(opp_face, 1);
723// other face of the opposite element (i.e. face "after" opp_face)
724 int f1 = zone_EF.elem_faces(elem_opp, 0);
725 int face_plus_2 = f1 != opp_face ? f1 : zone_EF.elem_faces(elem_opp, 1); //!< 2 faces per elem max - 1D other face of the current elem
726//
727 int elem = zone_EF.face_voisins(face, 0);
728 f1 = zone_EF.elem_faces(elem, 0);
729 int face_min_1 = f1 != face ? f1 : zone_EF.elem_faces(elem, 1);
730
731 for (int i1=0; i1<nb_som_face; i1++)
732 {
733 int som=face_sommets(face,i1);
734 int som_opp=face_sommets(opp_face,i1);
735 if (!hit[face-ndeb]) //!< first time we encounter one of the face of the pair
736 {
737 for (int j1=0; j1<nb_som_face; j1++)
738 {
739// heat equation skipping duplicated face: -T(i-1) + 2*T(i) - T(i+2)
740 int som_p2=face_sommets(face_plus_2,j1);
741 matrice.coef(som,som_p2) -= matrice.coef(som_opp, som_opp);
742 matrice.coef(som, som) += matrice.coef(som_opp, som_opp);
743 }
744 hit[face-ndeb] = true;
745 hit[opp_face-ndeb] = true;
746 }
747 else //!< we have already handled the opposite side of the wall
748 for (int j1=0; j1<nb_som_face; j1++)
749 {
750// Dirichlet T(face) = T(opposite_face) and cancelling right term in the tri-band
751 int som_m1=face_sommets(face_min_1, j1);
752 matrice.coef(som,som_opp) = -matrice.coef(som,som);
753 matrice.coef(som,som_m1) = 0;
754 }
755 }
756 }
757// matrice.imprimer(Cout);
758 }
759 else if (sub_type(Echange_interne_global_impose, la_cl.valeur()))
760 {
761 const Echange_interne_global_impose& la_cl_paroi = ref_cast(Echange_interne_global_impose, la_cl.valeur());
762 const Champ_front_calc_interne& Text = ref_cast(Champ_front_calc_interne, la_cl_paroi.T_ext().valeur());
763 const IntTab& fmap = Text.face_map();
764 const DoubleVect& surface_gap = la_cl_paroi.surface_gap();
765 for (int face=ndeb; face<nfin; face++)
766 {
767 double h=la_cl_paroi.h_imp(face-ndeb);
768 double tm=1./(nb_som_face*nb_som_face);
769// double flux=h*zone_EF.surface(face)*tm;
770 double flux=h*surface_gap(face-ndeb)*tm;
771 int opp_face = fmap(face-ndeb)+ndeb;
772
773 for (int i1=0; i1<nb_som_face; i1++)
774 {
775 int glob2=face_sommets(face,i1);
776 int glob3=face_sommets(opp_face,i1);
777 for (int j1=0; j1<nb_som_face; j1++)
778 {
779 int glob1=face_sommets(face,j1);
780 matrice.coef(glob1,glob2) += flux;
781 matrice.coef(glob1,glob3) -= flux;
782 }
783 }
784 }
785// matrice.imprimer(Cout);
786 }
787 else if (sub_type(Echange_global_impose, la_cl.valeur()))
788 {
789// matrice.imprimer(Cout);
790 const Echange_global_impose& la_cl_paroi = ref_cast(Echange_global_impose, la_cl.valeur());
791 for (int face=ndeb; face<nfin; face++)
792 {
793 double h=la_cl_paroi.h_imp(face-ndeb);
794 double tm=1./(nb_som_face*nb_som_face);
795 double flux=h*zone_EF.surface(face)*tm;
796 for (int i1=0; i1<nb_som_face; i1++)
797 {
798 int glob2=face_sommets(face,i1);
799 for (int j1=0; j1<nb_som_face; j1++)
800 {
801 int glob1=face_sommets(face,j1);
802 matrice.coef(glob1,glob2) += flux;
803 }
804 }
805 }
806 }
807 }
