94void EDO_Pression_th_VEF::calculer_grad(const DoubleTab& inco, DoubleTab& grad)
95{
96 assert(0);
97 const IntTab& face_voisins = la_zone->face_voisins();
98 const DoubleTab& face_normales = la_zone->face_normales();
99 const DoubleVect& volumes_entrelaces = la_zone->volumes_entrelaces();
100 const DoubleVect& volumes_entrelaces_Cl = ref_cast(Zone_Cl_VEF,la_zone_Cl->valeur()).volumes_entrelaces_Cl();
101
102 double diff;
103 int face,comp1;
104
105// Gradient d'un champ scalaire localise aux centres des elements
106// On initialise grad a zero
107 grad = 0;
108 if (1==1)
109 {
110 int elem1,elem2;
111// On traite les faces des joints:
112 for (face=0; face<la_zone->nb_faces_joint(); face++)
113 {
114 elem1 = face_voisins(face,0);
115 if (elem1==-1)
116 elem1 = face_voisins(face,1);
117 diff = -inco[elem1];
118 for (comp1=0; comp1<dimension; comp1++)
119 grad(face,comp1) = diff*face_normales(face,comp1);
120 }
121
122// On traite les conditions limites
123// Seule la condition aux limites de type Periodicite modifie le gradient
124 for (int n_bord=0; n_bord<la_zone->nb_front_Cl(); n_bord++)
125 {
126 const Cond_lim& la_cl = la_zone_Cl->les_conditions_limites(n_bord);
127 if (sub_type(Periodique,la_cl.valeur()))
128 {
129// const Periodique& la_cl_perio = (Periodique&) la_cl.valeur();
130 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
131 int num1 = le_bord.num_premiere_face();
132 int num2 = num1 + le_bord.nb_faces();
133 for (face=num1; face<num2; face++)
134 {
135 elem1 = face_voisins(face,0);
136 elem2 = face_voisins(face,1);
137 diff = inco[elem2] - inco[elem1];
138 for (int comp2=0; comp2<dimension; comp2++)
139 grad(face,comp2) = diff*face_normales(face,comp2);
140 }
141 }
142 }
143
144// On traite les faces internes
145// Cerr << "Faces internes." << finl;
146 for (face=la_zone->premiere_face_int(); face<la_zone->nb_faces(); face++)
147 {
148 elem1 = face_voisins(face,0);
149 elem2 = face_voisins(face,1);
150 diff = inco[elem2] - inco[elem1];
151 for (int comp=0; comp<dimension; comp++)
152 grad(face,comp) = diff*face_normales(face,comp);
153 }
154// On divise par le volume!!!
155 int premiere_fac_std=la_zone->premiere_face_std();
156 for (face=0; face<premiere_fac_std; face++)
157 {
158 if (volumes_entrelaces_Cl(face)!=0)
159 for (int comp=0; comp<dimension; comp++)
160 grad(face,comp) /=volumes_entrelaces_Cl(face);
161 else
162 for (int comp=0; comp<dimension; comp++)
163 grad(face,comp) =0.;
164 }
165 for (face=premiere_fac_std; face<la_zone->nb_faces(); face++)
166 {
167 for (int comp=0; comp<dimension; comp++)
168 grad(face,comp) /=volumes_entrelaces(face);
169 }
170 }
171 else
172 {
173 if (Objet_U::dimension==2)
174 {
175 int elem, nb_som = la_zone->zone().nb_som();
176 int som,nsom, nse = la_zone->zone().nb_som_elem();
177 const IntTab& som_elem = la_zone->zone().les_elems();
178 int s0,s1,elem0,elem1;
179 const IntTab& face_sommets = la_zone->face_sommets();
180 const DoubleTab& coord_sommets = la_zone->zone().domaine().coord_sommets();
181 const DoubleTab& xp = la_zone->xp();
182// const DoubleTab& xv = la_zone->xv();
183 DoubleTab incosom(nb_som);
184 DoubleTab volsom(nb_som);
185 incosom = 0;
186 volsom = 0;
187 double v,inc;
188// calcul de P aux sommets
189 for (elem=0 ; elem<la_zone->nb_elem() ; elem++)
190 {
191 v = la_zone->volumes(elem);
192 inc = inco[elem];
193 for (som=0 ; som<nse ; som++)
194 {
195 nsom = som_elem(elem,som);
196 incosom(nsom) += v*inc;
197 volsom(nsom) += v;
198 }
199 }
200 for (nsom=0 ; nsom<nb_som ; nsom++)
201 {
202 incosom(nsom) /= volsom(nsom);
203 }
204 double xs0,ys0,xs1,ys1,xe0,ye0,xe1,ye1, ds,de;
205// faces de bord
206 for (int n_bord=0; n_bord<la_zone->nb_front_Cl(); n_bord++)
207 {
208 const Cond_lim& la_cl = la_zone_Cl->les_conditions_limites(n_bord);
209
210 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
211 int num1 = le_bord.num_premiere_face();
212 int num2 = num1 + le_bord.nb_faces();
213 for (face=num1; face<num2; face++)
214 {
215 s0 = face_sommets(face,0);
216 s1 = face_sommets(face,1);
217 xs0 = coord_sommets(s0,0);
218 ys0 = coord_sommets(s0,1);
219 xs1 = coord_sommets(s1,0);
220 ys1 = coord_sommets(s1,1);
221 v = volumes_entrelaces(face);
222 ds = (incosom(s1) - incosom(s0))/((xs1-xs0)*(xs1-xs0) + (ys1-ys0)*(ys1-ys0));
223 grad(face,0) = v* (ds*(xs1-xs0));
224 grad(face,1) = v* (ds*(ys1-ys0));
225 }
226 }
227
228// faces internes
229 for (face=la_zone->premiere_face_int() ; face<la_zone->nb_faces(); face++)
230 {
231 s0 = face_sommets(face,0);
232 s1 = face_sommets(face,1);
233 elem0 = face_voisins(face,0);
234 elem1 = face_voisins(face,1);
235 xs0 = coord_sommets(s0,0);
236 ys0 = coord_sommets(s0,1);
237 xs1 = coord_sommets(s1,0);
238 ys1 = coord_sommets(s1,1);
239 xe0 = xp(elem0,0);
240 ye0 = xp(elem0,1);
241 xe1 = xp(elem1,0);
242 ye1 = xp(elem1,1);
243 v = volumes_entrelaces(face);
244 ds = (incosom(s1) - incosom(s0))/((xs1-xs0)*(xs1-xs0) + (ys1-ys0)*(ys1-ys0));
245 de = (inco(elem1) - inco(elem0))/((xe1-xe0)*(xe1-xe0) + (ye1-ye0)*(ye1-ye0));
246 grad(face,0) = v* (ds*(xs1-xs0) + de*(xe1-xe0));
247 grad(face,1) = v* (ds*(ys1-ys0) + de*(ye1-ye0));
248 }
249 }
250 else
251 {
252 Cerr<<"ATTENTION : gradient de l'inco mal calcule en 3D"<<finl;
253 }
254 }

257double EDO_Pression_th_VEF::masse_totale(double P,const DoubleTab& T)
258{
259 int nb_faces = la_zone->nb_faces();
260
261// assert(tab.dimension(0)==nb_faces);
262 const Loi_Etat_base& loi_ = ref_cast(Loi_Etat_base,le_fluide_->loi_etat().valeur());
263
264 DoubleVect tmp;
265 tmp.copy(T, ArrOfDouble::NOCOPY_NOINIT); //!< just copy the structure
266 if (!sub_type(Loi_Etat_Multi_GP_QC,loi_))
267 {
268 for (int i = 0; i < nb_faces; i++)
269 tmp[i] = loi_.calculer_masse_volumique(P, T[i]);
270 }
271 else
272 {
273 const Loi_Etat_Multi_GP_QC& loi_mel_GP = ref_cast(Loi_Etat_Multi_GP_QC,loi_);
274 const DoubleTab& Masse_mol_mel = loi_mel_GP.masse_molaire();
275 for (int i = 0; i < nb_faces; i++)
276 {
277 double r = 8.3143 / Masse_mol_mel[i];
278 tmp[i] = loi_mel_GP.calculer_masse_volumique(P, T[i], r);
279 }
280 }
281 const double M = Champ_P1NC::calculer_integrale_volumique(la_zone.valeur(), tmp, FAUX_EN_PERIO);
282 return M;
283}
