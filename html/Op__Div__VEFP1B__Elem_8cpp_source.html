50static int chercher_arete(int elem, int somi, int somj,
51 const IntTab& elem_aretes,
52 const IntTab& aretes_som)
53{
54 if(somi>somj)
55 {
56 int k=somi;
57 somi=somj;
58 somj=k;
59 }
60 for(int i_arete=0; i_arete<6; i_arete++)
61 {
62 int arete=elem_aretes(elem, i_arete);
63 int som1=aretes_som(arete,0);
64 if(somi==som1)
65 {
66 int som2=aretes_som(arete,1);
67 if(somj==som2)
68 return arete;
69 }
70 }
71 return -1;
72}

74static int verifier(const Op_Div_VEFP1B_Elem& op,
75 int& init,
76 const Zone_VEF_PreP1b& zone_VEF,
77 const DoubleTab& vit,
78 DoubleTab& div)
79{
80 init=1;
81 DoubleTab v(vit);
82 v=1;

84 r=0;
85 op.ajouter(v, r);
86 Cerr << "div(1,..,1) = " << r << finl;
87 Debog::verifier("div(1,..,1) =",r);
88 const DoubleTab& xv = zone_VEF.xv();

90 {
91 for (int i=0; i< v.dimension(0); i++)
92 v(i,1)=0;
93 }
94 r=0;
95 op.ajouter(v, r);
96 Cerr << "div(x,0) = " << r << finl;
97 Debog::verifier("div(x,0) =",r);
98 return 1;

101DoubleTab& Op_Div_VEFP1B_Elem::ajouter_elem(const DoubleTab& vit, DoubleTab& div) const
102{
103 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,la_zone_vef.valeur());
104 assert(zone_VEF.get_alphaE());
105 const Zone& zone = zone_VEF.zone();
106 const DoubleTab& face_normales = zone_VEF.face_normales();
107 const IntTab& elem_faces=zone_VEF.elem_faces();
108 const IntTab& face_voisins=zone_VEF.face_voisins();
109 int nfe=zone.nb_faces_elem();
110 int nb_elem=zone.nb_elem();
111 int elem,indice,face,comp;
112 double pscf,signe;
113
114 for(elem=0; elem<nb_elem; elem++)
115 {
116 pscf=0;
117 for(indice=0; indice<nfe; indice++)
118 {
119 face = elem_faces(elem,indice);
120 signe=1;
121 if(elem!=face_voisins(face,0))
122 signe=-1;
123 for(comp=0; comp<dimension; comp++)
124 pscf+=signe*vit(face, comp)*
125 face_normales(face,comp);
126 }
127 div(elem)+=pscf;
128 }
129 assert_invalide_items_non_calcules(div);
130 return div;
131}

133int find_cl_face(const Zone& zone, const int face)
134{
135 int i;
136 const int nb_cl = zone.nb_front_Cl();
137 for (i = 0; i < nb_cl; i++)
138 {
139 const Frontiere& fr = zone.frontiere(i);
140 const int debut = fr.num_premiere_face();
141 if (face >= debut && face < debut + fr.nb_faces())
142 return i;
143 const ArrOfInt& faces_virt = fr.get_faces_virt();
144 const int nb_faces_virt = faces_virt.size_array();
145 for (int j = 0; j < nb_faces_virt; j++)
146 if (face == faces_virt[j])
147 return i;
148 }
149 return -1;
150}

299DoubleTab& Op_Div_VEFP1B_Elem::ajouter_som(const DoubleTab& vit, DoubleTab& div, DoubleTab& flux_b) const
300{
301 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,la_zone_vef.valeur());
302 assert(zone_VEF.get_alphaS());
303 const Zone& zone = zone_VEF.zone();
304 const Domaine& dom=zone.domaine();
305 const DoubleTab& face_normales = zone_VEF.face_normales();
306 const IntTab& som_elem=zone.les_elems();
307 const IntTab& elem_faces=zone_VEF.elem_faces();
308 const IntTab& face_voisins=zone_VEF.face_voisins();
309 int nfe=zone.nb_faces_elem();
310 int nb_elem_tot=zone.nb_elem_tot();
311 int nps=zone_VEF.numero_premier_sommet();
312 int elem,indice,face,comp,som;
313 ArrOfDouble sigma(dimension);
314
315 int nb_face_diri=0;
316 double coeff_som=1./(dimension*(dimension+1));
317 int modif_traitement_diri=zone_VEF.get_modif_div_face_dirichlet();
318 double psc, signe;
319
320 const Zone_Cl_VEF& zcl=ref_cast(Zone_Cl_VEF,la_zcl_vef.valeur());
321 ArrOfInt indice_diri(dimension+1);
322 for(elem=0; elem<nb_elem_tot; elem++)
323 {
324 if (modif_traitement_diri)
325 coeff_som=calculer_coef_som(elem,nb_face_diri,indice_diri,zcl,zone_VEF);
326
327 sigma = 0;
328 for(indice=0; indice<nfe; indice++)
329 {
330 face = elem_faces(elem,indice);
331 for(comp=0; comp<dimension; comp++)
332 {
333 sigma[comp]+=vit(face, comp);
334 }
335 }
336// on retire la contribution des faces dirichlets
337 for (int fdiri=0; fdiri<nb_face_diri; fdiri++)
338 {
339 int indice2=indice_diri[fdiri];
340 face = elem_faces(elem,indice2);
341 for(comp=0; comp<dimension; comp++)
342 {
343 sigma[comp]-=vit(face, comp);
344 }
345 }
346 for(indice=0; indice<nfe; indice++)
347 {
348 som = nps+dom.get_renum_som_perio(som_elem(elem,indice));
349 face = elem_faces(elem,indice);
350 psc=0;
351 signe=1;
352 if(elem!=face_voisins(face,0))
353 signe=-1;
354 for(comp=0; comp<dimension; comp++)
355 {
356 psc+=sigma[comp]*face_normales(face,comp);
357 }
358 div(som)+=signe*coeff_som*psc;
359 nb_degres_liberte(som-nps)++;
360 }
361 }
362 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
363 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
364 const IntTab& face_sommets = zone_VEF.face_sommets();
365 int nb_bords =les_cl.size();
366 for (int n_bord=0; n_bord<nb_bords; n_bord++)
367 {
368 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
369 {
370 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
371 int nb_faces_bord = le_bord.nb_faces();
372 int nb_faces_bord_tot = le_bord.nb_faces_tot();
373 assert(le_bord.nb_faces()==zone_VEF.zone().frontiere(n_bord).nb_faces());
374 if (!sub_type(Periodique,la_cl.valeur()))
375 {
376 int libre=1;
377 if (sub_type(Dirichlet,la_cl.valeur())
378 ||sub_type(Dirichlet_homogene,la_cl.valeur())
379 ||sub_type(Dirichlet_entree_fluide,la_cl.valeur())
380 ||sub_type(Symetrie,la_cl.valeur()))
381 libre=0;
382// On boucle sur les faces de bord reelles et virtuelles
383 for (int ind_face=0; ind_face<nb_faces_bord_tot; ind_face++)
384 {
385 face = le_bord.num_face(ind_face);
386 double flux = 0. ;
387 for (comp=0; comp<dimension; comp++)
388 {
389 flux += vit(face,comp) * face_normales(face,comp) ;
390 }
391 if (ind_face<nb_faces_bord) flux_b(face,0) = flux;
392 flux*=1./dimension;
393 for(indice=0; indice<(nfe-1); indice++)
394 {
395 som=dom.get_renum_som_perio(face_sommets(face,indice));
396 div(nps+som)+=flux;
397 if (libre) nb_degres_liberte(som)++;
398 }
399 }
400 }
401 else
402 {
403 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
404// On boucle sur les faces de bord reelles et virtuelles
405 for (int ind_face=0; ind_face<nb_faces_bord_tot; ind_face++)
406 {
407 face = le_bord.num_face(ind_face);
408 int face_associee = le_bord.num_face(la_cl_perio.face_associee(ind_face));
409 double flux = 0. ;
410 double flux_perio = 0. ;
411 for (comp=0; comp<dimension; comp++)
412 {
413 flux += vit(face,comp) * face_normales(face,comp) ;
414 flux_perio += vit(face_associee,comp) * face_normales(face_associee,comp) ;
415 }
416 if (ind_face<(nb_faces_bord/2))
417 {
418 flux_b(face,0) = -flux;
419 flux_b(face_associee,0) = flux_perio;
420 }
421 }
422 }
423 }
424 }
425 return div;
426}

428DoubleTab& Op_Div_VEFP1B_Elem::ajouter_aretes(const DoubleTab& vit, DoubleTab& div) const
429{
430 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,la_zone_vef.valeur());
431 assert(zone_VEF.get_alphaA());
432 const Zone& zone = zone_VEF.zone();
433 const DoubleTab& face_normales = zone_VEF.face_normales();
434 const IntTab& som_elem=zone.les_elems();
435 const IntTab& elem_faces=zone_VEF.elem_faces();
436 const IntTab& face_voisins=zone_VEF.face_voisins();
437 const ArrOfInt& renum_arete_perio=zone_VEF.get_renum_arete_perio();
438 const ArrOfInt& ok_arete=zone_VEF.get_ok_arete();
439 int npa=zone_VEF.numero_premiere_arete();
440 int nb_elem_tot=zone.nb_elem_tot();
441 int elem,comp;
442 ArrOfDouble sigma(dimension);
443
444 const IntTab& aretes_som=zone_VEF.zone().aretes_som();
445 const IntTab& elem_aretes=zone_VEF.zone().elem_aretes();
446 for(elem=0; elem<nb_elem_tot; elem++)
447 {
448 for(int isom=0; isom<3; isom++)
449 {
450 int somi = som_elem(elem,isom);
451 int facei = elem_faces(elem,isom);
452 double signei=1.;
453 if (face_voisins(facei,0) != elem)
454 signei=-1.;
455 for(int jsom=isom+1; jsom<4; jsom++)
456 {
457 int somj = som_elem(elem,jsom);
458 int facej = elem_faces(elem,jsom);
459 double signej=1.;
460 if (face_voisins(facej,0) != elem)
461 signej=-1.;
462 int arete = renum_arete_perio[chercher_arete(elem, somi, somj,
463 elem_aretes, aretes_som)];
464 if(ok_arete[arete])
465 {
466 int niinij=0;
467 for(int ksom=0; ksom<4; ksom++)
468 {
469 if((ksom==isom)||(ksom==jsom))
470 {
471 if(niinij==ksom)
472 niinij++;
473 double psc=0;
474 for(comp=0; comp<3; comp++)
475 psc+=(signei*face_normales(facei,comp)+
476 signej*face_normales(facej,comp))
477 *vit(elem_faces(elem,ksom),comp);
478 div(npa+arete)-=1./15.*psc;
479 }
480 else
481 {
482 if(niinij==ksom)
483 niinij++;
484 while((niinij==jsom)||(niinij==isom))
485 niinij++;
486 assert(niinij<4);
487 assert(niinij!=isom);
488 assert(niinij!=jsom);
489 assert(niinij!=ksom);
490 int facek = elem_faces(elem,niinij);
491 double signek=1.;
492 if (face_voisins(facek,0) != elem)
493 signek=-1.;
494 double psc=0;
495 for(comp=0; comp<3; comp++)
496 psc+=signek*face_normales(facek,comp)
497 *vit(elem_faces(elem,ksom),comp);
498 div(npa+arete)-=2./15.*psc;
499 niinij=ksom;
500 }
501 }
502 }
503 }
504 }
505 }
506 return div;
507}

566DoubleTab& Op_Div_VEFP1B_Elem::ajouter(const DoubleTab& vitesse_face_absolue, DoubleTab& div) const
567{
568 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,la_zone_vef.valeur());
569 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
570// Quelques verifications:
571// L'espace virtuel de vitesse_face_absolue doit etre a jour (Le test est fait si check_enabled==1)
572 assert_espace_virtuel_vect(vitesse_face_absolue);
573// On s'en fiche de l'espace virtuel de div a l'entree, mais on fait += dessus.
574 assert_invalide_items_non_calcules(div, 0.);
575
576#ifndef NDEBUG
577// On s'assure que la periodicite est respectee sur vitesse_face_absolue (Voir FA814)
578 int nb_comp=vitesse_face_absolue.dimension(1);
579 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
580 {
581 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
582 if (sub_type(Periodique,la_cl.valeur()))
583 {
584 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
585 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
586 int nb_faces_bord=le_bord.nb_faces();
587 for (int ind_face=0; ind_face<nb_faces_bord; ind_face++)
588 {
589 int ind_face_associee = la_cl_perio.face_associee(ind_face);
590 int face = le_bord.num_face(ind_face);
591 int face_associee = le_bord.num_face(ind_face_associee);
592// PL : test a 1.e-4 car certains cas ne respectent pas strictement (lente derive?)
593 for (int comp=0; comp<nb_comp; comp++)
594 if (!est_egal(vitesse_face_absolue(face, comp),vitesse_face_absolue(face_associee, comp),1.e-4))
595 {
596 Cerr << "vit1("<<face<<","<<comp<<")=" << vitesse_face_absolue(face, comp) << finl;
597 Cerr << "vit2("<<face_associee<<","<<comp<<")=" << vitesse_face_absolue(face_associee, comp) << finl;
598 Cerr << "Delta=" << vitesse_face_absolue(face, comp)-vitesse_face_absolue(face_associee, comp) << finl;
