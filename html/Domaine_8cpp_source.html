216//! Ajoute des noeuds (ou sommets) au domaine<br>(sans verifier les doublons)
217void Domaine::ajouter(const DoubleTab& soms)
218{
219 int oldsz=sommets.dimension(0);
220 int ajoutsz=soms.dimension(0);
221 int dim=soms.dimension(1);
222 sommets.resize(oldsz+ajoutsz,dim);
223 for(int i=0; i<ajoutsz; i++)
224 for(int k=0; k<dim; k++)
225 sommets(oldsz+i,k)=soms(i,k) ;
226}

228// Precondition:
229// Parametre: DoubleTab& soms
230// Signification: le tableau contenant les coordonnees
231// des noeuds a ajouter au domaine
232// Valeurs par defaut:
233// Contraintes: reference constante
234// Acces: entree
235// Parametre: IntVect& nums
236// Signification: le tableau des nouveaux numeros apres
237// ajout des nouveaux noeuds et elimination
238// des doublons.
239// Valeurs par defaut:
240// Contraintes:
241// Acces: sortie
242//
243// Signification:
244// Contraintes:
245// Exception: des noeuds double ont ete trouve
246// Effets de bord:
247// Postcondition:
248//
249//
250//
251//! Ajoute des noeuds au domaine avec elimination des noeuds double<br>au retour nums contient les nouveaux numeros des noeuds de soms<br>apres elimination des doublons.
252void Domaine::ajouter(const DoubleTab& soms, IntVect& nums)
253{
254 int oldsz=sommets.dimension(0);
255 int ajoutsz=soms.dimension(0);
256 int dim=soms.dimension(1);
257 nums.resize(ajoutsz);
258 nums=-1;
259 if(oldsz!=0)
260 {
261 assert(dim==sommets.dimension(1));
262 Octree_Double octree;
263 octree.build_nodes(les_sommets(), 0 /* ne pas inclure les sommets virtuels */);
264
265 int i, k;
266 int compteur=0;
267 ArrOfDouble tab_coord(dim);
268 ArrOfInt liste_sommets;
269 liste_sommets.set_smart_resize(1);
270 for( i=0; i< ajoutsz; i++)
271 {
272 for (int j = 0; j < dim; j++)
273 tab_coord[j] = soms(i,j);
274 octree.search_elements_box(tab_coord, epsilon_, liste_sommets);
275 octree.search_nodes_close_to(tab_coord, les_sommets(), liste_sommets, epsilon_);
276 const int nb_sommets_proches = liste_sommets.size_array();
277 if (nb_sommets_proches == 0)
278 {
279// Aucun sommet du premier domaine n'est proche du sommet i.
280// Garder i.
281 }
282 else if (nb_sommets_proches == 1)
283 {
284// Un sommet est confondu avec le sommet i a epsilon_ pres.
285// Ne pas garder le sommet
286 nums(i) = liste_sommets[0];
287 compteur++;
288 }
289 else
290 {
291// Plusieurs sommets du domaine initial sont dans un rayon de epsilon.
292// epsilon est trop grand.
293 Cerr << "Error : several nodes of the domain 1 are within radius epsilon="
294 << epsilon_ << " of point " << tab_coord << ". We must reduce epsilon. " << finl;
295 Process::exit();
296 }
297 }
298 Cerr << compteur << " double nodes were found \n";
299 sommets.resize(oldsz+ajoutsz-compteur,dim);
300 compteur=0;
301 for( i=0; i<ajoutsz; i++)
302 if(nums(i)==-1)
303 {
304 nums(i)=oldsz+compteur;
305 compteur++;
306 for( k=0; k<dim; k++)
307 sommets(nums(i),k)=soms(i,k) ;
308 }
309 }
310 else
311 {
312 sommets=soms;
313// if som has a descriptor, delete it:
314 sommets.set_md_vector(MD_Vector());
315 for(int i=0; i<ajoutsz; i++)
316 {
317 nums(i)=i;
318 }
319 }
320}

502DoubleTab Domaine::getBoundingBox() const
503{
504 DoubleTab BB(dimension, 2);
505 int nbsom=coord_sommets().dimension(0);
506 for (int j=0; j<dimension; j++)
507 {
508 double min_=0.5*DMAXFLOAT;
509 double max_=-0.5*DMAXFLOAT;
510 for (int i=0; i<nbsom; i++)
511 {
512 double c = coord_sommets()(i,j);
513 min_ = (c < min_ ? c : min_);
514 max_ = (c > max_ ? c : max_);
515 }
516 BB(j,0) = min_;
517 BB(j,1) = max_;
518 }
519 return BB;
520}

522void Domaine::imprimer() const
523{
524 Cerr << "==============================================" << finl;
525 Cerr << "The extreme coordinates of the domain " << le_nom() << " are:" << finl;
526// Il n'existe pas de recherche du min et du max dans DoubleTab donc je code:
527 DoubleTab BB = getBoundingBox();
528 for (int j=0; j<dimension; j++)
529 {
530 double min_ = mp_min(BB(j,0));
531 double max_ = mp_max(BB(j,1));
532 if (j==0) Cerr << "x ";
533 if (j==1) Cerr << "y ";
534 if (j==2) Cerr << "z ";
535 Cerr << "is between " << min_ << " and " << max_ << finl;
536 }
537 Cerr << "==============================================" << finl;
538// We recompute volumes (cause stored in Zone_VF and so not available from Domaine...):
539 DoubleVect volumes;
540 DoubleVect inverse_volumes;
541 zone(0).calculer_volumes(volumes,inverse_volumes);
542 Cerr << "==============================================" << finl;
543 Cerr << "The volume cells of the domain " << le_nom() << " are:" << finl;
544 const int i_vmax = imax_array(volumes);
545 const int i_vmin = imin_array(volumes);
546 const double vmin_local = (i_vmin < 0) ? 1e40 : volumes[i_vmin];
547 const double vmax_local = (i_vmax < 0) ? -1e40 : volumes[i_vmax];
548 const double volmin = mp_min(vmin_local);
549 const double volmax = mp_max(vmax_local);
550 double volume_total = mp_somme_vect(volumes);
551 const int nb_elem = zone(0).nb_elem();
552 double volmoy = volume_total / Process::mp_sum(nb_elem);
553 Cerr << "sum(volume cells)= " << volume_total << finl;
554 Cerr << "mean(volume cells)= " << volmoy << finl;
555 Cerr << "min(volume cells)= " << volmin << finl;
556 Cerr << "max(volume cells)= " << volmax << finl;
557 if (volmin*1000<volmoy)
558 {
559 Cerr << "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!" << finl;
560 Cerr << "Warning, a cell volume is more than 1000 times smaller than the average cell volume. Check your mesh." << finl;
561 Cerr << "!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!" << finl;
562 }
563 Cerr << "==============================================" << finl;
564}

566// Build the faces mesh:
567void Domaine::buildUFacesMesh(const Zone_dis_base& zone_dis_base) const
568{
569#ifdef MEDCOUPLING_
570 MCAuto<DataArrayInt> desc(DataArrayInt::New());
571 MCAuto<DataArrayInt> descIndx(DataArrayInt::New());
572 MCAuto<DataArrayInt> revDesc(DataArrayInt::New());
573 MCAuto<DataArrayInt> revDescIndx(DataArrayInt::New());
574 faces_mesh_ = mesh_->buildDescendingConnectivity(desc, descIndx, revDesc, revDescIndx);
575// Renumber faces to have the same numbering than Domaine_dis
576 std::size_t size = faces_mesh_->getNumberOfCells();
577 IntVect renum(size);
578// Compute Center of Mass
579 MCAuto<DataArrayDouble> xv_med = faces_mesh_->computeCellCenterOfMass();
580// On boucle sur les elements des tableaux Zone_VF::elem_faces et desc
581// Ensuite on compare geometriquement les centres des faces de ces tableaux pour trouver les relations
582// Boucle sur les mailles
583 const IntTab& elem_faces = ref_cast(Zone_VF, zone_dis_base).elem_faces();
584 int nb_elem = elem_faces.dimension(0);
585 assert(nb_elem == (int)descIndx->getNbOfElems()-1);
586// Centre des faces TRUST:
587 const DoubleTab& xv = ref_cast(Zone_VF, zone_dis_base).xv();
588// int nb_faces = xv.dimension_tot(0);
589 int nb_comp = xv.dimension_tot(1), f;
590
591// Boucle sur les elements
592 MCAuto<DataArrayInt> renum_local(DataArrayInt::New());
593 for (int elem=0; elem<nb_elem; elem++)
594 {
595// Cerr << "elem=" << elem << finl;
596 int nb_face_elem = 0;
597 for (int i = 0; i < elem_faces.dimension(1) && elem_faces(elem, i) >= 0; i++)
598 nb_face_elem++;
599// xv1, xv2 tableaux temporaires des centres des faces de l'element elem pour comparaison
600 MCAuto<DataArrayDouble> xv1(DataArrayDouble::New());
601 xv1->alloc(nb_face_elem, nb_comp);
602 MCAuto<DataArrayDouble> xv2(DataArrayDouble::New());
603 xv2->alloc(nb_face_elem, nb_comp);
604 for (int i = 0; i < elem_faces.dimension(1) && (f = elem_faces(elem, i)) >= 0; i++)
605 {
606 /*
607 Cerr << "\tface=" << face;
608 for (int j=0; j<nb_comp; j++) Cerr << " " << xv(face, j);
609 Cerr << finl; */
610// Face globale MED
611 int index = descIndx->getIJ(elem, 0);
612 int face_med = desc->getIJ(index + i, 0);
613 /*
614 Cerr << "\tface_med=" << face_med;
615 for (int j=0; j<nb_comp; j++) Cerr << " " << xv_med->getIJ(face_med, j);
616 Cerr << finl; */
617// Centre des faces TRUST et MED
618 for (int j=0; j<nb_comp; j++)
619 {
620 xv1->setIJ(i, j, xv(f, j));
621 xv2->setIJ(i, j, xv_med->getIJ(face_med, j));
622 }
623 }
624 renum_local = xv2->findClosestTupleId(xv1);
625 for (int i=0; i<renum_local->getNumberOfTuples(); i++)
