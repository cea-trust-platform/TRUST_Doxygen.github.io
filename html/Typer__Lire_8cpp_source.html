61//! Read an object.
62Entree& Typer_Lire::interpreter(Entree& is)
63{
64// Example in a data file:
65// Typer_Lire name type { ... }
66 Nom name;
67 is >> name; //!< The object name is read from the input stream is
68
69 if (objet_existant(name)) //nom d'un objet existant -> on le lit
70 {
71 Objet_U& object=objet(name);
72 return is >> object; //!< Then "{ ... }" is read by the object readOn
73 }
74 else //pas le nom d'un objet existant -> on lit le type, on cree l'objet, puis on le lit
75 {
76// Lire name type { ... }
77 DerObjU ref;
78 Nom type;
79 is >> type;
80 if (type=="{")
81 {
82 Objet_U& object=objet(name); //!< Pour faire planter avec le message d'erreur habituel
83 return is >> object;
84 }
85 ref.typer(type);
86// on ajoute l'objet a l'interprete courant...
87 Objet_U& obj = Interprete_bloc::interprete_courant().ajouter(name, ref);
88 return is >> obj; //et on lit
89 }
90}
