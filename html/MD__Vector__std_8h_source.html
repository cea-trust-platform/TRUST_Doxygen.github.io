81template<typename _TYPE_>
82void MD_Vector_std::initialize_comm_template(const Echange_EV_Options& opt, Schema_Comm_Vecteurs& comm_, TRUSTVect<_TYPE_>& v) const
83{
84 const int reverse = (opt.get_op() != Echange_EV_Options::SYNC);
85 const int have_items_to_recv = (items_to_recv_.get_data().size_array() > 0);
86 const int have_blocs_to_recv = (blocs_to_recv_.get_data().size_array() > 0);
87 const int nprocs = pe_voisins_.size_array();
88 for (int i = 0; i < nprocs; i++)
89 {
90 const int pe = pe_voisins_[i];
91 int sz1 = items_to_send_.get_list_size(i), sz2 = 0;
92
93 if (have_items_to_recv) sz2 = items_to_recv_.get_list_size(i);
94 if (have_blocs_to_recv) sz2 += blocs_items_count_[i];
95 if (reverse)
96 {
97// Schema a l'envers: lecture dans les items to recv, ecriture dans les items to send
98 int tmp = sz1;
99 sz1 = sz2;
100 sz2 = tmp;
101 }
102 comm_.add_send_area_template < _TYPE_ > (pe, sz1 * v.line_size());
103 comm_.add_recv_area_template < _TYPE_ > (pe, sz2 * v.line_size());
104 }
105}

109//! This is the first part of the echange_espace_virtuel() process.<br>We take the data that has to be sent to other processors in "vect" and put it in appropriate send buffers in "buffers". The data taken depends on "opt".<br>Preconditon: The buffers must have been initialized to the appropriate size and neighbour processors.
110template<typename _TYPE_>
111void MD_Vector_std::prepare_send_data_template(const Echange_EV_Options& opt, Schema_Comm_Vecteurs& buffers, TRUSTVect<_TYPE_>& vect) const
112{
113 const int line_size = vect.line_size();
114 switch(opt.get_op())
115 {
116 case Echange_EV_Options::SYNC:
117 read_from_vect_items(line_size, pe_voisins_, items_to_send_, vect, buffers);
118 break;
119 case Echange_EV_Options::SUM:
120 case Echange_EV_Options::MAX:
121 case Echange_EV_Options::MINCOL1:
122 if (items_to_recv_.get_data().size_array() > 0)
123 read_from_vect_items(line_size, pe_voisins_, items_to_recv_, vect, buffers);
124 if (blocs_to_recv_.get_data().size_array() > 0)
125 read_from_vect_blocs(line_size, pe_voisins_, blocs_to_recv_, blocs_items_count_, vect, buffers);
126 break;
127 default:
128 Cerr << "Error in MD_Vector_std.tpp prepare_send_data_template : operation not implemented" << finl;
129 Process::exit();
130 }
131}

133template<typename _TYPE_>
134void MD_Vector_std::process_recv_data_template(const Echange_EV_Options& opt, Schema_Comm_Vecteurs& buffers, TRUSTVect<_TYPE_>& vect) const
135{
136 const int line_size = vect.line_size();
137 switch(opt.get_op())
138 {
139 case Echange_EV_Options::SYNC:
140 if (items_to_recv_.get_data().size_array() > 0)
141 write_to_vect_items(line_size, pe_voisins_, items_to_recv_, vect, buffers);
142 if (blocs_to_recv_.get_data().size_array() > 0)
143 write_to_vect_blocs(line_size, pe_voisins_, blocs_to_recv_, blocs_items_count_, vect, buffers);
144 break;
145 case Echange_EV_Options::SUM:
146 add_to_vect_items(line_size, pe_voisins_, items_to_send_, vect, buffers);
147 break;
148 case Echange_EV_Options::MAX:
149 max_to_vect_items(line_size, pe_voisins_, items_to_send_, vect, buffers);
150 break;
151 case Echange_EV_Options::MINCOL1:
152 mincol1_to_vect_items(line_size, pe_voisins_, items_to_send_, vect, buffers);
153 break;
154 default:
155 Cerr << "Error in MD_Vector_std.tpp process_recv_data_template : operation not implemented" << finl;
156 Process::exit();
157 }
158}
