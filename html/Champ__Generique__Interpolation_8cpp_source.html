131//! Initialisation de la classe: initialisation du domaine d'interpolation<br>Parametres valides : "" => domaine natif, ou le nom d'un objet Domaine connu de l'interprete
132int Champ_Generique_Interpolation::set_domaine(const Nom& nom_domaine, int exit_on_error)
133{
134 int ok = 0;
135 if (nom_domaine == "")
136 {
137 domaine_.reset(); //!< reference nulle => domaine natif du champ
138 ok = 1;
139 }
140 else
141 {
142 const Objet_U& ob = interprete().objet(nom_domaine);
143 if (!sub_type(Domaine, ob))
144 {
145 Cerr << "Error in Champ_Generique_Interpolation::set_domaine(nom_domaine)\n"
146 << " The object " << nom_domaine << " is not a domain" << finl;
147 if (exit_on_error)
148 {
149 exit();
150 }
151 }
152 else
153 {
154 domaine_ = ref_cast(Domaine, ob);
155 }
156 }
157 return ok;
158}

194 const Noms nom_champ = get_property("nom");
195 const Noms syno = get_property("synonyms");

217//! Interpolation du champ source a l'aide de Champ_base::calculer_champ_xxx_post
218const Champ_base& Champ_Generique_Interpolation::get_champ_with_calculer_champ_post(Champ& espace_stockage) const
219{
220 Champ espace_stockage_source;
221 const Champ_base& source0 = get_source(0).get_champ(espace_stockage_source);
222 Champ source_bis;
223
224 if (optimisation_sous_maillage_==-1)
225 {
226 source_bis=source0;
227// source.typer(source0.que_suis_je());
228
229 }
230 const Champ_base& source = ((optimisation_sous_maillage_==-1)?source_bis.valeur():source0);
231 Nom nom_champ_interpole;
232
233// Domaine sur lequel on interpole le champ :
234// si domaine_ est une ref nulle, on prend le domaine natif du champ.
235 const Domaine& domaine = get_ref_domain();
236 const Noms compo = get_property("composantes");
237 const Noms nom_champ = get_property("nom");
238 const Noms syno = get_property("synonyms");
239
240 int ncomp;
241// On fixe identifiant_appel_ a nom_champ[0] dans le cas ou il est egal a ??
242 if (identifiant_appel_=="??")
243 {
244 ncomp = -1;
245 }
246 else
247 {
248 ncomp = Champ_Generique_base::composante(identifiant_appel_,nom_champ[0],compo,syno);
249 }
250
251// Creation du champ espace_stockage
252 Nature_du_champ nature_source = (ncomp==-1)?source.nature_du_champ():scalaire;
253 nature_source = source.nature_du_champ();
254 int nb_comp = source.nb_comp();
255
256 Champ_Fonc es_tmp;
257 espace_stockage = creer_espace_stockage(nature_source,nb_comp,es_tmp);
258
259// double default_value=-1e35;
260// espace_stockage.valeurs()=default_value;
261 int decal=10;
262 if (optimisation_sous_maillage_==-1)
263 {
264 espace_stockage.valeurs()=0;
265// premier appel avec maillage different (ou on a force) , on essaye de voir si on peut optimiser
266// on champ la source on y met val(i)=i, pour recuperer le numero de la maille apres
267 DoubleTab& val=source_bis.valeurs();
268
269 int dim0 = val.dimension_tot(0);
270 for (int i=0; i<dim0; i++)
271 for (int j=0; j<nb_comp; j++)
272 val(i,j) = i+decal;
273 }
274
275// Evaluation des valeurs du champ espace_stockage
276 DoubleTab& espace_valeurs = espace_stockage->valeurs();
277
278 if (optimisation_sous_maillage_==1)
279 {
280 const DoubleTab& val_temp= source.valeurs();
281 int dim0 = espace_valeurs.dimension(0);
282 for (int i=0; i<dim0; i++)
283 if (ncomp==-1)
284 for (int j=0; j<nb_comp; j++)
285 {
286 int n=renumerotation_maillage_[i];
287 if (n>=0)
288 espace_valeurs(i,j) = val_temp(n,j);
289 }
290 else espace_valeurs(i, ncomp) = val_temp[renumerotation_maillage_[i]];
291
292 espace_valeurs.echange_espace_virtuel();
293 return espace_stockage.valeur();
294 }
295
296 int imax=0;
297// int jmax=0;;
298 imax = espace_valeurs.dimension(0);
299// if (espace_valeurs.nb_dim()==2)
300// jmax = espace_valeurs.dimension(1);
301
302 if (localisation_ == "elem")
303 {
304 const int nb_elements = domaine.zone(0).nb_elem();
305
306 if (ncomp==-1)
307 {
308 DoubleTab val_temp;
309 source.calculer_valeurs_elem_post(val_temp,
310 nb_elements,
311 nom_champ_interpole,
312 domaine);
313
314 for (int i_val=0; i_val<imax; i_val++)
315 for (int j_val=0; j_val<nb_comp; j_val++)
316 espace_valeurs(i_val,j_val) = val_temp(i_val,j_val);
317
318 }
319 else
320// On construit un tableau de valeurs a nb_comp composantes meme si ncomp!=-1
321 {
322 DoubleTab val_temp;
323 source.calculer_valeurs_elem_compo_post(val_temp,
324 ncomp,
325 nb_elements,
326 nom_champ_interpole,
327 domaine);
328
329 int dim0 = val_temp.dimension(0);
330 for (int i=0; i<dim0; i++)
331 for (int j=0; j<nb_comp; j++)
332 if (j==ncomp)
333 espace_valeurs(i,j) = val_temp(i);
334 }
335
336 }
337 else if (localisation_=="som")
338 {
339 const int nb_sommets = domaine.nb_som();
340// PL: mise a jour de l'espace virtuel de la source:
341 Champ copie_source;
342 copie_source = source;
343 copie_source.valeurs().echange_espace_virtuel();
344 if (ncomp==-1)
345 {
346 DoubleTab val_temp;
347 copie_source.valeur().calculer_valeurs_som_post(val_temp,
348 nb_sommets,
349 nom_champ_interpole,
350 domaine);
351 for (int i_val=0; i_val<imax; i_val++)
352 for (int j_val=0; j_val<nb_comp; j_val++)
353 espace_valeurs(i_val,j_val) = val_temp(i_val,j_val);
354 }
355 else
356// On construit un tableau de valeurs a nb_comp composantes meme si ncomp!=-1
357 {
358 DoubleTab val_temp;
359 copie_source.valeur().calculer_valeurs_som_compo_post(val_temp,
360 ncomp,
361 nb_sommets,
362 nom_champ_interpole,
363 domaine);
364
365 int dim0 = val_temp.dimension(0);
366 for (int i=0; i<dim0; i++)
367 for (int j=0; j<nb_comp; j++)
368 if (j==ncomp)
369 espace_valeurs(i,j) = val_temp(i);
370 }
371 }
372 else
373 {
374// const Nom& nom_source = source.que_suis_je();
375// if (((nom_source=="Champ_P1NC") || (nom_source=="Champ_Fonc_P1NC")) && (localisation_=="faces"))
376// {
377 espace_stockage->affecter(source);
378 /* else
379 {
380 Cerr << "Champ_Generique_Interpolation::get_champ localisation uncoded " << localisation_ << " for " <<source.que_suis_je()<<finl;
381 exit();
382 }
383// }
384 */
385 }
386
387
388
389 if (optimisation_sous_maillage_==-1)
390 {
391// on recalcule le bon champ
392
393 DoubleTab& val=espace_valeurs;
394 int dim0 = val.dimension_tot(0);
395 ArrOfInt& renumerotation_maillage=ref_cast_non_const(ArrOfInt, renumerotation_maillage_);
396 renumerotation_maillage.resize_array(dim0);
397 int ok=1;
398
399 for (int i=0; (i<dim0)&& ok; i++)
400 {
401 double v;
402 v=espace_valeurs(i,0) ;
403
404 int iproche=-1;
405 if (v>0)
406 {
407 iproche=int(v+0.1); //!< on ajoute 0.1 pour le cas ou v vaut i-epsilon
408 if (std::fabs(iproche-v)>1e-5)
409 {
410 ok=0;
411 Cerr<<nom_champ[0]<<" optimisation ko "<< finl;
412
413 }
414 }
415 renumerotation_maillage[i]=iproche-decal;
416
417 }
418
419 int test=(int)mp_min(ok);
420 if (test)
421 {
422 ref_cast_non_const(Champ_Generique_Interpolation,(*this)).optimisation_sous_maillage_=0;
423 Champ espace_stockage_test;
424 get_champ_with_calculer_champ_post(espace_stockage_test);
425 ref_cast_non_const(Champ_Generique_Interpolation,(*this)).optimisation_sous_maillage_=test;
426 get_champ_with_calculer_champ_post(espace_stockage);
427 espace_stockage_test.valeurs()-=espace_stockage.valeurs();
428 double dmax= mp_max_abs_vect(espace_stockage_test.valeurs());
429 if (dmax > 1e-7)
430 {
431 Cerr<<nom_champ[0]<<" optimisation ko "<< dmax<<finl;
432 exit();
433 }
434 else
435 Cerr<<nom_champ[0]<<" optimisation ok "<< dmax<<finl;
436 }
437
438 else
439 {
440 if (optimisation_demande_==1)
441 {
442 Cerr<<nom_champ[0]<<" optimisation not allowed "<<finl;
443 exit();
444 }
445 }
446 ref_cast_non_const(Champ_Generique_Interpolation,(*this)).optimisation_sous_maillage_=test;
447 return get_champ_with_calculer_champ_post(espace_stockage);
448 }
449
450 espace_valeurs.echange_espace_virtuel();
451 return espace_stockage.valeur();
452}

521const Noms Champ_Generique_Interpolation::get_property(const Motcle& query) const
522{
523
524 Motcles motcles(2);
525 motcles[0] = "composantes";
526 motcles[1] = "synonyms";
527 int rang = motcles.search(query);
528 switch(rang)
529 {
530
531 case 0:
532 {
533 Noms source_compos = get_source(0).get_property("composantes");
534 int nb_comp = source_compos.size();
535 Noms compo(nb_comp);
536
537 if (compo_.size()>1)
538 {
539// Le champ a ete cree par macro et on a rempli l attribut compo_ du Champ_Generique
540// Cela permet de reproduire les noms de composantes dans les lml
541 compo=compo_;
542 }
543 else
544 {
545 for (int i=0; i<nb_comp; i++)
546 {
547 Nom nume(i);
548 compo[i] = nom_post_+nume;
549 }
550 }
551 return compo;
552 break;
553 }
554 case 1:
555 {
556 if (syno_.size()>0)
557 {
558 return syno_;
559 }
560 return get_source(0).get_property("synonyms");
561 break;
562 }
563
564 }
565
566 return Champ_Gen_de_Champs_Gen::get_property(query);
567
