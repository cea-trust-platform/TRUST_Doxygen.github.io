131//! Renvoie 1 si l'element ielem de la zone associee a<br>l'element geometrique contient le point<br>de coordonnees specifiees par le parametre "pos".<br>Renvoie 0 sinon.
132/** \return int<br>
133Signification: 1 si le point de coordonnees specifiees<br>
134appartient a l'element ielem<br>
1350 sinon<br>
136Contraintes: */
137int Quadrangle_VEF::contient(const ArrOfDouble& pos, int element) const
138{
139 assert(pos.size_array()==2);
140 const Zone& zone=ma_zone.valeur();
141 const Domaine& dom=zone.domaine();
142 int som0 = zone.sommet_elem(element,0);
143 int som1 = zone.sommet_elem(element,1);
144 int som2 = zone.sommet_elem(element,2);
145 int som3 = zone.sommet_elem(element,3);
146// On regarde tout d'abord si le point cherche n'est pas un des
147// sommets du quadrangle
148 if( (est_egal(dom.coord(som0,0),pos[0]) && est_egal(dom.coord(som0,1),pos[1]))
149 || (est_egal(dom.coord(som1,0),pos[0]) && est_egal(dom.coord(som1,1),pos[1]))
150 || (est_egal(dom.coord(som2,0),pos[0]) && est_egal(dom.coord(som2,1),pos[1]))
151 || (est_egal(dom.coord(som3,0),pos[0]) && est_egal(dom.coord(som3,1),pos[1])) )
152 return 1;
153 double prod,p0,p1,p2,p3;
154// Calcul de prod = 01 vectoriel 02 selon z
155 prod = (dom.coord(som1,0)-dom.coord(som0,0))*(dom.coord(som2,1)-dom.coord(som0,1))
156 - (dom.coord(som1,1)-dom.coord(som0,1))*(dom.coord(som2,0)-dom.coord(som0,0));
157 double signe;
158 if (prod >= 0)
159 signe = 1;
160 else
161 signe = -1;
162// Calcul de p0 = 0M vectoriel 1M selon z
163 p0 = (pos[0]-dom.coord(som0,0))*(pos[1]-dom.coord(som1,1))
164 - (pos[1]-dom.coord(som0,1))*(pos[0]-dom.coord(som1,0));
165 p0 *= signe;
166// Calcul de p1 = 1M vectoriel 3M selon z
167 p1 = (pos[0]-dom.coord(som1,0))*(pos[1]-dom.coord(som3,1))
168 - (pos[1]-dom.coord(som1,1))*(pos[0]-dom.coord(som3,0));
169 p1 *= signe;
170// Calcul de p2 = 3M vectoriel 2M selon z
171 p2 = (pos[0]-dom.coord(som3,0))*(pos[1]-dom.coord(som2,1))
172 - (pos[1]-dom.coord(som3,1))*(pos[0]-dom.coord(som2,0));
173 p2 *= signe;
174// Calcul de p3 = 2M vectoriel 0M selon z
175 p3 = (pos[0]-dom.coord(som2,0))*(pos[1]-dom.coord(som0,1))
176 - (pos[1]-dom.coord(som2,1))*(pos[0]-dom.coord(som0,0));
177 p3 *= signe;
178 double epsilon=std::fabs(prod)*Objet_U::precision_geom;
179 if ((p0>-epsilon) && (p1>-epsilon) && (p2>-epsilon) && (p3>-epsilon))
180 return 1;
181 else
182 return 0;
183}

236//! Calcule les volumes des elements de la zone associee.
237void Quadrangle_VEF::calculer_volumes(DoubleVect& volumes) const
238{
239 const Zone& zone=ma_zone.valeur();
240 const DoubleTab& coord = zone.domaine().coord_sommets();
241 int S0,S1,S2,S3;
242 ArrOfDouble xg(dimension);
243 IntTab face_sommet_global;
244 face_sommet_global.resize(4,2);
245 DoubleTab pos(3,dimension);
246 int numface;
247
248 int size_tot = zone.nb_elem_tot();
249 assert(volumes.size_totale()==size_tot);
250 for (int num_poly=0; num_poly<size_tot; num_poly++)
251 {
252 S0 = zone.sommet_elem(num_poly,0);
253 S1 = zone.sommet_elem(num_poly,1);
254 S2 = zone.sommet_elem(num_poly,2);
255 S3 = zone.sommet_elem(num_poly,3);
256
257// on construit le tableau de travail face_sommet_global
258// les faces seront ordonnee suivant Xmin, Xmax, Ymin, Ymax, Zmin,Zmax
259// toujours en gardant la numerotation de TRUST
260
261 face_sommet_global(0,0) =S0;
262 face_sommet_global(0,1) =S2;
263
264 face_sommet_global(1,0) =S0;
265 face_sommet_global(1,1) =S1;
266
267 face_sommet_global(2,0) =S1;
268 face_sommet_global(2,1) =S3;
269
270 face_sommet_global(3,0) =S3;
271 face_sommet_global(3,1) =S2;
272
273// calcul du centre de gravite de l'element
274 for (int j=0; j<dimension; j++)
275 pos(0,j)=(coord(S0,j)+coord(S1,j)+coord(S2,j)+coord(S3,j))*0.25;
276
277 double v = 0.;
278// boucle sur les faces de l'element que l'on decoupe en 4 triangles.
279// constitue des deux sommets de la face consideree et du centre de gravite de l'element
280 for(numface=0; numface<nb_faces(0); numface++)
281 {
282 for (int i=0; i<2; i++)
283 {
284 int Si = face_sommet_global(numface,i);
285 for (int j=0; j<dimension; j++)
286 pos(i+1,j) = coord(Si,j);
287 }
288 v += aire_triangle(pos);
289 }
290 volumes[num_poly] = v;
291 }
292}

309void Quadrangle_VEF::reordonner()
310{
311 Zone& zone=ma_zone.valeur();
312 const Domaine& dom=zone.domaine();
313 IntTab& elem=zone.les_elems();
314
315 ArrOfInt S(4);
316 ArrOfInt NS(4);

320 for (num_poly=0; num_poly<nb_elem; num_poly++)
321 {
322 for(int i=0; i<4; i++)
323 {
324 S[i] = elem(num_poly,i);
325 }
326
327// adaptation de Hexaedre_VEF ::reordonne
328
329 const DoubleTab& coord=dom.les_sommets();
330 DoubleTab v(3,3);
331 for (int i=1; i<4; i++)
332 for (int dir=0; dir<2; dir++)
333 v(i-1,dir)=coord(S[i],dir)-coord(S[0],dir);
334 ArrOfDouble prod_(3);
335 int opp=-1;
336 for (int op=0; op<3; op++)
337 {
338 DoubleTab prod_v(3,3);
339 for (int i=0; i<3; i++)
340 {
341// prod_v(i,0)=v(op,1)*v(i,2)-v(op,2)*v(i,1);
342// prod_v(i,1)=v(op,2)*v(i,0)-v(op,0)*v(i,2);
343 prod_v(i,2)=v(op,0)*v(i,1)-v(op,1)*v(i,0);
344 }
345// Cerr<<prod_v<<finl;
346 double prod=0;
347 int i1=-1,i2=-1;
348 if (op==0)
349 {
350 i1=1;
351 i2=2;
352 }
353 else if (op==1)
354 {
355 i1=0;
356 i2=2;
357 }
358 else if (op==2)
359 {
360 i1=0;
361 i2=1;
362 }
363 for (int dir=0; dir<3; dir++)
364 prod+=(prod_v(i1,dir)*prod_v(i2,dir));
365 prod_[op]=prod;
366 if (prod<0)
367 {
368 if (opp!=-1)
369 {
370 op=2;
371// on a deux produits negatifs on ne fait rien
372 }
373 opp=op;
374
375 }
376 }
377 if (opp!=2)
378 {
379 int i2=2;
380 int i1=1;
381 int i3=3;
382 int i2b=i2;
383 if (opp==0) i2b=i1;
384 int tmp=elem(num_poly,i2b);
385 elem(num_poly,i2b)=elem(num_poly,i3);
386 elem(num_poly,i3)=tmp;
387 Cerr << "Permutation of local nodes "<<i2b<<" and "<<i3<<" on the element " <<num_poly<<" prod "<<prod_<<finl;
388 }
389
390 }
