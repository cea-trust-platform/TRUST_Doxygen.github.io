94void Terme_Source_Canal_perio_VDF_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
95{
96 const Zone_VF& zone_VF = la_zone_VDF.valeur();
97 const Zone_Cl_dis_base& zone_Cl_dis = la_zone_Cl_VDF.valeur();
98 const IntVect& orientation = la_zone_VDF->orientation();
99 const DoubleVect& porosite_surf = zone_VF.porosite_face();
100 const DoubleVect& volumes_entrelaces = zone_VF.volumes_entrelaces();
101 int ncomp;
102 ArrOfDouble s(source());
103
104// Boucle sur les conditions limites pour traiter les faces de bord
105 int n_bord,ndeb,nfin;
106 for (n_bord=0; n_bord<zone_VF.nb_front_Cl(); n_bord++)
107 {
108
109// pour chaque Condition Limite on regarde son type
110// Si face de Dirichlet ou de Symetrie on ne fait rien
111// Si face de Neumann on calcule la contribution au terme source
112
113 const Cond_lim& la_cl = zone_Cl_dis.les_conditions_limites(n_bord);
114
115 if (sub_type(Neumann_sortie_libre,la_cl.valeur())
116 ||sub_type(Periodique,la_cl.valeur())
117 ||sub_type(Symetrie,la_cl.valeur()))
118 {
119
120 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
121 ndeb = le_bord.num_premiere_face();
122 nfin = ndeb + le_bord.nb_faces();
123
124 for (int num_face=ndeb; num_face<nfin; num_face++)
125 {
126 double vol = volumes_entrelaces(num_face)*porosite_surf(num_face);
127 ncomp = orientation(num_face);
128 secmem(num_face) += s[ncomp]*vol;
129 }
130 }
131 else if ( (sub_type(Dirichlet,la_cl.valeur()))
132 ||
133 (sub_type(Dirichlet_homogene,la_cl.valeur()))
134 )
135 {
136// do nothing
137 ;
138 }
139 }
140
141// Boucle sur les faces internes
142 ndeb = zone_VF.premiere_face_int();
143 int nb_faces = zone_VF.nb_faces();
144 for (int num_face = ndeb; num_face<nb_faces; num_face++)
145 {
146 double vol = volumes_entrelaces(num_face)*porosite_surf(num_face);
147 ncomp = orientation(num_face);
148 secmem(num_face)+= s[ncomp]*vol;
149 }
150
151}

154void Terme_Source_Canal_perio_VDF_Face::calculer_debit(double& debit_e) const
155{
156 const Zone_VF& zone_VF = la_zone_VDF.valeur();
157 const Zone_Cl_dis_base& zone_Cl_dis = la_zone_Cl_VDF.valeur();
158 const DoubleTab& vitesse = equation().inconnue().valeurs();
159 const DoubleVect& porosite_surf = zone_VF.porosite_face();
160 int ndeb,nfin,num_face;
161 int nb_bords = zone_VF.nb_front_Cl();
162 for (int n_bord=0; n_bord<nb_bords; n_bord++)
163 {
164 const Cond_lim& la_cl = zone_Cl_dis.les_conditions_limites(n_bord);
165
166 if (sub_type(Periodique,la_cl.valeur()))
167 {
168 const Periodique& perio = ref_cast(Periodique,la_cl.valeur());
169 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
170 if (bord_periodique_ == le_bord.le_nom())
171 {
172 int axe = perio.direction_periodicite();
173 assert(axe==direction_ecoulement_);
174 debit_e=0.;
175 ndeb = le_bord.num_premiere_face();
176 nfin = ndeb + le_bord.nb_faces()/2;
177
178 if (equation().probleme().is_dilatable()==1)
179 {
180// Si l'on est en Quasi/Weakly Compressible, il faut conserver
181// le debit massique et non pas le debit volumique.
182// C'est pour cela que dans le cas QC/WC, on multiplie les vecteurs vitesse
183// par la masse volumique discretisee aux faces pour que lorsqu'on integre sur la surface,
184// on obtienne bien un debit massique et non pas un debit volumique.
185 const DoubleTab& tab_rho_face = ref_cast(Fluide_Dilatable_base,equation().milieu()).rho_discvit();
186
187 for (num_face=ndeb; num_face<nfin; num_face++)
188 {
189 double debit_face = porosite_surf[num_face]*vitesse[num_face] * std::fabs(zone_VF.face_normales(num_face,axe));
190 debit_e += tab_rho_face[num_face] * debit_face;
191 }
192 }
193 else
194 {
195 for (num_face=ndeb; num_face<nfin; num_face++)
196 {
197 double debit_face = porosite_surf[num_face]*vitesse[num_face] * std::fabs(zone_VF.face_normales(num_face,axe));
198 debit_e += debit_face;
199 }
200 }
201 }
202 }
203 }
204 debit_e = mp_sum(debit_e);
205}
