74DoubleTab& Champ_P1iP1B_implementation::valeur_aux_elems(const DoubleTab& positions, const IntVect& les_polys, DoubleTab& val) const
75{
76 int som;
77 double xs,ys,zs=0;
78// zs bien initialise en 3D
79
80 const Zone_VEF_PreP1b& zvef = zone_vef();
81 const Zone& zone_geom = zvef.zone();
82 const DoubleTab& coord = zone_geom.domaine().coord_sommets();
83 const IntTab& sommet_poly = zone_geom.les_elems();
84 int prs=zvef.numero_premier_sommet();
85 int nb_compo_=le_champ().nb_comp();
86 int dimension=Objet_U::dimension;
87 const int les_polys_size=les_polys.size();
88
89// TODO : FIXME
90// For FT the resize should be done in its good position and not here ...
91 if (val.nb_dim() == 1 ) val.resize(les_polys_size,1);
92
93 assert((val.dimension(0) == les_polys.size())||(val.dimension_tot(0) == les_polys.size()));
94 assert(val.line_size() == nb_compo_);
95
96// Filtrage du champ dans le tableau champ_filtre_
97 champ_filtre_=filtrage(zvef,le_champ());
98
99 if (nb_compo_ == 1)
100 {
101 for(int rang_poly=0; rang_poly<les_polys_size; rang_poly++)
102 {
103// On initialise
104 val(rang_poly,0) = 0.;
105
106// On prend le_poly
107 int le_poly=les_polys(rang_poly);
108 if (le_poly != -1)
109 {
110// Contribution P0
111 if (zvef.get_alphaE())
112 val(rang_poly,0) += champ_filtre_(le_poly);
113// Contribution P1
114 if (zvef.get_alphaS())
115 {
116 xs = positions(rang_poly,0);
117 ys = positions(rang_poly,1);
118 if (dimension == 3)
119 zs = positions(rang_poly,2);
120
121// Calcul par les fonctions de forme P1 du champ filtre
122// au point (xs,yz,zs)
123 for (int i=0; i<dimension+1; i++)
124 {
125 som = prs+sommet_poly(le_poly,i);
126 double champ_filtre_som=champ_filtre_(som);
127 double li=0.;
128
129 if (dimension == 2)
130 li=coord_barycentrique_P1_triangle(sommet_poly,
131 coord, xs, ys,
132 le_poly, i);
133 else if (dimension == 3)
134 li=coord_barycentrique_P1_tetraedre(sommet_poly,
135 coord, xs, ys, zs,
136 le_poly, i);
137 val(rang_poly,0) += champ_filtre_som * li;
138 }
139 }
140 }
141 }
142 }
143 else //!< nb_compo_ > 1
144 {
145 Cerr << "Vous en etes deja la vous ?" << finl;
146 Process::exit();
147 }
148 return val;
149}

261void assembler(const Zone_VEF_PreP1b& zone_VEF, Matrice& matrice)
262{
263 matrice.typer("Matrice_Morse_Sym");
264 Matrice_Morse_Sym& MatPoisson=ref_cast(Matrice_Morse_Sym, matrice.valeur());
265 int nb_som_tot = zone_VEF.zone().nb_som_tot();
266 int nb_arete_tot = zone_VEF.zone().nb_aretes_tot();
267 int nnz=nb_som_tot+nb_arete_tot;
268 const IntTab& aretes_som=zone_VEF.zone().aretes_som();
269 const ArrOfInt& renum_arete_perio=zone_VEF.get_renum_arete_perio();
270 const Domaine& dom=zone_VEF.zone().domaine();
271
272 IntLists voisins(nb_som_tot);
273 DoubleLists coeffs(nb_som_tot);
274 DoubleVect diag(nb_som_tot);
275// On parcourt toutes les aretes non periodiques:
276 for(int arete=0; arete<nb_arete_tot; arete++)
277 {
278 if (renum_arete_perio[arete]==arete)
279 {
280 int som1=dom.get_renum_som_perio(aretes_som(arete, 0));
281 int som2=dom.get_renum_som_perio(aretes_som(arete, 1));
282 if(som1>som2)
283 {
284 int tmp=som1;
285 som1=som2;
286 som2=tmp;
287 }
288 voisins[som1].add(som2);
289 coeffs[som1].add(1);
290 diag(som1)++;
291 diag(som2)++;
292 }
293 }
294 for (int i=0; i<nb_som_tot; i++)
295 {
296 if (diag(i)==0)
297 {
298 assert(i!=dom.get_renum_som_perio(i));
299 diag(i)=1; //!< Sommets periodiques
300 }
301 }
302 MatPoisson.dimensionner(nb_som_tot, nnz) ;
303 MatPoisson.remplir(voisins, coeffs, diag) ;
304 MatPoisson.compacte() ;
305 MatPoisson.set_est_definie(1);
306// Cerr << "Matrice P1";MatPoisson.imprimer(Cerr);
307}

310static double coeff_inv = 1 / coeff;
311double second_membre(const Zone_VEF_PreP1b& zone_VEF, ArrOfDouble& Pa, DoubleVect& secmem)
312{
313 int nb_arete_tot = zone_VEF.zone().nb_aretes_tot();
314 const IntTab& aretes_som=zone_VEF.zone().aretes_som();
315 const ArrOfInt& renum_arete_perio=zone_VEF.get_renum_arete_perio();
316 const Domaine& dom=zone_VEF.zone().domaine();
317 secmem=0;
318// On parcourt toutes les aretes non periodiques
319 for(int arete=0; arete<nb_arete_tot; arete++)
320 {
321 if (renum_arete_perio[arete]==arete)
322 {
323 double x=Pa[arete];
324 int som1=dom.get_renum_som_perio(aretes_som(arete, 0));
325 int som2=dom.get_renum_som_perio(aretes_som(arete, 1));
326 secmem(som1)+=x;
327 secmem(som2)+=x;
328 }
329 }
330 secmem*=coeff;
331 secmem.echange_espace_virtuel();
332 double norme=mp_norme_vect(secmem);
333// Cerr << "||somme Pa||= " << norme << finl;
334 return norme;
335}

337void corriger(const Zone_VEF_PreP1b& zone_VEF, DoubleTab& champ_filtre_, Matrice& matrice, const int Condition_Neumann_imposee_)
338{
339 int nb_elem=zone_VEF.nb_elem();
340 int nb_som=zone_VEF.nb_som();
341
342// Tableaux d'acces aux valeurs Pk et Ps
343 if (!zone_VEF.get_alphaE() || !zone_VEF.get_alphaS())
344 {
345 Cerr << "Erreur dans Champ_P1iP1B_impl.cpp: corriger(...), le champ doit avoir une partie sommets et elements" << finl;
346 Process::exit();
347 }
348 DoubleTab_parts parties_P(champ_filtre_);
349 DoubleVect& Pk = parties_P[0]; //!< partie elements
350 DoubleVect& Ps = parties_P[1]; //!< partie sommets
351
352// Filtrage si support arete
353 if (zone_VEF.get_alphaA())
354 {
355 if (zone_VEF.get_renum_arete_perio().size_array()==0)
356 {
357 Cerr<<"We try to Champ_P1iP1B_impl::corriger but get_renum_arete_perio"<<finl;
358 }
359 }
360 if (zone_VEF.get_alphaA()&& zone_VEF.get_renum_arete_perio().size_array())
361 {
362 DoubleVect& Pa = parties_P[2]; //!< partie aretes
363
364// Si premier passage on assemble la matrice
365 if (!matrice.non_nul()) assembler(zone_VEF, matrice);
366
367// Construction du second membre
368 DoubleVect secmem;
369 zone_VEF.zone().domaine().creer_tableau_sommets(secmem);
370 second_membre(zone_VEF, Pa, secmem);
371
372// Calcul de la correction
373 DoubleVect solution;
374 solution.copy(secmem, ArrOfDouble::NOCOPY_NOINIT); //!< Copie de la structure
375 solution = 0.;
376
377 SolveurSys solveur;
378 solveur.typer("Solv_GCP");
379 Precond p;
380 p.typer("SSOR");
381 ref_cast(Solv_GCP,solveur.valeur()).set_precond(p);
382 solveur.nommer("Pa_filter_solver");
383 solveur.resoudre_systeme(matrice, secmem, solution);
384
385// Application de la periodicite sur la solution:
386 const Domaine& dom=zone_VEF.zone().domaine();
387 for(int i=0; i<nb_som; i++)
388 {
389 int som=dom.get_renum_som_perio(i);
390 if (som!=i)
391 solution(i)=solution(som);
392 }
393
394// Correction de Ps:
395 for(int som=0; som<nb_som; som++)
396 Ps(som) += solution(som);
397
398// Correction de Pa:
399 int nb_arete = zone_VEF.zone().nb_aretes();
400 const IntTab& aretes_som=zone_VEF.zone().aretes_som();
401 const ArrOfInt& ok_arete=zone_VEF.get_ok_arete();
402 for(int arete=0; arete<nb_arete; arete++)
403 {
404 if(!ok_arete[arete] && Pa(arete)!=0)
405 {
406 Cerr << "Pa(arete_superflue)!=0 dans Champ_P1iP1B_implementation::corriger" << finl;
407 Cerr << "Contacter le support TRUST." << finl;
408 Cerr << "S'il s'agit d'une reprise au format xyz d'un calcul" << finl;
409 Cerr << "alors il faut ecraser le fichier .ok_arete de votre calcul de reprise" << finl;
410 Cerr << "par le fichier .ok_arete de votre calcul precedant afin d'avoir les" << finl;
411 Cerr << "memes aretes superflues." << finl;
412 Process::exit();
413 }
414 int som1=aretes_som(arete, 0);
415 int som2=aretes_som(arete, 1);
416 double sigma=solution(som1)+solution(som2);
417 Pa(arete)-=sigma*coeff_inv;
418 }
419
420// Correction de Pk:
421 const IntTab& les_elems = zone_VEF.zone().les_elems();
422 int nb_som_par_elem=les_elems.dimension_tot(1);
423 for(int elem=0; elem<nb_elem; elem++)
424 {
425 double sigma=0;
426 for (int som=0; som<nb_som_par_elem; som++)
427 sigma+=solution(les_elems(elem,som));
428 Pk(elem)-=sigma/4;
429 }
430 }
431
432// Filtrage L2:
433 double moyenne_K = mp_moyenne_vect(Pk); //!< Calcul de la moyenne du champ aux elements
434 Pk -= moyenne_K; //!< Correction des elements du champ_filtre_
435 Ps += moyenne_K; //!< Correction des sommets du champ_filtre_
436
437// On retranche a Ps sa moyenne si pas de Cl de Neumann pour avoir une moyenne nulle
438// Il faudrait peut etre mieux faire cela dans Assembleur_P_VEFPreP1B::modifier_solution
439// si cela est possible...
440 assert(Condition_Neumann_imposee_!=-1);
441
442 if (!Condition_Neumann_imposee_)
443 {
444// L'espace virtuel de Ps n'a pas besoin d'etre a jour pour calculer correctement la moyenne
445 Ps -= mp_moyenne_vect(Ps); //!< Correction des sommets du champ_filtre_
446 }
447
448 champ_filtre_.echange_espace_virtuel(); //!< Mise a jour des espaces virtuels
449}

517// du centre de gravite des faces de la frontiere

519 for (int dim=0; dim<Objet_U::dimension; dim++)

527
528 for (int i=0; i<nb_faces_fr; i++)
529 {
530 x(i,0) = val_interp(i);
531 }
532// Useless ? x.echange_espace_virtuel();
