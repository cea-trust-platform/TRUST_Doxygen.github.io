122//! Associe l'operateur a la zone_dis, la zone_Cl_dis,<br>et a l'inconnue de son equation.
123void Operateur_base::completer()
124{
125 assert(mon_equation.non_nul());
126 const Equation_base& eqn = equation();
127 const Zone_dis& zdis= eqn.zone_dis();
128
129 const Zone_Cl_dis& zcl = le_champ_inco.non_nul() ? le_champ_inco.valeur()->zone_Cl_dis() : eqn.zone_Cl_dis();
130 const Champ_Inc& inco = le_champ_inco.non_nul() ? le_champ_inco.valeur() : eqn.inconnue();
131 associer(zdis, zcl, inco);
132 const Conds_lim& les_cl = zcl->les_conditions_limites();
133 for (int i = 0; i < les_cl.size(); i++)
134 {
135 const Frontiere_dis_base& la_fr = les_cl[i].frontiere_dis();
136 col_width_ = std::max(col_width_, la_fr.le_nom().longueur());
137 }
138 int w_suffix = 3; //!< pour ajout _Mx (moment)
139
140// pour les champs a plusieurs composantes, le header des colonnes des fichiers .out prend la forme
141// Time cl1_compo1 cl1_compo2 ...
142// on prend en compte la longueur de compo1, compo2, etc...
143 Noms noms_compo_courts(inco->noms_compo());
144 if (noms_compo_courts.size() > 1)
145 for (int i = 0; i < noms_compo_courts.size(); ++i)
146 {
147 noms_compo_courts[i] = Motcle(noms_compo_courts[i]).getSuffix(eqn.inconnue().le_nom());
148 w_suffix = std::max(w_suffix, noms_compo_courts[i].longueur());
149 }
150 col_width_ += w_suffix;
151}

409//! Ouverture/creation d'un fichier d'impression d'un operateur<br>A surcharger dans les classes derivees.
410void Operateur_base::ouvrir_fichier(SFichier& os,const Nom& type, const int flag) const
411{
412
413// flag nul on n'ouvre pas le fichier
414 if (flag==0)
415 return ;
416
417 if (!je_suis_maitre())
418 {
419 Cerr<<"Only master process can open "<<type<<" file. See Operateur_base::ouvrir_fichier()" << finl;
420 Process::abort();
421 }
422 const Equation_base& eqn = equation();
423 const Probleme_base& pb=eqn.probleme();
424 const Schema_Temps_base& sch=pb.schema_temps();
425 const int precision = sch.precision_impr(), wcol = std::max(col_width_, sch.wcol()), gnuplot_header = sch.gnuplot_header();
426 os.set_col_width(wcol);
427 Nom nomfichier(out_);
428 if (type!="") nomfichier+=(Nom)"_"+type;
429 nomfichier+=".out";
430 struct stat f;
431// On cree le fichier a la premiere impression avec l'en tete ou si le fichier n'existe pas
432 if (stat(nomfichier,&f) || (sch.nb_impr()==1 && !pb.reprise_effectuee()))
433 {
434 os.ouvrir(nomfichier);
435
436 std::vector<std::string> comp_m = {"_Mx", "_My", "_Mz"};
437 os << (Nom)"# Printing on the boundaries of the equation "+eqn.que_suis_je()+" of the problem "+eqn.probleme().le_nom() << finl;
438 os << "# " << (type=="moment" ? "Moment of " : "") << description() << finl;
439
440 if (!gnuplot_header) os << "#";
441 os.set_col_width(wcol - !gnuplot_header);
442 os.add_col("Time");
443 os.set_col_width(wcol);
444 const Conds_lim& les_cls=eqn.inconnue()->zone_Cl_dis().les_conditions_limites();
445
446 if (flux_bords_.nb_dim()!=2)
447 {
448 Cerr << "flux_bords_ not dimensioned for the operator " << que_suis_je() << finl;
449 Process::exit();
450 }
451
452// s'il y a plusieurs composantes par CL, on se sert des noms de composante de l'inconnue
453 int nb_compo = flux_bords_.line_size();
454 if (type=="moment" && dimension == 2) nb_compo=1;
455 Noms noms_compo_courts(eqn.inconnue()->noms_compo());
456 if (nb_compo > 1)
457 for (int i = 0; i < noms_compo_courts.size(); ++i)
458 noms_compo_courts[i] = Motcle(noms_compo_courts[i]).getSuffix(eqn.inconnue().le_nom());
459
460// ecriture de l'entete des colonnes de la forme
461// Time cl1 cl2 cl3 ...
462// ou s'il y a plusieurs composantes :
463// Time cl1_compo1 cl1_compo2 cl2 compo1 cl2_compo2 ...
464 for (int num_cl=0; num_cl<les_cls.size(); num_cl++)
465 {
466 const Frontiere_dis_base& la_fr = les_cls[num_cl].frontiere_dis();
467 if (type!="sum" || eqn.zone_dis().zone().Bords_a_imprimer_sum().contient(la_fr.le_nom()))
468 {
469 Nom ch = la_fr.le_nom();
470 if (type=="moment")
471 {
472 if (dimension==2)
473 os.add_col((ch + comp_m[2]).getChar());
474 else
475 for (int d = 0; d < dimension; ++d)
476 os.add_col((ch + comp_m[d]).getChar());
477 }
478 else
479 {
480 if (nb_compo > 1)
481 for (int d = 0; d < nb_compo; ++d)
482 {
483 os.add_col((ch + "_" + noms_compo_courts[d]).getChar());
484 }
485 else os.add_col(ch.getChar());
486// if periodic BC, we write the boundary name twice on the header of .out files
487 if ((sub_type(Periodique,les_cls[num_cl].valeur())) && (Objet_U::nom_du_cas()+"_"+eqn.probleme().le_nom()+"_Force_pression"!=out_))
488 {
489 if (nb_compo > 1)
490 for (int d = 0; d < nb_compo; ++d)
491 {
492 os.add_col((ch + "_" + noms_compo_courts[d]).getChar());
493 }
494 else os.add_col(ch.getChar());
495 }
496 }
497 }
498 }
499 if (type!="sum" && type!="moment")
500 if (Objet_U::nom_du_cas()+"_"+eqn.probleme().le_nom()+"_Force_pression"!=out_)
501 {
502 if (nb_compo > 1)
503 for (int d = 0; d < nb_compo; ++d)
504 os.add_col((Nom("Total_") + noms_compo_courts[d]).getChar());
505 else os.add_col("Total");
506 }
507
508 os << finl;
509 }
510// Sinon on l'ouvre
511 else
512 {
513 os.ouvrir(nomfichier,ios::app);
514 }
515 os.precision(precision);
516 os.setf(ios::scientific);
517}

589// Calcul du tableau de valeurs d une quantite lie a un operateur pour postraitement
590// Options reconnues : "stabilite" pour dt_stab
591// "flux_bords" ou "flux_surfacique_bords" pour flux_bords_
592//
593void Operateur_base::calculer_pour_post(Champ& espace_stockage,const Nom& option, int comp) const
594{
595 if (Motcle(option)=="flux_bords" || Motcle(option)=="flux_surfacique_bords")
596 {
597 bool surfacique = (Motcle(option)=="flux_surfacique_bords");
598 DoubleTab& es_valeurs = espace_stockage->valeurs();
599 es_valeurs = 0.;
600 const Zone_Cl_dis_base& zcl_dis = equation().zone_Cl_dis();
601 const Zone_dis_base& zdis = equation().zone_dis().valeur();
602 int nb_front = zdis.nb_front_Cl();
603
604 if (flux_bords_.nb_dim()==2)
605 {
606 DoubleVect aire;
607 for (int n_bord=0; n_bord<nb_front; n_bord++)
608 {
609 const Cond_lim& la_cl = zcl_dis.les_conditions_limites(n_bord);
610 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
611 if (surfacique) la_cl.frontiere_dis().frontiere().faces().calculer_surfaces(aire);
612 int ndeb = le_bord.num_premiere_face();
613 int nfin = ndeb + le_bord.nb_faces();
614
615 for (int face=ndeb; face<nfin; face++)
616 es_valeurs(face) = flux_bords_(face,comp) / (surfacique ? aire(face-ndeb) : 1.);
617 }
618 }
619 }
