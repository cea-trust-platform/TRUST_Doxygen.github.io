77 {

80
81 if(flush_left_)

87 *out_stream_ << std::setw(column_widths_.at(j_))
88 << input;
89
90 if (j_ == get_num_columns()-1)
91 {
92 *out_stream_ << "│\n";
93 i_ = i_ + 1;
94 j_ = 0;
95 }
96 else
97 {
98 *out_stream_ << separator_;
99 j_ = j_ + 1;
100 }
101
102 return *this;
103 }
