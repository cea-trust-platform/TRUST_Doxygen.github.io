28template <Champ_Don_Type _TYPE_>
29void TRUSTChamp_Don_generique<_TYPE_>::mettre_a_jour(double t)
30{
31 static constexpr bool IS_TXYZ = (_TYPE_ == Champ_Don_Type::TXYZ);
32
33 if (!IS_TXYZ) /* XYZ ou LU */
34 Champ_Don_base::mettre_a_jour(t);
35 else
36 {
37 changer_temps(t);
38 int nb_elems = mon_domaine->zone(0).nb_elem();
39 DoubleTab& mes_val = valeurs();
40 DoubleTab positions(nb_elems, dimension);
41 mettre_a_jour_positions(positions);
42 eval_fct(positions, temps(), mes_val);
43 mes_val.echange_espace_virtuel();
44 }
45}

47template <Champ_Don_Type _TYPE_>
48double TRUSTChamp_Don_generique<_TYPE_>::valeur_a_compo(const DoubleVect& x, int ncomp) const
49{
50 static constexpr bool IS_XYZ = (_TYPE_ == Champ_Don_Type::XYZ);
51 if (!IS_XYZ) return Champ_Don_base::valeur_a_compo(x, ncomp); //!< appel simple si TXYZ ou LU !
52
53 if (ncomp > nb_compo_) erreur_champ_(__func__); //!< asking ncomp > nb_compo_ ?????
54
55 DoubleTab positions(1, dimension);
56 DoubleTab val_fct(1);
57 positions(0, 0) = x(0);
58 positions(0, 1) = x(1);
59 if (dimension > 2) positions(0, 2) = x(2);
60
61 eval_fct(positions, val_fct, ncomp);
62 return val_fct(0);
63}

65template <Champ_Don_Type _TYPE_>
66void TRUSTChamp_Don_generique<_TYPE_>::mettre_a_jour_positions(DoubleTab& positions)
67{
68 double x, y, z;
69 const int nb_elems = mon_domaine->zone(0).nb_elem();
70 const IntTab& les_Polys = mon_domaine->zone(0).les_elems();
71 for (int num_elem = 0; num_elem < nb_elems; num_elem++)
72 {
73 x = y = z = 0;
74 int nb_som = 0;
75 for (int s = 0, num_som; s < les_Polys.dimension(1); s++)
76 if ((num_som = les_Polys(num_elem, s)) >= 0)
77 {
78 x += mon_domaine->coord(num_som, 0);
79 y += mon_domaine->coord(num_som, 1);
80 if (dimension > 2) z += mon_domaine->coord(num_som, 2);
81 nb_som++;
82 }
83 positions(num_elem, 0) = x / nb_som;
84 positions(num_elem, 1) = y / nb_som;
85 if (dimension > 2) positions(num_elem, 2) = z / nb_som;
86 }
87}

252template <Champ_Don_Type _TYPE_> template<Champ_Don_Type T>
253enable_if_t<T == Champ_Don_Type::LU, DoubleTab&>
254TRUSTChamp_Don_generique<_TYPE_>::valeur_aux_elems_(const DoubleTab&, const IntVect& les_polys, DoubleTab& val) const
255{
256 if (val.nb_dim() > 2) erreur_champ_(__func__); //!< DoubleTab val don't have 2 entries
257
258 int p;
259 const DoubleTab& ch = valeurs();
260 val = 0.;
261
262 for (int rang_poly = 0; rang_poly < les_polys.size(); rang_poly++)
263 if ((p = les_polys(rang_poly)) != -1)
264 for (int n = 0; n < nb_compo_; n++) val(rang_poly, n) = ch(p, n);
265
266 return val;
