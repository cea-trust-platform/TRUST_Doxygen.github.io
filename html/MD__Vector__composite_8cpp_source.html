79static void append_items(ArrOfInt& dest, const ArrOfInt& src, int offset = 0, int multiplier = 1)
80{
81 int i = dest.size_array();
82 const int n = src.size_array();
83 dest.resize_array(i + n * multiplier);
84 for (int j = 0; j < n; j++)
85 {
86 int x = offset + src[j] * multiplier;
87 for (int k = 0; k < multiplier; k++)
88 dest[i++] = x++;
89 }
90}

92static void append_blocs(ArrOfInt& dest, const ArrOfInt& src, int offset = 0, int multiplier = 1)
93{
94 int i = dest.size_array();
95 const int n = src.size_array();
96 dest.resize_array(i + n);
97 for (int j = 0; j < n; j++, i++)
98 dest[i] = offset + src[j] * multiplier;
99}

102static int find_in_array(const ArrOfInt& a, int x)
103{
104 int n;
105 for (n = a.size_array() - 1; n >= 0; n--)

107 break;

111// Append all items in "src" to the "dest" descriptor, duplicating src items "multiplier" times
112// and adding the "offset" to item indexes. Also merge pe_voisins_ lists.
113static void append_global_md(MD_Vector_std& dest, const MD_Vector_std& src, int offset, int multiplier)
114{
115// List sizes for items_to_send_, items_to_recv_ and blocs_to_recv_ of new descriptor
116 ArrOfInt x_sz, y_sz, z_sz;
117// Data of these lists:
118 ArrOfInt x_data, y_data, z_data;
119 ArrOfInt new_blocs_items_count, new_nb_items_to_items;
120 x_sz.set_smart_resize(1);
121 y_sz.set_smart_resize(1);
122 z_sz.set_smart_resize(1);
123 x_data.set_smart_resize(1);
124 y_data.set_smart_resize(1);
125 z_data.set_smart_resize(1);
126 new_blocs_items_count.set_smart_resize(1);
127 new_nb_items_to_items.set_smart_resize(1);
128
129 ArrOfInt tmp;
130 tmp.set_smart_resize(1);
131
132 ArrOfInt pe_list(dest.pe_voisins_);
133 pe_list.set_smart_resize(1);
134 append_blocs(pe_list, src.pe_voisins_);
135 array_trier_retirer_doublons(pe_list);
136 const int np = pe_list.size_array();
137
138 new_blocs_items_count.resize_array(np);
139 new_nb_items_to_items.resize_array(np);
140
141 for (int i = 0; i < np; i++)
142 {
143// Number of items to recv, send and blocs to recv for this processor:
144 int nx = 0, ny = 0, nz = 0;
145 int blocs_count = 0;
146// Is processor pe_list[i] in initial "dest" descriptor ?
147 int j;
148
149 for (j = dest.pe_voisins_.size_array()-1; j >= 0; j--)
150 if (dest.pe_voisins_[j] == pe_list[i])
151 break;
152 const int pe = pe_list[i];
153
154// Insert items to send and items to recv from dest
155 j = find_in_array(dest.pe_voisins_, pe);
156 if (j >= 0)
157 {
158 const int count_items_single = dest.nb_items_to_items_[j];
159 nx += count_items_single;
160// take only single items to send
161 tmp.resize_array(0);
162 tmp.resize_array(count_items_single);
163 for (int k = 0; k < count_items_single; k++)
164 tmp[k] = dest.items_to_send_(j, k);
165 append_blocs(x_data, tmp);
166
167 ny += dest.items_to_recv_.get_list_size(j);
168 dest.items_to_recv_.copy_list_to_array(j, tmp);
169 append_blocs(y_data, tmp);
170 }
171
172// Insert items to send and items to recv from src, with multiplier
173 j = find_in_array(src.pe_voisins_, pe);
174 if (j >= 0)
175 {
176 const int count_items_single = src.nb_items_to_items_[j];
177 nx += count_items_single * multiplier;
178// take only single items to send
179 tmp.resize_array(0);
180 tmp.resize_array(count_items_single);
181 for (int k = 0; k < count_items_single; k++)
182 tmp[k] = src.items_to_send_(j, k);
183 append_items(x_data, tmp, offset, multiplier);
184
185 ny += src.items_to_recv_.get_list_size(j) * multiplier;
186 src.items_to_recv_.copy_list_to_array(j, tmp);
187 append_items(y_data, tmp, offset, multiplier);
188 }
189 const int single_items_count = nx;
190// Insert blocs to send and blocs to recv from dest
191 j = find_in_array(dest.pe_voisins_, pe);
192 if (j >= 0)
193 {
194 const int count_items_tot = dest.items_to_send_.get_list_size(j);
195 const int count_items_single = dest.nb_items_to_items_[j];
196 nx += count_items_tot - count_items_single;
197 tmp.resize_array(0);
198 tmp.resize_array(count_items_tot - count_items_single);
199 for (int k = count_items_single; k < count_items_tot; k++)
200 tmp[k - count_items_single] = dest.items_to_send_(j, k);
201 append_blocs(x_data, tmp);
202
203 nz += dest.blocs_to_recv_.get_list_size(j);
204 dest.blocs_to_recv_.copy_list_to_array(j, tmp);
205 append_blocs(z_data, tmp);
206
207 blocs_count += dest.blocs_items_count_[j];
208 }
209// Insert blocs to send and blocs to recv from src, with multiplier
210 j = find_in_array(src.pe_voisins_, pe);
211 if (j >= 0)
212 {
213 const int count_items_tot = src.items_to_send_.get_list_size(j);
214 const int count_items_single = src.nb_items_to_items_[j];
215 nx += (count_items_tot - count_items_single) * multiplier;
216 tmp.resize_array(0);
217 tmp.resize_array(count_items_tot - count_items_single);
218 for (int k = count_items_single; k < count_items_tot; k++)
219 tmp[k - count_items_single] = src.items_to_send_(j, k);
220 append_items(x_data, tmp, offset, multiplier);
221
222 nz += src.blocs_to_recv_.get_list_size(j);
223 src.blocs_to_recv_.copy_list_to_array(j, tmp);
224 append_blocs(z_data, tmp, offset, multiplier);
225
226 blocs_count += src.blocs_items_count_[j] * multiplier;
227 }
228
229 x_sz.append_array(nx);
230 y_sz.append_array(ny);
231 z_sz.append_array(nz);
232 new_blocs_items_count[i] = blocs_count;
233 new_nb_items_to_items[i] = single_items_count;
234 }
235
236 dest.items_to_send_.set_list_sizes(x_sz);
237 dest.items_to_send_.set_data(x_data);
238 dest.items_to_recv_.set_list_sizes(y_sz);
239 dest.items_to_recv_.set_data(y_data);
240 dest.blocs_to_recv_.set_list_sizes(z_sz);
241 dest.blocs_to_recv_.set_data(z_data);
242 dest.pe_voisins_ = pe_list;
243 dest.blocs_items_count_ = new_blocs_items_count;
244 dest.nb_items_to_items_ = new_nb_items_to_items;
245}

283//! Append the "part" descriptor to the composite vector.<br>By default, shape=0 says that for each item in "part", one item will<br>be appended to the composite descriptor.<br>If shape=n, each items in "part" will be duplicated n times in the<br>composite descriptor.<br>The difference between 0 and 1 is when we create parts with DoubleTab_parts<br>(see class DoubleTab_parts)<br>Exemples:<br>1) P1Bubble pressure fields have one, two or three parts with shape=1.<br>(see Zone_VEF_PreP1b::discretiser_suite())<br>2) One can create a composite field containing<br>- velocity degrees of freedom at the faces<br>- pressure d.o.f. at the elements<br>- one multiscalar d.o.f at nodes<br>MD_Vector_composite mdc;<br>mdc.add_part(zonevf.md_vector_faces(), dimension);<br>mdc.add_part(zone.md_vector_elements(), 0 /* scalar */);<br>mdc.add_part(domaine.md_vector_sommets(), 1 /* multiscalar with 1 component */);<br>MD_Vector md;<br>md.copy(mdc);<br>I will create a DoubleVect of this form by attaching this descriptor:<br>DoubleVect v;<br>MD_Vector_tools::creer_tableau_distribue(md, v);<br>DoubleTab_Parts parts(v);<br>// parts[0] has nb_dim()==2, dimensions are [ zonevf.nb_faces_tot, 3 ]<br>// parts[1] has nb_dim()==1, dimension [ zone.nb_elem_tot ] /* speciel case shape==0 */<br>// parts[2] has nb_dim()==2, dimension [ domaine.nb_som_tot, 1 ]<br>I can duplicate items in the descriptor by creating DoubleTab objects:<br>The following code will create an array containing 12 d.o.f for each face<br>followed by 4 d.o.f for each element and 4 d.o.f for each node:<br>DoubleTab tab(0, 4);<br>MD_Vector_tools::creer_tableau_distribue(md, tab);<br>DoubleTab_Parts parts(tab);<br>// parts[0] has nb_dim()==3, dimensions are [ zonevf.nb_faces_tot, 3, 4 ]<br>// parts[1] has nb_dim()==2, dimension [ zone.nb_elem_tot, 4 ] /* speciel case shape==0 */<br>// parts[2] has nb_dim()==3, dimension [ domaine.nb_som_tot, 1, 4 ]
284void MD_Vector_composite::add_part(const MD_Vector& part, int shape, Nom name)
285{
286 assert(part.non_nul());
287 assert(shape >= 0);
288 if (data_.size() == 0)
289 {
290 nb_items_tot_ = 0;
291 nb_items_seq_tot_ = 0;
292 nb_items_seq_local_ = 0;
293 }
294 const int multiplier = (shape == 0) ? 1 : shape;
295 data_.add(part);
296 const int offset = nb_items_tot_;
297 parts_offsets_.set_smart_resize(1);
298 parts_offsets_.append_array(offset);
299 shapes_.set_smart_resize(1);
300 shapes_.append_array(shape);
301 names_.add(name);
302
303 nb_items_tot_ += part.valeur().get_nb_items_tot() * multiplier;
304 if (data_.size() > 1 || part.valeur().get_nb_items_reels() < 0)
305 nb_items_reels_ = -1; //!< size() will be invalid for arrays with this descriptor
306 else
307 nb_items_reels_ = part.valeur().get_nb_items_reels() * multiplier;
308 nb_items_seq_tot_ += part.valeur().nb_items_seq_tot() * multiplier;
309 nb_items_seq_local_ += part.valeur().nb_items_seq_local() * multiplier;
310
311 append_blocs(blocs_items_to_sum_, part.valeur().get_items_to_sum(), offset, multiplier);
312 append_blocs(blocs_items_to_compute_, part.valeur().get_items_to_compute(), offset, multiplier);
313
314 if (sub_type(MD_Vector_std, part.valeur()))
315 append_global_md(global_md_, ref_cast(MD_Vector_std, part.valeur()), offset, multiplier);
316 else if (sub_type(MD_Vector_composite, part.valeur()))
317 append_global_md(global_md_, ref_cast(MD_Vector_composite, part.valeur()).global_md_, offset, multiplier);
318 else
319 {
320 Cerr << "Internal error in MD_Vector_composite::add_part: unknown part type " << part.valeur().que_suis_je() << finl;
321 Process::exit();
322 }
323}
