107void Champ_P1NC::cal_rot_ordre1(DoubleTab& vorticite) const
108{
109 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
110 const int nb_elem = zone_VEF.nb_elem();
111 const int nb_elem_tot = zone_VEF.nb_elem_tot();
112 DoubleTab gradient_elem(nb_elem_tot,dimension,dimension);
113
114 gradient_elem=0.;
115 vorticite=0;
116 gradient(gradient_elem);
117 int num_elem;
118
119 switch(dimension)
120 {
121 case 2 :
122 {
123 for (num_elem=0; num_elem<nb_elem; num_elem++)
124 {
125 vorticite(num_elem)=gradient_elem(num_elem,1,0)-gradient_elem(num_elem,0,1);
126 }
127 }
128 break;
129 case 3 :
130 {
131 for (num_elem=0; num_elem<nb_elem; num_elem++)
132 {
133 vorticite(num_elem,0)=gradient_elem(num_elem,2,1)-gradient_elem(num_elem,1,2);
134 vorticite(num_elem,1)=gradient_elem(num_elem,0,2)-gradient_elem(num_elem,2,0);
135 vorticite(num_elem,2)=gradient_elem(num_elem,1,0)-gradient_elem(num_elem,0,1);
136 }
137 }
138 }
139
140 vorticite.echange_espace_virtuel();
141 return ;

144void calculer_gradientP1NC(const DoubleTab& variable,
145 const Zone_VEF& zone_VEF,
146 const Zone_Cl_VEF& zone_Cl_VEF,
147 DoubleTab& gradient_elem)
148{
149 const DoubleTab& face_normales = zone_VEF.face_normales();
150 const int nb_faces = zone_VEF.nb_faces_tot();
151 const int nb_elem = zone_VEF.nb_elem_tot();
152 const IntTab& face_voisins = zone_VEF.face_voisins();
153 int premiere_face_int = zone_VEF.premiere_face_int();
154
155 int dimension=Objet_U::dimension;
156 const int nb_comp = variable.line_size();
157
158 gradient_elem=0.;
159 int icomp,fac,i,elem1,elem2,elem,num_face;
160 int n_bord;
161 int ind_face;
162
163
164// Cas du calcul du gradient d'un tableau de vecteurs
165 if (nb_comp!=1)
166 {
167 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
168 {
169 const Cond_lim& la_cl =
170 zone_Cl_VEF.les_conditions_limites(n_bord);
171 const Front_VF& le_bord =
172 ref_cast(Front_VF,la_cl.frontiere_dis());
173 int num1 = 0;
174 int num2 = le_bord.nb_faces_tot();
175 int nb_faces_bord = le_bord.nb_faces();
176
177 if (sub_type(Periodique,la_cl.valeur()))
178 {
179 for (ind_face=num1; ind_face<num2; ind_face++)
180 {
181 num_face = le_bord.num_face(ind_face);
182 elem1=face_voisins(num_face,0);
183 elem2=face_voisins(num_face,1);
184 for (icomp=0; icomp<nb_comp; icomp++)
185 for (i=0; i<dimension; i++)
186 {
187 if (ind_face<nb_faces_bord)
188 {
189 gradient_elem(elem1, icomp, i) +=
190 0.5*face_normales(num_face,i)*variable(num_face,icomp);
191 gradient_elem(elem2, icomp, i) -=
192 0.5*face_normales(num_face,i)*variable(num_face,icomp);
193 }
194 else
195 {
196// Correction pour prise en compte des faces virtuelles periodiques
197// On retranche la contribution en exces ajoutee quand on traite
198// les faces entre nb_faces et nb_faces_tot (dans le cas des faces periodiques)
199
200 gradient_elem(elem1, icomp, i) -=
201 0.5*face_normales(num_face,i)*variable(num_face,icomp);
202 gradient_elem(elem2, icomp, i) +=
203 0.5*face_normales(num_face,i)*variable(num_face,icomp);
204 }
205 }
206 }
207 }
208 else
209 for (ind_face=num1; ind_face<nb_faces_bord; ind_face++)
210 {
211 fac = le_bord.num_face(ind_face);
212 elem1=face_voisins(fac,0);
213 for (icomp=0; icomp<nb_comp; icomp++)
214 for (i=0; i<dimension; i++)
215 {
216 gradient_elem(elem1, icomp, i) +=
217 face_normales(fac,i)* variable(fac,icomp);
218 }
219 }
220 }
221 for (fac=premiere_face_int; fac<nb_faces; fac++)
222 {
223 elem1=face_voisins(fac,0);
224 elem2=face_voisins(fac,1);
225// Deroulement de la boucle pour optimiser un peu (-20% sur 12000 mailles):
226 if (elem1>=0 && elem2>=0)
227 {
228 if (dimension==2)
229 {
230 for (icomp=0; icomp<nb_comp; icomp++)
231 {
232 double var = variable(fac,icomp);
233 gradient_elem(elem1, icomp, 0) += face_normales(fac,0)*var;
234 gradient_elem(elem2, icomp, 0) -= face_normales(fac,0)*var;
235 gradient_elem(elem1, icomp, 1) += face_normales(fac,1)*var;
236 gradient_elem(elem2, icomp, 1) -= face_normales(fac,1)*var;
237 }
238 }
239 else
240 {
241 for (icomp=0; icomp<nb_comp; icomp++)
242 {
243 double var = variable(fac,icomp);
244 gradient_elem(elem1, icomp, 0) += face_normales(fac,0)*var;
245 gradient_elem(elem2, icomp, 0) -= face_normales(fac,0)*var;
246 gradient_elem(elem1, icomp, 1) += face_normales(fac,1)*var;
247 gradient_elem(elem2, icomp, 1) -= face_normales(fac,1)*var;
248 gradient_elem(elem1, icomp, 2) += face_normales(fac,2)*var;
249 gradient_elem(elem2, icomp, 2) -= face_normales(fac,2)*var;
250 }
251 }
252 }
253 else
254 {
255 for (icomp=0; icomp<nb_comp; icomp++)
256 for (i=0; i<dimension; i++)
257 {
258 if (elem1>=0) gradient_elem(elem1, icomp, i) += face_normales(fac,i)*variable(fac,icomp);
259 if (elem2>=0) gradient_elem(elem2, icomp, i) -= face_normales(fac,i)*variable(fac,icomp);
260 }
261 }
262 }
263
264 }
265// Cas du calcul du gradient d'un tableau de scalaire dans un tableau gradient_elem(,,)
266 else if (gradient_elem.nb_dim()==3)
267 {
268 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
269 {
270 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
271 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
272 int num1 = 0;
273 int num2 = le_bord.nb_faces_tot();
274 int nb_faces_bord = le_bord.nb_faces();
275
276 if (sub_type(Periodique,la_cl.valeur()))
277 {
278 for (ind_face=num1; ind_face<num2; ind_face++)
279 {
280 num_face = le_bord.num_face(ind_face);
281 elem1=face_voisins(num_face,0);
282 elem2=face_voisins(num_face,1);
283 for (i=0; i<dimension; i++)
284 {
285 if (ind_face<nb_faces_bord)
286 {
287 gradient_elem(elem1, 0, i) +=
288 0.5*face_normales(num_face,i)*variable(num_face);
289 gradient_elem(elem2, 0, i) -=
290 0.5*face_normales(num_face,i)*variable(num_face);
291 }
292 else
293 {
294// Correction pour prise en compte des faces virtuelles periodiques
295// On retranche la contribution en exces ajoutee quand on traite
296// les faces entre nb_faces et nb_faces_tot (dans le cas des faces periodiques)
297 gradient_elem(elem1, 0, i) -=
298 0.5*face_normales(num_face,i)*variable(num_face);
299 gradient_elem(elem2, 0, i) +=
300 0.5*face_normales(num_face,i)*variable(num_face);
301
302 }
303
304 }
305 }
306 }
307 else
308 for (ind_face=num1; ind_face<nb_faces_bord; ind_face++)
309 {
310 fac = le_bord.num_face(ind_face);
311 elem1=face_voisins(fac,0);
312 for (i=0; i<dimension; i++)
313 {
314 gradient_elem(elem1, 0, i) +=
315 face_normales(fac,i)* variable(fac);
316 }
317 }
318 }
319 for (fac=premiere_face_int; fac<nb_faces; fac++)
320 {
321 elem1=face_voisins(fac,0);
322 elem2=face_voisins(fac,1);
323 for (i=0; i<dimension; i++)
324 {
325 if (elem1>=0)
326 gradient_elem(elem1, 0, i) += face_normales(fac,i)*variable(fac);
327 if (elem2>=0)
328 gradient_elem(elem2, 0, i) -= face_normales(fac,i)*variable(fac);
329
330 }
331 }
332 }
333// Cas du calcul du gradient d'un tableau de scalaire dans un tableau gradient_elem(,)
334 else
335 {
336 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
337 {
338 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
339 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
340 int num1 = 0;
341 int num2 = le_bord.nb_faces_tot();
342 int nb_faces_bord = le_bord.nb_faces();
343
344 if (sub_type(Periodique,la_cl.valeur()))
345 {
346 for (ind_face=num1; ind_face<num2; ind_face++)
347 {
348 num_face = le_bord.num_face(ind_face);
349 elem1=face_voisins(num_face,0);
350 elem2=face_voisins(num_face,1);
351 for (i=0; i<dimension; i++)
352 {
353 if (ind_face<nb_faces_bord)
354 {
355 gradient_elem(elem1, i) +=
356 0.5*face_normales(num_face,i)*variable(num_face);
357 gradient_elem(elem2, i) -=
358 0.5*face_normales(num_face,i)*variable(num_face);
359 }
360 else
361 {
362// Correction pour prise en compte des faces virtuelles periodiques
363// On retranche la contribution en exces ajoutee quand on traite
364// les faces entre nb_faces et nb_faces_tot (dans le cas des faces periodiques)
365 gradient_elem(elem1,i) -=
366 0.5*face_normales(num_face,i)*variable(num_face);
367 gradient_elem(elem2,i) +=
368 0.5*face_normales(num_face,i)*variable(num_face);
369 }
370 }
371 }
372 }
373 else
374 for (ind_face=num1; ind_face<nb_faces_bord; ind_face++)
375 {
376 fac = le_bord.num_face(ind_face);
377 elem1=face_voisins(fac,0);
378 for (i=0; i<dimension; i++)
379 {
380 gradient_elem(elem1, i) +=
381 face_normales(fac,i)* variable(fac);
382 }
383 }
384 }
385 for (fac=premiere_face_int; fac<nb_faces; fac++)
386 {
387 elem1=face_voisins(fac,0);
388 elem2=face_voisins(fac,1);
389 for (i=0; i<dimension; i++)
390 {
391 if (elem1>=0)
392 gradient_elem(elem1, i) += face_normales(fac,i)*variable(fac);
393 if (elem2>=0)
394 gradient_elem(elem2, i) -= face_normales(fac,i)*variable(fac);
395
396 }
397 }
398 }
399
400 const DoubleVect& inverse_volumes = zone_VEF.inverse_volumes();
401 if (gradient_elem.nb_dim()==3)
402 for ( elem=0; elem<nb_elem; elem++)
403 for (icomp=0; icomp<nb_comp; icomp++)
404 for (i=0; i<dimension; i++)
405 gradient_elem(elem,icomp,i) *= inverse_volumes(elem);
406 else
407 for ( elem=0; elem<nb_elem; elem++)
408 for (i=0; i<dimension; i++)
409 gradient_elem(elem,i) *= inverse_volumes(elem);
410}

494 ndeb = le_bord.num_premiere_face();
495 nfin = ndeb + le_bord.nb_faces();

497 double moy;
498
499 for (num_face=ndeb; num_face<nfin; num_face++)
500 {
501 voisine = la_cl_perio.face_associee(num_face-ndeb) + ndeb;
502 for (int comp=0; comp<nb_compo; comp++)
503 {
504 if (ch_tab(num_face,comp) != ch_tab(voisine,comp))
505 {
506 moy = 0.5*(ch_tab(num_face,comp) + ch_tab(voisine,comp));
507 ch_tab(num_face,comp) = moy;
508 ch_tab(voisine,comp) = moy;
509 }
510 }
511 }
512 }
513 }
514 ch_tab.echange_espace_virtuel();
515}
