69//! Renvoie la derivee en temps de l'inconnue de l'equation.<br>Le calcul est le suivant:<br>d(inconnue)/dt = M^{-1} * (sources - somme(Op_{i}(inconnue))) / rho

73DoubleTab& Convection_Diffusion_Fluide_Dilatable_Proto::derivee_en_temps_inco_sans_solveur_masse_impl
74(Convection_Diffusion_Fluide_Dilatable_base& eqn, DoubleTab& derivee, const bool is_expl)
75{
76 /*
77 * ATEENTION : THIS IS A GENERIC METHOD THAT IS USED TO SOLVE EXPLICITLY (OR DIFFUSION IMPLICIT)
78 * A THERMAL OR SPECIES CONV/DIFF EQUATION. SO THE VARIABLE CAN EITHER BE THE TEMPERATURE T OR
79 * THE MASS FRACTION Y.
80 *
81 * For the species equation for example, we have :
82 * d(rho Y)/dt + div( rho*u*Y ) = div( rho*D*grad(Y) )
83 *
84 * With a non conservative formulation we write:
85 * rho d(Y)/dt + Y d(rho)/dt + Y div (rho*u) + rho*u grad(Y) = div( rho*D*grad(Y) )
86 *
87 * Using the mass equation, we get :
88 * rho d(Y)/dt + rho*u grad(Y) = div( rho*D*grad(Y) )
89 *
90 * divide by rho => d(Y)/dt = ( div( rho*D*grad(Y) ) + S ) / rho - u grad(Y)
91 *
92 * This is what we code here with derivee = d(Y)/dt
93 *
94 *
95 * Similarly for the temperature equation, we code
96 *
97 * derivee = d(T)/dt = ( div( lambda*grad(T) ) + S ) / (rho*Cp) - u grad(T)
98 */
99
100 const Schema_Temps_base& sch = eqn.schema_temps();
101 int diffusion_implicite=sch.diffusion_implicite();
102 eqn.zone_Cl_dis().les_conditions_limites().set_modifier_val_imp(0);
103
104 /*
105 * FIRST TERM : diffusive
106 * derivee = div( rho*D*grad(Y) ) or div( lambda*grad(T) )
107 */
108 Fluide_Dilatable_base& fluide_dil = eqn.fluide();
109 if ( !is_thermal() /* species equation */ || (is_thermal() && fluide_dil.type_fluide()=="Gaz_Parfait" ))
110 {
111 if (!diffusion_implicite) eqn.operateur(0).ajouter(derivee);
112 }
113 else //!< Thermal equation + Gaz reel => No diffusion implicit ...
114 {
115 eqn.operateur(0).ajouter(fluide_dil.temperature(),derivee);
116 if (diffusion_implicite)
117 {
118 Cerr << "Error: diffusion implicit not implemented in Convection_Diffusion_Chaleur_Fluide_Dilatable_base" << finl;
119 Process::exit();
120 }
121 }
122
123 eqn.zone_Cl_dis().les_conditions_limites().set_modifier_val_imp(1);
124 derivee.echange_espace_virtuel();
125
126// Add source term (if any, but for temperatur eit is sure !!! )
127 eqn.sources().ajouter(derivee);
128
129// On divise derivee par rho si espece ou gaz reel... sinon par rho*Cp !
130 const DoubleTab& tab_rho = fluide_dil.masse_volumique().valeurs();
131 const int n = tab_rho.dimension(0);
132 assert (tab_rho.line_size() == 1);
133
134 if (is_thermal() && fluide_dil.type_fluide()=="Gaz_Parfait")
135 {
136 fluide_dil.update_rho_cp(sch.temps_courant());
137 const DoubleTab& rhoCp = eqn.get_champ("rho_cp_comme_T").valeurs();
138 for (int som=0 ; som<n ; som++) derivee(som) /= rhoCp(som);
139 }
140 else
141 for (int som=0 ; som<n ; som++) derivee(som) /= tab_rho(som);
142
143 derivee.echange_espace_virtuel();
144
145
146 /*
147 * SECOND TERM : convective
148 * = - u grad(Y) = [ Y div (rho*u) - div( rho*u*Y ) ] / rho
149 * or
150 * = - u grad(T) = [ T div (rho*u) - div( rho*u*T ) ] / rho*Cp
151 */
152 DoubleTrav convection(derivee);
153
154// Add Y div (rho*u) or T div (rho*u)
155 const DoubleTab& inco = eqn.inconnue().valeurs();
156 calculer_div_u_ou_div_rhou(convection);
157
158 for (int som=0 ; som<n ; som++) convection(som) *= (-inco(som));
159
160// Add convection operator: - div( rho*u*Y ) or -div ( rho*u*T )
161 eqn.operateur(1).ajouter(convection);
162
163// Divide by rho if necessary
164 if (is_generic())
165 for (int som=0 ; som<n ; som++) convection(som) /= tab_rho(som);
166
167 /*
168 * TOTAL TERM : diffusive + convective + sources
169 */
170 derivee+=convection;
171
172 if (diffusion_implicite)
173 {
174 const DoubleTab& Tfutur=eqn.inconnue().futur();
175 DoubleTrav secmem(derivee);
176 secmem=derivee;
177 eqn.solv_masse().appliquer(secmem);
178 derivee = Tfutur;
179
180 is_thermal() ? eqn.solv_masse()->set_name_of_coefficient_temporel("rho_cp_comme_T") :
181 eqn.solv_masse()->set_name_of_coefficient_temporel("masse_volumique");
182
183 eqn.Gradient_conjugue_diff_impl(secmem,derivee);
184 eqn.solv_masse()->set_name_of_coefficient_temporel("no_coeff");
185 }
186
187// 100% explicite
188 if (!sch.diffusion_implicite() && is_expl)
189 {
190 eqn.solv_masse().appliquer(derivee);
191 derivee.echange_espace_virtuel();
192 }
193
194 return derivee;
195} /* END derivee_en_temps_inco_sans_solveur_masse_impl */

197void Convection_Diffusion_Fluide_Dilatable_Proto::assembler_impl
198( Convection_Diffusion_Fluide_Dilatable_base& eqn,
199 Matrice_Morse& matrice, const DoubleTab& inco, DoubleTab& resu)
200{
201 /*
202 * ATEENTION : THIS IS A GENERIC METHOD THAT IS USED TO SOLVE IMPLICITLY A THERMAL OR SPECIES
203 * CONV/DIFF EQUATION. SO THE VARIABLE CAN EITHER BE THE TEMPERATURE T OR THE MASS FRACTION Y.
204 *
205 * See the previous method derivee_en_temps_inco_sans_solveur_masse_impl for more description
206 * of the treated equations.
207 */
208
209// Elie Saikali
210// Je garde ce test pour debug ... TODO : a voir si on peut virer tout ça...
211 int test_op=0;
212 {
213 char* theValue = getenv("TRUST_TEST_OPERATEUR_IMPLICITE");
214 if (theValue != NULL) test_op=2;
215 }
216 {
217 char* theValue = getenv("TRUST_TEST_OPERATEUR_IMPLICITE_BLOQUANT");
218 if (theValue != NULL) test_op=1;
219 }
220
221 Fluide_Dilatable_base& fluide_dil = eqn.fluide();
222 const DoubleTab& tab_rho = fluide_dil.masse_volumique().valeurs();
223 const int n = tab_rho.dimension(0);
224
225// ajout diffusion (avec rho, D et Y / ou lambda et T)
226 eqn.operateur(0).l_op_base().contribuer_a_avec(inco, matrice );
227
228// Add source term (if any)
229 eqn.sources().contribuer_a_avec(inco,matrice);
230
231 const IntVect& tab1= matrice.get_tab1();
232 DoubleVect& coeff=matrice.get_set_coeff();
233 DoubleVect coeff_diffusif(coeff);
234
235// on calcule les coefficients de l'op de convection on obtient les coeff de div (rho*u*Y)
236// ou div(rho*u*T). dans le cas thermique, il faudrait multiplier par cp puis divisier par rho cp
237// on le fera d'un coup...
238 coeff=0.;
239 eqn.operateur(1).l_op_base().contribuer_a_avec(inco, matrice );
240
241// on calcule div(rho * u)
242 DoubleTrav derivee2(resu);
243 calculer_div_u_ou_div_rhou(derivee2);
244
245 if (!is_thermal()) //espece
246 {
247 for (int som=0 ; som<n ; som++)
248 {
249 double inv_rho = 1. / tab_rho(som);
250 for (int k=tab1(som)-1; k<tab1(som+1)-1; k++) coeff(k)= (coeff(k)*inv_rho+coeff_diffusif(k)*inv_rho);
251
252 matrice(som,som)+=derivee2(som)*inv_rho;
253 }
254 }
255 else if (is_thermal() && fluide_dil.type_fluide()=="Gaz_Parfait")
256 {
257 fluide_dil.update_rho_cp(eqn.schema_temps().temps_courant());
258 const DoubleTab& rhoCp = eqn.get_champ("rho_cp_comme_T").valeurs();
259
260 for (int som=0 ; som<n ; som++)
261 {
262 double inv_rho = 1. / tab_rho(som);
263 if (!is_generic()) inv_rho = 1.;
264 double rapport = 1. / rhoCp(som);
265
266// il faut multiplier toute la ligne de la matrice par rapport
267 for (int k=tab1(som)-1; k<tab1(som+1)-1; k++) coeff(k)= (coeff(k)*inv_rho+coeff_diffusif(k)*rapport);
268
269// ajout de Tdiv(rhou )/rho
270 matrice(som,som) += derivee2(som)*inv_rho;
271 }
272 }
273 else
274 {
275 Cerr<<"The implicit algorithm is available only for perfect gas."<<finl;
276 Process::exit();
277 }
278
279// on a la matrice approchee on recalcule le residu;
280 resu=0;
281 derivee_en_temps_inco_sans_solveur_masse_impl(eqn,resu,false /* implicit */);
282 matrice.ajouter_multvect(inco,resu);
283
284 if (test_op)
285 {
286 DoubleTrav diff(resu), conv(resu);
287 eqn.operateur(0).l_op_base().contribuer_au_second_membre(diff);
288 eqn.operateur(1).l_op_base().contribuer_au_second_membre(conv);
289 eqn.sources().ajouter(diff);
290 double Cp = -5.;
291 int is_cp_unif= sub_type(Champ_Uniforme,fluide_dil.capacite_calorifique().valeur());
292 const DoubleTab& tab_cp = fluide_dil.capacite_calorifique().valeurs();
293 if (is_cp_unif) Cp=tab_cp(0,0);
294
295 for (int som=0 ; som<n ; som++)
296 {
297 if (!is_cp_unif) Cp=tab_cp(som);
298 double inv_rho=1./tab_rho(som);
299 if (!is_generic()) inv_rho=1;
300 double rapport=1./(tab_rho(som)*Cp);
301 diff(som)=resu(som)-conv(som)*inv_rho-diff(som)*rapport;
302 }
303 eqn.solv_masse().appliquer(diff);
304 double err=mp_max_abs_vect(diff);
305 Cerr << eqn.que_suis_je() <<" : Erreur assemblage = " << err << finl;;
306
307 if (err > 1.e-5)
308 {
309 DoubleVect& diff_=diff;
310 Cerr<<" size "<< diff_.size()<<finl;
311 for (int i=0; i<diff_.size(); i++)
312 if (std::fabs(diff_(i))>1e-5) Cerr<<i << " "<< diff_(i)<< " "<<finl;
313
314 if (test_op==1)
315 {
316 Cerr<<" pb max case "<<imin_array(diff)<<" ou " <<imax_array(diff)<<finl;
317 Process::exit();
318 }
319 }
320 }
321} /* END assembler_impl */

323void Convection_Diffusion_Fluide_Dilatable_Proto::assembler_blocs(Convection_Diffusion_Fluide_Dilatable_base& eqn,matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl)
324{
325 const std::string& nom_inco = eqn.inconnue().le_nom().getString();
326 Matrice_Morse *mat = matrices.count(nom_inco)?matrices.at(nom_inco):NULL;
327
328 Fluide_Dilatable_base& fluide_dil = eqn.fluide();
329 const DoubleTab& tab_rho = fluide_dil.masse_volumique().valeurs();
330 const int n = tab_rho.dimension(0);
331
332 Matrice_Morse mat_diff(*mat);
333 DoubleTab secmem_tmp(secmem);
334 eqn.operateur(0).l_op_base().ajouter_blocs({{nom_inco, &mat_diff}}, secmem_tmp, semi_impl);
335 statistiques().end_count(assemblage_sys_counter_);
336
337 statistiques().begin_count(source_counter_);
338 for (int i = 0; i < eqn.sources().size(); i++)
339 eqn.sources()(i).valeur().ajouter_blocs({{nom_inco, &mat_diff}}, secmem_tmp, semi_impl);
340 statistiques().end_count(source_counter_);
341
342 statistiques().begin_count(assemblage_sys_counter_);
343 DoubleVect& coeff_diffusif=mat_diff.get_set_coeff();
344
345 const IntVect& tab1= mat->get_tab1();
346 DoubleVect& coeff=mat->get_set_coeff();
347 coeff = 0;
348 eqn.operateur(1).l_op_base().ajouter_blocs(matrices, secmem_tmp, semi_impl);
349
350// on calcule div(rho * u)
351 calculer_div_u_ou_div_rhou(secmem);
352
353 if (!is_thermal()) //espece
354 {
355 for (int som=0 ; som<n ; som++)
356 {
357 double inv_rho = 1. / tab_rho(som);
358 for (int k=tab1(som)-1; k<tab1(som+1)-1; k++) coeff(k)= (coeff(k)*inv_rho+coeff_diffusif(k)*inv_rho);
359
360 if(mat) (*mat)(som,som)+=secmem(som)*inv_rho;
361 }
362 }
363 else if (is_thermal() && fluide_dil.type_fluide()=="Gaz_Parfait")
364 {
365 fluide_dil.update_rho_cp(eqn.schema_temps().temps_courant());
366 const DoubleTab& rhoCp = eqn.get_champ("rho_cp_comme_T").valeurs();
367
368 for (int som=0 ; som<n ; som++)
369 {
370 double inv_rho = 1. / tab_rho(som);
371 if (!is_generic()) inv_rho = 1.;
372 double rapport = 1. / rhoCp(som);
373
374// il faut multiplier toute la ligne de la matrice par rapport
375 for (int k=tab1(som)-1; k<tab1(som+1)-1; k++) coeff(k)= (coeff(k)*inv_rho+coeff_diffusif(k)*rapport);
376
377// ajout de Tdiv(rhou )/rho
378 if(mat) (*mat)(som,som) += secmem(som)*inv_rho;
379 }
380 }
381 else
382 {
383 Cerr<<"The implicit algorithm is available only for perfect gas."<<finl;
384 Process::exit();
385 }
386
387// on a la matrice approchee on recalcule le residu;
388 secmem=0;
389 derivee_en_temps_inco_sans_solveur_masse_impl(eqn,secmem,false /* implicit */);
390 if(mat) mat->ajouter_multvect(eqn.inconnue().valeurs(),secmem);
391}
