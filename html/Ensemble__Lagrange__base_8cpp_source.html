43void Ensemble_Lagrange_base::remplir_sommets_tmp(DoubleTab& soms_tmp)
44{
45// Premier cas : les sommets ont ete lus dans un fichier
46// On copie sommets_lu_ dans un tableau temporaire sommets_tmp
47
48 if (sommets_lu_.dimension(0)!=0)
49 {
50 const DoubleTab& som_lu = sommets_lu() ;
51 int dim0=som_lu.dimension(0);
52 int dim1=som_lu.dimension(1);
53 soms_tmp.resize(dim0,dim1);
54 soms_tmp = som_lu;
55 }
56 else if (nb_marqs_par_sz().size()!=0)
57// Deuxieme cas : les sommets sont crees dans des sous zones
58// On remplit le tableau temporaire
59 generer_marqueurs_sz(soms_tmp);
60 else
61 soms_tmp.resize(0,0);
62}
63
64// Generation des coordonnees des particules sur une ou plusieurs sous zones
65// Pour chacune des sous zones :
66// - Etape 1 : On determine le min et max pour x, y et z en tenant compte de tous les processeurs
67// - Etape 2 : Le processeur maitre va calculer les coordonnees des particules
68// -soit avec une distribution uniforme
69// -soit avec une distribution aleatoire
70// - Etape 3 : Le processeur maitre envoie les valeurs des coordonnees aux autres processeurs
71
72void Ensemble_Lagrange_base::generer_marqueurs_sz(DoubleTab& soms_tmp)
73{
74 const Zone& mazone = ma_zone_.valeur();
75 const Domaine& dom = mazone.domaine();
76 int nb_sz,nb_marq_sz, old_size, dim;
77 soms_tmp.resize(0);
78 dim=Objet_U::dimension;
79 nb_sz=nb_marqs_sz.size();
80
81 for (int i=0; i<nb_sz; i++)
82 {
83 const Nom& nom = nom_sz[i];
84 const Sous_Zone& sz = dom.ss_zone(nom);
85 int nb_elem = sz.nb_elem_tot();
86
87 int nb_som = mazone.nb_som_elem();
88 int le_poly,poly,le_som;
89
90 double xmoy,ymoy,zmoy,deltax,deltay,deltaz;
91 double xmin,xmax,ymin,ymax,zmin,zmax;
92 xmin = ymin = zmin = 1.e8;
93 xmax = ymax = zmax= -1.e8;
94
95 if (dim==2)
96 {
97 zmin = zmax = 0.;
98 }
99
100// Etape 1 : On determine le min et max en tenant compte de tous les processeurs
101 if (nb_elem!=0)
102 {
103 poly = sz(0);
104 le_som = 0;
105
106 xmin=dom.coord(mazone.sommet_elem(poly,le_som),0);
107 ymin=dom.coord(mazone.sommet_elem(poly,le_som),1);
108 if (dim==3)
109 zmin=dom.coord(mazone.sommet_elem(poly,le_som),2);
110 xmax=xmin;
111 ymax=ymin;
112 if (dim==3)
113 zmax=zmin;
114
115 for (le_poly=0; le_poly<nb_elem; le_poly++)
116 for(le_som=0; le_som<nb_som; le_som++)
117 {
118 poly = sz(le_poly);
119 xmin=std::min(xmin,dom.coord(mazone.sommet_elem(poly,le_som),0));
120 ymin=std::min(ymin,dom.coord(mazone.sommet_elem(poly,le_som),1));
121 if (dim==3)
122 zmin=std::min(zmin,dom.coord(mazone.sommet_elem(poly,le_som),2));
123 xmax=std::max(xmax,dom.coord(mazone.sommet_elem(poly,le_som),0));
124 ymax=std::max(ymax,dom.coord(mazone.sommet_elem(poly,le_som),1));
125 if (dim==3)
126 zmax=std::max(zmax,dom.coord(mazone.sommet_elem(poly,le_som),2));
127 }
128 } //Fin if (nb_elem!=0)
129
130
131 xmin = Process::mp_min(xmin);
132 xmax = Process::mp_max(xmax);
133 ymin = Process::mp_min(ymin);
134 ymax = Process::mp_max(ymax);
135 if (dim==3)
136 {
137 zmin = Process::mp_min(zmin);
138 zmax = Process::mp_max(zmax);
139 }
140
141// Etape 2 : Le processeur maitre va calculer les coordonnees des particules
142 if (je_suis_maitre())
143 {
144
145 xmoy = (xmax+xmin)/2.;
146 ymoy = (ymax+ymin)/2.;
147 zmoy = (zmax+zmin)/2.;
148
149 deltax = (xmax-xmin)/2.;
150 deltay = (ymax-ymin)/2.;
151 deltaz = (zmax-zmin)/2.;
152
153 nb_marq_sz = nb_marqs_sz(i);
154 old_size = soms_tmp.dimension(0);
155 soms_tmp.resize(old_size+nb_marq_sz,dim);
156 int marq;
157
158// Cas d une distribution aleatoire
159 if (nb_marqs_par_dir(i,0)==0)
160 {
161// double a,b,c;
162 srand48(1);
163 for (marq=0; marq<nb_marq_sz; marq++)
164 {
165 soms_tmp(old_size+marq,0) = xmoy + deltax*(-1.+2.*drand48());
166 soms_tmp(old_size+marq,1) = ymoy + deltay*(-1.+2.*drand48());
167 if (dim==3)
168 soms_tmp(old_size+marq,2) = zmoy + deltaz*(-1.+2.*drand48());
169 }
170 }
171// Cas d une distribution uniforme
172 else
173 {
174 double dx, dy, dz;
175 int nx_max = nb_marqs_par_dir(i,0);
176 int ny_max = nb_marqs_par_dir(i,1);
177 dx = 2.*deltax/double(nx_max+1);
178 dy = 2.*deltay/double(ny_max+1);
179
180 if (dim==2)
181 {
182 for (int ny=1; ny<ny_max+1; ny++)
183 for (int nx=1; nx<nx_max+1; nx++)
184 {
185 marq = nx + (ny-1)*nx_max-1;
186 soms_tmp(old_size+marq,0) = xmin + nx*dx;
187 soms_tmp(old_size+marq,1) = ymin + ny*dy;
188 }
189 }
190 else if (dim==3)
191 {
192 int nz_max = nb_marqs_par_dir(i,2);
193 dz = 2.*deltaz/double(nz_max+1);
194 for (int nz=1; nz<nz_max+1; nz++)
195 for (int ny=1; ny<ny_max+1; ny++)
196 for (int nx=1; nx<nx_max+1; nx++)
197 {
198 marq = nx + (ny-1)*nx_max + (nz-1)*nx_max*ny_max-1;
199 soms_tmp(old_size+marq,0) = xmin + nx*dx;
200 soms_tmp(old_size+marq,1) = ymin + ny*dy;
201 soms_tmp(old_size+marq,2) = zmin + nz*dz;
202 }
203 }
204 }
205 }//Fin de if je_suis_maitre()
206
207// Etape 3 : Le processeur maitre envoie les valeurs des coordonnees calculees (soms_tmp)
208// aux autres processeurs
209 if (je_suis_maitre())
210 for(int p=1; p<Process::nproc(); p++)
211 envoyer(soms_tmp,0, p, 0);
212 for(int p=1; p<Process::nproc(); p++)
213 recevoir(soms_tmp,0,p,0);
214 }
215}
