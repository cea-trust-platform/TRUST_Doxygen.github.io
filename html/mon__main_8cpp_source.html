51static int init_petsc(True_int argc, char **argv, int with_mpi,int& trio_began_mpi_)
52{
53#ifdef PETSCKSP_H
54 static char help[] = "TRUST may solve linear systems with Petsc library.\n\n" ;
55 Nom pwd(::pwd());
56// On initialise Petsc
57#ifdef MPI_INIT_NEEDS_MPIRUN
58 True_int flag;
59 MPI_Initialized(&flag);
60// si MPI initialise ou si argc>2
61 if ((argc>2)||(flag))
62 {
63 PetscInitialize(&argc, &argv, (char*)0, help);
64 }
65#else
66 PetscInitialize(&argc, &argv, (char*)0, help);
67#endif
68// Bizarrerie qui se produit sur une machine (ioulia, MPICH natif): PetscInitialize change le pwd()
69// en sequentiel et si le binaire n'est pas dans le repertoire de l'etude, le pwd est perdu...
70 int ierr;
71 if (!with_mpi)
72 {
73 ierr=chdir(pwd);
74 if (ierr)
75 {
76 Cerr << "Error on chdir into mon_main.cpp. Contact TRUST support." << finl;
77 Process::exit();
78 }
79 }
80// Equivalent de -abort_on_error (aucune erreur PETSc n'est tolere):
81 PetscPushErrorHandler(PetscAbortErrorHandler,PETSC_NULL);
82// Desactive le signal handler en optimise pour eviter d'etre trop bavard
83// et de "masquer" les messages d'erreur TRUST:
84 PetscPopSignalHandler();
85
86 bool error_handlers = false;
87 char* theValue = getenv("TRUST_ENABLE_ERROR_HANDLERS");
88 if (theValue != NULL) error_handlers = true;
89#ifndef NDEBUG
90 error_handlers = true;
91#endif
92 if (error_handlers)
93 {
94 Cerr << "Enabling error handlers catching SIGFPE and SIGABORT and giving a trace of where the fault happened." << finl;
95 install_handlers();
96 }
97#else
98// MPI_Init pour les machines ou Petsc n'est pas
99// installe: ex AIX avec MPICH: il faut que argc et argv soit passes
100// correctement et pas comme dans Comm_Group_MPI::init_group_trio
101// sinon message: xm_348262: p4_error: Command-line arguments are missing: 0
102#ifdef MPI_
103 True_int flag;
104 MPI_Initialized(&flag);
105 if (!flag)
106 {
107 MPI_Init(&argc,&argv);
108 trio_began_mpi_=1;
109 }
110#endif
111#endif
112
113 return 1;
114}

132void init_cuda()
133{
134 char* local_rank_env;
135 cudaError_t cudaRet;
136
137 /* Recuperation du rang local du processus via la variable d'environnement
138 positionnee par Slurm, l'utilisation de MPI_Comm_rank n'etant pas encore
139 possible puisque cette routine est utilisee AVANT l'initialisation de MPI */
140 local_rank_env = getenv("SLURM_LOCALID");
141 if (local_rank_env)
142 {
143 int local_rank = atoi(local_rank_env);
144 if (local_rank==0) printf("The MPI library has CUDA-aware support and TRUST will try using this feature...\n");
145 /* Definition du GPU a utiliser pour chaque processus MPI */
146 int num_devices = 0;
147 cudaGetDeviceCount(&num_devices);
148 cudaRet = cudaSetDevice(local_rank % num_devices);
149 if(cudaRet != cudaSuccess)
150 {
151 printf("Error: cudaSetDevice failed\n");
152 abort();
153 }
154 else
155 {
156 if (local_rank==0) printf("init_cuda() done!");
157 }
158 }
159 else
160 {
161 printf("Error : can't guess the local rank of the task\n");
162 abort();
163 }
164}
165#endif

171void mon_main::init_parallel(const int argc, char **argv, int with_mpi, int check_enabled, int with_petsc)
172{
173#ifdef PETSC_HAVE_CUDA
174// Necessaire sur JeanZay pour utiliser GPU Direct (http://www.idris.fr/jean-zay/gpu/jean-zay-gpu-mpi-cuda-aware-gpudirect.html)
175// mais performances moins bonnes (trust PAR_gpu_3D 2) donc desactive en attendant d'autres tests:
176// Absolument necessaire sur JeanZay (si OpenMPU-Cuda car sinon plantages lors des IO)
177// Voir: https://www.open-mpi.org/faq/?category=runcuda#mpi-cuda-aware-support pour activer ou non a la compilation !
178#if defined(MPIX_CUDA_AWARE_SUPPORT) && MPIX_CUDA_AWARE_SUPPORT
179 init_cuda();
180#endif /* MPIX_CUDA_AWARE_SUPPORT */
181#endif
182 Nom arguments_info="";
183 int must_mpi_initialize = 1;
184 if (with_petsc != 0)
185 {
186 if (init_petsc(argc, argv, with_mpi,trio_began_mpi_))
187 {
188 must_mpi_initialize = 0; //!< Deja fait par Petsc
189 arguments_info += "Petsc initialization succeeded.\n";
190 }
191 else
192 {
193 arguments_info += "Petsc initialization failed (not compiled ?). Not fatal...\n";
194 }
195 }
196 else
197 {
198 arguments_info += "Petsc initialization skipped (-petsc=0).\n";
199 }
200#ifdef MPI_
201 Comm_Group_MPI::set_must_mpi_initialize(must_mpi_initialize);
202#else
203// GF pour eviter warning
204 if (must_mpi_initialize>10) abort();
205#endif
206// ***************** Initialisation du parallele *************************
207 Comm_Group::set_check_enabled(check_enabled);
208
209 if (with_mpi)
210 {
211 if (!init_parallel_mpi(groupe_trio_))
212 {
213 Cerr << "MPI initialization failed (not compiled ?). Fatal." << finl;
214 Process::exit();
215 }
216 }
217 else
218 {
219 groupe_trio_.typer("Comm_Group_NoParallel");
220 }
221
222// Initialisation des groupes de communication.
223 PE_Groups::initialize(groupe_trio_);
224 arguments_info += "Parallel engine initialized : ";
225 arguments_info += groupe_trio_.valeur().que_suis_je();
226 arguments_info += " with ";
227 arguments_info += Nom(Process::nproc());
228 arguments_info += " processors\n";
229
230 if (Process::je_suis_maitre())
231 Cerr << arguments_info;
232}

234void mon_main::finalize()
235{
236#ifdef MPI_
237// MPI_Group_free before MPI_Finalize
238 if (sub_type(Comm_Group_MPI,groupe_trio_.valeur()))
239 ref_cast(Comm_Group_MPI,groupe_trio_.valeur()).free();
240#endif
241#ifdef PETSCKSP_H
242// On PetscFinalize que si c'est necessaire
243 PetscBool isInitialized;
244 PetscInitialized(&isInitialized);
245 if (isInitialized==PETSC_TRUE)
246 {
247 PetscPopErrorHandler(); //!< Removes the latest error handler that was pushed with PetscPushErrorHandler in init_petsc
248#ifdef MPI_
249 if (sub_type(Comm_Group_MPI,PE_Groups::current_group()))
250 PETSC_COMM_WORLD = ref_cast(Comm_Group_MPI,PE_Groups::current_group()).get_mpi_comm();
251#endif
252 PetscFinalize();
253 }
254#endif
255#ifdef MPI_
256 if (trio_began_mpi_)
257 {
258// On MPI_Finalize si MPI_Initialized and not MPI_Finalized
259 True_int flag;
260 MPI_Initialized(&flag);
261 if (flag)
262 {
263 MPI_Finalized(&flag);
264 if (!flag)
265 MPI_Finalize();
266 }
267 }
268#endif
269}
