60int Champ_front_instationnaire_base::initialiser(double temps, const Champ_Inc_base& inco)
61{
62 if (!Champ_front_base::initialiser(temps,inco))
63 return 0;
64
65 temps_defaut=temps;
66 changer_temps_futur(temps,0);
67
68 for(int i=0; i<les_valeurs->nb_cases(); i++)
69 {
70 les_valeurs[i].valeurs().resize(1,nb_comp());
71 }
72
73 Gpoint_=valeurs(); //!< pour dimensionner
74 Gpoint_=0.;
75
76 return 1;
77}

180// Description :
181// Calcule le taux d'accroissement du champ entre t1 et t2
182// et le stocke dans Gpoint_
183void Champ_front_instationnaire_base::Gpoint(double t1, double t2)
184{
185 DoubleTab& v1=valeurs_au_temps(t1);
186 DoubleTab& v2=valeurs_au_temps(t2);
187 int dim=v1.dimension(1);
188
189 Gpoint_.resize(dim);
190 for (int i=0; i<dim; i++)
191 Gpoint_(i) = (v2(0,i)-v1(0,i)) / (t2-t1);
192}
