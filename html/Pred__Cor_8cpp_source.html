87int Pred_Cor::faire_un_pas_de_temps_eqn_base(Equation_base& eqn)
88{
89 DoubleTab& xi=eqn.inconnue().valeurs();
90 DoubleTab& xipls1=eqn.inconnue().futur();
91
92 DoubleTab dudt(xi);
93
94// Etape PREDICTOR
95
96 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
97
98// On tourne la roue pour que les operateurs utilisent les champs au temps futur
99 eqn.inconnue().avancer();
100 eqn.derivee_en_temps_inco(dudt);
101
102// Xi+1=Xi+dt_*dX/dt(Xi)
103 xipls1 = dudt;
104 xipls1 *= dt_;
105 xipls1 += xi;
106
107 xipls1.echange_espace_virtuel();
108
109// Etape CORRECTOR
110 DoubleTab dudtpls1(xipls1);
111 eqn.derivee_en_temps_inco(dudtpls1);
112 eqn.inconnue().reculer();
113
114 dudtpls1 += dudt;
115 dudtpls1 *= 0.5;
116
117// Xi+1=Xi+dt_*0.5*(dX/dt(Xi)+dX/dt(Xi+1))
118 xipls1 = dudtpls1 ;
119 xipls1 *= dt_;
120 xipls1 += xi ;
121
122 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
123 update_critere_statio(dudtpls1, eqn);
124
125 return 1;
126}
