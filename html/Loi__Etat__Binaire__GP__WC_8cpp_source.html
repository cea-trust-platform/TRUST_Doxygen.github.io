60void Loi_Etat_Binaire_GP_WC::calculer_masse_volumique()
61{
62 const DoubleTab& tab_Y1 = le_fluide->inco_chaleur().valeurs();
63 Fluide_Weakly_Compressible& FWC = ref_cast(Fluide_Weakly_Compressible,le_fluide.valeur());
64 DoubleTab& tab_rho = le_fluide->masse_volumique().valeurs();
65 const DoubleTab& pres = FWC.pression_th_tab();
66 const int n = tab_rho.size();
67 for (int som=0 ; som<n ; som++)
68 {
69 tab_rho_np1(som) = calculer_masse_volumique(pres(som,0),tab_Y1(som,0));
70 tab_rho(som,0) = 0.5 * ( tab_rho_n(som) + tab_rho_np1(som) );
71 }
72 tab_rho.echange_espace_virtuel();
73 tab_rho_np1.echange_espace_virtuel();
74 le_fluide->calculer_rho_face(tab_rho_np1);
