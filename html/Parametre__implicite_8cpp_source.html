45// XD parametre_implicite parametre_equation_base parametre_implicite -1 Keyword to change for this equation only the parameter of the implicit scheme used to solve the problem.
46// XD attr seuil_convergence_implicite floattant seuil_convergence_implicite 1 Keyword to change for this equation only the value of seuil_convergence_implicite used in the implicit scheme.
47// XD attr seuil_convergence_solveur floattant seuil_convergence_solveur 1 Keyword to change for this equation only the value of seuil_convergence_solveur used in the implicit scheme
48// XD attr solveur solveur_sys_base solveur 1 Keyword to change for this equation only the solver used in the implicit scheme
49// XD attr resolution_explicite rien resolution_explicite 1 To solve explicitly the equation whereas the scheme is an implicit scheme.
50// XD attr equation_non_resolue rien equation_non_resolue 1 Keyword to specify that the equation is not solved.
51// XD attr equation_frequence_resolue chaine equation_frequence_resolue 1 Keyword to specify that the equation is solved only every n time steps (n is an integer or given by a time-dependent function f(t)).
52Entree& Parametre_implicite::readOn(Entree& is)
53{
54 Nom eq_freq_resolue="0";
55 double seuil_convergence_solveur_prov=-1;
56 Param param(que_suis_je());
57 param.ajouter("solveur",&le_solveur_);
58 param.ajouter( "seuil_convergence_implicite",&seuil_convergence_implicite_);
59 param.ajouter( "seuil_convergence_solveur",&seuil_convergence_solveur_prov);
60
61 param.ajouter("seuil_generation_solveur",&seuil_generation_solveur_);
62 param.ajouter("seuil_test_preliminaire_solveur",&seuil_test_preliminaire_solveur_);
63 param.ajouter("seuil_verification_solveur",&seuil_verification_solveur_);
64
65 param.ajouter( "resolution_explicite_diffusion_implicite",&seuil_diffusion_implicite_);
66 param.ajouter_flag( "resolution_explicite",&calcul_explicite_);
67 param.ajouter( "equation_frequence_resolue",&eq_freq_resolue);
68 param.lire_avec_accolades_depuis(is);
69 if (param.get_list_mots_lus().rang("seuil_test_preliminaire_solveur")>0)
70 seuil_test_preliminaire_lu_=1;
71 if (seuil_convergence_solveur_prov>0)
72 set_seuil_solveur_avec_seuil_convergence_solveur(seuil_convergence_solveur_prov);
73 if (seuil_diffusion_implicite_>0)
74 calcul_explicite_=1;
75 equation_frequence_resolue_.setString(eq_freq_resolue);
76 equation_frequence_resolue_.parseString();
77 le_solveur_.nommer("solveur_implicite");
78 return is;
79}

83void Parametre_implicite::set_seuil_solveur_avec_seuil_convergence_solveur(double seuil_convergence_solveur)

89 #ifndef NDEBUG

91 Process::exit();
92 #endif
93 */
94 if (seuil_generation_solveur_<0)
95 seuil_generation_solveur_=seuil_convergence_solveur;
96 else
97 {
98 Cerr<< "You can not use seuil_generation_solveur and seuil_convergence_solveur in the same block" <<finl;
99 exit();
100 }
101 if (seuil_verification_solveur_<0)
102 seuil_verification_solveur_=seuil_convergence_solveur;
103 else
104 {
105 Cerr<< "You can not use seuil_verification_solveur and seuil_convergence_solveur in the same block" <<finl;
106 exit();
107 }
108 if (seuil_test_preliminaire_solveur_<0)
109 seuil_test_preliminaire_solveur_=seuil_convergence_solveur;
110 else
111 {
112 Cerr<< "You can not use seuil_test_preliminaire_solveur and seuil_convergence_solveur in the same block" <<finl;
113 exit();
114 }
115}
