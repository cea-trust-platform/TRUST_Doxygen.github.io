253int Postraitement_lata::lire_motcle_non_standard(const Motcle& mot, Entree& is)
254{
255 if (mot=="format")

257 lire_format(is);
258 return 1;

263 return 1;
264 }

266 {

271 {
272 Nom expression;
273 is >> expression;

276 fdt_post.setString(expression);
277 fdt_post.addVar("t");
278 fdt_post.parseString();
279 dt_post_ = fdt_post.eval();
280 return 1;
281 }
282 else
283 return Postraitement_base::lire_motcle_non_standard(mot,is);
284 return 1;
285}

317//! lecture d'une liste de champs a postraiter<br>On attend ceci :<br>champs sommets|elements|faces (le mot cle champs a deja ete lu)<br>{<br>nom_du_champ<br>[ nom_du_champ ]<br>[ ... ]<br>}<br>Les champs lus sont ajoutes a la liste correspondante (sommets, elements, ...),<br>les doublons sont retires.<br>Le nom du champ doit etre compris par le probleme (pb.comprend_champ(...))<br>Attention: les champs postraites aux faces doivent etre discretises aux faces<br>(pas d'interpolation, on postraite le champ tel quel).
318void Postraitement_lata::lire_champs(Entree& is)
319{
320 Motcle motcle;
321 is >> motcle;
322 Localisation loc = SOMMETS;
323 if (motcle == "sommets")
324 loc = SOMMETS;
325 else if (motcle == "elements")
326 loc = ELEMENTS;
327 else if (motcle == "faces")
328 loc = FACES;
329 else
330 {
331 Cerr << "Error in Postraitement_lata::lire_champs :" << finl;
332 Cerr << " We read : " << motcle << finl;
333 Cerr << " We expected \"sommets\", \"elements\" or \"faces\" after \"champs\"" << finl;
334 exit();
335 }
336 Motcles& liste =
337 (loc == SOMMETS) ? liste_champs_aux_sommets
338 : (loc == ELEMENTS) ? liste_champs_aux_elements
339 : liste_champs_aux_faces;
340
341 is >> motcle;
342 if (motcle != "{")
343 {
344 Cerr << "Error in Postraitement_lata::lire_champs :" << finl;
345 Cerr << " We read : " << motcle << finl;
346 Cerr << " We expected { after sommets/elements/faces." << finl;
347 exit();
348 }
349 Probleme_base& pb = mon_probleme.valeur();
350 REF(Champ_base) refch_base;
351
352 while (1)
353 {
354 is >> motcle;
355 if (motcle == "}")
356 break;
357 Noms liste_noms;
358 pb.get_noms_champs_postraitables(liste_noms);
359 pb.creer_champ(motcle);
360 int ok=0;
361 for (int i=0; i<liste_noms.size(); i++)
362 if (Motcle(liste_noms[i])==motcle) ok=1;
363
364 if (!ok)
365 {
366 Cerr << "Error in Postraitement_lata::lire_champs :\n";
367 Cerr << " The field " << motcle;
368 Cerr << " is not understood by the problem " << pb.le_nom() << finl;
369 exit();
370 }
371
372// Si un champ aux faces est demande, il faut que le champ d'origine
373// soit aux faces (on ne sait pas interpoler le champ)
374 if (loc == FACES)
375 {
376 refch_base = pb.get_champ(motcle);
377
378 const int nb_valeurs =
379 refch_base.valeur().valeurs().dimension(0);
380 const int nb_faces =
381 pb.domaine_dis().zone_dis(0).valeur().face_sommets().dimension(0);
382 if (nb_valeurs != nb_faces)
383 {
384 Cerr << "Error in Postraitement_lata::lire_champs :\n";
385 Cerr << " The field " << motcle;
386 Cerr << " can not be postprocessed to the faces.\n";
387 Cerr << " (Fields allowed: only fields discretized to the faces)" << finl;
388 exit();
389 }
390 }
391 if (!liste.contient_(motcle))
392 liste.add(motcle);
393 }
394}

401//! Evident...<br>Parametre: loc<br>valeurs autorisees: SOMMETS, ou ELEMENTS ou FACES<br>Parametre: liste<br>une liste de champs compris par le probleme.
402void Postraitement_lata::postraiter_liste_champs(Localisation loc, const Motcles& liste)
403{
404 if (verbose_stream_)
405 {
406 *verbose_stream_ << "Postraitement_lata::postraiter_liste_champs ";
407 switch(loc)
408 {
409 case SOMMETS:
410 *verbose_stream_ << "sommets" << finl;
411 break;
412 case ELEMENTS:
413 *verbose_stream_ << "elements" << finl;
414 break;
415 case FACES:
416 *verbose_stream_ << "faces" << finl;
417 break;
418 default:
419 Cerr << "Localization " << (int)loc << " not recognized in Postraitement_lata::postraiter_liste_champs" << finl;
420 exit();
421 }
422 }
423
424 const Comm_Group * comm_group = 0;
425 if (fichiers_multiples_ == 0)
426 comm_group = & PE_Groups::groupe_TRUST();
427
428 const int nb_champs = liste.size();
429 if (nb_champs == 0)
430 return;
431
432// Postraitement des champs :
433 const Probleme_base& pb = mon_probleme.valeur();
434 const Zone& la_zone = pb.domaine().zone(0);
435 const Zone_dis_base& zone_dis = pb.domaine_dis().zone_dis(0).valeur();
436 const int nb_elem = la_zone.nb_elem();
437 const int nb_sommets = la_zone.nb_som();
438 DoubleTab& valeurs = private_data_->valeurs;
439 DoubleTab& centres_de_gravites = private_data_->centres_de_gravites;
440
441 IntVect& les_polys = private_data_->les_polys;
442 REF(Champ_base) & le_champ = private_data_->le_champ;
443
444 if (loc == ELEMENTS)
445 {
446//
447// Ouille... faut calculer ca ???
448//
449 const Zone_VF& zvf = ref_cast(Zone_VF,zone_dis);
450 centres_de_gravites = zvf.xp();
451
452 les_polys.resize(nb_elem);
453 int i;
454 for (i = 0; i < nb_elem; i++)
455 les_polys(i) = i;
456 }
457
458 for (int i = 0; i < nb_champs; i++)
459 {
460 const Motcle nom_du_champ = liste[i];
461 if (verbose_stream_)
462 *verbose_stream_ << "Postraitement du champ " << nom_du_champ << finl;
463
464
465 le_champ = pb.get_champ(nom_du_champ);
466
467 {
468 if (loc == FACES)
469 valeurs.resize(0);
470 else if (loc == SOMMETS )
471 {
472 int dimension0 = (loc==SOMMETS) ? nb_sommets : nb_elem;
473 const int dimension1 = le_champ.valeur().nb_comp();
474 valeurs.resize(dimension0, dimension1);
475 }
476 else
477 {
478 int dimension0 = (loc==SOMMETS) ? nb_sommets : nb_elem;
479 if (le_champ.valeur().nb_comp() == 1)
480 valeurs.resize(dimension0);
481 else
482 {
483 const int dimension1 = le_champ.valeur().nb_comp();
484 valeurs.resize(dimension0, dimension1);
485 }
486 }
487 }
488
489 DoubleTab * valeurs_a_ecrire = 0;
490
491 switch (loc)
492 {
493 case SOMMETS:
494// Interpolation du champ aux sommets
495 le_champ.valeur().valeur_aux_sommets(pb.domaine(), valeurs);
496 valeurs_a_ecrire = &valeurs;
497 break;
498 case ELEMENTS:
499// Interpolation du champ aux elements
500 le_champ.valeur().valeur_aux_elems(centres_de_gravites, les_polys, valeurs);
501 valeurs_a_ecrire = &valeurs;
502 break;
503 case FACES:
504// Pour un champ aux faces, aucune interpolation, on ecrit le tableau tel quel.
505 valeurs_a_ecrire = &le_champ.valeur().valeurs();
506 break;
507 default:
508 exit();
509 }
510
511 ecrire_champ(lata_basename_,
512 nom_du_champ, //!< nom du champ
513 CHAMP, //!< type
514 'I', //!< champ instationnaire
515 pb.domaine().le_nom(),
516 pb.le_nom(),
517 temps_,
518 loc,
519 *valeurs_a_ecrire,
520 format_,
521 comm_group);
522 }
523// On libere la memoire des tableaux temporaires
524// (mais on garde les objets)
525// valeurs.resize(0);
526 valeurs.reset();
527 centres_de_gravites.resize(0);
528 les_polys.resize(0);
529}

535void Postraitement_lata::postraiter(int forcer)

590 postraiter_liste_champs(FACES, liste_champs_aux_faces);
591
592 temps_dernier_post_ = temps_;
593}

595//
596//! renvoie 1 si l'intervalle de temps demande entre deux<br>postraitements est ecoule.
597int Postraitement_lata::dt_post_ecoule(double& dt_post,
598 double temps_dernier_post)
599{
600 if (dt_post<=temps_ - temps_dernier_post_ || dt_post==0)
601 return 1;
602 else
603 {
604 fdt_post.setVar("t",temps_);
605 dt_post = fdt_post.eval();
606// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
607 double i, j, epsilon = 1.e-8;
608 modf(temps_/dt_post + epsilon, &i);
609 modf(temps_dernier_post/dt_post + epsilon, &j);
610 return ( i>j );
611 }
612}
