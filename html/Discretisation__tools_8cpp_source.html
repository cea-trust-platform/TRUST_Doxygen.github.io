23void Discretisation_tools::nodes_to_cells(const Champ_base& Hn, Champ_base& He)
24{
25 const DoubleTab& tabHn=Hn.valeurs();
26 DoubleTab& tabHe=He.valeurs();
27 const Zone_dis_base& zone_dis_base=He.zone_dis_base();
28
29 assert(tabHe.dimension_tot(0)==zone_dis_base.nb_elem_tot());
30 assert(tabHn.dimension_tot(0)==zone_dis_base.nb_som_tot());
31
32 const IntTab& les_elems=zone_dis_base.zone().les_elems();
33 int nb_som_elem=les_elems.dimension(1);
34 int nb_elem=zone_dis_base.nb_elem();
35
36 tabHe=0;
37 for (int ele=0; ele<nb_elem; ele++)
38 {
39 for (int s=0; s<nb_som_elem; s++)
40 {
41 int sglob=les_elems(ele,s);
42 tabHe(ele)+=tabHn(sglob);
43 }
44 }
45 double inv_nb_som_elem=1./(nb_som_elem);
46 tabHe*=inv_nb_som_elem;
47 tabHe.echange_espace_virtuel();
48}
49void Discretisation_tools::cells_to_nodes(const Champ_base& He, Champ_base& Hn)
50{
51
52 DoubleTab& tabHn=Hn.valeurs();
53 const DoubleTab& tabHe=He.valeurs();
54 Debog::verifier("elno entreee",tabHe);
55 assert_espace_virtuel_vect(tabHe);
56 const Zone_dis_base& zone_dis_base=He.zone_dis_base();
57
58 assert(tabHe.dimension_tot(0)==zone_dis_base.nb_elem_tot());
59 assert(tabHn.dimension_tot(0)==zone_dis_base.nb_som_tot());
60
61 const IntTab& les_elems=zone_dis_base.zone().les_elems();
62 const DoubleVect& volumes=ref_cast(Zone_VF,zone_dis_base).volumes();
63
64 tabHn=0;
65 DoubleTrav vsom(tabHn.dimension_tot(0));
66 int nb_som_elem=les_elems.dimension(1);
67 int nb_elem_tot=zone_dis_base.nb_elem_tot();
68 int N = tabHn.line_size();
69
70 for (int e = 0; e < nb_elem_tot; e++)
71 for (int s = 0; s < nb_som_elem; s++)
72 {
73 int sglob = les_elems(e, s);
74 for (int n = 0; n < N; n++)
75 tabHn(sglob, n) += tabHe(e, n) * volumes(e);
76 vsom(sglob) += volumes(e);
77 }
78
79 for (int s = 0; s < zone_dis_base.nb_som(); s++)
80 for (int n = 0; n < N; n++)
81 tabHn(s, n) /= vsom(s);
82
83 tabHn.echange_espace_virtuel();
84 Debog::verifier("elno sortie",tabHn);
85}

101 {

103 {
104

112void Discretisation_tools::cells_to_faces(const Champ_base& He, Champ_base& Hf)
113{
114
115 DoubleTab& tabHf=Hf.valeurs();
116 const DoubleTab& tabHe=He.valeurs();
117 Debog::verifier("element_face entreee",tabHe);
118 assert_espace_virtuel_vect(tabHe);
119 const Zone_dis_base& zone_dis_base=He.zone_dis_base();
120
121 const Zone_VF& zone_vf= ref_cast(Zone_VF,zone_dis_base);
122// en realite on fait P1B vers face
123// assert(tabHe.dimension_tot(0)==zone_dis_base.nb_elem_tot());
124 assert(tabHf.dimension_tot(0)==zone_vf.nb_faces_tot());
125
126 const IntTab& elem_faces=zone_vf.elem_faces();
127 const DoubleVect& volumes=zone_vf.volumes();
128 const DoubleVect& volumes_entrelaces=zone_vf.volumes_entrelaces();
129
130 tabHf=0;
131 int nb_face_elem=elem_faces.dimension(1);
132 int nb_elem_tot=zone_dis_base.nb_elem_tot();
133
134 double coeffb=nb_face_elem;
135 double coeffi=coeffb;
136 if (zone_vf.que_suis_je()=="Zone_VDF")
137 {
138 coeffb=1;
139 coeffi=2;
140 }
141
142// TODO : FIXME
143// XXX : codage pas coherent... Ã  voir : volumes ou volumes_entrelaces ???
144 if (tabHe.line_size() == 1)
145 {
146 DoubleTab vol_tot(tabHf);
147 for (int ele=0; ele<nb_elem_tot; ele++)
148 for (int s=0; s<nb_face_elem; s++)
149 {
150 tabHf(elem_faces(ele,s))+=tabHe(ele,0)*volumes(ele);
151 vol_tot(elem_faces(ele,s)) += volumes(ele);
152 }
153 for (int f = 0; f < zone_vf.nb_faces(); f++)
154 tabHf(f) /= vol_tot(f);
155 }
156 else
157 {
158// TODO : factorize these cases ...
159 if (tabHf.nb_dim()==1)
160 {
161 assert(coeffi==2);
162 assert(coeffb==1);
163// VDF
164// abort();
165 for (int ele=0; ele<nb_elem_tot; ele++)
166 {
167 for (int s=0; s<nb_face_elem; s++)
168 {
169 int face=elem_faces(ele,s);
170 for (int r = 0; r < zone_vf.dimension; r++)
171 {
172// Change of basis N.K.N, with N the normal of the face, and K the tensorial coefficient to get the value of the diffusivity
173// on the direction of the surface normal.
174 double normOnSurf = zone_vf.face_normales(face, r) / zone_vf.face_surfaces(face);
175 tabHf(face) += tabHe(ele, r) * normOnSurf * normOnSurf * volumes(ele);
176 }
177 }
178 }
179 for (int f=0; f<zone_vf.premiere_face_int(); f++)
180 tabHf(f)/=volumes_entrelaces(f)*coeffb;
181 for (int f=zone_vf.premiere_face_int(); f<zone_vf.nb_faces(); f++)
182 tabHf(f)/=volumes_entrelaces(f)*coeffi;
183 }
184 else
185 {
186// abort();
187 int nb_comp=tabHf.dimension(1);
188 for (int ele=0; ele<nb_elem_tot; ele++)
189 {
190 for (int s=0; s<nb_face_elem; s++)
191 {
192 int face=elem_faces(ele,s);
193 for (int comp=0; comp<nb_comp; comp++)
194 tabHf(face,comp)+=tabHe(ele,comp)*volumes(ele);
195 }
196 }
197
198 for (int f=0; f<zone_vf.premiere_face_int(); f++)
199 for (int comp=0; comp<nb_comp; comp++)
200 tabHf(f,comp)/=volumes_entrelaces(f)*coeffb;
201 for (int f=zone_vf.premiere_face_int(); f<zone_vf.nb_faces(); f++)
202 for (int comp=0; comp<nb_comp; comp++)
203 tabHf(f,comp)/=volumes_entrelaces(f)*coeffi;
204
205 }
206 }
207 tabHf.echange_espace_virtuel();
208// Cerr<<min_array(tabHe)<<" elem "<<max_array(tabHe)<<finl;
209// Cerr<<min_array(tabHf)<<" face "<<max_array(tabHf)<<finl;
210
211 Debog::verifier("element_face sortie",tabHf);
212}
