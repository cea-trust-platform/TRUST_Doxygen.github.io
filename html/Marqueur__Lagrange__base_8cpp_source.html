47void Marqueur_Lagrange_base::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
48{
49 const Zone_dis_base& zone_dis=pb.equation(0).zone_dis();
50 const Zone& zone = pb.domaine().zone(0);
51 double temps = pb.schema_temps().temps_courant();
52 Nom nom="densite_particules";

54 dis.discretiser_champ("champ_elem",zone_dis,nom,unite,1,temps,densite_particules_);
55 champs_compris_.ajoute_champ(densite_particules_.valeur());
56
57 ensemble_points().associer_zone(zone);
58 if (t_debut_integr_==-1.)
59 t_debut_integr_ = pb.schema_temps().temps_init();
60}
