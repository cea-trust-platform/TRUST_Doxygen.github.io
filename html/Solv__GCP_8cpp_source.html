55Entree& Solv_GCP::readOn(Entree& is )
56{
57 int precond_nul;
58 int impr,quiet;
59 Param param((*this).que_suis_je());
60 param.ajouter("seuil",&seuil_,Param::REQUIRED);
61 param.ajouter("nb_it_max",&nb_it_max_);
62 param.ajouter_flag("impr",&impr);
63 param.ajouter_flag("quiet",&quiet);
64 param.ajouter_flag("save_matrice|save_matrix",&save_matrice_);
65 param.ajouter("precond",&le_precond_);
66 param.ajouter_flag("precond_nul",&precond_nul);
67 param.ajouter_flag("precond_diagonal", &precond_diag_);
68 param.ajouter_flag("optimized", &optimized_);
69 param.lire_avec_accolades_depuis(is);
70// Obligation de definir un precond
71 if (!le_precond_.non_nul() && precond_nul==0 && precond_diag_==0)
72 {
73 Cerr << "You forgot to define a preconditionner with the keyword precond." << finl;
74 Cerr << "If you don't want a preconditionner, add for the solver definition:" << finl;
75 Cerr << "precond_nul" << finl;
76 Process::exit();
77 }
78 if (precond_nul)
79 {
80 le_precond_.detach();
81 }
82 assert(seuil_>0);
83 fixer_limpr(impr);
84 if (quiet)
85 fixer_limpr(-1);
86 return is;
87}

116void Solv_GCP::prepare_data(const Matrice_Base& matrice, const DoubleVect& secmem, DoubleVect& solution)
117{
118 if (reinit_ == 0)
119 {
120// Reconstruction de toute la structure (tableaux d'index et coefficients)
121
122 if (secmem.line_size() != 1)
123 {
124 Cerr << "Error line_size>1 not coded (GCP)" << finl;
125 exit();
126 }
127
128 const Matrice_Bloc& mat_bloc = ref_cast(Matrice_Bloc, matrice);
129 const Matrice_Morse_Sym& mat = ref_cast(Matrice_Morse_Sym, mat_bloc.get_bloc(0,0).valeur());
130 const Matrice_Morse& mat_virt = ref_cast(Matrice_Morse, mat_bloc.get_bloc(0,1).valeur());
131
132// Determination du nombre d'items reellement utilises:
133 {
134 const int sztot_source = secmem.size_array();
135 const int sz = secmem.size();
136 renum_.reset();
137 renum_.resize(sztot_source, Array_base::NOCOPY_NOINIT);
138 renum_ = 0;
139// Retirer les items virtuels
140 int i;
141 for (i = sz; i < sztot_source; i++)
142 renum_[i] = -1;
143 renum_.set_md_vector(secmem.get_md_vector());
144 const ArrOfInt& tab2 = mat_virt.get_tab2();
145 const int n = tab2.size_array();
146 for (i = 0; i < n; i++)
147 {
148// Attention: tab2 de la partie reele-virtuelle contient des indices
149// relatifs au debut de la partie virtuelle (d'ou "+ sz")
150 const int j = tab2[i]-1 + sz; //!< fortran -> c
151 renum_[j] = 0;
152 }
153 }
154// Determination du nombre de lignes non vides de mat_virt
155 int nb_lignes_mat_virt = 0;
156 {
157 const int n = mat_virt.get_tab1().size_array() - 1;
158 for (int i = 0; i < n; i++)
159 if (mat_virt.get_tab1()(i+1) - mat_virt.get_tab1()(i) > 0)
160 nb_lignes_mat_virt++;
161 }
162
163// Descripteur contenant uniquement les items utiles:
164 MD_Vector md;
165 MD_Vector_tools::creer_md_vect_renum_auto(renum_, md);
166 const int sz_tot = md.valeur().get_nb_items_tot();
167 const int sz = md.valeur().get_nb_items_reels();
168
169// Calcul de la taille memoire requise:
170 int mem_size = 0;
171 mem_size += sz_tot * sizeof(double); //!< vecteurs avec espace virtuel (tmp_p_)
172 mem_size += sz * sizeof(double) * 3; //!< vecteurs sans espace virtuel
173 const int nb_lignes_mat = sz;
174// matrice reel/reel
175 mem_size += (sz + 1) * sizeof(int); //!< pour tab1_
176 int nnz_reel_reel = 0;
177 assert(mat.get_tab1().size_array() == sz + 1);
178 assert(mat.get_tab2().size_array() == mat.get_coeff().size_array());
179 if (! precond_diag_)
180 {
181 nnz_reel_reel = mat.get_coeff().size_array();
182 }
183 else
184 {
185// On ne stocke pas les coefficients diagonaux:
186 nnz_reel_reel = mat.get_coeff().size_array() - sz;
187 }
188 mem_size += nnz_reel_reel * sizeof(double); //!< pour les coefficients
189 mem_size += nnz_reel_reel * sizeof(int); //!< pour les indices matrice reel/virtuel
190//
191 mem_size += (nb_lignes_mat_virt+1) * sizeof(int); //!< pour tab1_
192 const int nnz_reel_virtuel = mat_virt.get_coeff().size_array();
193 assert(mat_virt.get_tab2().size_array() == nnz_reel_virtuel);
194 mem_size += nnz_reel_virtuel * sizeof(double); //!< pour les coefficients
195 mem_size += nnz_reel_virtuel * sizeof(int); //!< pour les indices taille de tmp_mat_virt_.lignes_non_vides_
196//
197 mem_size += nb_lignes_mat_virt * sizeof(int);
198// aligner la taille sur un multiple de 8
199 if (mem_size % 8 != 0)
200 mem_size = (mem_size/8+1)*8;
201
202// Allocation des tableaux:
203// (on met d'abord tous les tableaux de double, puis a la fin les tableaux d'entiers
204// sinon il faut ajouter du padding pour aligner si on remet des double apres de int)
205//
206 Journal() << "Solv_GCP::prepare allocating data chunk : " << mem_size << " bytes" << finl;
207 tmp_data_block_.resize_array(mem_size/8, Array_base::NOCOPY_NOINIT);
208
209 double *ptr = tmp_data_block_.addr();
210 resu_.ref_data(ptr, sz);
211 ptr += sz;
212 residu_.ref_data(ptr, sz);
213 ptr += sz;
214 tmp_p_avec_items_virt_.ref_data(ptr, sz_tot); //!< avec espace virtuel
215 tmp_p_avec_items_virt_.set_md_vector(md);
216// tmp_p_ pointe sur la meme zone:
217 tmp_p_.ref_data(ptr, sz); //!< sans l'espace virtuel
218 ptr += sz_tot;
219 tmp_solution_.ref_data(ptr, sz);
220 ptr += sz;
221// Allocation des tableaux pour les matrices:
222 tmp_mat_.get_set_coeff().ref_data(ptr, nnz_reel_reel);
223 ptr += nnz_reel_reel;
224 tmp_mat_virt_.get_set_coeff().ref_data(ptr, nnz_reel_virtuel);
225 ptr += nnz_reel_virtuel;
226// On a fini les double, on passe aux tableaux d'entiers:
227 int * iptr = (int*)ptr;
228 tmp_mat_.get_set_tab1().ref_data(iptr, nb_lignes_mat + 1);
229 iptr += nb_lignes_mat + 1;
230 tmp_mat_.get_set_tab2().ref_data(iptr, nnz_reel_reel);
231 iptr += nnz_reel_reel;
232 tmp_mat_virt_.lignes_non_vides_.ref_data(iptr, nb_lignes_mat_virt);
233 iptr += nb_lignes_mat_virt;
234 tmp_mat_virt_.get_set_tab1().ref_data(iptr, nb_lignes_mat_virt + 1);
235 iptr += nb_lignes_mat_virt + 1;
236 tmp_mat_virt_.get_set_tab2().ref_data(iptr, nnz_reel_virtuel);
237 iptr += nnz_reel_virtuel;
238// Allocation terminee.
239 assert(((char*)iptr) <= ((char*)tmp_data_block_.addr() + mem_size));
240
241// Remplissages des tableaux d'index (tab1_, tab2_ et lignes_non_vides_)
242 if (! precond_diag_)
243 {
244 tmp_mat_.get_set_tab1().inject_array(mat.get_tab1());
245 {
246// remplissage de tab2 (renumerotation eventuelle)
247 for (int i = 0; i < nnz_reel_reel; i++)
248 {
249 int j = mat.get_tab2()(i)-1; //!< fortran->c
250 int rj = renum_[j];
251 assert(rj < sz_tot);
252 tmp_mat_.get_set_tab2()(i) = rj+1; //!< c->fortran
253 }
254 }
255 tmp_mat_.set_nb_columns( sz_tot );
256 }
257 else
258 {
259// Construction de la matrice D^(-1/2) * A * D^(-1/2)
260// on ne stocke pas les coeffs diagonaux
261// Le remplissage de tab1_ n'est pas trivial, du coup:
262 {
263 int src_index = 0; //!< index dans mat.tab2_ et coeff_
264 int dest_index = 0; //!< index dans tmp_mat_.tab2_ et coeff_
265 int i_ligne;
266 for (i_ligne = 0; i_ligne < nb_lignes_mat; i_ligne++)
267 {
268// A chaque ligne on a un coefficient de moins que dans la matrice d'origine
269// (on ne met pas le coeff diagonal)
270 tmp_mat_.get_set_tab1()(i_ligne) = dest_index + 1; //!< indice fortran du debut de ligne
271 const int ncoeff = mat.get_tab1()(i_ligne+1) - mat.get_tab1()(i_ligne) - 1;
272// Ne pas inserer le coeff diagonal
273 assert(mat.get_tab2()(src_index) == i_ligne + 1); //!< index fortran
274 src_index++;
275// Inserer les autres coeffs:
276 for (int i = 0; i < ncoeff; i++, src_index++, dest_index++)
277 tmp_mat_.get_set_tab2()(dest_index) = mat.get_tab2()(src_index);
278
279 }
280// Fin de la derniere ligne:
281 tmp_mat_.get_set_tab1()(i_ligne) = dest_index + 1; //!< indice fortran du debut de ligne
282 }
283 tmp_mat_.set_nb_columns( sz_tot );
284 }
285// remplissage de tmp_mat_virt_
286 {
287 tmp_mat_virt_.get_set_tab1()[0] = 1;
288 int i_ligne_dest = 0;
289 int dest_index = 0;
290 for (int i_ligne = 0; i_ligne < nb_lignes_mat; i_ligne++)
291 {
292 const int count = mat_virt.get_tab1()(i_ligne+1) - mat_virt.get_tab1()(i_ligne);
293 if (count > 0)
294 {
295 tmp_mat_virt_.lignes_non_vides_[i_ligne_dest] = i_ligne + 1; //!< indice fortran
296 tmp_mat_virt_.get_set_tab1()[i_ligne_dest] = dest_index + 1; //!< index fortran
297 i_ligne_dest++;
298 int src_index = mat_virt.get_tab1()(i_ligne) - 1; //!< fortran->c
299 for (int i = 0; i < count; i++, src_index++, dest_index++)
300 {
301// mat_virt contient des indices fortran relatifs au debut de la partie virtuelle,
302// on transform en indice C, relatif au vecteur complet
303 int j = mat_virt.get_tab2()(src_index) + sz - 1;
304 int rj = renum_[j];
305// on stocke dans tmp_mat_virt des indices de colonnes relatifs au vecteur complet
306// (pas seulement la partie virtuelle)
307 tmp_mat_virt_.get_set_tab2()[dest_index] = rj + 1; //!< indice fortran
308 }
309 }
310 }
311// Fin de la derniere ligne:
312 tmp_mat_virt_.get_set_tab1()[i_ligne_dest] = dest_index + 1; //!< index fortran
313 }
314 reinit_ = 1;
315 }
316
317 if (reinit_ < 2)
318 {
319 const Matrice_Bloc& mat_bloc = ref_cast(Matrice_Bloc, matrice);
320 const Matrice_Morse_Sym& mat = ref_cast(Matrice_Morse_Sym, mat_bloc.get_bloc(0,0).valeur());
321 const Matrice_Morse& mat_virt = ref_cast(Matrice_Morse, mat_bloc.get_bloc(0,1).valeur());
322 if (!precond_diag_)
323 {
324 tmp_mat_.get_set_coeff().inject_array(mat.get_coeff());
325 tmp_mat_virt_.get_set_coeff().inject_array(mat_virt.get_coeff());
326 }
327 else
328 {
329// calcul de D^(-1/2)
330 exit();
331// calcul du produit D^(-1/2) * A * D^(-1/2)
332
333 }
334 reinit_ = 2;
335 }
336
337 if (!precond_diag_)
338 {
339 tmp_solution_.inject_array(solution, tmp_solution_.size());
340 resu_.inject_array(secmem, resu_.size_array());
341 }
342 else
343 {
344 exit();
345 }
346}

348// Calcul de vx = vx * alpha - vy
349static void multiply_sub(DoubleVect& vx, DoubleVect& vy, double alpha)
350{
351 int n = vx.size_reelle_ok() ? vx.size() : vx.size_totale();
352 assert(vy.size_totale() >= n);
353 double *x_ptr = vx.addr();
354 double *y_ptr = vy.addr();
355 for (n = n - 1; n > 0; n -= 2, x_ptr += 2, y_ptr += 2)
356 {
357 double a = x_ptr[0] * alpha - y_ptr[0];
358 double b = x_ptr[1] * alpha - y_ptr[1];
359 x_ptr[0] = a;
360 x_ptr[1] = b;
361 }
362// n etait-il impair au depart ?
363 if (n == 0)
364 x_ptr[0] = x_ptr[0] * alpha - y_ptr[0];
365}
366
367// Calcul de vx += alpha * vy
368// Valeur de retour: somme locale sur ce processeur des vx[i]*vx[i] (apres ajout)
369// Attention: pas code pour les items communs
370static double ajoute_alpha_v_norme(DoubleVect& vx, double alpha, DoubleVect& vy)
371{
372 int n = vx.size();
373 assert(vy.size() == n);
374 double *x_ptr = vx.addr();
375 double *y_ptr = vy.addr();
376 double norme1 = 0., norme2 = 0.;
377 for (n = n - 1; n > 0; n -= 2, x_ptr += 2, y_ptr += 2)
378 {
379 double a = x_ptr[0] + alpha * y_ptr[0];
380 double b = x_ptr[1] + alpha * y_ptr[1];
381 x_ptr[0] = a;
382 x_ptr[1] = b;
383 norme1 += a * a;
384 norme2 += b * b;
385 }
386// n etait-il impair au depart ?
387 if (n == 0)
388 {
389 double a = x_ptr[0] + alpha * y_ptr[0];
390 x_ptr[0] = a;
391 norme1 += a * a;
392 }
393 return norme1 + norme2;
394}

396int Solv_GCP::resoudre_(const Matrice_Base& matrice,
397 const DoubleVect& secmem,
398 DoubleVect& solution,
399 int nmax)
400{
401 const int n_items_reels = solution.size_reelle_ok() ? solution.size_reelle() : solution.size_totale();
402 {
403 const int nb_items_seq = solution.get_md_vector().valeur().nb_items_seq_tot();
404 const int ls = secmem.line_size();
405 const int nb_inco_tot = nb_items_seq * ls;
406 nmax = std::max(nb_inco_tot, nmax);
407 }
408
409 const int avec_precond = le_precond_.non_nul();
410 const int precond_requires_echange_espace_virtuel =
411 avec_precond && (le_precond_.valeur().get_flag_updated_input());
412
413 const int optimized = optimized_;
414
415 if (optimized)
416 {
417 prepare_data(matrice, secmem, solution);
418 }
419 else
420 {
421 resu_.reset();
422 residu_.reset();
423 tmp_p_avec_items_virt_.reset();
424 resu_.copy(solution, Array_base::NOCOPY_NOINIT);
425 residu_.copy(solution, Array_base::NOCOPY_NOINIT);
426 tmp_p_avec_items_virt_.copy(solution, Array_base::NOCOPY_NOINIT);
427 tmp_p_.ref(tmp_p_avec_items_virt_);
428 tmp_solution_.ref(solution);
429 resu_.inject_array(secmem);
430 }
431
432 tmp_p_avec_items_virt_.inject_array(tmp_solution_, n_items_reels);
433 tmp_p_avec_items_virt_.echange_espace_virtuel();
434// On n'a pas besoin que residu ait son espace virtuel a jour
435// En revanche, on a besoin de l'espace virtuel de tmp_p_...
436 if (optimized)
437 {
438 tmp_mat_.multvect_(tmp_p_avec_items_virt_, residu_);
439// calcul du produit, le produit scalaire n'est pas utilise:
440 tmp_mat_virt_.ajouter_mult_vect_et_prodscal(tmp_p_avec_items_virt_, residu_);
441 }
442 else
443 {
444 matrice.multvect(tmp_p_avec_items_virt_, residu_);
445 }
446// ATTENTION: on suppose que secmem a ete copie dans resu_
447 operator_sub(residu_, resu_, VECT_REAL_ITEMS); //!< ne pas toucher a l'espace virtuel
448
449 if (avec_precond)
450 {
451 if (precond_requires_echange_espace_virtuel)
452 residu_.echange_espace_virtuel();
453
454 if (optimized)
455 le_precond_.valeur().preconditionner(tmp_mat_, residu_, tmp_p_);
456 else
457 le_precond_.valeur().preconditionner(matrice, residu_, tmp_p_);
458
459 }
460 else
461 {
462 tmp_p_.inject_array(residu_, n_items_reels);
463 }
464 double dold = mp_prodscal(residu_, tmp_p_);
465
466 operator_negate(tmp_p_, VECT_REAL_ITEMS);
467
468 double norme = mp_norme_vect(residu_);
469 double norme_b = mp_norme_vect(resu_);
470
471 if (limpr()==1)
472 {
473 double norme_relative=(norme_b>DMINFLOAT?norme/(norme_b+DMINFLOAT):norme);
474 Cout << "Norm of the residue: " << norme << " (" << norme_relative << ")" << finl;
475 }
476 int niter = 0;
477 int nb_it_max=nmax;
478 if (nb_it_max_>-1)
479 nb_it_max=nb_it_max_;
480 while ( ( norme > seuil_ ) && (niter++ < nmax) &&( niter<nb_it_max))
481 {
482// Precondition pour multvect
483// (le seul echange espace virtuel de l'algo sauf si le precond en a besoin)
484 tmp_p_avec_items_virt_.echange_espace_virtuel();
485// En revanche, on n'a pas besoin de l'espace virtuel a jour de resu:
486 double resu_scalaire_p_local;
487 if (optimized)
488 {
489 resu_scalaire_p_local = tmp_mat_.multvect_et_prodscal(tmp_p_avec_items_virt_, resu_);
490 resu_scalaire_p_local += tmp_mat_virt_.ajouter_mult_vect_et_prodscal(tmp_p_avec_items_virt_, resu_);
491 }
492 else
493 {
494 matrice.multvect(tmp_p_avec_items_virt_, resu_);
495 resu_scalaire_p_local = local_prodscal(resu_, tmp_p_avec_items_virt_);
496 }
497 const double resu_scalaire_p = mp_sum(resu_scalaire_p_local);
498 const double alfa = dold / resu_scalaire_p;
499 ajoute_alpha_v(tmp_solution_, alfa, tmp_p_, VECT_REAL_ITEMS);
500
501 double norme_residu_locale;
502 if (optimized)
503 {
504 norme_residu_locale = ajoute_alpha_v_norme(residu_, alfa, resu_);
505 }
506 else
507 {
508 ajoute_alpha_v(residu_, alfa, resu_);
509 norme_residu_locale = local_carre_norme_vect(residu_);
510 }
511
512 if(avec_precond)
513 {
514 if (precond_requires_echange_espace_virtuel)
515 residu_.echange_espace_virtuel();
516 if (optimized)
517 le_precond_.valeur().preconditionner(tmp_mat_, residu_, resu_);
518 else
519 le_precond_.valeur().preconditionner(matrice, residu_, resu_);
520
521 double residu_scalaire_resu = local_prodscal(residu_, resu_);
522 norme = norme_residu_locale;
523// optimisation: on calcule en une seule fois les deux sommes
524 mpsum_multiple(residu_scalaire_resu, norme);
525 assert(residu_scalaire_resu >= 0);
526 multiply_sub(tmp_p_, resu_, residu_scalaire_resu / dold);
527 dold = residu_scalaire_resu;
528 }
529 else
530 {
531 const double dnew = mp_carre_norme_vect(residu_);
532 norme = mp_sum(norme_residu_locale);
533 assert(dnew >= 0);
534 multiply_sub(tmp_p_, residu_, dnew / dold);
535 dold = dnew;
536 }
537 norme = sqrt(norme);
538
539 if (limpr()==1)
540 {
541 Cout << norme << " ";
542 if ((niter % 15) == 0) Cout << finl ;
543 }
544 }
545 if ((nb_it_max_<0)&& (norme > seuil_))
546 {
547 Cerr << "No convergence after : " << niter << " iterations\n";
548 Cerr << " Residue : "<< norme << "\n";
549 Cerr << " threshold : "<< seuil_ << "\n";
550 Cerr << "Change your data set." << finl;
551 exit();
552 }
553
554 if (optimized)
555 solution.inject_array(tmp_solution_, n_items_reels);
556
557// The user wants a result with updated virtual space:
