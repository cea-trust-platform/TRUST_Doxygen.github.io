70const Champ_base& Champ_Generique_Tparoi_VEF::get_champ(Champ& espace_stockage) const
71{
72
73 Champ source_espace_stockage;
74 const Champ_base& source = get_source(0).get_champ(source_espace_stockage);
75
76 const Zone_dis_base& zone_dis = get_ref_zone_dis_base();
77 Nature_du_champ nature_source = source.nature_du_champ();
78 Noms noms;
79 Noms unites;
80 noms.add("bidon");
81 unites.add("bidon");
82 int nb_comp = 1;
83 double temps;
84 temps=0.;
85
86 Champ_Fonc espace_stockage_fonc;
87 const Discretisation_base& discr = get_discretisation();
88 Motcle directive = get_directive_pour_discr();
89 discr.discretiser_champ(directive,zone_dis,nature_source,noms,unites,nb_comp,temps,espace_stockage_fonc);
90 espace_stockage = espace_stockage_fonc;
91
92 DoubleTab& valeurs_espace = espace_stockage->valeurs();
93 const DoubleTab& source_so_val = source.valeurs();
94 const DoubleTab& inconnue = source_so_val;
95 valeurs_espace = source_so_val;
96
97// Sur toutes les faces ou le flux est impose si le flux turbulent pas calcule on recalcule la temperature
98// on recuprere l'eqn de temp
99 const Equation_base& my_eqn=ref_cast(Champ_Inc_base,source).equation();
100 const RefObjU& modele_turbulence = my_eqn.get_modele(TURBULENCE);
101 if ( modele_turbulence.non_nul() && sub_type(Modele_turbulence_scal_base,modele_turbulence.valeur()))
102 {
103 const Modele_turbulence_scal_base& mod_turb_scal = ref_cast(Modele_turbulence_scal_base,modele_turbulence.valeur());
104 const Turbulence_paroi_scal& loiparth = mod_turb_scal.loi_paroi();
105
106 if( loiparth.valeur().use_equivalent_distance() )
107 {
108// const Paroi_scal_hyd_base_VEF& paroi_scal_vef = ref_cast(Paroi_scal_hyd_base_VEF,loiparth.valeur());
109
110 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF,my_eqn.zone_dis().valeur());
111 const IntTab& elem_faces = zone_VEF.elem_faces();
112 const DoubleVect& vol = zone_VEF.volumes();
113 const IntTab& face_voisins = zone_VEF.face_voisins();
114 const DoubleTab& face_normale = zone_VEF.face_normales();
115 const Zone_Cl_VEF& zone_Cl_VEF=ref_cast(Zone_Cl_VEF,my_eqn.zone_Cl_dis().valeur());
116 int nb_dim_pb=Objet_U::dimension;
117 DoubleVect le_mauvais_gradient(nb_dim_pb);
118 int nb_front=zone_VEF.nb_front_Cl();
119 for (int n_bord=0; n_bord<nb_front; n_bord++)
120 {
121 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
122 const Cond_lim_base& cl_base=la_cl.valeur();
123 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
124
125// Les lois de parois ne s'appliquent qu'aux cas ou la CL
126// est de type temperature imposee, car dans les autres cas
127// (flux impose et adiabatique) le flux a la paroi est connu et fixe.
128 int ldp_appli=0;
129 if ((sub_type(Neumann_paroi,cl_base))||(sub_type(Neumann_homogene,cl_base)))
130 {
131 ldp_appli=1;
132// if (paroi_scal_vef.get_flag_calcul_ldp_en_flux_impose() )
133 if (loiparth.valeur().get_flag_calcul_ldp_en_flux_impose() )
134 ldp_appli=0;
135 }
136
137 if (ldp_appli)
138 {
139// const DoubleVect& d_equiv = paroi_scal_vef.equivalent_distance(n_bord);
140 const DoubleVect& d_equiv = loiparth.valeur().equivalent_distance(n_bord);
141// d_equiv contient la distance equivalente pour le bord
142// Dans d_equiv, pour les faces qui ne sont pas paroi_fixe (eg periodique, symetrie, etc...)
143// il y a la distance geometrique grace a l'initialisation du tableau dans la loi de paroi.
144
145 int num1 = 0;
146 int num2 = le_bord.nb_faces_tot();
147 for (int ind_face=num1; ind_face<num2; ind_face++)
148 {
149// Tf est la temperature fluide moyenne dans le premier element
150// sans tenir compte de la temperature de paroi.
151 double Tf=0.;
152// double bon_gradient=0.; // c'est la norme du gradient de temperature normal a la paroi calculee a l'aide de la loi de paroi.
153 le_mauvais_gradient=0.;
154 int num_face = le_bord.num_face(ind_face);
155 int elem1 = face_voisins(num_face,0);
156 if (elem1==-1) elem1 = face_voisins(num_face,1);
157 double surface_face = zone_VEF.face_surfaces(num_face);
158
159 int nb_faces_elem = zone_VEF.zone().nb_faces_elem();
160 for (int i=0; i<nb_faces_elem; i++)
161 {
162 int j = elem_faces(elem1,i);
163 if ( j != num_face )
164 {
165 double surface_pond = 0.;
166 for (int kk=0; kk<nb_dim_pb; kk++)
167 surface_pond -= (face_normale(j,kk)*zone_VEF.oriente_normale(j,elem1)*face_normale(num_face,kk)*
168 zone_VEF.oriente_normale(num_face,elem1))/(surface_face*surface_face);
169 Tf+=inconnue(j)*surface_pond;
170 }
171
172 for(int kk=0; kk<nb_dim_pb; kk++)
173 le_mauvais_gradient(kk)+=inconnue(j)*face_normale(j,kk)*zone_VEF.oriente_normale(j,elem1);
174 }
175 le_mauvais_gradient/=vol(elem1);
176// mauvais_gradient = le_mauvais_gradient.n
177 double mauvais_gradient=0;
178 for(int kk=0; kk<nb_dim_pb; kk++)
179 mauvais_gradient+=le_mauvais_gradient(kk)*face_normale(num_face,kk)/surface_face;
180
181// valeurs_espace(num_face) est la temperature de paroi : Tw.
182// On se fiche du signe de bon gradient car c'est la norme du gradient
183// de temperature dans l'element.
184// Ensuite ce sera multiplie par le vecteur normal a la face de paroi
185// qui lui a les bons signes.
186// bon_gradient=(Tf-inconnue(num_face))/d_equiv(ind_face)*(-zone_VEF.oriente_normale(num_face,elem1));
187// on doit faire bon=mauvais...
188// on fait le calcul de facon a avoir bon_grad =mauva
189 valeurs_espace(num_face)=Tf-mauvais_gradient*d_equiv(ind_face)*(-zone_VEF.oriente_normale(num_face,elem1));
190 }
191 }
192 }
193 }
194 }
195 DoubleTab& espace_valeurs = espace_stockage->valeurs();
196 espace_valeurs.echange_espace_virtuel();
197
198 return espace_stockage.valeur();
199}
