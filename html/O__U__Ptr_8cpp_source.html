102//! Verifie si le pointeur est valide. Le pointeur est valide<br>si cle_==-1<br>ou si la_memoire().objet_u(cle_) a le meme object_id_ que<br>celui enregistre dans ptr_object_id_.<br>Si le pointeur n'est pas valide, arret du programme.<br>Renvoie l'adresse de l'objet pointe (de type Objet_U)
103Objet_U * O_U_Ptr::get_O_U_Ptr_check() const
104{
105 if (cle_ != -1 || ptr_object_id_ != -1)
106 {
107 Objet_U& objet = Memoire::Instance().objet_u(cle_);
108 Objet_U * addr = &objet;
109 const int id = objet.get_object_id();
110 if (id == ptr_object_id_)
111 return addr;
112 else
113 {
114 Cerr << "(PE" << me() << ") ";
115 Cerr << "Error in O_U_Ptr::get_O_U_Ptr_check() : id != ptr_object_id_\n";
116 Cerr << " Pointer type " << le_type();
117 Cerr << "\n cle_ = " << cle_;
118 Cerr << "\n ptr_object_id_ = " << ptr_object_id_;
119 Cerr << "\n id = " << id;
120 std::cerr << "\n &la_memoire().objet_u(cle_) = " << addr << std::endl;
121// Si ca plante a cet endroit, c'est que l'objet en reference
122// a ete detruit et que la reference est encore utilisee.
123 assert(0);
124 exit();
125 }
126 }
127 return 0;
128}

132//! Verifie que l'objet pointe par ptr est d'un type acceptable<br>pour le pointeur (avec get_info_ptr)
133int O_U_Ptr::check_O_U_Ptr_type(const Objet_U * ptr) const
134{
135 if (ptr == 0)
136 return 1; //!< Le pointeur nul est valide
137
138 const Objet_U& objet = *ptr;
139// On verifie que l'objet est du bon type :
140// type accepte par le pointeur :
141 const Type_info& type_info_ptr = get_info_ptr();
142// type de l'objet :
143 const Type_info& type_info_obj = *(objet.get_info());
144// Peut-on convertir type_info_obj en type_info_ptr ?
145 if (! type_info_ptr.can_cast(&type_info_obj))
146 {
147 Cerr << "(PE" << me() << ") ";
148 Cerr << "Error in O_U_Ptr::get_O_U_Ptr_check() : Type error\n";
149 Cerr << " Pointer type " << le_type();
150 Cerr << "\n cle_ = " << cle_;
151 Cerr << "\n ptr_object_id_ = " << ptr_object_id_;
152 std::cerr << "\n &memoire.objet_u(cle_) = " << ptr;
153 Cerr << "\n Type accepted by the pointer : " << type_info_ptr.name();
154 Cerr << "\n Object type in reference : " << type_info_obj.name();
155 assert(0);
156 exit();
157 }
158 return 1;
159}
