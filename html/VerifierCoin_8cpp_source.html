83//! Fonction principale de l'interprete:<br>resoudre un probleme<br>On cherche dynamiquement le type du probleme a resoudre<br>on resoud le probleme et on effectue les postraitements.
84/** \return Entree\&<br>
85Signification: le flot d'entree modifie<br>
86Contraintes: */
87Entree& VerifierCoin::interpreter_(Entree& is)
88{
89// XD verifiercoin interprete verifiercoin -1 This keyword subdivides inconsistent 2D/3D cells used with VEFPreP1B discretization. Must be used before the mesh is discretized. NL1 The Read_file option can be used only if the file.decoupage_som was previously created by TRUST. This option, only in 2D, reverses the common face at two cells (at least one is inconsistent), through the nodes opposed. In 3D, the option has no effect. NL2 The expert_only option deactivates, into the VEFPreP1B divergence operator, the test of inconsistent cells.
90// XD attr domain_name ref_domaine dom 0 Name of the domaine
91// XD attr bloc verifiercoin_bloc bloc 0 not_set
92// XD verifiercoin_bloc objet_lecture nul 1 not_set
93// XD attr Read_file|Lire_fichier chaine filename 1 name of the *.decoupage_som file
94// XD attr expert_only rien expert_only 1 to not check the mesh
95
96
97 associer_domaine(is);
98 Domaine& dom=domaine();
99 Zone& zone=dom.zone(0);
100 if (zone.type_elem()->que_suis_je() != "Triangle" && zone.type_elem()->que_suis_je() != "Tetraedre")
101 {
102 Cerr << "Error for "<<que_suis_je() <<" interpreter : it can be applied only for triangular or tetraedral meshing." << finl;
103 exit();
104 }
105
106 Nom decoup_som("");
107 Param param(que_suis_je());
108 param.ajouter("Lire_fichier|Read_file",&decoup_som);
109 param.ajouter_flag("expert_only",&expert_only);
110 param.lire_avec_accolades_depuis(is);
111 int lecture_decoupage_som=(decoup_som!=""?1:0);
112
113 if (Process::nproc()>1 && expert_only==0)
114 {
115 Cerr << que_suis_je() << " interpreter can be used only for sequential calculation." << finl;
116 Cerr << "For parallel calculation, please use "<< que_suis_je() <<" interpreter" << finl;
117 Cerr << "during the domain partitioning step." << finl;
118 exit();
119 }
120// Pas de changement du maillage si option expert_only utilise
121 Journal() << "expert_only= " << expert_only << finl;
122 if (!expert_only)
123 {
124 Scatter::uninit_sequential_domain(dom);
125 DoubleTab xp;
126 zone.calculer_centres_gravite(xp);
127 DoubleTab& sommets = dom.les_sommets();
128 int nbsom=sommets.dimension(0);
129
130 IntTab& les_elems=zone.les_elems();
131 int nbelem=zone.nb_elem();
132 int elem,ns,ns1,ne,somm;
133
134 IntLists sommets_associes(nbsom);
135 IntLists elements_associes(nbsom);
136
137 IntVect test_double(nbsom);
138 test_double=0;
139
140// On parcourt les elements et pour un element ne, on parcourt ses sommets. Pour chaque sommet on detecte les
141// sommets "voisins". Une fois tous les elements parcourus, s il n y a pas un doublon dans les sommets voisins
142// d un sommet somm, il faut decouper l element (ou les elements) qui porte ce sommet.
143
144 ne=0;
145
146 while (ne<nbelem)
147 {
148 ns=0;
149
150 while (ns<dimension+1)
151 {
152
153 somm = les_elems(ne,ns);
154 elements_associes[somm].add_if_not(ne);
155 ns1=0;
156 while ((ns1<dimension+1) && (test_double(somm)==0))
157 {
158 if (les_elems(ne,ns1)!=somm)
159 {
160 if (sommets_associes[somm].contient(les_elems(ne,ns1)))
161 test_double(somm)=1;
162 else
163 sommets_associes[somm].add(les_elems(ne,ns1));
164 }
165 ns1++;
166 }
167
168 ns++;
169 }
170 ne++;
171 }
172
173// On decoupe les elements pour le sommet qui pose probleme
174// PQ : 25/05/07
175// - soit de maniere automatique (tout sommet rattache qu'a un seul element)
176// - soit a partir d'une liste de sommets (liste generee lors d'un pre-calcul et tenant compte du type de CL)
177// avec comme option de decoupage dans ce cas (lue dans le fichier "decoupage_som") :
178//
179// 0 : decoupage traditionnel (centre de gravite)
180// 1 : decoupage en passant par le sommet oppose de l'element voisin
181
182 int option_decoupage=-1;
183 int somm_lu,elem_opp,somm_opp,somm1,somm2;
184 int dim_cas,nbsom_cas;
185
186 LecFicDiffuse fic;
187 if (lecture_decoupage_som)
188 {
189 Cerr<<"The file" << decoup_som <<" is checked before reading."<<finl;
190 fic.ouvrir(decoup_som);
191 if(fic.good())
192 {
193 fic >> option_decoupage;
194 fic >> dim_cas;
195 fic >> nbsom_cas;
196 if(dim_cas!=dimension)
197 {
198 Cerr << "Error for " <<que_suis_je()<<"::interpreter_" << finl;
199 Cerr << "The file " << decoup_som << " is planned for a case of dimension " << dim_cas << "D" << finl;
200 Cerr << "while the considered domain " << dom.le_nom() << " has a dimension " << dimension << "D." << finl;
201 Process::exit();
202 }
203 if(nbsom!=nbsom_cas)
204 {
205 Cerr << "Error for " <<que_suis_je()<<"::interpreter_" << finl;
206 Cerr << "The nodes number (" <<nbsom_cas<< ") readen in the file " << decoup_som << finl;
207 Cerr << "dot not correspond to those of the meshing (" << nbsom << ")" << finl;
208 Cerr << "Please check that the file you have specified " << decoup_som << finl;
209 Cerr << "is the one to consider for this meshing." << finl;
210 Process::exit();
211 }
212 }
213 else
214 {
215 Cerr << "Error for " <<que_suis_je()<<"::interpreter_" << finl;
216 Cerr << "Problem while trying to open the file " << decoup_som << finl;
217 Process::exit();
218 }
219 Cerr << "option_decoupage " << option_decoupage << finl;
220 }
221
222 for (somm=0; somm<nbsom; somm++)
223 {
224
225// On decoupe les elements pour le sommet qui pose probleme
226
227 if (test_double(somm)==0)
228 {
229 if(lecture_decoupage_som)
230 {
231 fic >> somm_lu >> somm_opp >> somm1 >> somm2;
232 if (dimension==3)
233 {
234 int somm3;
235 fic >> somm3;
236 }
237 fic >> elem >> elem_opp;
238 if(somm_lu!=-1) somm = somm_lu; //!< -1 indice de fin de fichier
239 if(nbsom<=somm)
240 {
241 Cerr << "Error in VerifierCoin::interpreter" << finl;
242 Cerr << "The node " << somm << " is not found." << finl;
243 Cerr << "Check the .Zones files are up to date with your mesh file." << finl;
244 Process::exit();
245 }
246 }
247 Cerr<<"-> VerifierCoin is applied on the node "<<somm<< " of coordinates: ";
248 for(int dir=0; dir<dimension; dir++) Cerr<<sommets(somm,dir)<<" ";
249 Cerr<<"..."<<finl;
250
251 IntList_Curseur liste_elem(elements_associes[somm]);
252 int size_elem= elements_associes[somm].size();
253
254 if (option_decoupage==1 && dimension==2 && somm_lu!=-1) //!< inversion des sommets
255 {
256 les_elems(elem,0) = somm ;
257 les_elems(elem,1) = somm_opp ;
258 les_elems(elem,2) = somm1 ;
259
260 les_elems(elem_opp,0) = somm ;
261 les_elems(elem_opp,1) = somm_opp ;
262 les_elems(elem_opp,2) = somm2 ;
263 }
264
265 else //!< creation d'un nouveau sommet au centre de gravite de l'element
266 {
267 for (int k=0; k<size_elem; k++)
268 {
269 elem=liste_elem.valeur();
270
271// On cree un sommet au centre de gravite de l'element
272 int nouveau_sommet=sommets.dimension(0);
273 sommets.resize(nouveau_sommet+1, dimension);
274 for(int j=0; j<dimension; j++)
275 sommets(nouveau_sommet,j)=xp(elem,j);
276
277// On divise l'element elem en dimension+1 elements
278 int oldsz=les_elems.dimension(0);
279 les_elems.resize(oldsz+dimension, dimension+1);
280 Cerr << "-> The element number " << elem <<" is cut in " << dimension+1 << " elements." << finl;
281// On recupere les sommets de l'element elem
282 int i0=les_elems(elem,0);
283 int i1=les_elems(elem,1);
284 int i2=les_elems(elem,2);
285
286// On affecte les sommets aux nouveaux elements
287 les_elems(elem,0)=i0;
288 les_elems(elem,1)=i1;
289 les_elems(elem,2)=nouveau_sommet;
290
291 les_elems(oldsz,0)=i1;
292 les_elems(oldsz,1)=i2;
293 les_elems(oldsz,2)=nouveau_sommet;
294
295 les_elems(oldsz+1,0)=i0;
296 les_elems(oldsz+1,1)=i2;
297 les_elems(oldsz+1,2)=nouveau_sommet;
298
299 if (dimension==3)
300 {
301 int i3=les_elems(elem,3);
302
303 les_elems(elem,3)=i3;
304 les_elems(oldsz,3)=i3;
305 les_elems(oldsz+1,3)=i3;
306
307 les_elems(oldsz+2,0)=i0;
308 les_elems(oldsz+2,1)=i1;
309 les_elems(oldsz+2,2)=i2;
310 les_elems(oldsz+2,3)=nouveau_sommet;
311 }
312
313 mettre_a_jour_sous_zone(zone,elem,oldsz,dimension);
314
315 ++liste_elem;
316 }
317 }//!< option_decoupage
318 }
319
320 }
321 Scatter::init_sequential_domain(dom);
322 }
323
324 return is;
325}
