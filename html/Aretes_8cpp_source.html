38// Description :
39// affecte a l'arete numero
40// les faces f1, f2, f3, f4
41// l'arete est de type2_ type ou :
42// type2_ = -1 si arete coin
43// type2_ = 0 si arete bord
44// type2_ = 1 si arete mixte
45// type2_ = 2 si arete interne
46// l'arete est de type1_ dir ou :
47// type1_ = 0 si arete XY
48// type1_ = 1 si arete XZ
49// type1_ = 2 si arete YZ
50// En dimension 2 il n'y a que des aretes XY
51void Aretes::affecter(int& numero_a, int dir, int type, int nb_face,
52 int f1, int f2, int f3, int f4, const ArrOfInt& est_une_plaque)
53{
54 int nb_plaques = 0;
55 nb_plaques += f1>=0 ? est_une_plaque[f1] : 0;
56 nb_plaques += f2>=0 ? est_une_plaque[f2] : 0;
57 nb_plaques += f3>=0 ? est_une_plaque[f3] : 0;
58 nb_plaques += f4>=0 ? est_une_plaque[f4] : 0;
59 int coin = -1 ;
60 int bord = 0 ;
61 if (type>0)
62 {
63 if (nb_plaques!=0) return;
64 if( (f1<nb_face) || (f2<nb_face) || (f3<nb_face) || (f4<nb_face) )
65 numero_a++;
66 else
67 return;
68 }
69 if (type==bord)
70 {
71 if( (f1<nb_face) || (f2<nb_face) || (f3<nb_face) )
72 numero_a++;
73 else
74 return;
75 }
76 if (type==coin)
77 {
78 if( (f1<nb_face) && (f2<nb_face) && (f3<nb_face) && (f4<nb_face) )
79 numero_a++;
80 else
81 return;
82 }
83
84 faces_(numero_a, 0)=f1;
85 faces_(numero_a, 1)=f2;
86 faces_(numero_a, 2)=f3;
87 faces_(numero_a, 3)=f4;
88 type1_(numero_a)=dir;
89 type2_(numero_a)=type;
90
91 if(type >0)
92 {
93 assert(faces_(numero_a, 0) !=-1);
94 assert(faces_(numero_a, 1) !=-1);
95 assert(faces_(numero_a, 2) !=-1);
96 assert(faces_(numero_a, 3) !=-1);
97 }
98}

100void Aretes::calculer_centre_de_gravite(Zone_VDF& zone)
101{
102 const IntTab& so = zone.face_sommets();
103 const DoubleTab& co = zone.zone().domaine().les_sommets();
104 DoubleTab& xa_ = zone.xa();
105 int i,j,k;
106 int f0=-1,f1=-1,s00,s01,s10,s11;
107 int type;
108 int nb_aretes = faces_.dimension(0);
109// const IntVect& orient = zone.orientation();
110// Calcul des ccordonnees de l'arete
111 if(dimension==2)
112 {
113 xa_.resize(nb_aretes,2);
114 for(i=0; i<nb_aretes; i++)
115 {
116 type = type2_(i);
117 if((type==2)||(type==1)) //!< arete interne ou mixte
118 {
119 f0 = faces_(i,0);
120 s00 = so(f0,0);
121 s01 = so(f0,1);
122 f1 = faces_(i,1);
123 s10 = so(f1,0);
124 s11 = so(f1,1);
125 if((s00==s10)||(s00==s11))
126 {
127 xa_(i,0) = co(s00,0);
128 xa_(i,1) = co(s00,1);
129 }
130 else if((s01==s10)||(s01==s11))
131 {
132 xa_(i,0) = co(s01,0);
133 xa_(i,1) = co(s01,1);
134 }
135 else
136 {
137 Cerr<<"Erreur on a pas trouve de sommets communs"<<finl;
138 exit();
139 }
140 }
141 else if((type == 0)||(type == -1)) //!< arete bord ou coin
142 {
143 for(j=0; j<4; j++)
144 {
145 f0=faces_(i,j);
146 if(f0!=-1) break;
147 }
148 assert(j<4);
149 k=j+1;
150 s00 = so(f0,0);
151 s01 = so(f0,1);
152 for(j=k; j<4; j++)
153 {
154 f1=faces_(i,j);
155 if(f1!=-1) break;
156 }
157 assert(j<4);
158 s10 = so(f1,0);
159 s11 = so(f1,1);
160 if((s00==s10)||(s00==s11))
161 {
162 xa_(i,0) = co(s00,0);
163 xa_(i,1) = co(s00,1);
164 }
165 else if((s01==s10)||(s01==s11))
166 {
167 xa_(i,0) = co(s01,0);
168 xa_(i,1) = co(s01,1);
169 }
170 else
171 {
172 Cerr<<"Erreur on a pas trouve de sommets communs"<<finl;
173 exit();
174 }
175 }
176 }
177 }
178 else if(dimension==3)
179 {
180 xa_.resize(nb_aretes,3);
181 for(i=0; i<nb_aretes; i++)
182 {
183 type = type2_(i);
184 if((type==2)||(type==1)) //!< arete interne ou mixte
185 {
186 f0 = faces_(i,0);
187 f1 = faces_(i,1);
188 int s0,s1,s0j;
189 int deux;
190 for(j=0; j<4; j++)
191 {
192 s0j=so(f0,j);
193 for(k=0; k<4; k++)
194 if(s0j==so(f1,k)) break;
195 if(k<4) break;
196 }
197 assert(j<4);
198 s0 = s0j;
199 deux = j+1;
200 for(j=deux; j<4; j++)
201 {
202 s0j=so(f0,j);
203 for(k=0; k<4; k++)
204 if(s0j==so(f1,k)) break;
205 if(k<4) break;
206 }
207 assert(j<4);
208 s1 = s0j;
209 xa_(i,0) = (co(s0,0)+co(s1,0))/2.0;
210 xa_(i,1) = (co(s0,1)+co(s1,1))/2.0;
211 xa_(i,2) = (co(s0,2)+co(s1,2))/2.0;
212 }
213 else if((type == 0)||(type == -1)) //!< arete bord ou coin
214 {
215 int f,fdeux;
216 int s0,s1,s0j;
217 int deux;
218 for(f=0; f<4; f++)
219 {
220 f0=faces_(i,f);
221 if(f0!=-1) break;
222 }
223 assert(f<4);
224 fdeux=f+1;
225 for(f=fdeux; f<4; f++)
226 {
227 f1=faces_(i,f);
228 if(f1!=-1) break;
229 }
230 for(j=0; j<4; j++)
231 {
232 s0j=so(f0,j);
233 for(k=0; k<4; k++)
234 if(s0j==so(f1,k)) break;
235 if(k<4) break;
236 }
237 assert(j<4);
238 s0 = s0j;
239 deux = j+1;
240 for(j=deux; j<4; j++)
241 {
242 s0j=so(f0,j);
243 for(k=0; k<4; k++)
244 if(s0j==so(f1,k)) break;
245 if(k<4) break;
246 }
247 assert(j<4);
248 s1 = s0j;
249 xa_(i,0) = (co(s0,0)+co(s1,0))/2.0;
250 xa_(i,1) = (co(s0,1)+co(s1,1))/2.0;
251 xa_(i,2) = (co(s0,2)+co(s1,2))/2.0;
252 }
253 }

270{
271 int tmp;
272 tmp = faces_(a1, 0);

274 faces_(a2, 0)=tmp;
275 tmp = faces_(a1, 1);
276 faces_(a1, 1) = faces_(a2, 1);
277 faces_(a2, 1)=tmp;
278 tmp = faces_(a1, 2);
279 faces_(a1, 2) = faces_(a2, 2);
280 faces_(a2, 2)=tmp;
281 tmp = faces_(a1, 3);
282 faces_(a1, 3) = faces_(a2, 3);
283 faces_(a2, 3)=tmp;
284 tmp = type1_(a1);
285 type1_(a1)=type1_(a2);
286 type1_(a2)=tmp;
287 tmp = type2_(a1);
288 type2_(a1)=type2_(a2);
289 type2_(a2)=tmp;
290}
291// Description :
292// reoordonne le tableaux des aretes
293// avec d'abord les aretes coins (elles n'ont que deux faces)

297void Aretes::trier(int& nb_aretes_coin, int& nb_aretes_bord,
298 int& nb_aretes_mixte, int& nb_aretes_interne)
299{
300//

303 int bord = 0 ;
304 int mixte = 1 ;
305 int interne = 2 ;
306 int nb_aretes = type1_.size();
307 int courante=0;
308 int arete;
309 while( (courante<nb_aretes)&&(type2_(courante)==coin) )
310 {
311 courante++;
312 nb_aretes_coin++;
313 }
314 for(arete=courante; arete<nb_aretes; arete++)
315 {
316 if(type2_(arete)==coin)
317 {
318 swap(arete, courante);
319 while( (courante<nb_aretes)&&(type2_(courante)==coin) )
320 {
321 courante++;
322 nb_aretes_coin++;
323 }
324// arete=courante;

326 }
327 while( (courante<nb_aretes)&&(type2_(courante)==bord) )
328 {
329 courante++;
330 nb_aretes_bord++;
331 }
332 for(arete=courante; arete<nb_aretes; arete++)
333 {
334 if(type2_(arete)==bord)
335 {
336 swap(arete, courante);
337 assert(type2_(courante) == bord);

392void Aretes::trier_pour_debog(int& nb_aretes_coin, int& nb_aretes_bord,
393 int& nb_aretes_mixte, int& nb_aretes_interne,const DoubleTab&
394 xv)
395{
396
397 ArrOfDouble XVref(dimension),XVref2(dimension);
398 int nb_aretes = type1_.size();
399 int arete;
400 for (int boucle=0; boucle<3; boucle++)
401 {
402 int deb=nb_aretes - nb_aretes_interne;
403 int fin=nb_aretes;
404 if (boucle==1)
405 {
406 fin=deb;
407 deb-=nb_aretes_mixte;
408 }
409 else if (boucle==2)
410 {
411 fin=deb;
412 deb-= nb_aretes_bord;
413 }
414 for ( arete=deb; arete<fin; arete++)
415 {
416 if ((boucle==2)&&(type2_(arete)!=0))
417 {
418 Cerr<<"gros pb "<<arete<<finl;
419 exit();
420 }
421 int ref=faces_(arete,0);
422 for (int i=0; i<dimension; i++) XVref[i]=xv(ref,i);
423 int marq=-1;
424 int ref2;
425 for (int arete2=arete; arete2<fin; arete2++)
426 {
427 ref2=faces_(arete2,0);
428 for (int i=0; i<dimension; i++) XVref2[i]=xv(ref2,i);
429 int test=-1;
430// int testsa=-1;
431 if (dimension==2)
432 {
433 if (sup_strict(XVref2[1],XVref[1])) test=1;
434 else if ((XVref2[1]==XVref[1])&&(sup_strict(XVref2[0],XVref[0]))) test=0;
435 }
436 else if (dimension==3)
437 {
438 if (XVref2[2]>XVref[2]) test=2;
439 else if (XVref2[2]==XVref[2])
440 {
441 if (XVref2[1]>XVref[1]) test=1;
442 else if ((XVref2[1]==XVref[1])&&(XVref2[0]>XVref[0])) test=0;
443 }
444 }
445 if (test!=-1)
446 {
447 marq=arete2;
448 XVref=XVref2;
449 }
450 }
451 if (marq!=-1) swap(arete,marq);
452 }
453 }
454 for (arete=nb_aretes - nb_aretes_interne; arete<nb_aretes*0; arete++)
455 {
456 Cerr <<me()<<" arete "<<arete<<" "<< faces_(arete,0)<<" "<< faces_(arete,1)<<" "<< faces_(arete,2)<<" "<< faces_(arete,3)<<finl;
457 }
