29// XD extrudebord interprete extrudebord 1 Class to generate an extruded mesh from a boundary of a tetrahedral or an hexahedral mesh. NL2 Warning: If the initial domain is a tetrahedral mesh, the boundary will be moved in the XY plane then extrusion will be applied (you should maybe use the Transformer keyword on the final domain to have the domain you really want). You can use the keyword Ecrire_Fichier_Meshtv to generate a meshtv file to visualize your initial and final meshes. NL2 This keyword can be used for example to create a periodic box extracted from a boundary of a tetrahedral or a hexaedral mesh. This periodic box may be used then to engender turbulent inlet flow condition for the main domain.NL2 Note that ExtrudeBord in VEF generates 3 or 14 tetrahedra from extruded prisms.

181 vect_dir(0)*Kx+vect_dir(1)*Ky+vect_dir(2)*Kz);

199// Pour que l'on ait plus de problemes de precision ensuite quand on manipule
200// le domaine extrude, on impose les coordonnes de la frontiere nom_front du domaine
201// initial sur les coordonnees de la frontiere devant du domaine extrude
202// Il faut egalement recalculer proprement pour la frontiere derriere
203 const Faces& faces=zone.frontiere(zone.rang_frontiere(nom_front)).faces();
204 int nb_faces=faces.nb_faces();

207 const Zone& zone2=dom_surfacique.zone(0);
208 const Faces& faces2=zone2.frontiere(zone2.rang_frontiere("devant")).faces();
209 const Faces& faces3=zone2.frontiere(zone2.rang_frontiere("derriere")).faces();
210 for (int j=0; j<nb_faces; j++)
211 for (int k=0; k<nb_som_faces; k++)
212 {
213 int som = faces.sommet(j,k);
214 int som2 = faces2.sommet(j,k);
215 int som3 = faces3.sommet(j,k);
216 for (int i=0; i<3; i++)
217 {
218 dom_surfacique.coord(som2,i) = dom.coord(som,i);
219 dom_surfacique.coord(som3,i) = dom.coord(som,i) + vect_dir(i);
220 }
221 }
222 Cerr << "ExtrudeBord does not create any more PERIO boundary." << finl;
223 Cerr << "Use RegroupeBord if you want to create periodic boundary condition..." << finl;

226void ExtrudeBord::extruder_hexa_old(Nom& nom_front, Nom& nom_dom_surfacique, DoubleVect& vect_dir, int nbpas)
227{
228
229 const Domaine& dom=domaine();
230
231 const Zone& zone=dom.zone(0);
232 int nbfr=zone.nb_front_Cl();
233
234
235 Domaine& dom_surfacique=ref_cast(Domaine, objet(nom_dom_surfacique));
236
237 Zone& zone2=dom_surfacique.add(*(new Zone()));
238 zone2.associer_domaine(dom_surfacique);
239 zone2.typer("Hexaedre");
240 zone2.nommer(nom_dom_surfacique);
241 DoubleTab& sommets2=dom_surfacique.les_sommets();
242 IntTab& les_elems2=zone2.les_elems();
243 int i;
244
245 for (int l=0; l<nbfr; l++)
246 {
247 const Frontiere& fr=zone.frontiere(l);
248 const Nom& nomfr=fr.le_nom();
249
250 if(nomfr==nom_front)
251 {
252 int nbfaces=fr.nb_faces();
253 const IntTab& som=fr.les_sommets_des_faces();
254 IntTab newsom(4*nbfaces);
255 IntTab compt(4*nbfaces);
256 newsom=-100;
257 compt=0;
258
259 int nbsombord,trouve;
260 nbsombord = 0;
261
262// ******** Table de correspondance des sommets de bords : dom_volumique - dom_surfacique
263
264 for (i=0; i<nbfaces; i++)
265 for (int j=0; j<4; j++)
266 {
267 trouve = 0;
268 int sommet=som(i,j);
269
270 for (int k=0; k<nbsombord+1; k++)
271 {
272 if(sommet==newsom(k))
273 {
274 k=nbsombord+1;
275 trouve = 1;
276 break;
277 }
278 }
279
280 if (trouve==0)
281 {
282 newsom[nbsombord]=sommet;
283 nbsombord++;
284 }
285 }
286
287 newsom.resize(nbsombord);
288 compt.resize(nbsombord);
289 sommets2.resize((nbpas+1)*nbsombord,3);
290
291// ******** Creation des sommets deduits par translation
292
293 for (int j=0; j<nbpas+1; j++)
294 for (i=0; i<nbsombord; i++)
295 {
296 sommets2(i+j*nbsombord,0)=dom.coord(newsom(i),0)+j*vect_dir(0)/nbpas;
297 sommets2(i+j*nbsombord,1)=dom.coord(newsom(i),1)+j*vect_dir(1)/nbpas;
298 sommets2(i+j*nbsombord,2)=dom.coord(newsom(i),2)+j*vect_dir(2)/nbpas;
299 }
300
301
302// ******** Definition des tableaux de correspondance ELEM<->SOMMETS
303
304 les_elems2.resize(nbpas*nbfaces,8);
305
306 int sommet0,sommet1,sommet2,sommet3;
307 int newsom0=-1,newsom1=-1,newsom2=-1,newsom3=-1;
308
309 for (i=0; i<nbfaces; i++)
310 {
311 sommet0=som(i,0);
312 sommet1=som(i,1);
313 sommet2=som(i,2);
314 sommet3=som(i,3);
315
316
317 for(int j=0; j<nbsombord; j++)
318 {
319 if(newsom[j]==sommet0) newsom0=j;
320 if(newsom[j]==sommet1) newsom1=j;
321 if(newsom[j]==sommet2) newsom2=j;
322 if(newsom[j]==sommet3) newsom3=j;
323 }
324
325 for(int k=0; k<nbpas; k++)
326 {
327 les_elems2(i+k*nbfaces,0)=newsom0+k*nbsombord;
328 les_elems2(i+k*nbfaces,1)=newsom1+k*nbsombord;
329 les_elems2(i+k*nbfaces,2)=newsom2+k*nbsombord;
330 les_elems2(i+k*nbfaces,3)=newsom3+k*nbsombord;
331 les_elems2(i+k*nbfaces,4)=newsom0+(k+1)*nbsombord;
332 les_elems2(i+k*nbfaces,5)=newsom1+(k+1)*nbsombord;
333 les_elems2(i+k*nbfaces,6)=newsom2+(k+1)*nbsombord;
334 les_elems2(i+k*nbfaces,7)=newsom3+(k+1)*nbsombord;
335 }
336 }
337
338
339
340// ******** Definition des frontieres periodiques
341
342
343 Bord& bordperio=zone2.faces_bord().add(Bord());
344 bordperio.nommer("PERIO");
345 bordperio.typer_faces("QUADRANGLE_3D");
346 bordperio.dimensionner(0);
347 IntTab faces_perio(2*nbfaces,4);
348
349 for (i=0; i<nbfaces; i++)
350 {
351 sommet0=som(i,0);
352 sommet1=som(i,1);
353 sommet2=som(i,2);
354 sommet3=som(i,3);
355
356
357 for(int j=0; j<nbsombord; j++)
358 {
359 if(newsom[j]==sommet0) newsom0=j;
360 if(newsom[j]==sommet1) newsom1=j;
361 if(newsom[j]==sommet2) newsom2=j;
362 if(newsom[j]==sommet3) newsom3=j;
363 }
364
365 faces_perio(i,0)=newsom0;
366 faces_perio(i,1)=newsom1;
367 faces_perio(i,2)=newsom2;
368 faces_perio(i,3)=newsom3;
369
370 faces_perio(i+nbfaces,0)=newsom0+nbpas*nbsombord;
371 faces_perio(i+nbfaces,1)=newsom1+nbpas*nbsombord;
372 faces_perio(i+nbfaces,2)=newsom2+nbpas*nbsombord;
373 faces_perio(i+nbfaces,3)=newsom3+nbpas*nbsombord;
374 }
375
376
377 bordperio.ajouter_faces(faces_perio);
378
379
380
381
382// ******** Definition des frontieres parois
383
384
385 Bord& bordparoi=zone2.faces_bord().add(Bord());
386 bordparoi.nommer("PAROI");
387 bordparoi.typer_faces("QUADRANGLE_3D");
388 bordparoi.dimensionner(0);
389
390
391 int nbsombordparoi=0;
392
393// ***** recherche des sommets appartenant au contour de la face de bord *****
394
395 for (i=0; i<nbfaces; i++)
396 for (int j=0; j<4; j++)
397 for (int k=0; k<nbsombord; k++)
398 if(som(i,j)==newsom(k)) compt(k)++; //!< nb de faces se rapportant a chacun des sommets
399
400 for (int k=0; k<nbsombord; k++)
401 if(compt(k)<3) nbsombordparoi++; //!< nb de sommets a extruder pour constituer les faces de parois
402
403// PQ : 05/05/04 : ATTENTION : des sommets internes peuvant etre connectes qu'a
404// 3 faces (cf. maillage en "O" d'un cylindre)
405// -> on se place donc dans le cas ou les sommets de bords ne sont connectes qu'a
406// 2 elements (excluant de fait les bords types VDF en marche d'escalier).
407
408
409// **** Extension des aretes pour former les faces de la frontiere paroi
410
411 IntTab faces_paroi(nbsombordparoi*nbpas,4);
412 int ip=0;
413
414 for (i=0; i<nbfaces; i++)
415 {
416 sommet0=som(i,0);
417 sommet1=som(i,1);
418 sommet2=som(i,2);
419 sommet3=som(i,3);
420
421 int compt0=4,compt1=4,compt2=4,compt3=4;
422
423 for(int j=0; j<nbsombord; j++)
424 {
425 if(newsom[j]==sommet0)
426 {
427 newsom0=j;
428 compt0=compt(j);
429 }
430 if(newsom[j]==sommet1)
431 {
432 newsom1=j;
433 compt1=compt(j);
434 }
435 if(newsom[j]==sommet2)
436 {
437 newsom2=j;
438 compt2=compt(j);
439 }
440 if(newsom[j]==sommet3)
441 {
442 newsom3=j;
443 compt3=compt(j);
444 }
445 }
446
447
448 if( (compt0<3) && (compt1<3) ) //!< arete[0-1]
449 {
450 for(int k=0; k<nbpas; k++)
451 {
452 faces_paroi(ip+k,0)=newsom0+k*nbsombord;
453 faces_paroi(ip+k,1)=newsom0+(k+1)*nbsombord;
454 faces_paroi(ip+k,2)=newsom1+k*nbsombord;
455 faces_paroi(ip+k,3)=newsom1+(k+1)*nbsombord;

457 ip+=nbpas;
458 }
459

463 for(int k=0; k<nbpas; k++)
464 {
465 faces_paroi(ip+k,0)=newsom1+k*nbsombord;
466 faces_paroi(ip+k,1)=newsom1+(k+1)*nbsombord;
467 faces_paroi(ip+k,2)=newsom3+k*nbsombord;
468 faces_paroi(ip+k,3)=newsom3+(k+1)*nbsombord;
469 }
470 ip+=nbpas;
471 }
472
473
474 if( (compt3<3) && (compt2<3) ) //!< arete[3-2]
475 {
476 for(int k=0; k<nbpas; k++)
477 {
478 faces_paroi(ip+k,0)=newsom3+k*nbsombord;
479 faces_paroi(ip+k,1)=newsom3+(k+1)*nbsombord;
480 faces_paroi(ip+k,2)=newsom2+k*nbsombord;
481 faces_paroi(ip+k,3)=newsom2+(k+1)*nbsombord;

484 }

490 {

500 bordparoi.ajouter_faces(faces_paroi);
501
502 }//fin de la condition sur la frontiere a extruder
503 }//fin de parcours de toutes les frontieres
