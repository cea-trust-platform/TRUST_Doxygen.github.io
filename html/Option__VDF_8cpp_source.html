110int Option_VDF::lire_motcle_non_standard(const Motcle& mot, Entree& is)
111{
112 Motcle motlu;
113 if (mot=="traitement_coins")
114 {
115 is >>motlu;
116 if (test_mot(motlu))
117 {
118 Cerr<< motlu <<" will affect the action of the method Zone_Cl_VDF::completer"<<finl;
119 Option_VDF::traitement_coins=1;
120 }
121 else
122 Option_VDF::traitement_coins=0;
123 return 1;
124 }
125 else if (mot=="P_imposee_aux_faces")
126 {
127 is >>motlu;
128 if (test_mot(motlu))
129 {
130 Cerr<<motlu<<" will affect the action of the operator gradient for Neumann_sortie_libre boundary condition" <<finl;
131 Cerr<<motlu<<" and therefore the assembleur action will also be affected."<<finl;
132 Option_VDF::coeff_P_neumann=2.;
133 }
134 else
135 Option_VDF::coeff_P_neumann=1.;
136 return 1;
137 }
138 else
139 {
140 Cerr << mot << "is not a keyword understood by " << que_suis_je() << "in methode lire_motcle_non_standard"<< finl;
141 exit();
142 }
143 return -1;
144}
