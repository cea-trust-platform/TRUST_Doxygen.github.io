43void test_un_solveur(SolveurSys& solveur, const Matrice_Base& matrice, const DoubleVect& secmem, DoubleVect& solution, int nmax, ArrOfDouble& temps, double seuil_verification=DMAXFLOAT)
44{
45 DoubleVect solution_ref(solution);
46 int n=temps.size_array();
47 Stat_Counter_Id solv_sys_counter_l= statistiques().new_counter(1, "SolveurSys::resoudre_systeme", 0);
48
49 for (int i=0; i<n; i++)
50 {
51 solution=solution_ref;
52
53// etape de resolution
54 Stat_Results stat_resol_0;
55
56 statistiques().get_stats(solv_sys_counter_l, stat_resol_0);
57 Cout<<"------------------------------------"<<finl;
58 Cout<<"Try " << i << " of solver " << solveur <<finl;
59// solveur.valeur().fixer_limpr(0);
60 statistiques().begin_count(solv_sys_counter_l);
61 solveur.nommer("test_solver");
62
63 solveur.resoudre_systeme(matrice,secmem,solution);
64 statistiques().end_count(solv_sys_counter_l);
65
66 Stat_Results stat_resol;
67 statistiques().get_stats(solv_sys_counter_l, stat_resol);
68// on recupere un delta time et non un time absolu !!
69 double time_resol=stat_resol.time-stat_resol_0.time;
70 DoubleVect test(secmem);
71 test*=-1;
72 matrice.ajouter_multvect(solution,test);
73// test-=secmem;
74 double norme=mp_norme_vect(test);
75// statistiques().dump("Statistiques de resolution du probleme");
76// print_statistics_analyse("Statistiques de resolution du probleme", 1);
77
78 Cout<<"CPU= " <<time_resol<<" s , ||Ax -b||= " << norme << finl;
79 Process::imprimer_ram_totale();
80 temps[i]=Process::mp_max(time_resol);
81
82 if (norme>seuil_verification)
83 {
84 Cerr<<"residue calculated greater than the threshold value indicated ("<<seuil_verification<<")"<<finl;
85 Cerr<<" one will not use the solver "<<solveur<<finl;
86 temps=1.e37;
87 }
88 }

90int test_solveur(SolveurSys& solveur, const Matrice_Base& matrice , const DoubleVect& secmem , DoubleVect& solution , int nmax, ArrOfDouble& temps, Entree& list_solveur, double seuil_verification=DMAXFLOAT)
91{
92 DoubleVect solution_ref(solution);
93 int numero=0,numero_best=-1;
94 double best_time=1e36;
95 Motcle motsolveur("solveur"),motlu;
96 list_solveur >> motlu;
97 int dernier=temps.size_array()-1;
98 while (list_solveur.good())
99 {
100 if (motlu!=motsolveur)
101 {
102 Cerr<<" One expected "<<motsolveur <<" and not "<<motlu<<finl;
103 Process::exit();
104 }
105 numero++;
106 SolveurSys newsolveur;
107 list_solveur>>newsolveur;
108 solution=solution_ref;
109 test_un_solveur(newsolveur, matrice, secmem ,solution, nmax, temps, seuil_verification);
110 if (temps[dernier]<best_time)
111 {
112 solveur=newsolveur;
113 best_time=temps[dernier];
114 numero_best=numero;
115 }
116 list_solveur>>motlu;
117 }
118 if (numero_best==-1)
119 {
120 Cerr<<" None of the tested solvers give the expected residue" <<finl;
121 Cerr<< " does the same solver must be kept ? "<<finl;
122// exit();
123 }
124 temps=best_time;
125 return numero_best;
126}

128void generate_defaut(const Matrice_Base& matrice, const double seuil, Sortie& sortie, int limpr=0)
129{
130 Nom impr(" impr " );
131 if (limpr==0) impr=" ";
132 if (limpr==-1) impr=" quiet ";
133 if (Process::je_suis_maitre())
134 {
135 if((!sub_type(Matrice_Morse_Sym,matrice))&&(!sub_type(Matrice_Bloc,matrice)))
136 {
137 sortie <<" solveur gen { seuil "<<seuil <<" "<<impr<< " solv_elem bicgstab precond ilu { type 2 filling 10 } }" <<finl;
138 sortie <<" solveur gen { seuil "<<seuil <<" "<<impr<< " solv_elem bicgstab precond ilu { type 2 filling 20 } }" <<finl;
139 sortie <<" solveur gmres { diag seuil "<<seuil <<" "<<impr<<"}"<<finl;
140#ifdef __PETSCKSP_H
141 sortie <<" solveur petsc bicgstab { precond diag { } seuil "<<seuil <<" "<<impr<<"}"<<finl;
142#endif
143 }
144 else
145 {
146// on a une matrice de type pression (?)
147// Mise a jour des solveurs testes 24/05/2012
148 sortie <<" solveur gcp { precond ssor { omega 1.6 } seuil "<<seuil <<" "<<impr<<"}"<<finl;
149#ifdef __PETSCKSP_H
150 sortie <<" solveur petsc gcp { precond ssor { omega 1.6 } seuil "<<seuil <<" "<<impr<<"}"<<finl;
151 if (Process::nproc()<512)
152 sortie <<" solveur petsc cholesky { impr }"<< finl;
153 else
154 {
155// Pour les tres grands calculs on bascule de Cholesky a BICGSTAB ILU(1) par bloc
156 sortie <<" solveur petsc bicgstab { precond block_jacobi_icc { level 1 } seuil "<<seuil <<" "<<impr<<"}"<<finl;
157// Voire CG ILU(1) par bloc car BICGSTAB peut avoir du mal a converger lors de la projection initiale...
158 sortie <<" solveur petsc gcp { precond block_jacobi_icc { level 1 } seuil "<<seuil <<" "<<impr<<"}"<<finl;
159 }
160// SPAI n'a jamais fait ses preuves (comme tous les preconditionnements de Hypre), on l'enleve
161// sortie <<" solveur petsc gcp { precond spai { level 2 epsilon 0.2 } seuil "<<seuil <<" "<<impr<<"}"<<finl;
162#endif
163 }
164 }

166Entree& Test_solveur::interpreter(Entree& is)
167{
168 Matrice matrice;
169 DoubleVect secmem,solution;
170 Nom fichier_secmem("Secmem.sa");
171 Nom fichier_solution("Solution.sa");
172 Nom fichier_matrice("Matrice.sa");
173 Nom fichier_solveur;
174 int pas_de_solution_init=0,ascii;
175 double seuil_verification=DMAXFLOAT;
176 SolveurSys solveur;
177 double seuil_list=0;
178 int limpr_;
179 int nb_test=2; //!< On teste 2 fois un solveur car la premiere fois, le cout du preconditionnement peut penaliser
180 Param param((*this).que_suis_je());
181 param.ajouter("fichier_secmem",&fichier_secmem); //!< nom du fichier contenant le second membre (Secmem.sa par defaut)
182 param.ajouter("fichier_matrice",&fichier_matrice); //!< nom du fichier contenant la matrice (Matrice.sa par defaut)
183 param.ajouter("fichier_solution",&fichier_solution); //!< nom du fichier contenant la solution (Solution.sa par defaut)
184 param.ajouter("nb_test",&nb_test); //!< nb de resolution pour mesurer le temps (un seul preconditionnemt)
185 param.ajouter_flag("impr",&limpr_); //!< impr des solveurs
186 param.ajouter("solveur",&solveur); //!< pour specifier un solveur
187 param.ajouter("fichier_solveur",&fichier_solveur); //!< pour specifier un fichier contenant des solveurs
188 param.ajouter("genere_fichier_solveur",&seuil_list); //!< genere le fichier de solveur avec un seuil donne
189 param.ajouter("seuil_verification",&seuil_verification); //!< verifie si la soulution trouvee par la resolution est telle que Ax -b < seuil_verification
190 param.ajouter_flag("pas_de_solution_initiale",&pas_de_solution_init); //!< pas_de_solution_initiale : on n'initialise pas la resolution avec la solution
191 param.ajouter_flag("ascii",&ascii); //!< dans le cas ou les fichiers sont ascii
192 param.lire_avec_accolades_depuis(is);
193 int binaire=1;
194 if (ascii)
195 binaire=0;
196// On relit la matrice et le secmem
197 {
198 LecFicDistribue entree;
199 entree.set_bin(binaire);
200 entree.ouvrir(fichier_matrice);
201 entree>>matrice;
202 Cout<<" size of system "<<matrice.valeur( ).nb_colonnes()<<finl;
203// matrice.valeur().imprimer_formatte(Cout);
204 }
205 {
206 LecFicDistribue entree;
207 entree.set_bin(binaire);
208 entree.ouvrir(fichier_secmem);
209 MD_Vector_tools::restore_vector_with_md(secmem,entree);
210 Cout<<" size of system "<<secmem.size_totale()<<finl;
211 }
212
213
214 if (pas_de_solution_init)
215 {
216 solution=secmem;
217 solution=0.;
218 }
219 else
220 {
221 LecFicDistribue entree;
222 entree.set_bin(binaire);
223 entree.ouvrir(fichier_solution);
224
225 MD_Vector_tools::restore_vector_with_md(solution,entree);
226 Cout<<" size of system "<<solution.size_totale()<<finl;
227 solution.set_md_vector(secmem.get_md_vector());
228 }
229
230 if (seuil_list!=0)
231 {
232 if (fichier_solveur==Nom())
233 fichier_solveur="list_solveur";
234 if (je_suis_maitre())
235 {
236 SFichier list_solveur(fichier_solveur);
237 generate_defaut(matrice,seuil_list,list_solveur,limpr_);
238 }
239 }
240
241 secmem.echange_espace_virtuel();
242 solution.echange_espace_virtuel();
243// Champ_Inc bidon;
244// bidon.typer("Champ_P0_VDF");
245 ArrOfDouble temps(nb_test);
246 if (fichier_solveur==Nom())
247 test_un_solveur(solveur, matrice , secmem , solution , -10, temps,seuil_verification);
248 else
249 {
250 LecFicDiffuse list_solveur(fichier_solveur);
251 int numero_best=test_solveur(solveur, matrice , secmem , solution , -10, temps,list_solveur,seuil_verification);
252 Cout <<"------------------------------------------------"<<finl;
253 Cout <<"Best solver : number "<<numero_best<<" "<<solveur<<finl;
254 Cout <<"Best CPU time = "<<temps[0]<<finl;
255 Cout <<"------------------------------------------------"<<finl;
256 }
257 return is;
258}

300// a la premiere iteration
301// genere le fichier fichier_solveur_ contenant la liste des solveurs a tester

305void Solv_Optimal::prepare_resol(const Matrice_Base& matrice, const DoubleVect& secmem, DoubleVect& solution, int nmax)
306{
307 if (n_resol_==0)
308 {
309// premiere chose on genere le fichier par defaut, puis on cree le solveur avec le premier du fichier
310
311 if ((!fichier_solveur_non_recree_)&&(je_suis_maitre()))
312 {
313 SFichier list_solveur(fichier_solveur_);
314 generate_defaut(matrice,seuil_,list_solveur,limpr());
315 }
316 LecFicDiffuse list_solveur2(fichier_solveur_);
317 Nom mot;
318 list_solveur2>>mot;
319 list_solveur2>>le_solveur_;
320 }
321// OK
322 n_resol_++;
323 if ((n_resol_>=3)&&((n_resol_-3)%freq_recalc_==0))
324 {
325 LecFicDiffuse list_solveur(fichier_solveur_);
326 int nb_ite;
327 if (n_reinit_<2)
328 nb_ite=2; //!< Matrice constante (on resout 2 fois pour ne retenir que le 2eme temps exempt d'eventuel preconditionnement)
329 else
330 nb_ite=1; //!< Matrice non constante (on resout 1 seul fois)
331 ArrOfDouble temps(nb_ite);
332 int numero_best=test_solveur(le_solveur_, matrice , secmem , solution , nmax, temps,list_solveur,seuil_);
333 Cout <<"------------------------------------------------"<<finl;
334 Cout <<"Best solver : number "<<numero_best<<" "<<le_solveur_<<finl;
335 Cout <<"Best CPU time = "<<temps[0]<<finl;
336 Cout <<"------------------------------------------------"<<finl;
337 if (je_suis_maitre())
338 {
339 Nom best("best_");
340 best+=fichier_solveur_;
341 SFichier sf(best);
342 sf << le_solveur_ <<finl;
343 }
344 }
