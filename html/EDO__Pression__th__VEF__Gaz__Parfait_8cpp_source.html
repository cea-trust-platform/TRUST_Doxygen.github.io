53//! Resoud l'EDO
54/** \return double<br>
55Signification: La nouvelle valeur de la pression<br>
56Contraintes: */
57double EDO_Pression_th_VEF_Gaz_Parfait::resoudre(double Pth_n)
58{
59
60 int traitPth = le_fluide_->getTraitementPth();
61
62 if (traitPth==2) //!< Pth constant
63 return Pth_n;
64
65 double present = le_fluide_->vitesse()->equation().schema_temps().temps_courant();
66 double dt = le_fluide_->vitesse()->equation().schema_temps().pas_de_temps();
67 double futur = present+dt;
68 const DoubleTab& tempnp1 = le_fluide_->inco_chaleur()->valeurs(futur); //!< T(n+1)
69 const DoubleTab& tempn = le_fluide_->inco_chaleur()->valeurs(present); //!< T(n)
70 int nb_faces = la_zone->nb_faces();
71 double Pth=0;
72
73 const Zone_VEF& zone_vef = ref_cast(Zone_VEF, la_zone.valeur());
74
75 if (traitPth == 0) //!< EDO
76 {
77
78 const DoubleTab& tab_rho = le_fluide_->masse_volumique().valeurs(); //!< n+1/2
79
80 const double rho_moy = Champ_P1NC::calculer_integrale_volumique(zone_vef, tab_rho, FAUX_EN_PERIO);
81 DoubleVect tmp(tab_rho); //!< copie de rho
82 assert(tmp.size() == nb_faces);
83 const double invdt = 1. / dt;
84 for (int i = 0; i < nb_faces; i++)
85 tmp[i] *= (tempnp1[i] - tempn[i]) / tempnp1[i] * invdt;
86 double S = Champ_P1NC::calculer_integrale_volumique(zone_vef, tmp, FAUX_EN_PERIO);
87
88//
89 S /= rho_moy;
90// double Pth = (Pth_n + S*dt/V);
91// double Pth = Pth_n/(1- S*dt);
92 Pth = Pth_n/(1- S*dt);
93// test pour faire apparaitre la difference sur Pth
94// Pth *= 1e10;
95 }
96
97 else if (traitPth == 1) //!< Conservation masse (WEC mars 2008)
98 {
99
100// On veut Masse(n+1) - Masse(n) = dt * Debits_aux_bords
101// or Masse = Pth / R * sum(Vi/Ti) en volume
102// et Debit = Pth / R * sum(Sj.Uj/Tj) aux faces Dirichlet
103
104// Donc si on note masse_n = sum(Vi/Ti(n))
105// debit_u_imp = sum(Sj.Uj(impose)/Tj(n+1)) pour les vitesses imposees
106// et debit_rho_u_imp = sum(Sj.Uj(impose)/Tj(n+1)) pour les rho_u imposes
107// ou U(impose) = (rho.U)(impose) / rho(Pth(n),T(n+1))
108// il faut Pth(n+1)*masse_np1 - Pth(n)*masse_n = Pth(n+1) * debit_u_imp * dt
109// + Pth(n) * debit_rho_u_imp * dt
110
111// Attention il faudrait prendre en compte les porosites dans les integrales !!!
112
113 double debit_u_imp=0,debit_rho_u_imp=0;
114
115// On veut que rho_np1 soit calcule avec T(n+1) et Pth_n pour les CLs en rho_u impose
116 le_fluide_->calculer_masse_volumique();
117
118// Calcul de masse_n et masse_np1
119 DoubleVect tmp;
120 tmp.copy(tempn, ArrOfDouble::NOCOPY_NOINIT); //!< copier uniquement la structure
121 for (int i = 0; i < nb_faces; i++)
122 tmp[i] = 1. / tempn[i];
123 const double masse_n = Champ_P1NC::calculer_integrale_volumique(zone_vef, tmp, FAUX_EN_PERIO);
124 for (int i = 0; i < nb_faces; i++)
125 tmp[i] = 1. / tempnp1[i];
126 const double masse_np1 = Champ_P1NC::calculer_integrale_volumique(zone_vef, tmp, FAUX_EN_PERIO);
127
128// Calcul de debit_u_imp et debit_rho_u_imp
129 for (int n_bord=0; n_bord<la_zone->nb_front_Cl(); n_bord++)
130 {
131 const Cond_lim_base& la_cl = la_zone_Cl->les_conditions_limites(n_bord).valeur();
132 if (sub_type(Dirichlet, la_cl))
133 {
134 const Dirichlet& diri=ref_cast(Dirichlet,la_cl);
135 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
136 int ndeb = la_front_dis.num_premiere_face();
137 int nfin = ndeb + la_front_dis.nb_faces();
138 for (int face=ndeb; face<nfin; face++)
139 {
140 double debit_v=0;
141 for (int d=0; d<dimension; d++)
142 debit_v += la_zone->face_normales(face,d)*diri.val_imp_au_temps(futur,face-ndeb,d)/tempnp1(face);
143 int n0 = la_zone->face_voisins(face,0);
144 if (n0 == -1)
145 debit_v=-debit_v;
146 if (sub_type(Frontiere_ouverte_rho_u_impose, la_cl))
147 debit_rho_u_imp+=debit_v;
148 else
149 debit_u_imp+=debit_v;
150 }
151 }
152 else if (sub_type(Neumann_sortie_libre, la_cl))
153 {
154 Cerr<<la_cl.que_suis_je()<<" est incompatible avec le traitement conservation_masse."<<finl;
155 exit();
156 }
157 }
158// On fait la somme sur les procs
159 debit_u_imp=mp_sum(debit_u_imp);
160 debit_rho_u_imp=mp_sum(debit_rho_u_imp);
161
162// Calcul de Pth(n+1)
163 Pth = Pth_n * (masse_n - dt*debit_rho_u_imp) / (masse_np1 + dt*debit_u_imp);
164 }
165
166 return Pth;
167}
