67//! methode outil pour ajouter un item a un tableau du genre "blocs" contenant<br>des series de blocs... alonge le bloc precedent ou commence un nouveau bloc si l'item n'est pas contigu avec le bloc precedent.
68void MD_Vector_base2::append_item_to_blocs(ArrOfInt& blocs, int item)
69{
70 int n = blocs.size_array();
71 assert(n % 2 == 0); //!< le tableau contient des blocs, donc un nombre pair d'elements
72 assert(n == 0 || item >= blocs[n - 1]); //!< les items doivent etre ajoutes dans l'ordre croissant
73 if (n == 0 || blocs[n - 1] != item)
74 {
75// nouveau bloc
76 blocs.append_array(item);
77 blocs.append_array(item + 1);
78 }
79 else blocs[n - 1] = item + 1;
80}
