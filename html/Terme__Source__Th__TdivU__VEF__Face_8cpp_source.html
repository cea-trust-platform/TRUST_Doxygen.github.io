53{
54 int nb_eqn = pb.nombre_d_equations();
55 int ok_vit=0;

58 {
59 const Equation_base& eqn = pb.equation(i);
60 if (sub_type(Navier_Stokes_std,eqn))

74 exit();

76 if (!ok_temp)
77 {
78 Cerr << "Error TRUST in " << que_suis_je() << finl;
79 Cerr << "Thermal equation not found" << finl;
80 exit();
81 }

156DoubleTab& Terme_Source_Th_TdivU_VEF_Face::ajouter(DoubleTab& resu) const
157{
158 Terme_Source_Th_TdivU_VEF_Face& me_non_const=ref_cast_non_const(Terme_Source_Th_TdivU_VEF_Face,*this);
159 me_non_const.modifier_zone_cl();
160 if(resu.line_size() > 1)
161 {
162 Cerr << "Error in " << que_suis_je() << finl;
163 Cerr << "The source term Source_Th_TdivU_VEF_P1NC is only dedicated to thermal equation" << finl;
164 exit();
165 }
166 const DoubleTab& temperature=equation().inconnue().valeurs();
167 const int nb_faces = temperature.dimension_tot(0);
168
169// Calcul de TdivU d'apres le schema de convection
170 DoubleTab temp(temperature);
171 DoubleTab TdivU(resu);
172 TdivU=0.;
173 temp=1.;
174 const Operateur& Op_conv=equation().operateur(1);
175 Operateur_base& optype=ref_cast_non_const(Operateur_base,Op_conv.l_op_base());
176// We save the boundary fluxes:
177 DoubleTab flux_bords_backup = optype.flux_bords();
178 optype.associer_zone_cl_dis(ma_zonecl_);
179 Op_conv.ajouter(temp,TdivU);
180 for(int face=0; face<nb_faces; face++)
181 {
182 TdivU[face]*=temperature(face);
183 }
184 if (mp_max_vect(TdivU)==0)
185 {
186 Cerr << finl;
187 Cerr << "******* Warning *******" << finl;
188 Cerr << "You are using source term TdivU into one transport equation." << finl;
189 Cerr << "It seems useless cause TdivU=0" << finl;
190 Cerr << "May be the advection operator is calculated with the non-conservative formulation." << finl;
191 }
192
193// la partie en TdivU est un bout de l'operateur de convection -> il doit etre multiplie par rhoCp
194 const double rhoCp = equation().milieu().capacite_calorifique().valeurs()(0, 0) * equation().milieu().masse_volumique().valeurs()(0, 0);
195 TdivU *= rhoCp;
196 resu-=TdivU;
197// on remet la bonne zone_cl_dis
198 optype.associer_zone_cl_dis(zonecl_sa.valeur());
199// We reset flux_bords
200 optype.flux_bords() = flux_bords_backup;
201 return resu;
202}
