229bool Multi_Sch_ThHyd::iterateTimeStep(bool& converged)
230{
231 Probleme_base& prob=pb_base();
232 double temps=temps_courant_+dt_;
233 int nb_eqn=pb_base().nombre_d_equations();
234 for(int i=nb_eqn-1; i>-1; i--)
235 {
236 Equation_base& eqn_i=prob.equation(i);
237 if (eqn_i.equation_non_resolue())
238 {
239 Cout<< "====================================================" << finl;
240 Cout<< eqn_i.que_suis_je()<<" equation is not solved."<<finl;
241 Cout<< "====================================================" << finl;
242// On calcule une fois la derivee pour avoir les flux bord
243 if (eqn_i.schema_temps().nb_pas_dt()==0)
244 {
245 DoubleTab inconnue_valeurs(eqn_i.inconnue().valeurs());
246 eqn_i.derivee_en_temps_inco(inconnue_valeurs);
247 }
248 }
249 else
250 {
251 eqn_i.zone_Cl_dis().mettre_a_jour(temps);
252 faire_un_pas_de_temps_eqn_base(eqn_i);
253 }
254 }
255 converged=true;
256 return true;
257
258}
