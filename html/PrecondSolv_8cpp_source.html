46//! Calcule la solution du systeme lineaire: A * solution = b<br>avec la methode de relaxation PrecondSolv.
47int PrecondSolv::preconditionner_(const Matrice_Base& matrice,
48 const DoubleVect& b,
49 DoubleVect& solution)
50{
51 double norme=mp_norme_vect(b);
52 if (norme > 0.)
53 {
54 DoubleVect b2(b);
55// Certains solveurs peuvent avoir besoin de l'espace virtuel,
56// on calcule tout le vecteur:
57 assert_espace_virtuel_vect(b2);
58 operator_multiply(b2, 1. / norme, VECT_ALL_ITEMS);
59
60 solveur.resoudre_systeme(matrice, b2, solution);
61
62 if (echange_ev_solution_)
63 {
64// On veut renvoyer un vecteur avec espace virtuel a jour
65// Plutot que de refaire un echange on multiplie tous les elements
66// mais il faut que l'espace virtuel soit deja a jour avant:
67 assert_espace_virtuel_vect(solution);
68 operator_multiply(solution, norme, VECT_ALL_ITEMS);
69 }
70 else
71 {
72 operator_multiply(solution, norme);
73 }
74 }
75 else
76 {
77 if (echange_ev_solution_)
78 {
79 operator_egal(solution, 0., VECT_ALL_ITEMS);
80 }
81 else
82 {
83 operator_egal(solution, 0.);
84 }
85 }
86 return 1;
87}
