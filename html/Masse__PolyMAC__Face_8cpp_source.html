63DoubleTab& Masse_PolyMAC_Face::appliquer_impl(DoubleTab& sm) const
64{
65 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
66 const IntTab& f_e = zone.face_voisins();
67 const DoubleVect& pf = zone.porosite_face();
68 int i, e, f, n, N = equation().inconnue().valeurs().line_size();
69 const DoubleTab *a_r = sub_type(QDM_Multiphase, equation()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.champ_conserve().passe() : NULL,
70 &vfd = zone.volumes_entrelaces_dir();
71 double fac;
72
73// vitesses aux faces
74 for (f = 0; f < zone.nb_faces(); f++)
75 for (n = 0; n < N; n++)
76 {
77 for (fac = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++) fac += vfd(f, i) * (a_r ? (*a_r)(e, n) : 1);
78 sm(f, n) /= pf(f) * fac; //vitesse calculee
79 }
80
81 sm.echange_espace_virtuel();
82 return sm;
83}

85void Masse_PolyMAC_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
86{
87 const std::string& nom_inc = equation().inconnue().le_nom().getString();
88 if (!matrices.count(nom_inc)) return; //rien a faire
89 Matrice_Morse& mat = *matrices.at(nom_inc), mat2;
90 const Zone_PolyMAC& zone = la_zone_PolyMAC;
91 const DoubleTab& inco = equation().inconnue().valeurs();
92 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
93 const Masse_ajoutee_base *corr = pbm && pbm->has_correlation("masse_ajoutee") ? &ref_cast(Masse_ajoutee_base, pbm->get_correlation("masse_ajoutee").valeur()) : NULL;
94 int i, f, m, n, N = inco.line_size();
95
96 IntTrav sten(0, 2);
97 sten.set_smart_resize(1);
98 for (f = 0, i = 0; f < zone.nb_faces(); f++)
99 for (n = 0; n < N; n++, i++) //faces reelles
100 if (corr)
101 for (m = 0; m < N; m++) sten.append_line(i, N * f + m);
102 else sten.append_line(i, i);
103
104 Matrix_tools::allocate_morse_matrix(inco.size_totale(), inco.size_totale(), sten, mat2);
105 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
106}

108void Masse_PolyMAC_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, double dt, const tabs_t& semi_impl, int resoudre_en_increments) const
109{
110 const DoubleTab& inco = equation().inconnue().valeurs(), &passe = equation().inconnue().passe();
111 Matrice_Morse *mat = matrices[equation().inconnue().le_nom().getString()]; //facultatif
112 const Zone_PolyMAC& zone = la_zone_PolyMAC;
113 const IntTab& f_e = zone.face_voisins();
114 const DoubleVect& pf = zone.porosite_face(), &vf = zone.volumes_entrelaces();
115 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
116 const DoubleTab& vfd = zone.volumes_entrelaces_dir(), &rho = equation().milieu().masse_volumique().passe(),
117 *alpha = pbm ? &pbm->eq_masse.inconnue().passe() : NULL, *a_r = pbm ? &pbm->eq_masse.champ_conserve().passe() : NULL;
118 const Masse_ajoutee_base *corr = pbm && pbm->has_correlation("masse_ajoutee") ? &ref_cast(Masse_ajoutee_base, pbm->get_correlation("masse_ajoutee").valeur()) : NULL;
119 int i, e, f, m, n, N = inco.line_size(), cR = rho.dimension_tot(0) == 1;
120
121 /* faces : si CLs, pas de produit par alpha * rho en multiphase */
122 DoubleTrav masse(N, N), masse_e(N, N); //masse alpha * rho, contribution
123 for (f = 0; f < zone.nb_faces(); f++) //faces reelles
124 {
125 if (!pbm)
126 for (masse = 0, n = 0; n < N; n++) masse(n, n) = vf(f); //pas Pb_Multiphase ou CL -> pas de alpha * rho
127 else for (masse = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
128 {
129 for (masse_e = 0, n = 0; n < N; n++) masse_e(n, n) = (*a_r)(e, n); //partie diagonale
130 if (corr) corr->ajouter(&(*alpha)(e, 0), &rho(!cR * e, 0), masse_e); //partie masse ajoutee
131 for (n = 0; n < N; n++)
132 for (m = 0; m < N; m++) masse(n, m) += vfd(f, i) * masse_e(n, m); //contribution au alpha * rho de la face
133 }
134 for (n = 0; n < N; n++)
135 {
136 for (m = 0; m < N; m++) secmem(f, n) += pf(f) / dt * masse(n, m) * (passe(f, m) - resoudre_en_increments * inco(f, m));
137 if (mat)
138 for (m = 0; m < N; m++)
139 if (masse(n, m)) (*mat)(N * f + n, N * f + m) += pf(f) / dt * masse(n, m);
140 }
141 }
142}

155// sert a imposer les CLs de Dirichlet en multiphase (ou la variation de P_bord ne permet de corriger que v_melange)
156DoubleTab& Masse_PolyMAC_Face::corriger_solution(DoubleTab& x, const DoubleTab& y, int incr) const
157{
158 const Zone_PolyMAC& zone = la_zone_PolyMAC;
159 const Conds_lim& cls = la_zone_Cl_PolyMAC->les_conditions_limites();
160 const IntTab& fcl = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur()).fcl();
161 const DoubleTab& nf = zone.face_normales(), &vit = equation().inconnue().valeurs();
162 const DoubleVect& fs = zone.face_surfaces();
163 int f, n, N = x.line_size(), d, D = dimension;
164
165 for (f = 0; f < zone.nb_faces_tot(); f++)
166 if (fcl(f, 0) == 2 || fcl(f, 0) == 4)
167 for (n = 0; n < N; n++) x(f, n) = incr ? -vit(f, n) : 0; //Dirichlet homogene / Symetrie: on revient a 0
168 else if (fcl(f, 0) == 3)
169 for (n = 0; n < N; n++)
170 for (x(f, n) = incr ? -vit(f, n) : 0, d = 0; d < D; d++) //Dirichlet : valeur de la CL
171 x(f, n) += nf(f, d) / fs(f) * ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), N * d + n);
172
173 return x;
174}
