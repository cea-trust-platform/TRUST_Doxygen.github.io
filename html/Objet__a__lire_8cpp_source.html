199void Objet_a_lire::read(Motcle const& motcle, Entree& is)
200{
201 switch (type)
202 {
203 case INTEGER :
204 if (dictionnaire_noms.size() == 0)
205 {
206 is>>*int_a_lire;
207 }
208 else
209 {
210 Motcle motlu;
211 is >> motlu;
212 const int rang = dictionnaire_noms.search(motlu);
213 if (rang < 0)
214 {
215 Cerr << "Error while reading parameter " << name
216 << "\n Found: " << motlu
217 << "\n Expected one on these keywords: " << dictionnaire_noms << finl;
218 barrier();
219 exit();
220 }
221 *int_a_lire = dictionnaire_valeurs[rang];
222 ptrParam& ptr=dictionnaire_params[rang];
223 if (ptr.non_nul())
224 {
225 ptr.valeur().lire_avec_accolades_depuis(is);
226 }
227 }
228 break;
229 case DOUBLE :
230 is >> (*double_a_lire);
231 break;
232 case OBJECT :
233 is >> (*obj_a_lire);
234 break;
235 case FLAG :
236 (*flag_a_lire)=1;
237 break;
238 case NON_STD :
239 (*objet_lu).lire_motcle_non_standard(motcle,is);
240 break;
241 case DERIV :
242 {
243 Motcle type_complet(prefixe_deriv);
244 Motcle the_type;
245 is >> the_type;
246 type_complet += the_type;
247 Entree_complete is2(type_complet, is);
248 is2 >> (*obj_a_lire);
249 break;
250 }
251 case ArrOfInt_size_imp:
252 {
253 ArrOfInt& arr=*arrofint_a_lire;
254 int size=arr.size_array();
255// on ne veut pas de taille nulle pour etre sur que l'utilisateur a bien fait dimensionner...
256 assert(size>0);
257 for (int i=0; i<size; i++) is >>arr[i];
258 break;
259 }
260 case ArrOfDouble_size_imp:
261 {
262 ArrOfDouble& arr=*arrofdouble_a_lire;
263 int size=arr.size_array();
264// on ne veut pas de taille nulle pour etre sur que l'utilisateur a bien fait dimensionner...
265 assert(size>0);
266 for (int i=0; i<size; i++) is >>arr[i];
267 break;
268 }
269 case PARAM:
270 {
271 param_interne.valeur().lire_avec_accolades_depuis(is);
272 break;
273 }
274 default :
275 Cerr << "Invalid parameter type" << finl;
276 Process::exit();
277 break;
278 }
