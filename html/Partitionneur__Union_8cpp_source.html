31Entree& Partitionneur_Union::readOn(Entree& is)
32{
33 Nom mot, fic;
34 is >> mot;
35 if (mot != "{")
36 {
37 Cerr << "Partitionneur_Union : list { sous_zone decoupage } expected." << finl;
38 abort();
39 }
40 for (is >> mot; mot != "}"; is >> mot)

42 is >> fic;

44 {
45 Cerr << "Partitionneur_Union : partition file expected for sub-zone " << mot << finl;
46 abort();
47 }
48 fic_ssz[mot.getString()] = fic.getString();
49 }
50 return is;
51}

69void Partitionneur_Union::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
70{
71 elem_part.resize(ref_domaine_->zone(0).nb_elem());
72 elem_part = -1;
73 for (auto &&kv : fic_ssz)
74 {
75// on recupere la sous-zone par son nom et le decoupage en ouvrant le fichier...
76 const Sous_Zone& ssz = ref_cast(Sous_Zone, Interprete::objet(kv.first));
77 EFichier file;
78 file.ouvrir(kv.second.c_str());
79 IntVect dec_ssz;
80 file >> dec_ssz;
81 file.close();
82// ... et on remplit un morceau de elem_part avec
83 if (dec_ssz.size_array() != ssz.nb_elem_tot())
84 {
85 Cerr << "Partitionneur_Union : incoherent element number for sub-zone " << kv.first << finl;
86 Process::exit();
87 }
88 for (int i = 0; i < ssz.nb_elem_tot(); i++)
89 elem_part[ssz(i)] = dec_ssz(i);
90 }
91}
