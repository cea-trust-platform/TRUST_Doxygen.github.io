25int meme_point(const ArrOfInt& x, const DoubleTab& tab, int rang_tab)
26{
27 int dime = x.size_array();
28 assert(tab.dimension(1) == dime);
29 int k;
30 for (k=0; k<dime; k++)
31 if (std::fabs(x[k]-tab(rang_tab,k)) >= 1.e-12)
32 break;
33 return (k == dime);
34}

74//! Lit le champ sur un flot d'entree.
75/** \return Entree\&<br>

77Contraintes: */
78Entree& Champ_Don_Face_lu::readOn(Entree& is)
79{
80 int dim;
81 Motcle motlu;
82 Nom nom;
83 is >> nom;
84 Probleme_base& pb= ref_cast(Probleme_base, Interprete::objet(nom));
85 associer_zone_dis_base(pb.domaine_dis().zone_dis(0).valeur());
86 const Zone_VDF& zvdf = ref_cast(Zone_VDF,zone_dis_base());
87 int nb_faces = zvdf.nb_faces();
88 is >> dim;
89 dimensionner(nb_faces, dim);
90 is >> nom;
91 if (Process::je_suis_maitre())
92 Cerr << "Champ_Don_Face_lu: lecture du fichier " << nom << finl;
93 EFichier fic(nom);
94 fic.set_check_types(1);//Remplace UFichier
95
96// Lecture des valeurs dans le fichier fic
97 DoubleTab& mes_val = valeurs();
98 const DoubleTab& xv = zvdf.xv();
99 ArrOfInt point(dimension);
100 ArrOfInt val_lu(dim);
101 ArrOfBit flag(nb_faces);
102 flag = 0;
103
104 int nb_val_lues = 0;
105 fic >> nb_val_lues;
106 int num_face = -1;
107 int nb_faces_affectees = 0;
108 int i;
109 for (i = 0; i < nb_val_lues; i++)
110 {
111 int k;
112 for (k=0; k<dimension; k++)
113 fic >> point[k];
114 for (k=0; k<dim; k++)
115 fic >> val_lu[k];
116
117// On cherche la face correspondante en partant de la derniere trouvee.
118 for (int j = 0; j < nb_faces; j++)
119 {
120 num_face++;
121 if (num_face >= nb_faces)
122 num_face = 0;
123
124 if (meme_point(point,xv,num_face))
125 {
126 if (dim == 1)
127 mes_val(num_face) = val_lu[0];
128 else
129 for (k=0; k < dim; k++)
130 mes_val(num_face,k) = val_lu[k];
131 if (! flag.testsetbit(num_face))
132 nb_faces_affectees++;
133 break;
134 }
135 }
136 }
137
138 int erreur = mp_sum(nb_faces - nb_faces_affectees);
139 if (erreur)
140 {
141 if (Process::je_suis_maitre())
142 Cerr << "Erreur dans Champ_Don_Face_lu.\n"
143 << erreur << " faces n'ont pas ete affectees.\n"
144 << "Voir la liste des faces dans les fichiers .log" << finl;
145 Journal() << "Champ_Don_Face_lu. Liste des faces non affectees :\n";
146 for (i=0; i<nb_faces; i++)
147 if (!flag[i])
148 Journal() << i << " " << xv(i, 0) << " " << xv(i, 1)
149 << " " << ((dimension == 3) ? xv(i, 3) : 0.) << finl;
150 barrier(); //!< Attend que tout le monde ait fini d'ecrire
151 exit();
152 }
153 return is;
154}
