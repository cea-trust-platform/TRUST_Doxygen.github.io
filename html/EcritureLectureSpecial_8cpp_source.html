144int ecrit(Sortie& fich, const ArrOfBit& items_to_write, const DoubleTab& pos, const DoubleTab& val)
145{
146 const int nb_dim = val.nb_dim();
147 const int nb_comp = (nb_dim == 2) ? val.dimension(1) : 1;
148 const int nb_val = items_to_write.size_array();
149 const int dim = pos.dimension(1);
150
151 if (EcritureLectureSpecial::get_Output().finit_par("MPIIO"))
152 {
153// No bufferisation needed for Parallel IO
154 int jmax = (dim + nb_comp) * nb_val;
155 ArrOfDouble tmp(jmax);
156 int j=0;
157 for (int p = 0; p < nb_val; p++)
158 {
159// On n'ecrit que les items reels non communs afin d'avoir un fichier .xyz
160// de meme taille quelque soit le decoupage et surtout de pouvoir le relire
161// quelque soit le decoupage et les supports
162 if (items_to_write[p])
163 {
164 for (int k = 0; k < dim; k++)
165 tmp[j++] = pos(p, k);
166 if (nb_dim == 1)
167 tmp[j++] = val(p);
168 else
169 for (int k = 0; k < nb_comp; k++)
170 tmp[j++] = val(p, k);
171 }
172 }
173 fich.put(tmp.addr(), j, dim + nb_comp /* nb colonnes en ascii */);
174 }
175 else
176 {
177// Bufferisation needed for EcrFicPartage
178 int jmax = (dim + nb_comp) * 128;
179 ArrOfDouble tmp(jmax);
180 int j = 0;
181 for (int p = 0; p < nb_val; p++)
182 {
183// On n'ecrit que les items reels non communs afin d'avoir un fichier .xyz
184// de meme taille quelque soit le decoupage et surtout de pouvoir le relire
185// quelque soit le decoupage et les supports
186 if (items_to_write[p])
187 {
188 for (int k = 0; k < dim; k++)
189 tmp[j++] = pos(p, k);
190 if (nb_dim == 1)
191 tmp[j++] = val(p);
192 else
193 for (int k = 0; k < nb_comp; k++)
194 tmp[j++] = val(p, k);
195 if (j == jmax)
196 {
197 fich.put(tmp.addr(), j, dim + nb_comp /* nb colonnes en ascii */);
198// On flushe regulierement en sequentiel car sur certains tres gros maillages
199// stack overflow possible...
200 if (Process::nproc()==1) fich.syncfile();
201 j = 0;
202 }
203 }
204 }
205 if (j)
206 fich.put(tmp.addr(), j, dim + nb_comp /* nb colonnes en ascii */);
207 }
208 return 8 * (dim + nb_comp) * nb_val; //!< Bytes
209}

213//! Partie "interieure" de l'ecriture, appellee par la methode en dessous.<br>Methode recursive, si le tableau a ecrire a un descripteur MD_Vector_composite
214static int ecriture_special_part2(const Zone_VF& zvf, Sortie& fich, const DoubleTab& val)
215{
216 const MD_Vector& md = val.get_md_vector();
217 int bytes = 0;
218 if (sub_type(MD_Vector_composite, md.valeur()))
219 {
220// Champs p1bulles et autres: appel recursif pour les differents sous-tableaux:
221 ConstDoubleTab_parts parts(val);
222 int n = zvf.que_suis_je() == "Zone_PolyMAC" || zvf.que_suis_je() == "Zone_PolyMAC_P0" ? 1 : parts.size();//on saute les variables auxiliaires de Champ_{P0,Face}_PolyMAC
223 for (int i = 0; i < n; i++)
224 bytes += ecriture_special_part2(zvf, fich, parts[i]);
225 }
226 else if (sub_type(MD_Vector_std, md.valeur()))
227 {
228 ArrOfBit items_to_write;
229 MD_Vector_tools::get_sequential_items_flags(md, items_to_write);
230 const DoubleTab& coords = get_ref_coordinates_items(zvf, md);
231 bytes += ecrit(fich, items_to_write, coords, val);
232 fich.syncfile();
233 }
234 else
235 {
236 Cerr << "EcritureLectureSpecial::ecriture_special_part: Error, unknown Metadata vector type : "
237 << md.valeur().que_suis_je() << finl;
238 Process::exit();
239 }
240 return bytes;
241}

250//! codage de l'ecriture des positions et des valeurs de val
251int EcritureLectureSpecial::ecriture_special(const Zone_VF& zvf, Sortie& fich, const DoubleTab& val)
252{
253 const MD_Vector& md = val.get_md_vector();
254 if (!md.non_nul())
255 {
256 Cerr << "EcritureLectureSpecial::ecriture_special: error, cannot save an array with no metadata" << finl;
257 Process::exit();
258 }
259 const int nb_items_seq = md.valeur().nb_items_seq_tot();
260 if (nb_items_seq == 0)
261 return 0;
262
263 const int nb_dim = val.nb_dim();
264 const int nb_comp = (nb_dim == 2) ? val.dimension(1) : 1;
265 const int dim = Objet_U::dimension;
266 const int n = nb_items_seq * (nb_comp + dim);
267
268
269 if (Process::je_suis_maitre())
270 {
271 fich << (int)1 << finl;
272 fich << n << finl ;
273 fich << (int)1 << finl;
274 fich << n << finl ;
275 fich << n <<finl;
276 }
277
278 int bytes = ecriture_special_part2(zvf, fich, val);
279
280 if (Process::je_suis_maitre())
281 {
282 fich << n << finl;
283 fich << (int)0 << finl;
284 fich << (int)0 << finl;
285 fich << (int)0 << finl;
286 fich << (int)1 << finl;
287 fich << (int)0 << finl;
288 fich << n << finl;
289 fich << finl;
290 fich << (int)1 << finl;
291 fich << (int)0 << finl;
292 fich << (int)0 <<finl;
293 }
294 fich.syncfile();
295 return bytes;
296}

319//! Reciproque de la methode ecrit(...), lit uniquement les items sequentiels<br>(donc pas les items communs recus d'un autre processeur)<br>On verifie a la fin qu'on a bien lu exactement le nombre d'items attendus, s'il en manque<br>c'est que le epsilon n'est pas bon (ou qu'on a change le maillage...)<br>Valeur de retour: nombre total d'items sequentiels lus (sur tous les procs)
320static int lire_special(Entree& fich, const DoubleTab& coords, DoubleTab& val, const double epsilon)
321{
322 const int dim = coords.dimension(1);
323 const int nb_dim = val.nb_dim();
324 const int nb_comp = (nb_dim == 1) ? 1 : val.dimension(1);
325
326 const MD_Vector& md_vect = val.get_md_vector();
327// Dans un premier temps, 1 si l'item est a lire, 0 s'il est lu par un autre processeur.
328// Une fois que l'item est lu, on met le flag a 2.
329 ArrOfInt items_to_read;
330 const int n_to_read = MD_Vector_tools::get_sequential_items_flags(md_vect, items_to_read);
331 Octree_Double octree;
332// Build an octree with "thick" nodes (epsilon size)
333 octree.build_nodes(coords, 0 /* do not include virtual elements */, epsilon);
334 const ArrOfInt& floor_elements = octree.floor_elements();
335
336// Le fichier contient ce nombre de lignes pour cette partie du tableau (nombre total d'items sequentiels)
337 const int ntot = Process::mp_sum(n_to_read);
338
339// On lit dans le fichier par blocs de buflines_max parce qu'il y a
340// un broadcast reseau a chaque comm:
341 const int buflines_max = 2048; //!< pas trop, histoire d'avoir plusieurs blocs dans les cas tests
342 DoubleTab buffer(buflines_max, dim + nb_comp);
343 int bufptr = buflines_max;
344 ArrOfInt items;
345 items.set_smart_resize(1);
346
347 double max_epsilon_needed = epsilon;
348// Combien de fois on a trouve plusieurs candidats a moins de epsilon ?
349 int error_too_many_matches = 0;
350// Combien de fois on est tombe plusieurs fois sur le meme sommet a lire ?
351 int error_duplicate_read = 0;
352// Combien d'items a-t-on lu ?
353 int count_items_read = 0;
354
355// Boucle sur les items sequentiels du fichier:
356 int pourcent=0;
357 for (int i = 0; i < ntot; i++)
358 {
359 int tmp=(i*10)/(ntot-1);
360 if (tmp>pourcent || i==0)
361 {
362 pourcent=tmp;
363 Cerr<<"\r"<<pourcent*10<<"% of data has been found."<<flush;
364 }
365 if (bufptr == buflines_max)
366 {
367 bufptr = 0;
368 int n = std::min(buflines_max, ntot - i) * (dim + nb_comp);
369 assert(n <= buffer.size_array());
370 fich.get(buffer.addr(), n);
371 }
372 const double x = buffer(bufptr, 0);
373 const double y = buffer(bufptr, 1);
374 const double z = (dim == 3) ? buffer(bufptr, 2) : 0.;
375// Recherche des items correspondant potentiellement au point (x,y,z)
376 int index = -1;
377 int nb_items_proches = octree.search_elements(x, y, z, index);
378 if (nb_items_proches > 0)
379 {
380 items.resize_array(nb_items_proches, ArrOfInt::NOCOPY_NOINIT);
381// Voir doc de Octree_Double::search_elements: on copie les indices des items proches dans items:
382 for (int j = 0; j < nb_items_proches; j++)
383 items[j] = floor_elements[index++];
384// On reduit la liste pour avoir uniquement les items a moins de epsilon
385 const int item_le_plus_proche = octree.search_nodes_close_to(x, y, z, coords, items, epsilon);
386 nb_items_proches = items.size_array();
387 if (nb_items_proches == 1)
388 {
389 const int flag = items_to_read[item_le_plus_proche];
390 if (flag == 1)
391 {
392// Ok, il faut lire cette valeur
393 items_to_read[item_le_plus_proche] = 2;
394 count_items_read++;
395 if (nb_dim == 1)
396 {
397 val(item_le_plus_proche) = buffer(bufptr, dim);
398 }
399 else
400 {
401 for (int j = 0; j < nb_comp; j++)
402 val(item_le_plus_proche, j) = buffer(bufptr, dim + j);
403 }
404 }
405 else if (flag == 0)
406 {
407// Cet item n'est pas a moi, ne pas le lire
408 }
409 else
410 {
411// Erreur, on a deja lu cet item !!! epsilon est trop grand (ou erreur a la sauvegarde ???)
412 error_duplicate_read++;
413 }
414 }
415 else if (nb_items_proches == 0)
416 {
417// ok, le sommet est sur un autre processeur (ou epsilon trop petit ??)
418 }
419 else
420 {
421// Erreur: epsilon est trop grand, on a plusieurs candidats a moins de epsilon
422// Calcul de la distance avec le deuxieme plus proche pour afficher un message d'erreur a la fin:
423 for (int ii = 0; ii < nb_items_proches; ii++)
424 {
425 const int i_coord = items[ii];
426 if (i_coord == item_le_plus_proche)
427 continue; //!< celui-la est sans doute le bon, il faut un epsilon superieur a cette valeur la...
428 double xx = 0;
429 for (int j = 0; j < dim; j++)
430 {
431 double yy = coords(i_coord, j) - buffer(bufptr, j);
432 xx += yy * yy;
433 }
434// On propose de mettre un epsilon au maximum egal a 1/10 de la distance avec le deuxieme point le plus proche:
435 xx = sqrt(xx) * 0.1;
436 if (max_epsilon_needed > xx)
437 max_epsilon_needed = xx;
438 }
439 error_too_many_matches++;
440 }
441 }
442 bufptr++;
443 }
444 Cerr << finl;
445// Erreurs ?
446 int err = (count_items_read != n_to_read) || (error_too_many_matches > 0) || (error_duplicate_read > 0);
447 err = Process::mp_sum(err);
448 if (err)
449 {
450 error_too_many_matches = Process::mp_sum(error_too_many_matches);
451 error_duplicate_read = Process::mp_sum(error_duplicate_read);
452 max_epsilon_needed = Process::mp_min(max_epsilon_needed);
453 if (Process::je_suis_maitre())
454 {
455 if (error_too_many_matches)
456 {
457 Cerr << "Error in EcritureLectureSpecial: error_too_many_matches = " << error_too_many_matches
458 << ", epsilon is too large. Suggested value: " << max_epsilon_needed << finl;
459 if (max_epsilon_needed==0)
460 {
461 Cerr << "It could be because your mesh has two boundaries which match exactly." << finl;
462 Cerr << "It is possible to do calculation with this property but xyz restart process" << finl;
463 Cerr << "is impossible because it can't detect the differences between faces of the two boundaries..." << finl;
464 Cerr << "Try to do a classic restart with .sauv files." << finl;
465 }
466 }
467 else if (error_duplicate_read)
468 {
469 Cerr << "Error in EcritureLectureSpecial: error_duplicate_read = " << error_duplicate_read
470 << ", probably epsilon too large. " << finl;
471 }
472 else
473 {
474 Cerr << "Error in EcritureLectureSpecial: Some items were not found: epsilon too small (or the mesh has changed?)" << finl;
475 }
476 }
477 Process::barrier();
478 Process::exit();
479 }
480 return ntot;
481}

484static int lecture_special_part2(const Zone_VF& zvf, Entree& fich, DoubleTab& val)
485{
486 const MD_Vector& md = val.get_md_vector();
487
488 int ntot = 0;
489 if (sub_type(MD_Vector_composite, md.valeur()))
490 {
491// Champs p1bulles et autres: appel recursif pour les differents sous-tableaux:
492 DoubleTab_parts parts(val);
493 const int n = parts.size();
494 for (int i = 0; i < n; i++)
495 ntot += lecture_special_part2(zvf, fich, parts[i]);
496 }
497 else if (sub_type(MD_Vector_std, md.valeur()))
498 {
499 const DoubleTab& coords = get_ref_coordinates_items(zvf, md);
500 const double epsilon = zvf.zone().domaine().epsilon();
501 ntot += lire_special(fich, coords, val, epsilon);
502 }
503 else
504 {
505 Cerr << "EcritureLectureSpecial::lecture_special_part2: Error, unknown Metadata vector type : "
506 << md.valeur().que_suis_je() << finl;
507 Process::exit();
508 }
509 return ntot;
510}
