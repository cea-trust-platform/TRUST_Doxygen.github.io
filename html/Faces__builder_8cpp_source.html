62//! A partir de la description des elements de la zone et des frontieres<br>(bords, raccords, faces internes et joints) :<br>Remplissage des structures suivantes:<br>- pour les frontieres de la zone: fixer_num_premiere_face<br>- les_faces.faces_sommets (faces reeles)<br>- les_faces.faces_voisins (faces reeles)<br>- elem_faces (pour les faces reeles des elements reels)<br>(on initialise elem_faces de taille nb_elem_reels, nb_faces_par_elem)<br>- joints.items_communs(FACE)
63void Faces_builder::creer_faces_reeles(Zone& zone,
64 const Static_Int_Lists& connect_som_elem,
65 Faces& les_faces,
66 IntTab& elem_faces)
67{
68 les_elements_ptr_ = & zone.les_elems();
69
70 connectivite_som_elem_ptr_ = & connect_som_elem;
71// La connectivite doit contenir les sommets virtuels
72 assert(connect_som_elem.get_nb_lists() == zone.nb_som_tot());
73
74// Remplissage du tableau des faces de l'element de reference
75
76 is_polyedre_=0;
77 if (sub_type(Poly_geom_base,zone.type_elem().valeur()))
78 {
79 is_polyedre_=1;
80 }
81 else
82 zone.type_elem().valeur().get_tab_faces_sommets_locaux(faces_element_reference_old_);
83// Tableau de taille (nb_faces, nb_sommets par face),
84// pour chaque face, les indices de ses sommets dans le domaine.
85// L'ordre des sommets est celui donne par l'element de reference,
86// pour celui des elements voisins de la face qui a le plus petit
87// indice.
88 IntTab& faces_sommets = les_faces.les_sommets();
89
90// Tableau de taille (nb_faces, 2) contenant pour chaque face
91// les indices des deux elements voisins. Si "i_face" a un seul voisin,
92// faces_voisins_(i_face, 1) = -1;
93 IntTab& faces_voisins = les_faces.voisins();
94
95// Initialisation des references utilisees dans check_erreur_faces
96 faces_sommets_ = faces_sommets;
97 face_elem_ = faces_voisins;
98 ref_zone_ = zone;
99
100// Le tableau des faces des elements:
101// dimension(0) = nombre d'elements,
102// dimension(1) = nombre de faces par element
103// elem_faces(i,j) = indice de la face j de l'element i dans les
104// tableaux faces_sommets et faces_voisins
105// (les faces de l'element sont dans l'ordre donne par faces_element_reference)
106// espaces distants et virtuels appropries pour les elements
107 const int nb_elements = les_elements().dimension(0);
108 const int nb_faces_par_element = faces_element_reference(0).dimension(0) ? faces_element_reference(0).dimension(0) : 4;
109 elem_faces.resize(nb_elements, nb_faces_par_element);
110 elem_faces = -1;
111
112 const int nb_sommets_par_face = faces_element_reference(0).dimension(0) ? faces_element_reference(0).dimension(1) : 3;
113// On ajoute chaque face avec resize(n+1,...), donc smart_resize:
114// Calcul du nombre theorique de faces:
115 const int nb_faces_front = zone.nb_faces_frontiere() + zone.nb_faces_joint();
116 int nb_faces_prevision = (nb_elements * nb_faces_par_element + nb_faces_front) / 2;
117 if (is_polyedre_)
118 {
119// les faces sont toutes deja connues....
120 const Poly_geom_base& poly=ref_cast(Poly_geom_base,ref_zone_->type_elem().valeur());
121 nb_faces_prevision=(poly.get_somme_nb_faces_elem()+ nb_faces_front) / 2;;
122 }
123// Allocation memoire pour le nombre de faces prevu pour eviter de reallouer
124// de la memoire n fois (voir set_smart_resize)
125 faces_sommets.set_smart_resize(1);
126 faces_sommets.resize(nb_faces_prevision, nb_sommets_par_face);
127 faces_sommets.resize(0, nb_sommets_par_face);
128 faces_voisins.set_smart_resize(1);
129 faces_voisins.resize(nb_faces_prevision, 2);
130 faces_voisins.resize(0, 2);
131
132// ******** Traitement des frontieres **********
133// attention, on initialise "num_premiere_face" pour les frontieres !
134
135// Creation des faces de bord
136 {
137 Bords& bords = zone.faces_bord();
138 const int n = bords.size();
139 for (int i = 0; i < n; i++)
140 {
141 Frontiere& frontiere = bords[i];
142 creer_faces_frontiere(1, /* un element voisin par face */
143 frontiere,
144 faces_sommets,
145 faces_voisins,
146 elem_faces);
147 }
148 }
149// Raccords
150 {
151 Raccords& raccords = zone.faces_raccord();
152 const int n = raccords.size();
153 for (int i = 0; i < n; i++)
154 {
155 Frontiere& frontiere = raccords[i].valeur();
156 creer_faces_frontiere(1, /* un element voisin par face */
157 frontiere,
158 faces_sommets,
159 faces_voisins,
160 elem_faces);
161 }
162 }
163
164// Faces de bord "internes"
165 {
166 Faces_Internes& faces_int = zone.faces_int();
167 const int n = faces_int.size();
168 for (int i = 0; i < n; i++)
169 {
170 Frontiere& frontiere = faces_int[i];
171 creer_faces_frontiere(2, /* deux elements voisin par face */
172 frontiere,
173 faces_sommets,
174 faces_voisins,
175 elem_faces);
176 }
177
178// On duplique les faces internes : pour chaque face qui a deux
179// voisins, on cree une deuxieme face identique avec le deuxieme voisin,
180// on efface le deuxieme voisin de la face d'origine et on change
181// la face voisins de deuxieme voisin:
182 if (n > 0)
183 {
184 Cerr << "Faces_builder::creer_faces_reeles not coded for the internal faces of boundary" << finl;
185 Process::exit();
186// A faire selon l'ancienne version de zone2... et a tester !
187 }
188 }
189
190// Faces de joint
191 {
192 Joints& joints = zone.faces_joint();
193 const int n = joints.size();
194 for (int i = 0; i < n; i++)
195 {
196 Frontiere& frontiere = joints[i];
197 creer_faces_frontiere(2, /* elements voisins par face */
198 frontiere,
199 faces_sommets,
200 faces_voisins,
201 elem_faces);
202// Remplissage de items_communs(FACE)
203// Les faces de joint sont dans le meme ordre en local et sur le voisin.
204 Joint& joint = joints[i];
205 ArrOfInt& indices_faces =
206 joint.set_joint_item(Joint::FACE).set_items_communs();
207 const int nb_faces = joint.nb_faces();
208 indices_faces.resize_array(nb_faces);
209 const int num_premiere_face = joint.num_premiere_face();
210 for (int i2 = 0; i2 < nb_faces; i2++)
211 indices_faces[i2] = num_premiere_face + i2;
212 }
213 }
214
215// *********************************************
216// Faces internes
217
218 creer_faces_internes(faces_sommets,
219 elem_faces,
220 faces_voisins);
221
222// *********************************************
223// C'est fini: on verifie qu'on a bien le nombre de faces prevu
224 if (faces_sommets.dimension(0) != nb_faces_prevision)
225 {
226 Cerr << "Error in Faces_builder::creer_faces_reeles:\n"
227 << " number of faces does not match predicted number of faces.\n"
228 << " (problem with faces_internes ?)" << finl;
229 Process::exit();
230 }
231
232// RAZ attribut smart_resize des tableaux faces_sommets et faces_voisins.
233 faces_sommets.set_smart_resize(0);
234 faces_voisins.set_smart_resize(0);
235// RAZ des attributs de la classe
236 reset();
237}

294//! ajoute une face reelle dans faces_sommets et faces_voisins.
295int Faces_builder::ajouter_une_face(const ArrOfInt& une_face,
296 const int elem0,
297 const int elem1,
298 IntTab& faces_sommets,
299 IntTab& faces_voisins)
300{
301 int i;
302 const int num_new_face = faces_sommets.dimension(0);
303 const int nb_sommets_par_face = faces_sommets.dimension(1);
304 const int new_size = num_new_face + 1;
305
306 assert(une_face.size_array() == nb_sommets_par_face);
307 faces_sommets.resize(new_size, nb_sommets_par_face);
308 for (i = 0; i < nb_sommets_par_face; i++)
309 faces_sommets(num_new_face, i) = une_face[i];
310
311 faces_voisins.resize(new_size, 2);
312 faces_voisins(num_new_face, 0) = elem0;
313 faces_voisins(num_new_face, 1) = elem1;
314
315 return num_new_face;
316}

318int Faces_builder::chercher_face_element(const IntTab& elem_som,
319 const IntTab& faces_element_ref,
320 const ArrOfInt& une_face,
321 const int elem)
322{
323#ifdef old
324#ifndef NDEBUG
325 ArrOfInt sommets_element(8);
326#else
327 int sommets_element[8];
328#endif
329#endif
330
331 const int nb_faces_element = faces_element_ref.dimension(0);
332 const int nb_sommets_par_face = faces_element_ref.dimension(1);
333
334 int i_face, i_som, i_som2;
335 for (i_face = 0; i_face < nb_faces_element; i_face++)
336 {
337 for (i_som = 0; i_som < nb_sommets_par_face; i_som++)
338 {
339 const int sommet_elem_ref = faces_element_ref(i_face, i_som);
340 int sommet_domaine ;
341 if (sommet_elem_ref==-1)
342 sommet_domaine=-1;
343 else
344 sommet_domaine = elem_som(elem, sommet_elem_ref);
345 for (i_som2 = 0; i_som2 < nb_sommets_par_face; i_som2++)
346 if (une_face[i_som2] == sommet_domaine) //!< si sommet trouve, stop
347 break;
348 if (i_som2 == nb_sommets_par_face) //!< si sommet non trouve, stop
349 break;
350 }
351 if (i_som == nb_sommets_par_face) //!< si tous les sommets ont ete trouves, stop
352 break;
353 }
354 if (i_face == nb_faces_element) //!< si face non trouvee
355 return -1;
356 else
357 return i_face;
358}

398//! Insere les faces de la frontiere donnee dans les trois tableaux,<br>a la suite des faces deja presentes dans faces_sommets.<br>Remplissage de :<br>frontiere.num_premiere_face<br>Completion de :<br>faces_sommets<br>elem_faces<br>faces_voisins
399void Faces_builder::creer_faces_frontiere(const int nb_voisins_attendus,
400 Frontiere& frontiere,
401 IntTab& faces_sommets,
402 IntTab& faces_voisins,
403 IntTab& elem_faces) const
404{
405 assert(nb_voisins_attendus == 1 || nb_voisins_attendus == 2);
406
407 const Static_Int_Lists& som_elem = connectivite_som_elem();
408 const int nb_sommets_par_face = faces_element_reference(0).dimension(0) ? faces_element_reference(0).dimension(1) : 3;
409 const int num_premiere_face = faces_sommets.dimension(0);
410 const int nb_elem_reels = elem_faces.dimension(0);
411 frontiere.fixer_num_premiere_face(num_premiere_face);
412
413 const Faces& faces_frontiere = frontiere.faces();
414 const IntTab& sommets_faces_fr = faces_frontiere.les_sommets();
415 const int nb_faces = faces_frontiere.nb_faces();
416 ArrOfInt une_face(nb_sommets_par_face);
417 ArrOfInt voisins;
418 voisins.set_smart_resize(1);
419 ArrOfInt liste_faces_erreur0;
420 liste_faces_erreur0.set_smart_resize(1);
421 ArrOfInt liste_faces_erreur1;
422 liste_faces_erreur1.set_smart_resize(1);
423 ArrOfInt liste_faces_erreur2;
424 liste_faces_erreur2.set_smart_resize(1);
425 ArrOfInt liste_faces_erreur3;
426 liste_faces_erreur3.set_smart_resize(1);
427
428 int i_face;
429 for (i_face = 0; i_face < nb_faces; i_face++)
430 {
431 {
432 int nb_sommets_par_face_fr=sommets_faces_fr.dimension(1);
433 for (int i = 0; i < std::min(nb_sommets_par_face, nb_sommets_par_face_fr); i++)
434 une_face[i] = sommets_faces_fr(i_face, i);
435 for (int i = std::min(nb_sommets_par_face, nb_sommets_par_face_fr); i < nb_sommets_par_face; i++)
436 une_face[i] = -1;
437 }
438// Quels sont les elements voisins de cette face ?
439 find_adjacent_elements(som_elem, une_face, voisins);
440 const int nb_voisins = voisins.size_array();
441 const int elem0 = (nb_voisins > 0) ? voisins[0] : -1;
442 const int elem1 = (nb_voisins > 1) ? voisins[1] : -1;
443 const int indice_face =
444 ajouter_une_face(une_face, elem0, elem1, faces_sommets, faces_voisins);
445
446 switch(nb_voisins)
447 {
448 case 0:
449 {
450// Erreur: la face n'a pas de voisin
451 liste_faces_erreur0.append_array(indice_face);
452 break;
453 }
454 case 1:
455 case 2:
456 {
457 if (nb_voisins_attendus == nb_voisins)
458 {
459 int i_voisin;
460 for (i_voisin = 0; i_voisin < nb_voisins; i_voisin++)
461 {
462 const int elem = voisins[i_voisin];
463// Quelle est la face de l'element ?
464 const int i_face_elem = chercher_face_element(une_face, elem);
465 if (i_face_elem >= 0)
466 {
467// Si c'est un element reel, on associe la face
468 if (elem < nb_elem_reels)
469 {
470 if (elem_faces(elem, i_face_elem) < 0)
471 {
472 elem_faces(elem, i_face_elem) = indice_face;
473 }
474 else
475 {
476// Erreur: cette face existe deja (dans cette frontiere ou une autre)
477 liste_faces_erreur3.append_array(indice_face);
478 }
479 }
480 }
481 else
482 {
483// Erreur: la face n'est pas une face de l'element.
484 liste_faces_erreur0.append_array(indice_face);
485 }
486 }
487 }
488 else
489 {
490// Erreur, on attendait pas ce nombre de voisins.
491 liste_faces_erreur1.append_array(indice_face);
492 }
493 break;
494 }
495 default:
496// Erreur, plus de deux voisins, c'est n'importe quoi...
497 liste_faces_erreur2.append_array(indice_face);
498 }
499 }
500 Nom msg;
501 msg = "Boundary \"";
502 msg += frontiere.le_nom();
503 msg += "\" contains faces which do not belong to any element.";
504 check_erreur_faces(msg, liste_faces_erreur0);
505
506 msg = "Boundary \"";
507 msg += frontiere.le_nom();
508 msg += "\" contains faces that belong to ";
509 msg += Nom(3-nb_voisins_attendus);
510 msg += " elements.\n";
511 switch(nb_voisins_attendus)
512 {
513 case 1:
514 msg += "These faces should have only 1 neighbouring element.";
515 break;
516 case 2:
517 msg += "These faces should have 2 neighbouring elements.";
518 break;
519 default:
520 msg = "Internal error.";
521 }
522 if (sub_type(Joint, frontiere))
523 {
524// Deux sources d'erreur possibles: les faces de joint sont fausses
525// ou bien la zone ne contient pas les elements virtuels (il faut
526// au moins que la zone contienne les elements virtuels voisins des
527// faces de joint).
528 msg += "(Error in a Joint object: internal error in the mesh splitter or scatter ? )\n";
529 }
530 check_erreur_faces(msg, liste_faces_erreur1);
531
532 msg = "Boundary \"";
533 msg += frontiere.le_nom();
534 msg += "\" contains faces that belong to more than 2 elements.\n";
535 check_erreur_faces(msg, liste_faces_erreur2);
536
537 msg = "Boundary \"";
538 msg += frontiere.le_nom();
539 msg += "\" contains faces that already exist in another boundary or in this one.\n";
540 check_erreur_faces(msg, liste_faces_erreur3);
541}

546//! Construction des faces interieures au domaine<br>(faces qui ont deux voisins et qui ne sont pas des "faces_bord_internes")<br>Les faces de joint ont deja ete creees.
547void Faces_builder::creer_faces_internes(IntTab& faces_sommets,
548 IntTab& elem_faces,
549 IntTab& faces_voisins) const
550{
551 const IntTab& elem_som = les_elements();
552 const Static_Int_Lists& som_elem = connectivite_som_elem();
553// const IntTab & faces_elem_ref = faces_element_reference();
554 const int nb_elem = elem_som.dimension(0);
555 const int nb_faces_par_element = faces_element_reference(0).dimension(0);
556 const int nb_sommets_par_face = nb_faces_par_element ? faces_element_reference(0).dimension(1) : 3;
557
558// Tableau temporaire dans lequel on stocke les indices des sommets
559// de la face en cours de traitement
560 ArrOfInt une_face(nb_sommets_par_face);
561// Tableau temporaire (liste des elements voisins d'une face)
562 ArrOfInt voisins;
563 voisins.set_smart_resize(1);
564// Liste des faces n'ayant qu'un seul voisin et qui ne figurent pas
565// dans les faces de bord (ce sont des erreurs):
566 ArrOfInt liste_faces_frontiere_non_declarees;
567 liste_faces_frontiere_non_declarees.set_smart_resize(1);
568 ArrOfInt liste_faces_joint_non_declarees;
569 liste_faces_joint_non_declarees.set_smart_resize(1);
570// Liste des faces presentant une erreur de connectivite (plus de
571// deux elements voisins, ou connection a des sommets qui ne
572// sont pas une face de l'element:
573 ArrOfInt liste_faces_erreurs_connectivite;
574 liste_faces_erreurs_connectivite.set_smart_resize(1);
575
576// Boucle sur les elements
577 int i_elem;
578 for (i_elem = 0; i_elem < nb_elem; i_elem++)
579 {

581// Boucle sur les faces de l'element
582 for (i_face = 0; i_face < nb_faces_par_element; i_face++)

585// L'indice de cette face dans le tableau faces_sommets.
586// Il vaut -1 si la face n'a pas encore ete creee,
587 int indice_face = elem_faces(i_elem, i_face);
588
589// Calcul des indices des sommets de la face dans le domaine:
590 int i;
591// Attention il ne faut laisser l'appel ici...
592 const IntTab& faces_elem_ref = faces_element_reference(i_elem);
593
594 for (i = 0; i < nb_sommets_par_face; i++)
595 {
596// indice du sommet sur l'element de reference

598// indice du sommet dans le domaine

601 else

604 une_face[i] = i_som;
605 }
606 }
607 if (une_face[0]==-1)
608 {
609// on a une face bidon on ne fait rien
610 elem_faces(i_elem, i_face) = -1;
611 }
612 else
613 {
614// Recherche des elements voisins de cette face.
615// Le tableau "voisins" est classe dans l'ordre croissant.
616 find_adjacent_elements(som_elem, une_face, voisins);
617
618 const int nb_voisins = voisins.size_array();
619 assert (nb_voisins > 0); //!< Il devrait au moins y avoir i_elem !!! (ou alors on a une face constitues de -1);
620
621 if (nb_voisins == 1) //!< ***** La face a 1 voisin ********
622 {
623
624 assert(voisins[0] == i_elem); //!< L'element voisin est forcement i_elem Une face ayant un seul element voisin est une face de frontiere.
625//
626 if (indice_face >= 0)
627 {
628// Ok, c'est normal, les frontieres ont deja ete traitees
629 }
630 else
631 {
632// Erreur: la face n'existe pas encore. Elle devrait avoir ete
633// creee a partir des frontieres (creer_faces_frontiere)

635 faces_sommets, faces_voisins);
636 liste_faces_frontiere_non_declarees.append_array(indice_face);
637 }
638
639 }
640 else if (nb_voisins == 2) //!< ***** La face a 2 voisins ********
641 {
642
643 const int elem0 = voisins[0];
644 const int elem1 = voisins[1];
645 assert(elem0 < elem1);
646 if (indice_face >= 0)
647 {
648// La face a deje ete creee.
649 }
650 else
651 {
652// La face n'existe pas encore.
653 if (elem0 == i_elem)
654 {
655// Les voisins sont classes: elem0 < elem1
656// donc c'est la premiere fois qu'on parcourt cette face dans la boucle
657// sur les elements.
658 indice_face = ajouter_une_face(une_face, elem0, elem1,
659 faces_sommets, faces_voisins);
660
661// Ou est cette face sur l'element voisin ?
662 const int i_face_elem1 = chercher_face_element(une_face, elem1);
663 if (i_face_elem1 >= 0)
664 {
665 if (elem1 < nb_elem) //!< Element voisin reel ?
666 elem_faces(elem1, i_face_elem1) = indice_face;
667 }
668 else
669 {
670// Erreur, les sommets de la face sont des sommets de l'element elem1
671// mais ne sont pas sur une face de cet element. Erreur de
672// connectivite du maillage.
673 liste_faces_erreurs_connectivite.append_array(indice_face);
674 }
675 if (elem1 >= nb_elem)
676 {
677// Erreur : le voisin est un element virtuel, cette face
678// devrait etre dans les faces de joint, donc deja creee.
679 liste_faces_joint_non_declarees.append_array(indice_face);
680 }
681 }
682 else
683 {
684 assert(elem1 == i_elem);
685 indice_face = ajouter_une_face(une_face, elem0, elem1,

687// On aurait deja du creer cette face car elle est voisine de elem0
688// qui est deja traite (indice plus petit). Si on arrive ici,
689// c'est que les sommets de "une_face" appartiennent bien a l'elem0,
690// mais qu'ils ne sont pas sur une face de cet element. C'est une
691// erreur de connectivite.
692 liste_faces_erreurs_connectivite.append_array(indice_face);
693 }
694 }
695
696 }
697 else //!< ***** La face a > 2 voisins ********
698 {
699 if (indice_face < 0)
700 {
701 const int elem0 = voisins[0];
702 const int elem1 = voisins[1];
703 indice_face = ajouter_une_face(une_face, elem0, elem1,
704 faces_sommets, faces_voisins);
705 }
706 liste_faces_erreurs_connectivite.append_array(indice_face);
707 }
708
709// Si la face n'existait pas, on l'a creee et on a mis son indice
710// dans indice_face. Sinon on a trouve l'indice de la face existante.
711 assert(indice_face >= 0);
712 elem_faces(i_elem, i_face) = indice_face; /* WRITE elem_faces */
713 }
714 }
715 }

718 {

720 const char * const msg2 = "Joint faces are incomplete: internal error in the mesh splitter\n";
721 const char * const msg3 = "Connectivity error in the mesh elements. Possible errors:\n- one face of one element belongs to more than 2 elements\n- two element have at least 3 common nodes but these nodes are not faces of these elements\n";
722 check_erreur_faces(msg1, liste_faces_frontiere_non_declarees);
723 check_erreur_faces(msg2, liste_faces_joint_non_declarees);
724 check_erreur_faces(msg3, liste_faces_erreurs_connectivite);
725 }

844//! Cette methode permet de faire un echange espace virtuel d'un tableau aux aretes<br>sans passer par le descripteur des aretes. On utilise le tableau elem_aretes et l'echange<br>espace virtuel des elements
845static void echanger_tableau_aretes(const IntTab& elem_aretes, int nb_aretes_reelles, ArrOfInt& tab_aretes)
846{
847 const int moi = Process::me();
848
849 const int nb_elem = elem_aretes.dimension(0);
850 const int nb_elem_tot = elem_aretes.dimension_tot(0);
851 const int nb_aretes_elem = elem_aretes.dimension(1);
852 int i;
853
854// **********************
855// I) Echange pour mettre a jour les items communs
856// Algo un peu complique pour mettre a jour les items communs: pour chaque arete reele,
857// la valeur de tab_aretes doit etre egale a la valeur initiale de tab_arete donnee par
858// le processeur de rang le plus petit parmi ceux qui partagent l'arete (c'est a dire
859// les processeurs qui ont un element adjacent a cette arete).
860
861// Tableau permettant de connaitre le processeur proprietaire d'une arete reele
862 ArrOfInt pe_arete(nb_aretes_reelles);
863 pe_arete= moi;
864// Tableau qui donne, pour chaque element, le processeur proprietaire
865 IntVect pe_elem(nb_elem_tot);
866 pe_elem= moi; //!< initialise avec "moi"
867 {
868 pe_elem.set_md_vector(elem_aretes.get_md_vector());
869 pe_elem.echange_espace_virtuel();
870// On range dans pe_arete le numero du plus petit processeur proprietaire des
871// elements adjacents a cette arete
872// Inutile de parcourir les elements reels, on va trouver pe_elem[i]==moi...
873// Si l'arete se trouve sur un processeur de rang inferieur, on lui attribue
874 for (i = nb_elem; i < nb_elem_tot; i++)
875 for (int pe = pe_elem[i], j = 0, a; j < nb_aretes_elem && (a = elem_aretes(i, j)) >= 0; j++)
876 if (a < nb_aretes_reelles && pe_arete[a] > pe)
877 pe_arete[a] = pe;
878 }
879// On suppose que l'espace virtuel des elements contient au moins une couche d'elements virtuels
880// (tous les voisins des elements reels par des sommets) alors les aretes reelles sont
881// echangees (pas encore les aretes virtuelles)
882// Dans ce cas, pe_arete est maintenant correctement rempli pour les aretes reelles.
883
884 IntTab tmp;
885 tmp.copy(elem_aretes, Array_base::NOCOPY_NOINIT); //!< copier uniquement la structure
886
887// Copier tab_aretes dans la structure tmp (on sait echanger tmp, pas tab_aretes)
888 for (i = 0; i < nb_elem; i++)
889 for (int j = 0, a; j < nb_aretes_elem && (a = elem_aretes(i, j)) >= 0; j++)
890 tmp(i, j) = tab_aretes[a];
891
892// 2) Echange du tableau
893 tmp.echange_espace_virtuel();
894
895// 3) On reverse dans la partie reelle de tab_aretes les valeurs prises dans tmp:
896// pour une arete partagee par plusieurs procs, c'est le proc de rang le plus petit
897// qui donne la valeur
898// Inutile de parcourir les elements reels, la valeur ne changerait pas
899 for (i = nb_elem; i < nb_elem_tot; i++)
900 for (int pe = pe_elem[i], j = 0, a; j < nb_aretes_elem && (a = elem_aretes(i, j)) >= 0; j++)
901 if (a < nb_aretes_reelles && pe_arete[a] == pe)
902 tab_aretes[a] = tmp(i, j);
903
904// tab_aretes contient maintenant des valeurs correctes pour toutes les aretes reeles
905// (les items communs sont a jour). On fait encore un echange en passant par tmp pour
906// mettre a jour les items virtuels:
907
908// ******************
909// II) echange pour mettre a jour l'espace virtuel des aretes
910
911// Copier encore une fois tab_aretes dans la structure tmp
912 for (i = 0; i < nb_elem; i++)
913 for (int j = 0, a; j < nb_aretes_elem && (a = elem_aretes(i, j)) >= 0; j++)
914 tmp(i, j) = tab_aretes[a];
915
916// Echange du tableau
917 tmp.echange_espace_virtuel();
918// Recopie de tmp dans tab_aretes
919 for (i = nb_elem; i < nb_elem_tot; i++)
920 for (int j = 0, a; j < nb_aretes_elem && (a = elem_aretes(i, j)) >= 0; j++)
921 tab_aretes[a] = tmp(i, j);
922}
