109// input : a vector of coefficients organized lines by lines
110// for example, let's say that we have nbl lines and nbc columns
111// then the first nbc coefficients are coefficients associated to the first line of the matrix
112// then the next nbc coefficients are coefficients associated to the second line of the matrix, ect...

122 {
123 for(int j=0; j<nb_cols; j++)
124 {

148// this function fills the matrix morse_matrix
149void Matrice_Dense::convert_to_morse_matrix( Matrice_Morse& morse_matrix ) const
150{
151 int nb_lines = nb_lignes( );
152 int nb_cols = nb_colonnes( ) ;
153 int nnz = nb_lines * nb_cols ;
154
155 morse_matrix.dimensionner( nb_lines , nb_cols , nnz );
156
157 int size_tab1 = nb_lines + 1 ;
158 int size_tab2 = nnz ;
159 morse_matrix.get_set_tab1().resize( size_tab1 );
160 morse_matrix.get_set_tab2().resize( size_tab2 );
161 morse_matrix.get_set_coeff().resize( nnz );
162
163 for(int i=0; i<size_tab1; i++)
164 {
165 morse_matrix.get_set_tab1()( i ) = /*already registred*/i*nb_cols + /*fortran index*/ 1 ;
166 }
167 int count = 0;
168 for(int i=0 ; i<nb_lines ; i++)
169 {
170 for(int j=0; j<nb_cols; j++)
171 {
172 morse_matrix.get_set_tab2()( count ) = j+1 ;
173 morse_matrix.get_set_coeff()( count ) = Matrix_( i , j );
174 count++;
175 }
176 }
177}

211//! Operation de multiplication-accumulation (saxpy) matrice vecteur.<br>Operation: resu = resu + Matrix_ * x
212DoubleVect& Matrice_Dense::ajouter_multvect_( const DoubleVect& x , DoubleVect& resu ) const
213{
214 const int nb_lines = nb_lignes( ) ;
215 const int nb_cols = nb_colonnes( );
216 assert( x.size_array( ) == nb_cols );
217// the test is in this order because size( ) maybe invalid
218 assert( resu.size_array( ) == nb_lines || resu.size( ) == nb_lines );
219
220// for a given i such that 0 <= i <= nb_lines we have
221// resu( i ) = resu( i ) + Matrix_( i , j ) * x( j )
222// with 0 <= j <= nb_cols
223 for(int i=0; i<nb_lines; i++)
224 {
225 double old_resu = resu( i );
226 double prod_mat_vect = 0; //!< store the result of Matrix_( i , j ) * x( j ) with 0<= j <=nb_cols
227 for(int j=0; j<nb_cols; j++)
228 {
229 prod_mat_vect += Matrix_( i , j ) * x( j );
230 }
231 resu( i ) = old_resu + prod_mat_vect ;
232 }
233
234 return resu;
235}

260// return a bool which indicates if the other matrix is the same or not
261// return true : Matrix_ and other_matrix are the same for a given tolerance tol
262// return false : Matrix_ and other_matrix are NOT the same for a given tolerance tol
263bool Matrice_Dense::is_the_same( const Matrice_Dense& other_matrix , const double tol ) const
264{
265 bool same = true;
266 int nb_lines = nb_lignes( );
267 int nb_cols = nb_colonnes( );
268// check if the other has same number of lines/columns
269 if( other_matrix.nb_lignes( ) != nb_lines )
270 {
271 same = false;
272 }
273 if( other_matrix.nb_colonnes( ) != nb_cols )
274 {
275 same = false;
276 }
277// check each coefficient
278 for(int i=0; i<nb_lines; i++)
279 {
280 for(int j=0; j<nb_cols; j++)
281 {
282 if( std::fabs( other_matrix( i , j ) - Matrix_( i , j )) >= tol )
283 {
284 same = false;
285 }
286 }
287 }
288 return same;
289}

295DoubleVect& Matrice_Dense::ajouter_multvectT_(const DoubleVect& x,DoubleVect& resu) const
296{
297
298 const int nb_lines = nb_lignes( ) ;
299 const int nb_cols = nb_colonnes( );
300 assert( x.size_array( ) == nb_lines );
301// the test is in this order because size( ) maybe invalid
302 assert( resu.size_array( ) == nb_cols || resu.size( ) == nb_cols );
303
304// for a given i such that 0 <= i <= nb_cols we have
305// resu( i ) = resu( i ) + Matrix_( j , i ) * x( j )
306// with 0 <= j <= nb_lines
307 for(int i=0; i<nb_cols; i++)

311 for(int j=0; j<nb_lines; j++)

313 prod_mat_vect += Matrix_( j , i ) * x( j );
314 }
315 resu( i ) = old_resu + prod_mat_vect ;
316 }

370void Matrice_Dense::inverse()
371{
372// This method compute the inverse of the matrix.
373// It uses the LAPACK library.
374 const int nbLines = nb_lignes();
375 const int nbCols = nb_colonnes();
376 int infoerr = 0;
377 ArrOfInt ipiv(nbLines);
378 ArrOfDouble work(nbLines);
379
380 if (nbLines != nbCols)
381 {
382 Cerr << "Error in Matrice_Dense::inverse" << finl;
383 Cerr << "The matrix is not a square matrix !" << finl;
384 Process::abort( );
385 }
386
387 F77NAME(DGETRF)(&nbLines, &nbLines, Matrix_.addr(), &nbLines, ipiv.addr(), &infoerr);
388 assert(infoerr == 0);
389 F77NAME(DGETRI)(&nbLines, Matrix_.addr(), &nbLines, ipiv.addr(), work.addr(), &nbLines, &infoerr);
390 assert(infoerr == 0);
391
392 return;
393}

395void Matrice_Dense::multiplyToRight(const Matrice_Dense& B, Matrice_Dense& RES) const
396{
397// This method allows to perform the multiplication of two matrix.
398// It is the "right" multiplication : (*this) * B = RES
399
400 assert(nb_colonnes() == B.nb_lignes());
401 assert(nb_lignes() == RES.nb_lignes());
402 assert(B.nb_colonnes() == RES.nb_colonnes());
403
404 for (int i = 0; i < nb_lignes(); ++i)
405 {
406 for (int j = 0; j < B.nb_colonnes(); ++j)
407 {
408 RES.set_coefficient(i, j, 0.0);
409 for (int k = 0; k < nb_colonnes(); ++k)
410 {
411 RES.set_coefficient(i, j, RES(i, j) + Matrix_(i, k) * B(k, j));
412 }
413 }
414 }
415 return;
416}
