54void extract_coef_local(ArrOfDouble& coeff_reactifs,const Nom& reactifs_,const Motcles& list_var)
55{
56 int sz=list_var.size();
57 assert(sz==list_var.size());
58
59 Parser_U parser;
60 parser.setNbVar(sz);
61 for (int i=0; i<sz; i++)
62 parser.addVar(list_var[i]);
63
64 Nom copie(reactifs_);
65 parser.setString(copie);
66 parser.parseString();
67 coeff_reactifs.resize_array(sz);
68 for (int i=0; i<sz; i++)
69 {
70 for (int j=0; j<sz; j++)
71 parser.setVar(j,0);
72 parser.setVar(i,1);
73 double val=parser.eval();
74 coeff_reactifs[i]=val;
75 }
76 Cerr<<reactifs_<<" "<<list_var<<" "<<coeff_reactifs<<finl;
77 for (int i=0; i<sz; i++) Cerr<<coeff_reactifs[i]<<"*"<<list_var[i]<<"+";
78 Cerr<<finl;
79}

90void Reaction::completer(const Motcles& list_var,const ArrOfDouble& masse_molaire)
91{
92// test
93
94 ArrOfDouble coeff_reactifs;
95 ArrOfDouble coeff_produits;
96 extract_coef(coeff_reactifs,coeff_produits,list_var,masse_molaire);
97 int sz=list_var.size();
98 double mreactif=0,mproduit=0;
99 coeff_Y_.resize_array(sz);
100 coeff_Y_=0;
101 coeff_stoechio_.resize_array(sz);
102 coeff_stoechio_=0;
103 coeff_activite_.resize_array(sz);
104 coeff_activite_=0;
105 for (int p=0; p<sz; p++)
106 {
107 mreactif+=masse_molaire[p]*coeff_reactifs[p];
108 mproduit+=masse_molaire[p]*coeff_produits[p];
109 coeff_Y_[p]+=masse_molaire[p]*coeff_reactifs[p];
110 coeff_Y_[p]-=masse_molaire[p]*coeff_produits[p];
111 coeff_stoechio_[p]+=coeff_reactifs[p];
112 coeff_stoechio_[p]-=coeff_produits[p];

116 Cerr<<" erreur dans la chimie de "<<(*this)<<finl;

118 Cerr<<" mproduit "<< mproduit<<finl;
119 exit();
120 }
121// coeff_Y_/=mreactif;
122 save_alias_=list_var;
123 Cerr<<" Extraction des activites "<<finl;
124 extract_coef_local(coeff_activite_,activite_,list_var);

298double Reaction::calcul_proportion_implicite(ArrOfDouble& C_temp,const ArrOfDouble& C0, double deltat, double seuil, double& proportion_directe) const
299{
300
301 int nbc=C0.size_array();
302// double proportion_directe_es;
303 double proportion=-1, omegapoint, dy_omega, proportion_inverse=-1;
304
305 double rho_mel=1 ; //tab_rho(elem); //!< kg/m3
306 double T_elem=1 ; //temperature(elem); //!< K
307 omegapoint=constante_taux_reaction_; //!< 1/s // prefacteur du taux de reaction volumique local omegapoint*=pow(T_elem,beta_)*exp(-Ea_/(R_gaz_parfait*T_elem)); // Arrhenius et facteur temperature
308//
309 dy_omega=omegapoint/rho_mel; //!< m3/kg/s // prefacteur du taux de reaction volumique local proportion=dy_omega*deltat; // m3/kg a ce stade si une seule activite exposant 1
310//
311 double produit_activite=1;
312 double produit_contre=1;
313 double R_gaz_parfait=8.3143;
314
315 double securite=1-1e-6;
316
317 if (contre_reaction_>0)
318 securite=0.5;
319 ArrOfDouble& C=C_temp;
320 C=C0;
321
322 int ite=0;
323 double dmax=2*seuil;
324
325 while((dmax>seuil)&&(ite<1000))
326// while(ite<5)
327 {
328
329 dmax=0;
330 produit_activite=1;
331 produit_contre=1;
332 proportion=dy_omega*deltat;
333// calcul du taux de reaction brut
334 for (int i=0; i<nbc; i++)
335 {
336 if (coeff_Y_[i]>0.) //!< c'est un reactif gazeux => produit par son activite
337 {
338 produit_activite*=pow(C[i],coeff_activite_[i]);
339 }
340 else if (contre_reaction_>0)
341 if (coeff_Y_[i]<0.)
342 {
343 produit_contre*=pow(C[i],coeff_activite_[i]);
344 }
345 } //!< mol/kg a ce stade
346
347 assert(produit_contre>=0);
348 proportion_directe=produit_activite*proportion;
349 if (contre_reaction_>0)
350 {
351 proportion_inverse=produit_contre/(contre_reaction_*exp(-c_r_Ea_/(R_gaz_parfait*T_elem)))*proportion; //!< contre-Arrhenius
352 proportion=proportion_directe-proportion_inverse;
353 }
354 else
355 proportion=proportion_directe;
356
357// proportion*=produit_activite;
358 if (1)
359 {
360// conditionnement en temps pour ne pas consommer plus que ce qui est present
361 if (proportion>0)
362 for (int i=0; i<nbc; i++)
363 {
364 if (coeff_Y_[i]>0.) //!< c'est un reactif
365 {
366 if (proportion > C[i]/coeff_stoechio_[i]*securite) Cerr<<" on limite" <<finl;
367 proportion=std::min(proportion,C[i]/coeff_stoechio_[i]*securite);
368 }
369 }
370 else
371 for (int i=0; i<nbc; i++)
372 {
373 if (coeff_Y_[i]<0.) //!< c'est un reactif car on est dans le cas contre treaction
374 {
375// on prend le max car proportion et Y_i(elem)/coeff_Y_[i] sont negatifs
376 proportion=std::max(proportion,C[i]/coeff_stoechio_[i]*securite);
377 }
378 }
379 }
380 proportion_max_sur_delta_t_=std::max(proportion_max_sur_delta_t_,std::fabs(proportion));
381// if (ite==0)
382// proportion_directe_es=proportion_directe;
383 for (int i=0; i<nbc; i++)
384
385 if (1)
386 {
387 if (coeff_Y_[i]!=0.) //!< c'est un reactif car on est dans le cas contre treaction
388 {
389 double pond=-1;
390 if (coeff_Y_[i]>0)
391 {
392 if (C[i]>0)
393 pond=proportion_directe*coeff_stoechio_[i]/C[i];
394 else
395 pond=0;
396 }
397 else
398 {
399 if ((contre_reaction_>0)&&(C[i]>0))
400 pond=-proportion_inverse*coeff_stoechio_[i]/C[i];
401 else
402 pond=0;
403 }
404 if (pond<0)
405 exit();
406// pond=0;
407 double nc=((C0[i]-proportion*coeff_stoechio_[i]/2.+pond/2.*C[i])/(1.+pond/2.)-C[i]);
408 double dc=std::fabs(nc);
409 C[i]+=nc;
410 if (dc>dmax)
411 dmax=dc;
412// Cerr<<ite<<" i "<< i<< " "<<C(i)-C0(i)<<" dc"<< dmax <<finl;
413 }
414 }
415 else
416 {
417 double nc=C0[i]-proportion*coeff_stoechio_[i]/2.-C[i];
418 C[i]+=nc;
419 double dc=std::fabs(nc);
420 if (dc>dmax)
421 dmax=dc;
422 }
423
424
425 ite++;
426// Cerr<<ite<< " ici " <<proportion <<" "<<dmax<<" direct " <<proportion_directe<<finl;
427// dmax=0;
428
429 }
430// Cerr<<"convergence en "<<ite <<" "<<dmax<<" "<<proportion_directe<<finl;
431
432 return proportion;
433}
