177// Add M/dt into matrix
178Matrice_Base& Solveur_Masse_base::ajouter_masse(double dt, Matrice_Base& matrice, int penalisation) const
179{
180 Matrice_Morse& matmo=ref_cast(Matrice_Morse, matrice);
181 if (has_interface_blocs())
182 {
183 penalisation_flag_ = penalisation;
184 DoubleTrav inco(equation().inconnue().valeurs());
185 ajouter_blocs({{ equation().inconnue().le_nom().getString(), &matmo }}, inco, dt, {}, 0); //on tente ajouter_blocs()
186 return matrice;
187 }
188
189 DoubleTrav diag(equation().inconnue().valeurs());
190 const int sz = equation().inconnue().valeurs().dimension_tot(0) * diag.line_size();
191 diag=1.;
192 appliquer(diag); //!< M-1
193 int prems=0;
194 if(penalisation)
195 {
196 if (penalisation_==0)
197 {
198 double penal=0;
199 for(int i=0; i<sz; i++)
200 penal=std::max(penal, matmo(i,i));
201 penal=mp_max(penal);
202 penalisation_=(mp_max_vect(diag)/dt + penal)*1.e3;
203 prems=1;
204 }
205 }
206 else
207 penalisation_=0;
208 for(int i=0; i<sz; i++)
209 {
210 if (diag.addr()[i]==0)
211 {
212 if(prems)
213 {
214 matmo(i,i)+=penalisation_;
215 prems=0;
216 }
217 matmo(i,i)+=penalisation_/dt;
218 }
219 else
220 matmo(i,i)+=1./(diag.addr()[i]*dt); //!< M/dt
221 }
222 return matrice;
223}

225// Add M*y/dt to x
226DoubleTab& Solveur_Masse_base::ajouter_masse(double dt, DoubleTab& x, const DoubleTab& y, int penalisation) const
227{
228 if (has_interface_blocs())
229 {
230 penalisation_flag_ = penalisation;
231 const std::string& nom_inco = equation().inconnue().le_nom().getString();
232 ajouter_blocs({}, x, dt, {{nom_inco,y}}, 0); //on tente ajouter_blocs()
233 return x;
234 }
235
236 int sz=y.size();
237 DoubleTab diag;
238 diag.copy(equation().inconnue().valeurs(), Array_base::NOCOPY_NOINIT);
239 diag=1.;
240 appliquer(diag); //!< M-1
241 if (penalisation)
242 {
243 if (penalisation_==0)
244 penalisation_ = mp_max_vect(diag)*1.e3;
245 }
246 else
247 penalisation_=1;
248
249 for(int i=0; i<sz; i++)
250 {
251 if (diag.addr()[i]==0)
252 x.addr()[i]=penalisation_*y.addr()[i];
253 else
254 x.addr()[i]+=1./(diag.addr()[i]*dt)*y.addr()[i];
255 }
256// x.echange_espace_virtuel();Debog::verifier("Solveur_Masse::ajouter_masse",x);
257 return x;
258}
