60void Perte_Charge_Reguliere_VEF_P1NC::remplir_num_faces(Nom& un_nom_sous_zone)
61{
62 Cerr << " Perte_Charge_Reguliere_VEF_P1NC::remplir_num_faces " << finl;
63 const Domaine& le_domaine = equation().probleme().domaine();
64 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
65
66 const IntTab& elem_faces = zone_VEF.elem_faces();
67 const IntTab& face_voisins = zone_VEF.face_voisins();
68 const DoubleVect& volumes = zone_VEF.volumes();
69 const Sous_Zone& la_sous_zone = le_domaine.ss_zone(un_nom_sous_zone);
70 int nb_poly_ss_zone = la_sous_zone.nb_elem_tot();
71
72 int nfe=zone_VEF.zone().nb_faces_elem();
73 int nfac;
74 int num_poly;
75
76 num_faces.resize(nb_poly_ss_zone*4);
77 corr_front_ss.resize(nb_poly_ss_zone*4);
78
79 corr_front_ss = 1. ;
80 nfac = 0;
81
82 IntVect num_loc(zone_VEF.nb_elem_tot());
83 IntVect fait(zone_VEF.nb_faces_tot());
84 fait = -1;
85 num_loc = -1;
86
87 int num_poly_vois;
88 int num_elem,num_face, face_loc;
89
90 for (num_elem=0; num_elem<nb_poly_ss_zone; num_elem++)
91 {
92 num_loc[la_sous_zone(num_elem)] = num_elem;
93 }
94
95 for (num_elem=0; num_elem<nb_poly_ss_zone; num_elem++)
96 {
97 num_poly = la_sous_zone(num_elem);
98 for (face_loc = 0; face_loc < nfe ; face_loc++)
99 {
100 num_face = elem_faces(num_poly,face_loc);
101 if (fait(num_face) ==-1)
102 {
103 fait(num_face) = 1 ;
104 num_faces[nfac++] = num_face;
105 for (int i=0; i<2; i++ )
106 {
107 num_poly_vois = face_voisins(num_face,i);
108 if (num_poly_vois != -1)
109 if (num_loc[num_poly_vois] == -1) //!< le poly voisin n'est pas dans la sous_zone
110 {
111 corr_front_ss[nfac-1] = volumes(num_poly)/(volumes(num_poly)+volumes(num_poly_vois)) ;
112 }
113 }
114 }
115 }
116 }
117 num_faces.resize(nfac);
118 corr_front_ss.resize(nfac);
119}

121DoubleTab& Perte_Charge_Reguliere_VEF_P1NC::ajouter(DoubleTab& resu) const
122{
123// Cerr << " Perte_Charge_Reguliere_VEF_P1NC::ajouter " << finl;
124 const Zone_VEF& zone_VEF = ref_cast(Zone_VEF,equation().zone_dis().valeur());
125 const DoubleVect& volumes_entrelaces = zone_VEF.volumes_entrelaces();
126 const DoubleVect& porosite_face = zone_VEF.porosite_face();
127 const DoubleTab& vit = la_vitesse->valeurs();
128 const IntTab& face_voisins = zone_VEF.face_voisins();
129
130 double U_abs;
131 double Cf,CK,Reynolds;
132 int numfa,n0,n1;
133
134 const Champ_Don& nu = le_fluide->viscosite_cinematique();
135 const DoubleTab& visco = nu.valeurs();
136 double d_visco=-1;
137 int l_visco_unif=0;
138
139 if (sub_type(Champ_Uniforme,nu.valeur()))
140 {
141 const Champ_Uniforme& ch_nu = ref_cast(Champ_Uniforme,nu.valeur());
142 d_visco = ch_nu(0,0);
143 l_visco_unif = 1;
144 }
145
146
147 if (couronne_tube == 1)
148 {
149 Cerr << " PQ : 17/06/08 : Cette option a ete supprimee dans Perte_Charge_Reguliere_VEF_P1NC" << finl;
150 Cerr << " Possibilite de retrouver la meme fonctionnalite a l'aide du mot cle : " << finl;
151 Cerr << " Perte_charge_anisotrope (voir doc) " << finl;
152 exit();
153 }
154
155 int premiere_face_int=zone_VEF.premiere_face_int();
156 int direction = direction_perte_charge();
157 int nb_faces = num_faces.size();
158 for (int i=0; i<nb_faces; i++)
159 {
160 numfa = num_faces[i];
161
162 if (!l_visco_unif)
163 {
164 n0 = face_voisins(numfa,0);
165 n1 = face_voisins(numfa,1);
166 if (numfa < premiere_face_int)
167 {
168 if (n0 != -1)
169 d_visco = visco[n0];
170 else
171 d_visco = visco[n1];
172 }
173 else
174 d_visco = 0.5*(visco[n0]+visco[n1]);
175 }
176// GF La vitesse pour caluler le Rey ne depend pas de la porosite
177// et on utilise la norme
178 U_abs=0;
179 for (int j =0; j<dimension; j++ )
180 U_abs += vit(numfa,j)*vit(numfa,j);
181
182
183 U_abs = sqrt(U_abs) ;
184
185
186 Reynolds = U_abs*D()/d_visco;
187
188 if (!Cf_utilisateur)
189 Cf = calculer_Cf_blasius(Reynolds);
190 else
191 Cf = CF();
192
193 CK = -0.5*Cf/D();
194
195 resu(numfa,direction) += CK*U_abs*vit(numfa,direction)*volumes_entrelaces(numfa)*porosite_face(numfa)*corr_front_ss(i);
196
197 }
198
199 return resu;
200}
