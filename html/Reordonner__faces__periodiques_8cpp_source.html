27static inline void calculer_vecteur_2faces(const DoubleTab& coord,
28 const IntTab& faces,
29 const int i_face1,
30 const int i_face2,
31 ArrOfDouble& vect)
32{
33 const int nb_som_faces = faces.dimension(1);
34 const int dim = coord.dimension(1);
35 assert(vect.size_array() == dim);
36 vect = 0.;
37// Calcul du vecteur entre le centre de la face i et le centre de la face i+n
38 for (int j = 0; j < nb_som_faces; j++)
39 {
40 const int sommet1 = faces(i_face1, j);
41 const int sommet2 = faces(i_face2, j);
42 for (int compo = 0; compo < dim; compo++)
43 vect[compo] += coord(sommet2, compo) - coord(sommet1, compo);
44 }
45 vect /= nb_som_faces;
46}

49static double local_norme_vect(const DoubleVect& dv)
50{
51 double x=0.0;
52 for(int i=0; i< dv.size_reelle(); i++)
53 x += dv(i)*dv(i);
54 x = sqrt(x);
55 return x;
56}

58void Reordonner_faces_periodiques::chercher_direction_perio(ArrOfDouble& direction_perio, const Domaine& dom, const Nom& bord)
59{
60 const DoubleTab& sommets = dom.coord_sommets();
61 const int dim = sommets.dimension(1);
62 direction_perio.resize_array(dim);
63 direction_perio = 0.;
64 const Frontiere& front = dom.zone(0).frontiere(bord);
65 const int nb_faces = front.nb_faces();
66 if (nb_faces == 0)
67 return;
68 const IntTab& faces = front.faces().les_sommets();
69 const int nb_som_face = faces.dimension(1);
70 DoubleTab normale(1, dim);
71 IntTab une_face(1, nb_som_face);
72 int i;
73 for (i = 0; i < nb_som_face; i++)
74 une_face(0, i) = faces(0, i);
75 dom.zone(0).type_elem().calculer_normales(une_face, normale);
76 normale /= local_norme_vect(normale);
77
78 ArrOfDouble delta(nb_faces);
79 ArrOfDouble vect(dim);
80 for (i = 1; i < nb_faces; i++)
81 {
82 calculer_vecteur_2faces(sommets, faces, 0, i, vect);
83 double x = 0.;
84 for (int j = 0; j < dim; j++)
85 x += vect[j] * normale(0,j);
86 delta[i] = x;
87 }
88 const double min = min_array(delta);
89 const double max = max_array(delta);
90 double facteur = (std::fabs(min) > std::fabs(max)) ? min : max;
91 for (i = 0; i < dim; i++)
92 direction_perio[i] = normale(0, i) * facteur;
93 Cerr << "Periodicity direction for " << dom.le_nom() << "/" << bord << " " << direction_perio;
94}

106//! Reordonne le tableau "faces" selon la convention des faces periodiques:<br>D'abord les faces d'une extremite, puis dans le meme ordre, les faces jumelles.<br>Attention, l'algorithme est en n carre (lent), et ne fonctionne qu'en sequentiel.<br>Parametre: zone<br>Signification: la zone a laquelle appartiennent les faces<br>Parametre: direction_perio<br>Signification: le vecteur qui separe le centre d'une face au centre de la face opposee<br>Parametre: faces<br>Signification: le tableau des faces (pour chaque face, indices de ses sommets) a reordonner<br>Valeur de retour: 1 si ok, 0 si on n'a pas trouve de face jumelle a une face a precision_geom pres.
107int Reordonner_faces_periodiques::reordonner_faces_periodiques(const Domaine& domaine,
108 IntTab& faces,
109 const ArrOfDouble& direction_perio,
110 const double epsilon)
111{
112// Modif B.M. 04/06/2010: j'autorise l'operation en parallele car c'est utilise par
113// l'interprete MaillerParallel...
114// PL 18/11/2010: Je deplace neanmoins l'interdiction de l'utilisation de l'interprete en // dans le jeu de donnees (voir ::interpreter_)
115 const int nb_faces = faces.dimension(0);
116 const int nb_som_faces = faces.dimension(1);
117 const int dim = domaine.les_sommets().dimension(1);
118// Calcul des coordonnees des centres des faces:
119 DoubleTab centres(nb_faces, 3);
120 {
121 const DoubleTab& coord = domaine.les_sommets();
122 const double inv_nb_som = 1. / (double) nb_som_faces;
123 for (int i = 0; i < nb_faces; i++)
124 {
125 for (int j = 0; j < nb_som_faces; j++)
126 {
127 const int sommet = faces(i, j);
128 for (int k = 0; k < dim; k++)
129 centres(i, k) += coord(sommet, k) * inv_nb_som;
130 }
131 }
132 }
133
134// Construction d'un octree contenant les centres des faces:
135 Octree_Double octree;
136 octree.build_nodes(centres, 0 /* do not include virtual nodes */);
137
138// Pour chaque face, on cherche sa face periodique associee (dont le centre
139// est decale de direction_perio).
140
141// Pour chaque face, son nouvel indice dans le tableau des faces
142 ArrOfInt renum_faces(nb_faces);
143 renum_faces= -1;
144 ArrOfInt nodes_list;
145 nodes_list.set_smart_resize(1);
146 ArrOfDouble coord(dim);
147 int count = 0;
148 for (int i_face = 0; i_face < nb_faces; i_face++)
149 {
150 if (renum_faces[i_face] >= 0)
151 continue; //!< Face deja traitee, on passe Cherche la face opposee dans les deux directions (-1. et +1.)
152//
153 double facteur;
154 int i_face2 = -1;
155 for (facteur = -1.; facteur < 1.5; facteur += 2.)
156 {
157 for (int i = 0; i < dim; i++)
158 coord[i] = centres(i_face, i) + facteur * direction_perio[i];
159 octree.search_elements_box(coord, epsilon, nodes_list);
160 i_face2 = octree.search_nodes_close_to(coord, centres, nodes_list, epsilon);
161 if (i_face2 >= 0)
162 break;
163 }
164 if (i_face2 >= 0)
165 {
166 if (renum_faces[i_face2] >= 0)
167 {
168 Cerr << "====================================================" << finl;
169 Cerr << "Error in reordonner_faces_periodiques: the face " << i_face
170 << " of " << centres(i_face,0) << " " << centres(i_face,1) << " "
171 << ((dim==3)?centres(i_face,2):0.)
172 << " center already has a face twin."
173 << finl;
174 Cerr << "Possible problem: the boundary is not periodic. Check your mesh." << finl;
175 return 0;
176 }
177 int f0 = (facteur > 0.) ? i_face : i_face2;
178 int f1 = (facteur > 0.) ? i_face2: i_face;
179 renum_faces[f0] = count;
180 renum_faces[f1] = count + nb_faces / 2;
181 count++;
182 }
183 else
184 {
185 Cerr << "====================================================" << finl;
186 Cerr << "Error in reordonner_faces_periodiques: the face " << i_face << " of " << centres(i_face,0) << " " << centres(i_face,1) << " "
187 << ((dim==3)?centres(i_face,2):0.) << finl;
188 Cerr << "center has no face twin into the specified direction in the list of faces." << finl;
189 return 0;
190 }
191 }
192// Reordonner les faces:
193 const IntTab oldfaces(faces);
194 for (int i = 0; i < nb_faces; i++)
195 {
196 const int new_i = renum_faces[i];
197 for (int j = 0; j < nb_som_faces; j++)
198 faces(new_i, j) = oldfaces(i, j);
199 }
200 return 1;
201}

208//! essaie de verifier si les faces du bord num_bord sont ordonnees suivant la convention<br>des faces periodiques. On stocke dans vecteur_delta la direction de periodicite presumee (intervalle<br>mesure entre la premiere face et la face jumelle), et erreur le max de l'erreur par rapport a cette<br>mesure pour les autres faces.<br>En parallele, l'erreur est le max sur tous les processeurs<br>Valeur de retour: 1 si ok, 0 si l'erreur est superieure a precision_geom.
209int Reordonner_faces_periodiques::check_faces_periodiques(const Frontiere& frontiere,
210 ArrOfDouble& vecteur_delta,
211 ArrOfDouble& erreur,
212 const int verbose)
213{
214 const int dim = Objet_U::dimension;
215 vecteur_delta.resize_array(dim);
216 vecteur_delta = 0.;
217 erreur.resize_array(dim);
218 erreur = 0.;
219
220 if (verbose && Process::je_suis_maitre())
221 {
222 Cerr << "Check periodic faces to the boundary : " << frontiere.le_nom() << finl;
223 }
224 const IntTab& faces = frontiere.faces().les_sommets();
225 const int nb_faces = faces.dimension(0);
226 if (nb_faces % 2 != 0)
227 {
228 Cerr << "Error in Check_faces_periodiques to the boundary " << frontiere.le_nom()
229 << "\n The number of faces is odd : " << nb_faces << finl;
230 Process::exit();
231 }
232 const int n = nb_faces / 2;
233 const DoubleTab coord = frontiere.zone().domaine().les_sommets();
234
235 int i;
236// Calculer un vecteur delta (tous les procs n'ont pas forcement des faces de ce bord)
237 vecteur_delta = -1.e37;
238 if (n > 0)
239 calculer_vecteur_2faces(coord, faces, 0, n, vecteur_delta);
240 for (i = 0; i < dim; i++)
241 vecteur_delta[i] = Process::mp_max(vecteur_delta[i]);
242
243// Calculer pour chaque face l'erreur par rapport a ce vecteur delta.
244 ArrOfDouble vect(dim);
245 for (i = 0; i < n; i++)
246 {
247 calculer_vecteur_2faces(coord, faces, i, i+n, vect);
248// Calcul de la difference entre vect et vecteur_delta:
249 for (int compo = 0; compo < dim; compo++)
250 erreur[compo] = std::max(erreur[compo], std::fabs(vecteur_delta[compo] - vect[compo]));
251 }
252 double maxerr = 0.;
253// Calcul du max sur tous les procs:
254 for (i = 0; i < dim; i++)
255 {
256 erreur[i] = Process::mp_max(erreur[i]);
257 maxerr = std::max(maxerr, erreur[i]);
258 }
259
260 if (verbose && Process::je_suis_maitre())
261 {
262 Cerr << " Delta vector = [ ";
263 for (i = 0; i < dim; i++) Cerr << vecteur_delta[i] << " ";
264 Cerr << "] error = [ ";
265 for (i = 0; i < dim; i++) Cerr << erreur[i] << " ";
266 Cerr << "]" << finl;
267 }
268 if (!(maxerr < Objet_U::precision_geom))
269 {
270 if (Process::je_suis_maitre())
271 {
272 Cerr << " This boundary is not detected as periodic (geometric error > precision_geom)" << finl;
273 message();
274 if (Process::nproc()>1) Cerr << "Or you forgot to define the periodic boundary in the Decouper keyword." << finl;
275 } //!< attendre qu'on ait ecrit pour continuer (sinon risque de exit() avant d'avoir affiche le message)
276 Process::barrier();
277 return 0;
278 }
279 return 1;
280}

282void Reordonner_faces_periodiques::renum_som_perio(const Domaine& domaine,
283 const Noms& liste_bords_periodiques,
284 ArrOfInt& renum_som_perio,
285 const int calculer_espace_virtuel)
286{
287 int i;
288 const int nb_som = domaine.nb_som();
289 IntTab renum(nb_som);
290 for (i = 0; i < nb_som; i++)
291 renum[i] = renum_som_perio[i];
292
293 const DoubleTab coord = domaine.coord_sommets();
294 const int dim = coord.dimension(1);
295
296// Etape 1: pour chaque sommet reel, trouver un sommet associe (si plusieurs directions
297// de periodicite, un sommet peut etre associe a plusieurs autres).
298
299 const int nb_bords_perio = liste_bords_periodiques.size();
300 for (int i_bord = 0; i_bord < nb_bords_perio; i_bord++)
301 {
302 const Nom& nom_bord = liste_bords_periodiques[i_bord];
303 const Frontiere& front = domaine.zone(0).bord(nom_bord);
304// Direction periodique de ce bord:
305 ArrOfDouble delta;
306 ArrOfDouble erreur;
307 if (!check_faces_periodiques(front, delta, erreur, 1 /* verbose */))
308 exit();
309// Tableau pointant vers tous les sommets de toutes les faces
310// (on cast le IntTab en ArrOfInt)
311 const IntTab& faces_sommets = front.les_sommets_des_faces();
312 const int nb_som_face = faces_sommets.dimension(1);
313 const int nb_faces = faces_sommets.dimension(0) / 2;
314// Boucle sur les faces d'un cote du domaine (premiere moitie des faces)
315 for (int i_face = 0; i_face < nb_faces; i_face++)
316 {
317
318 for (int i_som = 0; i_som < nb_som_face; i_som++)
319 {
320 const int sommet = faces_sommets(i_face, i_som);
321
322// Trouver le sommet associe
323// Comme les frontieres sont ordonnees (voir check_faces_periodiques),
324// le sommet est forcement un des sommets de la face opposee.
325// Le vecteur qui va de "sommet" a "sommet_oppose" doit etre egal a "delta"
326// la direction de periodicite.
327 const int i_face_opposee = i_face + nb_faces;
328 int sommet_opp = -1;
329 int i_som_opp;
330 for (i_som_opp = 0; i_som_opp < nb_som_face; i_som_opp++)
331 {
332 sommet_opp = faces_sommets(i_face_opposee, i_som_opp);
333 for (i = 0; i < dim; i++)
334 {
335 double epsilon = std::fabs((coord(sommet_opp, i) - coord(sommet, i)) - delta[i]);
336 if (epsilon > Objet_U::precision_geom)
337 break;
338 }
339 if (i == dim)
340// On a trouve le sommet oppose
341 break;
342 }
343 if (i_som_opp >= nb_som_face)
344 {
345 Cerr << "[PE" << Process::me() << "] Error in Reordonner_faces_periodiques::renum_som_perio\n"
346 << " An opposite node has not been found\n"
347 << " Boundary " << nom_bord << "\n Face 1: " << i_face << "\n Node: " << sommet << finl;
348 Cerr << " May be you should define the periodic boundary " << nom_bord << finl;
349 message();
350 exit();
351 }
352 renum[sommet_opp] = sommet;
353 }
354 }
355 }
356
357// Deuxieme etape: faire pointer tous les sommets periodiques lies entre eux vers le meme sommet
358
359 for (i = 0; i < nb_som; i++)
360 {
361 int j = renum[i];
362// Parcourir les sommet relies pour cette chaine:
363 while (j != renum[j])
364 j = renum[j];
365 renum[i] = j;
366 }
367// Calcul des valeurs pour les sommets virtuels.
368// Les sommets opposes aux sommets virtuels doivent etre connus, donc erreurs fatales.
369 if (calculer_espace_virtuel)
370 {
371 const MD_Vector& md_sommets = domaine.les_sommets().get_md_vector();
372 Scatter::construire_espace_virtuel_traduction(md_sommets, md_sommets, renum, 1 /* erreurs fatales */);
373 }
374
375// Recopie du resultat dans le tableau renum_som_perio
376 assert(renum.dimension_tot(0) == domaine.nb_som_tot());
377 renum_som_perio = renum;
378}
