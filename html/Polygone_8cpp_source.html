57void Polygone::rebuild_index()
58{
59
60 const IntTab& les_elems = ma_zone.valeur().les_elems();
61 int nb_elem=les_elems.dimension_tot(0);
62 ArrOfInt PolygonIndex_OK(nb_elem+1);
63 PolygonIndex_OK[0]=0;
64 for (int ele=0; ele<nb_elem; ele++)
65 {
66 int nbf=get_nb_som_elem_max();
67 while (les_elems(ele,nbf-1)<0)
68 nbf--;
69 PolygonIndex_OK[ele+1]= PolygonIndex_OK[ele]+nbf;
70 }
71 ArrOfInt FacesIndex_OK(PolygonIndex_OK[nb_elem]);
72 int f=0;
73 for (int ele=0; ele<nb_elem; ele++)
74 {
75 for (int ss=0; ss<PolygonIndex_OK[ele+1]-PolygonIndex_OK[ele]; ss++)
76
77 FacesIndex_OK[f++]= les_elems(ele,ss);
78 }
79 assert(f==PolygonIndex_OK[nb_elem]);
80
81 FacesIndex_=FacesIndex_OK;
82 PolygonIndex_=PolygonIndex_OK;
83}

176int contient_triangle(const ArrOfDouble& pos,int som0,int som1,int som2,const DoubleTab& coord)
177{
178
179 double prod,p0,p1,p2;
180
181// Il faut donc determiner le sens (trigo ou anti trigo) pour la numerotation :
182// Calcul de prod = 01 vectoriel 02 selon z
183// prod > 0 : sens trigo
184// prod < 0 : sens anti trigo
185 prod = (coord(som1,0)-coord(som0,0))*(coord(som2,1)-coord(som0,1))
186 - (coord(som1,1)-coord(som0,1))*(coord(som2,0)-coord(som0,0));
187 double signe;
188 if (prod >= 0)
189 signe = 1;
190 else
191 signe = -1;
192// Calcul de p0 = 0M vectoriel 1M selon z
193 p0 = (pos[0]-coord(som0,0))*(pos[1]-coord(som1,1))
194 - (pos[1]-coord(som0,1))*(pos[0]-coord(som1,0));
195 p0 *= signe;
196// Calcul de p1 = 1M vectoriel 2M selon z
197 p1 = (pos[0]-coord(som1,0))*(pos[1]-coord(som2,1))
198 - (pos[1]-coord(som1,1))*(pos[0]-coord(som2,0));
199 p1 *= signe;
200// Calcul de p2 = 2M vectoriel 0M selon z
201 p2 = (pos[0]-coord(som2,0))*(pos[1]-coord(som0,1))
202 - (pos[1]-coord(som2,1))*(pos[0]-coord(som0,0));
203 p2 *= signe;
204 double epsilon=std::fabs(prod)*Objet_U::precision_geom;
205 if ((p0>-epsilon) && (p1>-epsilon) && (p2>-epsilon))
206 return 1;
207 else
208 return 0;
209}
