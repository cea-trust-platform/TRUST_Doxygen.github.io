82DoubleTab& Op_Curl_VEFP1B::ajouter(const DoubleTab& vitesse,
83 DoubleTab& curl) const
84{
85 Cerr << "Je suis dans operateur curl" << finl;
86
87 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
88 const Zone& zone = zone_VEF.zone();
89// int prems=zone_VEF.premiere_face_int();
90 if ( dimension != 2 )
91 {
92 Cerr << "Pour l'instant seule la 2D est etudiee. " << finl;
93 Process::exit();
94 }
95
96 int face0=0,face1=0,face2=0;
97 int numero_triangle=0;
98 int face_globale=0,face_opp=0;
99
100 DoubleTab vecteur_normal0(dimension);
101 DoubleTab vecteur_normal1(dimension);
102 DoubleTab vecteur_normal2(dimension);
103
104// On traite les faces internes i.e. sans conditions aux limites
105// ATTENTION: la base de la vorticite c'est: l'ensemble des
106// fonctions indicatrices des elements + l'ensemble des fonctions
107// chapeaux du P1 moins la derniere de ces fonctions de forme
108
109// Partie P0 de la vorticite
110 for (int numero_elem = 0; numero_elem < zone.nb_elem(); numero_elem++)
111 {
112
113// REM: on peut generaliser cette partie a la 3D en
114// faisant une boucle avec zone.nb_faces_element()
115
116// Il nous faut d'abord les numeros des 3 faces
117// qui appartiennent a cet element K
118 face0 = zone_VEF.elem_faces(numero_elem,0);
119 face1 = zone_VEF.elem_faces(numero_elem,1);
120 face2 = zone_VEF.elem_faces(numero_elem,2);
121
122// Ensuite, il nous faut les vecteurs tangents de
123// ces trois faces.
124 vecteur_normal0 = vecteur_normal(face0,numero_elem);
125 vecteur_normal1 = vecteur_normal(face1,numero_elem);
126 vecteur_normal2 = vecteur_normal(face2,numero_elem);
127
128 int modulo;
129 for (int composante=0; composante<dimension; composante++)
130 {
131// La partie P0 a ete teste avec les fonctions (1,0);
132// (0,1);(x,0) et (0,x).
133// Tous les resultats sont corrects
134
135 modulo = (composante+1)%2;
136 curl(numero_elem) += pow(-1.,modulo) * (
137 vitesse(face0,composante)*vecteur_normal0(modulo)+
138 vitesse(face1,composante)*vecteur_normal1(modulo)+
139 vitesse(face2,composante)*vecteur_normal2(modulo) );
140
141 }
142
143 Cerr << "Element curl(" << numero_elem << ") " << curl(numero_elem) << finl;
144 }
145
146// Partie P1 de la vorticite
147
148 for (int numero_som = 0; numero_som < zone.nb_som()-1; numero_som++)
149 {
150 for (int num_loc_elem = 0; num_loc_elem < elem_som_size(numero_som);
151 num_loc_elem++)
152
153 {
154// for num_loc_elem
155
156// On recupere le numero global du triangle
157 numero_triangle = elements_pour_sommet(numero_som,num_loc_elem);
158
159// On recupere le numero global de la face opposee a "numero_som"
160// dans le triangle "numero_triangle"
161 face_opp=zone_VEF.numero_sommet_local(numero_som,numero_triangle);
162 face_opp=zone_VEF.elem_faces(numero_triangle,face_opp);
163
164// On recupere le vecteur normal de cette face opposee.
165 vecteur_normal1=vecteur_normal(face_opp,numero_triangle);
166
167 for (int num_loc_face = 0 ; num_loc_face < zone.nb_faces_elem();
168 num_loc_face++)
169
170 {
171// for num_loc_face
172
173// On recupere le numero global de la face "num_loc_face"
174 face_globale=zone_VEF.elem_faces(numero_triangle,num_loc_face);
175
176// //Si "face_globale" est une arete interne alors on effectue le bon
177// //traitement
178// if (face_globale >= zone_VEF.premiere_face_int() )
179 {
180// On calcule les vecteurs normaux associes a ces faces.
181 vecteur_normal0=vecteur_normal(face_globale,numero_triangle);
182
183// Enfin on calcule la contribution au curl de chacune de
184// ces faces pour chacun des 2 triangles.
185 int modulo;
186 for (int composante=0; composante<dimension; composante++)
187 {
188// Partie P1 teste avec les fonctions (1,0);(0,1)
189// (x,0) et (0,x).
190// Tous les tests sont corrects
191
192 modulo = (composante+1)%2;
193
194// Partie (lambda_s,curl u)
195 curl(zone.nb_elem()+numero_som) += -pow(-1.,modulo) *
196 1./(dimension+1) *
197 vitesse(face_globale,composante) *
198 vecteur_normal0(modulo);
199
200// Partie (rot lambda_s, u)
201 curl(zone.nb_elem()+numero_som) += pow(-1.,modulo)*
202 1./(dimension*(dimension+1))*
203 vitesse(face_globale,composante)*
204 vecteur_normal1(modulo);
205 }
206
207 } //!< fin du if
208
209 }//!< fin du for num_loc_face
210
211 /* Pour le moment, on ne travaille que sur des vitesse H10 */
212 /* C'est le travail de these */
213 /* Par consequent, inutile de traiter les faces du bord */
214
215 }//!< fin du for num_loc_elem
216
217 Cerr << "Sommet curl(" << numero_som << ") " << curl(zone.nb_elem()+numero_som) << finl;
218
219 }//!< fin du for sur les sommets
220
221 Cerr << "je sors de OpCurl" << finl;
222
223 return curl;
224}

226DoubleTab
227Op_Curl_VEFP1B::vecteur_normal(const int face, const int elem) const
228{
229 assert(dimension == 2);
230
231 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
232 DoubleTab le_vecteur_normal(dimension);
233
234 for (int composante = 0; composante<dimension; composante++)
235
236 le_vecteur_normal(composante) = zone_VEF.face_normales(face,composante)
237 * zone_VEF.oriente_normale(face,elem);
238
239 return le_vecteur_normal;
240}

243// Tableau qui stocke a la place "i", tous les elements
244// du maillage qui contiennent le sommet de numero global "i"
245int Op_Curl_VEFP1B::elements_pour_sommet()
246{
247 const Zone& zone = la_zone_vef.valeur().zone();
248 int numero_global_som;
249 elements_pour_sommet_.dimensionner(zone.nb_som());
250
251 for (int numero_elem = 0; numero_elem < zone.nb_elem(); numero_elem++)
252 for (int numero_som_loc=0; numero_som_loc < zone.nb_som_elem(); numero_som_loc++)
253 {
254 numero_global_som = zone.sommet_elem(numero_elem,numero_som_loc);
255 elements_pour_sommet_[numero_global_som].add_if_not(numero_elem);
256 }
257
258 return 1;
259}
