19// XD runge_kutta_rationnel_ordre_2 schema_temps_base runge_kutta_rationnel_ordre_2 -1 This is the Runge-Kutta rational scheme of second order. The method is described in the note: Wambeck - Rational Runge-Kutta methods for solving systems of ordinary differential equations, at the link: https://link.springer.com/article/10.1007/BF02252381. Although rational methods require more computational work than linear ones, they can have some other properties, such as a stable behaviour with explicitness, which make them preferable. The CFD application of this RRK2 scheme is described in the note: https://link.springer.com/content/pdf/10.1007\%2F3-540-13917-6_112.pdf.

37//! Effectue un pas de temps de Runge Kutta rationnel d'ordre 2, sur l'equation passee en parametre.<br>Le schema de Runge Kutta rationel d'ordre 2:<br>g1=hf(y0)<br>g2=hf(y0+c2g1)<br>y1=y0+(g1g1)/(b1g1+b2g2)<br>ou ab/d = (a(b,d)+b(d,a)-d(a,b)/(d,d)<br>y1=y0+(2g1(g1,b1g1+b2g2)-(b1g1+b2g2)(g1,g1)/(b1g1+b2g2,b1g1+b2g2)<br>y1=y0+(2g1(g1,"g2")-("g2")(g1,g1)/("g2","g2")<br>ordre2 si b2c2=-1/2<br>b2c2<=-1/2 A0 stabilite et I stabilite<br>b2c2<= 1/(2cos(alpha)(2-cos(alpha))) O<=alpha<pi/2 Aalpha stabilite
38int RRK2::faire_un_pas_de_temps_eqn_base(Equation_base& eqn)
39{
40// Warning sur les 100 premiers pas de temps si facsec est egal a 1 pour faire reflechir l'utilisateur
41 if (nb_pas_dt() >= 0 && nb_pas_dt() <= NW && facsec_ == 1) print_warning(NW);
42
43 const double b1 = 2.0, b2 = -1, c2 = 0.5;
44
45 DoubleTab& present = eqn.inconnue().valeurs(), &futur = eqn.inconnue().futur();
46
47// g1=futur=f(y0)
48 DoubleTab g1(present), g2(present); //!< just for initializing the array structure ...
49
50// sauv=y0
51 DoubleTrav sauv(present);
52 sauv = present;
53
54 eqn.derivee_en_temps_inco(g1);
55
56// g1=hf(y0)
57 g1 *= dt_;
58
59// present=y0+c2g1
60 present.ajoute(c2, g1);
61
62// g2=futur=f(y0+c2g1)
63 eqn.derivee_en_temps_inco(g2);
64
65// g2=b2"g2"
66 g2 *= (b2 * dt_);
67
68// g2=b2"g2" + b1g1
69 g2.ajoute(b1, g1);
70
71// normeg2=("g2","g2")
72 double normeg2 = mp_carre_norme_vect(g2) + DMINFLOAT;
73// normeg1=-(g1,g1)
74 double normeg1 = -mp_carre_norme_vect(g1);
75// psc1=2(g1,"g2")
76 double psc1 = 2.0 * mp_prodscal(g1, g2);
77
78// y1=y0+(2g1(g1,"g2")-("g2")(g1,g1)/("g2","g2")
79 futur = g1;
80 futur *= psc1;
81 futur.ajoute(normeg1, g2);
82 futur /= normeg2;
83 present = sauv;
84 futur /= dt_;
85 update_critere_statio(futur, eqn);
86 futur *= dt_;
87 futur += sauv;
88 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(), temps_courant() + pas_de_temps());
89 futur.echange_espace_virtuel();
90
91 return 1;
92}
