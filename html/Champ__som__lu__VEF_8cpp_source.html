146double coord_barycentrique3D(const IntTab& polys, const DoubleTab& coord, double x, double y, double z, int le_poly, int i)
147{
148 int som0, som1, som2, som3;
149
150 switch(i)
151 {
152 case 0:
153 {
154 som0 = polys(le_poly, 0);
155 som1 = polys(le_poly, 1);
156 som2 = polys(le_poly, 2);
157 som3 = polys(le_poly, 3);
158 break;
159 }
160 case 1:
161 {
162 som0 = polys(le_poly, 1);
163 som1 = polys(le_poly, 2);
164 som2 = polys(le_poly, 3);
165 som3 = polys(le_poly, 0);
166 break;
167 }
168 case 2:
169 {
170 som0 = polys(le_poly, 2);
171 som1 = polys(le_poly, 3);
172 som2 = polys(le_poly, 0);
173 som3 = polys(le_poly, 1);
174 break;
175 }
176 case 3:
177 {
178 som0 = polys(le_poly, 3);
179 som1 = polys(le_poly, 0);
180 som2 = polys(le_poly, 1);
181 som3 = polys(le_poly, 2);
182 break;
183 }
184 default:
185 {
186 som0 = som1 = som2 = som3 = -1;
187 assert(0);
188 Process::exit();
189 break;
190 }
191 }
192
193 double xp = (coord(som2, 1) - coord(som1, 1)) * (coord(som0, 2) - coord(som1, 2)) - (coord(som2, 2) - coord(som1, 2)) * (coord(som0, 1) - coord(som1, 1));
194 double yp = (coord(som2, 2) - coord(som1, 2)) * (coord(som0, 0) - coord(som1, 0)) - (coord(som2, 0) - coord(som1, 0)) * (coord(som0, 2) - coord(som1, 2));
195 double zp = (coord(som2, 0) - coord(som1, 0)) * (coord(som0, 1) - coord(som1, 1)) - (coord(som2, 1) - coord(som1, 1)) * (coord(som0, 0) - coord(som1, 0));
196 double den = xp * (coord(som3, 0) - coord(som1, 0)) + yp * (coord(som3, 1) - coord(som1, 1)) + zp * (coord(som3, 2) - coord(som1, 2));
197
198 xp = (coord(som2, 1) - coord(som1, 1)) * (z - coord(som1, 2)) - (coord(som2, 2) - coord(som1, 2)) * (y - coord(som1, 1));
199 yp = (coord(som2, 2) - coord(som1, 2)) * (x - coord(som1, 0)) - (coord(som2, 0) - coord(som1, 0)) * (z - coord(som1, 2));
200 zp = (coord(som2, 0) - coord(som1, 0)) * (y - coord(som1, 1)) - (coord(som2, 1) - coord(som1, 1)) * (x - coord(som1, 0));
201 double num = xp * (coord(som3, 0) - coord(som1, 0)) + yp * (coord(som3, 1) - coord(som1, 1)) + zp * (coord(som3, 2) - coord(som1, 2));
202
203 assert(den != 0.);
204 double coord_bary = num / den;
205 assert(sup_ou_egal(coord_bary, 0) && inf_ou_egal(coord_bary, 1));
206 return coord_bary;
207}

209double coord_barycentrique2D(const IntTab& polys, const DoubleTab& coord, double x, double y, int le_poly, int i)
210{
211 int som0 = polys(le_poly, i), som1 = polys(le_poly, 1);
212
213 if (i == 1) som1 = polys(le_poly, 0);
214
215 double xref = coord(som1, 0) - coord(som0, 0);
216 double yref = coord(som1, 1) - coord(som0, 1);
217
218 double xp = x - coord(som0, 0);
219 double yp = y - coord(som0, 1);
220
221 double pscal = (xref * xp) + (yref * yp);
222 double norm = (xref * xref) + (yref * yref);
223
224 assert(norm != 0.);
225 double coord_bary = pscal / norm;
226 assert(sup_ou_egal(coord_bary, 0) && inf_ou_egal(coord_bary, 1));
227 return coord_bary;
228}
