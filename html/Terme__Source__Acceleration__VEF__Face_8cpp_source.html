59//! Fonction outil pour Terme_Source_Acceleration_VEF_Face::ajouter<br>Ajout des contributions d'une liste contigue de faces du terme source de translation:<br>s_face = terme_source * rho<br>resu += integrale (s_face) sur le volume de controle de la vitesse.<br>On traite les cas suivants:<br>rho = reference nulle (=> rho = 1.) sinon rho != nul<br>faces de bord => sortie libre<br>periodicite<br>symetrie car en VEF la vitesse sur la face de bord peut ne pas etre nul (V_tangentielle)<br>faces_internes
60static void TSAVEF_ajouter_liste_faces(const int premiere_face, const int derniere_face,
61 const DoubleVect& volumes_entrelaces,
62 const DoubleVect& volumes_elements,
63 const DoubleVect& porosite_surf,
64 const IntTab& face_voisins,
65 const REF(Champ_base) & ref_rho,
66 const DoubleTab& terme_source,
67 DoubleTab& s_face,
68 DoubleTab& resu)
69{
70 int num_face;
71// Pointeur constant sur tableau constant.
72// Pointeur nul si ref_rho_ est une reference nulle.
73 const DoubleTab * const rho_elem =
74 (ref_rho.non_nul()) ? &(ref_rho.valeur().valeurs()) : 0;
75 const int dim = Objet_U::dimension;
76
77 for (num_face=premiere_face; num_face<derniere_face; num_face++)
78 {
79 const double vol = volumes_entrelaces(num_face)*porosite_surf(num_face);
80 double src[3] = {0., 0., 0.};
81 int j;
82
83 for (j = 0; j < dim; j++)
84 src[j] = terme_source(num_face, j);
85
86 double rho = 1.;
87
88// Calcul d'un rho moyen sur le volume de controle de la vitesse
89 if (rho_elem)
90 {
91 const int elem0 = face_voisins(num_face,0);
92 const int elem1 = face_voisins(num_face,1);
93 double rho0 = 0, rho1 = 0, vol0 = 0, vol1 = 0;
94 if (elem0 >= 0)
95 {
96 rho0 = (*rho_elem)(elem0);
97 vol0 = volumes_elements(elem0);
98 }
99 if (elem1 >= 0)
100 {
101 rho1 = (*rho_elem)(elem1);
102 vol1 = volumes_elements(elem1);
103 }
104 rho = (rho0 * vol0 + rho1 * vol1) / (vol0 + vol1);
105 }
106
107 for (j = 0; j < dim; j++)
108 {
109 double a = src[j] * rho;
110 s_face(num_face, j) = a;
111// Integrale sur le volume de controle :
112 resu(num_face, j) += a * vol;
113 }
114 }
115}

128//! Ajoute le terme (la_source_ * rho * volume_entrelace) au champ resu.<br>On suppose que resu est discretise comme la vitesse.<br>L'espace virtuel de "resu" n'est PAS mis a jour !<br>Commentaire sur le schema: L'acceleration d/dt(v) est calculee aux elements,<br>puis multipliee par "rho" aux elements, puis evaluee aux faces par une<br>moyenne sur les elements voisins de la face. C'est un premier essai, pas<br>forcement la meilleure idee. Comme l'acceleration depend de la vitesse qui<br>est aux faces, on passe par deux interpolations successives.<br>Effet de bord:<br>On met (la_source_ * rho) dans terme_source_post_
129DoubleTab& Terme_Source_Acceleration_VEF_Face::ajouter(DoubleTab& resu) const
130{
131 const Zone_VF& zone = la_zone_VEF_.valeur();
132 const Zone_Cl_dis_base& zone_Cl = la_zone_Cl_VEF_.valeur();
133 const IntTab& face_voisins = zone.face_voisins();
134 const DoubleVect& porosite_surf = zone.porosite_face();
135 const DoubleVect& volumes_entrelaces = zone.volumes_entrelaces();
136
137 DoubleTab& s_face = get_set_terme_source_post().valeur().valeurs();
138 s_face = 0.;
139
140// Calcul de la_source_ en fonction des champs d'acceleration et de la
141// vitesse du fluide.
142 const int dim = Objet_U::dimension;
143 const int nb_faces = resu.dimension(0);
144 DoubleTab acceleration_aux_faces(nb_faces, dim);
145 calculer_la_source(acceleration_aux_faces);
146
147// Boucle sur les conditions limites pour traiter les faces de bord
148
149 for (int n_bord = 0; n_bord < zone.nb_front_Cl(); n_bord++)
150 {
151// pour chaque Condition Limite on regarde son type
152// Si face de Dirichlet on ne fait rien
153// Si face de Neumann, Periodique ou de Symetrie on calcule la contribution au terme source
154 const Cond_lim& la_cl = zone_Cl.les_conditions_limites(n_bord);
155 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
156 const int ndeb = le_bord.num_premiere_face();
157 const int nfin = ndeb + le_bord.nb_faces();
158
159 TSAVEF_ajouter_liste_faces(ndeb, nfin,
160 volumes_entrelaces,
161 zone.volumes(),
162 porosite_surf,
163 face_voisins,
164 ref_rho_,
165 acceleration_aux_faces,
166 s_face,
167 resu);
168
169 }
170// Boucle sur les faces internes
171 {
172 const int ndeb = zone.premiere_face_int();
173 const int nfin = zone.nb_faces();
174 TSAVEF_ajouter_liste_faces(ndeb, nfin,
175 volumes_entrelaces,
176 zone.volumes(),
177 porosite_surf,
178 face_voisins,
179 ref_rho_,
180 acceleration_aux_faces,
181 s_face,
182 resu);
183 }
184
185 {
186// Force la periodicite
187 int nb_comp=resu.line_size();
188 for (int n_bord=0; n_bord<zone.nb_front_Cl(); n_bord++)
189 {
190 const Cond_lim& la_cl = zone_Cl.les_conditions_limites(n_bord);
191 if (sub_type(Periodique,la_cl.valeur()))
192 {
193 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
194 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
195 int nb_faces_bord=le_bord.nb_faces();
196 ArrOfInt fait(nb_faces_bord);
197 fait = 0;
198 for (int ind_face=0; ind_face<nb_faces_bord; ind_face++)
199 {
200 if (fait[ind_face] == 0)
201 {
202 int ind_face_associee = la_cl_perio.face_associee(ind_face);
203 fait[ind_face] = 1;
204 fait[ind_face_associee] = 1;
205 int face = le_bord.num_face(ind_face);
206 int face_associee = le_bord.num_face(ind_face_associee);
207 for (int comp=0; comp<nb_comp; comp++)
208 {
209 double val = 0.5*(resu(face_associee, comp)+resu(face, comp));
210 resu(face, comp)=resu(face_associee, comp) = val;
211 }
212 }//!< if fait
213 }//!< for face
214 }//!< sub_type Perio
215 }
216 }
217 return resu;
218}
