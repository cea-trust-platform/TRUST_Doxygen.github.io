114// Returns one of the "real" dimensions of the multi-dimensionnal array, as defined by:
115// dimension(0) = size_reelle() / line_size(), or 0 if line_size()==0
116// and, for i >= 1 : dimension(i) is equal to dimension_tot(i)
117// If TRUSTVect<_TYPE_>::size_reelle_ok() returns 0, it is invalid to ask for dimension(0). You can only ask for dimension_tot(0) (see TRUSTVect<_TYPE_>::size_reelle_ok())
118template<typename _TYPE_>
119inline int TRUSTTab<_TYPE_>::dimension(int i) const
120{
121 assert(i >= 0 && i < nb_dim_);
122// Si dimension_[0] == -1, c'est que c'est un vecteur distribue et que l'attribut size() est invalide. Il faut alors utiliser dimension_tot pour ce tableau.
123 assert(dimensions_[i] >= 0);
124 return dimensions_[i];
125}

127// Returns the total dimensions of the multi-dimensionnal array, including virtual items (used in parallel distributed arrays)
128template<typename _TYPE_>
129inline int TRUSTTab<_TYPE_>::dimension_tot(int i) const
130{
131 assert(i >= 0 && i < nb_dim_);
132 return (i == 0) ? dimension_tot_0_ : dimensions_[i];
133}

135// Adds 1 to dimension_tot(0) and puts a in the added line.
136// Precondition: line_size() must be equal to 1 and the array must be resizable.
137template<typename _TYPE_>
138inline void TRUSTTab<_TYPE_>::append_line(_TYPE_ a)
139{
140 assert(TRUSTVect<_TYPE_>::line_size() == 1);
141 assert(dimension_tot_0_ * TRUSTVect<_TYPE_>::line_size() == TRUSTVect<_TYPE_>::size_array());
142 const int n = dimension_tot_0_;
143 dimensions_[0] = ++dimension_tot_0_;
144 TRUSTVect<_TYPE_>::resize_vect_(n+1, Array_base::COPY_NOINIT);
145 _TYPE_ * ptr = TRUSTVect<_TYPE_>::addr() + n;
146 ptr[0] = a;
147}

195// fait pointer le tableau sur le vecteur v et en associant la meme structure parallele.
196// Attention, si line_size du vecteur v est different de 1, on cree un tableau bidimensionnel (on peut avoir un vecteur
197// de ce type si on copie un Tab dans un Vect puis on prend une ref sur ce Vect).
198// Precondition: le vecteur v doit vraiment etre de type Vect ! (sinon utiliser DoubleTab::ref(const DoubleTab &)
199template<typename _TYPE_>
200inline void TRUSTTab<_TYPE_>::ref(const TRUSTVect<_TYPE_>& v)
201{
202 assert(std::string(typeid(v).name()).find("TRUSTVect") != std::string::npos);
203 TRUSTVect<_TYPE_>::ref(v);
204 const int l = v.line_size();
205// Attention: En prenant la ref, on est oblige de conserver l'attribut line_size du Vect (sinon echange_espace_virtuel ne fonctionnera pas car
206// on n'aura pas le bon facteur multiplicatif des items geometriques). Si on voulait creer un tableau monodimensionnel avec line_size > 1,
207// le tableau devient invalide car on n'a plus line_size = produit des dimensions > 1.
208// On peut le faire a condition de laisser tomber le md_vector_ en faisant tab.ref_array(v) au lieu de tab.ref(v)
209 if (l == 1) nb_dim_ = 1;
210 else
211 {
212 nb_dim_ = 2;
213 dimensions_[1] = l;
214 }
215
216 if (v.size_reelle_ok())
217 {
218 int sz = v.size_reelle();
219 dimensions_[0] = sz / l;
220 }
221 else dimensions_[0] = -1;
222
223 dimension_tot_0_ = TRUSTVect<_TYPE_>::size_array() / l;
224 assert(verifie_LINE_SIZE());
225}

227// fait pointer le tableau sur le tableau t en recuperant la structure parallele. Attention, on fige le tableau qui ne pourra plus etre resize
228template<typename _TYPE_>
229inline void TRUSTTab<_TYPE_>::ref(const TRUSTTab& src)
230{
231 TRUSTVect<_TYPE_>::ref(src);
232 nb_dim_ = src.nb_dim_;
233 for (int i = 0; i < MAXDIM_TAB; i++) dimensions_[i] = src.dimensions_[i];
234 dimension_tot_0_ = src.dimension_tot_0_;
235 assert(verifie_LINE_SIZE());
236}

262// fait pointer le tableau sur une sous-partie du tableau t definie par la valeur du premier indice et ne nombre de "lignes" du tableau
263// a recuperer (une ligne = toutes les valeurs tab(i,j,k,...) pour un i donne). Le nombre de dimensions du tableau est le meme que pour t,
264// les dimension(i) pour i>=1 sont les memes et dimension(0) = nb_lines.
265template<typename _TYPE_>
266inline void TRUSTTab<_TYPE_>::ref_tab(TRUSTTab<_TYPE_>& t, int start_line, int nb_lines)
267{
268 if (nb_lines < 0) nb_lines = t.dimension_tot_0_ - start_line;
269 assert(start_line >= 0 && nb_lines >= 0 && start_line + nb_lines <= t.dimension_tot_0_);
270 const int l_size = t.line_size();
271 TRUSTVect<_TYPE_>::ref_array(t, start_line * l_size, nb_lines * l_size);
272 assert(!TRUSTVect<_TYPE_>::get_md_vector().non_nul() && TRUSTVect<_TYPE_>::size_reelle() == TRUSTVect<_TYPE_>::size_array());
273 TRUSTVect<_TYPE_>::set_line_size_(l_size);
274 nb_dim_ = t.nb_dim_;
275 dimension_tot_0_ = nb_lines;
276 dimensions_[0] = nb_lines;
277 for (int i = 1; i < MAXDIM_TAB; i++) dimensions_[i] = t.dimensions_[i];
278 assert(verifie_LINE_SIZE());
279}

386// redimensionne le tableau (nb_dim_ sera egal a tailles.size_array() et dimension(i) a tailles[i].
387// Precondition: identiques a TRUSTVect::resize_vect_()
388template<typename _TYPE_>
389inline void TRUSTTab<_TYPE_>::resize(const TRUSTArray<int>& tailles, Array_base::Resize_Options opt)
390{
391 nb_dim_ = tailles.size_array();
392 if (nb_dim_ <= 0 || nb_dim_ > MAXDIM_TAB)
393 {
394 Cerr << "Internal error in TRUSTTab::resize(const ArrOfInt & tailles, ...) \n" << " wrong dimensions number " << nb_dim_ << finl;
395 Process::exit();
396 }
397 int l_size = 1;
398 for (int i = 0; i < nb_dim_; i++)
399 {
400 const int n = tailles[i];
401 dimensions_[i] = n;
402 if (i > 0)
403 l_size *= n;
404 if (n < 0)
405 {
406 Cerr << "Internal error in TRUSTTab::resize(const ArrOfInt & tailles, ...) \n" << " wrong dimensions: " << tailles << finl;
407 Process::exit();
408 }
409 }
410 dimension_tot_0_ = dimensions_[0];
411 TRUSTVect<_TYPE_>::set_line_size_(l_size);
412 TRUSTVect<_TYPE_>::resize_vect_(dimensions_[0] * l_size, opt);
413 assert(verifie_LINE_SIZE());
414}

425// copie la structure et les valeurs de src. Attention: appel invalide si src est un type derive de Vect
426// (sinon quoi faire, un tableau unidimensionnel, ou une copie de la structure ?)
427template<typename _TYPE_>
428inline TRUSTTab<_TYPE_>& TRUSTTab<_TYPE_>::operator=(const TRUSTVect<_TYPE_>& src)
429{
430 assert(std::string(typeid(src).name()).find("TRUSTVect") != std::string::npos);
431 TRUSTVect<_TYPE_>::copy_(src);
432
433// Idem que dans ref(TRUSTVect<_TYPE_>) pour le nombre de dimensions du tableau cree
434 const int l = src.line_size();
435 if (l == 1) nb_dim_ = 1;
436 else
437 {
438 nb_dim_ = 2;
439 dimensions_[1] = l;
440 }
441
442 if (src.size_reelle_ok())
443 {
444 int sz = src.size_reelle();
445 dimensions_[0] = sz / l;
446 assert(sz % l == 0);
447 }
448 else dimensions_[0] = -1;
449
450 dimension_tot_0_ = TRUSTVect<_TYPE_>::size_array() / l;
451 assert(verifie_LINE_SIZE());
452 return *this;
453}

588template<typename _TYPE_>
589inline void TRUSTTab<_TYPE_>::ajoute_produit_tensoriel(double alpha, const TRUSTTab<double>& x, const TRUSTTab<double>& y)
590{
591// Tableaux vus comme des tableaux unidimensionnels (pour ne pas avoir a gerer nb_dim)
592 const TRUSTVect<double>& vx = x, &vy = y;
593 TRUSTVect<double>& v = *this;
594
595 const int line_size_x = vx.line_size(), line_size_y = vy.line_size(), line_size_xy = v.line_size();
596 assert(line_size_xy == line_size_x * line_size_y);
597// Pour ne pas diviser par line_size()
598 assert(vx.size_totale() * line_size_xy == v.size_totale() * line_size_x);
599 assert(vy.size_totale() * line_size_xy == v.size_totale() * line_size_y);
600
601// blocs d'items a traiter (un bloc par defaut: tout le tableau)
602 int default_bloc[2];
603 default_bloc[0] = 0;
604 default_bloc[1] = (v.line_size() > 0) ? (v.size_totale() / v.line_size()) : 0;
605 const int *blocs = default_bloc;
606 int nb_blocs = 1;
607 if (v.get_md_vector().non_nul())
608 {
609 const TRUSTArray<int>& blk = v.get_md_vector().valeur().get_items_to_compute();
610 blocs = blk.addr();
611 nb_blocs = blk.size_array() / 2;
612 }
613
614 for (int i_bloc = 0; i_bloc < nb_blocs; i_bloc++)
615 {
616 const int debut = blocs[i_bloc*2], fin = blocs[i_bloc*2+1];
617 int v_index = debut * line_size_xy;
618 for (int i = debut; i < fin; i++)
619 for (int j = 0; j < line_size_x; j++)
620 {
621 double xval = vx[i * line_size_x + j];
622 for (int k = 0; k < line_size_y; k++)
623 {
624 double yval = vy[i * line_size_y + k];
625 v[v_index] += alpha * xval * yval;
626 v_index++;
627 }
628 }
629 }
630}

646// Decomposition d'une matrice en L.U: methode de Crout (diagonale de L =1)
647/** \return matrice A_ = assemblage (L-diagonale)+U */
648template<typename _TYPE_>
649inline int TRUSTTab<_TYPE_>::decomp_LU(int n, TRUSTArray<int>& index, TRUSTTab<double>& matLU)
650{
651 TRUSTArray<double> vv(n);
652 int i, j, k, imax = -1, cvg = 1;
653 double big, dum, sum, temp;
654 matLU = (*this);
655
656// Recupere le coeff max d'une ligne, stocke dans vv
657 for (i=0 ; i<n ; i++)
658 {
659 big = 0;
660 for (j=0 ; j<n ; j++)
661 if ((temp = std::fabs(matLU(i,j))) > big) big = temp;
662
663 if (big == 0)
664 {
665 Cerr <<"Singular matrix in LU decomposition"<<finl;
666 cvg = 0;
667 Process::exit();
668 }
669 vv[i] = 1./big;
670 }
671
672// calcul de la matrice matLU
673 for (j=0 ; j<n ; j++)
674 {
675 for (i=0 ; i<j ; i++)
676 {
677 sum = matLU(i,j);
678 for (k=0 ; k<i ; k++) sum -= matLU(i,k) * matLU(k,j);
679 matLU(i,j) = sum;
680 }
681
682 big = 0;
683 for (i=j ; i<n ; i++)
684 {
685 sum = matLU(i,j);
686 for (k=0 ; k<j ; k++) sum -= matLU(i,k) * matLU(k,j);
687 matLU(i,j) = sum;
688 if ((dum = vv[i]*std::fabs(sum)) >= big)
689 {
690 big = dum;
691 imax = i;
692 }
693 }
694
695 if (j != imax)
696 {
697 for (k=0 ; k<n ; k++)
698 {
699 dum = matLU(imax,k);
700 matLU(imax,k) = matLU(j,k);
701 matLU(j,k) = dum;
702 }
703 vv[imax] = vv[j];
704 }
705
706 index[j] = imax;
707 dum = 1./matLU(j,j);
708 for (i=j+1 ; i<n ; i++) matLU(i,j) *= dum;
709 }
710 return cvg;
711}

713// Resolution du systeme A_x=b : A_ contenant le decompostion LU de A (stockee dans une seule matrice)
714template<typename _TYPE_>
715inline void TRUSTTab<_TYPE_>::resoud_LU(int n, TRUSTArray<int>& index, const TRUSTArray<double>& b, TRUSTArray<double>& solution)
716{
717 int i,ii=-1,ip,j;
718 double sum;
719 solution = b;
720 for (i=0 ; i<n ; i++)
721 {
722 ip = index[i];
723 sum = solution[ip];
724 solution[ip] = solution[i];
725 if (ii!=-1)
726 for (j=ii ; j<i ; j++) sum -= (*this)(i,j)*solution[j];
727 else if (sum) ii =i;
728
729 solution[i] = sum;
730 }
731
732 for (i=n-1 ; i>=0 ; i--)
733 {
734 sum = solution[i];
735 for (j=i+1 ; j<n ; j++) sum -= (*this)(i,j)*solution[j];
736 solution[i] = sum/(*this)(i,i);
737 }
738}

740// Fonction utilisee pour le calcul du du/u (pour convergence implicite)
741// renvoie le max de abs(du(i)/u(i)). utilisation max_ = (u(n+1)-u(n)).max_du_u(u(n))
742template<typename _TYPE_>
743inline double TRUSTTab<_TYPE_>::max_du_u(const TRUSTTab<double>& u)
744{
745 assert(TRUSTVect<_TYPE_>::size_array() == u.size_array());
746 const double *du_ptr = TRUSTVect<_TYPE_>::addr();
747 const double *u_ptr = u.addr();
748 const double epsilon = 1.e-8;
749 double res = 0.;
750 for (int n = TRUSTVect<_TYPE_>::size_array(); n; n--)
751 {
752 double a = std::fabs(*du_ptr), b = std::fabs(*u_ptr), c = a / (b + epsilon);
753 if (b > 1.e-2 && c > res) res = c;
754 du_ptr++;
755 u_ptr++;
756 }
757 return res;
758}
