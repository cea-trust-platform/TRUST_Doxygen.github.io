71//! Calcule la masse volumique
72void Loi_Etat_GP_WC::calculer_masse_volumique()
73{
74 const DoubleTab& tab_ICh = le_fluide->inco_chaleur().valeurs();
75 Fluide_Weakly_Compressible& FWC = ref_cast(Fluide_Weakly_Compressible,le_fluide.valeur());
76 DoubleTab& tab_rho = le_fluide->masse_volumique().valeurs();
77 const DoubleTab& pres = FWC.pression_th_tab();
78 const int n = tab_rho.size();
79 for (int som=0 ; som<n ; som++)
80 {
81 tab_rho_np1(som) = calculer_masse_volumique(pres(som,0),tab_ICh(som,0));
82 tab_rho(som,0) = 0.5 * ( tab_rho_n(som) + tab_rho_np1(som) );
83 }
84 tab_rho.echange_espace_virtuel();
85 tab_rho_np1.echange_espace_virtuel();
86 le_fluide->calculer_rho_face(tab_rho_np1);
87 Debog::verifier("Loi_Etat_GP_WC::calculer_masse_volumique, tab_rho_np1",tab_rho_np1);
88 Debog::verifier("Loi_Etat_GP_WC::calculer_masse_volumique, tab_rho",tab_rho);
89}
