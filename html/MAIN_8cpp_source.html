35void usage()
36{
37 Cerr << "usage:\n";
38 Cerr << "TRUST_EXECUTABLE [CASE[.data]] [options]\n";
39 Cerr << " CASE is the basename of the trust data file (must have .data extension)\n";
40 Cerr << " If no CASE given, the current directory name is used\n";
41 Cerr << " -help_trust => print options\n";
42 Cerr << " -mpi => run in parallel with MPI (must run with mpirun)\n";
43 Cerr << " -check_enabled=0|1 => enables or disables runtime checking of parallel messages\n";
44 Cerr << " -debugscript=SCRIPT => execute \"SCRIPT n\" after parallel initialisation, n=processor rank\n";
45 Cerr << " -petsc=0 => disable call to PetscInitialize\n";
46 Cerr << " -journal=0..9 => select journal level (0=disable, 9=maximum verbosity)\n";
47 Cerr << " -journal_master => only master processor writes a journal (not compatible with journal_shared)\n";
48 Cerr << " -journal_shared => each processor writes in a single log file (not compatible with journal_master)\n";
49 Cerr << " -disable_ieee => Disable the detection of NaNs. The detection can also be de-activated with env variable TRUST_DISABLE_FP_EXCEPT set to non zero.\n";
50 Cerr << " -no_verify => Disable the call to verifie function (from Type_Verifie) to catch outdated keywords while reading data file.\n";
51 Cerr << " -disable_stop => Disable the writing of the .stop file.\n";
52 Cerr << finl;
53 Process::exit();
54}

285// Modif B.Mathieu 28/09/2004 :
286// Pour debugger trust, on donne parametre le nom
287// d'une commande systeme a executer lorsqu'on arrive a cet endroit.
288// Le plus pratique, c'est d'utiliser un shell script qui ouvre

322 if (data_file == "")
323 {
324// TRUST sans argument => nom du cas = nom du repertoire courant
325 Nom pwd(::pwd());

327 Cerr << "No command line argument. Data file name from directory name:\n "

333 i--;
334 }

341 }
342 }
343 data_file = pwd.substr_old(i + 2, l - i - 1); //!< Attention, voir Nom::substr()
344 }
345 if (master)
346 {
347 Cerr << "Data file : " << data_file << finl;
348 }
349// Si le nom du cas finit par .data, on l'enleve.
350 if (data_file.finit_par(".data"))
351 {
352 if (master)
353 {
354 Cerr << " (removing .data extension)" << finl;
355 }
356 data_file.prefix(".data");
357 }
358
359
360
361 if (master)
362 {
363 Cerr<<"Localisation etude: " << ::pwd() << finl;
364 Cerr<<"Nom du cas " << data_file << finl;
365 Cerr<<" code : "<< argv[0] << finl;
366 Cerr<<" version : 1.9.0 " << finl;
367 }
368
369 main_process->dowork(data_file);
370
371 if (master)
372 {
373 Cerr << "Arret des processes." << finl;
374 }
375 }
376
377// pour detruire les derniers octets
378 desalloue_pwd();
379 return (0);
