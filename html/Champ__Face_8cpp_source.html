83Champ_base& Champ_Face::affecter_(const Champ_base& ch)
84{
85 const DoubleTab& v = ch.valeurs();
86 DoubleTab& val = valeurs();
87 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
88 int nb_faces = zone_VDF.nb_faces();
89 const IntVect& orientation = zone_VDF.orientation();
90 int ori, n0, n1;
91
92 if (sub_type(Champ_Uniforme,ch))
93 {
94 for (int num_face=0; num_face<nb_faces; num_face++)
95 val(num_face) = v(0,orientation(num_face));
96 }
97 else if ( (sub_type(Champ_Uniforme_Morceaux,ch))
98 || (sub_type(Champ_Don_lu,ch)) )
99 {
100 int ndeb_int = zone_VDF.premiere_face_int();
101 const IntTab& face_voisins = zone_VDF.face_voisins();
102 int num_face;
103
104 for (num_face=0; num_face<ndeb_int; num_face++)
105 {
106 ori = orientation(num_face);
107 n0 = face_voisins(num_face,0);
108 if (n0 != -1)
109 val(num_face) = v(n0,ori);
110 else
111 val(num_face) = v(face_voisins(num_face,1),ori);
112 }
113
114 for (num_face=ndeb_int; num_face<nb_faces; num_face++)
115 {
116 ori = orientation(num_face);
117 n0 = face_voisins(num_face,0);
118 n1 = face_voisins(num_face,1);
119 val(num_face) = 0.5*(v(n0,ori)+v(n1,ori));
120 }
121 }
122 else
123 {
124// int ndeb_int = zone_VDF.premiere_face_int();
125// const IntTab& face_voisins = zone_VDF.face_voisins();
126 DoubleTab positionX(zone_VDF.nb_faces_X() ,dimension);
127 DoubleVect U(zone_VDF.nb_faces_X());
128 DoubleTab positionY(zone_VDF.nb_faces_Y() ,dimension);
129 DoubleVect V(zone_VDF.nb_faces_Y());
130 DoubleTab positionZ(zone_VDF.nb_faces_Z() ,dimension);
131 DoubleVect W(zone_VDF.nb_faces_Z());
132 const DoubleTab& xv=zone_VDF.xv();
133 int nbx=0;
134 int nby=0;
135 int nbz=0;
136 int num_face, k;
137
138 for (num_face=0; num_face<nb_faces; num_face++)
139 {
140 ori = orientation(num_face);
141 switch(ori)
142 {
143 case 0:
144 for(k=0; k<dimension; k++)
145 positionX(nbx, k)=xv(num_face, k);
146 nbx++;
147 break;
148 case 1:
149 for(k=0; k<dimension; k++)
150 positionY(nby, k)=xv(num_face, k);
151 nby++;
152 break;
153 case 2:
154 for(k=0; k<dimension; k++)
155 positionZ(nbz, k)=xv(num_face, k);
156 nbz++;
157 break;
158 }
159 }
160
161 ch.valeur_aux_compo(positionX, U, 0);
162 ch.valeur_aux_compo(positionY, V, 1);
163 if(dimension == 3)
164 ch.valeur_aux_compo(positionZ, W, 2);
165 nbx=nby=nbz=0;
166 for (num_face=0; num_face<nb_faces; num_face++)
167 {
168 ori = orientation(num_face);
169 switch(ori)
170 {
171 case 0:
172 val(num_face) = U(nbx++);
173 break;
174 case 1:
175 val(num_face) = V(nby++);
176 break;
177 case 2:
178 val(num_face) = W(nbz++);
179 break;
180 }
181 }
182 }
183 return *this;
184}

264// -Cas CL periodique : assure que les valeurs sur des faces periodiques
265// en vis a vis sont identiques. Pour cela on prend la demi somme des deux valeurs.
266void Champ_Face::verifie_valeurs_cl()
267{
268 const Zone_Cl_dis_base& zcl = zone_Cl_dis().valeur();
269 int nb_cl = zcl.nb_cond_lim();
270 DoubleTab& ch_tab = valeurs();
271 int ndeb,nfin,num_face;
272
273 for (int i=0; i<nb_cl; i++)
274 {
275 const Cond_lim_base& la_cl = zcl.les_conditions_limites(i).valeur();
276 if (sub_type(Periodique,la_cl))
277 {
278 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl);
279 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
280 ndeb = le_bord.num_premiere_face();
281 nfin = ndeb + le_bord.nb_faces();
282 int voisine;
283 double moy;
284
285 for (num_face=ndeb; num_face<nfin; num_face++)
286 {
287 voisine = la_cl_perio.face_associee(num_face-ndeb) + ndeb;
288 if (ch_tab[num_face] != ch_tab[voisine])
289 {
290 moy = 0.5*(ch_tab[num_face] + ch_tab[voisine]);
291 ch_tab[num_face] = moy;
292 ch_tab[voisine] = moy;
293 }
294 }
295 }
296 }
297 ch_tab.echange_espace_virtuel();
298}

314double Champ_Face_get_val_imp_face_bord(const double temp,int face,int comp, const Zone_Cl_VDF& zclo)
315{
316 const Zone_VDF& zone_vdf=zclo.zone_VDF();
317 int face_globale,face_locale;
318
319 face_globale=face+zone_vdf.premiere_face_bord(); //!< Maintenant numero dans le tableau global des faces.
320 const Zone_Cl_dis_base& zcl = zclo; //equation().zone_Cl_dis().valeur();
321// On recupere la CL associee a la face et le numero local de la face dans la frontiere.
322// assert(equation().zone_Cl_dis().valeur()==zclo);
323
324 const Cond_lim_base& cl=(face<zone_vdf.nb_faces()) ?
325 zcl.condition_limite_de_la_face_reelle(face_globale,face_locale) :
326 zcl.condition_limite_de_la_face_virtuelle(face_globale,face_locale);
327
328 int ori = zone_vdf.orientation()(face_globale);
329
330 const DoubleTab& vals=cl.champ_front()->valeurs_au_temps(temp);
331
332 int face_de_vals=vals.dimension(0)==1 ? 0 : face_locale;
333
334 if(sub_type(Symetrie,cl))
335 {
336 if (comp == ori)
337 return 0;
338 else
339 {
340 Cerr<<"You should call Champ_Face_get_val_imp_face_bord_sym and not Champ_Face_get_val_imp_face_bord"<<finl;
341 Process::exit();
342 return 1e9;
343 }
344 }
345
346 else if ( sub_type(Dirichlet_entree_fluide,cl) )
347 {
348 return vals(face_de_vals,comp);
349 }
350
351 else if ( sub_type(Dirichlet_paroi_fixe,cl) )
352 {
353 return 0;
354 }
355
356 else if ( sub_type(Dirichlet_paroi_defilante,cl) )
357 {
358 return vals(face_de_vals,comp);
359 }
360
361 return 0; //!< All other cases
362

364double Champ_Face_get_val_imp_face_bord_sym(const DoubleTab& tab_valeurs, const double temp,int face,int comp, const Zone_Cl_VDF& zclo)
365{
366 const Zone_VDF& zone_vdf=zclo.zone_VDF();
367 int face_globale,face_locale;
368
369 face_globale=face+zone_vdf.premiere_face_bord(); //!< Maintenant numero dans le tableau global des faces.
370 const Zone_Cl_dis_base& zcl = zclo; //equation().zone_Cl_dis().valeur();
371// On recupere la CL associee a la face et le numero local de la face dans la frontiere.
372// assert(equation().zone_Cl_dis().valeur()==zclo);
373
374 const Cond_lim_base& cl=(face<zone_vdf.nb_faces()) ?
375 zcl.condition_limite_de_la_face_reelle(face_globale,face_locale) :
376 zcl.condition_limite_de_la_face_virtuelle(face_globale,face_locale);
377
378 const IntTab& face_voisins = zone_vdf.face_voisins();
379 const IntTab& elem_faces = zone_vdf.elem_faces();
380 const DoubleVect& porosite = zone_vdf.porosite_face();
381 int ori = zone_vdf.orientation()(face_globale);
382
383 const DoubleTab& vals=cl.champ_front()->valeurs_au_temps(temp);
384
385 int face_de_vals=vals.dimension(0)==1 ? 0 : face_locale;
386
387 if(sub_type(Symetrie,cl))
388 {
389 if (comp == ori)
390 return 0;
391 else
392 {
393 int elem=0;
394 if (face_voisins(face_globale,0) != -1)
395 elem = face_voisins(face_globale,0);
396 else
397 elem = face_voisins(face_globale,1);
398 int comp2=comp+Objet_U::dimension;
399 return (tab_valeurs(elem_faces(elem,comp))*porosite[elem_faces(elem,comp)]
400 + tab_valeurs(elem_faces(elem,comp2))*porosite[elem_faces(elem,comp2)])
401 / (porosite[elem_faces(elem,comp)] + porosite[elem_faces(elem,comp2)]) ;
402 }
403 }
404
405 else if ( sub_type(Dirichlet_entree_fluide,cl) )
406 {
407 return vals(face_de_vals,comp);
408 }
409
410 else if ( sub_type(Dirichlet_paroi_fixe,cl) )
411 {
412 return 0;
413 }
414
415 else if ( sub_type(Dirichlet_paroi_defilante,cl) )
416 {
417 return vals(face_de_vals,comp);
418 }
419
420 return 0; //!< All other cases
421
422}
