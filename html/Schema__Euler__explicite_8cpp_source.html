25//! Effectue un pas de temps d'Euler explicite sur l'equation passee en parametre.
26int Schema_Euler_explicite::faire_un_pas_de_temps_eqn_base(Equation_base& eqn)
27{
28 DoubleTab& present = eqn.inconnue().valeurs(); //!< Un
29 DoubleTab& futur = eqn.inconnue().futur(); //!< Un+1
30 DoubleTab dudt(futur); //!< just for initializing the array structure ...
31
32// Boundary conditions applied on Un+1:
33 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(), temps_courant() + pas_de_temps());
34
35// On tourne la roue pour que les operateurs utilisent les champs au temps futur
36 eqn.inconnue().avancer();
37 eqn.derivee_en_temps_inco(dudt);
38 eqn.inconnue().reculer();
39
40// Un+1=Un+dt_*dU/dt
41 futur = dudt;
42 futur *= dt_;
43 futur += present;
44
45 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(), temps_courant() + pas_de_temps());
46 update_critere_statio(dudt, eqn);
47
48 return 1;
49}
