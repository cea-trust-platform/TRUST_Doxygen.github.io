94void Schema_Backward_Differentiation_base::store_equation_parameters(Equation_base& eqn, DoubleTab& stored_parameters)
95{
96 if (nb_pas_dt()>nb_pas_dt_seuil())
97 {

99 {

102 }

106// To take into account pressure constraint
107void Schema_Backward_Differentiation_base::modify_equation_parameters(Equation_base& eqn, DoubleTab& stored_parameters)
108{
109 if (nb_pas_dt()>nb_pas_dt_seuil())
110 {
111 if (sub_type(Navier_Stokes_std,eqn))
112 {
113 Navier_Stokes_std& navier_stokes = ref_cast(Navier_Stokes_std,eqn);
114
115 DoubleTab& pressure = navier_stokes.pression().valeurs(); //pressure at time tnplus1 that is to say P(n+1)
116 DoubleTab delta_pressure(pressure);
117
118 delta_pressure -= stored_parameters; //contains P(n+1) - P(n)
119 delta_pressure /= coefficients()[nb_valeurs_passees()+1]; //take into account time scheme coefficient
120 pressure = stored_parameters;
121 pressure += delta_pressure;
122 }
123 }
124}

126void Schema_Backward_Differentiation_base::add_multi_timestep_data(const Equation_base& eqn, Matrice_Base& mat_morse, DoubleTab& secmem) const
127{
128 double time_step = pas_de_temps();
129 double effective_time_step = -1.;
130
131// Remark for "pen" interger : we do not penalize matrices and right members even in the case
132// of Dirichlet or Symetrie boundary conditions
133 int pen = 0;
134 int offset = 0;
135 int i = 0;
136
137 effective_time_step = coefficients()[nb_valeurs_passees()+1] * time_step;
138
139// Modification due to present time and past times
140 DoubleTab dudt(eqn.inconnue().valeurs());
141 DoubleTab tmp(dudt);
142
143 dudt = 0.;
144 for (i=0; i<nb_valeurs_passees()+1; ++i)
145 {
146 offset = nb_valeurs_passees()-i + 1;//due to Initialiser_Champs() offset
147 tmp = eqn.inconnue().passe(offset);
148 tmp *= coefficients()[i] ;
149 dudt += tmp; //past times and present time if i==nb_valeurs_passees()
150 }
151
152 eqn.solv_masse().ajouter_masse(effective_time_step,secmem,dudt,pen);
153 eqn.solv_masse().ajouter_masse(effective_time_step,mat_morse,pen);
154}
