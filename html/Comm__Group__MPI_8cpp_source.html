151void Comm_Group_MPI::mp_collective_op(const double *x, double *resu, int n, Collective_Op op) const
152{
153#ifdef MPI_
154 if (n <= 0)
155 return;
156 switch(op)
157 {
158// Note B.M.: cast en non const a cause de l'interface de MPI,
159// plusieurs posts sur newsgroups incitent a penser que c'est ok
160// (x ne sera jamais modifie par la librairie, sauf bug !)
161 case COLL_SUM:
162 statistiques().begin_count(mpi_sumdouble_counter_);
163 mpi_error(MPI_Allreduce((int*) x, resu, n, MPI_DOUBLE, MPI_SUM, mpi_comm_));
164 statistiques().end_count(mpi_sumdouble_counter_);
165 break;
166 case COLL_MIN:
167 statistiques().begin_count(mpi_mindouble_counter_);
168 mpi_error(MPI_Allreduce((int*) x, resu, n, MPI_DOUBLE, MPI_MIN, mpi_comm_));
169 statistiques().end_count(mpi_mindouble_counter_);
170 break;
171 case COLL_MAX:
172 statistiques().begin_count(mpi_maxdouble_counter_);
173 mpi_error(MPI_Allreduce((int*) x, resu, n, MPI_DOUBLE, MPI_MAX, mpi_comm_));
174 statistiques().end_count(mpi_maxdouble_counter_);
175 break;
176 case COLL_PARTIAL_SUM:
177 internal_collective(x, resu, n, &op, -1 /* only one operation */, 0 /* recursion level */);
178 break;
179 }
180#endif

238//! Point de synchronisation de tous les processeurs du groupe (permet<br>de verifier que tout le monde est la...). Si check_enabled() est<br>non nul, on utilise le tag pour verifier que tous les processeurs<br>sont bien en train d'attendre le meme tag, sinon c'est une barriere<br>simple. Le tag doit verifier 0 <= tag < max_tag (soit 32).
239void Comm_Group_MPI::barrier(int tag) const
240{
241#ifdef MPI_
242 static const int max_tag = 32;
243 statistiques().begin_count(mpi_barrier_counter_);
244 assert(tag >= 0 && tag < max_tag);
245 if (check_enabled())
246 {
247// On fait la barriere avec des mpmin et mpmax pour verifier
248// que le tag est le meme sur tous les processeurs :
249// ATTENTION : il faut "int" et pas "entier" !!!
250 int tag_complet = get_new_tag() * max_tag + tag;
251 int min_tag, amax_tag;
252 mpi_error(MPI_Allreduce(& tag_complet, & min_tag, 1, MPI_ENTIER, MPI_MIN, mpi_comm_));
253 mpi_error(MPI_Allreduce(& tag_complet, & amax_tag, 1, MPI_ENTIER, MPI_MAX, mpi_comm_));
254 if (min_tag != tag_complet || amax_tag != tag_complet)
255 {
256 Cerr << "Error in Comm_Group_MPI::barrier(int tag)\n";
257 Cerr << " the tag is not identical on all the processes.\n";
258 Cerr << " (Loss of communications synchronisation)." << finl;
259 Process::Journal() << "Comm_Group_MPI::barrier\n Error : tag = " << tag << finl;
260 assert(0);
261 exit();
262 }
263 }
264 else
265 {
266// Barriere simple sans le tag :
267 mpi_error(MPI_Barrier(mpi_comm_));
268 }
269 statistiques().end_count(mpi_barrier_counter_);
270#endif
271}

284//! Demarre l'envoi et la reception des buffers.<br>Les buffers doivent rester valide jusqu'au retour de send_recv_finish().<br>Le graphe de communication et la taille des buffers doivent etre corrects !<br>send_list : liste des processeurs (numerotation sur groupe courant) a qui envoyer<br>send_size : taille en octets de chaque message<br>send_buffers : adresse des donnees a envoyer.<br>recv_... : idem pour les donnees en reception.
285void Comm_Group_MPI::send_recv_start(const ArrOfInt& send_list,
286 const ArrOfInt& send_size,
287 const char * const * const send_buffers,
288 const ArrOfInt& recv_list,
289 const ArrOfInt& recv_size,
290 char * const * const recv_buffers,
291 TypeHint typehint) const
292{
293#ifdef MPI_
294 statistiques().begin_count(mpi_sendrecv_counter_);
295 assert(mpi_nrequests_ < 0);
296
297 const int tag = get_new_tag();
298 int i, n;
299 mpi_nrequests_ = 0;
300 int msg_size = 0;
301
302 int divisor = 0;
303 MPI_Datatype datatype = MPI_CHAR;
304 assert(sizeof(int) == sizeof(int)); //!< Sinon il faut changer MPI_ENTIER !!!
305 switch(typehint)
306 {
307 case CHAR:
308 divisor = 1;
309 break;
310 case INT:
311 divisor = sizeof(int);
312 datatype = MPI_ENTIER;
313 break;
314 case DOUBLE:
315 divisor = sizeof(double);
316 datatype = MPI_DOUBLE;
317 break;
318 default:
319 Process::exit();
320 }
321
322// Astuce pour maximiser les chances que ca marche : on declare
323// la reception d'abord et l'envoi ensuite.
324 n = recv_list.size_array();
325 for (i = 0; i < n; i++)
326 {
327 int source = recv_list[i];
328 int sz = recv_size[i];
329 msg_size += sz;
330 assert(source >= 0 && source < nproc());
331 assert(mpi_nrequests_ < mpi_maxrequests_);
332 assert(sz % divisor == 0);
333 assert(mpi_requests_[mpi_nrequests_]==MPI_REQUEST_NULL);
334 mpi_error(MPI_Irecv(recv_buffers[i], sz / divisor,
335 datatype,
336 source, tag, mpi_comm_,
337 & mpi_requests_[mpi_nrequests_]));
338 mpi_nrequests_++;
339 }
340
341 n = send_list.size_array();
342 for (i = 0; i < n; i++)
343 {
344 int dest = send_list[i];
345 int sz = send_size[i];
346 msg_size += sz;
347 assert(dest >= 0 && dest < nproc());
348 assert(mpi_nrequests_ < mpi_maxrequests_);
349 assert(sz % divisor == 0);
350 mpi_error(MPI_Isend((char*) send_buffers[i], sz / divisor,
351 datatype,
352 dest, tag, mpi_comm_,
353 & mpi_requests_[mpi_nrequests_]));
354 mpi_nrequests_++;
355 }
356 current_msg_size_ = msg_size;
357#endif
358}

452//! constructeur du groupe "tous"<br>Attention, ce constructeur ne doit etre appele qu'une seule fois.<br>Le groupe est associe a trio_u_world_<br>Si must_mpi_initialize_==0, on suppose que MPI_Init a deja ete appele.<br>Apres l'appel a init_group_trio, il faut enregistrer le groupe dans PE_Groups<br>Voir PE_Groups::initialize()
453void Comm_Group_MPI::init_group_trio()
454{
455 must_finalize_ = 0;
456
457 if (mpi_status_ != 0)
458 {
459 Cerr << "Error : the construction of the global Comm_Group_MPI has already been done." << finl;
460 exit();
461 }
462
463 if (must_mpi_initialize_)
464 {
465 if (trio_u_world_ != MPI_COMM_WORLD)
466 {
467 Cerr << "Error in Comm_Group_MPI::init_group_trio(...) : you cannot ask to initialize MPI\n"
468 << " with something else than MPI_COMM_WORLD !" << finl;
469 exit();
470 }
471 must_finalize_ = 1;
472 True_int argc=0;
473 char** argv=NULL;
474 int errcode = MPI_Init(&argc, &argv);
475// int errcode = MPI_Init(0,0); Message d'erreur sur MPI Voltaire
476 if (errcode != MPI_SUCCESS)
477 {
478 Cerr << "Error in Comm_Group_MPI::init_group_trio()\n"
479 << " MPI_Init() failed (forget to run with mpirun ?)" << finl;
480 mpi_error(errcode);
481 }
482 }
483
484 True_int arank;
485 True_int nbproc;
486
487 mpi_error(MPI_Comm_size (trio_u_world_, & nbproc));
488 mpi_error(MPI_Comm_rank (trio_u_world_, & arank));
489
490 Comm_Group::init_group_trio(nbproc, arank);
491
492 mpi_comm_ = trio_u_world_;
493 MPI_Comm_group(mpi_comm_, &mpi_group_);
494
495// Initialisation des variables statiques de la classe
496// Un buffer a envoyer et un a recevoir par processeur
497// d'ou le maximum...
498 mpi_maxrequests_ = nbproc * 2;
499 mpi_status_ = new MPI_Status[mpi_maxrequests_];
500 mpi_requests_ = new MPI_Request[mpi_maxrequests_];
501
502 for (int r=0; r<mpi_maxrequests_; r++)
503 {
504 mpi_requests_[r]=MPI_REQUEST_NULL;
505 }
506 if (arank == 0)
507 {
508 if (trio_u_world_ == MPI_COMM_WORLD)
509 {
510 Cerr << "Initialized MPI with MPI_COMM_WORLD (using all processors)" << finl;
511 }
512 else
513 {
514 Cerr << "Initialized MPI with communicator!=MPI_COMM_WORLD: using " << (int)nbproc << " processors" << finl;
515 }
516 }
517}

527// Wrapper to MPI_Alltoallv. data type is MPI_CHAR
528void Comm_Group_MPI::all_to_allv(const void *src_buffer, int *send_data_size, int *send_data_offset,
529 void *dest_buffer, int *recv_data_size, int *recv_data_offset) const
530{
531 statistiques().begin_count(mpi_alltoall_counter_);
532 assert(src_buffer != dest_buffer);
533 void * ptr = (void *) src_buffer; //!< Cast a cause de l'interface de MPI_Alltoall
534
535 const int n = nproc();
536 int size;
537
538#ifdef INT_is_64_
539 std::vector<True_int> send_data_size_int(n);
540 std::vector<True_int> send_data_offset_int(n);
541 std::vector<True_int> recv_data_size_int(n);
542 std::vector<True_int> recv_data_offset_int(n);
543
544 auto cast_func = [](int i) -> True_int { return static_cast<True_int>(i); };
545 std::transform(send_data_size, send_data_size + n, send_data_size_int.begin(), cast_func);
546 std::transform(send_data_offset, send_data_offset + n, send_data_offset_int.begin(), cast_func);
547 std::transform(recv_data_size, recv_data_size + n, recv_data_size_int.begin(), cast_func);
548 std::transform(recv_data_offset, recv_data_offset + n, recv_data_offset_int.begin(), cast_func);
549
550 mpi_error(MPI_Alltoallv(ptr, send_data_size_int.data(), send_data_offset_int.data(), MPI_CHAR,
551 dest_buffer, recv_data_size_int.data(), recv_data_offset_int.data(), MPI_CHAR, mpi_comm_));
552 size = send_data_offset_int[n-1] + send_data_size_int[n-1] + recv_data_size_int[n-1] + recv_data_offset_int[n-1];
553#else
554 mpi_error(MPI_Alltoallv(ptr, send_data_size, send_data_offset, MPI_CHAR,
555 dest_buffer, recv_data_size, recv_data_offset, MPI_CHAR, mpi_comm_));
556 size = send_data_offset[n-1] + send_data_size[n-1] + recv_data_size[n-1] + recv_data_offset[n-1];
557
558#endif
559 statistiques().end_count(mpi_alltoall_counter_, size);
560
561
562}

599void Comm_Group_MPI::ptop_send_recv(const void * send_buf, int send_buf_size, int send_proc,
600 void * recv_buf, int recv_buf_size, int recv_proc) const
601{
602 statistiques().begin_count(mpi_sendrecv_counter_);
603 assert(mpi_nrequests_ < 0);
604 int dest = send_proc;
605 int src = recv_proc;
606 int tag = 1;
607 MPI_Status status;
608 if (send_proc < 0 && recv_proc < 0)
609 {
610// do nothing
611 }
612 else if (send_proc < 0 && recv_proc >= 0)
613 {

618 mpi_error(MPI_Send ((void*)send_buf, send_buf_size, MPI_CHAR, dest, tag, mpi_comm_));

700//! Renvoie la somme des x sur les processeurs precedents du groupe (moi non<br>compris). Le resultat sur le premier processeur du groupe est donc toujours 0.<br>Le resultat depend de l'ordre dans lequel les processeurs ont ete<br>fournis dans le constructeur.
701int Comm_Group_MPI::mppartial_sum(int x) const
702{
703 statistiques().begin_count(mpi_partialsum_counter_);
704// ATTENTION : il faut "int" et pas "entier" !!!
705 int somme = 0;
706 MPI_Status status;
707 int tag = get_new_tag();
708 int rang = rank();
709 int np = nproc();
710
711 if (rang > 0)
712 {
713// Recoit la somme partielle du precedent
714 mpi_error(MPI_Recv(& somme, 1, MPI_ENTIER, rang-1, tag, mpi_comm_, &status));
715 }
716 if (rang+1 < np)
717 {
718// Envoie la somme partielle au suivant
719// ATTENTION : il faut "int" et pas "entier" !!!
720 int s = somme + x;
721 mpi_error(MPI_Send(& s, 1, MPI_ENTIER, rang+1, tag, mpi_comm_));
722 }
723 statistiques().end_count(mpi_partialsum_counter_);
724 return somme;
725}
