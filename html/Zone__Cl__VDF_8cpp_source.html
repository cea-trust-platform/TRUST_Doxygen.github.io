71
72// Remplissage du tableau d'entiers type_arete_bord_
73// tableau intermediaire : les_faces_Cl
74// On remplit le tableau d'entiers local les_faces_Cl qui
75// donne la condition aux limites pour chaque face de bord
76// avec les conventions suivantes :
77// 0 pour une condition de paroi
78// 1 pour une condition d'entree ou de sortie de fluide (face "fluide")
79// 2 pour une condition de symetrie
80// 3 pour une condition de periodicite
81// 0 pour toute autre Cl.
82// On ne considere que les conditions aux limites qui interviennent pour
83// la quantite de mouvement

85 int nb_aretes_bord = la_zone_VDF.nb_aretes_bord();
86 IntVect les_faces_Cl;
87 la_zone_VDF.creer_tableau_faces_bord(les_faces_Cl);
88
89// Boucle sur les conditions aux limites pour remplir les_faces_Cl:
90
91 for (int n_bord=0; n_bord<la_zone_VDF.nb_front_Cl(); n_bord++)
92 {
93
94// pour chaque Condition Limite on regarde son type
95
96 const Cond_lim_base& la_cl = les_conditions_limites_[n_bord].valeur();
97
98 int numero_cl = 0;
99
100 if ( (sub_type(Dirichlet_paroi_fixe,la_cl)) ||
101 (sub_type(Dirichlet_paroi_defilante,la_cl)) )
102 numero_cl = 0;
103 else if ( (sub_type(Dirichlet_entree_fluide,la_cl)) ||
104 (sub_type(Neumann_sortie_libre,la_cl)) )
105 numero_cl = 1;
106 else if (sub_type(Symetrie,la_cl))
107 numero_cl = 2;
108 else if (sub_type(Periodique,la_cl))
109 numero_cl = 3;
110
111 const Frontiere& fr = la_cl.frontiere_dis().frontiere();
112 const int ndeb = fr.num_premiere_face();
113 const int nfin = ndeb + fr.nb_faces();
114 for (int i=ndeb; i<nfin; i++)
115 {
116 les_faces_Cl[i] = numero_cl;
117 num_Cl_face_[i] = n_bord;
118 }
119
120 }
121
122 les_faces_Cl.echange_espace_virtuel();
123
124// Boucle sur les aretes bord pour remplir type_arete_bord_
125
126 int face1,face2;
127 int rang1,rang2;
128 type_arete_bord_= TypeAreteBordVDF::VIDE;
129 int ndeb = la_zone_VDF.premiere_arete_bord();
130 int nfin = ndeb + nb_aretes_bord;
131 int num_arete_;
132
133 int decal_virt = la_zone_VDF.nb_faces();
134 int sz_faces_Cl = les_faces_Cl.size_reelle();
135 const ArrOfInt& ind_faces_virt_bord = la_zone_VDF.zone().ind_faces_virt_bord();
136 int num_arete;
137
138 for ( num_arete=ndeb; num_arete<nfin; num_arete++)
139 {
140 num_arete_=num_arete-ndeb;
141 face1 = la_zone_VDF.Qdm(num_arete,0);
142 face2 = la_zone_VDF.Qdm(num_arete,1);
143 rang1 = face1;
144 rang2 = face2;
145 if(rang1>=sz_faces_Cl) rang1=ind_faces_virt_bord[rang1-decal_virt];

147 if (les_faces_Cl[rang1] == 0)
148 {
149 if (les_faces_Cl[rang2] == 0)

153 else if (les_faces_Cl[rang2] == 2)

157 Cerr << "On traite une arete qui separe deux faces de meme orientation : " << finl;

161 Process::Journal()<<"ERREUR Faces_CL"<<finl;

165 assert(0);

167 }
168 }

225
226 for (num_arete=ndeb; num_arete<nfin; num_arete++)
227 {
228 num_arete_=num_arete-ndeb;
229// On cherche les 2 valeurs differentes de -1
230 fac1 = la_zone_VDF.Qdm(num_arete_,0);
231 fac2 = la_zone_VDF.Qdm(num_arete_,1);
232 fac3 = la_zone_VDF.Qdm(num_arete_,2);

234
235 IntVect f(2);
236 f = -2;

238 if (fac1 != -1)
239 {
240 f(i) = fac1;

242 }

244 {
245 f(i) = fac2;
246 i++;
247 }

369//! Impose les conditions aux limites a la valeur temporelle "temps" du<br>Champ_Inc
370void Zone_Cl_VDF::imposer_cond_lim(Champ_Inc& ch, double temps)
371{
372 static int init=0;
373 Champ_Inc_base& ch_base=ch.valeur();
374 DoubleTab& ch_tab = ch_base.valeurs(temps);
375 if (sub_type(Champ_P0_VDF,ch_base))
376 ;
377 else if(ch_base.nature_du_champ()==scalaire)
378 ;
379 else if (sub_type(Champ_Face,ch_base))
380 {
381 Champ_Face& ch_face = ref_cast(Champ_Face, ch_base);
382 const Zone_VDF& ma_zone_VDF = ch_face.zone_vdf();
383 int ndeb,nfin, num_face;
384
385 for(int i=0; i<nb_cond_lim(); i++)
386 {
387 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
388 if (sub_type(Periodique,la_cl))
389 {
390 if (init == 0)
391 {
392// On fait en sorte que le champ ait la meme valeur
393// sur deux faces de periodicite qui sont en face l'une de l'autre
394 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl);
395 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
396 ndeb = le_bord.num_premiere_face();
397 nfin = ndeb + le_bord.nb_faces();
398 int voisine;
399 double moy;
400 for (num_face=ndeb; num_face<nfin; num_face++)
401 {
402 voisine = la_cl_perio.face_associee(num_face-ndeb) + ndeb;
403 if ( ch_tab[num_face] != ch_tab[voisine] )
404 {
405// Cerr << "dans Zone_Cl_VDF::imposer_cond_lim : on reajuste les vitesses!! pour la face num=" << num_face << finl;
406// Cerr << "difference = ch_tab[num_face]-ch_tab[voisine]=" << ch_tab[num_face]-ch_tab[voisine] << finl;
407 moy = 0.5*(ch_tab[num_face] + ch_tab[voisine]);
408 ch_tab[num_face] = moy;
409 ch_tab[voisine] = moy;
410 }
411 }
412// Il ne faut pas le faire a la premiere cl mais une fois toutes les cl faites une fois, cas multi perio avec ci non perio
413// init = 1;
414 }
415 }
416 else if( sub_type(Symetrie,la_cl) )
417 {
418 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
419 ndeb = le_bord.num_premiere_face();
420 nfin = ndeb + le_bord.nb_faces();
421 for (num_face=ndeb; num_face<nfin; num_face++)
422 ch_tab[num_face] = 0;
423 }
424 else if ( sub_type(Dirichlet_entree_fluide,la_cl) )
425 {
426 const Dirichlet_entree_fluide& la_cl_diri = ref_cast(Dirichlet_entree_fluide,la_cl);
427 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
428 ndeb = le_bord.num_premiere_face();
429 nfin = ndeb + le_bord.nb_faces();
430
431 for (num_face=ndeb; num_face<nfin; num_face++)
432 {
433// WEC : optimisable (pour chaque face recherche le bon temps !)
434 ch_tab[num_face] = la_cl_diri.val_imp_au_temps(temps,num_face-ndeb,ma_zone_VDF.orientation(num_face));
435 }
436 }
437 else if ( sub_type(Dirichlet_paroi_fixe,la_cl) )
438 {
439 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
440 ndeb = le_bord.num_premiere_face();
441 nfin = ndeb + le_bord.nb_faces();
442 for (num_face=ndeb; num_face<nfin; num_face++)
443 ch_tab[num_face] = 0;
444 }
445 else if ( sub_type(Dirichlet_paroi_defilante,la_cl) )
446 {
447 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
448 ndeb = le_bord.num_premiere_face();
449 nfin = ndeb + le_bord.nb_faces();
450 for (num_face=ndeb; num_face<nfin; num_face++)
451 ch_tab[num_face] = 0;
452 }
453 }
454 init = 1;
455 }
456 else
457 {
458 Cerr << "Le type de Champ_Inc " << ch->que_suis_je() << " n'est pas prevu en VDF\n";
459 exit();
460 }
461 ch_tab.echange_espace_virtuel();
462 Debog::verifier("Zone_Cl_VDF::imposer_cond_lim ch_tab",ch_tab);
463}
