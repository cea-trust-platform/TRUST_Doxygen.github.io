107//! Calcul la viscosite mu en fonction de la consistance et de<br>l'indice de structure en utilisant la loi d'Ostwald. Pour<br>des viscsites tres faible et tres fortes on utilise une regression
108
109void Champ_Ostwald_VEF::calculer_mu(DoubleTab& mu_tab)
110{
111 const double d_n = mon_fluide_->indice_struct().valeurs()(0,0);
112
113 for (int i=0 ; i<nb_valeurs_nodales() ; i++)
114 {
115 if (sub_type(Champ_Uniforme, mon_fluide_->consistance().valeur()))
116 {
117 const double d_k = mon_fluide_->consistance().valeurs()(0,0);
118 if ( mu_tab[i] < 1.E-4 )
119 mu_tab[i] = d_k * pow( 0.5 * 1.E-4 , (d_n - 1.) / 2.);
120 else if ( mu_tab[i] > 1.E16 )
121 mu_tab[i] = d_k * pow( 0.5 * 1.E16 , (d_n - 1.) / 2.);
122 else
123 mu_tab[i] = d_k * pow( 0.5 * mu_tab[i] , (d_n - 1.) / 2.);
124 }
125 else //!< K varie en fonction de la temperature
126 {
127 const DoubleTab& K_tab = mon_fluide_->consistance().valeurs();
128 if ( mu_tab[i] < 1.E-4 )
129 mu_tab[i] = K_tab[i] * pow( 0.5 * 1.E-4 , (d_n - 1.) / 2.);
130 else if ( mu_tab[i] > 1.E16 )
131 mu_tab[i] = K_tab[i] * pow( 0.5 * 1.E16 , (d_n - 1.) / 2.);
132 else
133 mu_tab[i] = K_tab[i] * pow( 0.5 * mu_tab[i] , (d_n - 1.) / 2.);
134 }
135 }
136}

144void Champ_Ostwald_VEF::calculer_dscald(DoubleTab& dscald)
145{
146 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
147 const Zone_Cl_VEF& zcl_VEF = ref_cast(Zone_Cl_VEF,eq_hydraulique->zone_Cl_dis().valeur());
148 const DoubleTab& vit = eq_hydraulique->inconnue().valeurs();
149 int nb_elem = zone_VEF.nb_elem();
150 int nb_elem_tot = zone_VEF.nb_elem_tot();
151
152 int elem;
153
154 DoubleTab gradient_elem(nb_elem_tot,dimension,dimension);
155 Champ_P1NC::calcul_gradient(vit,gradient_elem,zcl_VEF);
156
157 if (dimension == 2)
158 for (elem=0; elem<nb_elem; elem++)
159 {
160 dscald(elem) = 4. * (gradient_elem(elem, 0, 0) * gradient_elem(elem, 0, 0) +
161 gradient_elem(elem, 1, 1) * gradient_elem(elem, 1, 1)) +
162 2. * ((gradient_elem(elem, 0, 1) + gradient_elem(elem, 1, 0)) *
163 (gradient_elem(elem, 0, 1) + gradient_elem(elem, 1, 0))) ;
164 }
165 else if (dimension == 3)
166 for (elem=0; elem<nb_elem; elem++)
167 {
168 dscald(elem) = 4. * (gradient_elem(elem, 0, 0) * gradient_elem(elem, 0, 0) +
169 gradient_elem(elem, 1, 1) * gradient_elem(elem, 1, 1) +
170 gradient_elem(elem, 2, 2) * gradient_elem(elem, 2, 2)) +
171 2. *(((gradient_elem(elem, 0, 1) + gradient_elem(elem, 1, 0)) *
172 (gradient_elem(elem, 0, 1) + gradient_elem(elem, 1, 0))) +
173 ((gradient_elem(elem, 2, 0) + gradient_elem(elem, 0, 2)) *
174 (gradient_elem(elem, 2, 0) + gradient_elem(elem, 0, 2))) +
175 ((gradient_elem(elem, 2, 1) + gradient_elem(elem, 1, 2)) *
176 (gradient_elem(elem, 2, 1) + gradient_elem(elem, 1, 2)))) ;
177 }
178}
