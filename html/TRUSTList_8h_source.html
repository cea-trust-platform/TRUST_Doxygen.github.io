30class TRUSTList : public TRUSTListElem<_TYPE_>

32public :

35 TRUSTList() : TRUSTListElem<_TYPE_>()
36 {
37 min_data = std::is_same<_TYPE_, double>::value ? DMAXFLOAT : IMAX;

42// Constructeur par copie
43 TRUSTList(const TRUSTList& a_list) : TRUSTListElem<_TYPE_>()
44 {
45 min_data = a_list.min_data;
46 max_data = a_list.max_data;
47 if (a_list.est_vide()) this->suivant_ = this;
48 else
49 {
50 this->data = a_list.data;
51 dernier_ = this;
52 if (a_list.suivant_)
53 {
54 TRUSTListElem<_TYPE_> *next = new TRUSTListElem<_TYPE_>(*a_list.suivant_); //Recursif !!
55 this->suivant_ = next;
56 }
57 else this->suivant_ = 0;
58 }
59 }

91//! : List_Curseur de reels int/double precision
92template<typename _TYPE_>
93class TRUSTList_Curseur
94{
95public :
96 TRUSTList_Curseur(const TRUSTList<_TYPE_>& a_list) : curseur(&(TRUSTListElem<_TYPE_>&) a_list) { if (a_list.est_vide()) curseur = 0; }
97 TRUSTList_Curseur(const TRUSTListElem<_TYPE_>& a_list) : curseur(&(TRUSTListElem<_TYPE_>&) a_list) { if (a_list.est_vide()) curseur = 0; }
98
99 inline operator bool() const { return (curseur != 0); } //!< Teste si le curseur est non nul
100
101// Avance le curseur dans la liste. Si le curseur est sur le dernier element, il devient nul
102 inline void operator++() { curseur = curseur->est_dernier() ? 0 : &(curseur->suivant()); }
103
104// retourne la valeur du curseur
105 inline _TYPE_ valeur() const { return curseur->valeur(); }
106 inline _TYPE_& valeur() { return curseur->valeur(); }
107
108// Affectation d'une liste a une liste_curseur
109 inline void operator=(const TRUSTList<_TYPE_>& a_list)
110 {
111 curseur = (&(TRUSTListElem<_TYPE_>&) a_list);
112 if (a_list.est_vide()) curseur = 0;
113 }
114
115// Retourne la liste associee a la liste_curseur
116 inline const TRUSTListElem<_TYPE_>& list() const { return *curseur; }
117 inline TRUSTListElem<_TYPE_>& list() { return *curseur; }
118
119private :
120 TRUSTListElem<_TYPE_>* curseur;
121};
