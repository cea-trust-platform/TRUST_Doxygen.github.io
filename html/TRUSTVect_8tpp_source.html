21// Taille de l'espace "reel" du vecteur. (si md_vector_ est nul, cette valeur est identique a size_array(),
22// sinon, soit elle est egale a md_vector_....get_nb_items(),
23// soit l'appel est invalide (en particulier si les items reels ne sont pas regroupes en debut de tableau)
24// On peut interroger le vecteur pour savoir si size_reelle() est valide avec size_reelle_ok()
25template<typename _TYPE_>
26inline int TRUSTVect<_TYPE_>::size_reelle() const
27{
28// Si cet assert plante, c'est que l'appel a ete declare invalide par le MD_Vect associe a ce vecteur.
29 assert(size_reelle_ >= 0);
30// Si cet assert plante, c'est que le tableau a ete redimensionne avec resize_array() au lieu de resize(). (invalide pour un Vect ou un Tab).
31 assert(TRUSTArray<_TYPE_>::size_array() == size_reelle_ || md_vector_.non_nul());
32 return size_reelle_;
33}

137// detache le tableau et l'attache a v (sauf si v==*this, ne fait rien)
138// Fait pointer le tableau sur la meme zone de memoire que v et copie le MD_Vector (utilise ArrOfDouble/Int::attach_array())
139// Attention, il devient alors interdit de resizer le tableau v ainsi que *this
140// Methode virtuelle reimplementee dans DoubleTab
141// Precondition:
142// L'objet ne doit pas etre un sous-type de DoubleVect (sinon mauvaise initialisation des dimensions.
143template<typename _TYPE_>
144inline void TRUSTVect<_TYPE_>::ref(const TRUSTVect& v)
145{
146 if (&v != this)
147 {
148 TRUSTArray<_TYPE_>::detach_array();
149 TRUSTArray<_TYPE_>::attach_array(v);
150 md_vector_ = v.md_vector_;
151 size_reelle_ = v.size_reelle_;

235// associe le md_vector au vecteur et initialise l'attribut size_reelle_ (voir methode size_reelle())
236// Si md_vector est nul, detache simplement le md_vector existant.
237// Precondition: le vecteur doit deja avoir la taille appropriee au nouveau md_vector, c'est a dire md_vector...get_nb_items_tot() * line_size_
238template<typename _TYPE_>
239inline void TRUSTVect<_TYPE_>::set_md_vector(const MD_Vector& md_vector)
240{
241 int size_r = TRUSTArray<_TYPE_>::size_array();
242 if (md_vector.non_nul())
243 {
244 size_r = md_vector.valeur().get_nb_items_reels();
245 if (size_r >= 0) size_r *= line_size_;
246 else size_r = -1; //!< Cas particulier ou la size_reelle ne veut rien dire
247
248 int size_tot = md_vector.valeur().get_nb_items_tot() * line_size_;
249 if (size_tot != TRUSTArray<_TYPE_>::size_array())
250 {
251 Cerr << "Internal error in TRUSTVect::set_md_vector(): wrong array size\n"
252 << " Needed size = " << md_vector.valeur().get_nb_items_tot() << " x " << line_size_
253 << "\n Actual size = " << TRUSTArray<_TYPE_>::size_array() << finl;
254 Process::exit();
255 }
256 if (line_size_ == 0)
257 {
258 Cerr << "Internal error in TRUSTVect::set_md_vector():\n"
259 << " cannot attach descriptor to empty array (line_size_ is zero)" << finl;
260 Process::exit();
261 }
262 }
263 size_reelle_ = size_r;
264 md_vector_ = md_vector;
265}

285// lecture d'un tableau pour reprise de calcul. On lit les valeurs "raw".
286// Attention, si le tableau n'est pas vide, il doit deja avoir la bonne taille et la bonne structure, sinon erreur !
287// Parameter resize_and_read if the array is sized AND read (by default, yes)
288template<typename _TYPE_>
289inline void TRUSTVect<_TYPE_>::lit(Entree& is, int resize_and_read)
290{
291 int sz = -1;
292 is >> sz;
293 if (resize_and_read)
294 {
295 if (TRUSTArray<_TYPE_>::size_array() == 0 && (!get_md_vector().non_nul()))
296 resize(sz, Array_base::NOCOPY_NOINIT);
297 else if (sz != TRUSTArray<_TYPE_>::size_array())
298 {
299// Si on cherche a relire un tableau de taille inconnue, le tableau doit etre reset() a l'entree. On n'aura pas la structure parallele du tableau !
300 Cerr << "Error in TRUSTVect::lit(Entree & is): array has already a structure with incorrect size" << finl;
301 Process::exit();
302 }
303 is.get(TRUSTArray<_TYPE_>::addr(), sz);
304 }
305 else
306 {
307// May be slow if large chunks are read:
308// Double tmp;
309// for (int i=0;i<sz;i++) is >> tmp;
310// So we bufferize:
311 const int sz_max = 128000;
312 int buffer_size = std::min(sz,sz_max);
313 TRUSTArray<_TYPE_> tmp(buffer_size);
314 while(sz>buffer_size)
315 {
316 is.get(tmp.addr(), buffer_size);
317 sz-=buffer_size;
318 }
319 is.get(tmp.addr(), sz);
320 }
321 int sz_reel = -2;
322 is >> sz_reel;
323 if (sz_reel >= 0)
324 {
325// Lecture de l'ancien format. Ignore les valeurs lues.
326 int sz_virt;
327 is >> sz_virt;
328 DescStructure toto;
329 is >> toto;
330 if (std::is_same<_TYPE_,double>::value)
331 {
332 TRUSTArray<int> it_communs;
333 is >> it_communs;
334 TRUSTArray<int> it_communs_tot;
335 is >> it_communs_tot;
336 }
337 }
