106DoubleVect& Matrice_Bloc::ajouter_multvect_( const DoubleVect& x,
107 DoubleVect& r ) const
108{
109 const double * const_x_addr = x.addr( );
110 double * x_addr = ( double * ) const_x_addr;
111 double * r_addr = r.addr( );
112
113 DoubleVect x_bloc; //!< Une sous-partie de x
114 DoubleVect r_bloc; //!< Une sous-partie de y
115
116 int r_bloc_debut = 0; //!< Premier element de r du bloc courant Boucle sur les lignes de blocs
117//
118 for ( int i_ligne = 0; i_ligne < N_; ++i_ligne )
119 {
120 const int r_bloc_size = get_bloc( i_ligne, 0 ).valeur( ).nb_lignes( );
121 r_bloc.ref_data( r_addr + r_bloc_debut, r_bloc_size );
122 assert( r_bloc_debut+r_bloc_size <= r.size_array( ) );
123 int x_bloc_debut = 0; //!< Premier element de x du bloc courant Boucle sur les colonnes de blocs
124//
125 for ( int i_colonne = 0; i_colonne < M_; ++i_colonne )
126 {
127 const Matrice& sub_bloc = get_bloc( i_ligne, i_colonne );
128 const int x_bloc_size = sub_bloc.valeur( ).nb_colonnes( );
129 x_bloc.ref_data( x_addr + x_bloc_debut, x_bloc_size );
130 assert( x_bloc_debut+x_bloc_size <= x.size_array( ) );
131 sub_bloc.valeur( ).ajouter_multvect_( x_bloc, r_bloc );
132 x_bloc_debut += x_bloc_size;
133 }
134 r_bloc_debut += r_bloc_size;
135 }
136 return r;
137}

198void Matrice_Bloc::get_stencil( IntTab& stencil ) const
199{
200 assert_check_block_matrix_structure( );
201 if( is_stencil_up_to_date_ )
202 {
203 stencil = stencil_;
204 return;
205 }
206
207 const int nb_line_blocks = nb_bloc_lignes( );
208 const int nb_column_blocks = nb_bloc_colonnes( );
209 const int nb_stencils = nb_line_blocks * nb_column_blocks;
210
211 VECT( IntTab ) local_stencils;
212 local_stencils.dimensionner( nb_stencils );
213
214 int imin = 0;
215 int imax = 0;
216 int jmin = 0;
217 int jmax = 0;
218
219 for ( int i=0; i<nb_line_blocks; ++i )
220 {
221 for ( int j=0; j<nb_column_blocks; ++j )
222 {
223 const Matrice_Base& local_matrix = get_bloc( i, j ).valeur( );
224
225 imax = imin + local_matrix.nb_lignes( );
226 jmax = jmin + local_matrix.nb_colonnes( );
227
228 IntTab& local_stencil = local_stencils[ i * nb_column_blocks + j ];
229 local_matrix.get_stencil( local_stencil );
230
231 const int size = local_stencil.dimension( 0 );
232 for ( int k=0; k<size; ++k )
233 {
234 local_stencil( k, 0 ) += imin;
235 local_stencil( k, 1 ) += jmin;
236 }
237 jmin = jmax;
238 }
239 imin = imax;
240 jmin = 0;
241 }
242
243 const int nb_lines = nb_lignes( );
244 ArrOfInt offsets( nb_lines + 1 );
245 offsets[ 0 ] = 0;
246
247 for ( int i=0; i<nb_stencils; ++i )
248 {
249 const IntTab& local_stencil = local_stencils[ i ];
250 const int size = local_stencil.dimension( 0 );
251
252 for ( int k=0; k<size; ++k )
253 {
254 const int line = local_stencil( k, 0 );
255 offsets[ line + 1 ] += 1;
256 }
257 }
258
259 for ( int i=0; i<nb_lines; ++i )
260 {
261 offsets[ i + 1 ] += offsets[ i ];
262 }
263
264 const int stencil_size = offsets[ nb_lines ];
265 stencil.resize( stencil_size, 2 );
266
267 stencil = -1;
268
269 for ( int i=0; i<nb_stencils; ++i )
270 {
271 const IntTab& local_stencil = local_stencils[ i ];
272 const int size = local_stencil.dimension( 0 );
273
274 for ( int k=0; k<size; ++k )
275 {
276 const int line = local_stencil( k, 0 );
277 const int column = local_stencil( k, 1 );
278 const int index = offsets[ line ];
279
280 assert( stencil( index, 0 ) < 0 );
281 assert( stencil( index, 1 ) < 0 );
282 assert( index < offsets[ line + 1 ] );
283
284 stencil( index, 0 ) = line;
285 stencil( index, 1 ) = column;
286 offsets[ line ] += 1;
287 }
288 }

384void Matrice_Bloc::get_stencil_and_coefficients( IntTab& stencil, ArrOfDouble& coefficients ) const
385{
386 assert_check_block_matrix_structure( );
387 const int nb_line_blocks = nb_bloc_lignes( );
388 const int nb_column_blocks = nb_bloc_colonnes( );
389
390 int imin = 0;
391 int imax = 0;
392 int jmin = 0;
393 int jmax = 0;
394
395 if( is_stencil_up_to_date_ )
396 {
397 stencil = stencil_;
398 ArrOfInt offsets = offsets_;
399 const int stencil_size = stencil_.dimension(0);
400 coefficients.resize_array( stencil_size );
401
402 IntTab local_stencil;
403 ArrOfDouble local_coeff;
404
405 for ( int i=0; i<nb_line_blocks; ++i )
406 {
407 for ( int j=0; j<nb_column_blocks; ++j )
408 {
409 const Matrice_Base& local_matrix = get_bloc( i, j ).valeur( );
410
411 imax = imin + local_matrix.nb_lignes( );
412 jmax = jmin + local_matrix.nb_colonnes( );
413
414 local_matrix.get_stencil_and_coefficients( local_stencil, local_coeff );
415
416 const int size = local_stencil.dimension( 0 );
417 for ( int k=0; k<size; ++k )
418 {
419 const int line = local_stencil( k, 0 ) + imin;
420 const double coefficient = local_coeff[ k ];
421 const int index = offsets[ line ];
422
423 coefficients[ index ] = coefficient;
424
425 offsets[ line ] += 1;
426 }
427 jmin = jmax;
428 }
429 imin = imax;
430 jmin = 0;
431 }
432 return;
433 }
434 const int nb_stencils = nb_line_blocks * nb_column_blocks;
435
436 VECT( IntTab ) vect_local_stencils;
437 vect_local_stencils.dimensionner( nb_stencils );
438
439 VECT( ArrOfDouble ) vect_local_coefficients;
440 vect_local_coefficients.dimensionner( nb_stencils );
441
442 for ( int i=0; i<nb_line_blocks; ++i )
443 {
444 for ( int j=0; j<nb_column_blocks; ++j )
445 {
446 const Matrice_Base& local_matrix = get_bloc( i, j ).valeur( );
447
448 imax = imin + local_matrix.nb_lignes( );
449 jmax = jmin + local_matrix.nb_colonnes( );
450
451 IntTab& local_stencil = vect_local_stencils[ i * nb_column_blocks + j ];
452 ArrOfDouble& local_coefficients = vect_local_coefficients[i * nb_column_blocks + j ];
453 local_matrix.get_stencil_and_coefficients( local_stencil, local_coefficients );
454
455 const int size = local_stencil.dimension( 0 );
456 for ( int k=0; k<size; ++k )
457 {
458 local_stencil( k, 0 ) += imin;
459 local_stencil( k, 1 ) += jmin;
460 }
461 jmin = jmax;
462 }
463 imin = imax;
464 jmin = 0;
465 }
466
467 const int nb_lines = nb_lignes( );
468 ArrOfInt offsets( nb_lines + 1 );
469 offsets[ 0 ] = 0;
470
471 for ( int i=0; i<nb_stencils; ++i )
472 {
473 const IntTab& local_stencil = vect_local_stencils[ i ];
474 const int size = local_stencil.dimension( 0 );
475
476 for ( int k=0; k<size; ++k )
477 {
478 const int line = local_stencil( k, 0 );
479 offsets[ line + 1 ] += 1;
480 }
481 }
482
483
484 for ( int i=0; i<nb_lines; ++i )
485 {
486 offsets[ i + 1 ] += offsets[ i ];
487 }
488
489 const int stencil_size = offsets[ nb_lines ];
490 stencil.resize( stencil_size, 2 );
491 coefficients.resize_array( stencil_size );
492
493 stencil = -1;
494
495 for ( int i=0; i<nb_stencils; ++i )
496 {
497 const IntTab& local_stencil= vect_local_stencils[ i ];
498 const ArrOfDouble& local_coefficients = vect_local_coefficients[ i ];
499
500 const int size = local_stencil.dimension( 0 );
501 assert( local_coefficients.size_array( ) == size );
502
503 for ( int k=0; k<size; ++k )
504 {
505 const int line = local_stencil( k, 0 );
506 const int column = local_stencil( k, 1 );
507 const double coefficient = local_coefficients[ k ];
508 const int index = offsets[ line ];
509
510 assert( stencil( index, 0 ) < 0 );
511 assert( stencil( index, 1 ) < 0 );
512 assert( index < offsets[ line + 1 ] );
513
514 stencil( index, 0 ) = line;
515 stencil( index, 1 ) = column;
516 coefficients[ index ] = coefficient;
517
518 offsets[ line ] += 1;
519 }
520 }
521}

663// Remplissage partiel (i premieres lignes, j premieres colonnes) d'une matrice bloc par une matrice morse symetrique ou non
664// RR | RV
665// ------
666// VR | VV
667void Matrice_Bloc::remplir(const IntLists& voisins, const DoubleLists& valeurs, const DoubleVect& terme_diag, const int i, const int n, const int j, const int m)
668{
669 Matrice_Morse& RR=ref_cast(Matrice_Morse,get_bloc(0,0).valeur());
670 Matrice_Morse& RV=ref_cast(Matrice_Morse,get_bloc(0,1).valeur());
671 Matrice_Morse& VR=ref_cast(Matrice_Morse,get_bloc(1,0).valeur());
672 Matrice_Morse& VV=ref_cast(Matrice_Morse,get_bloc(1,1).valeur());
673
674// Premiere passe pour le dimensionnement
675 int RR_compteur, RR_rang=0;
676 int RV_compteur, RV_rang=0;
677 int VR_compteur, VR_rang=0;
678 int VV_compteur, VV_rang=0;
679
680 int num_elem;
681 for (num_elem=0; num_elem<n; num_elem++)

847 RR.formeF();
848 RV.formeF();

850 VV.formeF();
851
852// Compactage de la matrice

854 RV.compacte();
855 VR.compacte();
