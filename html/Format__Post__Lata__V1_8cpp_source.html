154// On intercepte uniquement les tableaux FACES et ELEM_FACES
155// qui sont ecrits dans le meme fichier que le domaine.
156int Format_Post_Lata_V1::ecrire_faces_lata_V1(const Nom& basename,
157 const Format& format,
158 const Options_Para& option,
159 const Nom& id_du_domaine,
160 const IntTab& faces,
161 const IntTab& elem_faces,
162 const int nb_sommets)
163{
164 Nom nom_fichier_geom;
165 int nb_faces_tot = -1;
166 int decalage_faces = 1;
167 {
168 Nom basename_geom(basename);
169 Nom extension_geom(extension_lata());
170 extension_geom += ".";
171 extension_geom += id_du_domaine;
172 Fichier_Lata fichier_geom(basename_geom, extension_geom,
173 Fichier_Lata::APPEND, format, option);
174 nom_fichier_geom = fichier_geom.get_filename();
175
176// Faces
177// Les indices de sommets, elements et autres dans le fichier lata commencent a 1 :
178 int nb_col;
179 int decal;
180 if (option == SINGLE_FILE && Process::nproc() > 1)
181 {
182// Decalage a ajouter aux indices pour avoir une numerotation globale.
183 decal = mppartial_sum(nb_sommets) + 1;
184 decalage_faces = mppartial_sum(faces.dimension(0)) + 1;
185 }
186 else
187 {
188 decal = 1;
189 decalage_faces = 1;
190 }
191 nb_faces_tot = Format_Post_Lata::write_inttab(fichier_geom,
192 1, decal,
193 faces,
194 nb_col,
195 option);
196
197 Format_Post_Lata::write_inttab(fichier_geom,
198 1, decalage_faces,
199 elem_faces,
200 nb_col,
201 option);
202 }
203
204 Fichier_Lata_maitre fichier_lata(basename, extension_lata(),
205 Fichier_Lata::APPEND, option);
206 SFichier& sfichier = fichier_lata.get_SFichier();
207
208 Nom filename_joint_faces;
209 if (option == SINGLE_FILE && Process::nproc() > 1)
210 {
211 Nom extension;
212 extension = extension_lata();
213 extension += ".";
214 extension += "decal_faces_proc";
215 extension += ".";
216 extension += id_du_domaine;
217 {
218 Fichier_Lata fichier_champ(basename, extension,
219 Fichier_Lata::ERASE, format, option);
220 filename_joint_faces = fichier_champ.get_filename();
221 int nb_col = -1; //!< non utilise
222 IntTab data(1,2);
223 data(0, 0) = decalage_faces;
224 data(0, 1) = faces.dimension(0);
225 Format_Post_Lata::write_inttab(fichier_champ,0, 0 /* ne pas ajouter de decalage */,
226 data, nb_col, option);
227 }
228 }
229
230 if (fichier_lata.is_master())
231 {
232 sfichier << "nb_faces_tot=" << nb_faces_tot << finl;
233 if (option == SINGLE_FILE && Process::nproc() > 1)
234 {
235 const int nproc = Process::nproc();
236 sfichier << "joints_faces nproc=" << nproc << " decalage=" << remove_path(filename_joint_faces) << finl;
237 }
238 }
239 fichier_lata.syncfile();
240
241 return 1; //!< ok tout va bien
242}

245int Format_Post_Lata_V1::ecrire_domaine_lata_V1(const Nom& id_domaine,
246 const Motcle& type_elem,
247 const int dim,
248 const DoubleTab& sommets,
249 const IntTab& elements,
250 const Status& stat, const Nom& basename, const Format&
251 format, const Options_Para& option)
252{
253
254 if (stat > INITIALIZED)
255 {
256 return 0; //!< Ecriture tardive des domaines dynamiques non supportee
257 }
258
259// Construction du nom du fichier de geometrie
260 Nom basename_geom(basename);
261 Nom extension_geom(extension_lata());
262 extension_geom += ".";
263 extension_geom += id_domaine;
264
265 Nom nom_fichier_geom;
266// Indice global attribue au premier sommet de ce processeur:
267 int decalage_sommets = -1;
268 int decalage_elements = -1;
269 {
270 Fichier_Lata fichier_geom(basename_geom, extension_geom,
271 Fichier_Lata::ERASE, format,option);
272
273 nom_fichier_geom = fichier_geom.get_filename();
274 int nb_col;
275// Coordonnees des sommets
276 const int nb_sommets = sommets.dimension(0);
277
278 if (dim==2)
279 {
280// Ecriture en deux fois : tous les processeurs ecrivent une premiere rangee
281// de sommets, puis tous les processeurs ecrivent une deuxieme rangee.
282 for (int bloc_number = 0; bloc_number < 2; bloc_number++)
283 {
284 DoubleTab sommets2(nb_sommets, 3);
285 const double z = (bloc_number == 0) ? 0. : 1.;
286 for (int i = 0; i < nb_sommets; i++)
287 {
288 double x = sommets(i,0);
289 double y = sommets(i,1);
290 if (axi)
291 {
292 double r = x;
293 double theta = y;
294 x = r * cos(theta);
295 y = r * sin(theta);
296 }
297 sommets2(i,0) = x;
298 sommets2(i,1) = y;
299 sommets2(i,2) = z;
300 }
301 Format_Post_Lata::write_doubletab(fichier_geom, sommets2, nb_col, option,
302 bloc_number, 2);
303 }
304
305 }
306 else
307 {
308
309 if (axi)
310 {
311 DoubleTab sommets2(sommets);
312 for (int i = 0; i < nb_sommets; i++)
313 {
314 double r = sommets(i,0);
315 double theta = sommets(i,1);
316 double z = sommets(i,2);
317 sommets2(i,0) = r*cos(theta);
318 sommets2(i,1) = r*sin(theta);
319 sommets2(i,2) = z;
320 }
321 Format_Post_Lata::write_doubletab(fichier_geom, sommets2, nb_col,option);
322 }
323 else
324 {
325 Format_Post_Lata::write_doubletab(fichier_geom, sommets, nb_col,option);
326 }
327 }
328// Elements
329// Les indices de sommets, elements et autres dans le fichier lata commencent a 1 :
330 const int nb_elem = elements.dimension(0);
331 if (option == SINGLE_FILE && Process::nproc() > 1)
332 {
333// Tous les processeurs ecrivent dans le meme fichier. Il faut renumeroter les
334// sommets des elements pour avoir une numerotation globale.
335// On ajoute un pour numerotation fortran
336 decalage_sommets = 1 + mppartial_sum(nb_sommets);
337 decalage_elements = 1 + mppartial_sum(nb_elem);
338 }
339 else
340 {
341// Chaque processeur ecrit dans un fichier separe avec des indices locaux.
342// On ajoute un pour numerotation fortran
343 decalage_sommets = 1;
344 decalage_elements = 1;
345 }
346 if (dim==2)
347 {
348// Decalage a ajouter a un indice de sommet pour passer de la rangee z=0
349// a la rangee z=1
350 int decalage_z1 = nb_sommets;
351 if (option == SINGLE_FILE && Process::nproc() > 1)
352 decalage_z1 = mp_sum(nb_sommets);
353 const int nsomelem = (elements.nb_dim() > 1) ? elements.dimension(1) : 1;
354 IntTab elem2(nb_elem, nsomelem*2);
355 for (int i = 0; i < nb_elem; i++)
356 {
357 for (int j = 0; j < nsomelem; j++)
358 {
359 const int s = elements(i,j);
360 elem2(i,j) = s;
361 elem2(i,j + nsomelem) = s + decalage_z1;
362 }
363 }
364 Format_Post_Lata::write_inttab(fichier_geom, 1,decalage_sommets, elem2, nb_col, option);
365 }
366 else
367 {
368 Format_Post_Lata::write_inttab(fichier_geom, 1,decalage_sommets, elements, nb_col, option);
369 }
370 }
371// En mode parallele, on ecrit en plus des fichiers contenant les donnees paralleles
372// sur les sommets, les elements et les faces...
373// On ecrit dans le fichier maitre ceci :
374// joints_sommets NPROC FICHIER_NB_SOMMETS FICHIER_NB_SOMMETS_JOINTS FICHIER_JOINTS
375// NPROC est le nombre de parties
376// FICHIER_NB_SOMMETS est le nom du fichier contenant, pour chaque proc, le nombre de
377// sommets sur ce processeur
378// FICHIER_JOINTS est le nom du fichier contenant les joints
379// Dans FICHIER_NB_SOMMETS, on a un tableau contenant NPROC entiers, le nombre de sommets
380// ecrits par chaque processeur
381// Dans FICHIER_NB_SOMMETS_JOINTS, on a un tableau de NPROC entiers contenant pour chaque
382// proc le nombre de sommets de joint
383// Dans FICHIER_JOINTS on a un tableau contenant les sommets de joint de chaque processeur,
384// en numerotation globale : pour chaque sommet, indice du sommet local, indice du sommet
385// equivalent sur une autre sous-partie, numero de la sous-partie
386 Nom filename_decal_som;
387 Nom filename_decal_elem;
388 if (option == SINGLE_FILE && Process::nproc() > 1)
389 {
390 Nom extension;
391 extension = extension_lata();
392 extension += ".";
393 extension += "decal_som_proc";
394 extension += ".";
395 extension += id_domaine;
396
397 {
398 Fichier_Lata fichier_champ(basename, extension,
399 Fichier_Lata::ERASE, format, option);
400 filename_decal_som = fichier_champ.get_filename();
401 int nb_col = -1; //!< non utilise
402 IntTab data(1,2);
403 data(0, 0) = decalage_sommets;
404 data(0, 1) = sommets.dimension(0);
405 Format_Post_Lata::write_inttab(fichier_champ,0, 0 /* ne pas ajouter de decalage */,
406 data, nb_col, option);
407 }
408
409 extension = extension_lata();
410 extension += ".";
411 extension += "decal_elem_proc";
412 extension += ".";
413 extension += id_domaine;
414
415 {
416 Fichier_Lata fichier_champ(basename, extension,
417 Fichier_Lata::ERASE, format, option);
418 filename_decal_elem = fichier_champ.get_filename();
419 int nb_col = -1; //!< non utilise
420 IntTab data(1,2);
421 data(0, 0) = decalage_elements;
422 data(0, 1) = elements.dimension(0);
423 Format_Post_Lata::write_inttab(fichier_champ,0, 0 /* ne pas ajouter de decalage */,
424 data, nb_col, option);
425 }
426 }
427
428 Fichier_Lata_maitre fichier_lata(basename, extension_lata(),
429 Fichier_Lata::APPEND,option);
430 SFichier& sfichier = fichier_lata.get_SFichier();
431
432 int nb_som_tot = sommets.dimension(0);
433 int nb_elem_tot = elements.dimension(0);
434 if (option == SINGLE_FILE && Process::nproc() > 1)
435 {
436 nb_som_tot = mp_sum(nb_som_tot);
437 nb_elem_tot = mp_sum(nb_elem_tot);
438 }
439 if (fichier_lata.is_master())
440 {
441 sfichier << "Geometrie " << id_domaine;
442 sfichier << " " << remove_path(nom_fichier_geom) << finl;
443 switch(format)
444 {
445 case ASCII:
446 sfichier << "Format=ascii" << finl;
447 break;
448 case BINAIRE:
449 sfichier << "Format=binary" << finl;
450 break;
451 default:
452 Cerr << "Error in Format_Post_Lata_V1::ecrire_entete\n"
453 << " format not supported" << finl;
454 exit();
455 }
456 sfichier << "nb_som_tot=" << nb_som_tot << finl;
457 sfichier << "type_elem=" << type_elem << finl;
458 sfichier << "nb_elem_tot=" << nb_elem_tot << finl;
459 if (option == SINGLE_FILE && Process::nproc() > 1)
460 {
461 const int nproc = Process::nproc();
462 sfichier << "joints_sommets nproc=" << nproc << " decalage=" << remove_path(filename_decal_som) << finl;
463 sfichier << "joints_elements nproc=" << nproc << " decalage=" << remove_path(filename_decal_elem) << finl;
464 }
465 }
466 fichier_lata.syncfile();
467 return 1;
468}

514int Format_Post_Lata_V1::ecrire_champ_lata_V1(const Nom& id_du_champ,
515 const Nom& id_du_domaine,
516 const Nom& localisation,
517 const DoubleTab& data,const double temps,const Nom& basename,
518 const Format& format, const Options_Para& option)
519{
520// Verification que le nom du champ correspond bien au
521// nombre de coposante du champ grace a latav1_component_shape()
522// Cette partie de codage pourra etre supprime quand le plugin
523// LATA sera en mesure de gerer cela lui-meme (V2 ?)
524 int nb_comp,nb_comp_du_champ;
525 {
526 int shape=1;
527 shape=latav1_component_shape(id_du_champ);
528
529 Nom avec_s;
530 avec_s="";
531 nb_comp=1;
532 if (shape<0)
533 {
534 avec_s="s";
535 if (shape==-1)
536 nb_comp=dimension;
537 else if (dimension==3)
538 nb_comp=dimension;
539 }
540 else
541 nb_comp=shape;
542
543 nb_comp_du_champ=data.nb_dim();
544 if (nb_comp_du_champ>1)
545 nb_comp_du_champ=data.dimension(1);
546
547// On exclus tous les cas ou le champ qui est postraite aux faces, est scalaire mais interprete comme vectoriel
548// Typiquement le cas de la vitesse aux faces en VDF : nb_comp = 2 ou 3 et nb_comp_du_champ = 1 car 1 seul vitesse normal par face.
549 if ((Motcle)localisation=="FACES" && nb_comp_du_champ<nb_comp)
550 nb_comp_du_champ=nb_comp;
551
552 Nom avec_ss;
553 avec_ss="";
554 if (nb_comp_du_champ>1)
555 avec_ss="s";
556 }
557
558// Modification de id_du_champ car l interpolation modifie le nom (ajout de elem_nom_du_domaine
559// ou som_nom_du_domaine, on supprime cette extension)
560 Motcle id_du_champ_modifie(id_du_champ);
561 Motcle iddomaine(id_du_domaine);
562
563 if (nb_comp!=nb_comp_du_champ)
564 {
565 id_du_champ_modifie=Nom(nb_comp_du_champ);
566 id_du_champ_modifie+="_";
567 id_du_champ_modifie+=id_du_champ;
568 }
569// On utilise prefix avec un argument en majuscule
570 if ((Motcle)localisation=="SOM")
571 {
572 id_du_champ_modifie.prefix(id_du_domaine);
573 id_du_champ_modifie.prefix(iddomaine);
574//
575 id_du_champ_modifie.prefix("_SOM_");
576 }
577 else if ((Motcle)localisation=="ELEM")
578 {
579 id_du_champ_modifie.prefix(id_du_domaine);
580 id_du_champ_modifie.prefix(iddomaine);
581//
582 id_du_champ_modifie.prefix("_ELEM_");
583 }
584 else if ((Motcle)localisation=="FACES")
585 {
586 id_du_champ_modifie.prefix(id_du_domaine);
587 id_du_champ_modifie.prefix(iddomaine);
588 id_du_champ_modifie.prefix("_FACES_");
589 }
590
591 if ((Motcle)id_du_champ_modifie=="VITESSEX")
592 id_du_champ_modifie="VITESSEX";
593 if ((Motcle)id_du_champ_modifie=="VITESSEY")
594 id_du_champ_modifie="VITESSEY";
595
596 if ((Motcle)id_du_champ_modifie=="EPS")
597 id_du_champ_modifie="EPS";
598
599 const char * point = ".";
600// Construction du nom du fichier
601 Nom basename_champ(basename);
602 Nom extension_champ(extension_lata());
603 extension_champ += point;
604 extension_champ += id_du_champ_modifie;
605 extension_champ += point;
606 Motcle loc = localisation;
607
608 if (loc.debute_par("SOM"))
609 {
610 extension_champ += "SOM";
611 }
612 else if (loc.debute_par("ELEM"))
613 {
614 extension_champ += "ELEM";
615 }
616 else if (loc.debute_par("FACES"))
