56Entree& Terme_Source_Canal_RANS_LES_VDF_Elem::readOn(Entree& is )
57{
58 int compteur = 0;
59 Motcle mot_lu;
60 Motcle acc_ouverte("{");
61 Motcle acc_fermee("}");
62 nom_pb_rans="non_couple";
63
64 t_av=-1;
65 f_start = 0.;
66 moyenne = 0;//valeur par defaut
67// 0 => moyenne spatiale
68// 1 => moyenne temporelle glissante (moyenne en alpha)
69// 2 => moyenne temporelle
70
71 Motcles les_mots(6);
72 {
73 les_mots[0]="alpha_tau"; //coeff de relaxation du terme source
74 les_mots[1]="Ly"; //Hauteur du canal plan (utile pour la moyenne spatiale)
75 les_mots[2]="f_start"; //temps a partir duquel le terme source est active
76 les_mots[3]="t_av"; //temps de prise de moyenne (moyenne temporelle glissante)
77 les_mots[4]="type_moyenne"; //
78 les_mots[5]="nom_pb_rans"; //
79 }
80 is >> mot_lu;
81 if(mot_lu != acc_ouverte)
82 {
83 Cerr << "On attendait { a la place de " << mot_lu
84 << " lors de la lecture des parametres de la loi de paroi " << finl;
85 }
86 is >> mot_lu;
87 while(mot_lu != acc_fermee)
88 {
89 int rang=les_mots.search(mot_lu);
90 switch(rang)
91 {
92 case 0 :
93 is >> alpha_tau;
94 Cerr << "alpha_tau = " << alpha_tau << finl;
95 compteur++;
96 break;
97 case 1 :
98 is >> Ly;
99 Cerr << "Ly = "<< Ly << finl;
100 compteur++;
101 break;
102 case 2 :
103 is >> f_start;
104 Cerr << "f_start = "<< f_start << finl;
105 compteur++;
106 break;
107 case 3 :
108 is >> t_av;
109 Cerr << "t_av = " << t_av << finl;
110 Cerr << "type_moyenne : " << moyenne << finl;
111 if((moyenne==1)&&(t_av<=0))
112 {
113 Cerr << "La periode en temps de la moyenne temporelle glissante est non precisee !"
114 << finl;
115 exit();
116 }
117 compteur++;
118 break;
119 case 4 :
120 is >> moyenne;
121 Cerr << "type_moyenne : " << moyenne << finl;
122 compteur++;
123 break;
124 case 5 :
125 is >> nom_pb_rans;
126
127 compteur++;
128 break;
129 default :
130 {
131 Cerr << mot_lu << " n'est pas un mot compris" << finl;
132 Cerr << "Les mots compris sont : " << les_mots << finl;
133 exit();
134 }
135 }
136 is >> mot_lu;
137 }
138 Cerr << "nom_pb_rans = " << nom_pb_rans << finl;
139 Cerr << compteur << " motcles ont ete lus dans le readOn du terme de forcage thermique" << finl;
140
141 init();
142
143 return is;
144

228void Terme_Source_Canal_RANS_LES_VDF_Elem::mettre_a_jour(double temps)
229{
230// Cerr << "Je suis dans le mettre_a_jour" << finl;
231
232 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
233
234// vitesse=temperature
235
236 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
237 const double dt = mon_equation->schema_temps().pas_de_temps();
238 const double tps = mon_equation->schema_temps().temps_courant();
239 int nb_elems = zone_VDF.nb_elem();
240 const double dt_min = mon_equation->schema_temps().pas_temps_min();
241
242 int cptbis=0;
243
244 //****************************************************
245 //******* MaJ de la vitesse cible (RANS) ***********
246 //**************************************************
247 if(nom_pb_rans != "non_couple")
248 {
249 REF(Probleme_base) pb_rans;
250 Objet_U& obj=Interprete::objet(nom_pb_rans);
251
252 if( sub_type(Probleme_base, obj) )
253 {
254 pb_rans = ref_cast(Probleme_base, obj);
255 }
256
257 U_RANS = pb_rans.valeur().equation(1).inconnue().valeurs();
258
259 }
260 //***************************************************
261
262 if(moyenne==2)
263 {
264 //******************************************************
265 //*************** MOYENNE TEMPORELLE *******************
266 //******************************************************
267
268 //Calcul d'une premiere moyenne temporelle significative
269
270 if((tps>(f_start-t_av))&&(tps<f_start))
271 {
272 if (cptbis==0)
273 {
274 for (int num_elem=0; num_elem<nb_elems; num_elem++)
275 {
276 utemp_sum(num_elem) = vitesse(num_elem)*(tps-(f_start-t_av));
277 umoy(num_elem) = vitesse(num_elem);
278 }
279 cptbis = 3;
280 }
281 else
282 {
283 for (int num_elem=0; num_elem<nb_elems; num_elem++)
284 {
285 utemp_sum(num_elem) += vitesse(num_elem)*dt;
286 umoy(num_elem) = utemp_gliss(num_elem)/(tps-(f_start-t_av));
287 }
288 }
289
290 }
291
292
293 if(tps>=f_start)
294 {
295 for (int num_elem=0; num_elem<nb_elems; num_elem++)
296 {
297 utemp_sum(num_elem) += dt*vitesse(num_elem);
298 umoy(num_elem)=utemp_sum(num_elem)/(tps-(f_start-t_av));
299 }
300 cptbis=4;
301 }
302
303
304 //***********************************************************
305 //*************** FIN MOYENNE TEMPORELLE ***************
306 //*******************************************************
307 }
308
309 else if(moyenne==3)
310 {
311 if (tps>0)
312 {
313 if (cptbis==0)
314 {
315 for (int num_elem=0; num_elem<nb_elems; num_elem++)
316 {
317 utemp_sum(num_elem) = vitesse(num_elem)*dt;
318 umoy(num_elem) = vitesse(num_elem);
319 }
320 cptbis ++;
321 }
322 else
323 {
324 for (int num_elem=0; num_elem<nb_elems; num_elem++)
325 {
326 utemp_sum(num_elem) += vitesse(num_elem)*dt;
327 umoy(num_elem) = utemp_sum(num_elem)/(tps-dt_min+dt);
328 }
329 }
330 }
331
332 }
333 else
334 {
335 Cerr << "moyenne = " << moyenne << finl;
336 Cerr << "Probleme pour le choix du type de moyenne" << finl;
337 }
338 compteur_reprise++;
339
340}//fin mettre_a_jour

343void Terme_Source_Canal_RANS_LES_VDF_Elem::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
344{
345 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
346 int nb_elems = zone_VDF.nb_elem();
347 const DoubleVect& volume = zone_VDF.volumes();
348 const double tps = mon_equation->schema_temps().temps_courant();
349 const double dt = mon_equation->schema_temps().pas_de_temps();
350 const double dt_min = mon_equation->schema_temps().pas_temps_min();
351
352 //vitesse=temperature
353 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
354
355 double vol=0.;
356 SFichier fic_f("f_temp.dat", ios::app);
357 SFichier fic_finst("f_temp_inst.dat", ios::app);
358
359 double mbf = 0.; //maximum body force
360 double mbf2 = 0.; //maximum body force
361
362 static int cpt2=0;
363
364 // Calcul de la norme des vitesses au centre des elements
365
366 if(((tps>f_start)&&(compteur_reprise > 1))||((moyenne==3)&&(tps>dt_min)))
367 {
368 for(int num_elem = 0 ; num_elem<nb_elems ; num_elem++)
369 {
370 vol = volume(num_elem);
371
372 secmem(num_elem) += ((U_RANS(num_elem)-umoy(num_elem))/(tau(num_elem)*dt))*vol;
373
374 mbf2 +=((U_RANS(num_elem)-umoy(num_elem))/(tau(num_elem)*dt))*vol;
375
376 cpt2++;
377 }
378
379 mbf2 = mbf2/cpt2;
380
381 int num_elem=12;
382
383 mbf = (U_RANS(num_elem)-umoy(num_elem))
384 /(tau(num_elem)*dt)*vol;
385
386 fic_finst << tps << " " << mbf << " " << U_RANS(num_elem)
387 << " " << umoy(num_elem) << " " << vitesse(num_elem) << finl;
388
389 fic_f << tps << " " << mbf2 << " "
390 << U_RANS(num_elem) << " " << umoy(num_elem) << " "
391 << utemp_sum(num_elem) << " " << dt <<finl;
392 }//fin if f_start
393
394}
