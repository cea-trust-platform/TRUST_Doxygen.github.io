75void Force_Centrifuge_VDF_Face_Axi::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
76{
77 const Zone_VDF& zvdf = la_zone_VDF.valeur();
78 const Zone_Cl_VDF& zclvdf = la_zone_Cl_VDF.valeur();
79 const DoubleTab& vitesse = la_vitesse->valeurs();
80
81 int nb_elem = zvdf.nb_elem();
82 int nb_faces = zvdf.nb_faces();
83 DoubleTrav vit_sum(nb_faces);
84 vit_sum =0;
85 double U,V,coef;
86 int fac0,fac1,fac2,fac3;
87
88// Boucle sur les elements pour calculer vit_sum
89
90 for (int num_elem=0; num_elem<nb_elem; num_elem++)
91 {
92 fac0 = elem_faces(num_elem,0);
93 fac1 = elem_faces(num_elem,1);
94 fac2 = elem_faces(num_elem,dimension);
95 fac3 = elem_faces(num_elem,1+dimension);
96
97 V = 0.5*(vitesse(fac1)+vitesse(fac3));
98 U = 0.5*(vitesse(fac0)+vitesse(fac2));
99
100 vit_sum(fac0) += 0.5*(V*V);
101 vit_sum(fac2) += 0.5*(V*V);
102
103 vit_sum(fac1) += 0.5*(U*V);
104 vit_sum(fac3) += 0.5*(U*V);
105 }
106
107// Boucle sur les conditions limites pour traiter les faces de bord
108
109 int ndeb,nfin,ori,num_face;
110
111 for (int n_bord=0; n_bord<zvdf.nb_front_Cl(); n_bord++)
112 {
113
114// pour chaque Condition Limite on regarde son type
115// Si face de Dirichlet ou de Symetrie on ne fait rien
116// Si face de Neumann on calcule la contribution au terme source
117
118 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
119
120 if (sub_type(Neumann_sortie_libre,la_cl.valeur()))
121 {
122 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
123 ndeb = le_bord.num_premiere_face();
124 nfin = ndeb + le_bord.nb_faces();
125
126 for (num_face=ndeb; num_face<nfin; num_face++)
127 {
128 ori = orientation(num_face);
129 if (ori == 0)
130 {
131 coef = volume_entrelaces(num_face)*porosite_surf(num_face)/xv(num_face,0);
132 secmem(num_face) += 0.5*(vit_sum(num_face)*coef);
133 }
134 else if (ori == 1)
135 {
136 coef = volume_entrelaces(num_face)*porosite_surf(num_face)/xv(num_face,0);
137 secmem(num_face) -= 0.5*(vit_sum(num_face)*coef);
138 }
139 }
140 }
141 else if (sub_type(Symetrie,la_cl.valeur()))
142 ;
143 else if ( (sub_type(Dirichlet,la_cl.valeur()))
144 ||
145 (sub_type(Dirichlet_homogene,la_cl.valeur()))
146 )
147 {
148// do nothing
149 ;
150 }
151 }
152
153// Boucle sur les faces internes
154
155 ndeb = zvdf.premiere_face_int();
156 nfin = zvdf.nb_faces();
157
158 for (num_face=ndeb; num_face<nfin; num_face++)
159 {
160 ori = orientation(num_face);
161 if (ori == 0)
162 {
163 coef = volume_entrelaces(num_face)*porosite_surf(num_face)/xv(num_face,0);
164 secmem(num_face) += 0.5*(vit_sum(num_face)*coef);
165 }
166 else if (ori == 1)
167 {
168 coef = volume_entrelaces(num_face)*porosite_surf(num_face)/xv(num_face,0);
169 secmem(num_face) -= 0.5*(vit_sum(num_face)*coef);
170 }
171 }
172}
