85template<AJOUTE_SCAL _T_>
86DoubleTab& Op_Diff_EF::ajouter_scalaire_template(const DoubleTab& tab_inconnue, DoubleTab& resu) const
87{
88 static constexpr bool IS_GEN = (_T_ == AJOUTE_SCAL::GEN), IS_D3_8 = (_T_ == AJOUTE_SCAL::D3_8), IS_D2_4 = (_T_ == AJOUTE_SCAL::D2_4);
89
90 const Zone_EF& zone_ef = ref_cast(Zone_EF, equation().zone_dis().valeur());
91
92 const int N = IS_GEN ? resu.line_size() : 1;
93 const int nb_som_elem = IS_D3_8 ? 8 : ( IS_D2_4 ? 4 : zone_ef.zone().nb_som_elem() /* IS_GEN */);
94 const int const_dimension = IS_D3_8 ? 3 : ( IS_D2_4 ? 2 : Objet_U::dimension /* IS_GEN */);
95 const int dim_fois_nbn = nb_som_elem * const_dimension;
96
97 const DoubleVect& volumes_thilde = zone_ef.volumes_thilde(), &volumes = zone_ef.volumes();
98
99 const DoubleTab& bij = zone_ef.Bij();
100 int nb_elem_tot = zone_ef.zone().nb_elem_tot();
101 const IntTab& elems = zone_ef.zone().les_elems();
102
103 const double *bij_ptr = bij.addr();
104 const double *inco_ptr = tab_inconnue.addr();
105
106#define bij_(elem,i,j) bij_ptr[elem*dim_fois_nbn+i*const_dimension+j]
107#define inconnue_(som,a) inco_ptr[som * N + a]
108#define resu_(som,a) resu_ptr[som * N + a]
109
110 ArrOfDouble pr(N);
111 for (int elem = 0; elem < nb_elem_tot; elem++)
112 if (elem_contribue(elem))
113 {
114 double pond = volumes_thilde(elem) / volumes(elem) / volumes(elem) * nu_(elem);
115
116 for (int i1 = 0; i1 < nb_som_elem; i1++)
117 {
118 int glob = elems(elem, i1);
119 for (int yy = 0; yy < N; yy++)
120 pr[yy] = 0;
121 for (int i2 = 0; i2 < nb_som_elem; i2++)
122 {
123 int glob2 = elems(elem, i2);
124
125 {
126 double prod = 0;
127 for (int b = 0; b < const_dimension; b++)
128 {
129 prod += bij_(elem,i1,b)*bij_(elem,i2,b);
130 }
131 for (int n = 0; n < N; n++) pr[n] += prod * inconnue_(glob2, n);
132 }
133 }
134 resu(glob) -= pr[0] * pond;
135 }
136 }
137
138// on ajoute la contribution des bords
139 ajouter_bords(tab_inconnue, resu);
140 return resu;
141
142#undef bij_
143#undef inconnue_
144#undef resu_
145}

148DoubleTab& Op_Diff_EF::ajouter_vectoriel_template(const DoubleTab& tab_inconnue, DoubleTab& resu) const
149{
150 static constexpr bool IS_GEN = (_T_ == AJOUTE_VECT::GEN), IS_D3_8 = (_T_ == AJOUTE_VECT::D3_8), IS_D2_4 = (_T_ == AJOUTE_VECT::D2_4);

152 const Zone_EF& zone_ef = ref_cast(Zone_EF, equation().zone_dis().valeur());
153
154 const int N = IS_D3_8 ? 3 : (IS_D2_4 ? 2 : resu.line_size() /* IS_GEN */);
155 const int const_dimension = IS_GEN ? Objet_U::dimension : N;
156 const int nb_som_elem = IS_D3_8 ? 8 : (IS_D2_4 ? 4 : zone_ef.zone().nb_som_elem() /* IS_GEN */);
157 const int dim_fois_nbn = nb_som_elem * const_dimension;
158
159 ArrOfInt marqueur_neuman;
160 remplir_marqueur_sommet_neumann(marqueur_neuman, zone_ef, la_zcl_EF.valeur(), transpose_partout_);
161
162 const DoubleVect& volumes_thilde = zone_ef.volumes_thilde(), &volumes = zone_ef.volumes();
163
164 const DoubleTab& bij = zone_ef.Bij();
165 int nb_elem_tot = zone_ef.zone().nb_elem_tot();
166 const IntTab& elems = zone_ef.zone().les_elems();
167
168 const double *bij_ptr = bij.addr();
169 const double *inco_ptr = tab_inconnue.addr();
170 double *resu_ptr = resu.addr();
171
172#define bij_(elem,i,j) bij_ptr[elem*dim_fois_nbn+i*const_dimension+j]
173#define inconnue_(som,a) inco_ptr[som * N + a]
174#define resu_(som,a) resu_ptr[som * N + a]
175
176 ArrOfDouble pr(N);
177
178 for (int elem = 0; elem < nb_elem_tot; elem++)
179 if (elem_contribue(elem))
180 {
181 double pond = volumes_thilde(elem) / volumes(elem) / volumes(elem) * nu_(elem);
182
183 for (int i1 = 0; i1 < nb_som_elem; i1++)
184 {
185 int glob = elems(elem, i1);
186 for (int yy = 0; yy < N; yy++)
187 pr[yy] = 0;
188 int transpose = (marqueur_neuman[glob] == 1 || N == 1) ? 0 : transpose_;
189 for (int i2 = 0; i2 < nb_som_elem; i2++)
190 {
191 int glob2 = elems(elem, i2);
192
193 {
194 double prod = 0;
195 double prod2 = 0;
196 for (int b = 0; b < const_dimension; b++)
197 {
198 prod += bij_(elem,i1,b)*bij_(elem,i2,b);
199 if (transpose)
200 prod2+=bij_(elem,i1,b)*inconnue_(glob2,b);
201 }
202 for (int n = 0; n < N; n++) pr[n] += prod * inconnue_(glob2, n) + prod2 * bij_(elem, i2, n);
203 }
204 }
205 for (int n = 0; n < N; n++) resu_(glob, n)-= pr[n] * pond;
206 }
207 }
208
209// on ajoute la contribution des bords
210 ajouter_bords(tab_inconnue, resu);
211 return resu;
212
213#undef bij_
214#undef inconnue_
215#undef resu_
216}
