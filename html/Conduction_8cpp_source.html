86// Modification par rapport a Conduction::lire_motcle_non_standard()!
87int Conduction::lire_motcle_non_standard(const Motcle& mot, Entree& is)
88{
89 Motcle motlu;
90 if (mot=="diffusion")
91 {
92 Cerr << "Reading and typing of the diffusion operator : " << finl;
93
94 terme_diffusif.associer_diffusivite(diffusivite_pour_transport());
95 is >> terme_diffusif;
96// le champ pour le dt_stab est le meme que celui de l'operateur. On prend toujours la diffusivite()
97// comme dans le cas de Conduction
98 terme_diffusif.associer_diffusivite_pour_pas_de_temps(milieu().diffusivite());
99 solveur_masse->set_name_of_coefficient_temporel("rho_cp_comme_T");
100
101 return 1;
102 }
103 else if (mot=="Traitement_particulier")
104 {
105 Cerr << "Reading and typing of the Traitement_particulier object : " << finl;
106 Nom type="Traitement_particulier_";
107 Motcle motbidon;
108 is >> motbidon;
109 if (motbidon == "{")
110 {
111 Motcle le_cas;
112 is >> le_cas;
113 if (le_cas == "}")
114 le_cas ="";
115 else
116 {
117 type+= le_cas;
118 type+= "_";
119 }
120 Nom discr=discretisation().que_suis_je();
121 if (discr == "VEFPreP1B")
122 discr = "VEF";
123 type+=discr;
124 Cerr << type << finl;
125 le_traitement_particulier.typer(type);
126 le_traitement_particulier.associer_eqn(*this);
127 le_traitement_particulier.lire(is);
128 le_traitement_particulier.preparer_calcul_particulier();
129 return 1;
130 }
131 else
132 {
133 Cerr << "Error while reading Traitement_particulier" << finl;
134 Cerr << "A { is expected." << finl;
135 exit();
136 return -1;
137 }
138 }
139 else
140 return Equation_base::lire_motcle_non_standard(mot,is);
141 return 1;
142}
