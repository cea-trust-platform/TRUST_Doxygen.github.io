72void Op_Div_VEF_Elem::volumique(DoubleTab& div) const
73{
74// PQ : 04/03
75 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
76 const DoubleVect& vol = zone_VEF.volumes();
77 int nb_elem=zone_VEF.zone().nb_elem_tot();
78 for(int num_elem=0; num_elem<nb_elem; num_elem++)
79 div(num_elem)/=vol(num_elem);
80}

82int Op_Div_VEF_Elem::impr(Sortie& os) const
83{
84 const int impr_bord=(la_zone_vef->zone().Bords_a_imprimer().est_vide() ? 0:1);
85 const Schema_Temps_base& sch = equation().probleme().schema_temps();
86 double temps = sch.temps_courant();
87
88 int nb_compo=flux_bords_.dimension(1);
89// On parcours les frontieres pour sommer les flux par frontiere dans le tableau flux_bord
90 DoubleVect bilan(nb_compo);
91 bilan = 0;
92 int nb_cl = la_zone_vef->nb_front_Cl();
93// flux_bords contains the sum of flux on each boundary:
94 DoubleTrav tab_flux_bords(3,nb_cl,nb_compo);
95 for (int num_cl=0; num_cl<nb_cl; num_cl++)
96 {
97 const Cond_lim& la_cl = la_zcl_vef->les_conditions_limites(num_cl);
98 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
99 int ndeb = frontiere_dis.num_premiere_face();
100 int nfin = ndeb + frontiere_dis.nb_faces();
101 int perio = (sub_type(Periodique,la_cl.valeur())?1:0);
102 for (int face=ndeb; face<nfin; face++)
103 for(int k=0; k<nb_compo; k++)
104 {
105 tab_flux_bords(0,num_cl,k)+=flux_bords_(face, k);
106 if(perio)
107 {
108 if(face<(ndeb+frontiere_dis.nb_faces()/2))
109 tab_flux_bords(1,num_cl,k)+=flux_bords_(face, k);
110 else
111 tab_flux_bords(2,num_cl,k)+=flux_bords_(face, k);
112 }
113 }
114 }
115// Sum on all CPUs:
116 mp_sum_for_each_item(tab_flux_bords);
117
118// Print
119 if(je_suis_maitre())
120 {
121 SFichier Flux_div;
122 ouvrir_fichier(Flux_div,"",1);
123 Flux_div.add_col(temps);
124 for (int num_cl=0; num_cl<nb_cl; num_cl++)
125 {
126 const Cond_lim& la_cl = la_zcl_vef->les_conditions_limites(num_cl);
127 int perio = (sub_type(Periodique,la_cl.valeur())?1:0);
128 for(int k=0; k<nb_compo; k++)
129 {
130 if(perio)
131 {
132 Flux_div.add_col(tab_flux_bords(1,num_cl,k));
133 Flux_div.add_col(tab_flux_bords(2,num_cl,k));
134 }
135 else
136 Flux_div.add_col(tab_flux_bords(0,num_cl,k));
137 bilan(k)+=tab_flux_bords(0,num_cl,k);
138 }
139 }
140
141 for(int k=0; k<nb_compo; k++)
142 Flux_div.add_col(bilan(k));
143 Flux_div << finl;
144 }
145
146 const LIST(Nom)& Liste_Bords_a_imprimer = la_zone_vef->zone().Bords_a_imprimer();
147 if (!Liste_Bords_a_imprimer.est_vide())
148 {
149 EcrFicPartage Flux_face;
150 ouvrir_fichier_partage(Flux_face,"",impr_bord);
151 for (int num_cl=0; num_cl<nb_cl; num_cl++)
152 {
153 const Frontiere_dis_base& la_fr = la_zcl_vef->les_conditions_limites(num_cl).frontiere_dis();
154 const Cond_lim& la_cl = la_zcl_vef->les_conditions_limites(num_cl);
155 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
156 int ndeb = frontiere_dis.num_premiere_face();
157 int nfin = ndeb + frontiere_dis.nb_faces();
158 if (la_zone_vef->zone().Bords_a_imprimer().contient(la_fr.le_nom()))
159 {
160 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps " << temps << " : " << finl;
161 for (int face=ndeb; face<nfin; face++)
162 {
163 if (dimension==2)
164 Flux_face << "# Face a x= " << la_zone_vef->xv(face,0) << " y= " << la_zone_vef->xv(face,1) << " flux=" ;
165 else if (dimension==3)
166 Flux_face << "# Face a x= " << la_zone_vef->xv(face,0) << " y= " << la_zone_vef->xv(face,1) << " z= " << la_zone_vef->xv(face,2) << " flux=" ;
167 for(int k=0; k<nb_compo; k++)
168 Flux_face << " " << flux_bords_(face, k);
169 Flux_face << finl;
170 }
171 Flux_face.syncfile();
172 }
173 }
174 }
175 return 1;
176}
