20//! Verifie si le vecteur v a son espace virtuel a jour. Cette fonction doit etre appelee simultanement par tous les processeurs qui partagent le vecteur.<br>(pour cela, on cree une copie qu'on echange, puis on compare a l'original).
21int check_espace_virtuel_vect(const DoubleVect& v)
22{
23// On fait une copie:
24 DoubleVect w(v);
25// On echange
26 w.echange_espace_virtuel();
27// Norme de w :
28 const double norme_w = norme_array(w);
29// On compare : difference entre les deux vecteurs au sens
30// ArrOfDouble (pas d'echange espace virtuel)
31 w.ArrOfDouble::operator-=(v);
32 const double ecart = norme_array(w);
33 if (ecart > (norme_w + 1e-30) * 1e-12)
34 return 0;
35 else
36 return 1;
37}

39//! Idem que check_espace_virtuel_vect(const DoubleVect \& v)
40int check_espace_virtuel_vect(const IntVect& v)
41{
42// On fait une copie:
43 IntVect w(v);
44// On echange
45 w.echange_espace_virtuel();
46// On compare:
47 const int n = v.size_array();
48 for (int i = 0; i < n; i++)
49 if (v[i] != w[i])
50 return 0;
51 return 1;
52}
