101int Champ_Elem_PolyMAC::imprime(Sortie& os, int ncomp) const
102{
103 const Zone_dis_base& zone_dis = zone_dis_base();
104 const Zone& zone = zone_dis.zone();
105 const DoubleTab& coord=zone.domaine().coord_sommets();
106 const int nb_som = zone.domaine().nb_som();
107 const DoubleTab& val = valeurs();
108 int som;
109 os << nb_som << finl;
110 for (som=0; som<nb_som; som++)
111 {
112 if (dimension==3)
113 os << coord(som,0) << " " << coord(som,1) << " " << coord(som,2) << " " ;
114 if (dimension==2)
115 os << coord(som,0) << " " << coord(som,1) << " " ;
116 if (nb_compo_ == 1)
117 os << val(som) << finl;
118 else
119 os << val(som,ncomp) << finl;
120 }
121 os << finl;
122 Cout << "Champ_Elem_PolyMAC::imprime FIN >>>>>>>>>> " << finl;
123 return 1;

138void Champ_Elem_PolyMAC::init_auxiliary_variables()
139{
140 const Zone_PolyMAC& zone = ref_cast( Zone_PolyMAC,la_zone_VF.valeur());
141 const IntTab& f_e = zone.face_voisins();
142 for (int n = 0; n < nb_valeurs_temporelles(); n++)
143 if (futur(n).size_reelle_ok())
144 {
145 DoubleTab& vals = futur(n);
146 vals.set_md_vector(MD_Vector()); //on enleve le MD_Vector...
147 vals.resize_dim0(zone.mdv_elems_faces.valeur().get_nb_items_tot()); //...on dimensionne a la bonne taille...
148 vals.set_md_vector(zone.mdv_elems_faces); //...et on remet le bon MD_Vector
149 /* initialisation des variables aux faces : par celle de l'elem amont */
150 for (int f = 0, ne_tot = zone.nb_elem_tot(); f < zone.nb_faces(); f++)
151 for (int m = 0, e = f_e(f, 0); m < vals.dimension(1); m++)
152 vals(ne_tot + f, m) = vals(e, m);
153 vals.echange_espace_virtuel();
154 }
155}

183void Champ_Elem_PolyMAC::init_fcl() const
184{
185 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,la_zone_VF.valeur());
186 const Conds_lim& cls = ma_zone_cl_dis.valeur().les_conditions_limites();
187 int i, f, n;
188
189 fcl_.resize(zone.nb_faces_tot(), 3);
190 for (n = 0; n < cls.size(); n++)
191 {
192 const Front_VF& fvf = ref_cast(Front_VF, cls[n].frontiere_dis());
193 int idx = sub_type(Echange_externe_impose, cls[n].valeur()) + 2 * sub_type(Echange_global_impose, cls[n].valeur())
194 + 4 * sub_type(Neumann_paroi, cls[n].valeur()) + 5 * (sub_type(Neumann_homogene, cls[n].valeur()) || sub_type(Neumann_sortie_libre, cls[n].valeur()) || sub_type(Symetrie, cls[n].valeur()))
195 + 6 * sub_type(Dirichlet, cls[n].valeur()) + 7 * sub_type(Dirichlet_homogene, cls[n].valeur());
196 if (sub_type(Echange_contact_PolyMAC, cls[n].valeur()) || sub_type(Echange_contact_PolyMAC_P0, cls[n].valeur())) idx = 3;
197 if (!idx) Cerr << "Champ_Elem_PolyMAC : CL non codee rencontree!" << finl, Process::exit();
198 for (i = 0; i < fvf.nb_faces_tot(); i++)
199 f = fvf.num_face(i), fcl_(f, 0) = idx, fcl_(f, 1) = n, fcl_(f, 2) = i;
200 }
201 fcl_init_ = 1;
202}

204DoubleTab& Champ_Elem_PolyMAC::valeur_aux_faces(DoubleTab& dst) const
205{
206 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, zone_dis_base());
207 const IntTab& f_e = zone.face_voisins();
208 const DoubleTab& src = valeurs();
209
210 /* vals doit etre pre-dimensionne */
211 int i, e, f, n, N = (src.nb_dim() == 1 ? 1 : src.dimension(1));
212 assert(dst.dimension(0) == zone.xv().dimension(0) && N == (dst.nb_dim() == 1 ? 1 : dst.dimension(1)));
213
214 if (src.dimension_tot(0) > zone.nb_elem_tot()) //on a les valeurs aux faces
215 for (f = 0; f < dst.dimension(0); f++)
216 for (n = 0; n < N; n++) dst(f, n) = src(zone.nb_elem_tot() + f, n);
217 else for (f = 0; f < dst.dimension(0); f++) //on prend (amont + aval) / 2
218 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
219 for (n = 0; n < N; n++)
220 dst(f, n) += src(e, n) * (f < zone.premiere_face_int() ? 1 : 0.5);
221
222 return dst;
223}
