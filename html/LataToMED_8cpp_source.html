83// dom_trio.les_sommets()=geom.nodes_;
84// mais geom.nodes est un FloatTab
85 DoubleTab& som= dom_trio.les_sommets();
86 int nx=1,ny=1,nz=1;
87 if (dom.get_domain_type()==Domain::UNSTRUCTURED)
88 {
89 const DomainUnstructured& geom = dom.cast_DomainUnstructured();
90 int dim1=geom.nodes_.dimension(0);
91 int dim2=geom.nodes_.dimension(1);
92 som.resize(dim1,dim2);
93 for (int i1=0; i1<dim1; i1++)
94 for (int i2=0; i2<dim2; i2++)
95 som(i1,i2)=geom.nodes_(i1,i2);
96 }
97 else
98 {
99 assert(dom.get_domain_type()==Domain::IJK);
100 const DomainIJK& geom = dom.cast_DomainIJK();
101 const int dim = geom.coord_.size();
102
103
104 nx=geom.coord_[0].size_array();
105 ny=geom.coord_[1].size_array();
106 nz=1;
107 if (dim>2) nz=geom.coord_[2].size_array();
108
109 som.resize(nx*ny*nz,dim);
110
111 for (int i=0; i<nx; i++)
112 for (int j=0; j<ny; j++)
113 for (int k=0; k<nz; k++)
114 {
115 int nn=i*ny*nz+j*nz+k;
116 som(nn,0)=geom.coord_[0][i];
117 som(nn,1)=geom.coord_[1][j];
118 if (dim>2)
119 som(nn,2)=geom.coord_[2][k];
120 }
121
122 }
123
124 Zone zone_bidon;
125 dom_trio.add(zone_bidon);
126 Zone& zone=dom_trio.zone(0);
127 zone.nommer("PAS_NOMME");
128 Nom type_elem=dom.lata_element_name(dom.elt_type_);
129 if (type_elem=="PRISM6") type_elem="PRISME";
130// zone.type_elem()=type_ele;
131 zone.typer(type_elem);
132
133 zone.type_elem().associer_zone(zone);
134 zone.associer_domaine(dom_trio);
135 if (dom.get_domain_type()==Domain::UNSTRUCTURED)
136 {
137 const DomainUnstructured& geom = dom.cast_DomainUnstructured();
138 zone.les_elems()=geom.elements_;
139 }
140 else
141 {
142
143 zone.les_elems().resize((nx-1)*(ny-1)*(nz-1),zone.nb_som_elem());
144 IntTab& elems=zone.les_elems();
145 for (int i=0; i<nx-1; i++)
146 for (int j=0; j<ny-1; j++)
147 for (int k=0; k<nz-1; k++)
148 {
149 int nn=i*(ny-1)*(nz-1)+j*(nz-1)+k;
150 nn=i+j*(nx-1)+k*((nx-1)*(ny-1));
151 elems(nn,0)=i*ny*nz+j*nz+k;
152 elems(nn,1)=(i+1)*ny*nz+j*nz+k;
153 elems(nn,2)=(i)*ny*nz+(j+1)*nz+k;
154 elems(nn,3)=(i+1)*ny*nz+(j+1)*nz+k;
155 if (elems.dimension(1)>4)
156 for (int ii=0; ii<4; ii++) elems(nn,4+ii)=elems(nn,ii)+1;
157 }
158// a coder
159// Process::exit();
160 }
161 zone.faces_bord().associer_zone(zone);
162 zone.faces_raccord().associer_zone(zone);
163 zone.faces_joint().associer_zone(zone);
164
165
166 zone.type_elem().associer_zone(zone);
167 zone.fixer_premieres_faces_frontiere();

184Entree& latatoother::interpreter(Entree& is)
185{
186 Cerr<<"syntax Latatoother::interpreter format_post_supp [nom_lata||NOM_DU_CAS] [nom_fichier_sortie||NOM_DU_CAS] "<<finl;
187 Nom nom_dom, nom_lata,nom_fic;
188 Nom format_post_supp;
189 is >>format_post_supp>> nom_lata>>nom_fic ;
190
191 std::set<std::string> formats( { "LML", "LATA", "LATA_V1", "LATA_V2", "MED" });
192 Nom tmp = format_post_supp;
193 if(formats.count(tmp.majuscule().getString())==0)
194 {
195 Cerr << "Error, format " << format_post_supp << " is not supported!" << finl;
196 Process::exit();
197 }
198
199// Creation d'un sous-groupe contenant uniquement le processeur maitre
200 ArrOfInt liste_pe(1);
201 DERIV(Comm_Group) group;
202// on se met en non axi le temps de la conversion
203 int axi_sa=axi;
204 axi=0;
205 PE_Groups::create_group(liste_pe, group);
206
207 if (PE_Groups::enter_group(group.valeur()))
208 {
209 if (nom_lata=="NOM_DU_CAS")
210 nom_lata=nom_du_cas()+".lata";
211 if (nom_fic=="NOM_DU_CAS")
212 nom_fic=nom_du_cas();
213 Nom filename(nom_lata);
214 LataOptions opt;
215 LataDB lata_db;
216 LataFilter filter;
217 set_Journal_level(0);
218
219 LataOptions::extract_path_basename(filename, opt.path_prefix, opt.basename);
220 opt.dual_mesh = true;
221 opt.faces_mesh = false;
222// on ne veut jamais creer le domaine IJK
223 opt.regularize = 0;
224 opt.regularize_tolerance = 1e-7;
225 read_any_format_options(filename, opt);
226// Read the source file to the lata database
227 read_any_format(filename, opt.path_prefix, lata_db);
228
229 filter.initialize(opt, lata_db);
230
231// On met dual_mesh a "false" si on lit un domaine de polyedres PolyMAC (pas encore supporte)
232// et on-reouvre la database:
233 Noms geom_names = filter.get_exportable_geometry_names();
234 int elt_type = filter.get_geometry(Domain_Id(geom_names[0], 1, -1)).elt_type_;
235 Cerr << elt_type << finl;
236 if (elt_type==Domain::polyedre || elt_type==Domain::polygone)
237 {
238 opt.dual_mesh = false;
239 filter.initialize(opt, lata_db);
240 }
241
242 Nom nom_pdb(nom_fic);
243 Nom format_post_(format_post_supp);
244 if (format_post_=="lata")
245 format_post_+="_V1";
246
247// copie de Postraiter_domaine
248
249 Nom type("Format_Post_");
250 type+=format_post_;
251 Format_Post post_typer;
252 post_typer.typer_direct(type);
253 Format_Post_base& post=ref_cast(Format_Post_base,post_typer.valeur());
254 Nom nom_2(nom_pdb);
255 Nom format_post_bis(format_post_);
256 if (format_post_bis.debute_par("lata"))
257 format_post_bis="lata";
258 Nom suffix(".");
259 suffix+=format_post_bis;
260 nom_2.prefix(suffix);
261// int est_le_premie_post=0;
262 int format_binaire_=0;
263 if (format_post_bis!="lata")
264 post.initialize_by_default(nom_2);
265 else
266 post.initialize(nom_2,format_binaire_,"SIMPLE");
267
268// ecriture des domaines
269 {
270
271 Noms geoms = filter.get_exportable_geometry_names();
272
273 for (int i = 0; i < geoms.size(); i++)
274 {
275 Domain_Id id(geoms[i], 1, -1);
276 const Domain& dom = filter.get_geometry(id);
277
278 Domaine dom_trio;
279 dom_trio.nommer(geoms[i]);
280 convert_domain_to_Domaine(dom,dom_trio);
281 int est_le_premier_post=(i==0);
282 post.ecrire_entete(0.,0,est_le_premier_post);
283 int reprise = 0;
284 double t_init = 0.;
285 post.preparer_post(dom_trio.le_nom(),est_le_premier_post,reprise,t_init);
286 post.ecrire_domaine(dom_trio, est_le_premier_post);
287// meddrive.ecrire_domaine(nom_fic,dom_trio,dom_trio.le_nom(),mode);
288
289
290 filter.release_geometry(dom);
291 }
292 }
293// les champs
294 const char *suffix_vector_names[] = { "X", "Y", "Z" };
295
296 for (int t = 1; t < lata_db.nb_timesteps(); t++)
297 {
298 int timestate=t;
299// if (filter.get_nb_timesteps() > 1) timestate++;
300 Noms geoms = filter.get_exportable_geometry_names();
301 double time=filter.get_timestep(t);
302 post.ecrire_temps(time);
303 for (int i = 0; i < geoms.size(); i++)
304 {
305 Domain_Id id(geoms[i], timestate, -1);
306 const Domain& dom = filter.get_geometry(id);
307
308 Domaine dom_trio;
309 dom_trio.nommer(geoms[i]);
310 convert_domain_to_Domaine(dom,dom_trio);
311 Field_UNames fields = filter.get_exportable_field_unames(geoms[i]);
312 for (int j = 0; j < fields.size(); j++)
313 {
314// const Nom& nom= fields[j].get_field_name();
315// if ((nom!=Motcle("ELEM_FACES"))&& (nom!=Motcle("FACES"))&& (nom!=Motcle("ELEMENTS")))
316 {
317
318 int est_le_premie_post=(i==0);
319 post.init_ecriture(time,-1.,est_le_premie_post,dom_trio);
320 Field_Id fieldid(fields[j],timestate,-1);
321
322 DoubleTab values;
323 try
324 {
325 const FieldFloat& field = filter.get_float_field(fieldid);
326 {
327 const FloatTab& values_lata = field.data_;
328 int dim1=values_lata.dimension(0);
329 int dim2=values_lata.dimension(1);
330 values.resize(dim1,dim2);
331 for (int i1=0; i1<dim1; i1++)
332 for (int i2=0; i2<dim2; i2++)
333 values(i1,i2)=values_lata(i1,i2);
334 }
335
336
337 Nom nom_type;
338 if (field.localisation_==LataField_base::ELEM)
339 nom_type="ELEM";//"CHAMPMAILLE";
340 else if (field.localisation_==LataField_base::SOM)
341 nom_type="SOM";//!< CHAMPPOINT";
342 else
343 {
344 Cerr<<nom_type<<" not coded ...... "<<finl;
345 Cerr<<__FILE__<<":"<<(int)__LINE__<<finl;
346 exit();
347
348 }
349 Nom type_elem=dom.lata_element_name(dom.elt_type_);
350 Noms unites(values.dimension(1));
351 Noms noms_post(values.dimension(1));
352 Cerr<<"Extraction "<< time<<" "<< geoms[i]<<" "<< fields[j].get_field_name()<<finl;
353 Nom nom_post,ajout;
354
355 nom_post=fields[j].get_field_name();
356// pour eviter qu'en med on est le champ aux faces et sur le dual avec le meme nom
357 if (dom_trio.le_nom().finit_par("_faces")) nom_post+="_Centre";
358 ajout="_";
359 ajout+=nom_type;
360 ajout+="_";
361 ajout+=geoms[i];
362 for (int ii=0; ii<values.dimension(1); ii++)
363 {
364 noms_post[ii]=nom_post;
365 if (values.dimension(1)==dom_trio.les_sommets().dimension(1))
366 noms_post[ii]+=suffix_vector_names[ii];
367 else
368 noms_post[ii]+=Nom(ii);
369 noms_post[ii]+=ajout;
370 }
371 nom_post+=ajout;
372// exit();
373// meddrive.ecrire_champ(type,nom_fic,geoms[i],fields[j].get_field_name(),values,unites,type_elem ,time,0);
374
375 post.ecrire_champ(dom_trio,unites,noms_post,-1,time,time,nom_post,geoms[i],nom_type,"iii",values);
376 filter.release_field(field);
377 }
378 catch(...)
379 {
380 Cerr<<fieldid.uname_.get_field_name()<<" is not a FloatField !!!"<<finl;
381 }
382 }
383 }
384 filter.release_geometry(dom);
385 }
386
387
388 }
389 int fin=1;
390 post.finir(fin);
391
392
393// delete [] les_domaines;
394// delete [] list_champ;
395 PE_Groups::exit_group();
396 }
397 barrier();
398// on remet eventuellement axi
399 axi=axi_sa;
400 return is;
401}

423 Nom nom1,nom2;
424

426 Nom instruction("med ");
427 instruction+=nom1;
428 instruction+=" ";
429 instruction+=nom2;

432 latatoother latatoother;

435 }
