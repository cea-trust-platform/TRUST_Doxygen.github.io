81//! Renvoie la valeur de la pression calculee pour conserver la masse
82/** \return Double<br>
83Signification: pression calculee<br>
84Contraintes: */
85double EDO_Pression_th_VDF::masse_totale(double P,const DoubleTab& T)
86{
87 int elem, nb_elem=la_zone->nb_elem();
88 const DoubleVect& volumes = la_zone->volumes();
89 const Loi_Etat_base& loi_ = ref_cast(Loi_Etat_base,le_fluide_->loi_etat().valeur());
90 double M=0;
91 if (!sub_type(Loi_Etat_Multi_GP_QC,loi_))
92 {
93 for (elem=0 ; elem<nb_elem ; elem++)
94 {
95 double v = volumes(elem);
96 M += v*loi_.calculer_masse_volumique(P,T[elem]);
97 }
98 }
99 else
100 {
101 const Loi_Etat_Multi_GP_QC& loi_mel_GP = ref_cast(Loi_Etat_Multi_GP_QC,loi_);
102 const DoubleTab& Masse_mol_mel = loi_mel_GP.masse_molaire();
103
104 for (elem=0 ; elem<nb_elem ; elem++)
105 {
106 double v = volumes(elem);
107 double r = 8.3143/Masse_mol_mel(elem);
108 M += v*loi_mel_GP.calculer_masse_volumique(P,T[elem],r);
109 }
110 }
111 M=Process::mp_sum(M);
112 return M;
113}

115void EDO_Pression_th_VDF::calculer_grad(const DoubleTab& inco, DoubleTab& resu)
116{
117 int face, n0, n1, ori;
118 double coef;
119 IntTab& face_voisins = la_zone->face_voisins();
120 IntVect& orientation = la_zone->orientation();
121 DoubleVect& porosite_surf = la_zone->porosite_face();
122 DoubleTab& xp = la_zone->xp();
123 DoubleVect& volume_entrelaces = la_zone->volumes_entrelaces();
124
125// Boucle sur les faces internes
126 for (face=la_zone->premiere_face_int(); face<la_zone->nb_faces(); face++)
127 {
128 n0 = face_voisins(face,0);
129 n1 = face_voisins(face,1);
130 ori = orientation(face);
131 coef = volume_entrelaces(face)*porosite_surf(face);
132 coef=1;
133 resu(face) += coef*(inco(n1)-inco(n0))/(xp(n1,ori)- xp(n0,ori));
134 }
135}
