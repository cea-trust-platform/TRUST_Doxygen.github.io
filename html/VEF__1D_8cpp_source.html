40void VEF_1D::vitesse(const Schema_Temps_base& sch,
41 Zone_dis& z,
42 Champ_Inc& ch) const
43{
44 Cerr << "Discretisation de la vitesse" << finl;
45 Zone_VEF& zone_VEF=ref_cast(Zone_VEF, z.valeur());
46 if (sub_type(Tri_VEF,zone_VEF.type_elem().valeur()) || sub_type(Tetra_VEF,zone_VEF.type_elem().valeur()))
47 {
48 ch.typer("Champ_P1NC");
49 Champ_P1NC& ch_vitesse=ref_cast(Champ_P1NC, ch.valeur());
50 ch_vitesse.associer_zone_dis_base(zone_VEF);
51 ch_vitesse.nommer("vitesse");
52 ch_vitesse.fixer_nb_comp(1);
53 ch_vitesse.fixer_nb_valeurs_temporelles(sch.nb_valeurs_temporelles());
54 ch_vitesse.fixer_nb_valeurs_nodales(zone_VEF.nb_faces());
55 ch_vitesse.fixer_unite("m/s");
56 ch_vitesse.changer_temps(sch.temps_courant());
57 }
58 else if (sub_type(Quadri_VEF,zone_VEF.type_elem().valeur()) || sub_type(Hexa_VEF,zone_VEF.type_elem().valeur()))
59 {
60 ch.typer("Champ_Q1NC");
61 Champ_Q1NC& ch_vitesse=ref_cast(Champ_Q1NC, ch.valeur());
62 ch_vitesse.associer_zone_dis_base(zone_VEF);
63 ch_vitesse.nommer("vitesse");
64 ch_vitesse.fixer_nb_comp(1);
65 ch_vitesse.fixer_nb_valeurs_temporelles(sch.nb_valeurs_temporelles());
66 ch_vitesse.fixer_nb_valeurs_nodales(zone_VEF.nb_faces());
67 ch_vitesse.fixer_unite("m/s");
68 ch_vitesse.changer_temps(sch.temps_courant());
69 }
70 else
71 {
72 Cerr << "Pb dans le typage des elements dans VEF_discretisation::vitesse" << finl;
73 exit();
74 }
75}
