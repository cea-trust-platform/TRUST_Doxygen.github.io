336// Parametre:
337// Signification:
338// Valeurs par defaut:
339// Contraintes:

348//! Cette methode est une sorte de main() du Problem<br>Elle peut etre utilisee si le probleme n'est couple a aucun autre.<br>(s'il n'a besoin d'aucun champ d'entree).
349/** \return bool<br>
350Signification: true=OK, false=error<br>
351Contraintes: */
352bool Probleme_U::run()
353{
354// Force the post process task at the beginning of the run
355 Cerr<<"First postprocessing, this can take some minutes"<<finl;
356 postraiter(1);
357 Cerr<<"First postprocessing OK"<<finl;
358
359 bool stop=false; //!< Does the Problem want to stop ?
360 bool ok=true; //!< Is the time interval successfully solved ?
361
362// Compute the first time step
363 double dt=computeTimeStep(stop);
364
365 statistiques().end_count(initialisation_calcul_counter_);
366// Print the initialization CPU statistics
367 if (!disable_TU)
368 {
369 if(GET_COMM_DETAILS)
370 statistiques().print_communciation_tracking_details("Statistiques d'initialisation du calcul", 0); //!< Into _comm.TU file
371
372 statistiques().dump("Statistiques d'initialisation du calcul", 0);
373 print_statistics_analyse("Statistiques d'initialisation du calcul", 0);
374 }
375 statistiques().reset_counters();
376 statistiques().begin_count(temps_total_execution_counter_);
377
378#ifdef VTRACE
379// VT_USER_END("Initialization");
380 VT_BUFFER_FLUSH();
381 VT_ON();
382 VT_USER_START("Resolution");
383#endif
384// Time step loop
385 int tstep = 0;
386 while(!stop)
387 {
388// Begin the CPU measure of the time step
389 statistiques().begin_count(timestep_counter_);
390
391 ok=false; //!< Is the time interval successfully solved ?
392
393// Loop on the time interval tries
394 while (!ok && !stop)
395 {
396
397// Prepare the next time step
398 if (!initTimeStep(dt))
399 return false;
400
401// Backup unknown fields if necessary
402 if (lsauv())
403 sauver();
404
405// Solve the next time step
406 ok=solveTimeStep();
407
408 if (!ok) //!< The resolution failed, try with a new time interval.
409 {
410 abortTimeStep();
411 dt=computeTimeStep(stop);
412 }
413 else //!< The resolution was successful, validate and go to the next time step.
414 validateTimeStep();
415 }
416 if (!ok) //!< Impossible to solve the next time step, the Problem has given up
417 break;
418
419// Compute the next time step length
420 dt=computeTimeStep(stop);
421
422// Stop the resolution if the Problem is stationnary
423 if (isStationary())
424 {
425 stop=true;
426 setStationary(stop);
427 }
428// Post process task (Force the post processing/prints at the end of the run (stop=1))
429 postraiter(stop);
430
431// Stop the CPU measure of the time step and print:
432 statistiques().end_count(timestep_counter_);
433 if (limpr())
434 {
435 double temps = statistiques().last_time(timestep_counter_);
436 Cout << finl << "clock: Total time step: " << temps << " s" << finl << finl;
437 }
438
439 if(JUMP_3_FIRST_STEPS && tstep < 3)
440 {
441// demarrage des compteurs CPU
442 if(tstep == 2)
443 {
444 statistiques().set_three_first_steps_elapsed(true);
445 }
446 }
447 else
448 {
449 statistiques().compute_avg_min_max_var_per_step(tstep);
450 }
451
452 tstep++;
453#ifdef VTRACE
454// Flush the buffer regulary to avoid setting VT_MAX_FLUSHES=0 variable...
455 VT_BUFFER_FLUSH();
456#endif
457 }
458#ifdef VTRACE
459 VT_USER_END("Resolution");
460 VT_OFF();
461#endif
462
463// Print the resolution CPU statistics
464 if (!disable_TU)
465 {
466 if(GET_COMM_DETAILS)
467 statistiques().print_communciation_tracking_details("Statistiques de resolution du probleme", 1); //!< Into _comm.TU file
468
469 statistiques().dump("Statistiques de resolution du probleme", 1); //!< Into _detail.TU file
470 print_statistics_analyse("Statistiques de resolution du probleme", 1); //!< Into .TU file
471 }
472
473// Reset the CPU counters
474 statistiques().reset_counters();
475 statistiques().begin_count(temps_total_execution_counter_);
476
477 return ok;
478}

480// Precondition: initialize
481// Parametre: double time
482// Signification: time to reach
483// Valeurs par defaut:
484// Contraintes:
485// Acces:
486// Exception:
487// Effets de bord:
488// Postcondition:
489// If the method succeeds, the present time is time.
490//
491//
492//
493//! This method has the same role as the method run, but it stops<br>when reaching the time given in parameter.<br>If this time cannot be reached, the method returns false.
494/** \return true=OK, false=error<br>
495Signification:<br>
496Contraintes: */
497bool Probleme_U::runUntil(double time)
498{
499
500// Error if time is already past
501 if (time<presentTime())
502 return false;
503
504 bool stop=false; //!< Does the Problem want to stop ?
505 bool ok=true; //!< Is the time interval successfully solved ?
506
507// Compute the first time step length
508 double dt=computeTimeStep(stop);
509
510// Boucle sur les pas de temps
511 while(!stop)
512 {
513
514 statistiques().begin_count(timestep_counter_);
515
516 ok=false;
517
518// Loop on the time interval tries
519 while (!ok && !stop)
520 {
521
522// Do not go past time
523 if (presentTime()+dt>time)
524 dt=time-presentTime();
525
526// Prepare the next time step
527 if (!initTimeStep(dt))
528 return false;
529
530// Solve the next time step
531 ok=solveTimeStep();
532
533 if (!ok) //!< The resolution failed, try with a new time interval.
534 {
535 abortTimeStep();
536 dt=computeTimeStep(stop);
537 }
538
539 else //!< The resolution was successful, validate and go to the next time step.
540//
541 validateTimeStep();
542 }
543 if (!ok) //!< Impossible to solve the next time step, the Problem has given up
544 break;
545
546// time was reached
547 if (presentTime()==time)
548 return true;
549
550// Compute the next time step length
551 dt=computeTimeStep(stop);
552
553 statistiques().end_count(timestep_counter_);
554 if (je_suis_maitre() && limpr())
555 {
556 double temps = statistiques().last_time(timestep_counter_);
557 Cout << "clock: Total time step: " << temps << " s" << finl;
558 }
559
560 postraiter(0);
561 }
562
563 return ok;
564}
