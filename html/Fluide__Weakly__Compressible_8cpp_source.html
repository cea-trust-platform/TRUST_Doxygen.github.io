29// XD fluide_weakly_compressible fluide_dilatable_base fluide_weakly_compressible -1 Weakly-compressible flow with a low mach number assumption; this means that the thermo-dynamic pressure (used in state law) can vary in space.
30// XD attr loi_etat loi_etat_base loi_etat 1 The state law that will be associated to the Weakly-compressible fluid.
31// XD attr sutherland bloc_sutherland sutherland 1 Sutherland law for viscosity and for conductivity.
32// XD attr traitement_pth chaine(into=["constant"]) traitement_pth 1 Particular treatment for the thermodynamic pressure Pth ; there is currently one possibility: NL2 1) the keyword \'constant\' makes it possible to have a constant Pth but not uniform in space ; it\'s the good choice when the flow is open (e.g. with pressure boundary conditions).
33// XD attr lambda field_base lambda_u 1 Conductivity (W.m-1.K-1).
34// XD attr mu field_base mu 1 Dynamic viscosity (kg.m-1.s-1).

192void Fluide_Weakly_Compressible::initialiser_pth_xyz(const Probleme_base& pb)
193{
194 Cerr<<"Initializing the thermo-dynamic pressure Pth_xyz ..." << finl;
195 int isVDF = 0; //!< VDF or VEF ??
196 if (pb.equation(0).discretisation().que_suis_je()=="VDF") isVDF = 1;
197
198// We know that mu is always stored on elems and that Champ_Don rho_xyz_ is evaluated on elements
199 assert (Pth_xyz_->valeurs().size() == viscosite_dynamique().valeurs().size());
200
201 if (isVDF)
202 {
203// Disc VDF => Pth_xyz_ on elems => we do nothing
204 Pth_tab_=Pth_n_tab_=Pth_xyz_->valeurs();
205 }
206 else
207 {
208// we use rho for affecter because the field is on the faces in VEF (as rho)
209 Champ_base& ch_rho = masse_volumique().valeur();
210 ch_rho.affecter_(Pth_xyz_);
211 Pth_tab_=Pth_n_tab_=ch_rho.valeurs();
212 }

240void Fluide_Weakly_Compressible::checkTraitementPth(const Zone_Cl_dis& zone_cl)
241{
242// TODO : FIXME : we should be able to use this in closed cavities too !!
243 int pression_imposee=0;
244 int size=zone_cl.les_conditions_limites().size();
245 assert(size!=0);
246 for (int n=0; n<size; n++)
247 {
248 const Cond_lim& la_cl = zone_cl.les_conditions_limites(n);
249 if (sub_type(Neumann_sortie_libre, la_cl.valeur())) pression_imposee=1;

254 Cerr << "At least one open boundary is currently required for the weakly compressible problem !" << finl;

259void Fluide_Weakly_Compressible::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
260{
261 const Zone_dis_base& zone_dis=pb.equation(0).zone_dis();
262 double temps=pb.schema_temps().temps_courant();
263
264// In *_Melange_Binaire_WC we do not even have a temperature variable ...
265// it is the species mass fraction Y1... Although named ch_temperature
266 Champ_Don& ch_TK = ch_temperature();
267 if (pb.que_suis_je()=="Pb_Hydraulique_Melange_Binaire_WC" || pb.que_suis_je()=="Pb_Hydraulique_Melange_Binaire_Turbulent_WC")
268 dis.discretiser_champ("temperature",zone_dis,"fraction_massique","neant",1,temps,ch_TK);
269 else
270 dis.discretiser_champ("temperature",zone_dis,"temperature","K",1,temps,ch_TK);
271
272 if (type_fluide()!="Gaz_Parfait")
273 loi_etat()->champs_compris().ajoute_champ(ch_TK);
274
275 Fluide_Dilatable_base::discretiser(pb,dis);
276
277// XXX XXX : Champs pour WC : comme la temperature car elem en VDF et faces en VEF
278 Champ_Don& phydro = pression_hydro();
279 dis.discretiser_champ("temperature",zone_dis,"pression_hydro","Pa",1,temps,phydro);
280 champs_compris_.ajoute_champ(phydro);
281
282 Champ_Don& peos = pression_eos();
283 dis.discretiser_champ("temperature",zone_dis,"pression_eos","Pa",1,temps,peos);
284 champs_compris_.ajoute_champ(peos);
285
286// Seulement pour multi-especes
287 if (pb.que_suis_je()=="Pb_Thermohydraulique_Especes_WC")
288 {
289 Champ_Don& yn = fraction_massique_nonresolue();
290 dis.discretiser_champ("temperature",zone_dis,"fraction_massique_nonresolue","neant",1,temps,yn);
291 champs_compris_.ajoute_champ(yn);

357//! Calcule la pression hydrostatique = rho*g*z
358void Fluide_Weakly_Compressible::calculer_pression_hydro()
359{
360 DoubleTab& tab_Phydro = pression_hydro_.valeurs();
361 const Zone_dis_base& zone_dis= pression_-> zone_dis_base();
362 const Zone_VF& zone = ref_cast(Zone_VF, zone_dis);
363 int is_VDF = zone_dis.que_suis_je() == "Zone_VDF" ? 1 : 0;
364 const DoubleTab& centres_de_gravites = is_VDF ? zone.xp() : zone.xv();
365 const DoubleTab& tab_rho = rho_np1(), grav = gravite().valeurs();
366 const int n = tab_Phydro.dimension_tot(0);
367 assert (n == tab_rho.dimension_tot(0) && a_gravite());
368 for (int i=0 ; i<n ; i++)
369 {
370 double gz=0.;
371 for (int dir=0; dir<dimension; dir++) gz+=centres_de_gravites(i,dir)*grav(0,dir);
372
373 tab_Phydro(i) = tab_rho(i)*gz;
374 }
375 if (use_total_hydro_pressure()) remplir_champ_pression_for_EOS();
376}
