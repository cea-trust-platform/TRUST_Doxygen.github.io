48void SSOR::prepare_(const Matrice_Base& la_matrice, const DoubleVect& secmem)
49{
50 if (get_status() >= REINIT_ALL)
51 {
52 md_secmem_ = secmem.get_md_vector();
53 line_size_ = secmem.line_size();
54// Pour le prochain preconditionnement, verifier la matrice
55 avec_assert_ = 1;
56
57 if (nproc() == 1 || !(md_secmem_.non_nul())) algo_items_communs_ = 0;
58 else
59 {
60// Nombre d'items sequentiels sur ce proc
61 const int sz_tot = secmem.size_totale();
62 items_a_traiter_.reset();
63 items_a_traiter_.resize(sz_tot / line_size_, line_size_, Array_base::NOCOPY_NOINIT);
64 items_a_traiter_.set_md_vector(md_secmem_);
65 int n = MD_Vector_tools::get_sequential_items_flags(md_secmem_, items_a_traiter_, line_size_);
66 int sz = md_secmem_.valeur().get_nb_items_reels();
67
68 if (sz < 0) //!< size() est invalide, les items reels ne sont pas groupes a debut !
69 algo_items_communs_ = 1;
70 else
71 {
72 assert(sz >= n);
73 if (mp_sum(sz) > mp_sum(n)) algo_items_communs_ = 1; //!< Il y a des items partages parmi les items reels
74 else
75 {
76 algo_items_communs_ = 0;
77 items_a_traiter_.reset();
78 }
79 }
80 }
81 }
82
83 if (get_status() >= REINIT_COEFF) { /* Do nothing */ }
84 Precond_base::prepare_(la_matrice, secmem);
85}

88//! Calcule la solution du systeme lineaire: A * solution = b avec la methode de relaxation SSOR.
89int SSOR::preconditionner_(const Matrice_Base& la_matrice, const DoubleVect& b, DoubleVect& solution)
90{
91// pour compatibilite historique:
92 if (omega_ <= 0. || omega_ >= 2)
93 {
94 operator_egal(solution, b);
95 return 1;
96 }
97
98 operator_egal(solution, b);
99
100 if (sub_type(Matrice_Morse_Sym, la_matrice))
101 {
102 const Matrice_Morse_Sym& matrice = ref_cast(Matrice_Morse_Sym, la_matrice);
103 ssor(matrice, solution);
104 }
105 else if (sub_type(Matrice_Bloc_Sym, la_matrice))
106 {
107// Matrice correspondant a un vecteur multi-localisation (MD_Vector_composite)
108 const Matrice_Bloc_Sym& matrice = ref_cast(Matrice_Bloc_Sym, la_matrice);
109 ssor(matrice, solution);
110 }
111 else if (sub_type(Matrice_Bloc, la_matrice))
112 {
113// On suppose une matrice reelle-reelle et une matrice reelle-virtuelle
114 const Matrice_Bloc& mat = ref_cast(Matrice_Bloc, la_matrice);
115 const Matrice_Morse_Sym& matrice = ref_cast(Matrice_Morse_Sym, mat.get_bloc(0, 0).valeur());
116 ssor(matrice, solution);
117 }
118 else
119 {
120 Cerr << "SSOR::preconditionner not coded for type " << la_matrice.que_suis_je() << finl;
121 exit();
122 }
123 if (echange_ev_solution_) solution.echange_espace_virtuel();
124 return 1;
125}

127void traite_diagonale(const double omega, const ArrOfInt& tab1, const ArrOfInt& tab2, const ArrOfDouble& coeff, DoubleVect& vecteur)
128{
129 const int nb_lignes_a_traiter = vecteur.size_reelle_ok() ? vecteur.size_reelle() : vecteur.size_totale();
130
131 const double psi = (2. - omega) / omega;
132 const double *coeff_fortran = coeff.addr() - 1; //!< indexable par index fortran
133 const int *tab1_ptr = tab1.addr();
134 double *vect_ptr = vecteur.addr();
135 for (int i = nb_lignes_a_traiter; i; i--, tab1_ptr++, vect_ptr++)
136 {
137 const int j = *tab1_ptr;
138// Coefficient diagonale de la ligne i:
139 const double coeff_i_i = coeff_fortran[j];
140 (*vect_ptr) *= psi * coeff_i_i;
141 }
142}

146template<descente_enum _TYPE_>
147void descente_generique(const double omega, const ArrOfInt& tab1, const ArrOfInt& tab2, const ArrOfDouble& coeff, DoubleVect& vecteur)
148{
149 static constexpr bool IS_NORMAL_ASSERT = (_TYPE_ == descente_enum::NORMAL_ASSERT), IS_DIAG_ASSERT = (_TYPE_ == descente_enum::DIAG_ASSERT), NOT_DIAG = (_TYPE_ != descente_enum::DIAG && _TYPE_ != descente_enum::DIAG_ASSERT);
150
151 const int nb_lignes_a_traiter = vecteur.size_reelle_ok() ? vecteur.size_reelle() : vecteur.size_totale();
152// pointeur "fortran" vers le tableau solution (indexable avec index fortran) le pointeur est constant, pas le tableau pointe.
153 double * const sol_fortran = vecteur.addr() - 1;
154 const int *tab1_ptr = tab1.addr();
155 assert(nb_lignes_a_traiter <= tab1.size_array() + 1);
156 assert(*tab1_ptr == 1); //!< sinon 2 lignes ci-dessous fausses.
157 const int *tab2_ptr = tab2.addr();
158 const double *coeff_ptr = coeff.addr();
159 int last_tab1_de_i = *tab1_ptr;
160 tab1_ptr++;
161 for (int i = 1; i <= nb_lignes_a_traiter; i++, tab1_ptr++)
162 {
163 const int tab1_de_i = *tab1_ptr; //!< = tab1[i]
164 const int nvois = tab1_de_i - last_tab1_de_i;
165
166 if (IS_NORMAL_ASSERT || IS_DIAG_ASSERT) assert(nvois >= 0 && (tab1_de_i - 1) <= tab2.size_array());
167
168 last_tab1_de_i = tab1_de_i;
169// Ce test doit rester, sinon on pollue les autres lignes du vecteur sol avec des valeurs dependant des items communs et virtuels
170 {
171 double v_i;
172 if (NOT_DIAG)
173 {
174// Le premier coeff doit etre le coef diagonal et doit etre strictement positif
175 if (IS_NORMAL_ASSERT) assert(nvois >= 1 && (*tab2_ptr) == i && (*coeff_ptr) > 0.);
176
177 const double omega_coeff_i_i = omega / (*coeff_ptr);
178 v_i = sol_fortran[i] *= omega_coeff_i_i;
179 tab2_ptr++;
180 coeff_ptr++;
181 }
182 else //!< PRECOND DIAG !!
183 {
184// Pas de coefficient diagonal stocke (ni dans tab2 ni dans coeff), la diagonale vaut 1:
185 v_i = sol_fortran[i] *= omega;
186 }
187
188 for (int j = nvois-1; j; j--, tab2_ptr++, coeff_ptr++)
189 {
190 const int i2 = *tab2_ptr; //!< indice de colonne pour le prochain coefficient
191 const double coeff_i_i2 = *coeff_ptr;
192// la matrice n'a que des coeffs diagonaux superieurs et on a deja traite la diagonale, donc i2 > i. Pas d'items virtuels autorises !
193 if (IS_NORMAL_ASSERT || IS_DIAG_ASSERT) assert(i2 > i && i2 <= nb_lignes_a_traiter);
194// B.M.: ... en revanche, le test sur les items communs est superflu ici car la valeur sera annulee (voir **Annulation items communs**),
195// gain de perfs si on ne fait pas le test
196 sol_fortran[i2] -= coeff_i_i2 * v_i;
197 }
198 }
199 }
200}

227// Descente sur un bloc extradiagonal. Methode appelee par SSOR(const Matrice_bloc & ...)
228// vecteur: de taille "nombre de lignes de la matrice", vecteur2: "nombre de colonnes"
229void descente_bloc_extradiag_assert(const ArrOfInt& tab1, const ArrOfInt& tab2, const ArrOfDouble& coeff, const DoubleVect& vecteur, DoubleVect& vecteur2)
230{
231 const int nb_lignes = vecteur.size_reelle_ok() ? vecteur.size_reelle() : vecteur.size_totale();
232 const double *vecteur_ptr = vecteur.addr();
233 double *vecteur2_fortran_ptr = vecteur2.addr() - 1;
234 const int *tab2_fortran_ptr = tab2.addr() - 1;
235 const double *coeff_fortran_ptr = coeff.addr() - 1;
236 assert(coeff.size_array() == tab2.size_array());
237 for (int i_ligne = 0; i_ligne < nb_lignes; i_ligne++, vecteur_ptr++)
238 {
239 {
240 const double v_i = *vecteur_ptr;
241 int index = tab1[i_ligne];
242 const int index_fin = tab1[i_ligne + 1];
243// Il peut n'y avoir aucun coefficient sur la ligne => index_fin == index
244 assert(index > 0 && index_fin >= index && index_fin <= tab2.size_array() + 1);
245 const int *tab2_ptr = tab2_fortran_ptr + index;
246 const double *coeff_ptr = coeff_fortran_ptr + index;
247 for (; index < index_fin; index++, tab2_ptr++, coeff_ptr++)
248 {
249 const int i_colonne = *tab2_ptr;
250 assert(i_colonne >= 1 && i_colonne <= vecteur2.size_array());
251 const double c = *coeff_ptr;
252// pas de test item commun sur les colonnes, voir **Annulation items communs**
253 vecteur2_fortran_ptr[i_colonne] -= c * v_i;
254 }
255 }
256 }

259template<descente_enum _TYPE_>
260void descente_generique(const double omega, const ArrOfInt& tab1, const ArrOfInt& tab2, const ArrOfDouble& coeff, DoubleVect& vecteur, const ArrOfInt& items_a_traiter)
261{
262 static constexpr bool IS_NORMAL_ASSERT = (_TYPE_ == descente_enum::NORMAL_ASSERT), IS_DIAG_ASSERT = (_TYPE_ == descente_enum::DIAG_ASSERT), NOT_DIAG = (_TYPE_ != descente_enum::DIAG && _TYPE_ != descente_enum::DIAG_ASSERT);
263
264 const int nb_lignes_a_traiter = vecteur.size_reelle_ok() ? vecteur.size_reelle() : vecteur.size_totale();
265// pointeur "fortran" vers le tableau solution (indexable avec index fortran) le pointeur est constant, pas le tableau pointe.
266 double * const sol_fortran = vecteur.addr() - 1;
267 const int *flags_ptr = items_a_traiter.addr();
268 assert(nb_lignes_a_traiter <= items_a_traiter.size_array());
269 const int *tab1_ptr = tab1.addr();
270 assert(nb_lignes_a_traiter <= tab1.size_array() + 1);
271 assert(*tab1_ptr == 1); //!< sinon 2 lignes ci-dessous fausses.
272 const int *tab2_ptr = tab2.addr();
273 const double *coeff_ptr = coeff.addr();
274 int last_tab1_de_i = *tab1_ptr;
275 tab1_ptr++;
276 for (int i = 1; i <= nb_lignes_a_traiter; i++, tab1_ptr++)
277 {
278 const int tab1_de_i = *tab1_ptr; //!< = tab1[i]
279 const int nvois = tab1_de_i - last_tab1_de_i;
280
281 if (IS_NORMAL_ASSERT || IS_DIAG_ASSERT) assert(nvois >= 0 && (tab1_de_i - 1) <= tab2.size_array());
282
283 last_tab1_de_i = tab1_de_i;
284// Ce test doit rester, sinon on pollue les autres lignes du vecteur sol avec des valeurs dependant des items communs et virtuels
285 if (IS_NORMAL_ASSERT || IS_DIAG_ASSERT) assert((flags_ptr - items_a_traiter.addr()) == (i-1));
286
287 const int item_a_traiter = *(flags_ptr++);
288 if (item_a_traiter)
289 {
290 double v_i;
291 if (NOT_DIAG)
292 {
293// Le premier coeff doit etre le coef diagonal et doit etre strictement positif
294 if (IS_NORMAL_ASSERT) assert(nvois >= 1 && (*tab2_ptr) == i && (*coeff_ptr) > 0.);
295
296 const double omega_coeff_i_i = omega / (*coeff_ptr);
297 v_i = sol_fortran[i] *= omega_coeff_i_i;
298 tab2_ptr++;
299 coeff_ptr++;
300 }
301 else //!< PRECOND DIAG !!
302 {
303// Pas de coefficient diagonal stocke (ni dans tab2 ni dans coeff), la diagonale vaut 1:
304 v_i = sol_fortran[i] *= omega;
305 }
306
307 for (int j = nvois-1; j; j--, tab2_ptr++, coeff_ptr++)
308 {
309 const int i2 = *tab2_ptr; //!< indice de colonne pour le prochain coefficient
310 const double coeff_i_i2 = *coeff_ptr;
311// la matrice n'a que des coeffs diagonaux superieurs et on a deja traite la diagonale, donc i2 > i.
312 if (IS_NORMAL_ASSERT || IS_DIAG_ASSERT) assert(i2 > i && i2 <= vecteur.size_totale());
313
314// B.M.: ... en revanche, le test sur les items communs est superflu ici car
315// la valeur sera annulee (voir **Annulation items communs**), gain de perfs si on ne fait pas le test
316 sol_fortran[i2] -= coeff_i_i2 * v_i;
317 }
318 }
319 else
320 {
321// **Annulation items communs**
322// c'est la derniere fois qu'on ecrit dans sol_fortran[i] car la matrice est diagonale superieure. Pour eviter le test sur items_a_traiter_
323// dans la remontee, on annule la solution pour les items communs et virtuels (sol_fortran[i] ne sera plus modifie ensuite dans la descente)
324 sol_fortran[i] = 0.;
325 coeff_ptr += nvois;
326 tab2_ptr += nvois;
327 }
328 }
329// **Annulation items communs** : Pour la remontee il faut annuler les valeurs dans les cases virtuelles
330 {
331 const int fin = vecteur.size_totale();
332 for (int i = nb_lignes_a_traiter+1; i <= fin; i++)
333 sol_fortran[i] = 0.;
334 }
335}

396enum class remontee_enum { NORMAL , NORMAL_ASSERT , DIAG_OK_ASSERT , DIAG , DIAG_ASSERT };
397
398template<remontee_enum _TYPE_>
399void remontee_generique(const double omega, const ArrOfInt& tab1, const ArrOfInt& tab2, const ArrOfDouble& coeff, DoubleVect& vecteur)
400{
401 static constexpr bool IS_NORMAL_ASSERT = (_TYPE_ == remontee_enum::NORMAL_ASSERT), IS_DIAG_OK_ASSERT = (_TYPE_ == remontee_enum::DIAG_OK_ASSERT), IS_DIAG_ASSERT = (_TYPE_ == remontee_enum::DIAG_ASSERT),
402 NOT_DIAG = (_TYPE_ != remontee_enum::DIAG && _TYPE_ != remontee_enum::DIAG_ASSERT);
403
404 const int nb_lignes_a_traiter = vecteur.size_reelle_ok() ? vecteur.size_reelle() : vecteur.size_totale();
405 const double psi = IS_DIAG_OK_ASSERT ? -1e10 : (2. - omega) / omega;
406
407// pointeur "fortran" vers le tableau solution (indexable avec index fortran)
408 const double *const sol_fortran = vecteur.addr() - 1;
409 const int *tab1_ptr = tab1.addr() + nb_lignes_a_traiter;
410 int last_tab1_de_i = *tab1_ptr;
411 tab1_ptr--;
412// On ne va pas a la fin de tab2 car on n'est pas sur que nb_lignes_a_traiter = tab1.size_array() :
413// -2 car last_tab1_de_i est l'indice du premier coefficient de la ligne suivante en fortran
414// (dont -1 pour fortran->c et -1 pour passer au dernier coeff de la ligne precedente)
415 const int *tab2_ptr = tab2.addr() + last_tab1_de_i - 2;
416 const double *coeff_ptr = coeff.addr() + last_tab1_de_i - 2;
417 double *soli_ptr = vecteur.addr() + nb_lignes_a_traiter - 1;
418 for (int i = nb_lignes_a_traiter; i; i--, tab1_ptr--, soli_ptr--)
419 {
420 const int tab1_de_i = *tab1_ptr; //!< = tab1[i]
421 const int nvois = last_tab1_de_i - tab1_de_i;
422
423 if (IS_NORMAL_ASSERT || IS_DIAG_OK_ASSERT || IS_DIAG_ASSERT) assert(nvois >= 0 && tab1_de_i > 0);
424
425 last_tab1_de_i = tab1_de_i;
426// Ce test doit rester car il ne faut pas modifier sol[i] pour les items communs
427// et virtuels (ils sont nuls et doivent le rester pour ne pas polluer les autres lignes):
428 if (1)
429 {
430// Operation "diagonale":
431 double x = 0.;
432 for (int j = nvois - 1; j; j--, tab2_ptr--, coeff_ptr--)
433 {
434 const int i2 = *tab2_ptr;
435 const double coeff_i_i2 = *coeff_ptr;
436// Operation nulle pour les items communs et virtuels
437// (on a annule le second membre lors de la descente)
438 x += coeff_i_i2 * sol_fortran[i2];
439 }
440
441 if (NOT_DIAG)
442 {
443// ici coeff_ptr est le coeff diagonal
444 if (IS_NORMAL_ASSERT || IS_DIAG_OK_ASSERT) assert((*tab2_ptr) == i);
445
446 const double coeff_i_i = *coeff_ptr;
447 const double omega_coeff_i_i = omega / coeff_i_i;
448// Si IS_DIAG_OK_ASSERT : La diagonale a deja ete multipliee par psi * coeff_i_i

527template<remontee_enum _TYPE_>
528void remontee_generique(const double omega, const ArrOfInt& tab1, const ArrOfInt& tab2, const ArrOfDouble& coeff, DoubleVect& vecteur, const ArrOfInt& items_a_traiter)
529{
530 static constexpr bool IS_NORMAL_ASSERT = (_TYPE_ == remontee_enum::NORMAL_ASSERT), IS_DIAG_OK_ASSERT = (_TYPE_ == remontee_enum::DIAG_OK_ASSERT), IS_DIAG_ASSERT = (_TYPE_ == remontee_enum::DIAG_ASSERT),
531 NOT_DIAG = (_TYPE_ != remontee_enum::DIAG && _TYPE_ != remontee_enum::DIAG_ASSERT);
532
533 const int nb_lignes_a_traiter = vecteur.size_reelle_ok() ? vecteur.size_reelle() : vecteur.size_totale();
534 const int *flags_ptr = items_a_traiter.addr() + nb_lignes_a_traiter - 1;
535 assert(nb_lignes_a_traiter <= items_a_traiter.size_array());
536 const double psi = IS_DIAG_OK_ASSERT ? -1e10 : (2. - omega) / omega;
537
538// pointeur "fortran" vers le tableau solution (indexable avec index fortran)
539 const double * const sol_fortran = vecteur.addr() - 1;
540 const int *tab1_ptr = tab1.addr() + nb_lignes_a_traiter;
541 int last_tab1_de_i = *tab1_ptr;
542 tab1_ptr--;
543// On ne va pas a la fin de tab2 car on n'est pas sur que nb_lignes_a_traiter = tab1.size_array() :
544// -2 car last_tab1_de_i est l'indice du premier coefficient de la ligne suivante en fortran
545// (dont -1 pour fortran->c et -1 pour passer au dernier coeff de la ligne precedente)
546 const int *tab2_ptr = tab2.addr() + last_tab1_de_i - 2;
547 const double *coeff_ptr = coeff.addr() + last_tab1_de_i - 2;
548 double * soli_ptr = vecteur.addr() + nb_lignes_a_traiter - 1;
549 for (int i = nb_lignes_a_traiter; i; i--, tab1_ptr--, soli_ptr--)
550 {
551 const int tab1_de_i = *tab1_ptr; //!< = tab1[i]
552 const int nvois = last_tab1_de_i - tab1_de_i;
553 if (IS_NORMAL_ASSERT || IS_DIAG_OK_ASSERT || IS_DIAG_ASSERT) assert(nvois >= 0 && tab1_de_i > 0);
554
555 last_tab1_de_i = tab1_de_i;
556// Ce test doit rester car il ne faut pas modifier sol[i] pour les items communs
557// et virtuels (ils sont nuls et doivent le rester pour ne pas polluer les autres lignes):
558 if (*(flags_ptr--))
559 {
560// Operation "diagonale":
561 double x = 0.;
562 for (int j = nvois-1; j; j--, tab2_ptr--, coeff_ptr--)
563 {
564 const int i2 = *tab2_ptr;
565 const double coeff_i_i2 = *coeff_ptr;
566// Operation nulle pour les items communs et virtuels
567// (on a annule le second membre lors de la descente)
568 x += coeff_i_i2 * sol_fortran[i2];
569 }
570 if (NOT_DIAG)
571 {
572// ici coeff_ptr est le coeff diagonal
573 if (IS_NORMAL_ASSERT || IS_DIAG_OK_ASSERT) assert((*tab2_ptr) == i);
574
575 const double coeff_i_i = *coeff_ptr;
576 const double omega_coeff_i_i = omega / coeff_i_i;
577// Si IS_DIAG_OK_ASSERT : La diagonale a deja ete multipliee par psi * coeff_i_i
578 *soli_ptr = IS_DIAG_OK_ASSERT ? ((*soli_ptr) - x) * omega_coeff_i_i : (*soli_ptr) * psi * omega - x * omega_coeff_i_i;
579 coeff_ptr--;
580 tab2_ptr--;
581 }
582 else //!< PRECOND DIAG !!
583 {
584// Preconditionnement diagonal, pas de coefficient diagonal stocke:
585 *soli_ptr = ((*soli_ptr) * psi - x) * omega;
586 }
587 }
588 else
589 {
590 assert((*soli_ptr) == 0.);
591 coeff_ptr -= nvois;
592 tab2_ptr -= nvois;
593 }
594 }
595}
