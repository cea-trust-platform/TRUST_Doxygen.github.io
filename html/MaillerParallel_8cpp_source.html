47struct BlocData
48{
49 void add_bloc(Domaine& domaine, const ArrsOfDouble& coord_ijk) const;
50// xmin_tot = 0 implicitement
51// xmax_tot = nombre total de sommets dans le maillage
52// xmax = indice du dernier sommet + 1
53// xmin = indice du premier sommet (donc xmax - xmin = nombre de sommets du bloc courant)
54 ArrOfInt xmax_tot_;
55 ArrOfInt xmin_, xmax_;
56 Noms bord_xmin_, bord_xmax_;
57 inline int nb_som(int i) const
58 {
59 return xmax_[i] - xmin_[i];
60 }
61 inline int nb_elem(int i) const
62 {
63 return xmax_[i] - xmin_[i] - 1;
64 }
65 void add_faces(Zone& zone, const Nom& nombord, int offset_sommets, int dir, int cote_max) const;
66};

68void BlocData::add_bloc(Domaine& domaine, const ArrsOfDouble& coord_ijk) const
69{
70 if (nb_elem(0) < 1 || nb_elem(1) < 1 || nb_elem(2) < 1)
71 {
72 Cerr << "Error : not enough nodes in a block, increase mesh size or decrease processor count" << finl;
73 Process::exit();
74 }
75 const int dim = 3;
76// Les sommets
77 const int old_nb_som = domaine.les_sommets().dimension(0);
78 {
79 DoubleTab& sommets = domaine.les_sommets();
80 const int nsom = nb_som(0) * nb_som(1) * nb_som(2);
81 sommets.resize(old_nb_som + nsom, dim);
82
83 int index = old_nb_som;
84 int i, j, k;
85 for (k = xmin_[2]; k < xmax_[2]; k++)
86 {
87 for (j = xmin_[1]; j < xmax_[1]; j++)
88 {
89 for (i = xmin_[0]; i < xmax_[0]; i++)
90 {
91 sommets(index, 0) = coord_ijk[0][i];
92 sommets(index, 1) = coord_ijk[1][j];
93 if (dim==3)
94 sommets(index, 2) = coord_ijk[2][k];
95 index++;
96 }
97 }
98 }
99 }
100// Les elements
101 {
102 assert(dim==3);
103 Zone& zone = domaine.zone(0);
104 IntTab& elements = zone.les_elems();
105 const int nb_som_par_element = 8;
106
107// stride des indices de sommets:
108 const int dx = 1;
109 const int dy = nb_som(0);
110 const int dz = dy * nb_som(1);
111// nombre d'elements dans chaque direction
112 const int nx = nb_elem(0);
113 const int ny = nb_elem(1);
114 const int nz = nb_elem(2);
115 const int nb_elem_old = zone.les_elems().dimension(0);
116 const int nbelem = nx * ny * nz;
117 elements.resize(nb_elem_old + nbelem, nb_som_par_element);
118
119 int index = nb_elem_old;
120 int i, j, k;
121 for (k = 0; k < nz; k++)
122 {
123 for (j = 0; j < ny; j++)
124 {
125 for (i = 0; i < nx; i++)
126 {
127 int n = old_nb_som + k * dz + j * dy + i * dx;
128 elements(index, 0) = n;
129 elements(index, 1) = n + dx;
130 elements(index, 2) = n + dy;
131 elements(index, 3) = n + dx + dy;
132 elements(index, 4) = n + dz;
133 elements(index, 5) = n + dz + dx;
134 elements(index, 6) = n + dz + dy;
135 elements(index, 7) = n + dz + dx + dy;
136 index++;
137 }
138 }
139 }
140 }
141// Les bords
142 Zone& zone = domaine.zone(0);
143 for (int dir = 0; dir < 3; dir++)
144 {
145 if (xmin_[dir] == 0)
146 add_faces(zone, bord_xmin_[dir], old_nb_som, dir, 0);
147 if (xmax_[dir] == xmax_tot_[dir])
148 add_faces(zone, bord_xmax_[dir], old_nb_som, dir, 1);
149 }
150}

152void BlocData::add_faces(Zone& zone, const Nom& nombord, int offset_sommets, int dir, int cote_max) const
153{
154 int nfaces1, nfaces2;
155 int increment1, increment2;
156 int increment_dir;
157 switch(dir)
158 {
159 case 0:
160 nfaces1 = nb_elem(1);
161 nfaces2 = nb_elem(2);
162 increment1 = nb_som(0);
163 increment2 = nb_som(0) * nb_som(1);
164 increment_dir = 1;
165 break;
166 case 1:
167 nfaces1 = nb_elem(0);
168 nfaces2 = nb_elem(2);
169 increment1 = 1;
170 increment2 = nb_som(0) * nb_som(1);
171 increment_dir = nb_som(0);
172 break;
173 default:
174 nfaces1 = nb_elem(0);
175 nfaces2 = nb_elem(1);
176 increment1 = 1;
177 increment2 = nb_som(0);
178 increment_dir = nb_som(0)*nb_som(1);
179 break;
180 }
181 if (cote_max)
182 offset_sommets += increment_dir * (nb_som(dir)-1);
183
184 IntTab& faces = zone.frontiere(nombord).faces().les_sommets();
185 int count = faces.dimension(0);
186 zone.frontiere(nombord).faces().dimensionner(count + nfaces1 * nfaces2);
187 for (int i = 0; i < nfaces2; i++)
188 {
189 int sommet = offset_sommets;
190 for (int j = 0; j < nfaces1; j++)
191 {
192 faces(count, 0) = sommet;
193 faces(count, 1) = sommet + increment1;
194 faces(count, 2) = sommet + increment2;
195 faces(count, 3) = sommet + increment1 + increment2;
196 sommet += increment1;
197 count++;
198 }
199 offset_sommets += increment2;
200 }
201}

203static int bbox_intersection(const DoubleTab& bboxes, int i, int j, double epsilon)
204{
205 const int dim = bboxes.dimension(1) / 2;
206 for (int k = 0; k < dim; k++)
207 {
208 double min1 = bboxes(i, k) - epsilon;
209 double max1 = bboxes(i, dim+k) + epsilon;
210 double min2 = bboxes(j, k);
211 double max2 = bboxes(j, dim+k);
212 if (min1 > max2 || max1 < min2)
213 return 0;
214 }
215 return 1;
216}

223//! for a given set of local coordinates, finds which other processors<br>share the same nodes.<br>match must be sized to Process::nproc() and will be filled with, for each processor,<br>the indexes of coordinates that the corresponding processor shares with me.<br>Nodes are listed in the same order on couples of processors sharing a list of nodes<br>match[Process::me()] will be left empty.
224void find_matching_coordinates(const DoubleTab& coords,
225 ArrsOfInt& match,
226 double epsilon)
227{
228 int i, j;
229 const int dim = coords.dimension(1);
230 const int ncoord = coords.dimension(0);
231 const int nproc = Process::nproc();
232 const int moi = Process::me();
233// Fill nodes coordinates and bounding boxes
234 DoubleTab bounding_boxes(nproc, dim*2);
235 {
236 const double big_val = 1.e36;
237 ArrOfDouble min_coord(dim);
238 min_coord= big_val;
239 ArrOfDouble max_coord(dim);
240 max_coord= -big_val;
241
242 for (i = 0; i < ncoord; i++)
243 {
244 for (j = 0; j < dim; j++)
245 {
246 const double x = coords(i, j);
247 if (x > max_coord[j])
248 max_coord[j] = x;
249 if (x < min_coord[j])
250 min_coord[j] = x;
251 }
252 }
253// broadcast a tous les processeurs
254 for (i = 0; i < nproc; i++)
255 {
256 for (j = 0; j < dim; j++)
257 {
258 bounding_boxes(i, j) = min_coord[j];
259 bounding_boxes(i, j+dim) = max_coord[j];
260 }
261 }
262 envoyer_all_to_all(bounding_boxes, bounding_boxes);
263 }
264// Exchange boundary nodes coordinates with other processors
265 ArrOfInt all_pe;
266 all_pe.set_smart_resize(1);
267 for (i = 0; i < nproc; i++)
268 if (i != moi && bbox_intersection(bounding_boxes, moi, i, epsilon))
269 all_pe.append_array(i);
270
271 Schema_Comm schema;
272 schema.set_send_recv_pe_list(all_pe, all_pe);
273 schema.begin_comm();
274 Octree_Double octree;
275 octree.build_nodes(coords, 0 /* do not include virtual nodes */);
276
277// Send nodes coordinates to processors with non empty intersection
278// temporary array for octree functions:
279 ArrOfInt nodes_list;
280 nodes_list.set_smart_resize(1);
281 int ipe;
282 for (ipe = 0; ipe < all_pe.size_array(); ipe++)
283 {
284// Search nodes within the bounding box of the other processor:
285 const int pe = all_pe[ipe]; //!< processor number
286 const double xmin = bounding_boxes(pe, 0) - epsilon;
287 const double ymin = bounding_boxes(pe, 1) - epsilon;
288 const double zmin = (dim == 3) ? bounding_boxes(pe, 2) - epsilon : -epsilon;
289 const double xmax = bounding_boxes(pe, dim) + epsilon;
290 const double ymax = bounding_boxes(pe, dim+1) + epsilon;
291 const double zmax = (dim == 3) ? bounding_boxes(pe, dim+2) + epsilon : epsilon;
292 octree.search_elements_box(xmin, ymin, zmin, xmax, ymax, zmax, nodes_list);
293
294 Sortie& buffer = schema.send_buffer(pe);
295 const int n = nodes_list.size_array();
296 for (i = 0; i < n; i++)
297 {
298 const int som = nodes_list[i];
299 const double x = coords(som, 0);
300 const double y = coords(som, 1);
301 const double z = (dim==3) ? coords(som, 2) : 0;
302// test again if inside bounding box (octree test gives more nodes)
303 if (x>=xmin && x<=xmax && y>=ymin && y<=ymax && z>=zmin && z<=zmax)
304 {
305 buffer << x << y << z;
306 }
307 }
308 }
309 schema.echange_taille_et_messages();
310 for (ipe = 0; ipe < all_pe.size_array(); ipe++)
311 {
312 const int pe = all_pe[ipe]; //!< processor number
313 Entree& buffer = schema.recv_buffer(pe);
314 ArrOfInt& resu = match[pe];
315 resu.set_smart_resize(1);
316 resu.resize_array(0);
317 while(1)
318 {
319 double x, y, z;
320 buffer >> x >> y >> z;
321 if (buffer.eof())
322 break;
323// Seach for this node in my list
324 octree.search_elements_box(x-epsilon, y-epsilon, z-epsilon,
325 x+epsilon, y+epsilon, z+epsilon, nodes_list);
326// Refine search to keep only matching nodes:
327 octree.search_nodes_close_to(x, y, z, coords, nodes_list, epsilon);
328 const int n = nodes_list.size_array();
329 if (n > 1)
330 {
331 Cerr << "Error in automatic joint builder: more than one node within epsilon tolerance\n"
332 << " node coord = " << x << " " << y << " " << z << " tolerance= " << epsilon << finl;
333 Process::exit();
334 }
335 if (n == 1)
336 resu.append_array(nodes_list[0]);
337 }
338 }
339 schema.end_comm();
340}

346//! finds all faces of all elements that<br>- have only one neighbour elements<br>- and do not belong to a boundary<br>(these are joint faces)<br>Fills "faces" with the local node numbers of the faces that have been found.
347static void find_joint_faces(const Zone& zone, IntTab& faces)
348{
349 Static_Int_Lists som_elem;
350 const IntTab& elements = zone.les_elems();
351 const int nb_som = zone.domaine().nb_som();
352 construire_connectivite_som_elem(nb_som, elements, som_elem, 0 /* do not include virtual elements */);
353 const int nb_som_faces = zone.type_elem().valeur().nb_som_face();
354 faces.resize(0, nb_som_faces);
355 faces.set_smart_resize(1);
356 IntTab faces_element_reference;
357 zone.type_elem().valeur().get_tab_faces_sommets_locaux(faces_element_reference);
358 const int nb_faces_elem = faces_element_reference.dimension(0);
359// Mark faces of elements that are on a boundary
360 const int nb_elem = elements.dimension(0);
361 ArrOfBit boundary_faces(nb_elem * nb_faces_elem);
362 boundary_faces = 0;
363 ArrOfInt une_face(nb_som_faces);
364 ArrOfInt liste_elements;
365 liste_elements.set_smart_resize(1);
366 const int nb_boundaries = zone.nb_front_Cl();
367 for (int i_boundary = 0; i_boundary < nb_boundaries; i_boundary++)
368 {
369 const IntTab& faces_front = zone.frontiere(i_boundary).faces().les_sommets();
370 const int nb_faces_bord = faces_front.dimension(0);
371 for (int j = 0; j < nb_faces_bord; j++)
372 {
373 for (int k = 0; k < nb_som_faces; k++)
374 une_face[k] = faces_front(j, k);
375 find_adjacent_elements(som_elem, une_face, liste_elements);
376 if (liste_elements.size_array() != 1)
377 {
378 Cerr << "Error in MaillerParallel::find_joint_faces: boundary face ArrOfInt=" << une_face
379 << " does not have exactly 1 neighbour element: " << liste_elements;
380 Process::exit();
381 }
382 const int elem = liste_elements[0];
383 const int face_locale = Faces_builder::chercher_face_element(elements, faces_element_reference, une_face, elem);
384 boundary_faces.setbit(elem * nb_faces_elem + face_locale);
385 }
386 }
387// Browse all faces of all elements, find faces with only one neighbour
388 for (int elem = 0; elem < nb_elem; elem++)
389 {
390 for (int face_locale = 0; face_locale < nb_faces_elem; face_locale++)
391 {
392 if (boundary_faces[elem * nb_faces_elem + face_locale])
393 continue;
394 int i;
395 for (i = 0; i < nb_som_faces; i++)
396 {
397 const int sommet_local = faces_element_reference(face_locale, i);
398 une_face[i] = elements(elem, sommet_local);
399 }
400 find_adjacent_elements(som_elem, une_face, liste_elements);
401 if (liste_elements.size_array() == 1)
402 {
403 const int n = faces.dimension(0);
404 faces.resize_dim0(n+1);
405 for (i = 0; i < nb_som_faces; i++)
406 faces(n, i) = une_face[i];
407 }
408 }
409 }
410 Process::Journal() << "Domain " << zone.domaine().le_nom() << " has " << faces.dimension(0) << " undeclared boundary faces candidates for joint faces" << finl;
411}

413static void auto_build_joints(Zone& zone, const int epaisseur_joint)
414{
415 const double epsilon = 1e-10;
416 int i, j;
417// Find joint faces (faces with 1 neighbour element not registered in boundary faces)
418 IntTab faces;
419 find_joint_faces(zone, faces);
420
421 const DoubleTab& sommets = zone.domaine().les_sommets();
422// List of unique local boundary node numbers
423 const int dim = sommets.dimension(1);
424
425 {
426 ArrOfInt boundary_nodes_index;
427 boundary_nodes_index = faces; //!< copie du tableau
428 array_trier_retirer_doublons(boundary_nodes_index);
429 const int ncoord = boundary_nodes_index.size_array();
430// Fill nodes coordinates and bounding boxes
431 DoubleTab coord;
432 coord.resize(ncoord, dim, Array_base::NOCOPY_NOINIT);
433 for (i = 0; i < ncoord; i++)
434 {
435 const int som = boundary_nodes_index[i];
436 for (j = 0; j < dim; j++)
437 coord(i, j) = sommets(som, j);
438 }
439 ArrsOfInt match(Process::nproc());
440 find_matching_coordinates(coord, match, epsilon);
441
442 for (int pe = 0; pe < match.size(); pe++)
443 {
444 const ArrOfInt& list = match[pe];
445 const int n = list.size_array();
446 if (n > 0)
447 {
448 Joint& joint = zone.faces_joint().add(Joint());
449 joint.nommer("Joint_i");
450 joint.associer_zone(zone);
451 joint.affecte_epaisseur(epaisseur_joint);
452 joint.affecte_PEvoisin(pe);
453 joint.faces().typer(zone.type_elem().valeur().type_face());
454 ArrOfInt& sommets_joint = joint.set_joint_item(Joint::SOMMET).set_items_communs();
455 sommets_joint.resize_array(n, Array_base::NOCOPY_NOINIT);
456 for (i = 0; i < n; i++)
457 sommets_joint[i] = boundary_nodes_index[list[i]];
458 sommets_joint.ordonne_array();
459 Process::Journal() << "Joint sommets with pe " << pe << " has " << n << " nodes" << finl;
460 }
461 }
462 }
463
464 {
465// Find matching faces (match the coordinates of the centers)
466 const int nfaces = faces.dimension(0);
467 const int nb_som_faces = faces.dimension(1);
468 DoubleTab coord(nfaces, dim); //!< init with 0.
469 const double facteur = 1. / (double)nb_som_faces;
470 for (i = 0; i < nfaces; i++)
471 {
472 for (j = 0; j < nb_som_faces; j++)
473 {
474 const int som = faces(i,j);
475 for (int k = 0; k < dim; k++)
476 coord(i, k) += sommets(som, k) * facteur;
477 }
478 }
479 ArrsOfInt match(Process::nproc());
480 find_matching_coordinates(coord, match, epsilon);
481
482 for (int pe = 0; pe < match.size(); pe++)
483 {
484 const ArrOfInt& list = match[pe];
485 const int n = list.size_array();
486 if (n > 0)
487 {
488 REF(Joint) ref_joint;
489 int ii;
490 for (ii = 0; ii < zone.faces_joint().size(); ii++)
491 {
492 Joint& joint = zone.faces_joint()[ii];
493 if (joint.PEvoisin() == pe)
494 {
495 ref_joint = joint;
496 break;
497 }
498 }
499 if (!ref_joint.non_nul())
500 {
501 Joint& joint = zone.faces_joint().add(Joint());
502 joint.nommer("Joint_i");
503 joint.associer_zone(zone);
504 joint.affecte_epaisseur(epaisseur_joint);
505 joint.affecte_PEvoisin(pe);
506 joint.faces().typer(zone.type_elem().valeur().type_face());
507 ref_joint = joint;
508 }
509 Faces& les_faces = ref_joint.valeur().faces();
510 les_faces.dimensionner(n);
511 IntTab& faces_sommets = les_faces.les_sommets();
512 for (ii = 0; ii < n; ii++)
513 {
514 const int num_face = list[ii];
515 for (j = 0; j < nb_som_faces; j++)
516 faces_sommets(ii, j) = faces(num_face, j);
517 }
518 Process::Journal() << "Joint faces with pe " << pe << " has " << n << " faces" << finl;
519 }
520 }
521 }
522// Tri des joints dans l'ordre croissant des processeurs
523 Scatter::trier_les_joints(zone.faces_joint());
524}

526Entree& MaillerParallel::interpreter(Entree& is)
527{
528 if (dimension != 3)
529 {
530 Cerr << "Error: Mailler_Parallel is only coded for dimension 3" << finl;
531 barrier();
532 exit();
533 }
534
535 Nom nom_domaine;
536 ArrOfInt nb_noeuds;
537 ArrOfInt decoupage;
538 int epaisseur_joint;
539 ArrOfInt perio(3);
540 Noms fonctions_coord(3);
541 Noms nom_bords_min(3);
542 Noms nom_bords_max(3);
543 nom_bords_min[0] = "xmin";
544 nom_bords_min[1] = "ymin";
545 nom_bords_min[2] = "zmin";
546 nom_bords_max[0] = "xmax";
547 nom_bords_max[1] = "ymax";
548 nom_bords_max[2] = "zmax";
549 Noms fichier_coord(3);
550 IntTab mapping;
551 {
552 Param param(que_suis_je());
553 param.ajouter("domain", &nom_domaine, Param::REQUIRED);
554 param.ajouter("nb_nodes", &nb_noeuds, Param::REQUIRED);
555 param.ajouter("splitting", &decoupage, Param::REQUIRED);
556 param.ajouter("ghost_thickness", &epaisseur_joint, Param::REQUIRED);
557 param.ajouter_flag("perio_x", &perio[0]);
558 param.ajouter_flag("perio_y", &perio[1]);
559 param.ajouter_flag("perio_z", &perio[2]);
560 param.ajouter("function_coord_x", &fonctions_coord[0]);
561 param.ajouter("function_coord_y", &fonctions_coord[1]);
562 param.ajouter("function_coord_z", &fonctions_coord[2]);
563 param.ajouter("file_coord_x", &fichier_coord[0]);
564 param.ajouter("file_coord_y", &fichier_coord[1]);
565 param.ajouter("file_coord_z", &fichier_coord[2]);
566 param.ajouter("boundary_xmin", &nom_bords_min[0]);
567 param.ajouter("boundary_xmax", &nom_bords_max[0]);
568 param.ajouter("boundary_ymin", &nom_bords_min[1]);
569 param.ajouter("boundary_ymax", &nom_bords_max[1]);
570 param.ajouter("boundary_zmin", &nom_bords_min[2]);
571 param.ajouter("boundary_zmax", &nom_bords_max[2]);
572 param.ajouter("mapping", &mapping);
573 param.lire_avec_accolades(is);
574 }
575
576 ArrsOfDouble coord_ijk(3);
577 for (int dir=0; dir<3; dir++)
578 {
579 const int n = nb_noeuds[dir];
580 ArrOfDouble& coord = coord_ijk[dir];
581 coord.resize_array(n);
582 Cerr << "Building coordinates for direction " << dir << finl;
583 if (fonctions_coord[dir] != "??")
584 {
585 Nom chaine = fonctions_coord[dir];
586 Cerr << " Interpreting expression " << chaine << finl;
587 Parser_U p;
588 Noms noms_dir(3);
589 noms_dir[0]="X";
590 noms_dir[1]="Y";
591 noms_dir[2]="Z";
592 p.setNbVar(1);
593 p.setString(chaine);
594 p.addVar(noms_dir[dir]);
595 p.parseString();
596 for (int i = 0; i < n; i++)
597 {
598 p.setVar(0, (double) i / (double) (nb_noeuds[dir]-1));
599 coord[i] = p.eval();
600 if (i > 0 && coord[i] <= coord[i-1])
601 {
602 Cerr << "Error in fonctions_coord parameter " << chaine << " : must be strictly increasing" << finl;
603 Process::exit();
604 }
605 }
606 Cerr << " Build mesh covering [ " << coord[0] << " , " << coord[n-1] << " ]" << finl;
607 }
608 else if (fichier_coord[dir] != "??")
609 {
610 Cerr << " Reading file " << fichier_coord[dir] << finl;
611 if (Process::je_suis_maitre())
612 {
613 EFichier f(fichier_coord[dir]);
614 for (int i = 0; i < n; i++)
615 f >> coord[i];
616 }
617 envoyer_broadcast(coord, 0);
618 Cerr << " Found mesh covering [ " << coord[0] << " , " << coord[n-1] << " ]" << finl;
619 }
620 else
621 {
622 Cerr << " Uniform mesh covering [0,1]" << finl;
623 for (int i = 0; i < n; i++)
624 {
625 coord[i] = (double) i / (double)(n-1);
626 }
627 }
628 }
629
630 Objet_U& obj = objet(nom_domaine);
631 if(!sub_type(Domaine, obj))
632 {
633 Cerr << "obj : " << nom_domaine << " is not an object of type Domaine !" << finl;
634 exit();
635 }
636
637 Domaine& domaine = ref_cast(Domaine, obj);
638
639 const int numproc = Process::me();
640
641 static Stat_Counter_Id stats = statistiques().new_counter(0 /* Level */, "Mailler_Parallel", 0 /* Group */);
642
643 const int dim = Objet_U::dimension;
644 if (nb_noeuds.size_array() != dim)
645 {
646 Cerr << "MaillerParallel::construire_domaine wrong dimension for nb_noeuds : nb_noeuds=" << nb_noeuds << finl;
647 barrier();
648 exit();
649 }
650
651 if (decoupage.size_array() != dim)
652 {
653 Cerr << "MaillerParallel::construire_domaine wrong dimension for nb_noeuds : nb_noeuds=" << nb_noeuds << finl;
654 barrier();
655 exit();
656 }
657
658 {
659 int n = 1;
660 for (int i = 0; i < dim; i++)
661 n *= decoupage[i];
662 if (n != Process::nproc())
663 {
664 Cerr << "MaillerParallel::construire_domaine decoupage does not match nproc" << finl;
665 barrier();
666 exit();
667 }
668 }
669 statistiques().begin_count(stats);
670
671// Position du bloc correspondant a numproc dans le decoupage i,j,k:
672 ArrOfInt i_proc(dim);
673 if (mapping.dimension(0) == 0)
674 {
675// Pas de mapping fourni, on prend decoupe par defaut
676 int i;
677 int reste = numproc;
678
679 for (i = 0; i < dim; i++)
680 {
681// Nombre de parties dans la direction dim_2:
682 const int n_parties = decoupage[i];
683// Position du processeur courant dans cette direction:
684 i_proc[i] = reste % n_parties;
685 reste = reste / n_parties;
686 }
687 }
688 else
689 {
690 if (mapping.dimension(0) != Process::nproc() || mapping.dimension(1) != 3)
691 {
692 Cerr << "Error: processor mapping should be of dimension " << Process::nproc() << " x 3" << finl;
693 Process::exit();
694 }
695 i_proc[0] = mapping(numproc,0);
696 i_proc[1] = mapping(numproc,1);
697 i_proc[2] = mapping(numproc,2);
698 }
699// Indice du premier element du bloc associe au processeur dans la direction i:
700// Attention, en cas de periodique, i_premier_element sera negatif.
701 ArrOfInt i_premier_element(dim);
702// Taille du bloc associe au processeur
703 ArrOfInt nb_elements(dim);
704 {
705 int i;
706 for (i = 0; i < dim; i++)
707 {
708 const int nb_elem_tot = nb_noeuds[i] - 1; //!< Nb total de noeuds dans la direction i
709 i_premier_element[i] = nb_elem_tot * i_proc[i] / decoupage[i];
710 const int i_dernier_element = nb_elem_tot * (i_proc[i] + 1) / decoupage[i] - 1;
711 nb_elements[i] = i_dernier_element - i_premier_element[i] + 1;
712
713 if (perio[i] && decoupage[i] > 1)
714 {
715 int decalage = nb_elem_tot / decoupage[i] / 2;
716 i_premier_element[i] -= decalage;
717 }
718 }
719 }
720
721// On cree une zone pour le domaine
722 Zone empty_zone;

724// Nom de la zone
725 {
726 Nom nom_zone("Proc");
727 nom_zone += Nom(numproc);
728 nom_zone += "sur";
729 nom_zone += Nom(Process::nproc());
730 zone.nommer(nom_zone);
731 }
732 zone.associer_domaine(domaine);

734 Elem_geom& elem = zone.type_elem();
735 switch(dim)
736 {
737 case 3:
738 elem.typer("Hexaedre");
739 break;
740 default:
741 Cerr << "MaillerParallel::construire_domaine erreur" << finl;
742 exit();
743 }
744 elem.valeur().associer_zone(zone);
745 }
746
747 BlocData data;
748 Noms liste_bords_perio;
749 Bords& bords = zone.faces_bord();
750 data.bord_xmin_ = nom_bords_min;
751 data.bord_xmax_ = nom_bords_max;
752 for (int dir = 0; dir < 3; dir++)
753 {
754 Nom nomdir;
755 if (dir == 0) nomdir = "x";
756 else if (dir == 1) nomdir = "y";
757 else nomdir = "z";
758 if (perio[dir])
759 {
760 data.bord_xmax_[dir] = data.bord_xmin_[dir];
761 bords.add(Bord()).nommer(data.bord_xmin_[dir]);
762 }
763 else
764 {
765 bords.add(Bord()).nommer(data.bord_xmin_[dir]);
766 bords.add(Bord()).nommer(data.bord_xmax_[dir]);
767 }
768 }
769 data.xmax_tot_ = nb_noeuds;
770
771 ArrOfInt nblocs(3);
772 nblocs= 1;
773 for (int i = 0; i < 3; i++)
774 {
775 if (i_premier_element[i] < 0)
776 nblocs[i] = 2;
777 }
778
779 int num_bord;
780 for (num_bord = 0; num_bord < bords.size(); num_bord++)
781 {
782 Bord& bord = bords[num_bord];
783 bord.associer_zone(zone);
784 bord.faces().typer(zone.type_elem().valeur().type_face());
785// important pour dimensionner le linesize du tableau des faces pour les frontieres vides
786 bord.faces().dimensionner(0);
787 }
788
789 ArrOfInt ibloc(3);
790 for (ibloc[0] = 0; ibloc[0] < nblocs[0]; ibloc[0]++)
791 {
792 for (ibloc[1] = 0; ibloc[1] < nblocs[1]; ibloc[1]++)
793 {
794 for (ibloc[2] = 0; ibloc[2] < nblocs[2]; ibloc[2]++)
795 {
796
797 data.xmin_.resize_array(3);
798 data.xmax_.resize_array(3);
799 for (int dir = 0; dir < 3; dir++)
800 {
801 data.xmin_[dir] = i_premier_element[dir];
802 data.xmax_[dir] = i_premier_element[dir] + nb_elements[dir] + 1;
803
804 if (nblocs[dir] > 1)
805 {
806 if (ibloc[dir] == 0)
807 data.xmin_[dir] = 0;
808 else
809 {
810 data.xmin_[dir] += data.xmax_tot_[dir] - 1;
811 data.xmax_[dir] = data.xmax_tot_[dir];
812 }
813 }
814 }
815 Process::Journal() << "add bloc xmin=" << data.xmin_ << "xmax=" << data.xmax_ << "xtot=" << data.xmax_tot_;
816 data.add_bloc(domaine, coord_ijk);
817 }
818 }
819 }
820
821 {
822 for (int d = 0; d < 3; d++)
823 {
824 if (perio[d])
825 {
826 ArrOfDouble dir(3);
827 dir[d] = coord_ijk[d][nb_noeuds[d]-1] - coord_ijk[d][0];
828 IntTab& faces = domaine.zone(0).bord(nom_bords_min[d]).faces().les_sommets();
829 double epsilon = precision_geom;
830 Reordonner_faces_periodiques::reordonner_faces_periodiques(domaine, faces, dir, epsilon);
831 }
832 }
833 }
834
835
836 auto_build_joints(domaine.zone(0), epaisseur_joint);
837
838
839 statistiques().end_count(stats);
840 double maxtime = mp_max(statistiques().last_time(stats));
841 if (Process::je_suis_maitre())
842 Cerr << "Ending of the construction of the zones, time:" << maxtime << finl;
843
844 if (nproc() > 1)
845 {
846
847 statistiques().begin_count(stats);
848
849 Scatter::construire_correspondance_sommets_par_coordonnees(domaine);
850 Scatter::calculer_renum_items_communs(domaine.zone(0).faces_joint(), Joint::SOMMET);
851
852 statistiques().end_count(stats);
853 maxtime = mp_max(statistiques().last_time(stats));
854 Cerr << "Scatter::construire_correspondance_sommets_par_coordonnees fin, time:"
855 << maxtime << finl;
856
857 statistiques().begin_count(stats);
858
859 Scatter::construire_structures_paralleles(domaine, liste_bords_perio);
860
861 statistiques().end_count(stats);
862 maxtime = mp_max(statistiques().last_time(stats));
863 Cerr << "Scatter::construire_structures_paralleles, time:" << maxtime << finl;
864 }
865 else
866 {
867 Scatter::init_sequential_domain(domaine);
868 }
869
870 domaine.zone(0).nommer(domaine.le_nom());
871
872 Cerr << "MaillerParallel::construire_domaine : end" << finl;
873
874 return is;
875}
