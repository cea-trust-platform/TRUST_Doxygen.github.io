33Entree& Viscosite_turbulente_multiple::readOn(Entree& is)
34{
35 Motcle mot;
36 is >> mot;
37 if (mot != "{") Cerr << "Multiple turbulent viscosities : { expected instead of " << mot << finl, Process::exit();
38
39 for (is >> mot; mot != "}"; is >> mot)
40 if (viscs_turbs.count(mot.getString())) Process::exit(que_suis_je() + " : a correlation already exists for " + mot + " !");
41 else viscs_turbs[mot.getString()].typer_lire(pb_.valeur(), "viscosite_turbulente", is);
42 return is;
43}

45void Viscosite_turbulente_multiple::eddy_viscosity(DoubleTab& nu_t) const
46{
47// On fait appel aux correlations d'en_dessous
48 nu_t = 0;
49 DoubleTrav nu_loc = DoubleTrav(nu_t);
50 for (auto &&corr : viscs_turbs)
51 {
52 nu_loc = 0;
53 ref_cast(Viscosite_turbulente_base, corr.second.valeur()).eddy_viscosity(nu_loc);
54 for (int i = 0; i < nu_t.dimension_tot(0); i++)
55 for (int n = 0; n < nu_t.dimension(1); n++)
56 nu_t(i, n) += nu_loc(i,n);
57 }
58}
