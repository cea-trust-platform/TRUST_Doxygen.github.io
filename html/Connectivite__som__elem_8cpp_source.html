34//! construction de la structure som_elem pour la zone donnee<br>On cree pour chaque sommet i la liste des elements adjacents a ce sommet<br>(c'est la liste des elements k tels que il existe j tel que les_elems(k,j) == i)<br>Parametre: nb_sommets<br>nombre de sommets utilises dans les elements (som_elem contiendra<br>autant de listes). Si include_virtual==1, c'est le nombre de sommets<br>total, sinon c'est le nombre de sommets reels<br>Parametre: les_elems<br>tableau des elements (contient les numeros des sommets de chaque element)<br>Les valeurs du tableau doivent etre inferieurs a nb_sommets.<br>Parametre: som_elem<br>la structure dans laquelle on stocke le resultat. L'ancien<br>contenu est perdu. Chaque liste d'elements est triee dans l'ordre croissant<br>Parametre: include_virtual<br>0 => seuls les elements reels sont inclus dans la structure<br>1 => on inclut les elements virtuels (donc les sommets virtuels)
35void construire_connectivite_som_elem(const int nb_sommets,
36 const IntTab& les_elems,
37 Static_Int_Lists& som_elem,
38 const int include_virtual)
39{
40// Nombre d'elements du domaine
41 const int nb_elem = (include_virtual) ? les_elems.dimension_tot(0) : les_elems.dimension(0);
42// Nombre de sommets par element
43 const int nb_sommets_par_element = les_elems.dimension(1);
44
45// Construction d'un tableau initialise a zero : pour chaque sommet,
46// nombre d'elements voisins de ce sommet
47 ArrOfInt nb_elements_voisins(nb_sommets);
48
49// Premier passage : on calcule le nombre d'elements voisins de chaque
50// sommet pour creer la structure de donnees
51 int elem, i;
52
53 for (elem = 0; elem < nb_elem; elem++)
54 {
55 for (i = 0; i < nb_sommets_par_element; i++)
56 {
57 int sommet = les_elems(elem, i);
58// GF cas des polyedres
59 if (sommet==-1) break;
60 nb_elements_voisins[sommet]++;
61 }
62 }
63
64 som_elem.set_list_sizes(nb_elements_voisins);
65
66// On reutilise le tableau pour stocker le nombre d'elements dans
67// chaque liste pendant qu'on la remplit
68 nb_elements_voisins = 0;
69
70// Remplissage du tableau des elements voisins.
71 for (elem = 0; elem < nb_elem; elem++)
72 {
73 for (i = 0; i < nb_sommets_par_element; i++)
74 {
75 int sommet = les_elems(elem, i);
76// GF cas des polyedres
77 if (sommet==-1) break;
78 int n = (nb_elements_voisins[sommet])++;
79 som_elem.set_value(sommet, n, elem);
80 }
81 }
82
83// Tri de toutes les listes dans l'ordre croissant
84 som_elem.trier_liste(-1);
85}

100//! Cherche les elements qui contiennent tous les sommets<br>du tableau sommets_to_find (permet de trouver les elements<br>adjacents a une face ou une arete)<br>Parametre: som_elem<br>Signification: pour chaque sommet, liste triee des elements adjacents<br>(voir construire_connectivite_som_elem)<br>Parametre: sommets_to_find<br>Signification: une liste de sommets<br>Parametre: elements<br>Signification: resultat de la recherche: la liste des elements qui<br>contiennent tous les sommets de sommets_to_find.<br>Si sommets_to_find est vide, on renvoie un tableau vide.<br>(en cas d'appels repetes a cette fonction, il est<br>conseille de mettre le drapeau "smart_resize")
101void find_adjacent_elements(const Static_Int_Lists& som_elem,
102 const ArrOfInt& sommets_to_find,
103 ArrOfInt& elements)
104{
105 int nb_som_to_find = sommets_to_find.size_array();
106// on retire les sommets valant -1 (cas ou plusieurs types de faces)
107 while (sommets_to_find[nb_som_to_find-1]==-1) nb_som_to_find--;
108 if (nb_som_to_find == 0)
109 {
110 elements.resize_array(0);
111 return;
112 }
113// Algorithme: on initialise elements avec tous les elements adjacents
114// au premier sommet de la liste.
115// Puis pour chacun des autres sommets de la liste, on retire du tableau
116// "elements" les elements qui ne sont pas voisins du sommet.
117// A la fin, il ne reste que les elements qui sont dans toutes les listes.
118 {
119// Initialisation avec les elements adjacents au premier sommet
120 const int sommet = sommets_to_find[0];
121 som_elem.copy_list_to_array(sommet, elements);
122 }
123 int nb_elem_found = elements.size_array();
124 int i_sommet;
125 for (i_sommet = 1; i_sommet < nb_som_to_find; i_sommet++)
126 {
127 const int sommet = sommets_to_find[i_sommet];
128// Calcul des elements communs entre elements[.] et som_elem(sommet,.)
129// Nombre d'elements communs entre elements et la nouvelle liste de sommets
130 int nb_elems_restants = 0;
131// Nombre d'elements adjacents au "sommet"
132 const int nb_elem_liste = som_elem.get_list_size(sommet);
133// On suppose que les listes d'elements sont triees dans l'ordre croissant
134// On parcourt simultanement les deux listes et on conserve les elements
135// communs.
136 int i = 0;
137 int j = 0;
138 if (nb_elem_found == 0)
139 break;
140 if (nb_elem_liste > 0)
141 {
142 while (1)
143 {
144 const int elem_i = elements[i];
145 const int elem_j = som_elem(sommet, j);
146 if (elem_i == elem_j)
147 {
148// Element commun aux deux listes, on le garde
149 elements[nb_elems_restants] = elem_i;
150 nb_elems_restants++;
151 }
152 if (elem_i >= elem_j)
153 {
154 j++;
155 if (j >= nb_elem_liste)
156 break;
157 }
158 if (elem_j >= elem_i)
159 {
160 i++;
161 if (i >= nb_elem_found)
162 break;
163 }
164 }
165 }
166 else
167 {
168 nb_elems_restants = 0;
169 }
170 nb_elem_found = nb_elems_restants;
171 }
172 elements.resize_array(nb_elem_found);
173}
