189// Description :
190// Calcule le taux d'accroissement du champ entre t1 et t2

192void Champ_front_var_instationnaire::Gpoint(double t1, double t2)
193{
194 Gpoint_=valeurs_au_temps(t1);
195 if (t2>t1+DMINFLOAT)
196 {
197 Gpoint_ *= -1;
198 Gpoint_ += valeurs_au_temps(t2);
199 Gpoint_ /= (t2-t1);
200 }
201 else
202 Gpoint_=0;

204}
