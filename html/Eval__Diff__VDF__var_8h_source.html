27//! class Eval_Diff_VDF_var<br>Cette classe represente un evaluateur de flux diffusif<br>avec un coefficient de diffusivite qui n'est pas constant en espace.
28/** \sa Evaluateur_VDF */
29class Eval_Diff_VDF_var : public Eval_Diff_VDF
30{
31public:
32 inline void associer(const Champ_base&) override;
33 inline void mettre_a_jour() override;
34 inline const Champ_base& get_diffusivite() const override;
35 inline const Champ_base& diffusivite() const { return get_diffusivite(); }
36
37// Methods used by the flux computation in template class:
38 inline double nu_1_impl(int i, int compo) const { return dv_diffusivite(i); }
39 inline double nu_2_impl(int i, int compo) const { return dv_diffusivite(i); }
40 inline double nu_t_impl(int i, int compo) const { return 0.; }
41 inline double nu_lam_impl_face(int i, int j, int k, int l, int compo) const { return nu_2_impl_face(i,j,k,l,compo); }
42 inline double nu_lam_impl_face2(int i, int j, int compo) const { return nu_1_impl_face(i,j,compo); }
43 inline double tau_tan_impl(int i, int j) const { return 0.; }
44 inline bool uses_wall() const { return false; }
45 inline bool uses_mod() const { return false; }
46 inline const DoubleTab& get_k_elem() const { throw; } //!< pour F5 seulement ...
47
48 inline double compute_heq_impl(double d0, int i, double d1, int j, int compo) const
49 {
50 return 1./(d0/dv_diffusivite(i) + d1/dv_diffusivite(j));
51 }
52
53 inline double nu_1_impl_face(int i, int j, int compo) const { return 0.5*(dv_diffusivite(i)+dv_diffusivite(j)); }
54 inline double nu_2_impl_face(int i, int j, int k, int l, int compo) const
55 {
56 return 0.25*(dv_diffusivite(i)+dv_diffusivite(j)+dv_diffusivite(k)+dv_diffusivite(l));
57 }
58
59protected:
60 REF(Champ_base) diffusivite_;
61 DoubleVect dv_diffusivite;
62};
