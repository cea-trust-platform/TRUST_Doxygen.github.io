43Entree& Perte_Charge_Circulaire_VEF_P1NC::readOn(Entree& s )
44{
45 Cerr << "Perte_Charge_Circulaire_VEF_P1NC::readOn " << finl;
46 sous_zone=false;
47 int lambda_ortho_ok=0;
48 int lambda_ok=0;
49// Definition des mots-cles
50 Motcles les_mots(7);
51 les_mots[0] = "lambda";
52 les_mots[1] = "diam_hydr";
53 les_mots[2] = "sous_zone";
54 les_mots[3] = "direction";
55 les_mots[4] = "lambda_ortho";
56 les_mots[5] = "diam_hydr_ortho";
57 les_mots[6] = "implicite";
58
59// Lecture et interpretation
60 Motcle motlu, accolade_fermee="}", accolade_ouverte="{";
61 s >> motlu;
62 while (motlu != accolade_ouverte)
63 {
64 Cerr << "On attendait une { a la lecture d'un " << que_suis_je() << finl;
65 Cerr << "et non : " << motlu << finl;
66 exit();
67 }
68 s >> motlu;
69 while (motlu != accolade_fermee)
70 {
71 int rang=les_mots.search(motlu);
72 switch(rang)
73 {
74 case 0 : //!< lambda
75 {
76 lambda_ok=1;
77 Nom tmp;
78 s >> tmp;
79 Cerr << "Lecture et interpretation de la fonction " << tmp << " ... ";
80 lambda.setNbVar(3+dimension);
81 lambda.setString(tmp);
82 lambda.addVar("Re_tot");
83 lambda.addVar("Re_long");
84 lambda.addVar("t");
85 lambda.addVar("x");
86 if (dimension>1)
87 lambda.addVar("y");
88 if (dimension>2)
89 lambda.addVar("z");
90 lambda.parseString();
91 Cerr << " Ok" << finl;
92 break;
93 }
94 case 1: //!< diam_hydr
95 s >> diam_hydr;
96 break;
97 case 5: //!< diam_hydr_ortho
98 s >> diam_hydr_ortho;
99 break;
100 case 2: //!< sous_zone
101 s >> nom_sous_zone;
102 sous_zone=true;
103 break;
104 case 3: //!< direction
105 s >> v;
106 break;
107 case 4: //!< lambda_ortho
108 {
109 lambda_ortho_ok=1;
110 Nom tmp;
111 s >> tmp;
112 Cerr << "Lecture et interpretation de la fonction " << tmp << " ... ";
113 lambda_ortho.setNbVar(3+dimension);
114 lambda_ortho.setString(tmp);
115 lambda_ortho.addVar("Re_tot");
116 lambda_ortho.addVar("Re_ortho");
117 lambda_ortho.addVar("t");
118 lambda_ortho.addVar("x");
119 if (dimension>1)
120 lambda_ortho.addVar("y");
121 if (dimension>2)
122 lambda_ortho.addVar("z");
123 lambda_ortho.parseString();
124 Cerr << " Ok" << finl;
125 break;
126 }
127 case 6:
128 {
129 s>>implicite_;
130 break;
131 }
132 default : //!< non compris
133 Cerr << "Mot cle \"" << motlu << "\" non compris lors de la lecture d'un "
134 << que_suis_je() << finl;
135 exit();
136 }
137 s >> motlu;
138 }
139
140// Verification de la coherence
141 if (lambda_ok==0)
142 {
143 Cerr << "Il faut definir l'expression lamba(Re_long,Re_tot)" << finl;
144 exit();
145 }
146
147 if (lambda_ortho_ok==0)
148 {
149 Cerr << "Il faut definir l'expression lamba_ortho(Re_ortho,Re_tot)" << finl;
150 exit();
151 }
152
153 if ((!diam_hydr.non_nul())|| (diam_hydr->nb_comp()!=1))
154 {
155 Cerr << "Il faut definir le champ diam_hydr a une composante" << finl;
156 exit();
157 }
158 if ((!diam_hydr_ortho.non_nul())|| (diam_hydr_ortho->nb_comp()!=1))
159 {
160 Cerr << "Il faut definir le champ diam_hydr_ortho a une composante" << finl;
161 exit();
162 }
163 if (v->nb_comp()!=dimension)
164 {
165 Cerr << "Il faut definir le champ direction a " << dimension << " composantes" << finl;
166 exit();
167 }
168
169
170 Cerr << "Fin de Perte_Charge_Circulaire_VEF_P1NC::readOn" << finl;
171 return s;
172}

180void Perte_Charge_Circulaire_VEF_P1NC::coeffs_perte_charge(const DoubleVect& u, const DoubleVect& pos,
181 double t, double norme_u, double dh, double nu, double reynolds,double& coeff_ortho,double& coeff_long,double& u_l,DoubleVect& av_valeur) const
182{
183
184// calcul de dh_ortho
185 double dh_ortho=diam_hydr_ortho->valeur_a_compo(pos,0);
186
187// calcul de u.d/||d||
188// Calcul de v et ||v||^2
189 av_valeur.resize(dimension);
190
191 v.valeur().valeur_a(pos,av_valeur);
192// on norme v
193 {
194 double vcarre=0;
195 for (int dim=0; dim<dimension; dim++)
196 vcarre+=av_valeur[dim]*av_valeur[dim];
197 av_valeur/=sqrt(vcarre);
198 }
199// Calcul de u.v/||v||
200 u_l=0;
201
202 for (int dim=0; dim<dimension; dim++)
203 u_l+=u[dim]*av_valeur[dim];
204
205 double u_ortho=sqrt(std::max(norme_u*norme_u-u_l*u_l,0.0));
206 /* PL: To avoid a possible division by zero, we replace:
207 double nu=norme_u*dh/reynolds;
208// calcule de Re_l et Re_ortho
209// Calcul du reynolds
210 double Re_l=std::fabs(u_l)*dh/nu; */
211// By:
212 double Re_l=dh*std::fabs(u_l)/nu;
213 if(Re_l<1e-10) Re_l=1e-10;
214// PL: To avoid a possible division by zero, we replace:
215 /* double Re_ortho=u_ortho*dh_ortho/nu; */
216// By:
217 double Re_ortho=dh_ortho*u_ortho/nu;
218 if (Re_ortho<1e-10) Re_ortho=1e-10;
219// Calcul de lambda
220 lambda.setVar(0,reynolds);
221 lambda.setVar(1,Re_l);
222 lambda.setVar(2,t);
223 lambda.setVar(3,pos[0]);
224 if (dimension>1)
225 lambda.setVar(4,pos[1]);
226 if (dimension>2)
227 lambda.setVar(5,pos[2]);
228
229// Calcul de lambda_ortho
230 lambda_ortho.setVar(0,reynolds);
231 lambda_ortho.setVar(1,Re_ortho);
232 lambda_ortho.setVar(2,t);
233 lambda_ortho.setVar(3,pos[0]);
234 if (dimension>1)
235 lambda_ortho.setVar(4,pos[1]);
236 if (dimension>2)
237 lambda_ortho.setVar(5,pos[2]);
238 double l_ortho=lambda_ortho.eval(); //!< Pour ne pas evaluer 2 fois le parser
239 double l_long=lambda.eval();
240 coeff_ortho=l_ortho*u_ortho/2./dh_ortho;
241 coeff_long=l_long *std::fabs(u_l) /2./dh ;
242}
