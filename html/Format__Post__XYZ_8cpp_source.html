204Nom Format_Post_XYZ::construire_nom_xyz(double temps_,
205 const Nom& id_champ,
206 const Nom& id_du_domaine,
207 const Nom& localisation)
208{
209// recup de lata pour construire le nom du fichier
210// Un fichier xyz sera cree par champs et par pas de temps de post-traitement
211 Nom id_du_champ_modifie(id_champ);
212 Nom iddomaine(id_du_domaine);
213// On utilise prefix avec un argument en majuscule
214 if ((Motcle)localisation=="SOM")
215 {
216 id_du_champ_modifie.prefix(id_du_domaine);
217 id_du_champ_modifie.prefix(iddomaine);
218//
219 id_du_champ_modifie.prefix("_SOM_");
220 }
221 else if ((Motcle)localisation=="ELEM")
222 {
223 id_du_champ_modifie.prefix(id_du_domaine);
224 id_du_champ_modifie.prefix(iddomaine);
225//
226 id_du_champ_modifie.prefix("_ELEM_");
227 }
228 else if ((Motcle)localisation=="FACES")
229 {
230 id_du_champ_modifie.prefix(id_du_domaine);
231 id_du_champ_modifie.prefix(iddomaine);
232 id_du_champ_modifie.prefix("_FACES_");
233 }
234// Construction du nom du fichier
235 Nom extension_champ(id_du_champ_modifie);
236 extension_champ += ".";
237 extension_champ += localisation;
238 extension_champ += ".";
239 extension_champ += id_du_domaine;
240 extension_champ += ".";
241 char str_temps[100] = "0.0";
242 if (temps_ >= 0.)
243 sprintf(str_temps, "%.10f", temps_);
244 extension_champ += str_temps;
245 Nom nom_fich(xyz_basename_);
246 nom_fich += "." ;
247 nom_fich += extension_champ;
248 nom_fich +=".";
249 Nom format="xyz";
250 nom_fich += format;
251
252 return nom_fich;
253}

285int Format_Post_XYZ::ecrire_champ_xyz(const Domaine& domaine,const Noms& unite_,const Noms& noms_compo,
286 int ncomp,double temps_,
287 const Nom& id_du_champ,
288 const Nom& id_du_domaine,
289 const Nom& localisation,
290 const DoubleTab& data,Nom& nom_fich,
291 const DoubleTab& coord)
292{
293//
294 EcrFicPartage s;
295// J'ecris tout en une seule fois donc pas la peine d'ouvrir en mode append
296 s.ouvrir(nom_fich);
297 int dim = coord.dimension(1);
298 int nb_compo_=1;
299 if (data.nb_dim()>1)
300 nb_compo_=data.dimension(1);
301// On ecrit uniquement le n nombre de lignes contenues dans le fichier xyz
302// il correspond au nombre de coordonees
303 int n = data.dimension(0);
304 if(je_suis_maitre())
305 s << n << finl;
306 for (int j=0; j<n; j++)
307 {
308// Les coord du point j
309 for (int ncomp2=0; ncomp2<dim; ncomp2++)
310 s << coord(j, ncomp2) << " " ;
311// Les val du champs au point j
312 if (nb_compo_==1)
313 s << data(j) << " ";
314 else
315 for (int ncomp2=0; ncomp2<nb_compo_; ncomp2++)
316 s << data(j,ncomp2) << " ";
317 s << finl;
318 }
319 s.unlockfile();
320 s.syncfile();
321//
322
323 s.flush();
324 s.syncfile();
325
326 return 1;
327}
