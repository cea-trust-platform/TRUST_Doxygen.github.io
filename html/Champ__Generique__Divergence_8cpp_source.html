36void Champ_Generique_Divergence::completer(const Postraitement_base& post)
37{
38 Champ_Gen_de_Champs_Gen::completer(post);
39 const Probleme_base& Pb = get_ref_pb_base();
40// On ne complete l operateur que s il sagit du champ de vitesse
41
42 if (sub_type(Champ_Generique_refChamp,get_source(0)))
43 {
44 Champ espace_stockage;
45 if (get_source(0).get_champ(espace_stockage).le_nom()=="vitesse")
46 {
47 const Equation_base& eqn = Pb.equation(0);
48 Op_Div_.associer_eqn(eqn);
49 Op_Div_.typer();
50 Op_Div_.l_op_base().associer_eqn(eqn);
51
52// La methode completer() lance l association de la zone_discretisee
53// qui est recuperre par l equation associee a l operateur
54 Op_Div_->completer();
55 }
56 }
57}

76const Champ_base& Champ_Generique_Divergence::get_champ(Champ& espace_stockage) const
77{
78
79 Champ source_espace_stockage;
80 const Champ_base& source = get_source(0).get_champ(source_espace_stockage);
81
82 if (Op_Div_.non_nul())
83 {
84 Champ_Fonc es_tmp;
85 espace_stockage = creer_espace_stockage(scalaire,1,es_tmp);
86 Op_Div_.calculer(source.valeurs(),espace_stockage.valeurs());
87 }
88 else
89 {
90 const Noms nom = get_source(0).get_property("nom");
91 Cerr<<"We can apply a Champ_Generique_Divergence only to the velocity field"<<finl;
92 Cerr<<"The field treated here "<<nom[0]<<" concerns "<<source.le_nom()<<finl;
93 exit();
94 }
95
96 DoubleTab& espace_valeurs = espace_stockage->valeurs();
97 espace_valeurs.echange_espace_virtuel();
98 return espace_stockage.valeur();
99}

102Entity Champ_Generique_Divergence::get_localisation(const int index) const
103{
104 Entity loc;
105 Nom type_op = Op_Div_.valeur().que_suis_je();
106 if (((type_op=="Op_Div_VEF_P1NC") || (type_op=="Op_Div_VDF_Face")) && (index <= 0))

108 else

110 Cerr << "Error of type : Divergence operator " << type_op << " is not possible for the defined field " << nom_post_ << finl;

112 }

115const Noms Champ_Generique_Divergence::get_property(const Motcle& query) const
116{
117
118 Motcles motcles(2);
119 motcles[0] = "composantes";
120 motcles[1] = "unites";
121
122 int rang = motcles.search(query);
123 switch(rang)
124 {
125 case 0:
126 {
127 Noms compo(1);
128 Nom nume((int)0);
129 compo[0] = nom_post_+nume;
130
131 return compo;
132 break;
133 }
134
135 case 1:
136 {
137
138 Noms unites(1);
139 Noms source_unites = get_source(0).get_property("unites");
140 unites[0] = source_unites[0]+"/m";
141
142 return unites;
143 break;
144 }
145 }
146 return Champ_Gen_de_Champs_Gen::get_property(query);
147}
