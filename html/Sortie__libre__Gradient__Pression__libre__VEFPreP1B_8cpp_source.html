58void Sortie_libre_Gradient_Pression_libre_VEFPreP1B::completer()
59{
60
61 Cerr << "Sortie_libre_Gradient_Pression_libre_VEFPreP1B::completer()" << finl;
62 const Zone_Cl_dis_base& la_zone_Cl = zone_Cl_dis();
63 const Equation_base& eqn = la_zone_Cl.equation();
64 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,eqn);
65
66 const Zone_VEF& ma_zone_VEF = ref_cast(Zone_VEF,eqn.zone_dis().valeur());
67
68 const Champ_P1_isoP1Bulle& pression = ref_cast(Champ_P1_isoP1Bulle,eqn_hydr.pression().valeur());
69
70 const IntTab& face_voisins = ma_zone_VEF.face_voisins();
71
72 pression_interne = pression;
73
74 const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
75 int nb_faces_loc = le_bord.nb_faces();
76
77 int ndeb = le_bord.num_premiere_face();
78 int nfin = ndeb + nb_faces_loc;
79 int face;
80 coeff.resize(nb_faces_loc);
81
82 if (dimension == 2)
83 for (face=ndeb; face<nfin ; face++)
84 {
85 int elem=face_voisins(face,0);
86 coeff[face-ndeb] = distance_2D(face,elem,ma_zone_VEF) * 3.;
87 }
88 if (dimension == 3)
89 for (face=ndeb; face<nfin ; face++)
90 {
91 int elem=face_voisins(face,0);
92 coeff[face-ndeb] = distance_3D(face,elem,ma_zone_VEF) * 4. ;
93 }
94
95 pression_temps_moins_un.resize(nb_faces_loc);
96 pression_temps_moins_deux.resize(nb_faces_loc);
97 pression_moins_un_temps_moins_un.resize(nb_faces_loc);
98 pression_moins_un_temps_moins_deux.resize(nb_faces_loc);
99 pression_moins_deux_temps_moins_un.resize(nb_faces_loc);
100 pression_moins_un.resize(nb_faces_loc,1);
101 pression_moins_deux.resize(nb_faces_loc,1);
102
103 Pimp.resize(nb_faces_loc);
104
105 Cerr << "Sortie_libre_Gradient_Pression_libre_VEFPreP1B::completer() ok" << finl;
106}

114

139void Sortie_libre_Gradient_Pression_libre_VEFPreP1B::mettre_a_jour(double temps)
140{
141
142 const Zone_Cl_dis_base& la_zone_Cl = zone_Cl_dis();
143 const Equation_base& eqn = la_zone_Cl.equation();
144 const Zone_VEF& ma_zone_VEF = ref_cast(Zone_VEF,eqn.zone_dis().valeur());
145 const DoubleTab& face_normale = ma_zone_VEF.face_normales();
146 const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
147
148 int nb_faces_loc = le_bord.nb_faces();
149 int ndeb = le_bord.num_premiere_face();
150 int nfin = ndeb + nb_faces_loc;
151
152 int face, face_loc, j ;
153
154 const DoubleTab& xv = ma_zone_VEF.xv();
155
156 DoubleTab geom ;
157 geom.resize(nb_faces_loc,dimension);
158
159 Cond_lim_base::mettre_a_jour(temps);
160
161 for (face=ndeb; face < nfin ; face++)
162 {
163 face_loc = face - ndeb ;
164
165 pression_temps_moins_deux(face_loc) = pression_temps_moins_un(face_loc);
166 pression_temps_moins_un(face_loc) = Pimp(face_loc);
167 pression_moins_un_temps_moins_deux(face_loc)= pression_moins_un_temps_moins_un(face_loc);
168 pression_moins_un_temps_moins_un(face_loc) = pression_moins_un(face_loc,0);
169 pression_moins_deux_temps_moins_un(face_loc)= pression_moins_deux(face_loc,0);
170 }
171
172 double orient ;
173 for (face=ndeb; face<nfin ; face++)
174 {
175 orient = 0. ;
176 for ( j= 0 ; j < dimension ; j++ )
177 {
178 orient += (face_normale(face,j) * face_normale(face,j));
179 }
180
181 orient = sqrt(orient) ;
182
183 for ( j= 0 ; j < dimension ; j++ )
184 {
185 geom(face-ndeb,j) = xv(face,j) ;
186 geom(face-ndeb,j)-= face_normale(face,j)/orient * coeff(face-ndeb) ; ;
187 }
188
189 }
190
191 pression_interne->valeur_aux(geom, pression_moins_un);
192
193
194 for (face=ndeb; face<nfin ; face++)
195 {
196 orient = 0. ;
197 for ( j= 0 ; j < dimension ; j++ )
198 {
199 orient += (face_normale(face,j) * face_normale(face,j));
200 }
201
202 orient = sqrt(orient) ;
203
204 for ( j= 0 ; j < dimension ; j++ )
205 {
206 geom(face-ndeb,j) = xv(face,j) ;
207 geom(face-ndeb,j) -= face_normale(face,j)/orient * 2. * coeff(face-ndeb) ;
208 }
209
210 }
211
212 pression_interne->valeur_aux(geom, pression_moins_deux);
213
214 double VPhiP ;
215
216 for (face=ndeb; face<nfin ; face++)
217 {
218 face_loc = face - ndeb ;
219 double pre_m_un_t_m_deux = pression_moins_un_temps_moins_deux(face_loc);
220 double pre_m_deux_t_m_un = pression_moins_deux_temps_moins_un(face_loc);
221 double pre_m_un = pression_moins_un(face_loc,0);
222
223 if( pre_m_un_t_m_deux == pre_m_un )
224 VPhiP = 0. ;
225 else
226 VPhiP = (pre_m_un_t_m_deux-pre_m_un)/
227 (pre_m_un+pre_m_un_t_m_deux-2*pre_m_deux_t_m_un);
228
229 if(VPhiP<=1.e-24) VPhiP = 0.0 ;
230 if(VPhiP>1.) VPhiP = 1.0 ;
231 assert(VPhiP<1.e12);
232
233 Pimp[face_loc] = (1-VPhiP)/(1+VPhiP) * pression_temps_moins_un(face_loc)
234 + (2*VPhiP)/(1+VPhiP) * pression_moins_un(face_loc,0);
235
236 }

255double Sortie_libre_Gradient_Pression_libre_VEFPreP1B::Grad_P_lib_VEFPreP1B(int face) const
256{
257 const Milieu_base& mil=ma_zone_cl_dis->equation().milieu();
258 const Champ_Uniforme& rho=ref_cast(Champ_Uniforme,mil.masse_volumique().valeur());
259 double d_rho = rho(0,0);
260 if (le_champ_front.valeurs().size()==1)
261 return le_champ_front(0,0)/d_rho;
262 else if (le_champ_front.valeurs().dimension(1)==1)
