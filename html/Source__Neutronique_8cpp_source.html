39{
40 Motcles les_mots(11);
41 {
42 les_mots[0] = "N";

57 Ci0_ok=0;

62 dt_impr = 1e10;
63 faire_un_pas_de_temps = &Source_Neutronique::faire_un_pas_de_temps_RK;
64 Motcle motlu, accolade_fermee="}", accolade_ouverte="{";
65 is >> motlu;

71 }
72 is >> motlu;
73 while (motlu != accolade_fermee)
74 {
75
76 int rang=les_mots.search(motlu);
77 switch(rang)
78 {
79 case 0 :
80 {
81 is >> N;
82 if (N<1)
83 {
84 Cerr << "Le nombre de groupes doit etre >=1 dans " << que_suis_je() << finl;
85 exit();
86 }
87 break;
88 }
89 case 1 :
90 {
91 if (N<1)
92 {
93 Cerr << "Il faut rentrer le nb de groupes N avant lambda dans " << que_suis_je() << finl;
94 exit();
95 }
96 lambda.resize(N);
97 for (int i=0; i<N; i++) is>>lambda(i);
98 break;
99 }
100 case 2 :
101 {
102 is >> Tvie;
103 break;
104 }
105 case 3 :
106 {
107 if (N<1)
108 {
109 Cerr << "Il faut rentrer le nb de groupes N avant beta dans " << que_suis_je() << finl;
110 exit();
111
112 }
113 beta.resize(N);
114 beta_som=0.;
115 for (int i=0; i<N; i++)
116 {
117 is>>beta(i);
118 beta_som+=beta(i);
119 }
120 break;
121 }
122 case 5 :
123 {
124 is >> P0;
125 break;
126 }
127 case 6 :
128 {
129 is >> dt_impr;
130 break;
131 }
132 case 7 :
133 {
134 Motcle nom_sch;

136 if (nom_sch == "euler_explicite")
137 {
138 faire_un_pas_de_temps = &Source_Neutronique::faire_un_pas_de_temps_EE;

146 Cerr << "On ne comprend le mot : " << motlu << "dans " << que_suis_je() << finl;
147 exit();
148 }
149

151 }

154 Nom tmp;

164 }

166 {
167 is >> f_xyz;

170 }

172 {

194

196 {
197 Cerr << "Il faut rentrer le nb de groupes N dans " << que_suis_je() << finl;
198 exit();
199
200 }
201

230void Source_Neutronique::completer()
231{
232
233 Source_base::completer();
234
235
236 Un.resize(N+1);
237 Unp1.resize(N+1);
238 matA.resize(N+1,N+1);
239 matA = 0.;
240 for (int i=1; i<N+1; i++)
241 {
242 matA(0,i) = lambda(i-1);
243 matA(i,i) = -lambda(i-1);
244 matA(i,0) = beta(i-1)/Tvie;
245 }
246 Un(0) = P0;
247 if (Ci0_ok == 0)
248 for (int i=1; i<N+1; i++)
249 Un(i) = beta(i-1)*Un(0)/Tvie/lambda(i-1);
250 else
251 for (int i=1; i<N+1; i++)
252 Un(i) = Ci0(i-1);
253}

256/**
257 * Effectue resu = m*v
258 */void Source_Neutronique::mul(DoubleTab& m, DoubleVect& v, DoubleVect& resu)
259{
260 resu = 0.;
261 resu(0) = m(0,0)*v(0);
262 for (int i=1; i<N+1; i++)
263 {
264 resu(0)+= m(0,i)*v(i);
265 resu(i) = m(i,0)*v(0) + m(i,i)*v(i);
266 }

270/**
271 * Schema de Runge Kutta classique
272 */
273void Source_Neutronique::faire_un_pas_de_temps_RK()
274{
275 static const double a1 = 1./6.;
276 static const double a2 = 1./3.;
277
278 DoubleVect fn1(Un);
279 DoubleVect fn2(Un);
280 DoubleVect fn3(Un);
281 DoubleVect fn4(Un);
282 const double dt2 = dt/2.;
283
284// Un1 = Un+dt/2*f(tn,un)
285 mul(matA,Un, fn1);
286 Unp1 = Un;
287 Unp1.ajoute(dt2, fn1);
288
289// Un2 = Un+dt/2*f(tn+dt/2,Un1)
290 mettre_a_jour_matA(temps_courant+dt2);
291
292 mul(matA,Unp1, fn2);
293 Unp1 = Un;
294 Unp1.ajoute(dt2, fn2);
295
296// Un3 = Un+dt*f(tn+dt/2,Un2)
297 mul(matA,Unp1, fn3);
298 Unp1 = Un;
299 Unp1.ajoute(dt, fn3);
300
301// Un+1 = Un+dt*(a1*f(tn,un) + a2*f(tn+dt/2,Un1) + a2*f(tn+dt/2,Un2) + a1*f(tn+dt/2,Un3))
302 mettre_a_jour_matA(temps_courant+dt);
303 mul(matA,Unp1, fn4);
304 Unp1 = Un;
305 Unp1.ajoute(a1*dt, fn4);
306 Unp1.ajoute(a2*dt, fn3);
307 Unp1.ajoute(a2*dt, fn2);
308 Unp1.ajoute(a1*dt, fn1);
309}

336/**
337 * Aller_au_temps : permet d'avancer avec le pas de temps de stabilite propre de l'equation.
338 */
339void Source_Neutronique::aller_au_temps(double temps)
340{
341 while (temps_courant<temps)
342 {
343 const double dt_stab = 0.8*Tvie/beta_som*std::max(std::fabs(rho(temps_courant,Tmoy)-1),1.);
344 dt = std::min(dt_stab, temps-temps_courant);
345 (this->*faire_un_pas_de_temps)();
346 Un = Unp1;
347 temps_courant += dt;
348 mettre_a_jour_matA(temps_courant);
349 if (limpr(temps_courant,dt)) imprimer(temps_courant);
350 }
351
352}

374int Source_Neutronique::limpr(double le_temps_courant,double ddt) const
375{
376 static const double epsilon = 1.e-9;
377 const Schema_Temps_base& sch = mon_equation->schema_temps();
378// Si impression a chaque pas de temps, alors que pas demande, faire comme limpr() de Echange_contact_Correlation_VDF.cpp
379 if (sch.nb_pas_dt()==0)
380 return 0;
381 if (dt_impr<=ddt || ((sch.temps_final_atteint() || sch.nb_pas_dt_max_atteint() || sch.nb_pas_dt()<=1 || sch.stationnaire_atteint()) && dt_impr!=1e10))
382 return 1;
383 else
384 {
385// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
386 double i, j;
387 modf(le_temps_courant/dt_impr + epsilon, &i);
388 modf((le_temps_courant-ddt)/dt_impr + epsilon, &j);
389 return ( i>j );
390 }
391}
