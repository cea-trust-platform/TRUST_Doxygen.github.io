27{
28 const Zone_VDF& zvdf = ref_cast(Zone_VDF,zone_dis.valeur());

41void Op_Div_VDF_Elem::volumique(DoubleTab& div) const
42{
43 const Zone_VDF& zone_VDF = la_zone_vdf.valeur();
44 const DoubleVect& vol = zone_VDF.volumes();
45 const int nb_elem = zone_VDF.zone().nb_elem_tot();
