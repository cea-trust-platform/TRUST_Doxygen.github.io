87// Compute -Cf.psi.U.|U|/sqrt(K).dvol
88inline void Eval_Forchheimer_VEF_Face::calculer_terme_source(int num_face, DoubleVect& source, const DoubleVect& volumes) const
89{
90 int size=source.size();
91 for (int i=0; i<size; i++)
92 {
93 double U = vitesse->valeurs()(num_face,i);
94 source(i) = -Cf/sqrt(modK->getK(porosite))*volumes(num_face)*porosite_surf(num_face)*std::fabs(U)*U;
95 }
96}
