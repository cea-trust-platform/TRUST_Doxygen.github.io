33class Eval_Dift_VDF_const : public Eval_Diff_VDF_const, public Eval_Turbulence

40// Overloaded methods used by the flux computation in template class:
41 inline double nu_1_impl(int i, int compo) const
42 {
43 const double nu_lam = Eval_Diff_VDF_const::nu_1_impl(i,compo);
44 const double nu_turb = dv_diffusivite_turbulente(i);
45 return nu_lam+nu_turb;
46 }
47
48 inline double nu_2_impl(int i, int compo) const { return Eval_Diff_VDF_const::nu_2_impl(i,compo); }
49 inline double nu_t_impl(int i, int compo) const { return dv_diffusivite_turbulente(i); }
50
51 inline double nu_1_impl_face(int i, int j, int compo) const
52 {
53 return 0.5*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j));
54 }
55
56 inline double nu_2_impl_face(int i, int j, int k, int l, int compo) const
57 {
58 return 0.25*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j)+
59 dv_diffusivite_turbulente(k)+dv_diffusivite_turbulente(l));
60 }
61
62 inline double nu_lam_impl_face(int i, int j, int k, int l, int compo) const
63 {
64 return Eval_Diff_VDF_const::nu_2_impl_face(i,j,k,l,compo);
65 }
66
67 inline double nu_lam_impl_face2(int i, int j, int compo) const
68 {
69 return Eval_Diff_VDF_const::nu_1_impl_face(i,j,compo);
70 }
71
72 inline double compute_heq_impl(double d0, int i, double d1, int j, int compo) const
73 {
74 const double heq_lam = Eval_Diff_VDF_const::compute_heq_impl(d0, i, d1, j, compo);
75 const double heq_turb= 0.5*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j))/(d1+d0);
76 return heq_lam + heq_turb;
77 }
