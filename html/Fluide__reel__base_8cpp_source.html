51void Fluide_reel_base::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
52{
53 Cerr << "Medium discretization." << finl;
54
55 const Zone_dis_base& zone_dis = pb.equation(0).zone_dis();
56 const double temps = pb.schema_temps().temps_courant();
57
58 /* masse volumique, energie interne, enthalpie : champ_inc */
59 Champ_Inc rho_inc, ei_inc, h_inc;
60 int nc = pb.equation(0).inconnue()->nb_valeurs_temporelles();
61 dis.discretiser_champ("champ_elem", zone_dis, "masse_volumique", "kg/m^3", 1, nc, temps, rho_inc);
62 dis.discretiser_champ("champ_elem", zone_dis, "energie_interne", "J/kg", 1, nc, temps, ei_inc);
63 dis.discretiser_champ("champ_elem", zone_dis, "enthalpie", "J/kg", 1, nc, temps, h_inc);
64 rho = rho_inc, e_int = ei_inc, h = h_inc;
65
66 dis.discretiser_champ("champ_elem", zone_dis, "viscosite_dynamique", "kg/m/s", 1, temps, mu);
67 dis.discretiser_champ("champ_elem", zone_dis, "viscosite_cinematique", "m2/s", 1, temps, nu);
68 dis.discretiser_champ("champ_elem", zone_dis, "diffusivite", "m2/s", 1, temps, alpha);
69 dis.discretiser_champ("champ_elem", zone_dis, "conductivite", "W/m/K", 1, temps, lambda);
70 dis.discretiser_champ("champ_elem", zone_dis, "capacite_calorifique", "J/kg/K", 1, temps, Cp);
71}

73int Fluide_reel_base::initialiser(const double temps)
74{
75 const Equation_base& eqn = equation("temperature");
76 Champ_Inc_base& ch_rho = ref_cast(Champ_Inc_base, rho.valeur()),
77 &ch_e = ref_cast(Champ_Inc_base, e_int.valeur()),

98void Fluide_reel_base::mettre_a_jour(double t)
99{
100 double tp = ref_cast(Champ_Inc_base, rho.valeur()).temps(); //pour savoir si on va tourner la roue
101 rho.mettre_a_jour(t);
102 e_int.mettre_a_jour(t);
103 h.mettre_a_jour(t);
104
105 Cp.mettre_a_jour(t);
106 mu.mettre_a_jour(t);
107 lambda.mettre_a_jour(t);
108 nu.mettre_a_jour(t);
109 alpha.mettre_a_jour(t);
110
111 const Champ_Inc_base& ch_T = equation("temperature").inconnue().valeur(), &ch_p = ref_cast(Navier_Stokes_std, equation("vitesse")).pression().valeur();
112 const DoubleTab& temp = ch_T.valeurs(), &pres = ch_p.valeurs();
113
114 int i, Ni = mu.valeurs().dimension_tot(0);
115 DoubleTab& tab_Cp = Cp.valeurs(), &tab_mu = mu.valeurs(), &tab_lambda = lambda.valeurs(), &tab_nu = nu.valeurs(), &tab_alpha = alpha.valeurs();
116 const DoubleTab& tab_rho = masse_volumique().valeurs();
117 if (t > tp || first_maj_)
118 for (i = 0; i < Ni; i++) /* maj uniquement en fin de pas de temps */
119 {
120 const double T = (T_ref_ > 0) ? T_ref_ : temp(i, id_composite), P = (P_ref_ > 0) ? P_ref_ : pres(i);
121 tab_Cp(i) = cp_(T, P);
122 tab_mu(i) = mu_(T, P);
123 tab_lambda(i) = lambda_(T, P);
124 tab_nu(i) = tab_mu(i) / tab_rho(i);
125 tab_alpha(i) = tab_lambda(i) / tab_rho(i) / tab_Cp(i);
126 }
127 first_maj_ = 0;
128}

130int Fluide_reel_base::check_unknown_range() const
131{
132 int ok = 1, zero = 0, nl = rho.valeurs().dimension_tot(0); //on n'impose pas de contraintes aux lignes correspondant a des variables auxiliaires (eg pressions aux faces dans PolyMAC)
133 for (auto &&i_r : unknown_range())
134 {
135 const DoubleTab& vals = i_r.first == "pression" ? ref_cast(Navier_Stokes_std, equation("vitesse")).pression().valeurs() : equation(i_r.first).inconnue().valeurs();
136 double vmin = DBL_MAX, vmax = -DBL_MAX;
137 for (int i = 0, j = std::min(std::max(id_composite, zero), vals.dimension(1) - 1); i < nl; i++)
138 vmin = std::min(vmin, vals(i, j)), vmax = std::max(vmax, vals(i, j));
139 ok &= Process::mp_min(vmin) >= i_r.second[0] && Process::mp_max(vmax) <= i_r.second[1];
140 }
141 return ok;
142}

153{
154 if (!equation_.size()) return true; //pas d'equation associee -> ???

167void Fluide_reel_base::calculer_masse_volumique(const Objet_U& obj, DoubleTab& val, DoubleTab& bval, tabs_t& deriv)
168{
169 const Fluide_reel_base& fl = ref_cast(Fluide_reel_base, obj);
170 const Champ_Inc_base& ch_T = fl.equation("temperature").inconnue().valeur(),
171 &ch_p = ref_cast(Navier_Stokes_std, fl.equation("vitesse")).pression().valeur();
172 const DoubleTab& T = ch_T.valeurs(), &p = ch_p.valeurs();
173 int i, zero = 0, Ni = val.dimension_tot(0), Nb = bval.dimension_tot(0), n = std::max(fl.id_composite, zero), m = p.line_size() == T.line_size() ? n : 0,
174 incomp = fl.T_ref_ >= 0 && fl.P_ref_ >= 0;
175 double rho0 = incomp ? fl.rho_(fl.T_ref_, fl.P_ref_) : -1;
176 for (i = 0; i < Ni; i++) val(i) = rhoc(T(i, n), p(i, m));
177
178 const DoubleTab& bT = ch_T.valeur_aux_bords(), &bp = ch_p.valeur_aux_bords();
179 for (i = 0; i < Nb; i++) bval(i) = rhoc(bT(i, n), bp(i, m));
180
181 if (incomp) return; //pas de derivees en incompressible
182 DoubleTab& dp_rho = deriv["pression"], &dT_rho = deriv["temperature"];
183 for (dp_rho.resize(Ni, 1), i = 0; i < Ni; i++) dp_rho(i) = fl.dP_rho_(T(i, n), p(i, m));
184 for (dT_rho.resize(Ni, 1), i = 0; i < Ni; i++) dT_rho(i) = fl.dT_rho_(T(i, n), p(i, m));
185}

189void Fluide_reel_base::calculer_enthalpie(const Objet_U& obj, DoubleTab& val, DoubleTab& bval, tabs_t& deriv)
190{
191 const Fluide_reel_base& fl = ref_cast(Fluide_reel_base, obj);
192 const Champ_Inc_base& ch_T = fl.equation("temperature").inconnue().valeur(),
193 &ch_p = ref_cast(Navier_Stokes_std, fl.equation("vitesse")).pression().valeur();
194 const DoubleTab& T = ch_T.valeurs(), &p = ch_p.valeurs();
195 int i, zero = 0, Ni = val.dimension_tot(0), Nb = bval.dimension_tot(0), n = std::max(fl.id_composite, zero), m = p.line_size() == T.line_size() ? n : 0,
196 incomp = fl.T_ref_ >= 0 && fl.P_ref_ >= 0;
197 double h0 = incomp ? fl.h_(fl.T_ref_, fl.P_ref_) : -1, Cp0 = incomp ? fl.dT_h_(fl.T_ref_, fl.P_ref_) : -1;
198
199 for (i = 0; i < Ni; i++) val(i) = hc(T(i, n), p(i, m));
200
201 const DoubleTab& bT = ch_T.valeur_aux_bords(), &bp = ch_p.valeur_aux_bords();
202 for (i = 0; i < Nb; i++) bval(i) = hc(bT(i, n), bp(i, m));
203
204 DoubleTab *dp_h = incomp ? NULL : &deriv["pression"], &dT_h = deriv["temperature"];//une seule derivee en incompressible
205 if (dp_h)
206 for (dp_h->resize(Ni, 1), i = 0; i < Ni; i++) (*dp_h)(i) = fl.dP_h_(T(i, n), p(i, m));
207 for (dT_h.resize(Ni, 1), i = 0; i < Ni; i++) dT_h(i) = incomp ? Cp0 : fl.dT_h_(T(i, n), p(i, m));
208}

210void Fluide_reel_base::calculer_energie_interne(const Objet_U& obj, DoubleTab& val, DoubleTab& bval, tabs_t& deriv)
211{
212 const Fluide_reel_base& fl = ref_cast(Fluide_reel_base, obj);
213 const Champ_Inc_base& ch_T = fl.equation("temperature").inconnue().valeur(),
214 &ch_p = ref_cast(Navier_Stokes_std, fl.equation("vitesse")).pression().valeur();
215 const DoubleTab& T = ch_T.valeurs(), &p = ch_p.valeurs();
216 int i, zero = 0, Ni = val.dimension_tot(0), Nb = bval.dimension_tot(0), n = std::max(fl.id_composite, zero), m = p.line_size() == T.line_size() ? n : 0,
217 incomp = fl.T_ref_ >= 0 && fl.P_ref_ >= 0;
218 if (incomp) return calculer_enthalpie(obj, val, bval, deriv); //en incompressible, on prend e_int = h
219
220 for (i = 0; i < Ni; i++) val(i) = fl.h_(T(i, n), p(i, m)) - p(i, m) / fl.rho_(T(i, n), p(i, m));
221
222 const DoubleTab& bT = ch_T.valeur_aux_bords(), &bp = ch_p.valeur_aux_bords();
223 for (i = 0; i < Nb; i++) bval(i) = fl.h_(bT(i, n), bp(i, m)) - bp(i, m) / fl.rho_(bT(i, n), bp(i, m));
224
225 DoubleTab& dp_e = deriv["pression"], &dT_e = deriv["temperature"];
226 for (dp_e.resize(Ni, 1), i = 0; i < Ni; i++) dp_e(i) = fl.dP_h_(T(i, n), p(i, m)) - 1. / fl.rho_(T(i, n), p(i, m)) + p(i, m) * fl.dP_rho_(T(i, n), p(i, m)) / std::pow(fl.rho_(T(i, n), p(i, m)), 2);
227 for (dT_e.resize(Ni, 1), i = 0; i < Ni; i++) dT_e(i) = fl.dT_h_(T(i, n), p(i, m)) + p(i, m) * fl.dT_rho_(T(i, n), p(i, m)) / std::pow(fl.rho_(T(i, n), p(i, m)), 2);
228}
