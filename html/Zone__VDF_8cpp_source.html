94//! Reordonne les faces internes par orientation, on doit mettre a jour<br>tous les tableaux qui dependent des indices de faces internes, soit:<br>- faces_sommets<br>- faces_voisins<br>- elem_faces<br>- orientation<br>- Zone.faces_joint().items_communs(FACE)
95static void reordonner_vdf(const int nb_faces_front,
96 Faces_VDF& faces, IntTab& elem_faces,
97 IntVect& orientation, Joints& joints)
98{
99// Construction d'un int selon lequel on va trier les faces:
100// orientation * nb_faces + indice_face
101// Quand on trie par ordre croissant de cet int, on trie selon l'orientation
102// en preservant l'ordre initial des faces de meme orientation
103 const int nb_faces = faces.nb_faces();
104 int i, j;
105
106 ArrOfInt sort_key(nb_faces);
107// On ne trie pas les faces de bord, qui restent au debut:
108 for (i = 0; i < nb_faces_front; i++)
109 sort_key[i] = i;
110 for (; i < nb_faces; i++)
111 {
112 const int ori = orientation(i);
113 sort_key[i] = ori * nb_faces + i;
114 }
115
116 sort_key.ordonne_array();
117
118// Il suffit de revenir a l'index initial pour avoir les indices des
119// faces triees par orientation : sort_key[nouveau numero] = ancien numero
120 for (i = nb_faces_front; i < nb_faces; i++)
121 {
122 const int key = sort_key[i];
123 orientation[i] = key / nb_faces;
124 sort_key[i] = key % nb_faces;
125 }
126
127// On reordonne les faces:
128 {
129 IntTab& faces_sommets = faces.les_sommets();
130 IntTab old_tab(faces_sommets);
131 const int nb_som_faces = faces_sommets.dimension(1);
132 for (i = 0; i < nb_faces; i++)
133 {
134 const int old_i = sort_key[i];
135 for (j = 0; j < nb_som_faces; j++)
136 faces_sommets(i, j) = old_tab(old_i, j);
137 }
138 }
139
140 {
141 IntTab& faces_voisins = faces.voisins();
142 IntTab old_tab(faces_voisins);
143 for (i = 0; i < nb_faces; i++)
144 {
145 const int old_i = sort_key[i];
146 faces_voisins(i, 0) = old_tab(old_i, 0);
147 faces_voisins(i, 1) = old_tab(old_i, 1);
148 }
149 }
150
151// Calcul de la table inversee: reverse_index[ancien_numero] = nouveau numero
152 ArrOfInt reverse_index(nb_faces);
153 {
154 for (i = 0; i < nb_faces; i++)
155 {
156 const int jj = sort_key[i];
157 reverse_index[jj] = i;
158 }
159 }
160// Renumerotation de elem_faces:
161 {
162// Nombre d'indices de faces dans le tableau
163 const int nb_items = elem_faces.size();
164 ArrOfInt& array = elem_faces;
165 for (i = 0; i < nb_items; i++)
166 {
167 const int old = array[i];
168 array[i] = reverse_index[old];
169 }
170 }
171// Mise a jour des indices des faces de joint:
172 {
173 const int nb_joints = joints.size();
174 for (int i_joint = 0; i_joint < nb_joints; i_joint++)
175 {
176 Joint& joint = joints[i_joint];
177 ArrOfInt& indices_faces = joint.set_joint_item(Joint::FACE).set_items_communs();
178 const int nb_faces_bis = indices_faces.size_array();
179 assert(nb_faces_bis == joint.nb_faces()); //!< items_communs rempli ?
180 for (i = 0; i < nb_faces_bis; i++)
181 {
182 const int old = indices_faces[i]; //!< ancien indice local
183 indices_faces[i] = reverse_index[old];
184 }
185// Les faces de joint ne sont plus consecutives dans le
186// tableau: num_premiere_face n'a plus ne sens
187 joint.fixer_num_premiere_face(-1);
188 }
189 }
190}

219//! appel a Zone_VF::discretiser()<br>calcul des centres de gravite des elements<br>remplissage des connectivites elements/faces<br>on reordonne les connectivites faces/elements<br>de sorte que el1 est a gauche et el2 a droite.<br>calcul des porosites volumiques et surfaciques<br>generation des aretes<br>calcul des pas du maillage
220void Zone_VDF::discretiser()
221{
222 Zone_VF::discretiser();
223
224 Zone& zone_geom=zone();
225
226// Verification de la coherence entre l'element geometrique et la discretisation
227
228 const Elem_geom_base& elem_geom = zone_geom.type_elem().valeur();
229
230 if (dimension == 2)
231 {
232 if (!sub_type(Rectangle,elem_geom))
233 {
234 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
235 Cerr << " Seul le type Rectangle et Rectangle_Axi sont compatibles avec la discretisation VDF en dimension 2" << finl;
236 exit();
237 }
238 }
239 else if (dimension == 3)
240 {
241 if (!sub_type(Hexaedre,elem_geom))
242 {
243 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
244 Cerr << " Seul le type Hexaedre ou Hexadre_Axi sont compatibles avec la discretisation VDF en dimension 3" << finl;
245 exit();
246 }
247 }
248// Calcul de l'orientation des faces virtuelles
249// Note BM: pour compatibilite avec la version 1.5, orientation_
250// n'a pas d'espace virtuel.
251 MD_Vector md_nul;
252 creer_tableau_faces(orientation_);
253 orientation_.echange_espace_virtuel();
254 orientation_.set_md_vector(md_nul); //!< Detache la structure parallele
255
256// Application de la convention VDF sur face_voisin:
257// L'element face_voisin(i,0) doit avoir une coordonnee "ori" plus petite que la face
258// et l'element face_voisin(i,1) doit avoir une coordonnee plus grande.
259 {
260 const int nbr_faces_tot = face_voisins_.dimension_tot(0);
261 for (int i_face = 0; i_face < nbr_faces_tot; i_face++)
262 {
263 const int elem0 = face_voisins_(i_face, 0);
264 const int elem1 = face_voisins_(i_face, 1);
265 const int ori = orientation_[i_face];
266 const double x_face = xv(i_face, ori);
267 double delta = 0.;
268// L'element 0 doit avoir une coordonnee "ori" plus petite que la face
269// et l'element 1 doit avoir une coordonnee plus grande.
270 if (elem0 >= 0)
271 {
272 delta = x_face - xp(elem0, ori);
273 }
274 else
275 {
276 assert(elem1 >= 0); //!< Sinon, grosse erreur: elements voisins non renseignes
277 delta = xp(elem1, ori) - x_face;
278 }
279 if (delta < 0)
280 {
281// On inverse les deux elements
282 face_voisins_(i_face, 0) = elem1;
283 face_voisins_(i_face, 1) = elem0;
284 }
285 }
286 }
287
288 calculer_volumes_entrelaces();
289 Zone_VF::calculer_porosites();
290 Zone_VF::calculer_diametres_hydrauliques();
291 Zone_VF::calculer_coefficient_echange_thermique();
292 Zone_VF::calculer_coefficient_frottement();
293 genere_aretes();
294 calcul_h();
295 Cerr << "L'objet de type Zone_VDF a ete rempli avec succes " << finl;
296
297}

305//! remplissage des volumes entrelaces
306void Zone_VDF::calculer_volumes_entrelaces()
307{
308 Cerr << "On calcule les volumes entrelaces" << finl;
309 creer_tableau_faces(volumes_entrelaces_);
310 volumes_entrelaces_dir_.resize(nb_faces(), 2);
311 creer_tableau_faces(volumes_entrelaces_dir_);
312
313 const int nb_faces_front = premiere_face_int();
314 const int nbf = nb_faces();
315 for (int num_face = 0; num_face<nbf; num_face++)
316 {
317 const double f = (num_face < nb_faces_front) ? 2. : 1.;
318 for (int dir = 0; dir < 2; dir ++)
319 {
320 int elem = face_voisins_(num_face, dir);
321 if (elem != -1)
322 {
323 if ((axi) && (orientation_[num_face]==0))
324 volumes_entrelaces_dir_(num_face, dir) = f * 0.5 * xv_(num_face, 0) * volumes(elem) / xp(elem, 0);
325 else if ((bidim_axi) && (orientation_[num_face]==0))
326 {
327 const double r1 = xv(num_face, 0);
328 const double r2 = xp(elem, 0);
329 const double dz = dim_elem(elem, 1);
330 const double dr = std::fabs(r1 - r2);
331 const double r = std::min(r1, r2);
332 volumes_entrelaces_dir_(num_face, dir) = 2. * M_PI * (r * dr + 0.5 * dr * dr) * dz;
333 }
334 else
335 volumes_entrelaces_dir_(num_face, dir) = f * 0.5 * volumes(elem);
336
337 volumes_entrelaces_[num_face] += volumes_entrelaces_dir_(num_face, dir);
338 }
339 }
340 }
341 volumes_entrelaces_.echange_espace_virtuel();
342 volumes_entrelaces_dir_.echange_espace_virtuel();
343}

345static inline int face_vois(const Zone_VDF& zvdf, const Zone& zone, int face, int i)
346{
347 int elem=zvdf.face_voisins(face,i);
348 if(elem==-1)
349 {
350 face=zone.face_interne_conjuguee(face);
351 if(face!=-1)
352 {
353 elem=zvdf.face_voisins(face,i);
354 assert(elem!=-1);
355 }
356 }
357 return elem;
358}
