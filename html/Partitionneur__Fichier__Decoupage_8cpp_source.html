68//! Lit le contenu du fichier "filename_" et stocke le resultat dans elem_part
69void Partitionneur_Fichier_Decoupage::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
70{
71 if (! ref_domaine_.non_nul())
72 {
73 Cerr << "Error in Partitionneur_Fichier_Decoupage::construire_partition\n";
74 Cerr << " The domain has not been associated" << finl;
75 exit();
76 }
77 if (filename_ == "")
78 {
79 Cerr << "Error in Partitionneur_Fichier_Decoupage::construire_partition\n";
80 Cerr << " The file name has not been initialized" << finl;
81 exit();
82 }
83
84 Cerr << "Reading of splitting file : " << filename_ << finl;
85 EFichier file;
86 file.ouvrir(filename_);
87 if (!file.good())
88 {
89 Cerr << "Error in Partitionneur_Fichier_Decoupage::construire_partition\n";
90 Cerr << " Failed to open file " << filename_ << finl;
91 exit();
92 }
93 file >> elem_part;
94 file >> nb_parts_tot;
95 file.close();
96 const int sz = elem_part.size_array();
97 const int nelem = ref_domaine_.valeur().zone(0).nb_elem();
98 if (nelem != sz)
99 {
100 Cerr << "Error in Partitionneur_Fichier_Decoupage::construire_partition\n";
101 Cerr << " The file contains an array of " << sz << " values.\n";
102 Cerr << " The area contains " << nelem << " elements" << finl;
103 exit();
104 }
105 if (corriger_partition_)
106 {
107 Cerr << " Correction of the partition." << finl;
108 if (liste_bords_periodiques_.size() > 0)
109 corriger_bords_avec_liste(ref_domaine_.valeur(),
110 liste_bords_periodiques_,
111 0,
112 elem_part);
113 corriger_elem0_sur_proc0(elem_part);
114 }
115}
