128template<Champ_Morceaux_Type _TYPE_>
129DoubleTab& TRUSTChamp_Morceaux_generique<_TYPE_>::valeur_aux_elems(const DoubleTab&, const IntVect& les_polys, DoubleTab& val) const
130{
131 if (val.nb_dim() == 2)
132 {
133 assert((val.dimension(0) == les_polys.size()) || (val.dimension_tot(0) == les_polys.size()));
134 assert(val.dimension(1) == nb_compo_);
135 }
136 else erreur_champ_(__func__); //!< (val.nb_dim() > 2)
137
138 const DoubleTab& ch = valeurs();
139 val = 0.;
140 int p;
141
142 for (int rang_poly = 0; rang_poly < les_polys.size(); rang_poly++)
143 if ((p = les_polys(rang_poly)) != -1)
144 for (int n = 0; n < nb_compo_; n++) val(rang_poly, n) = ch(p, n);
145
146 return val;
147}

162{
163 assert(val.size() == les_polys.size());
164 int le_poly;

166

176template<Champ_Morceaux_Type _TYPE_>
177void TRUSTChamp_Morceaux_generique<_TYPE_>::mettre_a_jour(double time)
178{
179 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Morceaux_Type::UNIFORME);
180
181 if (IS_UNIFORME)
182 {
183 Champ_Don_base::mettre_a_jour(time);
184 return;
185 }
186
187 const IntTab& les_elems = mon_domaine->zone(0).les_elems();
188 const int nb_som_elem = mon_domaine->zone(0).nb_som_elem();
189
190 const Champ_base *ch = ref_pb.non_nul() ? &ref_pb->get_champ(nom_champ_parametre_) : nullptr;
191
192 DoubleTab& tab = valeurs();
193
194 for (int i = 0; i < mon_domaine->zone(0).nb_elem_tot(); i++)
195 {
196 /* xs : coordonnees du poly par barycentre des sommets -> pas top */
197 double xs[3] = { 0, };
198 int nb_som = 0, s, r;
199 for (int j = 0; j < nb_som_elem && (s = les_elems(i, j)) >= 0; j++)
200 for (r = 0, nb_som++; r < dimension; r++) xs[r] += mon_domaine->coord(s, r);
201
202 for (r = 0; r < dimension; r++) xs[r] /= nb_som;
203
204 /* calcul de chaque composante */
205 double val = ch ? ch->valeurs()(i, 0) : 0;
206 for (int k = 0; k < tab.dimension(1); k++)
207 {
208 Parser_U& psr = parser(parser_idx(i, k));
209 psr.setVar("x", xs[0]);
210 psr.setVar("y", xs[1]);
211 psr.setVar("z", xs[2]);
212 psr.setVar("t", time);
213
214 if (ref_pb.non_nul()) psr.setVar("val", val);
215 tab(i, k) = psr.eval();
216 }
217 }
218}

248template<Champ_Morceaux_Type _TYPE_>
249Entree& TRUSTChamp_Morceaux_generique<_TYPE_>::complete_readOn(const int dim, const Nom& qsj, Entree& is, Nom& nom)
250{
251 Motcle motlu;
252 int k, poly;
253
254 if (nom != "{")
255 {
256 Cerr << "Error while reading a " << qsj << " . We expected a { instead of " << nom << finl;
257 Process::exit();
258 }
259 is >> motlu;
260 if (motlu != Motcle("defaut"))
261 {
262 Cerr << "Error while reading a " << qsj << " . We expected defaut instead of " << nom << finl;
263 Process::exit();
264 }
265
266 /* parsers par defaut */
267 for (k = 0; k < dim; k++)
268 {
269 Parser_U psr;
270 Nom tmp;
271 is >> tmp;
272 psr.setNbVar(5);
273 psr.setString(tmp);
274 psr.addVar("t"), psr.addVar("x"), psr.addVar("y"), psr.addVar("z");
275 if (ref_pb.non_nul()) psr.addVar("val");
276 psr.parseString();
277
278 for (poly = 0; poly < mon_domaine->zone(0).nb_elem_tot(); poly++)
279 parser_idx(poly, k) = parser.size();
280
281 parser.add(psr);
282 }
283
284 is >> nom;
285 while (nom != Nom("}"))
286 {
287 REF (Sous_Zone) refssz = les_sous_zones.add(mon_domaine->ss_zone(nom));
288 Sous_Zone& ssz = refssz.valeur();
289 for (k = 0; k < dim; k++)
290 {
291 Parser_U psr;
292 Nom tmp;
293 is >> tmp;
294 psr.setNbVar(5);
295 psr.setString(tmp);
296 psr.addVar("t"), psr.addVar("x"), psr.addVar("y"), psr.addVar("z");
297 if (ref_pb.non_nul()) psr.addVar("val");
298 psr.parseString();
299
300 for (poly = 0; poly < ssz.nb_elem_tot(); poly++)
301 parser_idx(ssz(poly), k) = parser.size();
302
303 parser.add(psr);
304 }
305 is >> nom;
306 }
307 return is;
308}
