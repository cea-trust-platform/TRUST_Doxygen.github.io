139inline int Champ_Generique_base::composante(const Nom& nom_test,const Nom& nom,const Noms& composantes, const Noms& syno)
140{
141 Motcle motlu(nom_test);
142 if (motlu == Motcle(nom))
143 return -1;
144 for (int i=0; i<syno.size(); i++)
145 if (syno[i]==motlu) return -1;
146 int n = composantes.size();
147 Motcles les_noms_comp(n);
148 for (int i=0; i<n; i++)
149 les_noms_comp[i] = composantes[i];
150 int ncomp = les_noms_comp.search(motlu);
151 if (ncomp == -1)
152 {
153 Cerr << "Error TRUST, the identifier : " << nom_test << finl
154 << "did not designate the name of the field tested nor any of its components\n";
155 Cerr <<" the components of field named "<<nom <<" are :"<< finl;
156 for (int ii=0; ii<n; ii++)
157 Cerr << composantes[ii] << " ";
158 Cerr << finl;
159 exit();
160 }
161 return ncomp;
162}
