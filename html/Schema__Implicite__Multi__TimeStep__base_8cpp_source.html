63static void Implicite_Multi_TimeStep_calcul_fac_sec(double& residu_,double& residu_old,double& facsec_,const double facsec_max_,int& nb_ite_sans_accel_)
64{
65 double rap = 1.2;
66 int nb_ite_sans_accel_max = 20000;
67
68 if (residu_old==0)
69 {
70 residu_old=residu_;
71 nb_ite_sans_accel_=0;
72 }
73 else if ((residu_old>rap*residu_)||(nb_ite_sans_accel_>nb_ite_sans_accel_max))
74 {
75 facsec_*=sqrt(rap);
76 residu_old=residu_;
77 facsec_=std::min(facsec_,facsec_max_);
78 nb_ite_sans_accel_=0;
79 }
80
81 nb_ite_sans_accel_++;
82}

161int Schema_Implicite_Multi_TimeStep_base::Iterer_Pb(Probleme_base& pb,int ite)
162{
163 int i = 0, ok = 1;
164 int ii = 0;
165 bool convergence_pb = true;
166 bool convergence_eqn = false;
167 int drap = pb.is_dilatable();
168
169 for(i=0; i<pb.nombre_d_equations(); i++)
170 {
171// reverse equation order if quasi-compressible medium
172 if ((drap)&&(i<2))
173 ii = 1-i;
174 else
175 ii = i;
176
177 Equation_base& eqn = pb.equation(ii);
178 DoubleTab& present = eqn.inconnue().valeurs();
179 DoubleTab& futur = eqn.inconnue().futur();
180 double temps = temps_courant_+dt_;
181
182 DoubleTab stored_parameters;
183
184 authorized_equation(eqn);
185
186// imposer_cond_lim sert pour la pression et pour les echanges entre pbs
187 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
188 Cout<<"Solving equation "<<eqn.que_suis_je()<<finl;
189 const DoubleTab& inut=futur;
190
191 store_equation_parameters(eqn,stored_parameters);
192 convergence_eqn = le_solveur.valeur().iterer_eqn(eqn,inut, present, dt_,ite, ok);
193 modify_equation_parameters(eqn,stored_parameters);
194
195 convergence_pb = convergence_pb&&convergence_eqn;
196 futur = present;
197 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
198 present = futur;
199
200// La ligne suivante realise:
201// MAJ NS (donc MAJ inc)
202// MAJ modele de turbulence donc k-eps
203// eqn.inconnue().mettre_a_jour(temps);
204// eqn.inconnue().reculer();
205 changer_temps(eqn,temps);
206 }
207 return (convergence_pb==true);
208}

210void Schema_Implicite_Multi_TimeStep_base::test_stationnaire(Probleme_base& pb)
211{
212 int i = 0;
213 int j = 0;
214
215 for(i=0; i<pb.nombre_d_equations(); i++)
216 {
217// treatement of time convergency
218 DoubleTab& passe = pb.equation(i).inconnue().passe();
219 DoubleTab& present = pb.equation(i).inconnue().valeurs();
220 DoubleTab& futur = pb.equation(i).inconnue().futur();
221
222 futur = present;
223 pb.equation(i).zone_Cl_dis()->imposer_cond_lim(pb.equation(i).inconnue(),temps_courant()+pas_de_temps());
224 present -= passe;
225 present /= dt_;
226 update_critere_statio(present, pb.equation(i));
227 update_time_derivative(pb.equation(i),present);
228 for (j=0; j<nb_valeurs_passees(); j++)
229 {
230 pb.equation(i).inconnue().passe(j) = pb.equation(i).inconnue().passe(j+1);
231 }
232 }

249bool Schema_Implicite_Multi_TimeStep_base::iterateTimeStep(bool& converged)
250{
251 Probleme_base& prob=pb_base();
252
253 int convergence_pb = 0;
254 Initialiser_Champs(prob);
255
256 int ok=0;
257 int compteur;
258 while (!ok)
259 {
260 compteur=0;
261 Cout<<" "<<finl;
262 Cout<<que_suis_je()<<" : solving of problem "<<prob.que_suis_je()<<finl;
263
264 while ((!convergence_pb)&&(compteur<nb_ite_max))
265 {
266 compteur++;
267 prob.updateGivenFields();
268 convergence_pb = Iterer_Pb(prob,compteur);
269 Cout<<" "<<finl;;
270 }
271
272 if ((!convergence_pb)&&(compteur==nb_ite_max))
273 {
274 Cerr<<que_suis_je()<<" has not converged at t="<< temps_courant_ << " with dt =" << dt_<< finl;
275
276 double rap = sqrt(2.);
277 facsec_ /= rap;
278 dt_ /= rap;
279 ok =0;
280
281 Cerr<<"We restart with dt_ =" << dt_ <<finl;
282
283 prob.abortTimeStep();
284 prob.initTimeStep(dt_);
285
286// PL pour corriger plantage danbs U_in_var_impl
287 const int nb_eqn=prob.nombre_d_equations();
288 for(int ii=0; ii<nb_eqn; ii++)
289 {
290 Equation_base& eqn=prob.equation(ii);
291 mettre_a_jour_equation(eqn,temps_courant_+dt_);
292 }
293// Fin
294
295 Implicite_Multi_TimeStep_recommencer_pb(nb_valeurs_passees(),prob);
296 }
297 else
298 {
299 Cout<<"The problem "<<prob.que_suis_je()<<" has converged after "<<compteur<<" iterations."<<finl;
300 ok=1;
301 }
302
303 }
304// modification : prise en compte de la possibilite de modification du dt dans Itere_Pb
305// cas du solveur pour compressible : division du pas de temps par 2
306// si la convergence n'est pas atteinte en un nombre de pas de temps donne
307// double temps = dt_ + temps_courant_;
308 test_stationnaire(prob);
309
310 converged=true;
311
312 return true;
313}

315int Schema_Implicite_Multi_TimeStep_base::faire_un_pas_de_temps_pb_couple(Probleme_Couple& pbc, int& ok)
316{
317// Modif B.M. : Si on fait la sauvegarde entre derivee en temps inco et mettre a jour,
318// un calcul avec reprise n'est pas equivalent au calcul ininterrompu
319// (front-tracking notamment). Donc je mets la sauvegarde au debut du pas de temps.
320// if (lsauv())
321// for (int i=0;i<pbc.nb_problemes();i++)
322// ref_cast(Probleme_base,pbc.probleme(i)).sauver();
323
324 int convergence_pbc = 0;
325 int convergence_pb;
326 int i;
327
328 for(i=0; i<pbc.nb_problemes(); i++)
329 Initialiser_Champs(ref_cast(Probleme_base,pbc.probleme(i)));
330
331
332 int compteur;
333
334 while (!ok)
335 {
336 compteur=0;
337
338 while ((!convergence_pbc )&&(compteur<nb_ite_max))
339 {
340 compteur++;
341 convergence_pbc=1;
342
343
344 for(i=0; i<pbc.nb_problemes()*1; i++)
345 {
346 Probleme_base& pb = ref_cast(Probleme_base,pbc.probleme(i));
347 const int nb_eqn=pb.nombre_d_equations();
348 for(int ii=0; ii<nb_eqn; ii++)
349 {
350 pb.equation(ii).zone_Cl_dis()->calculer_coeffs_echange(temps_courant_+pas_de_temps());
351 }
352
353 }
354
355 for(i=0; i<pbc.nb_problemes(); i++)
356 {
357 pbc.probleme(i).updateGivenFields();
358 convergence_pb = Iterer_Pb(ref_cast(Probleme_base,pbc.probleme(i)),compteur);
359 convergence_pbc = convergence_pbc * convergence_pb ;
360 }
361
362 }
363 if ((!convergence_pbc)&&(compteur==nb_ite_max))
364 {
365 Cerr<<"The process is restarted. " <<finl;
366 facsec_/=2;
367 dt_/=2;
368 ok=0;
369 for(int i2=0; i2<pbc.nb_problemes(); i2++)
370 {
371 ref_cast(Probleme_base,pbc.probleme(i2)).abortTimeStep();
372 ref_cast(Probleme_base,pbc.probleme(i2)).initTimeStep(dt_);
373 Implicite_Multi_TimeStep_recommencer_pb(nb_valeurs_passees(),ref_cast(Probleme_base,pbc.probleme(i2)));
374 }
375 }
376 else ok=1;
377 }
378
379 double residu_1=0;
380 for(i=0; i<pbc.nb_problemes(); i++)
381 {
382 residu_1=residu_;
383 test_stationnaire(ref_cast(Probleme_base,pbc.probleme(i)));
384 residu_=std::max(residu_,residu_1);
385 }
386
387 return 1;
388}

390void Schema_Implicite_Multi_TimeStep_base::ajouter_inertie(Matrice_Base& mat_morse,DoubleTab& secmem,const Equation_base& eqn) const
391{
392 if (nb_pas_dt()>nb_pas_dt_seuil())
393 {
394 const Equation_base& eqn_bis = ref_cast(Equation_base,eqn);
395 int i = 0;
396 int offset = 0;
397 double time_step = pas_de_temps();
398
399// Building of a time-in-ascending-order table
400 DoubleTab times(nb_valeurs_temporelles_effectives());
401 assert(nb_valeurs_passees()+2 == nb_valeurs_temporelles_effectives());
402
403 for (i=0; i<nb_valeurs_passees(); ++i)
404 {
405 offset = nb_valeurs_passees()-i;
406 times[i] = eqn_bis.inconnue().valeur().recuperer_temps_passe(offset); //past time
407 }
408 times[nb_valeurs_passees()] = eqn_bis.inconnue().valeur().recuperer_temps_futur(0); //present time
409 times[nb_valeurs_passees()+1] = eqn_bis.inconnue().valeur().recuperer_temps_futur(1); //future time
410
411 compute_coefficients(time_step,times);
412 add_multi_timestep_data(eqn, mat_morse, secmem);
413 }
414 else
415 {
416// Backward Euler time scheme
417 Schema_Temps_base::ajouter_inertie(mat_morse,secmem,eqn);
418 }
419}

421int Schema_Implicite_Multi_TimeStep_base::faire_un_pas_de_temps_eqn_base(Equation_base& eqn)
422{
423// Necessary because of equations like Transport_K_Eps which must be time incremented but
424// are not equations of the solved problem and therefore are not seen by Schema_Implicite_Multi_TimeStep_base::iterateTimeStep
425
426 DoubleTab& passe = eqn.inconnue().passe();
427 DoubleTab& present = eqn.inconnue().valeurs();
428 DoubleTab& futur = eqn.inconnue().futur();
429 int compteur = 0, ok = 1;
430 int i = 0;
431 bool convergence_eqn = false;
432
433 DoubleTab stored_parameters;
434
435 authorized_equation(eqn);
436
437// Initialize time data
438 if ( (eqn.que_suis_je() != "Euler") || temps_courant_ ==0.)
439 {
440 for (i=nb_valeurs_passees(); i>-1; i--)
441 {
442 eqn.inconnue().passe(i+1) = eqn.inconnue().passe(i);
443 }
444 }
445
446// sert pour la pression et les couplages
447 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
448
449 Cout << finl;
450 while ((!convergence_eqn)&&(compteur<nb_ite_max))
451 {
452 compteur++;
453
454 Cout<<"==================================================================================" << finl;
455 Cout<<que_suis_je()<<": Implicit iteration " << compteur << " on the "<<eqn.que_suis_je() << " equation of the problem "<< eqn.probleme().le_nom()<< " :" <<finl;
456 Cout<<"==================================================================================" << finl;
457
458 const DoubleTab& inut=futur;
459 store_equation_parameters(eqn,stored_parameters);
460 convergence_eqn=le_solveur.valeur().iterer_eqn(eqn, inut, present, dt_, compteur, ok);
461 modify_equation_parameters(eqn,stored_parameters);
462 futur=present;
463 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
464 present=futur;
465 }
466
467// Calculation of time derivative at time n+1
468 present -= passe; //"present" contains unknown at time n+1 and "passe" contains unknown at time n
469 present/=dt_;
470
471 update_critere_statio(present, eqn);
472 update_time_derivative(eqn,present);
473
474// Update time data
475 for (i=0; i<nb_valeurs_passees(); i++)
476 {
477 eqn.inconnue().passe(i) = eqn.inconnue().passe(i+1);
478 }
479
480 return 1;
481}

483int Schema_Implicite_Multi_TimeStep_base::reprendre(Entree& s)
484{
485 Nom nom_fichier(nom_du_cas());
486 nom_fichier+=".dt_ev";
487 LecFicDiffuse test;
488 if (!test.ouvrir(nom_fichier))
489 {
490 Cerr << "*****************************************" << finl;
491 Cerr << "***************** WARNING ***************" << finl;
492 Cerr << "File " << nom_fichier << " does not exist." << finl;
493 Cerr << "In order to restart a calculation carried out with an implicit time scheme" << finl;
494 Cerr << "it is preferable to re-read the .dt_ev file to pick up some informations" << finl;
495 Cerr << "and in particular the facsec of the previous calculation." << finl;
496 Cerr << "TRUST will use facsec= " << facsec_ << "." << finl;
497 Cerr << "Else specify the facsec wanted value in your data file." << finl;
498 Cerr << "*****************************************" << finl;
499 return 1;
500 }
501
502// Reprise du facsec et du residu dans le fichier .dt_ev s'il existe
503 double facsec_lu=1.;
504 double residu_lu=0;
505 double facsec_lu_old;
506
507// Test ouverture du fichier
508 if (je_suis_maitre())
509 {
510 EFichier fichier;
511 fichier.ouvrir(nom_fichier);
512 Nom chaine;
513 double temps=0;
514 double dt;
515 std::string ligne;
516// Si en tete on lit
517 fichier >> chaine;
518 if (chaine=="#")
519 {
520// On lit la ligne complete
521 std::getline(fichier.get_ifstream(), ligne);
522 } //!< Sinon on reouvre
523 else
524 {
525 fichier.ouvrir(nom_fichier);
526 }
527// Recherche du pas de temps precedant tinit_
528 fichier >> temps;
529 while (!fichier.eof() && temps<tinit_)
530 {
531 facsec_lu_old = facsec_lu;
532 fichier >> dt;
533 fichier >> facsec_lu;
534 fichier >> residu_lu;
535// On lit le reste de la ligne
536 std::getline(fichier.get_ifstream(), ligne);
537 fichier >> temps;
538 if (facsec_lu_old != facsec_lu)
539 {
540 residu_old_ = residu_lu;
541 nb_ite_sans_accel_ = 1;
542 }
543 nb_ite_sans_accel_++;
544 }
545 }
546 envoyer_broadcast(facsec_lu, 0 /* pe source */);
547 envoyer_broadcast(residu_lu, 0 /* pe source */);
548 envoyer_broadcast(nb_ite_sans_accel_, 0 /* pe source */);
549 envoyer_broadcast(residu_old_, 0 /* pe source */);
550
551// On prend le facsec lu uniquement s'il est entre les
552// bornes specifiees dans le jeu de donnees
553// En effet, on peut faire un calcul explicite puis une reprise en implicite
554 residu_ = residu_lu;
555 if (facsec_lu>=facsec_)
556 {
557 if (facsec_lu<=facsec_max_)
558 facsec_ = facsec_lu;
559 else
560 facsec_ = facsec_max_;
561 if (Process::je_suis_maitre()) Cerr << "Facsec after reading in " << nom_fichier << " : " << facsec_ << finl;
562 if (Process::je_suis_maitre()) Cerr << "Residu after reading in " << nom_fichier << " : " << residu_ << finl;
563 }
564 else
565 {
566 if (Process::je_suis_maitre())
567 {
568 Cerr << "The readen facsec in " << nom_fichier << " : " << facsec_lu << finl;
569 Cerr << "is not used since it is lower than the facsec from the data set : " << facsec_ << finl;
570 }
571 }
572 return 1;
573}
