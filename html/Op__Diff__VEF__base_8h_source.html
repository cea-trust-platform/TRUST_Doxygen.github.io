83inline double Op_Diff_VEF_base::viscA(int i, int j, int num_elem, double diffu) const
84{
85 const Zone_VEF& zone=la_zone_vef.valeur();
86 const IntTab& face_voisins=zone.face_voisins();
87 const DoubleTab& face_normales=zone.face_normales();
88 const DoubleVect& inverse_volumes=zone.inverse_volumes();
89 double pscal = face_normales(i,0)*face_normales(j,0)
90 + face_normales(i,1)*face_normales(j,1);
91 if (Objet_U::dimension == 3)
92 pscal += face_normales(i,2)*face_normales(j,2);
93
94// *zone.porosite_elem(num_elem));
95 if ( (face_voisins(i,0) == face_voisins(j,0)) ||
96 (face_voisins(i,1) == face_voisins(j,1)) )
97 return -(pscal*diffu)*inverse_volumes(num_elem);
98 else
99 return (pscal*diffu)*inverse_volumes(num_elem);
100}

104inline double Op_Diff_VEF_base::viscA(int i, int j, int num_elem, DoubleTab& diffu) const
105{
106 const Zone_VEF& zone=la_zone_vef.valeur();
107 const IntTab& face_voisins=zone.face_voisins();
108 const DoubleTab& face_normales=zone.face_normales();
109 const DoubleVect& inverse_volumes=zone.inverse_volumes();
110
111 double DSiSj=0;
112 for (int k=0; k<dimension; k++)
113 for (int l=0; l<dimension; l++)
114 DSiSj += diffu(num_elem,k*dimension+l)*face_normales(i,k)*face_normales(j,l); //!< steph
115
116 if ( (face_voisins(i,0) == face_voisins(j,0)) ||
117 (face_voisins(i,1) == face_voisins(j,1)) )
118 return -DSiSj*inverse_volumes(num_elem);
119 else
120 return DSiSj*inverse_volumes(num_elem);
121}
122
123// dinhvan 14 sept 18: tableau diffu est vu comme bidimensionnel mais pas 1D, ni 3D,
124// si scalaire de type [[value]], si tenseur de type [nb_elem x nb_comp]
125// case of multi-scalar
126inline double Op_Diff_VEF_base::viscA(int i, int j, int num_elem, ArrOfDouble& diffu_ci_cj_elem) const
127{
128 const Zone_VEF& zone=la_zone_vef.valeur();
129 const IntTab& face_voisins=zone.face_voisins();
130 const DoubleTab& face_normales=zone.face_normales();
131 const DoubleVect& inverse_volumes=zone.inverse_volumes();
132
133 double DSiSj=0;
134 for (int k=0; k<dimension; k++)
135 for (int l=0; l<dimension; l++)
136 DSiSj += diffu_ci_cj_elem[k*dimension+l]*face_normales(i,k)*face_normales(j,l);
137
138 if ( (face_voisins(i,0) == face_voisins(j,0)) ||
139 (face_voisins(i,1) == face_voisins(j,1)) )
140 return -DSiSj*inverse_volumes(num_elem);
141 else
142 return DSiSj*inverse_volumes(num_elem);
143}
