83void diviser_par_rho_np1_face(Equation_base& eqn,DoubleTab& tab)
84{
85 Fluide_Dilatable_base& fluide_dil = ref_cast(Fluide_Dilatable_base,eqn.milieu());
86 const DoubleTab& rho = fluide_dil.rho_face_np1();
87 int nbdim = tab.nb_dim();
88 int taille_0_tot = tab.dimension_tot(0);
89
90 if (nbdim==1)
91 for (int i=0; i<taille_0_tot; i++)
92 tab(i) /= rho(i);
93 else
94 {
95 int dim = tab.dimension(1);
96 for (int i=0; i<taille_0_tot; i++)
97 for (int j=0; j<dim; j++)
98 tab(i,j) /= rho(i);
99 }
100}

102void iterer_eqn_expl(Equation_base& eqn,int nb_iter,double dt,DoubleTab& current,DoubleTab& dudt,
103 int& converge,const int no_qdm_)
104{
105 if (nb_iter>1)
106 {
107 converge = 1;
108 return;
109 }
110 if ((no_qdm_))
111 {
112 Cout<<eqn.que_suis_je()<<" equation is not solved."<<finl;
113 eqn.valider_iteration();
114 converge = 1;
115 return;
116
117 }
118 double dt_stab = eqn.calculer_pas_de_temps();
119
120// Voir Schema_Temps_base::limpr pour information sur modf
121 double n_sous_ite;
122 modf((dt/dt_stab), &n_sous_ite);
123 if (dt>dt_stab*n_sous_ite) n_sous_ite=n_sous_ite+1.;
124 double dt_calc = dt/n_sous_ite;
125 assert(dt_calc<=dt_stab);
126 for (double i=0; i<n_sous_ite; i=i+1.)
127 {
128 eqn.derivee_en_temps_inco(dudt);
129 dudt *= dt_calc;
130 current += dudt;
131 eqn.valider_iteration();
132 }
133 Cout <<"It is solved with " << n_sous_ite << " explicit sub-iterations."<<finl;
134 converge=1;
135}

137void iterer_eqn_expl_diffusion_implicite(Equation_base& eqn,int nb_iter,double dt,DoubleTab& current,DoubleTab& dudt,
138 int& converge,const int no_qdm_,const double seuil)
139{
140 if (nb_iter>1)
141 {
142 converge = 1;
143 return;
144 }
145 if ((no_qdm_))
146 {
147 Cout<<eqn.que_suis_je()<<" equation is not solved."<<finl;
148 eqn.valider_iteration();
149 converge = 1;
150 return;
151 }
152
153// dudt = current;
154 Schema_Temps_base& sch = eqn.probleme().schema_temps();
155 double seuil_sa = sch.seuil_diffusion_implicite();
156 int flag_sa = sch.diffusion_implicite();

158 sch.set_diffusion_implicite() = 1;
159 eqn.derivee_en_temps_inco(dudt);
160 dudt *= dt;
161 current += dudt;
162 eqn.valider_iteration();
163
164// On remet les parametres de diffusion implicite a leur ancienne valeur
165 sch.set_seuil_diffusion_implicite() = seuil_sa;
166 sch.set_diffusion_implicite() = flag_sa;
167 converge = 1;

170bool Simple::iterer_eqn(Equation_base& eqn,const DoubleTab& inut,DoubleTab& current,
171 double dt,int nb_iter, int& ok)
172{
173 DoubleTab dudt(current);
174 Parametre_implicite& param = get_and_set_parametre_implicite(eqn);
175
176
177 int converge=0;
178 int nb_pas_dt = eqn.schema_temps().nb_pas_dt();
179
180 double temps = eqn.schema_temps().temps_courant();
181 int freq = param.equation_frequence_resolue(temps);
182
183 if (freq!=0 && (nb_pas_dt%freq)!=0)
184 {
185 Cout<<"Time step : "<<nb_pas_dt<<" - "<<eqn.que_suis_je()<<" equation is not solved."<<finl;
186 eqn.valider_iteration();
187 return true;
188 }
189 if (eqn.equation_non_resolue())
190 {
191 Cout<<eqn.que_suis_je()<<" equation is not solved."<<finl;
192// on calcule une fois la derivee pour avoir les flux bord
193 if (eqn.schema_temps().nb_pas_dt()==0)
194 {
195 DoubleTab toto(eqn.inconnue().valeurs());
196 eqn.derivee_en_temps_inco(toto);
197 }
198 eqn.valider_iteration();
199 converge = 1;
200 return true;
201 }
202 if (param.calcul_explicite())
203 {
204 if (param.seuil_diffusion_implicite()<0)
205 iterer_eqn_expl(eqn,nb_iter,dt,current,dudt,converge,no_qdm_);
206 else
207 iterer_eqn_expl_diffusion_implicite(eqn,nb_iter,dt,current,dudt,converge,no_qdm_,param.seuil_diffusion_implicite());
208
209 return (converge==1);
210 }
211
212 double& seuil_convergence_implicite = param.seuil_convergence_implicite();
213 double seuil_verification_solveur = param.seuil_verification_solveur();
214 double seuil_test_preliminaire_solveur = param.seuil_test_preliminaire_solveur();
215 assert(seuil_verification_solveur>0);
216 SolveurSys& solveur = param.solveur();
217 double seuil_convg = seuil_convergence_implicite;
218 if (is_seuil_convg_variable)
219 {
220 double residu = ref_cast(Schema_Euler_Implicite,eqn.schema_temps()).residu_old();
221
222 if ((residu*dt*10<seuil_convg)&&(nb_iter==1))
223 {
224 seuil_convergence_implicite /= 1.1;
225 seuil_convg = seuil_convergence_implicite;
226 }
227 Cout<<"seuil_convg "<<seuil_convg<<finl;
228 }
229
230 converge = 0;
231 if ((no_qdm_) && (sub_type(Navier_Stokes_std,eqn)))
232 {
233 Matrice& matrice_en_pression_2 = ref_cast(Navier_Stokes_std, eqn).matrice_pression();
234 if (!matrice_en_pression_2.non_nul()) matrice_en_pression_2.detach();
235
236 converge = 1;
237 return true;
238 }
239
240//
241// Resolution implicite -par iterer_NS pour Navier_Stokes
242// -...solveur.resoudre_systeme()... pour les autres equations
243//
244
245 dudt = current; //!< pour pouvoir tester la convergence.
246 Matrice_Morse matrice;
247 if (!(sub_type(Navier_Stokes_std,eqn) && sub_type(SETS, *this))) //SETS et ICE gerent eux-memes leurs matrices
248 {
249 eqn.dimensionner_matrice(matrice);
250 matrice.get_set_coeff() = 0;
251 }
252
253 DoubleTrav resu(current);
254
255 if( sub_type(Navier_Stokes_std,eqn))
256 {
257 Navier_Stokes_std& eqnNS = ref_cast(Navier_Stokes_std,eqn);
258 DoubleTab& pression = eqnNS.pression().valeurs();
259 DoubleTrav secmem(pression);
260 pression.echange_espace_virtuel();
261 iterer_NS(eqnNS,current,pression,dt,matrice,seuil_verification_solveur,secmem,nb_iter,converge, ok);
262 }
263 else
264 {
265 solveur.valeur().reinit();
266 DoubleTrav resu_temp(current); /* residu en increments */
267 if (eqn.has_interface_blocs()) /* si assembler_blocs est disponible */
268 {
269 if (eqn.discretisation().que_suis_je().debute_par("PolyMAC"))
270 {
271 eqn.assembler_blocs_avec_inertie({{ eqn.inconnue().le_nom().getString(), &matrice }}, resu_temp, { });
272 resu = resu_temp;
273 matrice.ajouter_multvect(current, resu);
274 }
275 else
276 {
277 eqn.assembler_blocs_avec_inertie({{ eqn.inconnue().le_nom().getString(), &matrice }}, resu, { });
278 resu_temp = 0;
279 matrice.ajouter_multvect(current,resu_temp);
280 resu_temp -= resu;
281 }
282
283 }
284 else
285 {
286 eqn.assembler_avec_inertie(matrice,current,resu);
287 resu_temp = 0;
288 matrice.ajouter_multvect(current,resu_temp);
289 resu_temp -= resu;
290 }
291
292 if (seuil_test_preliminaire_solveur>0)
293 {
294 double norme_b=mp_norme_vect(resu_temp);
295 if (norme_b<seuil_test_preliminaire_solveur)
296 {
297// GF le test suivant est peut etre intelligent ?
298// if ( nb_iter>1)
299 {
300 converge = 1;
301 Cout<<"Resolution of system is not necessary: "<< norme_b <<" < "<< seuil_test_preliminaire_solveur <<finl;
302 }
303 }
304 }
305 if (converge==0)
306 {
307 int con = 0;
308 while (con==0)
309 {
310 con = 1;
311 solveur.resoudre_systeme(matrice,resu,current);
312 ok = eqn.milieu().check_unknown_range(); //verification que l'inconnue est dans les bornes du milieu
313
314 if (ok)
315 {
316 resu_temp = 0;
317 matrice.ajouter_multvect(current,resu_temp);
318 resu_temp -= resu;
319 double norme_resu = mp_norme_vect(resu_temp) ;
320
321 if (norme_resu>seuil_verification_solveur) con = 0;
322 if (con==0)
323 {
324 Cout<<"Residu of iterative solving : "<<norme_resu<<" instead of "<<seuil_verification_solveur<<finl;
325 Cout<<"Iterating on the linear system again:"<< finl;
326 seuil_verification_solveur *= 2;
327 con = 0;
328 }
329 }
330 else current = eqn.inconnue().passe(); //si ok == 0, on restaure la valeur passee de inco
331 }
332 converge = 0;
333 }
334 }
335
336//
337// Test de convergence de la solution entre deux iterations successives
338// Pas applique pour l inconnue de N_S avec alorithme PISO et Implicite
339//
340
341 if(!converge && ok)
342 {
343// permet de controler ce qui se passe
344// en particulier la positivite de K et de eps
345 eqn.valider_iteration();
346 dudt -= current;
347 double dudt_norme = mp_norme_vect(dudt);
348 converge = (dudt_norme < seuil_convg);
349 if (!converge)
350 {
351 Cout<<eqn.que_suis_je()<<" is not converged at the implicit iteration "<<nb_iter<<" ( ||uk-uk-1|| = "<<dudt_norme<<" > implicit threshold "<<seuil_convg<<" )"<<finl;
352 if (nb_iter>=10) Cout << "Consider lowering facsec_max value. Look at the reference manual for advice to set facsec_max value according to the problem type." << finl;
353 }
354 else
355 Cout<<eqn.que_suis_je()<<" is converged at the implicit iteration "<<nb_iter<<" ( ||uk-uk-1|| = "<<dudt_norme<<" < implicit threshold "<<seuil_convg<<" )"<<finl;
356 }
357
358 if(ok && eqn.discretisation().que_suis_je().debute_par("PolyMAC")) eqn.probleme().mettre_a_jour(eqn.schema_temps().temps_courant());
359 solveur->reinit();
360 return (ok && converge==1);
361}

363bool Simple::iterer_eqs(LIST(REF(Equation_base)) eqs, int nb_iter, int& ok)
364{
365// on recupere le solveur de systeme lineaire
366 Parametre_implicite& param = get_and_set_parametre_implicite(eqs[0]);
367 SolveurSys& solveur = param.solveur();
368 double seuil_convg = param.seuil_convergence_implicite();
369 int i, j, bs = 0; //bs : linesize commune des tableaux si > 0, 0 sinon
370
371 /* cle pour la memoization */
372 std::vector<intptr_t> key(eqs.size());
373 for (i = 0; i < eqs.size(); i++) key[i] = (intptr_t) &eqs[i].valeur();
374
375 int init = !mbloc.count(key); //premier passage
376 Matrice_Bloc& Mglob = mbloc[key];
377
378 if (init)
379 for (Mglob.dimensionner(eqs.size(), eqs.size()), i = 0; i < eqs.size(); i++)
380 for (j = 0; j < eqs.size(); j++) Mglob.get_bloc(i, j).typer("Matrice_Morse");
381
382 /* pour interface_blocs : si toutes les equations ont cette interface, on l'utilise */
383 int interface_blocs_ok = 1;
384 for (i = 0; i < eqs.size(); i++) interface_blocs_ok &= eqs[i]->has_interface_blocs();
385 std::vector<matrices_t> mats(eqs.size()); //ligne de matrices de l'equation i
386 for (i = 0; i < eqs.size(); i++)
387 for (j = 0; j < eqs.size(); j++)
388 {
389 Nom nom_i = eqs[j]->inconnue().le_nom();
390// champ d'un autre probleme : on ajoute un suffixe
391 if (eqs[i]->probleme().le_nom().getString() != eqs[j]->probleme().le_nom().getString()) nom_i += Nom("/") + eqs[j]->probleme().le_nom();
392 mats[i][nom_i.getString()] = &ref_cast(Matrice_Morse, Mglob.get_bloc(i, j).valeur());
393 }
394
395// Les inconues/residus ont-ils la meme forme?
396 for (bs = eqs[0]->inconnue().valeurs().line_size(), i = 1; i < eqs.size(); i++)
397 if (eqs[i]->inconnue().valeurs().line_size() != bs) bs = 0;
398
399// MD_Vector global
400 MD_Vector_composite mdc; //version composite
401 for (i = 0; i < eqs.size(); i++)
402 mdc.add_part(eqs[i]->inconnue().valeurs().get_md_vector(), bs ? 0 : eqs[i]->inconnue().valeurs().line_size());
403 MD_Vector mdv;
404 mdv.copy(mdc);
405
406 if (init) //1er passage -> dimensionnement des MD_Vector et des matrices
407 {
408 /* dimensionnement de la matrice globale */
409 if (interface_blocs_ok)
410 for (i = 0; i < eqs.size(); i++) eqs[i]->dimensionner_blocs(mats[i], {});
411 else for (i = 0; i < eqs.size(); i++)
412 for (j = 0; j < eqs.size(); j++)
413 {
414 Matrice_Morse& mat = ref_cast(Matrice_Morse, Mglob.get_bloc(i, j).valeur()), mat2;
415 int nl = eqs[i]->inconnue().valeurs().size_totale(), nc = eqs[j]->inconnue().valeurs().size_totale();
416 if (i == j) eqs[i]->dimensionner_matrice(mat);
417 eqs[i]->dimensionner_termes_croises(i == j ? mat2 : mat, eqs[j]->probleme(), nl, nc);
418 if (i == j) mat += mat2;
419 }
420 }
421 else for (i = 0; i < eqs.size(); i++)
422 for (j = 0; j < eqs.size(); j++) //passages suivantes -> il suffit de reallouer les tableaux coeff()
423 {
424 Matrice_Morse& mat = ref_cast(Matrice_Morse, Mglob.get_bloc(i, j).valeur());
425 mat.get_set_coeff().resize(mat.get_set_tab2().size());
426 }
427
428// tableaux de travail
429 DoubleTrav inconnues, residus, dudt;
430 if (bs) inconnues.resize(0, bs), residus.resize(0, bs), dudt.resize(0, bs); //pour que les tableaux aggreges aient la bonne line_size() si elle existe
431 MD_Vector_tools::creer_tableau_distribue(mdv, inconnues);
432 MD_Vector_tools::creer_tableau_distribue(mdv, residus);
433 MD_Vector_tools::creer_tableau_distribue(mdv, dudt);
434 DoubleTab_parts residu_parts(residus), inconnues_parts(inconnues), dudt_parts(dudt);
435
436// remplissage des inconnues
437 for(i = 0; i < eqs.size(); i++) inconnues_parts[i] = eqs[i]->inconnue().valeurs();
438 dudt = inconnues;
439
440// remplissage des matrices
441 if (interface_blocs_ok)
442 {
443 for (i = 0; i < eqs.size(); i++)
444 {
445 eqs[i]->assembler_blocs_avec_inertie(mats[i], residu_parts[i], {});
446 if (!eqs[i]->discretisation().que_suis_je().debute_par("PolyMAC"))
447 {
448 for (j = 0; j < eqs.size(); j++)
449 {
450 Nom nom_j = eqs[j]->inconnue().le_nom();
451 if (eqs[i]->probleme().le_nom().getString() != eqs[j]->probleme().le_nom().getString())
452 {
453 nom_j += Nom("/") + eqs[j]->probleme().le_nom();
454 mats[i][nom_j.getString()]->ajouter_multvect(inconnues_parts[j], residu_parts[i]);
455 }
456 }
457 }
458 }
459 if (eqs[0]->discretisation().que_suis_je().debute_par("PolyMAC")) Mglob.ajouter_multvect(inconnues, residus); //pour ne pas resoudre en increments
460 }
461 else for(i = 0; i < eqs.size(); i++)
462 for (j = 0; j < eqs.size(); j++)
463 {
464 Matrice_Morse& mat = ref_cast(Matrice_Morse, Mglob.get_bloc(i, j).valeur());
465 eqs[i]->ajouter_termes_croises(inconnues_parts[i], eqs[j]->probleme(), inconnues_parts[j], residu_parts[i]);
466 eqs[i]->contribuer_termes_croises(inconnues_parts[i], eqs[j]->probleme(), inconnues_parts[j], mat);
467 /* si i == j, alors assembler_avec_inertie() se charge du produit matrice/vecteur : sinon, on doit le faire a la main */
468 if (i == j) eqs[i]->assembler_avec_inertie(mat, inconnues_parts[i], residu_parts[i]);
469 else mat.ajouter_multvect(inconnues_parts[j], residu_parts[i]);
470 }
471
472// resolution
473 solveur.valeur().reinit();
474 solveur.resoudre_systeme(Mglob, residus, inconnues);
475 inconnues.echange_espace_virtuel();
476
477// mise a jour
478 bool converge = true;
479 for(i = 0; i < eqs.size(); i++)
480 {
481 dudt_parts[i] -= inconnues_parts[i];
482 double dudt_norme = mp_norme_vect(dudt_parts[i]);
483 eqs[i]->inconnue().valeurs() = inconnues_parts[i];
484
485 converge &= (dudt_norme < seuil_convg);
486 if (!converge)
487 {
488 Cout<<eqs[i]->que_suis_je()<<" is not converged at the implicit iteration "<<nb_iter<<" ( ||uk-uk-1|| = "<<dudt_norme<<" > implicit threshold "<<seuil_convg<<" )"<<finl;
489 if (nb_iter>=10) Cout << "Consider lowering facsec_max value. Look at the reference manual for advice to set facsec_max value according to the problem type." << finl;
490 }
491 else
492 Cout<<eqs[i]->que_suis_je()<<" is converged at the implicit iteration "<<nb_iter<<" ( ||uk-uk-1|| = "<<dudt_norme<<" < implicit threshold "<<seuil_convg<<" )"<<finl;
493 eqs[i]->inconnue().futur() = eqs[i]->inconnue().valeurs();
494 const double t = eqs[i]->schema_temps().temps_courant() + eqs[i]->schema_temps().pas_de_temps();
495 eqs[i]->zone_Cl_dis()->imposer_cond_lim(eqs[i]->inconnue(), t);
496 eqs[i]->inconnue().valeurs() = eqs[i]->inconnue().futur();
497 eqs[i]->inconnue().valeur().Champ_base::changer_temps(t);
498 }
499 for(i = 0; i < eqs.size(); i++) eqs[i]->probleme().mettre_a_jour(eqs[i]->schema_temps().temps_courant());
500
501// on desalloue les tableaux de coeffs
502 for (i = 0; i < eqs.size(); i++)
503 for (j = 0; j < eqs.size(); j++) ref_cast(Matrice_Morse, Mglob.get_bloc(i, j).valeur()).get_set_coeff().reset();
504
505 return converge;
506}

508void Simple::calculer_correction_en_vitesse(const DoubleTrav& correction_en_pression,DoubleTrav& gradP,DoubleTrav& correction_en_vitesse,const Matrice_Morse& matrice,const Operateur_Grad& gradient)
509{
510 int deux_entrees = 0;
511 if (correction_en_vitesse.nb_dim()==2) deux_entrees = 1;
512 gradient.valeur().multvect(correction_en_pression,gradP);
513 int nb_comp = 1;
514 if(deux_entrees)
515 nb_comp = correction_en_vitesse.dimension(1);
516
517 int nb_ligne_reel = correction_en_vitesse.dimension(0);
518 if ((Process::nproc()==1)&& (nb_ligne_reel!=matrice.ordre()/nb_comp)) abort();
519 if (deux_entrees==0)
520 {
521// D(Uk-1)^-1 resu
522 int i,j;
523 for(i=0; i<nb_ligne_reel; i++)
524
525 for (j=0; j<nb_comp; j++)
526 {
527// k=tab1(i*nb_comp+j)-1;
528 correction_en_vitesse(i) = -gradP(i)/matrice(i*nb_comp+j,i*nb_comp+j);
529 }
530 }
531 else
532 {
533 int i,j;
534 for(i=0; i<nb_ligne_reel; i++)
535 for (j=0; j<nb_comp; j++)
536 {
537// k = tab1(i*nb_comp+j)-1;

545// Entree : Uk-1 ; Pk-1
546// Sortie Uk ; Pk
547// k designe une iteration
548
549void Simple::iterer_NS(Equation_base& eqn,DoubleTab& current,DoubleTab& pression,
550 double dt,Matrice_Morse& matrice,double seuil_resol,DoubleTrav& secmem,int nb_ite,int& converge, int& ok)
551{
552 Parametre_implicite& param = get_and_set_parametre_implicite(eqn);
553 SolveurSys& solveur = param.solveur();
554
555 Navier_Stokes_std& eqnNS = ref_cast(Navier_Stokes_std,eqn);
556 DoubleTrav gradP(current);
557 DoubleTrav correction_en_pression(pression);
558 DoubleTrav correction_en_vitesse(current);
559 DoubleTrav resu(current);
560 int is_dilat = eqn.probleme().is_dilatable();
561
562// int deux_entrees = 0;
563// if (current.nb_dim()==2) deux_entrees = 1;
564 Operateur_Grad& gradient = eqnNS.operateur_gradient();
565 Operateur_Div& divergence = eqnNS.operateur_divergence();
566
567 /* int nb_comp = 1;
568 int nb_dim = current.nb_dim();
569 if (nb_dim==2)
570 nb_comp = current.dimension(1);
571 */
572
573 gradient.calculer(pression,gradP);
574// Construction de matrice et resu
575// matrice = A[Uk-1] = M/dt + CONV + DIFF
576// resu = A[Uk-1]Uk-1 -(A[Uk-1]Uk-1-Ss) + Sv + (M/dt)Uk-1 -BtPk-1
577 if (eqnNS.has_interface_blocs()) //si l'interface blocs est disponible, on l'utilise
578 eqnNS.assembler_blocs_avec_inertie({{ "vitesse", &matrice }}, resu);
579 else //sinon, on passe par ajouter/contribuer
580 {
581 resu -= gradP;
582 eqnNS.assembler_avec_inertie(matrice,current,resu);
583 }
584
585 solveur.valeur().reinit();
586
587// Resolution du systeme A[Uk-1]U* = -BtP* + Sv + Ss + (M/dt)Uk-1
588// current = U*
589 solveur.resoudre_systeme(matrice,resu,current);
590
591// Relaxation du champ de vitesse U*
592// U* = alpha U*_new + (1-alpha)*U*_old
593 if (nb_ite==1)
594 Ustar_old = current;
595 current *= alpha_ ;
596 current.ajoute(1.-alpha_,Ustar_old);
597 current.echange_espace_virtuel();
598 Ustar_old = current;
599
600// Construction de matrice_en_pression_2 = BD-1Bt[Uk-1]
601 Matrice& matrice_en_pression_2 = eqnNS.matrice_pression();
602 assembler_matrice_pression_implicite(eqnNS,matrice,matrice_en_pression_2);
603 SolveurSys& solveur_pression_ = eqnNS.solveur_pression();
604 solveur_pression_.valeur().reinit();
605
606// Calcul de secmem = BU* (en incompressible) BU* -drho/dt (en quasi-compressible)
607 if (is_dilat)
608 {
609 if (with_d_rho_dt_)
610 {
611 Fluide_Dilatable_base& fluide_dil = ref_cast(Fluide_Dilatable_base,eqn.milieu());
612 fluide_dil.secmembre_divU_Z(secmem);
613 secmem *= -1;
614 }
615 else secmem = 0;
616 divergence.ajouter(current,secmem);
617 }
618 else
619 divergence.calculer(current,secmem);
620 secmem *= -1;
621 secmem.echange_espace_virtuel();
622
623
624// Resolution du systeme (BD-1Bt)P' = BU* en incompressible
625// (BD-1Bt)P' = BU* -drho/dt en quasi-compressible
626// correction_en_pression = P'
627 solveur_pression_.resoudre_systeme(matrice_en_pression_2.valeur(),
628 secmem,correction_en_pression);
629
630// Resolution de DU' = BP'
631// correction_en_vitesse = U'
632 calculer_correction_en_vitesse(correction_en_pression,gradP,correction_en_vitesse,matrice,gradient);
633
634// Correction de la pression P = P* + beta_*P'
635// Correction de la vitesse U = U* + beta_u*U' (beta_u=1)
636
637 pression.ajoute(beta_,correction_en_pression);
638 eqnNS.assembleur_pression().valeur().modifier_solution(pression);
639
640 current += correction_en_vitesse;
641
642 if (is_dilat)
643 diviser_par_rho_np1_face(eqn,current);
644}
