112template<typename _TYPE_>
113typename std::enable_if<(std::is_same<_TYPE_, int>::value || std::is_same<_TYPE_, double>::value),void >::type
114inline assert_parallel(const _TYPE_ x)
115{
116 if (!Comm_Group::check_enabled())
117 return;
118 if (!is_parallel_object(x))
119 {
120 Cerr << "Fatal error in template<typename _TYPE_>assert_parallel(const _TYPE_ x). See log files" << finl;
121 Process::Journal() << "Error in template<typename _TYPE_>assert_parallel(const _TYPE_ x)" << "This processor has this value: " << x << finl;
122 Process::barrier();
123 Process::exit();
124 }
125}

127template<typename _TYPE_>
128inline void mp_collective_op(TRUSTArray<_TYPE_>& x, Comm_Group::Collective_Op op)
129{
130 int sz = x.size_array();
131 assert_parallel<_TYPE_>(sz);
132 if (sz > 0)
133 {
134 _TYPE_ *data = x.addr();
135 _TYPE_ *tmp = new _TYPE_[sz];
136 const Comm_Group& grp = PE_Groups::current_group();
137 grp.mp_collective_op(data, tmp, sz, op);
138 memcpy(data, tmp, sz * sizeof(_TYPE_));
139 delete[] tmp;
140 }
141}

154//! On suppose que les tableaux en entree et en sortie sont de taille nproc() . On envoie src[0] au proc 0,<br>src[1] au proc 1, etc... la valeur recue du processeur 0 et mise dans dest[0], processeur 1 dans dest[1], etc...<br>Il est autorise d'appeler la fonction avec le meme tableau src et dest.
155template<typename _TYPE_>
156inline int envoyer_all_to_all(const TRUSTArray<_TYPE_>& src, TRUSTArray<_TYPE_>& dest)
157{
158 const Comm_Group& grp = PE_Groups::current_group();
159 assert(src.size_array() == grp.nproc());
160 assert(dest.size_array() == grp.nproc());
161 if (src.addr() == dest.addr())
162 {
163 TRUSTArray<_TYPE_> tmp;
164 tmp.resize_array(grp.nproc(), Array_base::NOCOPY_NOINIT);
165 grp.all_to_all(src.addr(), tmp.addr(), sizeof(_TYPE_));
166 dest.inject_array(tmp);
167 }
168 else
169 grp.all_to_all(src.addr(), dest.addr(), sizeof(_TYPE_));
170 return 1;
171}
