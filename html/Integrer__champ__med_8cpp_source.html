62double calcul_surface_triangle(const ArrOfDouble& origine,const ArrOfDouble& point1, const ArrOfDouble& point2)
63{
64 double normal0=(point1[1]-origine[1])*(point2[2]-origine[2])-((point2[1]-origine[1])*(point1[2]-origine[2]));
65
66 double normal1=(point1[2]-origine[2])*(point2[0]-origine[0])-((point2[2]-origine[2])*(point1[0]-origine[0]));
67 double normal2=(point1[0]-origine[0])*(point2[1]-origine[1])-((point2[0]-origine[0])*(point1[1]-origine[1]));
68 double res=normal0*normal0+ normal1*normal1 + normal2*normal2 ;
69
70 return sqrt(res)/2.;
71}

73double portion_surface_elem(const ArrOfDouble& pointA,const ArrOfDouble& pointB, const ArrOfDouble& pointC,const double z)
74{
75 if (z<=pointA[2]) return 0;
76 if (z>pointC[2]) return calcul_surface_triangle(pointA,pointB,pointC);
77 ArrOfDouble pointD(3);
78// le point D est sur AC a hauteur de B.
79 pointD[2]=pointB[2];
80 double lambda=(pointB[2]-pointA[2])/(pointC[2]-pointA[2]);
81 for (int i=0; i<2; i++)
82 pointD[i]=pointA[i]+lambda*(pointC[i]-pointA[i]);
83 if (z<=pointB[2])
84 {
85// on calcul la surface du triangle inferieur
86// et on prend la proportion...
87 double surf_inf=calcul_surface_triangle(pointA,pointB,pointD);
88 double rap=1;
89 if (pointB[2]!=pointA[2])
90 rap=(z-pointA[2])/(pointB[2]-pointA[2]);
91 return surf_inf*rap*rap;
92 }
93 else
94 {
95// on calcul la surface du triangle inferieur
96// et on prend la proportion...
97// et on retire a la surface totale
98 double surf_sup=calcul_surface_triangle(pointB,pointD,pointC);
99 double rap=(z-pointC[2])/(pointB[2]-pointC[2]);
100 double surf_tot=calcul_surface_triangle(pointA,pointB,pointC);
101 return surf_tot-surf_sup*rap*rap;
102 }
103
104}
