57DoubleTab& Op_Div_EF::ajouter(const DoubleTab& vit, DoubleTab& div) const
58{
59 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
60
61
62 const DoubleTab& Bij_thilde=zone_ef.Bij_thilde();
63 int nb_elem=zone_ef.zone().nb_elem();
64 int nb_som_elem=zone_ef.zone().nb_som_elem();
65 const IntTab& elems=zone_ef.zone().les_elems() ;
66
67 for (int elem=0; elem<nb_elem; elem++)
68 {
69 for (int s=0; s<nb_som_elem; s++)
70 {
71 int som=elems(elem,s);
72 for (int i=0; i<dimension; i++)
73 {
74 div(elem)+=Bij_thilde(elem,s,i)*vit(som,i);
75 }
76// Cerr<<finl;
77 }
78 }
79// L'espace virtuel du tableau div n'est pas mis a jour par l'operateur,
80// assert(invalide_espace_virtuel(div));
81 declare_espace_virtuel_invalide(div);
82// calcul de flux bord
83
84 const IntTab& face_sommets=zone_ef.face_sommets();
85 int nb_som_face=zone_ef.nb_som_face();
86 const DoubleTab& face_normales = zone_ef.face_normales();
87 DoubleTab& tab_flux_bords = flux_bords_;
88 const DoubleVect& porosite_sommet=zone_ef.porosite_sommet();
89 tab_flux_bords.resize(zone_ef.nb_faces_bord(),1);
90 tab_flux_bords=0;
91 int premiere_face_int=zone_ef.premiere_face_int();
92 for (int face=0; face<premiere_face_int; face++)
93 {
94 for (int s=0; s<nb_som_face; s++)
95 {
96 int som=face_sommets(face,s);
97 for (int dir=0; dir<Objet_U::dimension; dir++)
98 tab_flux_bords(face,0)+=porosite_sommet(som)*vit(som,dir)*face_normales(face,dir)/nb_som_face;
99 }
100 }
101
102
103 return div;
104}

113int Op_Div_EF::impr(Sortie& os) const
114{
115
116 const int impr_bord=(la_zone_EF->zone().Bords_a_imprimer().est_vide() ? 0:1);
117 SFichier Flux_div;
118 ouvrir_fichier(Flux_div,"",je_suis_maitre());
119 EcrFicPartage Flux_face;
120 ouvrir_fichier_partage(Flux_face,"",impr_bord);
121 const Schema_Temps_base& sch = equation().probleme().schema_temps();
122 double temps = sch.temps_courant();
123 if (je_suis_maitre()) Flux_div.add_col(temps);
124
125 int nb_compo=flux_bords_.dimension(1);
126// On parcours les frontieres pour sommer les flux par frontiere dans le tableau flux_bord
127 DoubleVect flux_bord(nb_compo);
128 DoubleVect bilan(nb_compo);
129 bilan = 0;
130 for (int num_cl=0; num_cl<la_zone_EF->nb_front_Cl(); num_cl++)
131 {
132 flux_bord=0;
133 const Cond_lim& la_cl = la_zcl_EF->les_conditions_limites(num_cl);
134 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
135 int ndeb = frontiere_dis.num_premiere_face();
136 int nfin = ndeb + frontiere_dis.nb_faces();
137 for (int face=ndeb; face<nfin; face++)
138 for(int k=0; k<nb_compo; k++)
139 flux_bord(k)+=flux_bords_(face, k);
140 for(int k=0; k<nb_compo; k++)
141 flux_bord(k)=Process::mp_sum(flux_bord(k));
142
143 if(je_suis_maitre())
144 {
145 for(int k=0; k<nb_compo; k++)
146 {
147// Ajout pour impression sur fichiers separes
148 Flux_div.add_col(flux_bord(k));
149 bilan(k)+=flux_bord(k);
150 }
151 }
152 }
153
154 if(je_suis_maitre())
155 {
156 for(int k=0; k<nb_compo; k++)
157 {
158 Flux_div.add_col(bilan(k));
159 }
160 Flux_div << finl;
161 }
162
163 for (int num_cl=0; num_cl<la_zone_EF->nb_front_Cl(); num_cl++)
164 {
165 const Frontiere_dis_base& la_fr = la_zcl_EF->les_conditions_limites(num_cl).frontiere_dis();
166 const Cond_lim& la_cl = la_zcl_EF->les_conditions_limites(num_cl);
167 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
168 int ndeb = frontiere_dis.num_premiere_face();
169 int nfin = ndeb + frontiere_dis.nb_faces();
170 if (la_zone_EF->zone().Bords_a_imprimer().contient(la_fr.le_nom()))
171 {
172 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps " << temps << " : " << finl;
173 for (int face=ndeb; face<nfin; face++)
174 {
175 if (dimension==2)
176 Flux_face << "# Face a x= " << la_zone_EF->xv(face,0) << " y= " << la_zone_EF->xv(face,1) << " flux=" ;
177 else if (dimension==3)
178 Flux_face << "# Face a x= " << la_zone_EF->xv(face,0) << " y= " << la_zone_EF->xv(face,1) << " z= " << la_zone_EF->xv(face,2) << " flux=" ;
179 for(int k=0; k<nb_compo; k++)
180 Flux_face << " " << flux_bords_(face, k);
181 Flux_face << finl;
182 }
183 Flux_face.syncfile();
184 }
185 }
186
187 return 1;
188}

191void Op_Div_EF::volumique(DoubleTab& div) const
192{
193 const Zone_EF& zone_EF = la_zone_EF.valeur();
194 const DoubleVect& vol = zone_EF.volumes();
195 int nb_elem=zone_EF.zone().nb_elem_tot();
196 int num_elem;
197
198 for(num_elem=0; num_elem<nb_elem; num_elem++)
199 div(num_elem)/=vol(num_elem);
200}
