120void Champ_Q1_EF::gradient(DoubleTab& gradient_elem)
121{
122// Calcul du gradient de la vitesse pour le calcul de la vorticite
123// Gradient ordre 1 (valeur moyenne dans un element)
124// Order 1 gradient (mean value within an element)
125 const Zone_EF& zone_EF_ = zone_EF();
126 const DoubleTab& vitesse = equation().inconnue().valeurs();
127 const IntTab& elems= zone_EF_.zone().les_elems();
128 int nb_som_elem=zone_EF_.zone().nb_som_elem();
129 int nb_elems=zone_EF_.zone().nb_elem_tot();
130 const DoubleVect& volume_thilde=zone_EF_.volumes_thilde();
131 const DoubleTab& Bij_thilde=zone_EF_.Bij_thilde();
132
133 assert(gradient_elem.dimension_tot(0) == nb_elems);
134 assert(gradient_elem.dimension(1) == dimension); //!< line
135 assert(gradient_elem.dimension(2) == dimension); //!< column
136
137 operator_egal(gradient_elem, 0.); //!< Espace reel uniquement
138
139 for (int num_elem=0; num_elem<nb_elems; num_elem++)
140 {
141 for (int a=0; a<dimension; a++) //!< composante numero 1, component number 1
142 {
143 for (int b=0; b<dimension; b++) //!< composante numero 2, component number 2
144 {
145 for (int j=0; j<nb_som_elem; j++)
146 {
147 int s = elems(num_elem,j);
148 gradient_elem(num_elem,a,b) += vitesse(s,a)*Bij_thilde(num_elem,j,b);
149 }
150 gradient_elem(num_elem,a,b) /= volume_thilde(num_elem);
151 }
152 }
153 }
154}

156void Champ_Q1_EF::cal_rot_ordre1(DoubleTab& vorticite)
157{
158 const Zone_EF& zone_EF_ = zone_EF();
159 int nb_elems=zone_EF_.zone().nb_elem_tot();
160
161 DoubleTab gradient_elem(0, dimension, dimension);
162// le tableau est initialise dans la methode gradient():
163 zone_EF_.zone().creer_tableau_elements(gradient_elem, Array_base::NOCOPY_NOINIT);
164 gradient(gradient_elem);
165 Debog::verifier("apres calcul gradient",gradient_elem);
166 int num_elem;
167 switch(dimension)
168 {
169 case 2 :
170 {
171 for (num_elem=0; num_elem<nb_elems; num_elem++)
172 {
173 vorticite(num_elem)=gradient_elem(num_elem,1,0)-gradient_elem(num_elem,0,1);
174 }
175 }
176 break;
177 case 3 :
178 {
179 for (num_elem=0; num_elem<nb_elems; num_elem++)
180 {
181 vorticite(num_elem,0)=gradient_elem(num_elem,2,1)-gradient_elem(num_elem,1,2);
182 vorticite(num_elem,1)=gradient_elem(num_elem,0,2)-gradient_elem(num_elem,2,0);
183 vorticite(num_elem,2)=gradient_elem(num_elem,1,0)-gradient_elem(num_elem,0,1);
184 }
185 }
186 }
187 Debog::verifier("apres calcul vort",vorticite);
188 return ;
189}
