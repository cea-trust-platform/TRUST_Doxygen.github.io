25// XD create_domain_from_sous_zone interprete_geometrique_base create_domain_from_sous_zone 1 This keyword fills the domain domaine_final with the subzone par_sous_zone from the domain domaine_init. It is very useful when meshing several mediums with Gmsh. Each medium will be defined as a subzone into Gmsh. A MED mesh file will be saved from Gmsh and read with Lire_Med keyword by the TRUST data file. And with this keyword, a domain will be created for each medium in the TRUST data file.
26// 16/10/2017: desactivation du mot-cle par_sous_zones
27// create_domain_from_sous_zone interprete_geometrique_base create_domain_from_sous_zone 1 This keyword fills the domain domaine_final with the subzone par_sous_zone or with several subzones par_sous_zones from the domain domaine_init. It is very useful when meshing several mediums with Gmsh. Each medium will be defined as a subzone into Gmsh. A MED mesh file will be saved from Gmsh and read with Lire_Med keyword by the TRUST data file. And with this keyword, a domain will be created for each medium in the TRUST data file.

38Entree& Create_domain_from_sous_zone::interpreter_(Entree& is)
39{
40 Nom nom_dom,nom_dom_org;
41 Nom nom_sous_zone;
42 Noms vec_nom_ssz;
43
44 Param param(que_suis_je());
45 param.ajouter("domaine_final",&nom_dom,Param::REQUIRED); //!< XD_ADD_P ref_domaine new domain in which faces are stored
46 param.ajouter("par_sous_zone",&nom_sous_zone,Param::REQUIRED); //!< XD_ADD_P chaine a sub-area allowing to choose the elements
47 param.ajouter("domaine_init",&nom_dom_org,Param::REQUIRED); //!< XD_ADD_P ref_domaine initial domain 16/10/2017: desactivation du mot-cle par_sous_zones param.ajouter("par_sous_zones",&vec_nom_ssz); // listchaine several sub-zones allowing to choose the elements param.ajouter_condition("is_read_par_sous_zone_or_is_read_par_sous_zones","Interpreter Create_domain_from_sous_zone: one of the keywords par_sous_zone or par_sous_zones must be specified.");
48//
49//
50//
51 param.lire_avec_accolades_depuis(is);
52
53 associer_domaine(nom_dom);
54 Domaine& dom=domaine();
55
56 if (nproc()>1)
57 {
58 Cerr<<"Options par_sous_zone and par_sous_zones of "<<que_suis_je()<<" are for sequential." <<finl;
59 exit();
60 }
61
62 const Domaine& domaine_org=ref_cast(Domaine, objet(nom_dom_org));
63 IntTab marq_elem(domaine_org.zone(0).nb_elem());
64
65 if (vec_nom_ssz.size()==0)
66 vec_nom_ssz.add(Nom(nom_sous_zone));
67 else
68 Cerr << "\nWARNING: par_sous_zones option only available for subzones that do not touch each other!!\n" << finl;
69
70 int nb_poly = 0;
71 DomaineCutter cutter;
72 Noms vide;
73
74 for ( int i=0; i<vec_nom_ssz.size(); i++ )
75 {
76 const Sous_Zone& ssz=ref_cast(Sous_Zone,objet(vec_nom_ssz(i)));
77 nb_poly=ssz.nb_elem_tot();
78 for (int pol=0; pol<nb_poly; pol++)
79 marq_elem(ssz[pol])=1;
80 }
81
82 cutter.initialiser(domaine_org,marq_elem,2,1,vide,1);
83 cutter.construire_sous_domaine(1,dom);
84
85 Zone& zone=dom.zone(0);
86 Bords& bords=zone.faces_bord();
87
88 if (cutter.bords_internes().size()>0)
89 zone.faces_joint().vide();
90 else
91 {
92// on transforme les joints en bord
93 Joints& joints= zone.faces_joint();
94 for (int j=joints.size()-1; j>=0; j--)
95 {
96 Cout <<"The joint"<<j<<" becomes a boundary"<<finl;
97 Bord b;
98 ref_cast(Frontiere,b)=ref_cast(Frontiere,joints(j));
99 b.nommer(Nom("Couture_")+Nom(j));
100 bords.add(b);
101
102 joints.suppr(joints(j));
103 }
104 }
105
106 for (int b=bords.size()-1; b>=0; b--)
107 if (bords(b).nb_faces()==0)
108 {
109 Cout << bords(b).le_nom()<<" deleted"<<finl;
110 bords.suppr(bords(b));
111 }
112 Raccords& listrac=zone.faces_raccord();
113
114 for (int b=bords.size()-1; b>=0; b--)
115 {
116 if (cutter.bords_internes().rang(bords(b).le_nom())>=0)
117 {
118 Cout << bords(b).le_nom() <<" becomes a connection "<<finl;
119 Raccord racc_base;
120 racc_base.typer("Raccord_local_homogene");
121 Raccord_base& racc=racc_base.valeur();
122// on caste en Frontiere pour pouvoir faire la copie ...
123 ref_cast(Frontiere,racc)=ref_cast(Frontiere,bords(b));
124 listrac.add(racc_base);
125
126 bords.suppr(bords(b));
127 }
128 }
129
130// et les sous-zones?
131 const LIST(REF(Sous_Zone)) & liste_sous_zones = domaine_org.ss_zones();
132 int nb_sous_zones = liste_sous_zones.size();
133 const Sous_Zone& ssz=ref_cast(Sous_Zone,objet(vec_nom_ssz(0)));
134 ArrOfInt rev_ssz(domaine_org.zone(0).nb_elem());
135 rev_ssz = -1;
136 for (int i = 0; i < nb_poly; i++)
137 rev_ssz[ssz[i]] = i;
138
139 Nom jdd(" "), jdd_par(" ");
140 int ecr_jdd = 0;
141 for (int i = 0; i < nb_sous_zones; i++)
142 if (liste_sous_zones[i]->le_nom() != nom_sous_zone)
143 {
144// liste des elements de la sous-sous-zone
145 ArrOfInt polys;
146 polys.set_smart_resize(1);
147 for (int j = 0, k; j < liste_sous_zones[i]->nb_elem_tot(); j++)
148 if ((k = rev_ssz[liste_sous_zones[i].valeur()[j]]) >= 0)
149 polys.append_array(k);
150
151 if (!polys.size_array())
152 continue; //sous-sous-zone vide!
153
154 Nom nom_ssz(nom_dom + "_" + liste_sous_zones[i]->le_nom()), file_ssz(nom_ssz + ".file");
155
156// contribution aux JDDs des sous-sous-zones
157 jdd += Nom("export Sous_Zone ") + nom_ssz + "\n";
158 jdd += Nom("Associer ") + nom_ssz + " " + nom_dom + "\n";
159 jdd += Nom("Lire ") + nom_ssz + " { fichier " + file_ssz + " }" + "\n";
160 jdd_par += Nom("export Sous_Zone ") + nom_ssz + "\n";
161 jdd_par += Nom("Associer ") + nom_ssz + " " + nom_dom + "\n";
162 jdd_par += Nom("Lire ") + nom_ssz + " { fichier " + nom_ssz + ".ssz }" + "\n";
163 ecr_jdd = 1;
164
165// fichier de la sous-sous-zone
166 SFichier f_ssz(file_ssz);
167 f_ssz << polys;
168 }
169
170 if (ecr_jdd)
171 {
172 SFichier f_jdd(nom_dom + "_ssz.geo");
173 SFichier f_jdd_par(nom_dom + "_ssz_par.geo");
174 f_jdd << jdd;
175 f_jdd_par << jdd_par;
176 }
177
178 dom.nommer(nom_dom);
179 Scatter::init_sequential_domain(dom);
180
181 return is;
182
183}
