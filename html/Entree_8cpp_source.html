118//! Lecture d'une chaine dans ostream_<br>bufsize est la taille du buffer alloue pour ob (y compris<br>le caractere 0 final).<br>La chaine contient toujours un 0 meme en cas d'echec.<br>La methode renvoie 1 si la lecture est bonne, 0 sinon.<br>Si le buffer est trop petit, pour l'instant on fait exit() mais<br>par la suite on pourra tester: si strlen(ob)==bufsize-1, alors<br>refaire lire() jusqu'a arriver au bout. Si le lire() suivant<br>renvoie une chaine de longueur nulle, cela signifie que la taille de la<br>chaine etait exactement bufsize-1.<br>Attention: le comportement est different en binaire et en ascii.<br>En binaire, on lit la chaine jusqu'au prochain '\\0'.<br>En ascii, on lit la chaine jusqu'au prochain separateur (espace, tab, fin ligne)
119int Entree::get(char* ob, int bufsize)
120{
121 assert(istream_!=0);
122 assert(bufsize > 0);
123 ob[bufsize-1] = 1;
124 if(bin_)
125 {
126// En binaire, on lit jusqu'au prochain caractere 0
127// (y compris espaces, retours a la ligne etc)
128 int i;
129 for (i = 0; i < bufsize-1; i++)
130 {
131 (*istream_).read(ob+i, sizeof(char));
132 if (!error_handle(istream_->fail()))
133 ob[i] = 0;
134 if (ob[i] == 0)
135 break;
136 }
137 ob[i] = 0;
138 }
139 else
140 {
141// L'appel suivant permet a l'operateur>> de limiter le nombre
142// de caracteres lus. On lira au maximum bufsize-1 caracteres.
143 istream_->width(bufsize);
144 (*istream_) >> ob;
145 if (!error_handle(istream_->fail()))
146 ob[0] = 0;
147 }
148 if (ob[bufsize-1] == 0)
149 {
150// Note Benoit Mathieu:
151// Si on a rempli le buffer jusqu'au bout, c'est qu'il est probablement
152// trop petit. Lire la suite est dangereux car en ascii on ne sait pas
153// si on a pu lire pile poil la chaine (donc c'est ok), ou si la chaine
154// n'a pas ete lue en int. Il faut tester tres serieusement la stl et
155// ca depend sans doute de l'implementation. Donc, si le buffer est plein,
156// on plante le code et tant pis.
157 Cerr << "Error in Entree::lire(char* ob, int bufsize) : buffer too small" << finl;
158 Process::exit();
159 }
160 return (!istream_->fail());
161}

345//! Cette fonction est appellee par operateur>>, get, get_nom<br>ouvrir, fermer, lire, etc... en cas d'echec (lorsque fail() est mis)<br>Elle renvoie 0 s'il y a eu une erreur (passer par error_handle() qui<br>traite en inline le cas ou il n'y a pas d'erreur), et 1 s'il n'y a pas<br>d'erreur.<br>(par commodite de codage des methodes qui l'utilisent, on<br>ecrira "return error_handle(fail());"<br>Elle peut etre configuree pour<br>- renvoyer "0" en cas d'erreur et continuer l'execution du code<br>(cas d'un ancien code qui ne gere pas les exceptions mais teste<br>le drapeau fail() de temps en temps)<br>Dans ce cas les methodes operator>> continuent l'execution du code<br>meme en cas d'echec, le contenu des variables lues est indefini !<br>- faire Process::exit() (cas d'une portion de code dans laquelle<br>on ne veut pas faire de gestion d'erreur du tout et ou on suppose<br>que tout va toujours bien)<br>- lever une exception (permet une gestion rigoureuse des erreurs<br>et une information utilisateur optimale en fonction du contexte)<br>Voir aussi set_error_action()
346int Entree::error_handle_(int fail_flag)
347{
348 if (!fail_flag)
349 return 1;
350
351 switch(error_action_)
352 {
353 case ERROR_CONTINUE:
354 break;
355 case ERROR_EXIT:
356 Cerr << "Error while reading in Entree object. Exiting.\n";
357 if (istream_)
358 {
359// On n'utilise pas Entree::eof() car dans le cas d'un Lec_Fic_Dif
360// eof() est parallele et ca peut bloquer.
361 if (get_istream().eof())
362 Cerr << " End of file reached." << finl;
363 else
364 Cerr << " IO error (not an EOF error)." << finl;
365 }
366 Process::exit();
367 break;
368 case ERROR_EXCEPTION:
369 Entree_Sortie_Error e;
370 throw (e);
371 }
372 return 0;
373}

391int is_a_binary_file(Nom& filename)
392{
393// On parcourt les 1000 premiers octets
394// Detection tres imparfaite donc limite
