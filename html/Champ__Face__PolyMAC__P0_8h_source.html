59 /* utilitaire pour le calcul des termes sources : calcule le vecteur v_e + n_f (v_f - v_e. n_f)
60 retour : le vecteur, sa norme et les derivees de celle-ci selon v_e et v_f
61 */
62 inline double v_norm(const DoubleTab& val, const DoubleTab& val_f, int e, int f, int k, int l, double *v_ext, double *dnv) const
63 {
64 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0, zone_dis_base());
65 int d, D = dimension, nf_tot = zone.nb_faces_tot();
66 const DoubleTab& nf = zone.face_normales();
67 const DoubleVect& fs = zone.face_surfaces();
68 double scal = 0, vf = f >= 0 ? val_f(f, k) - (l >= 0 ? val_f(f, l) : 0) : 0, v_temp[3], *v = v_ext ? v_ext : v_temp;
69 for (d = 0; d < D; d++) v[d] = val(nf_tot + D * e + d, k) - (l >= 0 ? val(nf_tot + D * e + d, l) : 0);
70 if (f >= 0)
71 for (d = 0, scal = zone.dot(v, &nf(f, 0)) / fs(f); d < D; d++) v[d] += (vf - scal) * nf(f, d) / fs(f);
72 double nv = sqrt(zone.dot(v, v));
73 if (dnv)
74 for (d = 0; d < D; d++) dnv[d] = nv ? (v[d] - (f >= 0 ? vf * nf(f, d) / fs(f) : 0)) / nv : 0;
75 if (dnv) dnv[3] = f >= 0 && nv ? vf / nv : 0;
76 return nv;
77 }
