87void Polyedre::calculer_centres_gravite(DoubleTab& xp) const
88{
89
90 const Zone& zone=ma_zone.valeur();
91 const IntTab& elem=zone.les_elems();
92 const DoubleTab& coord=zone.domaine().coord_sommets();
93
94
95 int nb_elem;
96 if(xp.dimension(0)==0)
97 {
98 nb_elem = ma_zone->nb_elem_tot();
99 xp.resize(nb_elem,dimension);
100 }
101 else
102 nb_elem=xp.dimension(0);
103
104
105 for (int num_poly=0; num_poly<nb_elem; num_poly++)
106 {
107 double volume=0;
108 Vecteur3 xg(0,0,0);
109 int nb_som_max=elem.dimension(1);
110 int nb_som_reel;
111 for (nb_som_reel=0; nb_som_reel<nb_som_max; nb_som_reel++)
112 {
113 int n=elem(num_poly,nb_som_reel);
114 if (n<0)
115 break;
116 Vecteur3 S(coord(n,0),coord(n,1),coord(n,2));
117 xg+=S;
118 }
119 xg*=1./nb_som_reel;
120 Vecteur3 moinsS0(xg);
121 moinsS0*=-1;
122
123 Vecteur3 vraixg(0,0,0);
124
125 for (int f=PolyhedronIndex_[num_poly]; f<PolyhedronIndex_[num_poly+1]; f++)
126 {
127 int somm_loc3=Nodes_[FacesIndex_[f+1]-1];
128 int n3=elem(num_poly,somm_loc3);
129 Vecteur3 S3(coord(n3,0),coord(n3,1),coord(n3,2));
130 Vecteur3 S3sa(S3);
131 S3+= moinsS0;
132 for (int s=FacesIndex_[f]; s<FacesIndex_[f+1]-2; s++)
133 {
134 int somm_loc1=Nodes_[s];
135 int n1=elem(num_poly,somm_loc1);
136 int somm_loc2=Nodes_[s+1];
137 int n2=elem(num_poly,somm_loc2);
138 Vecteur3 S1(coord(n1,0),coord(n1,1),coord(n1,2));
139 Vecteur3 S2(coord(n2,0),coord(n2,1),coord(n2,2));
140
141 Vecteur3 xgl(xg);
142 xgl+=S3sa;
143 xgl+=S1;
144 xgl+=S2;
145 xgl*=0.25;
146 S1+= moinsS0;
147 S2+= moinsS0;
148 double vol_l= std::fabs(
149 S1[0] * ( S2[1] * S3[2] - S3[1] * S2[2] )
150 + S2[0] * ( S3[1] * S1[2] - S1[1] * S3[2] )
151 + S3[0] * ( S1[1] * S2[2] - S2[1] * S1[2] ) );
152 volume+=vol_l;
153 xgl*=vol_l;
154 vraixg+=xgl;
155 }
156 }
157 vraixg*=1./volume;
158 xp(num_poly,0)=vraixg[0];
159 xp(num_poly,1)=vraixg[1];
160 xp(num_poly,2)=vraixg[2];
161
162// Vecteur3 test= xg -vraixg;
163// Cerr<<num_poly<< "iiii "<<test[0]<< " "<<test[1]<<" "<<test[2]<<finl;
164 }
165
166// Cerr<<" xp "<<xp << finl;
167 return ;

253int Polyedre::contient(const ArrOfDouble& pos, int num_poly ) const
254{
255
256// on regarde si le point P est du meme cote que xg pour chaque face .
257 const Zone& zone=ma_zone.valeur();
258 const IntTab& elem=zone.les_elems();
259 const DoubleTab& coord=zone.domaine().coord_sommets();
260 Vecteur3 P(pos[0],pos[1],pos[2]);
261 Vecteur3 xg(0,0,0);
262 int nb_som_max=elem.dimension(1);
263 int nb_som_reel;
264 for (nb_som_reel=0; nb_som_reel<nb_som_max; nb_som_reel++)
265 {
266 int n=elem(num_poly,nb_som_reel);
267 if (n<0)
268 break;
269 Vecteur3 S(coord(n,0),coord(n,1),coord(n,2));
270 xg+=S;
271 }
272 xg*=1./nb_som_reel;
273
274 for (int f=PolyhedronIndex_[num_poly]; f<PolyhedronIndex_[num_poly+1]; f++)
275 {
276 Vecteur3 n(0,0,0);
277 int somm_loc3=Nodes_[FacesIndex_[f+1]-1];
278 int n3=elem(num_poly,somm_loc3);
279 Vecteur3 moinsS3(-coord(n3,0),-coord(n3,1),-coord(n3,2));
280
281 for (int s=FacesIndex_[f]; s<FacesIndex_[f+1]-2; s++)
282 {
283 int somm_loc1=Nodes_[s];
284 int n1=elem(num_poly,somm_loc1);
285 int somm_loc2=Nodes_[s+1];
286 int n2=elem(num_poly,somm_loc2);
287 Vecteur3 S1(coord(n1,0),coord(n1,1),coord(n1,2));
288 Vecteur3 S2(coord(n2,0),coord(n2,1),coord(n2,2));
289 S1+=moinsS3;
290 S2+=moinsS3;
291 Vecteur3 nTr;
292 Vecteur3::produit_vectoriel(S1,S2,nTr);
293 n+=nTr;
294
295 }
296
297
298
299
300 Vecteur3 S3G(xg);
301 S3G+=moinsS3;
302 Vecteur3 S3P(P);
303 S3P+=moinsS3;
304 double prod_scal1=Vecteur3::produit_scalaire(n,S3G);
305 double prod_scal2=Vecteur3::produit_scalaire(n,S3P);
306 if (prod_scal1*prod_scal2<-Objet_U::precision_geom*prod_scal1*prod_scal1)
307 return 0;
308 }
309
310 return 1;
311}

362void Polyedre::calculer_volumes(DoubleVect& volumes) const
363{
364 const Zone& zone=ma_zone.valeur();
365 const IntTab& elem=zone.les_elems();
366 const DoubleTab& coord=zone.domaine().coord_sommets();
367
368 int size_tot = zone.nb_elem_tot();
369 assert(volumes.size_totale()==size_tot);
370 for (int num_poly=0; num_poly<size_tot; num_poly++)
371 {
372 double volume=0;
373 Vecteur3 xg(0,0,0);
374 int nb_som_max=elem.dimension(1);
375 int nb_som_reel;
376 for (nb_som_reel=0; nb_som_reel<nb_som_max; nb_som_reel++)

379 if (n<0)

382 xg+=S;

385 Vecteur3 moinsS0(xg);
386 moinsS0*=-1;
387 for (int f=PolyhedronIndex_[num_poly]; f<PolyhedronIndex_[num_poly+1]; f++)
388 {
389 int somm_loc3=Nodes_[FacesIndex_[f+1]-1];
390 int n3=elem(num_poly,somm_loc3);
391 Vecteur3 S3(coord(n3,0),coord(n3,1),coord(n3,2));
392 S3+= moinsS0;
393 for (int s=FacesIndex_[f]; s<FacesIndex_[f+1]-2; s++)
394 {
395 int somm_loc1=Nodes_[s];
396 int n1=elem(num_poly,somm_loc1);

398 int n2=elem(num_poly,somm_loc2);
399 Vecteur3 S1(coord(n1,0),coord(n1,1),coord(n1,2));
400 Vecteur3 S2(coord(n2,0),coord(n2,1),coord(n2,2));
401 S1+= moinsS0;
402 S2+= moinsS0;
403 volume += std::fabs(
404 S1[0] * ( S2[1] * S3[2] - S3[1] * S2[2] )
405 + S2[0] * ( S3[1] * S1[2] - S1[1] * S3[2] )
406 + S3[0] * ( S1[1] * S2[2] - S2[1] * S1[2] ) );
407
408 }
409 }
410 volumes(num_poly)=volume/6.;
411 }

458void Polyedre::affecte_connectivite_numero_global(const ArrOfInt& Nodes,const ArrOfInt& FacesIndex,const ArrOfInt& PolyhedronIndex,IntTab& les_elems)
459{

464 nb_som_face_max_=0;

466 for (int ele=0; ele<nelem; ele++)
467 {
468 prov.vide();
469 int nbf=PolyhedronIndex[ele+1]-PolyhedronIndex[ele];
470 if (nbf>nb_face_elem_max_) nb_face_elem_max_=nbf;
471 for (int f=PolyhedronIndex[ele]; f<PolyhedronIndex[ele+1]; f++)

473// Cerr<<" ici "<<ele << " " <<f <<" "<<FacesIndex(f+1)-FacesIndex(f)<<finl;

476 for (int s=FacesIndex[f]; s<FacesIndex[f+1]; s++)
477 {
478 prov.add_if_not(Nodes[s]);
479 }
480 }
481 int nbsom=prov.size();
482 if (nbsom>nb_som_elem_max_) nb_som_elem_max_=nbsom;
483 }
484 Cerr<<" Polyhedron information nb_som_elem_max "<< nb_som_elem_max_<<" nb_som_face_max "<<nb_som_face_max_<<" nb_face_elem_max "<<nb_face_elem_max_<<finl;
485 les_elems.resize(nelem,nb_som_elem_max_);

488 for (int ele=0; ele<nelem; ele++)

502 while (perm)

504 perm=0;
505 for (int i=0; i<nbsom-1; i++)
506 if (prov[i]>prov[i+1])
507 {
508 perm=1;

510 prov[i]=prov[i+1];

512 }
513 }
514 for (int s=0; s<nbsom; s++) les_elems(ele,s)=prov[s];
515 }
516 FacesIndex_=FacesIndex;
517 PolyhedronIndex_=PolyhedronIndex;
518// determination de Nodes_...
519 Nodes_=Nodes;

523 {

527 {

530 if (les_elems(ele,sl)==somm_glob)
531 {
532 Nodes_[s]=sl;
533 break;
534 }
535 }
536 }
537 }
538 }
539 assert(min_array(Nodes_)>-1);
540}

578void Polyedre::ajouter_elements(const Elem_geom_base& type_elem, const IntTab& new_elems,IntTab& les_elems)

610
611 Nodes_.set_smart_resize(1);
612 int old_nodes_index= Nodes_.size_array();
613 Nodes_.resize_array(old_nodes_index+nb_new_elem*nb_face_new_elem*nb_som_face_new_elem);
614
615 int new_s=0;
616 for (int el=0; el<nb_new_elem; el++)
617 {
618 PolyhedronIndex_[nb_old_elem+el+1]=PolyhedronIndex_[nb_old_elem+el]+nb_face_new_elem;
619 for (int f=0; f<nb_face_new_elem; f++)

621 int nb_som_face_this_elem=0;
622 for (int s=0; s<nb_som_face_new_elem; s++)

624 {
625 Nodes_[old_nodes_index+new_s++]=faces_som_local(f,s);
626 nb_som_face_this_elem++;
627 }
628 if (nb_som_face_this_elem==4)
629 {
630// on inverse 3 et 4 en cas de quadrangle
631 int last=old_nodes_index+new_s-1;
632 swap(Nodes_[last],Nodes_[last-1]);
633 }
634 FacesIndex_[old_faces_index+(el*nb_face_new_elem)+f]=
635 FacesIndex_[old_faces_index+(el*nb_face_new_elem)+f-1]+nb_som_face_this_elem;
636
637 }
638 }
639 Nodes_.resize_array(old_nodes_index+new_s);
