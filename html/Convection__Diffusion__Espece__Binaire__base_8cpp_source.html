59int Convection_Diffusion_Espece_Binaire_base::lire_motcle_non_standard(const Motcle& mot, Entree& is)
60{
61 if (mot=="diffusion")
62 {
63 /*
64 * The diffusive term is div(rho*D*grad(Y))
65 * recall that mu_sur_Sc = rho*D and nu_sur_Sc = D
66 */
67
68 Cerr << "Reading and typing of the diffusion operator : " << finl;
69 terme_diffusif.associer_diffusivite(diffusivite_pour_transport());
70 ref_cast_non_const(Champ_base,terme_diffusif.diffusivite()).nommer("mu_sur_Schmidt");
71 is >> terme_diffusif;
72 terme_diffusif.associer_diffusivite_pour_pas_de_temps(diffusivite_pour_pas_de_temps());
73 return 1;
74 }
75 else
76 return Convection_Diffusion_Espece_Fluide_Dilatable_base::lire_motcle_non_standard(mot,is);
77
78}
