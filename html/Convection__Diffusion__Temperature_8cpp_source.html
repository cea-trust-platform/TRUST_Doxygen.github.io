497DoubleTab& Convection_Diffusion_Temperature::derivee_en_temps_inco(DoubleTab& derivee)
498{
499 if (!is_penalized)
500 {
501 return derivee_en_temps_inco_eq_base(derivee);
502 }
503 else
504 {
505 const double rhoCp = le_fluide->capacite_calorifique().valeurs()(0, 0) * le_fluide->masse_volumique().valeurs()(0, 0);
506// Specific code if temperature equation is penalized
507 derivee=inconnue().valeurs();
508// Mise en place d'une methode de mise en place d'une zone fantome
509 mise_en_place_zone_fantome(derivee);
510 DoubleTab& inc=inconnue().valeurs();
511 inc = derivee;
512
513 DoubleTrav secmem(derivee);
514
515 if (nombre_d_operateurs()>1)
516 {
517 derivee_en_temps_conv(secmem, derivee);
518 secmem *= rhoCp;
519 }
520
521// Transport des ibcs et de la couche limite
522 DoubleTrav secmem_conv_vr(derivee);
523// on advecte la variable avec la vitesse ibc
524 transport_ibc(secmem_conv_vr, derivee);
525// on retranche cette advection
526 secmem -= secmem_conv_vr;
527
528 DoubleTab derivee_conv;
529 derivee_conv.copy(secmem, Array_base::COPY_INIT);
530
531 les_sources.ajouter(secmem);

535

545 Equation_base::Gradient_conjugue_diff_impl(secmem, derivee);
546
547// penalisation de la temperature
548 penalisation_L2(derivee);
549 return derivee;
550 }

553DoubleTab& Convection_Diffusion_Temperature::derivee_en_temps_inco_eq_base(DoubleTab& derivee)
554{
555 derivee=0;
556 DoubleTrav secmem(derivee);
557
558 const double rhoCp = le_fluide->capacite_calorifique().valeurs()(0, 0) * le_fluide->masse_volumique().valeurs()(0, 0);
559
560// secmem = sum(operators) + sources + equation specific terms
561 if (schema_temps().diffusion_implicite())
562 {
563// Add convection operator only if equation has one
564 derivee = inconnue().valeurs();
565 if (nombre_d_operateurs()>1)
566 {
567 derivee_en_temps_conv(secmem, derivee);
568 secmem *= rhoCp;
569 }
570 }
571 else
572 {
573// Add all explicit operators
574 for(int i=0; i<nombre_d_operateurs(); i++)
575 if(operateur(i).l_op_base().get_decal_temps()!=1)
576 {
577 DoubleTrav secmem_tmp(secmem);
578 operateur(i).ajouter(secmem_tmp);
579 if (i == 1) secmem_tmp *= rhoCp;
580 secmem += secmem_tmp;
581 }
582 }
583 les_sources.ajouter(secmem);
584
585 if (calculate_time_derivative())
586 {
587// Store dI/dt(n) = M-1 secmem :
588 derivee_en_temps().valeurs()=secmem;
589 solveur_masse.appliquer(derivee_en_temps().valeurs());
590 schema_temps().modifier_second_membre((*this),secmem); //!< Change secmem for some schemes (eg: Adams_Bashforth)
591 }
592
593 corriger_derivee_expl(secmem); //!< Add specific term for an equation (eg: -gradP for Navier Stokes)
594
595 if (implicite_==0)
596 {
597 solveur_masse.appliquer(secmem); //!< M-1 * secmem
598 if (schema_temps().diffusion_implicite())
599 {
600// Solve: (1/dt + M-1*A)*dI = M-1 * secmem
601// where A is the diffusion
602 Equation_base::Gradient_conjugue_diff_impl(secmem, derivee);
603 }
604 else
605 {
606 derivee = secmem;
607 derivee.echange_espace_virtuel();
608 }
609 corriger_derivee_impl(derivee); //!< Solve specific implicit term for an equation (eg: pressure for Navier Stokes)
610 }
611 else if (implicite_>0)
612 {
613// TRUST support notices that this part has never been covered...
614// implicite
615// M dU/dt + AU* = f -BUn;
616// U* = Un+dt dU/dt
617// (M/dt + A) U* = f -BUn + M/dt Un
618//
619 double dt=schema_temps().pas_de_temps();
620 for(int i=0; i<nombre_d_operateurs(); i++)
621 {
622// boucle sur les operateurs
623 Operateur_base& op=operateur(i).l_op_base();
624 if(!op.get_matrice().non_nul())

1202{

1213// calcul de T_voisinage pour tous les elemnts

1215 const IntTab& elem_faces = zone_vf.elem_faces();
1216 const IntTab& faces_elem = zone_vf.face_voisins();
1217 const int nb_faces_elem = elem_faces.dimension(1);
1218 const DoubleTab& inc=inconnue().valeurs();
1219// inconnue doit etre scalaire
1220 assert(inc.line_size() == 1);
1221 DoubleTrav t_voisinage(inc);
1222 DoubleTrav u_voisinage(u);
1223 const int nb_elem = u.dimension_tot(0);
1224 for (int i_elem = 0; i_elem < nb_elem; i_elem++)
1225 {
1226 double somme = 0.;
1227 double somme_deriv = 0.;
1228 double coeff = 0.;
1229
1230 if( indic_pena_global(i_elem) != 0)
1231 {
1232// boucle sur les faces de l element
1233 for (int i_face = 0; i_face < nb_faces_elem; i_face++)
1234 {
1235 const int face = elem_faces(i_elem, i_face);
1236 const int voisin = faces_elem(face, 0) + faces_elem(face, 1) - i_elem;
1237 if (voisin >= 0)
1238 {
1239 if (indic_pena_global(voisin) == 0)
1240 {
1241 somme += inc(voisin);
1242 somme_deriv += u(voisin);
1243 coeff += 1.;
1244 }
1245 }
1246 }
1247 }
1248
1249 if (coeff > 0.)
1250 {
1251 t_voisinage(i_elem) = somme / coeff;
1252 u_voisinage(i_elem) = somme_deriv / coeff;
1253 }
1254 else
1255 {
1256// cas pas de voisin dans le fluide
1257 t_voisinage(i_elem) = inc(i_elem);
1258 u_voisinage(i_elem) = u(i_elem);
1259 }
1260 }
1261 t_voisinage.echange_espace_virtuel();
1262 u_voisinage.echange_espace_virtuel();
1263
1264// determination du champ global de penalisation L2
1265 DoubleTrav pena_glob(inc); //initialise a zero
1266 DoubleTrav tab_(inc); //initialise a zero
1267 DoubleTrav denom(inc); //initialise a zero
1268
1269// boucle ibc
1270 for ( int i = 0 ; i < ref_penalisation_L2_FTD.size() ; ++i)
1271 {
1272 Transport_Interfaces_base& nom_eq = ref_cast(Transport_Interfaces_base,ref_penalisation_L2_FTD[i].valeur());
1273 const DoubleTab& tab = tab_penalisation_L2_FTD[i];
1274// verification dimensions tab t obstacle = nb dim de l inconnu
1275 if ( tab.dimension(0) != inc.line_size() )
1276 {
1277 Cerr << " penalisation_L2: Les champs de temperature impose et calcule n'ont pas les memes dimensions" <<finl;
1278 Cerr << " Dimension du champ de temperature impose "<< tab.dimension(0) << finl;
1279 Cerr << " Dimension du champ de temperature calcule par TRUST "<<inc.line_size() <<finl;
1280 Cerr << " Nombre d'elements stockes au dans l'iconnue " <<inc.size() <<finl;
1281 Process::exit();
1282 }
1283 const DoubleTab& indicatrice = nom_eq.get_update_indicatrice().valeurs();
1284
1285 DoubleTrav pena_loc(indicatrice);
1286// Determination du terme de penalisation pour l ibc courante
1287// boucle sur les elements
1288 for (int j = 0; j<nb_elem ; ++j)
1289 {
1290// calcul de la fonction characteristique Ksi de l ibc courante -> pena_loc
1291 if (indicatrice(j) > 0.)
1292 {
1293 double coeff_p=1.;
1294 pena_loc(j) = coeff_p;
1295// calcul du denominateur du terme de penalisation ( Sigma_ibc Ksi_ibc(j) )
1296 denom(j) += coeff_p;
1297
1298// determination valeur a imposer T_ref
1299 double tref_j = 0.;
1300 switch (choix_pena)
1301 {
1302 case 0 :
1303// approximation de T_ref par T_ref = T_imp
1304 tref_j = tab(0);
1305 break;
1306 case 1 :
1307// approximation de T_ref par T_ref = T+dt*derivee
1308 tref_j = inc(j) + dt*u_old(j);
1309 break;
1310 case 2 :
1311// approximation de T_imp par T_ref= tau*T_imp + (1-tau)*T_voisin
1312// approximation de T_voisin par T_voisin= 1/n * somme(1,n)(T_environnant+dt*derivee_environnant)
1313 tref_j = (indicatrice(j)*tab(0)+(1.-indicatrice(j))*(t_voisinage(j)+dt*u_voisinage(j)));
1314 break;
1315 case 3 :
1316// approximation de T_imp par T_ref= [(1-tau)*T_voisin + (1/2)*T_imp] / (3/2 -tau)
1317// approximation de T_voisin par T_voisin= 1/n * somme(1,n)(T_environnant+dt*derivee_environnant)
1318 tref_j = ((1.-indicatrice(j))*(t_voisinage(j)+dt*u_voisinage(j))+0.5*tab(0))/(1.5-indicatrice(j));
1319 break;
1320 case 4 :
1321// approximation de T_imp par T_ref= T_imp si tau=1 sinon T_ref=T_voisin
1322 if (indicatrice(j) == 1.)
1323 {
1324 tref_j = tab(0);
1325 }
1326 else
1327 {
1328 tref_j = (t_voisinage(j)+dt*u_voisinage(j));
1329 }
1330 break;
1331 default:
1332 Cerr << "Penalisation_L2: choix_pena invalide "<< choix_pena <<finl;
1333 exit();
1334 break;
1335 }
1336// sauvegarde de Sigma_ibc Ksi_ibc(j) Tref_ibc pour post-traitement
1337 tab_(j) += tref_j;
1338// penalisation_global = Sigma_ibc Ksi_ibc(j)*(Tref_ibc - inc) / eta;
1339 pena_loc(j) *= (tref_j-inc(j));
1340 pena_loc(j) /= eta;
1341 pena_glob(j) += pena_loc(j);
1342 }
1343 }
1344 }
1345
1346// Realisation de la penalisation L2 de la derivee en temps: (dT_t_* + pena_glob) / ( 1 + (dt*Sigma_ibc Ksi_ibc(j)) / eta )
1347// Tref est la moyenne arithmetique de Tref_j lorsque 0<eta<<1
1348 for (int j = 0 ; j<nb_elem ; ++j)
1349 {
1350 u(j) = u_old(j) + pena_glob(j);
1351 u(j) /= (1. + denom(j)*dt/eta);
1352 }
1353 u.echange_espace_virtuel();
1354
1355// Sauvegarde des flux
1356 ecrire_fichier_pena_th(u_old,u,tab_,denom);
1357
1358// Debog::verifier("Convection_Diffusion_Temperature::penalisation_L2 u ",u);
1359 return u;
1360}
