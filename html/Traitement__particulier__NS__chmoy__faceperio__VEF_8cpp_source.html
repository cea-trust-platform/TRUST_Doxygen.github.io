68void Traitement_particulier_NS_chmoy_faceperio_VEF::init_calcul_stats(void)
69{
70 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
71 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF, zdisbase);
72 const Zone_Cl_VEF& zone_Cl_VEF = ref_cast(Zone_Cl_VEF,mon_equation->zone_Cl_dis().valeur() );
73 const DoubleTab& xv = zone_VEF.xv(); //!< centre de gravite des faces Imprime dans un fichier les faces periodiques (numero+coordonnees du centre)

75 int nb_front=zone_VEF.nb_front_Cl();
76 for (int n_bord=0; n_bord<nb_front; n_bord++)

102void Traitement_particulier_NS_chmoy_faceperio_VEF::calcul_chmoy_faceperio(double un_temps_deb, double temps, double dt)
103{
104 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
105 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF, zdisbase);
106 const Zone_Cl_VEF& zone_Cl_VEF = ref_cast(Zone_Cl_VEF,mon_equation->zone_Cl_dis().valeur() );
107 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
108
109// Calcul de la moyenne temporelle de la vitesse sur les frontieres periodiques:
110// moy(U)=1/(temps-temps_deb)*Integrale(t=temps_deb a temps)(U(t)*dt)
111 chmoy_faceperio*=(temps-un_temps_deb-dt);
112 int test_proc=0;
113 int nb_front=zone_VEF.nb_front_Cl();
114 for (int n_bord=0; n_bord<nb_front; n_bord++)
115 {
116 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
117 if (sub_type(Periodique,la_cl.valeur()))
118 {
119 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
120 int num1 = le_bord.num_premiere_face();
121 int num2 = num1 + le_bord.nb_faces();
122 for (int num_face=num1; num_face<num2; num_face++)
123 {
124 for(int i=0; i<Objet_U::dimension; i++)
125 chmoy_faceperio(num_face,i)+=dt*vitesse(num_face,i);
126 test_proc=1;
127 }
128 }
129 }
130
131 chmoy_faceperio/=(temps-un_temps_deb);
132
133 if (test_proc)
134 {
135 EcrFicPartage fic("chmoy_face_perio");
136 fic.setf(ios::scientific);
137 fic.precision(format_precision_geom);
138 for (int n_bord=0; n_bord<nb_front; n_bord++)
139 {
140 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
141 if (sub_type(Periodique,la_cl.valeur()))
142 {
143 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
144 int num1 = le_bord.num_premiere_face();
145 int num2 = num1 + le_bord.nb_faces();
146 int nb_faces_bord = mp_sum(le_bord.nb_faces());
147 if (Process::je_suis_maitre()) fic<<nb_faces_bord<<finl;
148 for (int num_face=num1; num_face<num2; num_face++)
149 fic<<num_face<<" "<<chmoy_faceperio(num_face,0)<<" "<<chmoy_faceperio(num_face,1)<<" "<<chmoy_faceperio(num_face,2)<<finl;
150 }
151 }
152 fic.syncfile();
153 fic.close();
154 }
