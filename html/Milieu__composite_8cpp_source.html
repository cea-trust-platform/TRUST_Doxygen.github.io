29Entree& Milieu_composite::readOn(Entree& is)
30{
31 int i = 0;
32 std::vector<std::pair<std::string, int>> especes;
33 Nom mot;
34 is >> mot;
35 if (mot != "{") Cerr << "Milieu_composite : { expected instead of " << mot << finl, Process::exit();
36 for (is >> mot; mot != "}"; is >> mot)
37 {
38 Cerr << "Milieu_composite : ajout de " << mot << " ... ";
39 if (Interprete::objet_existant(mot))
40 {
41 Fluide_base& fluide = ref_cast(Fluide_base, Interprete::objet(mot));
42 fluide.set_id_composite(i++);
43 fluides.add(fluide);
44 especes.push_back(check_fluid_name(fluide.le_nom()));
45 Cerr << "ok!" << finl;
46 }
47 else Process::exit("not found!");
48 }
49
50// interfaces
51 int N = fluides.size();
52 for (int n = 0; n < N; n++)
53 {
54 std::vector<Interface_base *> inter;
55 for (int m = 0; m < N; m++)
56 {
57 const std::string espn = especes[n].first, espm = especes[m].first;
58 const int pn = especes[n].second, pm = especes[m].second;
59 Nom mot_inter = Nom((std::string("interface_") + especes[n].first + "_" + especes[m].first)), //!< "Interface_[...]_[...]"
60 mot_sat = Nom((std::string("saturation_") + especes[n].first)); //"Saturation_[...]"
61 if ( pn != pm && (Interprete::objet_existant(mot_inter) || (espn == espm &&Interprete::objet_existant(mot_sat))))
62 {
63 Cerr << "Interface between fluid " << n << " : " << fluides[n].le_nom() << " and " << m << " : " << fluides[m].le_nom() << finl;
64 phases_melange[especes[n].first].insert(n), phases_melange[especes[n].first].insert(m);
65 inter.push_back(&ref_cast(Interface_base, Interprete::objet(Interprete::objet_existant(mot_sat) ? mot_sat : mot_inter)));
66 const Saturation_base *sat = sub_type(Saturation_base, *inter.back()) ? &ref_cast(Saturation_base, *inter.back()) : NULL;
67 if (sat && sat->get_Pref() > 0) //!< pour loi en e = e0 + cp * (T - T0)
68 {
69 const double hn = pn ? sat->Hvs(sat->get_Pref()) : sat->Hls(sat->get_Pref()),
70 hm = pm ? sat->Hvs(sat->get_Pref()) : sat->Hls(sat->get_Pref()),
71 T0 = sat->Tsat(sat->get_Pref());
72 fluides[n].set_h0_T0(hn, T0), fluides[m].set_h0_T0(hm, T0);
73 }
74 }
75 else inter.push_back(NULL);
76 }
77 tab_interface.push_back(inter);
78 }
79 return is;
80}

118{

138void Milieu_composite::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
139{
140 Cerr << "Composite medium discretization." << finl;
141
142 const int N = fluides.size(), nc = pb.equation(0).inconnue()->nb_valeurs_temporelles();
143 const Zone_dis_base& zone_dis = pb.equation(0).zone_dis();
144 const double temps = pb.schema_temps().temps_courant();
145
146// Fluide_base::discretiser(pb, dis);
147 for (int n = 0; n < N; n++)
148 fluides[n].discretiser(pb, dis);
149
150 /* masse volumique, energie interne, enthalpie : champ_inc */
151 Champ_Inc rho_inc, ei_inc, h_inc;
152 dis.discretiser_champ("champ_elem", zone_dis, "masse_volumique", "kg/m^3", N, nc, temps, rho_inc);
153 dis.discretiser_champ("champ_elem", zone_dis, "energie_interne", "J/m^3", N, nc, temps, ei_inc);
154 dis.discretiser_champ("champ_elem", zone_dis, "enthalpie", "J/m^3", N, nc, temps, h_inc);
155 rho = rho_inc, e_int = ei_inc, h = h_inc;
156
157 /* autres champs : champ_fonc */
158 dis.discretiser_champ("champ_elem", zone_dis, "viscosite_dynamique", "kg/m/s", N, temps, mu);
159 dis.discretiser_champ("champ_elem", zone_dis, "viscosite_cinematique", "m2/s", N, temps, nu);
160 dis.discretiser_champ("champ_elem", zone_dis, "diffusivite", "m2/s", N, temps, alpha);
161 dis.discretiser_champ("champ_elem", zone_dis, "conductivite", "W/m/K", N, temps, lambda);
162 dis.discretiser_champ("champ_elem", zone_dis, "capacite_calorifique", "J/kg/K", N, temps, Cp);
163
164 /* champs de quantites melange */
165 for (auto &&kv : phases_melange)
166 {
167 dis.discretiser_champ("champ_elem", zone_dis, Nom("masse_volumique_") + kv.first + "_melange", "kg/m^3", 1, temps, rho_m);
168 dis.discretiser_champ("champ_elem", zone_dis, Nom("enthalpie_") + kv.first + "_melange", "J/m^3", 1, temps, h_m);
169 champs_compris_.ajoute_champ(rho_m);
170 champs_compris_.ajoute_champ(h_m);
171 }
172
173 champs_compris_.ajoute_champ(rho);
174 champs_compris_.ajoute_champ(mu.valeur());
175 champs_compris_.ajoute_champ(nu.valeur());
176 champs_compris_.ajoute_champ(alpha.valeur());
177 champs_compris_.ajoute_champ(lambda.valeur());
178 champs_compris_.ajoute_champ(Cp.valeur());
179 champs_compris_.ajoute_champ(e_int);
180 champs_compris_.ajoute_champ(h);
181}

202void Milieu_composite::mettre_a_jour_tabs()
203{
204 const int N = fluides.size();
205
206 /* viscosite dynamique */
207 {
208 DoubleTab& tab = mu.valeurs();
209 const int Nl = mu.valeurs().dimension_tot(0);
210 for (int n = 0; n < N; n++)
211 {
212 const Champ_base& ch_n = fluides[n].viscosite_dynamique();
213 const int cch = sub_type(Champ_Uniforme, ch_n);
214 for (int i = 0; i < Nl; i++) tab(i, n) = ch_n.valeurs()(!cch * i, 0);
215 }
216 }
217
218 /* viscosite cinematique */
219 {
220 DoubleTab& tab = nu.valeurs();
221 const int Nl = nu.valeurs().dimension_tot(0);
222 for (int n = 0; n < N; n++)
223 {
224 const Champ_base& ch_n = fluides[n].viscosite_cinematique();
225 const int cch = sub_type(Champ_Uniforme, ch_n);
226 for (int i = 0; i < Nl; i++) tab(i, n) = ch_n.valeurs()(!cch * i, 0);
227 }
228 }
229
230 /* diffusivite */
231 {
232 DoubleTab& tab = alpha.valeurs();
233 const int Nl = alpha.valeurs().dimension_tot(0);
234 for (int n = 0; n < N; n++)
235 {
236 const Champ_base& ch_n = fluides[n].diffusivite();
237 const int cch = sub_type(Champ_Uniforme, ch_n);
238 for (int i = 0; i < Nl; i++) tab(i, n) = ch_n.valeurs()(!cch * i, 0);
239 }
240 }
241
242 /* conductivite */
243 {
244 DoubleTab& tab = lambda.valeurs();
245 const int Nl = lambda.valeurs().dimension_tot(0);
246 for (int n = 0; n < N; n++)
247 {
248 const Champ_base& ch_n = fluides[n].conductivite();
249 const int cch = sub_type(Champ_Uniforme, ch_n);
250 for (int i = 0; i < Nl; i++) tab(i, n) = ch_n.valeurs()(!cch * i, 0);
251 }
252 }
253
254 /* capacite calorifique */
255 {
256 DoubleTab& tab = Cp.valeurs();
257 const int Nl = Cp.valeurs().dimension_tot(0);
258 for (int n = 0; n < N; n++)
259 {
260 const Champ_base& ch_n = fluides[n].capacite_calorifique();
261 const int cch = sub_type(Champ_Uniforme, ch_n);
262 for (int i = 0; i < Nl; i++) tab(i, n) = ch_n.valeurs()(!cch * i, 0);
263 }
264 }
265
266 if (rho_m.non_nul() && h_m.non_nul())
267 {
268 DoubleTab& trm = rho_m.valeurs(), &thm = h_m.valeurs();
269 trm = 0, thm = 0;
270
271 const Equation_base& eqn = equation("alpha");
272 const DoubleTab& a = eqn.inconnue().valeurs(),
273 &r = rho.valeurs(), &ent = h.valeurs();
274 const Nom ph = (rho_m.le_nom().getSuffix("masse_volumique_")).getPrefix("_melange");
275 const int Nl = rho_m.valeurs().dimension_tot(0);
276// masse volumique melange
277 for (auto n : phases_melange[ph.getString()])
278 for (int i = 0; i < Nl; i++) trm(i) += a(i, n) * r(i, n);
279// enthalpie melange
280 for (auto n : phases_melange[ph.getString()])
281 for (int i = 0; i < Nl; i++) thm(i) += a(i, n) * r(i, n) * ent(i, n);
282 for (int i = 0; i < Nl; i++) thm(i) /= trm(i);
283 }
284}

301void Milieu_composite::calculer_masse_volumique(const Objet_U& obj, DoubleTab& val, DoubleTab& bval, tabs_t& deriv)
302{
303 const Milieu_composite& mil = ref_cast(Milieu_composite, obj);
304 const Zone_VF& zvf = ref_cast(Zone_VF, mil.equation_.begin()->second->zone_dis().valeur());
305 int i, Ni = val.dimension_tot(0), Nb = bval.dimension_tot(0), n, N = mil.fluides.size();
306 std::vector<const DoubleTab *> split(N);
307 for (n = 0; n < N; n++) split[n] = &mil.fluides[n].masse_volumique().valeurs();
308 for (i = 0; i < Ni; i++)
309 for (n = 0; n < N; n++) val(i, n) = (*split[n])(i * (split[n]->dimension(0) > 1), 0);
310
311 std::vector<DoubleTab> bsplit(N);
312 for (n = 0; n < N; n++)
313 if (mil.fluides[n].masse_volumique()->a_une_zone_dis_base())
314 bsplit[n] = mil.fluides[n].masse_volumique()->valeur_aux_bords();
315 else bsplit[n].resize(bval.dimension_tot(0), 1), mil.fluides[n].masse_volumique()->valeur_aux(zvf.xv_bord(), bsplit[n]);
316 for (i = 0; i < Nb; i++)
317 for (n = 0; n < N; n++) bval(i, n) = bsplit[n](i * (split[n]->dimension(0) > 1), 0);
318
319 /* derivees */
320 std::vector<const tabs_t *> split_der(N);
321 for (n = 0; n < N; n++) split_der[n] = sub_type(Champ_Inc_base, mil.fluides[n].masse_volumique().valeur()) ? &ref_cast(Champ_Inc_base, mil.fluides[n].masse_volumique().valeur()).derivees() : NULL;
322 std::set<std::string> noms_der;
323 for (n = 0; n < N; n++)
324 if (split_der[n])
325 for (auto &&n_d : *split_der[n]) noms_der.insert(n_d.first);
326 for (auto &&nom : noms_der)
327 {
328 for (n = 0; n < N; n++) split[n] = split_der[n] && split_der[n]->count(nom) ? &split_der[n]->at(nom) : NULL;
329 DoubleTab& der = deriv[nom];
330 for (der.resize(Ni, N), i = 0; i < Ni; i++)
331 for (n = 0; n < N; n++) der(i, n) = split[n] ? (*split[n])(i * (split[n]->dimension(0) > 1)) : 0;
332 }
333}
