53DoubleTab& Masse_VEF_P1NC::appliquer_impl(DoubleTab& sm) const
54{
55 const Zone_Cl_VEF& zone_Cl_VEF = ref_cast(Zone_Cl_VEF,la_zone_Cl_VEF.valeur());
56 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
57 const DoubleVect& volumes_entrelaces = zone_VEF.volumes_entrelaces();
58 const DoubleVect& volumes_entrelaces_Cl = zone_Cl_VEF.volumes_entrelaces_Cl();
59 const IntTab& face_voisins = zone_VEF.face_voisins();
60 const DoubleVect& porosite_face = zone_VEF.porosite_face();
61 int nfa = zone_VEF.nb_faces();
62 int num_std = zone_VEF.premiere_face_std();
63 int num_int = zone_VEF.premiere_face_int();
64 int face,comp,elem;
65 int nbcomp = sm.line_size();
66 if (nfa != sm.dimension(0))
67 {
68 Cerr << "erreur dans Masse_VEF_P1NC : ";
69 Cerr << "nombre de faces : " << nfa
70 << " taille du second membre : " << sm.dimension(0) << finl;
71 exit();
72 }
73
74// On traite les faces standard qui ne portent pas de conditions aux limites
75
76 for (face=num_std; face<nfa; face++)
77 {
78// elem1 = face_voisins(face,0);
79// elem2 = face_voisins(face,1);
80 for (comp=0; comp<nbcomp; comp++)
81 sm(face,comp) /= (volumes_entrelaces(face)*porosite_face(face)) ;
82 }
83// On traite les faces non standard
84// les faces des bord sont des faces non standard susceptibles de porter des C.L
85// les faces internes non standard ne portent pas de C.L
86
87// On traite les conditions aux limites
88
89 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
90 {
91
92 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
93 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
94 int num1 = le_bord.num_premiere_face();
95 int num2 = num1 + le_bord.nb_faces();
96
97 if ( (sub_type(Dirichlet,la_cl.valeur()))
98 ||
99 (sub_type(Dirichlet_homogene,la_cl.valeur()))
100 )
101// Pour les faces de Dirichlet on met sm a 0
102 for (face=num1; face<num2; face++)
103 for (comp=0; comp<nbcomp; comp++)
104 sm(face,comp) =0;
105 else
106
107 if ((sub_type(Symetrie,la_cl.valeur()))&&(zone_Cl_VEF.equation().inconnue()->nature_du_champ()==vectoriel))
108 {
109 const DoubleTab& normales = zone_VEF.face_normales();
110 for (face=num1; face<num2; face++)
111 {
112 double psc=0;
113 double surf=0;
114 for (comp=0; comp<nbcomp; comp++)
115 {
116 psc+=sm(face,comp)*normales(face,comp);
117 surf+=normales(face,comp)*normales(face,comp);
118 }
119 psc/=surf;
120 for(comp=0; comp<dimension; comp++)
121 {
122 sm(face,comp)-=psc*normales(face,comp);
123 sm(face,comp) /= (volumes_entrelaces_Cl(face)*
124 porosite_face(face));
125 }
126 }
127 }
128
129 else
130
131 for (face=num1; face<num2; face++)
132 {
133 elem = face_voisins(face,0);
134 if ( elem == -1 )
135 {
136 elem = face_voisins(face,1);
137 }
138 for (comp=0; comp<nbcomp; comp++)
139 sm(face,comp) /= (volumes_entrelaces_Cl(face)*porosite_face(face));
140 }
141 }
142
143// On traite les faces internes non standard
144 for (face=num_int; face<num_std; face++)
145 {
146// elem1 = face_voisins(face,0);
147// elem2 = face_voisins(face,1);
148 for (comp=0; comp<nbcomp; comp++)
149 sm(face,comp) /= (volumes_entrelaces_Cl(face)*porosite_face(face)) ;
150
151 }
152// sm.echange_espace_virtuel();
153// Debog::verifier("Masse_VEF_P1NC::appliquer, sm=",sm);
154 return sm;
155}

170Matrice_Base& Masse_VEF_P1NC::ajouter_masse(double dt, Matrice_Base& matrice, int penalisation) const
171{
172 if (penalisation||(la_zone_Cl_VEF.valeur().equation().inconnue()->nature_du_champ()!=vectoriel))
173 return Solveur_Masse_base::ajouter_masse(dt,matrice,penalisation);
174// Sinon on modifie temporairement la nature du champ pour que appliquer_impl ne projette pas sur n.
175 Champ_Inc_base& inco=ref_cast_non_const( Champ_Inc_base,la_zone_Cl_VEF.valeur().equation().inconnue().valeur());
176 inco.fixer_nature_du_champ(multi_scalaire);
177 Solveur_Masse_base::ajouter_masse(dt,matrice,penalisation);
178 inco.fixer_nature_du_champ(vectoriel);
179 return matrice;
