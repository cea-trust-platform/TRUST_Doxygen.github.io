295void Format_Post_base::postraiter_debug_valeurs_un_instant(const Nom& nom_fich,
296 const Domaine& dom,
297 const Zone_dis_base& zone_dis,
298 double temps_champ, double temps_courant,
299 const Nom& id_champ_post,const Nature_du_champ& nature_champ,
300 const Noms& noms_compos,const Noms& unites,const Motcle& loc_post,
301 const DoubleTab& data,
302 int ncomp,
303 int is_axi,
304 int champ,int stat,double dt_ch,double dt_stat,int reprise,
305 int est_le_premier_post, int est_le_dernier_post,
306 double t_init,double temps_post)
307{
308
309 const Nom& id_domaine = dom.le_nom();
310 const int nb_compo = noms_compos.size();
311 DoubleTab valeurs_tmp;
312 int dim0 = data.dimension(0);
313 int nb_dim_val = 1;
314
315 if (ncomp==-1)
316 {
317 if (data.nb_dim()>1)
318 {
319 nb_dim_val = data.dimension(1);
320 valeurs_tmp.resize(dim0,nb_dim_val);
321 for (int i=0; i<dim0; i++)
322 for (int j=0; j<nb_dim_val; j++)
323 valeurs_tmp(i,j) = data(i,j);
324 }
325 else
326 {
327 valeurs_tmp.resize(dim0,nb_dim_val);
328 for (int i=0; i<dim0; i++)
329 valeurs_tmp(i,0) = data(i);
330 }
331 }
332 else
333 {
334 valeurs_tmp.resize(dim0);
335 if (data.nb_dim()>1)
336 {
337 int nb_comp = data.dimension(1);
338 for (int i=0; i<dim0; i++)
339 for (int j=0; j<nb_comp; j++)
340 if (j==ncomp)
341 valeurs_tmp(i) = data(i,j);
342 }
343 else
344 valeurs_tmp = data;
345 }
346
347 initialize_by_default(nom_fich);
348 test_coherence(champ,stat,dt_ch,dt_stat);
349 ecrire_entete(temps_courant,reprise,est_le_premier_post);
350 preparer_post(id_domaine,est_le_premier_post,reprise,t_init);
351 completer_post(dom,is_axi,nature_champ,nb_compo,noms_compos,loc_post,id_champ_post);
352 ecrire_domaine(dom,est_le_premier_post);
353
354 if (loc_post=="FACES")
355 {
356 const Zone_VF& zone_vf = ref_cast(Zone_VF, zone_dis);
357 const IntTab& faces_sommets = zone_vf.face_sommets();
358 const int nb_sommets = dom.nb_som();
359 const int nb_faces = faces_sommets.dimension(0);
360 ecrire_item_int("FACES", //!< Identifiant
361 dom.le_nom(),
362 dom.zone(0).le_nom(),
363 "FACES", //!< localisation
364 "SOMMETS", //!< reference
365 faces_sommets,
366 nb_sommets);
367
368 const IntTab& elem_faces = zone_vf.elem_faces();
369 ecrire_item_int("ELEM_FACES", //!< Identifiant
370 dom.le_nom(),
371 dom.zone(0).le_nom(),
372 "ELEMENTS", //!< localisation
373 "FACES", //!< reference
374 elem_faces,
375 nb_faces);
376
377 }
378
379 ecrire_temps(temps_courant);
380 init_ecriture(temps_courant,temps_post,est_le_premier_post,dom);
381 Nom nature("scalar");
382 if (nature_champ==vectoriel) nature="vector";
383 ecrire_champ(dom,unites,noms_compos,ncomp,temps_champ,temps_courant,id_champ_post,id_domaine,loc_post,nature,valeurs_tmp);
384 finir_ecriture(temps_courant);
385 finir(est_le_dernier_post);
386
387}
