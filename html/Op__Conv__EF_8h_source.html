135 DoubleTab transp_loc(nb_som_elem, nb_comp0);

157#define bij_(elem,i,j) (IS_GEN ? bij(elem,i,j) : bij_ptr[elem*dim_fois_nbn+i*const_dimension+j])

167 for (int i1 = 0; i1 < nb_som_elem; i1++)
168 {
169 int glob = elems(elem, i1);
170 for (int b = 0; b < const_dimension; b++)
171 G_e[b] += G(glob, b);
172 for (int a = 0; a < nb_comp; a++)
173 transp_loc(i1, a) = transporte_(glob, a);

175 G_e *= inv_nb_som_elem;

177 double vol_elem = volumes(elem);
178 double inv_vol_elem = 1. / vol_elem;
179 double pond2 = volumes_thilde(elem) * inv_vol_elem * inv_vol_elem;

187 double pond3 = f * dotproduct_array(G_e, G_e);
188 if (transport_rhou)

190 if (is_not_lambda_unif)
191 pond3 += lambda(elem);
192 else
193 pond3 += lambda(0, 0);
194 pond3 *= volumes_thilde(elem) * inv_vol_elem * pow(vol_elem, 0.3333333333333333);
195 pond3 *= hourglass;
196
197 for (int a = 0; a < nb_comp; a++)
198 {
199 double coef2d = 0.042 * pond3;
200 double coef3d = coef2d * 0.5;
201
202 double t0 = -coef2d * (transp_loc_(0,a)+transp_loc_(1,a)+transp_loc_(2,a)+transp_loc_(3,a)
203 +transp_loc_(4,a)+transp_loc_(5,a)+transp_loc_(6,a)+transp_loc_(7,a));
204
205 double t3d = 0;
206 for (int i = 0; i < 8; i++)
207 t3d += mcoef3d[i] * transp_loc_(i, a);
208 t3d *= coef3d;
209 double t3db = coef3d * (transp_loc_(0,a)-transp_loc_(1,a)+transp_loc_(3,a)-transp_loc_(2,a)
210 -transp_loc_(4,a)+transp_loc_(5,a)-transp_loc_(7,a)+transp_loc_(6,a));
211 if (!est_egal(t3d, t3db, 1e-6))
212 assert(0);
213
214 for (int i = 0; i < 8; i++)
215 resu_(elems(elem,i),a)-=mcoef3d[i]*t3d+t0+coef2d*4.*(transp_loc_(i,a)+transp_loc_(sommetoppose[i],a));
216 }
217 }
218
219 {
220 for (int yy = 0; yy < nb_som_elem; yy++)
221 ge_bij[yy] = 0;
222 if ((centre_impl == 0) || (btd_impl == 0))
223 {
224 for (int i1 = 0; i1 < nb_som_elem; i1++)

226 double cb = 0;
227 for (int b = 0; b < const_dimension; b++)
228 cb += G_e[b] * bij_(elem, i1, b);
229 ge_bij[i1] = cb;

231 for (int i1 = 0; i1 < nb_som_elem; i1++)
232 {
233 double pond = 0;
234 if (centre_impl == 0)
235 pond = IPhi_thilde(elem, i1) * inv_vol_elem;
236 int glob = elems(elem, i1);
237// pr=0;
238 for (int yy = 0; yy < nb_comp; yy++)
239 pr[yy] = 0;
240
241 double cbbtd = pond;
242 if ((btd_impl == 0) && (type_op == amont))
243 cbbtd += ge_bij[i1] * (fpond2);
244 for (int i2 = 0; i2 < nb_som_elem; i2++)
245 {
246 const double cbi2 = ge_bij[i2];
247 double coef = cbbtd * cbi2;
