63void Partitionneur_Sous_Domaine::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
64{
65 if (filename_ == "")
66 {
67 Cerr << "Error in Partitionneur_Sous_Domaine::construire_partition\n";

69 exit();

71 if (filename_ssz_ == "")
72 {
73 Cerr << "Error in Partitionneur_Sous_Domaine::construire_partition\n";
74 Cerr << " The subzone file name has not been initialized" << finl;
75 exit();
76 }
77
78 Cerr << "Reading of splitting file : " << filename_ << finl;
79 EFichier file;
80 file.ouvrir(filename_);
81 if (!file.good())
82 {
83 Cerr << "Error in Partitionneur_Sous_Domaine::construire_partition\n";
84 Cerr << " Failed to open file " << filename_ << finl;
85 exit();
86 }
87 IntVect elem_part_glob;
88 file >> elem_part_glob;
89 file >> nb_parts_tot;
90 file.close();
91
92 Cerr << "Reading of subzone file : " << filename_ssz_ << finl;
93 file.ouvrir(filename_ssz_);
94 if (!file.good())
95 {
96 Cerr << "Error in Partitionneur_Sous_Domaine::construire_partition\n";
97 Cerr << " Failed to open file " << filename_ssz_ << finl;
98 exit();
99 }
100 ArrOfInt elem_ssz;
101 file >> elem_ssz;
102 file.close();
103
104 /* remplissage de elem_part */
105 elem_part.resize(elem_ssz.size_array());
106 for (int i = 0; i < elem_ssz.size_array(); i++)
107 nb_parts_tot = std::max(nb_parts_tot, 1 + (elem_part[i] = elem_part_glob[elem_ssz[i]]));
108
109}
