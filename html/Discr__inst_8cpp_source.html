79//! Idem que VEF_discretisation::discretiser_champ(... , Champ_Inc)<br>Traitement commun aux champ_fonc et champ_don.<br>Cette methode est privee (passage d'un Objet_U pas propre vu<br>de l'exterieur ...)
80void Discr_inst::discretiser_champ_fonc_don(
81 const Motcle& directive, const Zone_dis_base& z,
82 Nature_du_champ nature,
83 const Noms& noms, const Noms& unites,
84 int nb_comp, double temps,
85 Objet_U& champ) const
86{
87// Deux pointeurs pour acceder facilement au champ_don ou au champ_fonc,
88// suivant le type de l'objet champ.
89 Champ_Fonc * champ_fonc = 0;
90 Champ_Don * champ_don = 0;
91 if (sub_type(Champ_Fonc, champ))
92 champ_fonc = & ref_cast(Champ_Fonc, champ);
93 else
94 champ_don = & ref_cast(Champ_Don, champ);
95
96 Motcles motcles(2);
97 motcles[0] = "champ_elem"; //!< Creer un champ aux elements (de type P0)
98 motcles[1] = "champ_sommets"; //!< Creer un champ aux sommets
99
100 Nom type;
101 int rang = motcles.search(directive);
102// int default_nb_comp = 1;
103 const Domaine& domaine=z.zone().domaine();
104 int nb_som_elem = domaine.zone(0).nb_som_elem();
105 int nb_ddl=-1;
106 switch(rang)
107 {
108 case 0:
109 type = "Champ_Fonc_P0_MED";
110 nb_ddl = domaine.zone(0).nb_elem();
111 break;
112 case 1:
113 type = "Champ_Fonc_P1_MED";
114 nb_ddl = domaine.nb_som();
115 if (((dimension==2) && (nb_som_elem==4)) || ((dimension==3) && (nb_som_elem==8)))
116 {
117 type = "Champ_Fonc_Q1_MED";
118 }
119 break;
120
121 default:
122 assert(rang < 0);
123
124 break;
125 }
126
127 if (directive == demande_description)
128 Cerr << "Discr_inst : " << motcles;
129
130// Si on n'a pas compris la directive (ou si c'est une demande_description)
131// alors on appelle l'ancetre :
132 if (rang < 0)
133 {
134 if (champ_fonc)
135 Discretisation_base::discretiser_champ(directive, z, nature, noms, unites,
136 nb_comp, temps, *champ_fonc);
137 else
138 Discretisation_base::discretiser_champ(directive, z, nature, noms, unites,
139 nb_comp, temps, *champ_don);
140
141
142 return;
143 }
144
145
146
147 if (champ_fonc)
148 {
149 Champ_Fonc& ch=*champ_fonc;
150 ch.typer(type);
151 ch->associer_zone_dis_base(z);
152 ch->fixer_nb_comp(nb_comp);
153 ch->fixer_nature_du_champ(nature);
154 ch->fixer_nb_valeurs_nodales(nb_ddl);
155 }
156 else
157 {
158 assert(0);
159 throw;
160 }
161
162
163}
