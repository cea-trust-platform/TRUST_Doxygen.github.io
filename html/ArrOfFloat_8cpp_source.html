47//! Operateur copie (deep copy)
48const ArrOfFloat& ArrOfFloat::operator=(const ArrOfFloat& array)
49{
50 if (&array == this) //!< Self assignment, why not...
51 return *this;
52 if (array.size_ > memory_size_)
53 {
54 if (data_)
55 delete [] data_;
56 data_ = new float[array.size_];
57 memory_size_ = array.size_;
58 }
59 if (array.size_ > 0)
60 memcpy(data_, array.data_, array.size_ * sizeof(float));
61 size_ = array.size_;
62 return *this;
63}

86//! Augmentation de taille du tableau (appele uniquement si l'espace<br>memoire est insuffisant et qu'il faut reallouer de la memoire).<br>Les anciennes donnees sont recopiees dans le nouveau tableau,<br>le contenu des elements nouveaux est indefini.
87void ArrOfFloat::mem_resize(int size)
88{
89 assert(size > size_);
90 int newsize = size * grow_factor_num_ / grow_factor_denom_;
91 float * newdata = new float[newsize];
92
93// Copier les anciennes donnees dans le nouveau tableau
94 if (size_ > 0)
95 memcpy(newdata, data_, size_ * sizeof(float));
96
97// Effacer l'ancien tableau
98 if (data_)
99 delete [] data_;
100
101 data_ = newdata;
102 memory_size_ = newsize;
103 size_ = size;
104}

169static True_int fonction_compare_arroffloat_ordonner(const void * data1, const void * data2)
170{
171 const float x = *(const float*)data1;
172 const float y = *(const float*)data2;
173 if (x < y)
174 return -1;
175 else if (x > y)
176 return 1;
177 else

212//! Copie les elements source[first_element_source + i]<br>dans les elements (*this)[first_element_dest + i] pour 0 <= i < nb_elements<br>Les autres elements de (*this) sont inchanges.
213/** \return ArrOfDouble\&<br>
214Signification: *this<br>
215Contraintes: */
216ArrOfFloat& ArrOfFloat::inject_array(const ArrOfFloat& source,
217 int nb_elements,
218 int first_element_dest,
219 int first_element_source)
220{
221 assert(&source != this);
222 assert(nb_elements >= -1);
223 assert(first_element_dest >= 0);
224 assert(first_element_source >= 0);
225
226 if (nb_elements < 0)
227 nb_elements = source.size_array();
228
229 assert(first_element_source + nb_elements <= source.size_array());
230 assert(first_element_dest + nb_elements <= size_array());
231
232 if (nb_elements > 0)
233 {
234 float * addr_dest = addr() + first_element_dest;
235 const float * addr_source = source.addr() + first_element_source;
236// PL: On utilise le memcpy car c'est VRAIMENT plus rapide (10% +vite sur RNR_G20)
237 memcpy(addr_dest , addr_source, nb_elements * sizeof(float));
238 /*
239 int i;
240 for (i = 0; i < nb_elements; i++) {
241 addr_dest[i] = addr_source[i];
242 } */
243 }
244 return *this;
245}
