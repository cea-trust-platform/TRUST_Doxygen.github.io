78//! Call to the prepare() method if reinit() has been called previously, then<br>call to preconditionner_()...<br>If echange_ev_solution is not zero, the solution vector will have an updated virtual space.
79int Precond_base::preconditionner(const Matrice_Base& m, const DoubleVect& src, DoubleVect& solution)
80{
81 if (status_ != READY)
82 prepare(m, src);
83// implementation of the prepare() method must call ancestor methods until Precond_base::prepare();
84 assert(status_ == READY);
85 int ok = preconditionner_(m, src, solution);
86 return ok;
87}
