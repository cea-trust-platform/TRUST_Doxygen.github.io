59//! corrige la partition pour que l'element 0 du domaine initial<br>se trouve sur le premier sous-domaine de la partition.<br>On echange le premier sous-domaine et celui qui contient l'element 0.
60void Partitionneur_base::corriger_elem0_sur_proc0(IntVect& elem_part)
61{
62 Cerr << "Correction of the splitting to put the element 0 on processor 0." << finl;
63 int pe_to_xchange = elem_part[0];
64 envoyer_broadcast(pe_to_xchange, 0);
65 if (pe_to_xchange == 0)
66 {
67 Cerr << " No correction to be made" << finl;
68 return;
69 }
70
71 Cerr << " Exchange of parts 0 and " << pe_to_xchange << finl;
72 const int n = elem_part.size_reelle();
73 for (int i = 0; i < n; i++)
74 {
75 const int pe = elem_part[i];
76 if (pe == 0)
77 elem_part[i] = pe_to_xchange;
78 else if (pe == pe_to_xchange)
79 elem_part[i] = 0;
80 }

82}

95//! construction (taille et contenu) du tableau elements avec<br>pour chaque face du bord donne, l'indice de l'element de la zone adjacent<br>a cette face.<br>Parametre: som_elem<br>Signification: connectivite sommet-elements de la zone, calculee a l'aide<br>de construire_connectivite_som_elem<br>Parametre: faces<br>Signification: les faces du bord a traiter (pour chaque face, indices des sommets)<br>Parametre: nom_faces<br>Signification: un nom de bord a imprimer en cas d'erreur<br>Parametre: elements<br>Signification: le tableau a remplir.
96static void chercher_elems_voisins_faces(const Static_Int_Lists& som_elem,
97 const IntTab& faces,
98 const Nom& nom_faces,
99 ArrOfInt& elements)
100{
101 const int nb_faces = faces.dimension(0);
102 elements.resize_array(nb_faces);
103 if (nb_faces == 0)
104 return;
105 const int nb_som_faces = faces.dimension(1);
106 ArrOfInt une_face(nb_som_faces);
107 ArrOfInt voisins;
108 for (int i = 0; i < nb_faces; i++)
109 {
110 for (int j = 0; j < nb_som_faces; j++)
111 une_face[j] = faces(i, j);
112 find_adjacent_elements(som_elem, une_face, voisins);
113 const int nb_voisins = voisins.size_array();
114 if (nb_voisins != 1)
115 {
116 Cerr << "Error in chercher_elems_voisins_faces : the face " << i
117 << "\n of boundary " << nom_faces << " has " << nb_voisins
118 << " neighboring elements with the indices : " << voisins << finl;
119 Process::exit();
120 }
121 elements[i] = voisins[0];
122 }
123}

139//! Calcul d'un graphe de connectivite entre les elements lies par des faces periodiques.<br>Si l'element i est voisin de l'element j par une face periodique, alors il existe<br>k tel que graph(i,k)==j et il existe k2 tel que graph(j,k2)==i.<br>Parametre: zone<br>Signification: la zone a traiter<br>Parametre: liste_bords_periodiques<br>Signification: liste des noms des bords periodiques. ATTENTION: on suppose que les<br>faces des bords periodiques sont rangees selon la convention des bords periodiques.<br>Voir check_faces_periodiques().<br>Parametre: som_elem<br>Signification: la connectivite sommets-elements pour la zone donnee.<br>Parametre: graph<br>Signification: On y stocke le resultat.<br>Valeur de retour: nombre d'elements dans le graphe (egal au nombre de faces periodiques)
140int Partitionneur_base::calculer_graphe_connexions_periodiques(const Zone& zone,
141 const Noms& liste_bords_periodiques,
142 const Static_Int_Lists& som_elem,
143 const int my_offset,
144 Static_Int_Lists& graph)
145{
146 const int nb_elem = zone.nb_elem();
147
148// Pour chaque element, combient a-t-il de faces periodiques ?
149 ArrOfInt nb_faces_perio(nb_elem);
150// Liste de correspondances element0 <=> element1
151// entre l'element voisin d'une face et l'element voisin de la face periodique opposee
152 IntTab correspondances(0,2);
153 correspondances.set_smart_resize(1);
154
155// Premiere etape: remplissage de nb_faces_perio et correspondances
156// Parcours des bords periodiques
157 const int nb_bords = zone.nb_bords();
158 for (int i_bord = 0; i_bord < nb_bords; i_bord++)
159 {
160 const Bord& bord = zone.bord(i_bord);
161 if (!liste_bords_periodiques.contient_(bord.le_nom()))
162 continue;
163 Cerr << " Checking of the boundary " << bord.le_nom();
164 {
165 ArrOfDouble delta;
166 ArrOfDouble erreur;
167 const int ok = Reordonner_faces_periodiques::check_faces_periodiques(zone.bord(i_bord),
168 delta,
169 erreur);
170 const int d = delta.size_array();
171 Cerr << " Delta = ";
172 int i;
173 for (i = 0; i < d; i++) Cerr << delta[i] << " ";
174 Cerr << " Error = ";
175 for (i = 0; i < d; i++) Cerr << erreur[i] << " ";
176 Cerr << finl;
177 if (!ok)
178 {
179 Cerr << "You need to use the Corriger_frontiere_periodique keyword on the periodic boundaries." << finl;
180 Cerr << "See the reference manual to use this keyword on your data file." << finl;
181 exit();
182 }
183 }
184 ArrOfInt elems_voisins;
185 chercher_elems_voisins_faces(som_elem,
186 bord.faces().les_sommets(),
187 bord.le_nom(),
188 elems_voisins);
189
190// Parcours des faces du bord periodique, 2 a 2.
191// On suppose que les faces apparaissent dans l'ordre:
192// d'abord toutes les faces d'une extremite du domaine,
193// puis, dans le meme ordre, les faces de l'autre extremite.
194 assert(bord.nb_faces() % 2 == 0); //!< Nombre pair, forcement
195 const int nb_faces = bord.nb_faces() / 2;
196 for (int i = 0; i < nb_faces; i++)
197 {
198// Les indices des deux elements "voisins" par la face periodique:
199 int elem0 = elems_voisins[i];
200 int elem1 = elems_voisins[i+nb_faces]; //!< Indice de la face perio correspondante
201
202 ++nb_faces_perio[elem0 - my_offset*(elem0 >= my_offset)];
203 ++nb_faces_perio[elem1 - my_offset*(elem1 >= my_offset)];
204 if (elem0 == elem1)
205 {
206 Cerr << "Error in calculer_correspondance_faces_perio: the faces " << i
207 << " and " << i + nb_faces
208 << "\n of the boundary " << bord.le_nom()
209 << " are neighbors of the same element " << elem0 << finl;
210 }
211 const int n = correspondances.dimension(0);
212 correspondances.resize(n+1, 2);
213 correspondances(n, 0) = elem0;
214 correspondances(n, 1) = elem1;
215 }
216 }
217
218// Deuxieme etape:
219// Construction de "graph" a partir du tableau correspondances.
220 graph.set_list_sizes(nb_faces_perio);
221// On recycle le tableau nb_faces_perio pour stocker le nombre
222// d'elements deja remplis dans chaque liste:
223 nb_faces_perio = 0;
224 const int n = correspondances.dimension(0);
225 for (int i = 0; i < n; i++)
226 {
227 const int elem0 = correspondances(i, 0);
228 const int elem1 = correspondances(i, 1);
229
230 const int j0 = nb_faces_perio[elem0 - my_offset*(elem0 >= my_offset)]++;
231 graph.set_value(elem0 - my_offset*(elem0 >= my_offset), j0, elem1);
232 const int j1 = nb_faces_perio[elem1 - my_offset*(elem1 >= my_offset)]++;
233 graph.set_value(elem1 - my_offset*(elem1 >= my_offset), j1, elem0);
234 }
235 Cerr << " There is " << n*2 << " periodic connections." << finl;
236 return n * 2;
237}

252//! Modifie elem_part pour assurer les proprietes suivantes :<br>1) Les elements possedant un sommet de bord sont associes<br>a un processeur possedant une face de bord adjacente a ce sommet.<br>2) Si un processeur possede un sommet periodique reel, il<br>possede forcement le renum_som_perio associe (donc un element<br>qui possede ce sommet et un face periodique).<br>Cette propriete est indispensable pour le periodique (existence<br>de renum_som_perio pour tous les sommets).<br>Pour les autres bords, cette correction est peut-etre inutile, mais<br>pas sur. Sans cette correction, il peut exister des sommets de bord<br>isoles (un processeur possede un sommet de bord mais aucune face).<br>Si on cherche les sommets de bord en parcourant les faces de bord,<br>c'est faux. Avec cette correction, cet algorithme est correct.
253int Partitionneur_base::corriger_sommets_bord(const Domaine& domaine,
254 const Noms& liste_bords_perio,
255 const ArrOfInt& renum_som_perio,
256 const Static_Int_Lists& som_elem,
257 IntVect& elem_part)
258{
259 const int nb_som_tot = domaine.nb_som_tot();
260 const Zone& zone = domaine.zone(0);
261 const int nb_elem = zone.nb_elem();
262 const int nb_elem_tot = zone.nb_elem_tot();
263
264// Premiere etape :
265// Marquage des sommets de bord :
266 ArrOfBit sommet_bord(nb_som_tot);
267 ArrOfBit sommet_bord_perio(nb_som_tot);
268 sommet_bord = 0;
269 sommet_bord_perio = 0;
270// element_bord indique si l'element est adjacent a une face de bord
271 ArrOfBit element_bord(nb_elem_tot);
272// element_bord_perio indique si l'element est adjacent a une face de bord periodique
273 ArrOfBit element_bord_perio(nb_elem_tot);
274 element_bord = 0;
275 element_bord_perio = 0;
276
277 const int nb_bords = zone.nb_bords();
278 for (int i_bord = 0; i_bord < nb_bords; i_bord++)
279 {
280 const Bord& bord = zone.bord(i_bord);
281 const IntTab& faces_sommets = bord.faces().les_sommets();
282 const int nb_faces_bord = faces_sommets.dimension(0);
283 const int nb_som_face = faces_sommets.dimension(1);
284 ArrOfInt elems_voisins;
285 const int is_perio = (liste_bords_perio.contient_(bord.le_nom()));
286
287 chercher_elems_voisins_faces(som_elem,
288 faces_sommets,
289 bord.le_nom(),
290 elems_voisins);
291// Pour chaque element voisin des faces du bord, on marque cet element.
292// On associe a chaque sommet de bord l'indice de la partie la plus
293// petite qui contient une face de bord adjacente au sommet.
294 for (int i_face = 0; i_face < nb_faces_bord; i_face++)
295 {
296 const int elem_voisin = elems_voisins[i_face];
297 element_bord.setbit(elem_voisin);
298 if (is_perio)
299 element_bord_perio.setbit(elem_voisin);
300 for (int i_som = 0; i_som < nb_som_face; i_som++)
301 {
302 const int som = faces_sommets(i_face, i_som);
303 sommet_bord.setbit(som);
304 if (is_perio)
305 sommet_bord_perio.setbit(som);
306 }
307 }
308 }
309
310// Deuxieme etape:
311// Parcours des elements qui n'ont pas de face de bord mais qui ont
312// des sommets de bord:
313// On construit pour chaque sommet de bord de l'element, la liste des
314// "parties autorisees"
315// Si le sommet renum_som_perio est adjacent a une face de bord periodique,
316// les parties autorisees sont celles qui possedent une face periodique
317// adjacente au sommet.
318// Sinon, c'est les parties contenant une face de bord adjacente au sommet
319// On calcule l'intersection de ces listes et si l'element n'est pas dans
320// une partie autorisee, on en choisit une et on le met dedans.
321 int count = 0;
322 {
323 ArrOfInt parties_autorisees;
324 parties_autorisees.set_smart_resize(1);
325 ArrOfInt tmp;
326 tmp.set_smart_resize(1);
327 const IntTab& elements = zone.les_elems();
328 const int nb_som_elem = elements.dimension(1);
329 for (int elem = 0; elem < nb_elem; elem++)
330 {
331// L'element a-t-il un sommet periodique ?
332 int has_som_perio = 0;
333 int isom;
334 for (isom = 0; isom < nb_som_elem; isom++)
335 if (sommet_bord_perio[elements(elem, isom)])
336 has_som_perio = 1;
337// Boucle sur les sommets de l'element:
338 parties_autorisees.resize_array(0);
339 int nb_sommets_bord = 0; //!< Nombre de sommets de bord de l'element
340 for (isom = 0; isom < nb_som_elem; isom++)
341 {
342 const int som = elements(elem, isom);
343// Ne traiter que les sommets de bord perio si l'element a un bord perio,
344// sinon ne traiter que les sommets de bord.
345 if (has_som_perio && !sommet_bord_perio[som])
346 continue;
347 if (!sommet_bord[som])
348 continue;
349
350 nb_sommets_bord++;
351// On met dans tmp la liste des parties associees aux elements de bord adjacents
352 {
353 tmp.resize_array(0);
354 const int renum_som = renum_som_perio[som];
355 const int n = som_elem.get_list_size(renum_som);
356 for (int i = 0; i < n; i++)
357 {
358 const int elem2 = som_elem(renum_som, i);
359 int test;
360 if (has_som_perio)
361 test = element_bord_perio[elem2];
362 else
363 test = element_bord[elem2];
364 if (test)
365 {
366 const int p = elem_part[elem2];
367 tmp.append_array(p);
368 }
369 }
370 array_trier_retirer_doublons(tmp);
371 }
372// Calcul de l'intersection entre tmp et parties_autorisees
373 if (parties_autorisees.size_array() > 0)
374 array_calculer_intersection(parties_autorisees, tmp);
375 else
376 parties_autorisees = tmp;
377 }
378 if (nb_sommets_bord > 0)
379 {
380// Est-ce que l'element appartient a une partie autorisee ?
381 const int n = parties_autorisees.size_array();
382 if (n == 0)
383 {
384 Cerr << "Partitionneur_base::corriger_sommets_bord : Error. No part authorized because of the periodicity for the mesh element " << elem << finl;
385 elem_part[elem] = -1;
386 }
387 else
388 {
389 int i;
390 const int p = elem_part[elem];
391 for (i = 0; i < n; i++)
392 if (parties_autorisees[i] == p)
393 break;
394 if (i >= n)
395 {
396// Il faut affecter une autre partie:
397 elem_part[elem] = parties_autorisees[0];
398 count++;
399 }
400 }
401 }
402 }
403 Cerr << "Partitionneur_base::corriger_sommets_bord : " << count << " modified elements" << finl;
404 }
405 return count;
406}

411//! applique des corrections a elem_part pour que le<br>multi-periodique soit correct :<br>Si un sommet appartient a plusieurs frontieres periodiques,<br>tous les elements adjacents sont rattaches au meme processeur.
412int Partitionneur_base::corriger_multiperiodique(const Domaine& domaine,
413 const Noms& liste_bords_perio,
414 const ArrOfInt& renum_som_perio,
415 const Static_Int_Lists& som_elem,
416 IntVect& elem_part)
417{
418 const int nb_som = domaine.nb_som();
419 const Zone& zone = domaine.zone(0);
420 const int nb_elem = zone.nb_elem();
421
422// Pour chaque sommet periodique, selectionner une partie a laquelle il appartient
423// (la plus petite parmi les parties des elements periodiques adjacents)
424// Initialise a -1
425 ArrOfInt partie_associee(nb_som);
426 partie_associee= -1;
427// Pour chaque sommet, a quel(s) bords periodiques appartient-il
428// (somme de 2^n ou n est l'indice du nom du bord dans la liste des bords periodiques)
429// Initialise a 0
430 ArrOfInt marqueur_bord(nb_som);
431
432 const int nb_bords_perio = liste_bords_perio.size();
433 int deux_puissance_i_bord = 1;
434 for (int i_bord = 0; i_bord < nb_bords_perio; i_bord++)
435 {
436 const Bord& bord = zone.bord(liste_bords_perio[i_bord]);
437 const IntTab& faces_sommets = bord.faces().les_sommets();
438 const int nb_faces_bord = faces_sommets.dimension(0);
439 const int nb_som_face = faces_sommets.dimension(1);
440 ArrOfInt elems_voisins;
441 chercher_elems_voisins_faces(som_elem,
442 faces_sommets,
443 bord.le_nom(),
444 elems_voisins);
445 for (int i_face = 0; i_face < nb_faces_bord; i_face++)
446 {
447 const int elem_voisin = elems_voisins[i_face];
448 const int part = elem_part[elem_voisin];
449 for (int i_som = 0; i_som < nb_som_face; i_som++)
450 {
451 const int som = faces_sommets(i_face, i_som);
452 const int old_part = partie_associee[som];
453 if (old_part < 0 || old_part > part)
454 partie_associee[som] = part;
455 marqueur_bord[som] |= deux_puissance_i_bord; //!< Bitwise OR operator
456 }
457 }
458 deux_puissance_i_bord *= 2;
459 if (deux_puissance_i_bord > 65536)
460 {
461// De toutes facons, plus de 3 frontieres periodiques, c'est hautement suspect...
462 Cerr << "Error in Partitionneur_base::corriger_multiperiodique : there is too many periodic boundaries." << finl;
463 exit();
464 }
465 }
466
467// Transformer marqueur_bord:
468// 1 si le sommet appartient a plusieurs bords periodiques,
469// 0 sinon
470 for (int sommet = 0; sommet < nb_som; sommet++)
471 {
472// Compter le nombre de bits a 1 dans le marqueur
473 const int marq = marqueur_bord[sommet];
474 int n = 0;
475 for (int x = 1; x < marq; x = x * 2)
476 {
477 if (marq & x) //!< bitwise AND
478 n++;
479 }
480// Marqueur a 1 si le sommet appartient a plusieurs bords
481 marqueur_bord[sommet] = (n > 1);
482 }
483// Deuxieme etape: affecter les elements adjacents a un sommet multiperiodique
484// a la partie associee au sommet renum_som_perio de ce sommet.
485 const IntTab& les_elems = zone.les_elems();
486 const int nb_som_elem = les_elems.dimension(1);
487 int count = 0;
488 for (int elem = 0; elem < nb_elem; elem++)
489 {
490// Cet int vaudra -1 si aucun sommet de l'element est periodique,
491// sinon, c'est la plus petite des parties associees aux sommets periodiques
492 int new_part = -1;
493 for (int isom = 0; isom < nb_som_elem; isom++)
494 {
495 const int sommet = les_elems(elem, isom);
496 if (marqueur_bord[sommet])
497 {
498// Ce sommet appartient a plusieurs frontieres periodiques
499 const int renum = renum_som_perio[sommet];
500 const int part = partie_associee[renum];
501 assert(marqueur_bord[renum]);
502 assert(part > -1);
503 if (new_part < 0 || new_part > part)
504 new_part = part;
505 }
506 }
507 if (new_part >= 0 && new_part != elem_part[elem])
508 {
509 count++;
510 elem_part[elem] = new_part;
511 }
512 }
513 Cerr << "Partitionneur_base::corriger_multiperiodique : " << count << " modified elements" << finl;
514
515// Add another criteria to fix some problem when building renum_som_perio in parallel:
516// Check every periodic nodes are surrounded by elements on the same parts
517 ArrOfInt node(2);
518 int another_count=0;
519 int err=0;
520 for (int som=0; som<nb_som; som++)
521 {
522// Periodic nodes:
523 node[0]=som;
524 node[1]=renum_som_perio[som];
525 if (node[0]!=node[1])
526 {
527// Loop on each element surrounding the nodes
528 IntLists part(2);
529 for (int i=0; i<2; i++)
530 {
531 for (int i_elem=0; i_elem<som_elem.get_list_size(node[i]); i_elem++)
532 {
533 int elem = som_elem(node[i],i_elem);
534 part[i].add_if_not(elem_part[elem]);
535 }
536 }
537// Check if same number:
538 if (part[0].size()!=part[1].size())
539 {
540 Cerr << "Warning: Not the same number of parts around the periodic nodes " << node[0] << " and " << node[1] << " !" << finl;
541 Cerr << "We try to fix:" << finl;
542 int smaller = ( part[0].size() < part[1].size() ? 0 : 1);
543 int bigger = 1 - smaller;
544 int first_part = part[smaller][0]; //!< Take arbitrary the first part of the smallest list
545 for (int i=0; i<part[bigger].size(); i++)
546 {
547 int i_part = part[bigger][i];
548 if (!part[smaller].contient(i_part))
549 {
550// i_part -> first_part on all elements surrounding the node with more parts:

622//! corrige la partition elem_part pour qu'un element i se trouve sur la meme<br>partition elem_part[i] que tous les elements auxquels il est lie dans le graphe<br>(elements d'indices graph_elements_perio(i, j) pour tout j).<br>Parametre: graph_elements_perio<br>Signification: graphe calcule par la methode calculer_graphe_connexions_periodiques<br>Parametre: elem_part<br>Signification: pour chaque element, a quelle partie appartient-il.<br>Valeur de retour: nombre d'elements dont la partition a ete corrigee.
623int Partitionneur_base::corriger_bords_avec_graphe(const Static_Int_Lists& graph_elements_perio,
624 const Static_Int_Lists& som_elem,

626 const Noms& liste_bords_perio,
627 IntVect& elem_part)
628{
629// Algorithme: parcours de tous les elements dans l'ordre.
630// Pour chaque element, associer aux autres elements lies la partie a laquelle appartient
631// l'element courant. Comme le graphe est symetrique, si un element a deja ete traite,
632// on ne change rien les fois suivantes. Donc un seul passage suffit.
633 const int n = graph_elements_perio.get_nb_lists(); //elem_part.size_array();
634// assert(n == graph_elements_perio.get_nb_lists());

637 {
638 const int m = graph_elements_perio.get_list_size(i);
639 const int part = elem_part[i];
640 for (int j = 0; j < m; j++)
641 {
642 const int elem2 = graph_elements_perio(i,j);
643 if (elem_part[elem2] != part)
644 {
645 elem_part[elem2] = part;
646 count++;
647 }
648 }
649 }

651 << " modified periodic elements" << finl;
652

654 ArrOfInt renum_som_perio(nb_sommets_reels);
655// Initialisation du tableau renum_som_perio
656 for (int i = 0; i < nb_sommets_reels; i++)
657 renum_som_perio[i] = i;
658 int parallel_algo = Process::nproc() > 1;
659 Reordonner_faces_periodiques::renum_som_perio(domaine, liste_bords_perio, renum_som_perio,
660 parallel_algo /* pas d'espace virtuel en sequentiel */);
661

663 count += corriger_multiperiodique(domaine, liste_bords_perio, renum_som_perio, som_elem, elem_part);
664 count += corriger_sommets_bord(domaine, liste_bords_perio, renum_som_perio, som_elem, elem_part);
665 return count;
