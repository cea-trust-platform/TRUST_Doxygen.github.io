37Entree& Perte_Charge_Isotrope_PolyMAC_Face::readOn(Entree& s )
38{
39 Cerr << "Perte_Charge_Isotrope_PolyMAC_Face::readOn " << finl;
40 sous_zone=false;
41 int lambda_ok=0;
42
43// Definition des mots-cles
44 Motcles les_mots(4);
45 les_mots[0] = "lambda";
46 les_mots[1] = "diam_hydr";
47 les_mots[2] = "sous_zone";
48 les_mots[3] = "implicite";
49
50// Lecture et interpretation
51 Motcle motlu, accolade_fermee="}", accolade_ouverte="{";
52 s >> motlu;
53 while (motlu != accolade_ouverte)
54 {
55 Cerr << "On attendait une { a la lecture d'un " << que_suis_je() << finl;
56 Cerr << "et non : " << motlu << finl;
57 exit();
58 }
59 s >> motlu;
60 while (motlu != accolade_fermee)
61 {
62 int rang=les_mots.search(motlu);
63 switch(rang)
64 {
65 case 0 : //!< lambda
66 {
67 lambda_ok=1;
68 Nom tmp;
69 s >> tmp;
70 Cerr << "Lecture et interpretation de la fonction " << tmp << " ... ";
71 lambda.setNbVar(2+dimension);
72 lambda.setString(tmp);
73 lambda.addVar("Re");
74 lambda.addVar("t");
75 lambda.addVar("x");
76 if (dimension>1)
77 lambda.addVar("y");
78 if (dimension>2)
79 lambda.addVar("z");
80 lambda.parseString();
81 Cerr << " Ok" << finl;
82 break;
83 }
84 case 1: //!< diam_hydr
85 s >> diam_hydr;
86 break;
87 case 2: //!< sous_zone
88 s >> nom_sous_zone;
89 sous_zone=true;
90 break;
91 case 3:
92 {
93 s>>implicite_;
94 break;
95 }
96 default : //!< non compris
97 Cerr << "Mot cle \"" << motlu << "\" non compris lors de la lecture d'un "
98 << que_suis_je() << finl;
99 exit();
100 }
101 s >> motlu;
102 }
103
104// Verification de la coherence
105 if (lambda_ok==0)
106 {
107 Cerr << "Il faut definir l'expression lamba(Re)" << finl;
108 exit();
109 }
110
111 if (diam_hydr->nb_comp()!=1)
112 {
113 Cerr << "Il faut definir le champ diam_hydr a une composante" << finl;
114 exit();
115 }
116
117 Cerr << "Fin de Perte_Charge_Isotrope_PolyMAC_Face::readOn" << finl;
118 return s;
119}
