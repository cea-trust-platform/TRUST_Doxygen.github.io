62// -Initialisation de ref_eq_
63// -Initialisation de localisation_

65void Champ_Generique_Morceau_Equation::completer(const Postraitement_base& post)
66{
67 const Probleme_base& Pb = get_ref_pb_base();
68 int numero_eq_=-1;
69 if (sub_type(Champ_Generique_refChamp,get_source(0)))
70 {
71
72 Champ espace_stockage;
73 const Champ_base& mon_champ = get_source(0).get_champ(espace_stockage);
74 if (sub_type(Champ_Inc_base,mon_champ))
75 {
76 const Champ_Inc_base& mon_champ_inc = ref_cast(Champ_Inc_base,mon_champ);
77
78 {
79// On recupere l equation alors qu elle n est pas encore associee au Champ_Inc
80// On parcours les equations du probleme et on identifie celle qui correspond au champ inconnu
81
82 int nb_eq = Pb.nombre_d_equations();
83 int i=0;
84
85 while (i<nb_eq)
86 {
87 const Equation_base& eq_test = Pb.equation(i);
88 if ((eq_test.inconnue().le_nom() == mon_champ_inc.le_nom()))
89 {
90 numero_eq_=i;
91 break;
92 }
93 i++;
94 }
95 }
96 }
97 }
98
99 if (numero_eq_==-1)
100 {
101 Cerr<<"We can apply a Champ_Generique_Morceau_Equation only to an unknown field of the problem"<<finl;
102 exit();
103 }
104 ref_eq_ = Pb.equation(numero_eq_);
105
106 localisation_ = morceau().get_localisation_pour_post(option_);
107
108// Appel de la methode de la classe mere fait apres pour que
109// la methode nommer_source() dispose de ref_eq_ initialise
110 Champ_Gen_de_Champs_Gen::completer(post);
111}

113Champ_Fonc& Champ_Generique_Morceau_Equation::creer_espace_stockage(const Nature_du_champ& nature,
114 const int nb_comp,
115 Champ_Fonc& es_tmp) const
116{
117 Noms noms;
118 Noms unites;
119 for (int c=0; c<nb_comp; c++)
120 {
121// noms.add("bidon");
122 unites.add("bidon");
123 }
124 noms.add("bidon");
125 double temps;
126 temps = get_time();
127 const Discretisation_base& discr = get_discretisation();
128 Motcle directive;
129 if ((Motcle(option_)=="flux_bords") || (Motcle(option_)=="flux_surfacique_bords"))
130 directive = "champ_face"; //!< Pour avoir les flux_bords aux faces et non aux elements en VDF (ex: temperature)
131 else
132 directive = get_directive_pour_discr();
133 const Zone_dis_base& zone_dis = get_ref_zone_dis_base();
134
135 discr.discretiser_champ(directive,zone_dis,nature,noms,unites,nb_comp,temps,es_tmp);
136
137 if (directive=="pression")
138 {
139 const Zone_Cl_dis_base& zcl = get_ref_zcl_dis_base();
140 es_tmp->completer(zcl);
141 }
142
143 return es_tmp;
144}

152const Champ_base& Champ_Generique_Morceau_Equation::get_champ(Champ& espace_stockage) const
153{
154 Champ_Fonc es_tmp;
155 Nature_du_champ nature;
156 int nb_comp = -1;
157
158 if ((Motcle(option_)=="stabilite") || (Motcle(option_)=="flux_bords") || (Motcle(option_)=="flux_surfacique_bords"))
159 {
160 nature = scalaire;
161 nb_comp = 1;
162 }
163 else
164 {
165 Cerr << "Error. Contact TRUST support." << finl;
166 Process::exit();
167 }
168
169 espace_stockage = creer_espace_stockage(nature,nb_comp,es_tmp);
170 morceau().calculer_pour_post(espace_stockage,option_,compo_);
171 DoubleTab& es_val = espace_stockage.valeurs();
172 es_val.echange_espace_virtuel();
173 return espace_stockage.valeur();
174}

191const Noms Champ_Generique_Morceau_Equation::get_property(const Motcle& query) const
192{
193
194 Motcles motcles(2);
195 motcles[0] = "composantes";
196 motcles[1] = "unites";
197
198 int rang = motcles.search(query);
199 switch(rang)
200 {
201 case 0:
202 {
203 Noms compo(1);
204 Nom nume((int)1);
205 compo[0] = nom_post_+nume;
206
207 return compo;
208 break;
209 }
210
211 case 1:
212 {
213 Noms unites(1);
214 if (Motcle(option_)=="stabilite")
215 unites[0] = "s";
216 else if (Motcle(option_)=="flux_bords" || Motcle(option_)=="flux_surfacique_bords")
217 unites[0] = "bidon";
218
219 return unites;
220 break;
221 }
222
223 }
224 return Champ_Gen_de_Champs_Gen::get_property(query);
225}

247// Description
248// Nomme le champ en tant que source par defaut
249// nom_eq + "_" + type_morceau + "_" + type_option
250void Champ_Generique_Morceau_Equation::nommer_source()
251{
252 Motcle type_morceau, type_option;
253 if (Motcle(type_morceau_)=="operateur")
254 {
255 if (numero_morceau_==0)
256 type_morceau = "diffusion";
257 else if (numero_morceau_==1)
258 type_morceau = "convection";
259 else if (numero_morceau_==2)
260 type_morceau = "gradient";
261 else if (numero_morceau_==3)
262 type_morceau = "divergence";
263 else
264 {
265 Cerr<<"This operateur number is not available."<<finl;
266 exit();
267 }
268 }
269 else
270 {
271 Cerr<<"Currently the only pieces of equations considered for the postprocessing"<<finl;
272 Cerr<<"are of type operator"<<finl;
273 exit();
274 }
275
276 if (Motcle(option_)=="stabilite")
277 type_option = "dt";
278 else if (Motcle(option_)=="flux_bords")
279 {
280 type_option = "flux_bords";
281 }
282 else if (Motcle(option_)=="flux_surfacique_bords")
283 {
284 type_option = "flux_surfacique_bords";
285 }
286 else
287 {
288 Cerr<<"Currently the only options considered for a piece of equation are \"stabilite\" and \"flux_bords\"."<<finl;
289 exit();
290 }
291
292
293 if (nom_post_=="??")
294 {
295 Nom nom_post_source;
296 const Nom& nom_eq = ref_eq_->le_nom();
297 nom_post_source = nom_eq;
298 nom_post_source += "_";
299 nom_post_source += type_morceau;
300 nom_post_source += "_";
301 nom_post_source += type_option ;
302 if ((Motcle(option_)=="flux_bords") || (Motcle(option_)=="flux_surfacique_bords"))
303 {
304 Nom nume(compo_);
305 nom_post_source += nume;
306 }
307 nommer(nom_post_source);
308 }
309
310}
