106int Champ_front_softanalytique::initialiser(double temps, const Champ_Inc_base& inco)
107{
108
109 if (!Ch_front_var_stationnaire::initialiser(temps,inco))
110 return 0;
111
112 int dim=nb_comp();
113 double x,y,z;
114// int nbsf=faces.nb_som_faces();
115 int k;
116 DoubleTab& tab=valeurs();
117
118 const Zone_VF& zone_vf=ref_cast(Zone_VF,frontiere_dis().zone_dis());
119 const DoubleTab& xv=zone_vf.xv();
120
121 const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
122 int nb_faces_bord_tot=le_bord.nb_faces_tot();
123 for (int ind_face=0; ind_face<nb_faces_bord_tot; ind_face++)
124 {
125 int face = le_bord.num_face(ind_face);
126 x=y=z=0;
127 x=xv(face,0);
128 if(dimension>1)
129 y=xv(face,1);
130 if(dimension>2)
131 z=xv(face,2);
132
133 for( k=0; k<dim; k++)
134 {
135 fxyz[k].setVar("x",x);
136 fxyz[k].setVar("y",y);
137 fxyz[k].setVar("z",z);
138 tab(ind_face,k)=fxyz[k].eval();
139// Cout << " x y z " << x << " " << y << " " << z << " " << tab(i,k) << finl;
140 }
141 }
142
143 tab.echange_espace_virtuel();
144 return 1;
145}
