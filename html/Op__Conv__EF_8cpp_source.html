196void Op_Conv_EF::ajouter_contribution_sous_cond(const DoubleTab& transporte, Matrice_Morse& matrice,int btd_impl,int hourglass_impl,int centre_impl ) const
197{
198
199 const Champ_Inc_base& la_vitesse=vitesse_.valeur();
200 const DoubleTab& G=la_vitesse.valeurs();
201
202 int transport_rhou=0;
203 if (vitesse_.le_nom()=="rho_u") transport_rhou=1;
204
205 const DoubleTab& rho_elem=(transport_rhou==1 ? equation().probleme().get_champ("masse_volumique_melange").valeurs() : \
206 equation().probleme().get_champ("masse_volumique").valeurs());
207 int is_not_rho_unif = (rho_elem.size() == 1 ? 0 : 1);
208
209 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
210 int nb_comp = transporte.line_size();
211 const DoubleVect& volumes_thilde= zone_ef.volumes_thilde();
212 const DoubleVect& volumes= zone_ef.volumes();
213 const DoubleTab& IPhi_thilde=zone_ef.IPhi_thilde();
214
215 const DoubleTab& bij=zone_ef.Bij();
216 int nb_elem_tot=zone_ef.zone().nb_elem_tot();
217 int nb_som_elem=zone_ef.zone().nb_som_elem();
218 const IntTab& elems=zone_ef.zone().les_elems() ;
219 ArrOfDouble G_e(dimension);
220
221 double f=coefficient_btd();
222// DoubleTab transp_loc(nb_som_elem,nb_comp);
223// A DEPLACER !!!!!
224 const DoubleTab& lambda=ref_cast(Operateur_Diff_base,equation().operateur(0).l_op_base()).diffusivite().valeurs();
225 int is_not_lambda_unif=1;
226 if (lambda.size()==1)
227 is_not_lambda_unif=0;
228
229 int mcoef3d[8]= {1,-1,-1,1,-1,1,1,-1};
230 int sommetoppose[8]= {7,6,5,4,3,2,1,0};
231 for (int elem=0; elem<nb_elem_tot; elem++)
232 if (elem_contribue(elem))
233 {
234 G_e=0;
235 for (int i1=0; i1<nb_som_elem; i1++)
236 {
237 int glob=elems(elem,i1);
238 for (int b=0; b<dimension; b++)
239 G_e[b]+=G(glob,b);
240 }
241 G_e/=nb_som_elem;
242
243 double pond2=volumes_thilde(elem)/volumes(elem)/volumes(elem);
244
245 if (transport_rhou) pond2 /= (is_not_rho_unif ? rho_elem(elem) : rho_elem(0,0));
246
247 if ((hourglass)&&(nb_som_elem==8)&&(hourglass_impl==1))
248 {
249 double pond3=f*dotproduct_array(G_e,G_e);
250 if (transport_rhou) pond3 /= (is_not_rho_unif ? rho_elem(elem) : rho_elem(0,0));
251 if (is_not_lambda_unif)
252 pond3+=lambda(elem);
253 else
254 pond3+=lambda(0,0);
255 pond3*=volumes_thilde(elem)/volumes(elem)*pow(volumes(elem),0.3333333333333333);
256 pond3*=hourglass;
257
258 for (int a=0; a<nb_comp; a++)
259 {
260 double coef2d=0.042*pond3;
261 double coef3d=coef2d*0.5;
262
263 for (int i1=0; i1<8; i1++)
264 {
265 int n1=elems(elem,i1)*nb_comp+a;
266 for (int i2=0; i2<8; i2++)
267 {
268 int n2=elems(elem,i2)*nb_comp+a;
269 matrice.coef(n1,n2)+=mcoef3d[i1]*mcoef3d[i2]*coef3d-coef2d;
270 }
271 matrice.coef(n1,n1)+=coef2d*4.;
272 matrice.coef(n1,elems(elem,sommetoppose[i1])*nb_comp+a)+=coef2d*4.;
273 }
274 }
275 }
276
277 for (int i1=0; i1<nb_som_elem; i1++)
278 {
279 int glob=elems(elem,i1);
280
281 if (((btd_impl==1)&&(type_op==amont))||(centre_impl==1))
282 {
283 double cb=0;
284 for (int b=0; b<dimension; b++)
285 cb+=G_e[b]*bij(elem,i1,b);
286 cb*=(f*pond2);
287 for (int i2=0; i2<nb_som_elem; i2++)
288 {
289 int glob2=elems(elem,i2);
290
291 double ca=0;
292 if (centre_impl==1)
293 {
294 for (int b=0; b<dimension; b++)
295 ca+=G_e[b]*bij(elem,i2,b);
296 double pond=IPhi_thilde(elem,i1)/volumes(elem);
297
298 ca*=pond;
299 }
300 double cc=0;
301 if ((btd_impl==1)&&(type_op==amont))
302 {
303 for (int c=0; c<dimension; c++)
304 cc+=G_e[c]*bij(elem,i2,c);
305 cc*=cb;
306 }
307// resu(glob)-=cc*cb*transp_loc(i2,0);
308 for (int a=0; a<nb_comp; a++)
309 {
310 int n1=glob*nb_comp+a;
311 int n2=glob2*nb_comp+a;
312 matrice.coef(n1,n2)+=cc+ca;
313
314 }
315 }
316 }
317 }
318 }
319 return;
320}

332double Op_Conv_EF::calculer_dt_stab() const
333{
334 if (calcul_dt_stab_==0) return Operateur_base::calculer_dt_stab();
335
336 const Champ_Inc_base& la_vitesse=vitesse_.valeur();
337 const DoubleTab& G=la_vitesse.valeurs();
338
339 int transport_rhou=0;
340 if (vitesse_.le_nom()=="rho_u") transport_rhou=1;
341
342 const DoubleTab& rho_elem=(transport_rhou==1 ? equation().probleme().get_champ("masse_volumique_melange").valeurs() : \
343 equation().probleme().get_champ("masse_volumique").valeurs());
344 int is_not_rho_unif = (rho_elem.size() == 1 ? 0 : 1);
345
346 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
347 const DoubleVect& valeurs_diffusivite = ref_cast(Operateur_Diff_base,equation().operateur(0).l_op_base()).diffusivite().valeurs();
348 int is_not_lambda_unif = (valeurs_diffusivite.size() == 1 ? 0 : 1);
349
350 int autre_eq=0;
351 const DoubleVect& valeurs_diffusivite_p = ref_cast(Operateur_Diff_base,equation().probleme().equation(autre_eq).operateur(0).l_op_base()).diffusivite().valeurs();
352 if (calcul_dt_stab_==2)
353 {
354 if (&valeurs_diffusivite_p == &valeurs_diffusivite)
355 autre_eq=1;
356 }
357 const DoubleVect& valeurs_diffusivite_2 = ref_cast(Operateur_Diff_base,equation().probleme().equation(autre_eq).operateur(0).l_op_base()).diffusivite().valeurs();
358
359 const DoubleTab& coord=zone_ef.zone().domaine().les_sommets();
360
361 int nb_elem_tot=zone_ef.zone().nb_elem_tot();
362 int nb_som_elem=zone_ef.zone().nb_som_elem();
363 const IntTab& elems=zone_ef.zone().les_elems() ;
364 ArrOfDouble G_e(dimension);
365 max_ue_=1e-8 ; //!< comme G2
366 min_dx_=1e37;
367 double Max=0;
368 double dt_l=0,dt2=0;
369 for (int elem=0; elem<nb_elem_tot; elem++)
370 if (elem_contribue(elem))
371 {
372 G_e=0;
373 for (int i1=0; i1<nb_som_elem; i1++)
374 {
375 int glob=elems(elem,i1);
376 for (int b=0; b<dimension; b++)
377 G_e[b]+=G(glob,b);
378 }
379 G_e/=nb_som_elem;
380// if (transport_rhou) G_e/=rho_elem(elem);
381 double ml=0;
382 double dx2;
383// pas de temps de diffusion !!
384 double ml2=0;
385 double vx[3];
386 for (int d=0; d<3; d++)
387 {
388 dx2=std::fabs(coord(elems(elem,0),d)-coord(elems(elem,7),d));
389 dx2+=std::fabs(coord(elems(elem,1),d)-coord(elems(elem,6),d));
390 dx2+=std::fabs(coord(elems(elem,2),d)-coord(elems(elem,5),d));
391 dx2+=std::fabs(coord(elems(elem,3),d)-coord(elems(elem,4),d));
392 dx2*=0.25;
393 dx2*=dx2;
394
395 ml2+=dx2;
396// double p=G_e(d)*G_e(d)/dx2;
397 ml+=G_e[d]*G_e[d]/dx2;
398 vx[d]=G_e[d]*G_e[d];
399 }
400
401 double diffu=(is_not_lambda_unif?valeurs_diffusivite(elem):valeurs_diffusivite(0));
402
403 if (calcul_dt_stab_==2)
404 {
405 double diffu2=(is_not_lambda_unif?valeurs_diffusivite_2(elem):valeurs_diffusivite_2(0));
406 if (diffu2<diffu)
407 diffu=diffu2;
408 }
409 diffu /= (is_not_rho_unif ? rho_elem(elem) : rho_elem(0,0));
410
411// horreur pour NR G2 cas clotaire
412// diffu*=0.5;
413// if (elem==0) Cerr<<"uu "<<diffu<<finl;
414 if (transport_rhou)
415 {
416 double rho_e = (is_not_rho_unif ? rho_elem(elem) : rho_elem(0,0));
417 double inv_rho2= rho_e*rho_e;
418 inv_rho2=1./inv_rho2;
419 ml*=inv_rho2;
420 for (int d=0; d<dimension; d++)
421 vx[d]*=inv_rho2;
422 }
423// le .5* c'est pour avoir le meme dt que g2 mais faux
424 double dxe=sqrt(ml2);
425 if (dxe<min_dx_)
426 min_dx_=dxe;
427 ml2=diffu/ml2;
428 ml*=btd_;
429 ml=sqrt(ml2*ml2+ml)+ml2+DMINFLOAT;
430 /* if (ml2>0)
431 ml=ml/(sqrt(ml2*ml2+ml)-ml2);
432 else ml=-1.;
433 */
434 if (ml > Max) Max=ml;
435 double ue=sqrt(vx[0]+vx[1]+vx[2]);
436 if (ue>max_ue_) max_ue_=ue;
437
438 dt_l= 1./(Max+DMINFLOAT);
439
440 if (elem==0) dt2=dt_l;
441 if (btd_!=3.)
442 {
443 double dt0=dt2;
444 double s=1.;
445 while(s>0)
446 {
447 s=-1;
448 for (int d=0; d<dimension; d++)
449 {
450 double c=vx[d]*dt0;
451 if (std::fabs(c)>1e-16)
452 s+=c/(2.*diffu +c*btd_);
453 }
454 if (s>0)
455 {
456 dt0*=0.9;
457 Cout<<elem<<" " <<dt_l<<" "<<s<<" "<<dt0<<finl;
458 exit();
459 }
460 }
461 dt2=std::min(dt2,dt0);
462 dt_l=std::min(dt2,dt_l);
463 }
464 }
465 min_dx_=mp_min(min_dx_);
466 max_ue_=mp_max(max_ue_);
467 coefficient_correcteur_supg_.dimensionner(1,1);
468 coefficient_correcteur_supg_(0,0)=min_dx_/max_ue_;
469 double dt_stab=dt_l;
470 dt_stab = Process::mp_min(dt_stab);
471// astuce pour contourner le type const de la methode
472 Op_Conv_EF_base& op = ref_cast_non_const(Op_Conv_EF_base,*this);
473 op.fixer_dt_stab_conv(dt_stab);
474
475 return dt_stab;
476}
