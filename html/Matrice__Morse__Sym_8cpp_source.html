216//! Operation de multiplication-accumulation (saxpy) matrice matrice<br>(matrice represente par un tableau)<br>Operation: RESU = RESU + A*X
217/** \return DoubleTab\&<br>
218Signification: la matrice resultat de l'operation<br>
219Contraintes: */
220DoubleTab& Matrice_Morse_Sym::ajouter_multTab_(const DoubleTab& x, DoubleTab& resu) const
221{
222 if ( (x.nb_dim() == 1) && (resu.nb_dim() == 1))
223 {
224 ajouter_multvect(x,resu);
225 return resu;
226 }
227
228 assert_check_symmetric_morse_matrix_structure( );
229 int i,k,j;
230 double aij;
231 int comp;
232 int nb_com=x.dimension(1);
233 assert(resu.dimension(1)==nb_com);
234 double* t=new double[nb_com];
235 int n=nb_lignes();
236 for(i=0; i<n; i++)
237 {
238 for(comp=0; comp<nb_com; comp++)
239 t[comp] = coeff_(tab1_(i)-1)*x(i,comp);
240 for (k=tab1_(i); k<tab1_(i+1)-1; k++)
241 {
242 j = tab2_(k)-1;
243 aij = coeff_(k);
244 for(comp=0; comp<nb_com; comp++)
245 {
246 t[comp] += aij*x(j,comp);
247 resu(j,comp) += aij*x(i,comp);
248 }
249 }
250 for(comp=0; comp<nb_com; comp++)
251 resu(i,comp) += t[comp] ;
252 }
253 delete [] t;
254 return resu;
255}

257double Matrice_Morse_Sym::multvect_et_prodscal(const DoubleVect& x, DoubleVect& resu) const
258{
259 assert_check_symmetric_morse_matrix_structure( );
260
261 assert(x.size_totale() == nb_lignes() || x.size() == nb_lignes());
262 assert(resu.size_totale() == nb_lignes() || resu.size() == nb_lignes());
263 assert(m_ <= x.size_totale());
264
265 double prod_scal_local = 0.;
266 operator_egal(resu, 0.);
267
268 const int fin = nb_lignes() + 1;
269 {
270 const double * thecoef = get_coeff().addr() - 1;
271 const double * xx = x.addr() - 1;
272 double * res = resu.addr() - 1;
273 const int * index1 = get_tab1().addr() - 1;
274 const int * index2 = get_tab2().addr() - 1;
275#ifdef COMPILER_BLRTS_XLC
276#pragma disjoint (*coef, *xx, *res)
277#endif
278 int i;
279 for (i = 1; i < fin; i++)
280 {
281 int j = index1[i];
282 int j_next= index1[i+1];
283// int ncoeffs = j_next - j;
284 assert(i==index2[j]); //!< La diagonale meme nulle doit etre stockee dans une Mat_Morse_Sym
285 double xi = xx[i];
286 double resu_tmp = res[i] + thecoef[j] * xi;
287 j++;
288 for (; j < j_next; j++)
289 {
290 int nj = index2[j];
291 double coef_j = thecoef[j];
292 double xn = xx[nj];
293 resu_tmp += coef_j * xn;
294 res[nj] += coef_j * xi;
295 }
296 res[i] = resu_tmp;
297 prod_scal_local += resu_tmp * xi;
298 }
299 }
300
301 return prod_scal_local;
302}

462//! Fonction (hors classe) amie de la classe Matrice_Morse_Sym<br>NE FAIT RIEN : NON CODE
463/** \return Matrice_Morse_Sym<br>
464Signification:<br>
465Contraintes: */
466Matrice_Morse_Sym operator+(const Matrice_Morse_Sym& A,const Matrice_Morse_Sym& B)
467{
468// Il faut evidemment que les matrices soient de meme taille
469 if (A.nb_lignes() != B.nb_lignes())
470 {
471 Cerr << "Error in Matrice_Morse_Sym::operator+" << finl;
472 Cerr << "The 2 matrices must have the same dimension" << finl;
473 Cerr << "The first matrix is of dimension: " << A.nb_lignes()<< finl;
474 Cerr << "The second matrix is of dimension: " << B.nb_lignes()<< finl;
475 Process::exit();
476 }
477
478 Matrice_Morse_Sym somme(A.nb_lignes(),A.nb_coeff()+B.nb_coeff());
479 IntList colonnes_A_ligne_i,colonnes_B_ligne_i;
480 int min_colonnes_A,min_colonnes_B,non_nuls_ligne_i;
481 DoubleList coeffs_A_ligne_i,coeffs_B_ligne_i;
482 double coeff_A,coeff_B;
483
484// Pour s'assurer que compacte() va marcher
485 for (int i=0; i<somme.get_coeff().size(); i++)
486 somme.get_set_coeff()(i) = 0.;
487
488// On va remplir tab2_ par ordre croissant de colonnes
489 somme.get_set_tab1()(0) = 1;
490 for (int i=0; i<A.nb_lignes(); i++) //boucle sur les lignes
491 {
492// Pour eviter les effets de bord
493 if (!colonnes_A_ligne_i.est_vide()) colonnes_A_ligne_i.vide();
494 if (!colonnes_B_ligne_i.est_vide()) colonnes_B_ligne_i.vide();
495
496// Initialisation des colonnes_A et colonnes_B
497// ATTENTION a la numerotation C++
498 for (int j=0; j<A.get_tab1()(i+1)-A.get_tab1()(i); j++)
499 {
500 colonnes_A_ligne_i.add_if_not(A.get_tab2()(A.get_tab1()(i)+j-1));
501 coeffs_A_ligne_i.add(A.get_coeff()(A.get_tab1()(i)+j-1));
502 }//fin for
503
504 for (int j=0; j<B.get_tab1()(i+1)-B.get_tab1()(i); j++)
505 {
506 colonnes_B_ligne_i.add_if_not(B.get_tab2()(B.get_tab1()(i)+j-1));
507 coeffs_B_ligne_i.add(B.get_coeff()(B.get_tab1()(i)+j-1));
508 }//fin for
509
510// On initialise d'autres varibles
511 non_nuls_ligne_i = 0;
512
513// Debut de l'algorithme
514 while (!colonnes_A_ligne_i.est_vide() || !colonnes_B_ligne_i.est_vide())
515 {
516// Si l'une des listes est vide et pas l'autre
517 if (colonnes_A_ligne_i.est_vide() && !colonnes_B_ligne_i.est_vide())
518 {
519 somme.get_set_tab2()(somme.get_tab1()(i)+non_nuls_ligne_i-1) =
520 colonnes_B_ligne_i[0];
521
522 somme.get_set_coeff()(somme.get_tab1()(i)+non_nuls_ligne_i-1) =
523 coeffs_B_ligne_i[0] ;
524
525 colonnes_B_ligne_i.suppr(colonnes_B_ligne_i[0]);
526 coeffs_B_ligne_i.suppr(coeffs_B_ligne_i[0]);
527 non_nuls_ligne_i++;
528 }//fin if
529
530 if (colonnes_B_ligne_i.est_vide() && !colonnes_A_ligne_i.est_vide())
531 {
532 somme.get_set_tab2()(somme.get_tab1()(i)+non_nuls_ligne_i-1) =
533 colonnes_A_ligne_i[0];
534
535 somme.get_set_coeff()(somme.get_tab1()(i)+non_nuls_ligne_i-1) =
536 coeffs_A_ligne_i[0];
537
538 colonnes_A_ligne_i.suppr(colonnes_A_ligne_i[0]);
539 coeffs_A_ligne_i.suppr(coeffs_A_ligne_i[0]);
540 non_nuls_ligne_i++;
541 }//fin if
542
543// Aucune des deux listes n'est vide
544 min_colonnes_A = colonnes_A_ligne_i[0];
545 min_colonnes_B = colonnes_B_ligne_i[0];
546 coeff_A = coeffs_A_ligne_i[0];
547 coeff_B = coeffs_B_ligne_i[0];
548
549 if (min_colonnes_A < min_colonnes_B)
550 {
551 somme.get_set_tab2()(somme.get_tab1()(i)+non_nuls_ligne_i-1) = min_colonnes_A;
552 somme.get_set_coeff()(somme.get_tab1()(i)+non_nuls_ligne_i-1) = coeff_A;
553
554 colonnes_A_ligne_i.suppr(min_colonnes_A);
555 coeffs_A_ligne_i.suppr(coeff_A);
556 non_nuls_ligne_i++;
557 }//fin if
558
559 if (min_colonnes_B < min_colonnes_A)
560 {
561 somme.get_set_tab2()(somme.get_tab1()(i)+non_nuls_ligne_i-1) = min_colonnes_B;
562 somme.get_set_coeff()(somme.get_tab1()(i)+non_nuls_ligne_i-1) = coeff_B;
563
564 colonnes_B_ligne_i.suppr(min_colonnes_B);
565 coeffs_B_ligne_i.suppr(coeff_B);
566 non_nuls_ligne_i++;
567 }//fin if
568
569 if (min_colonnes_A == min_colonnes_B)
570 {
571 somme.get_set_tab2()(somme.get_tab1()(i)+non_nuls_ligne_i-1) = min_colonnes_A;
572 somme.get_set_coeff()(somme.get_tab1()(i)+non_nuls_ligne_i-1) = coeff_A+coeff_B;
573
574 colonnes_A_ligne_i.suppr(min_colonnes_A);
575 colonnes_B_ligne_i.suppr(min_colonnes_B);
576 coeffs_A_ligne_i.suppr(coeff_A);
577 coeffs_B_ligne_i.suppr(coeff_B);
578 non_nuls_ligne_i++;
579 }//fin if
580
581 }//!< fin while
582
583 somme.get_set_tab1()(i+1) = somme.get_tab1()(i)+non_nuls_ligne_i;//?????
584 }//!< fin for
585
586 somme.compacte();
587 somme.morse_matrix_structure_has_changed_=1;
588 return somme;
589}
