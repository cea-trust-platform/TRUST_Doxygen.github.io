53void Terme_Puissance_Thermique_Echange_Impose_P0_VDF::mettre_a_jour(double temps)
54{
55 int nb_elem = la_zone_VDF.valeur().nb_elem();
56 const Zone_VF& zone = la_zone_VDF.valeur();
57 const DoubleVect& volumes = zone.volumes();
58 const DoubleTab& himp = himp_.valeur().valeurs();
59 const DoubleTab& Text = Text_.valeur().valeurs();
60 const DoubleTab& T = equation().inconnue().valeurs();
61
62 bilan()(0) = 0;
63
64 for (int e = 0; e < nb_elem; e++)
65 bilan()(0) += himp.addr()[himp.dimension(0) > 1 ? e : 0] * volumes(e) * (Text.addr()[Text.dimension(0) > 1 ? e : 0] - T(e));
66
67 himp_.mettre_a_jour(temps);
68 Text_.mettre_a_jour(temps);
69}

107void Terme_Puissance_Thermique_Echange_Impose_P0_VDF::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
108{
109 int nb_elem=la_zone_VDF.valeur().nb_elem();
110 const Zone_VF& zone = la_zone_VDF.valeur();
111 const DoubleVect& volumes = zone.volumes();
112 const DoubleTab& himp = himp_.valeur().valeurs();
113 const DoubleTab& Text = Text_.valeur().valeurs();
114 const DoubleTab& T = equation().inconnue().valeurs();
115 const std::string& nom_inco = equation().inconnue().le_nom().getString();
116 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
117

120 if(mat) (*mat)(e, e) += volumes(e) * himp.addr()[himp.dimension(0) > 1 ? e : 0];
121 secmem(e) -= volumes(e) * himp.addr()[himp.dimension(0) > 1 ? e : 0] * (T(e) - Text.addr()[Text.dimension(0) > 1 ? e : 0]);
