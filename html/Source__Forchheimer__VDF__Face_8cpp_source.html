26Entree& Source_Forchheimer_VDF_Face::readOn(Entree& is )
27{
28 Motcles les_mots(3);
29 {
30 les_mots[0] = "modele_K";
31 les_mots[1] = "Cf";
32 les_mots[2] = "porosite";

34
35 Motcle motlu, accolade_fermee="}", accolade_ouverte="{";
36 is >> motlu;
37 if(motlu!=accolade_ouverte)
38 {
39 Cerr << "On attendait une { a la lecture d'une " << que_suis_je() << finl;
40 Cerr << "et non : " << motlu << finl;
41 exit();
42 }
43 is >> motlu;
44 while (motlu != accolade_fermee)

46 int rang=les_mots.search(motlu);
47 switch(rang)
48 {
49 case 0 :
50 {
51 Motcle mot;
52 is >> mot;
53 eval().modK.typer(mot);
54 is >> eval().modK.valeur();
55 break;
56 }
57 case 1 :
58 {
59 double c;
60 is >> c;
61 eval().setCf(c);
62 break;
63 }
64 case 2 :
65 {
66 double c;
67 is >> c;
68 eval().setPorosite(c);
69 break;
70 }
71 default :
72 {
73 Cerr << "Unknown keyword in Source_Forchheimer_VDF_Face " << finl;
74 exit();
75 }
76 }
77 is >> motlu;
78 }
79 set_fichier("Forcheimer");
80 set_description("Forchheimer term = Integral(-Cf/K*abs(vitesse)*vitesse*dv) [m/s2]");
