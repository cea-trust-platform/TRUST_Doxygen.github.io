57inline double Eval_Dift_VDF_const_Face::tau_tan_impl(int face, int k) const
58{
59 const int nb_faces = la_zone.valeur().nb_faces();
60 const ArrOfInt& ind_faces_virt_bord = la_zone.valeur().ind_faces_virt_bord();
61 int f = (face >= tau_tan_.dimension(0)) ? ind_faces_virt_bord[face-nb_faces] : face;
62
63 if(f >= tau_tan_.dimension_tot(0))
64 {
65 Cerr << "Erreur dans tau_tan " << finl;
66 Cerr << "dimension : " << tau_tan_.dimension(0) << finl;
67 Cerr << "dimension_tot : " << tau_tan_.dimension_tot(0) << finl;
68 Cerr << "face : " << face << finl;
69 Process::exit();
70 }
71 return tau_tan_(f,k);
72}
