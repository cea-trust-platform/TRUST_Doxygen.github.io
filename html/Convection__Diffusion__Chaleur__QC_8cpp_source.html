48int Convection_Diffusion_Chaleur_QC::lire_motcle_non_standard(const Motcle& mot, Entree& is)
49{
50 if (mot=="convection")
51 {
52 Convection_Diffusion_Chaleur_Fluide_Dilatable_base::lire_motcle_non_standard(mot,is);
53
54// l'equation de la chaleur en quasi compressible contient un terme source dP/dt
55 Cerr << "Source term creation of the energy equation :"<< finl;
56 Source t;
57 Source& so=les_sources.add(t);
58 Nom type_so = "Source_QC_Chaleur_";
59 Nom disc = discretisation().que_suis_je();
60 if (disc=="VEFPreP1B") disc = "VEF";
61 type_so += disc;
62 so.typer_direct(type_so);
63 so->associer_eqn(*this);
64 Cerr<<so->que_suis_je()<<finl;
65 Cerr << "Typing of the convection operator : ";
66 return 1;
67 }
68 else if (mot=="mode_calcul_convection")
69 {
70 if (terme_convectif.non_nul())
71 {
72 Cerr<<" The option "<<mot<<" must be indicated before the convection operator."<<finl;
73 Cerr<<" Please modify your data set."<<finl;
74 exit();
75 }
76 Motcles modes(3);
77 modes[0]="ancien"; //!< Default mode before 1.6.8 (wrong in VEF)
78 modes[1]="divuT_moins_Tdivu"; //!< OK in VDF and VEF
79 modes[2]="divrhouT_moins_Tdivrhou"; //!< Default mode after 1.6.8 (better)
80 Motcle mot2;
81 is>>mot2;
82 mode_convection_=modes.search(mot2);
83 if (mode_convection_==0)
84 {
85 Cerr << "WARNING! This mode is not supported anymore." << finl;
86 }
87 if (mode_convection_==-1)
88 {
89 Cerr<<" Allowed keywords for option mode_calcul_convection are :"<<finl;
90 Cerr<<modes<<finl;
91 exit();
92 }
93 return 1;
94 }
95 else
96 return Convection_Diffusion_Chaleur_Fluide_Dilatable_base::lire_motcle_non_standard(mot,is);
97 return 1;
98}

111void Convection_Diffusion_Chaleur_QC::calculer_div_u_ou_div_rhou(DoubleTab& Div) const
112{
113// No convective operator:
114 if (sub_type(Op_Conv_negligeable,operateur(1).l_op_base()))
115 {
116 Div=0;
117 return;
118 }
119
120 if (mode_convection_!=0)
121 Convection_Diffusion_Fluide_Dilatable_Proto::calculer_div_rho_u_impl(Div,*this);
122 else
123 {
124// Compute Div([rho]u) with Divergence operator
125 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,probleme().equation(0));
126 const DoubleTab& pression = eqn_hydr.pression().valeurs();
127 const Champ_Inc& vitesse = eqn_hydr.inconnue();
128
129// Div = Div([rho]u) located on pressure nodes:
130 DoubleTab Div_on_pressure_nodes(pression);
131 eqn_hydr.operateur_divergence().calculer(vitesse.valeurs(),Div_on_pressure_nodes);
132
133// Test on the discretization:
134 if (vitesse.valeurs().line_size() > 1) //!< VEF
135 {
136// VEF (temperature and pressure are not the same) => Div extrapolated on velocity nodes:
137 DoubleTab Div_on_temperature_nodes(inconnue().valeurs());
138 Div_on_pressure_nodes.echange_espace_virtuel();
139 ref_cast_non_const(Fluide_Quasi_Compressible,le_fluide.valeur()).divu_discvit(Div_on_pressure_nodes,Div_on_temperature_nodes);
140 const int nsom = Div_on_temperature_nodes.dimension_tot(0);
141 for (int i=0 ; i<nsom ; i++) Div(i)=Div_on_temperature_nodes(i);
142 }
143 else//!< VDF (temperature and pressure nodes are the same):
144 Div = Div_on_pressure_nodes;
145
146 Div *= -1;
147 }
148}
