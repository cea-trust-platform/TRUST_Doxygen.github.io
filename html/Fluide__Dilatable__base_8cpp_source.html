43void Fluide_Dilatable_base::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
44{
45 Cerr<<"Fluide_Dilatable_base::discretiser"<<finl;
46
47 const Zone_dis_base& zone_dis=pb.equation(0).zone_dis();
48 double temps=pb.schema_temps().temps_courant();
49
50// les champs seront nommes par le milieu_base
51 Champ_Don ch_rho;
52 dis.discretiser_champ("temperature",zone_dis,"masse_volumique_p","neant",1,temps,ch_rho);
53 rho = ch_rho.valeur();
54
55 Champ_Don& cp = capacite_calorifique();
56 if (!cp.non_nul() || !sub_type(Champ_Uniforme,cp.valeur())) //ie Cp non constant : gaz reels
57 {
58 Cerr<<"Heat capacity Cp is discretized once more for space variable case."<<finl;
59 dis.discretiser_champ("temperature",zone_dis,"cp_prov","neant",1,temps,cp);
60 }
61
62 if (!lambda.non_nul() || ((!sub_type(Champ_Uniforme,lambda.valeur())) && (!sub_type(Champ_Fonc_Tabule,lambda.valeur()))))
63 {
64// cas particulier etait faux en VEF voir quand cela sert (FM slt) : sera nomme par milieu_base
65 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,lambda);
66 }
67
68 dis.discretiser_champ("vitesse", zone_dis,"rho_comme_v","kg/m3",1,temps,rho_comme_v);
69 champs_compris_.ajoute_champ(rho_comme_v);
70
71 dis.discretiser_champ("champ_elem",zone_dis,"mu_sur_Schmidt","kg/(m.s)",1,temps,mu_sur_Sc);
72 champs_compris_.ajoute_champ(mu_sur_Sc);
73
74 dis.discretiser_champ("champ_elem",zone_dis,"nu_sur_Schmidt","m2/s",1,temps,nu_sur_Sc);
75 champs_compris_.ajoute_champ(nu_sur_Sc);
76
77 Champ_Don& ptot = pression_tot();
78 dis.discretiser_champ("champ_elem",zone_dis,"pression_tot","Pa",1,temps,ptot);
79 champs_compris_.ajoute_champ(ptot);
80
81 dis.discretiser_champ("temperature",zone_dis,"rho_gaz","kg/m3",1,temps,rho_gaz);
82 champs_compris_.ajoute_champ(rho_gaz);
83
84 Fluide_base::discretiser(pb,dis);
85}

103int Fluide_Dilatable_base::lire_motcle_non_standard(const Motcle& mot, Entree& is)
104{
105 Motcle motlu;
106 if (mot=="loi_etat")
107 {
108 is>>loi_etat_;
109 loi_etat_->associer_fluide(*this);
110 return 1;
111 }
112 else if (mot=="mu")
113 {
114 is>>mu;
115 mu->nommer("mu");
116 return 1;
117 }
118 else if (mot=="sutherland")
119 {
120 double mu0,T0,C=-1,Slambda=-1;
121 Nom prob;
122 is>>prob;
123 is>>motlu;
124 if (motlu!="MU0") warn_syntax_Sutherland();
125 is>>mu0;
126 is>>motlu;
127 if (motlu!="T0") warn_syntax_Sutherland();
128 is>>T0;
129 is>>motlu;
130 if (motlu=="SLAMBDA")
131 {
132 is >> Slambda;
133 is >> motlu;
134 }
135 if (motlu!="C") warn_syntax_Sutherland();
136 is>>C;
137
138 mu.typer("Sutherland");
139 Sutherland& mu_suth = ref_cast(Sutherland,mu.valeur());
140 mu_suth.set_val_params(prob,mu0,C,T0);
141 mu_suth.lire_expression();
142
143// On stocke la valeur de C (ici Slambda) pour construire(cf creer_champs_non_lus())
144// la loi de Sutherland qui concerne la conductivite
145 if (Slambda!=-1)
146 {
147 lambda.typer("Sutherland");
148 Sutherland& lambda_suth = ref_cast(Sutherland,lambda.valeur());
149 lambda_suth.set_prob(prob);
150 lambda_suth.set_Tref(T0);
151 lambda_suth.set_C(Slambda);
152 }
153 return 1;
154 }
155 else if ((mot=="beta_th") || (mot=="beta_co"))
156 {
157 Cerr<<"The keyword "<<mot<<" has not to be read for a "<<que_suis_je()<<" type medium."<<finl;
158 Cerr<<"Please remove it from your data set."<<finl;
159 Process::exit();
160 return -1;
161 }
162 else return Fluide_base::lire_motcle_non_standard(mot,is);
163}

253void Fluide_Dilatable_base::update_rho_cp(double temps)
254{
255 rho_cp_comme_T_.changer_temps(temps);
256 rho_cp_comme_T_.valeur().changer_temps(temps);
257 DoubleTab& rho_cp = rho_cp_comme_T_.valeurs();
258 if (sub_type(Champ_Uniforme,rho))

261 {
262// AB: rho_cp = rho.valeurs() turns rho_cp into a 2 dimensional array with 1 compo. We want to stay mono-dim:

265 }
266 if (sub_type(Champ_Uniforme, Cp.valeur()))
267 rho_cp *= Cp.valeurs()(0, 0);
268 else
269 tab_multiply_any_shape(rho_cp,Cp.valeurs());
270}

341void Fluide_Dilatable_base::creer_champs_non_lus()
342{
343// on s'occupe de lamda si mu uniforme et CP uniforme
344// on type lambda en champ uniforme et on met lambda=mu*Cp/Pr
345//
346 if (mu.non_nul())
347 {
348 if (!(lambda.non_nul())||(!sub_type(Champ_Fonc_Tabule,lambda.valeur())))
349 if ((sub_type(Champ_Uniforme,mu.valeur()))&&(sub_type(Loi_Etat_GP_base,loi_etat_.valeur())))
350 {
351 if (!sub_type(Loi_Etat_Multi_GP_QC,loi_etat_.valeur()))
352 {
353// Si mu uniforme et si la loi d'etat est celle d'un gaz parfait
354 double lold=-1;
355 if (lambda.non_nul())
356 lold=lambda.valeurs()(0,0);
357 lambda.typer(mu.valeur().le_type());
358 lambda=mu;
359
360 loi_etat_->calculer_lambda();
361 double lo=lambda.valeurs()(0,0);
362 if (lold!=-1)
363 {
364 if (!est_egal(lold,lo))
365 {
366 Cerr << "Error : mu, lambda, Cp and Prandtl are all specified in your data set." << finl;
367 Cerr << "It is observed that your imput values leads to :" << finl;
368 Cerr.precision(20);
369 Cerr << "mu/(lambda*Cp)=" << loi_etat_->Prandt()*(lo/lold) << " and Prandtl=" << loi_etat_->Prandt() << finl;
370 Cerr<<"Please modify your data set by specifying for instance Prandtl = "<<loi_etat_->Prandt()*(lo/lold)<<finl;
371 Process::exit();
372 }
373 }
374 }
375 }
376
377 if (lambda.non_nul())
378 {
379 if (sub_type(Sutherland,lambda.valeur()))
380 {
381 if (!sub_type(Champ_Uniforme,Cp.valeur()))
382 {
383 Cerr << "A sutherland law cannot be requested for the conductivity "<<finl;
384 Cerr << "by indicating Slambda if the heat capacity (Cp) is not uniform."<<finl;
385 Process::exit();
386 }
387 Sutherland& mu_suth = ref_cast(Sutherland,mu.valeur());
388 const double mu0 = mu_suth.get_A();
389 Sutherland& lambda_suth = ref_cast(Sutherland,lambda.valeur());
390 double lambda0 = mu0/loi_etat_->Prandt()*Cp(0,0);
391 lambda_suth.set_A(lambda0);
392 lambda_suth.lire_expression();
393 }
394 }
395 }
396}
