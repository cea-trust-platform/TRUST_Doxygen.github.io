71int EcrFicPartageMPIIO::ouvrir(const char* name,IOS_OPEN_MODE mode)
72{
73 MPI_Comm mpi_comm;
74 if (sub_type(Comm_Group_MPI,PE_Groups::current_group()))
75 mpi_comm = ref_cast(Comm_Group_MPI,PE_Groups::current_group()).get_mpi_comm();
76 else
77 mpi_comm = MPI_COMM_WORLD;
78 int MPI_OPEN = mode == ios::app ? MPI_MODE_APPEND : MPI_MODE_CREATE;
79 int MPI_MODE = (mode == ios::app || ios::out) ? MPI_MODE_WRONLY : MPI_MODE_RDONLY;
80 MPI_CHECK(MPI_File_open(mpi_comm, (char*)name, MPI_OPEN|MPI_MODE, MPI_INFO_NULL, &mpi_file_));
81// Set MPI errors fatal:
82 MPI_File_set_errhandler(mpi_file_,MPI_ERRORS_ARE_FATAL);
83// Set initial displacement:
84 disp_=0;
85// Ecriture d'un entete specifique pour le build int64 ?
86 if (mode==ios::out)
87 {
88 if (je_suis_maitre())
89 {
90 Nom marq("INT64");
91#ifdef INT_is_64_
92 (*this)<<marq.getChar();
93#endif
94 }
95 }
96 /*
97 else
98 {
99 Cerr<<"not coded in EcrFicPartageMPIIO::ouvrir"<<finl;
100 exit();
101 }*/
102 return 1;
103}

111void EcrFicPartageMPIIO::check()
112{
113 if (Process::me()>0)
114 {
115 Cerr << "Only master process can call EcrFicPartageMPIIO::operator <<(...)" << finl;
116 exit();
117 }
118 MPI_Offset offset; //!< Relative position
119 MPI_File_get_position(mpi_file_, &offset);
120 MPI_Offset disp; //!< Absolute position
121 MPI_File_get_byte_offset(mpi_file_, offset, &disp);
122 if (disp_!=disp)
123 {
124 Cerr << "Error in EcrFicPartageMPIIO::check()" << finl;
125 Cerr << "Decalage : disp_= " << (int)disp_ << " disp=" << (int)disp << finl;
126 Cerr << "Contact TRUST support." << finl;
127 }
128}

222// Function used for different MPI_Type
223// Good info/examples on MPI-IO:
224// See page 213: http://www.idris.fr/data/cours/parallel/mpi/IDRIS_MPI_cours_couleurs.pdf
225// See http://www.cac.cornell.edu/ranger/mpiadvtopics/fileptroffs.aspx
226// Default file view has etype=filetype=MPI_Byte and disp=0
227// MPI_File_get_position returns a relative offset in etype units of the current view
228// MPI_File_get_byte_offset converts a view relative offset (etype units) into an absolute byte position
229// MPI_File_set_view(file,disp,etype,...) disp should be specified in absolute bytes from the start of the file
230// sizeof(MPI_DOUBLE)=4 !!! Should use:MPI_Type_size( MPI_DOUBLE, &size ) to return 8 !!!
231// Ecriture d'une zone memoire pointee par ob contenant n MPI_TYPE :
232int EcrFicPartageMPIIO::put(MPI_Datatype MPI_TYPE, const void* ob, int n)
233{
234 MPI_Datatype etype;
235 etype=MPI_TYPE;
236 True_int sizeof_etype;
237 MPI_Type_size(etype,&sizeof_etype);
238
239// filetype is n etype contigous:
240 MPI_Datatype filetype;
241 MPI_Type_contiguous(n, etype, &filetype);
242 MPI_Type_commit(&filetype);
243
244// Before collecting operations, update disp_ on all processes:
245 envoyer_broadcast(disp_, 0);
246 MPI_Offset disp_me = disp_ + mppartial_sum(n) * sizeof_etype;
247// ROMIO hints:
248 if (Process::nproc()>1024)
249 {
250 char* ROMIO_HINTS=getenv("ROMIO_HINTS");
251 if (ROMIO_HINTS==NULL)
252 {
253 Cerr << "Warning, no ROMIO_HINTS detected on your massive parallel calculation." << finl;
254 Cerr << "Performances of MPI I/O could be improved with ROMIO hints." << finl;
255 Cerr << "Contact TRUST support." << finl;
256 }
257 }
258
259// Create info for using hints:
260 MPI_Info mpi_info;
261 MPI_Info_create(&mpi_info);
262 /*
263 MPI_Info_set(mpi_info, "romio_cb_read", "enable");
264 MPI_Info_set(mpi_info, "romio_cb_write", "enable");
265 // Un processus par node s'occupe du buffering:
266 MPI_Info_set(mpi_info, "cb_config_list", "*:1");
267
268 // Striping count (max 160 sur LUSTRE):
269 Nom striping_factor=(Nom)min(Process::nproc(),160);
270 MPI_Info_set(mpi_info, "striping_factor", (char*)striping_factor.getChar());
271
272 // Striping units in bytes (ideal is stripe size = I/O operations):
273 Nom striping_unit=(Nom)(n * sizeof_etype);
274 MPI_Info_set(mpi_info, "striping_unit", (char*)striping_unit.getChar());
275// Example (not used cause machine dependant):
276 */
277
278// Set a new view:
279 MPI_File_set_view(mpi_file_, disp_me, etype, filetype, (char*)"native", mpi_info);
280
281// Write all:
282 statistiques().begin_count(IO_EcrireFicPartageMPIIO_counter_);
283 MPI_File_write_all(mpi_file_, (void*)ob, n, etype, MPI_STATUS_IGNORE);
284 statistiques().end_count(IO_EcrireFicPartageMPIIO_counter_, n * sizeof_etype);
285
286// Update the position of the pointer file:
287 disp_+=mp_sum(n) * sizeof_etype;
288
289// Reset a simple view at this new position:
290 MPI_File_set_view(mpi_file_, disp_, MPI_BYTE, MPI_BYTE, (char*)"native", MPI_INFO_NULL);
291
292// Free
293 MPI_Info_free(&mpi_info);
294 MPI_Type_free(&filetype);
295 return 1;
296}
