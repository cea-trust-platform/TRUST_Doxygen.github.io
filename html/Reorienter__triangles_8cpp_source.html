50Reorienter_triangles::Sens Reorienter_triangles::test_orientation_triangle(IntTab& les_elems, int num_element, const DoubleTab& coord_sommets) const
51{
52 static const int SOM_Z = 0;
53 static const int SOM_A = 1;
54 static const int SOM_B = 2;
55
56 const int som_Z = les_elems(num_element,SOM_Z);
57 const int som_A = les_elems(num_element,SOM_A);
58 const int som_B = les_elems(num_element,SOM_B);
59
60 double ZA0 = coord_sommets(som_A,0) - coord_sommets(som_Z,0);
61 double ZB0 = coord_sommets(som_B,0) - coord_sommets(som_Z,0);
62 double ZA1 = coord_sommets(som_A,1) - coord_sommets(som_Z,1);
63 double ZB1 = coord_sommets(som_B,1) - coord_sommets(som_Z,1);
64
65// calcul pdt vect ZAxZB
66 double pdtvect = ZA0*ZB1 - ZA1*ZB0;
67
68 if (pdtvect<0.)
69 {
70// le pdt scalaire est negatif : il s'agit d'un triangle mal oriente
71#ifdef _AFFDEBUG
72 {
73 Process::Journal<<" element "<<num_element<<" indirect"<<finl;
74 }
75#endif
76 return INDIRECT;
77 }
78#ifdef _AFFDEBUG
79 {
80 Process::Journal<<" element "<<num_element<<" direct"<<finl;
81 }
82#endif
83
84 return DIRECT;
85}

87Reorienter_triangles::Sens Reorienter_triangles::reorienter_triangle(IntTab& les_elems, int num_element) const
88{
89 static const int SOM_A = 1;
90 static const int SOM_B = 2;
91
92// pour reorienter le triangle, on va permuter les sommets 1 et 2
93 int tmp;
94 tmp = les_elems(num_element,SOM_A);
95 les_elems(num_element,SOM_A) = les_elems(num_element,SOM_B);
96 les_elems(num_element,SOM_B) = tmp;
97 return DIRECT;
98}

100// Cette methode permet de reorienter les triangles dans le sens direct
101void Reorienter_triangles::Reorienter(Domaine& dom) const
102{
103 const DoubleTab& coord_sommets = dom.coord_sommets();
104 int nb_zones = dom.nb_zones();
105 int iz;
106
107// on balaye les zones du domaine
108 for(iz=0 ; iz<nb_zones ; iz++)
109 {
110 Zone& zone = dom.zone(iz);
111 if (zone.type_elem()->que_suis_je() == "Triangle" )
112 {
113// zone de triangles
114 IntTab& les_elems = zone.les_elems();
115 int nb_elems = les_elems.dimension(0);
116 int ielem;
117
118// balaye les triangles
119 for (ielem=0 ; ielem<nb_elems ; ielem++)
120 {
121 if (test_orientation_triangle(les_elems, ielem, coord_sommets)==INDIRECT)
122 {
123// triangle oriente en sens indirect -> a reorienter
124 reorienter_triangle(les_elems, ielem);
125
126#ifdef _AFFDEBUG
127 {
128 Process::Journal<<" #element reoriente "<<ielem<<finl;
129 static const int SOM_Z = 0;
130 static const int SOM_A = 1;
131 static const int SOM_B = 2;
132 const int som_Z = les_elems(ielem,SOM_Z);
133 const int som_A = les_elems(ielem,SOM_A);
134 const int som_B = les_elems(ielem,SOM_B);
135 Process::Journal<<" somZ= "<<som_Z<<" coords= "<<coord_sommets(som_Z, 0)<<" "<<coord_sommets(som_Z, 1)<<finl;
136 Process::Journal<<" somA= "<<som_A<<" coords= "<<coord_sommets(som_A, 0)<<" "<<coord_sommets(som_A, 1)<<finl;
137 Process::Journal<<" somB= "<<som_B<<" coords= "<<coord_sommets(som_B, 0)<<" "<<coord_sommets(som_B, 1)<<finl;
138 }
139#endif
140 }
141 }
142 }
143 }//fin balayage zones
144}
