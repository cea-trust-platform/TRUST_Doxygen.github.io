113//! Reordonne et verifie que l'on a bien des rectangles.
114int Rectangle::reordonner_elem()
115{
116 Zone& zone=ma_zone.valeur();
117 const Domaine& dom=zone.domaine();
118 IntTab& elem=zone.les_elems();
119
120 ArrOfInt S(4);
121 ArrOfInt NS(4);
122 DoubleTab co(4,2);
123 int i,j;
124 int num_poly;
125 double xmin, ymin;
126 const int nb_elem=zone.nb_elem();
127 for (num_poly=0; num_poly<nb_elem; num_poly++)
128 {
129 NS=-1;
130 for(i=0; i<4; i++)
131 {
132 S[i] = elem(num_poly,i);
133 for(j=0; j<2; j++)
134 co(i,j) = dom.coord(S[i], j);
135 }
136
137 xmin=std::min(co(0, 0), co(1, 0));
138 xmin=std::min(xmin, co(2, 0));
139 ymin=std::min(co(0, 1), co(1, 1));
140 ymin=std::min(ymin, co(2, 1));
141
142 for(i=0; i<4; i++)
143 if ( est_egal(co(i, 0),xmin) && est_egal(co(i, 1),ymin))
144 NS[0]=S[i];
145 for(i=0; i<4; i++)
146 if ( !est_egal(co(i, 0),xmin) && est_egal(co(i, 1),ymin))
147 NS[1]=S[i];
148 for(i=0; i<4; i++)
149 if ( est_egal(co(i, 0),xmin) && !est_egal(co(i, 1),ymin))
150 NS[2]=S[i];
151 for(i=0; i<4; i++)
152 if ( !est_egal(co(i, 0),xmin) && !est_egal(co(i, 1),ymin))
153 NS[3]=S[i];
154
155// Si un sommet vaut -1, ce n'est pas un Rectangle !
156 if (min_array(NS)==-1)
157 {
158 Cerr << "The element " << num_poly << " is not a rectangle." << finl;
159 return -1;
160 }
161// Sinon on remplit elem
162 for(i=0; i<4; i++)
163 elem(num_poly, i)=NS[i];
164 }
165 return 0;
166}

281//! Calcule les volumes des elements de la zone associee.
282void Rectangle::calculer_volumes(DoubleVect& volumes) const
283{
284 if (dimension==3)
285 {
286 Cerr << "Rectangle::calculer_volumes is not supported for 3D yet." << finl;
287 exit();
288 }
289 const Zone& zone=ma_zone.valeur();
290 const Domaine& dom=zone.domaine();
291 double dx,dy;
292 int S1,S2,S3;
293
294 int size_tot = zone.nb_elem_tot();
295 assert(volumes.size_totale()==size_tot);
296 for (int num_poly=0; num_poly<size_tot; num_poly++)
297 {
298 S1 = zone.sommet_elem(num_poly,0);
299 S2 = zone.sommet_elem(num_poly,1);
300 S3 = zone.sommet_elem(num_poly,2);
301 dx = dom.coord(S2,0) - dom.coord(S1,0);
302 dy = dom.coord(S3,1) - dom.coord(S1,1);
303 volumes[num_poly]= dx*dy;
304 }
305}

325//! Calcule les normales aux faces des elements de la zone associee.
326void Rectangle::calculer_normales(const IntTab& Face_sommets ,
327 DoubleTab& face_normales) const
328{
329 const Zone& zone_geom = ma_zone.valeur();
330 const DoubleTab& les_coords = zone_geom.domaine().coord_sommets();
331 int nbfaces = Face_sommets.dimension(0);
332 double x1,y1;
333 int n0,n1;
334 for (int numface=0; numface<nbfaces; numface++)
335 {
336 n0 = Face_sommets(numface,0);
337 n1 = Face_sommets(numface,1);
338 x1 = les_coords(n0,0) - les_coords(n1,0);
339 y1 = les_coords(n0,1) - les_coords(n1,1);
340 face_normales(numface,0) = -y1;
341 face_normales(numface,1) = x1;
342 }
343}
