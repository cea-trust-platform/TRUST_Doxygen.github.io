45//! Renvoie 1 si le bit e est mis, 0 sinon.
46inline int ArrOfBit::operator[](int e) const
47{
48 assert(e >= 0 && e < taille);
49 unsigned int i = (unsigned int) e;
50 unsigned int x = data[i >> SIZE_OF_INT_BITS];
51 unsigned int flag = 1 << (i & DRAPEAUX_INT);
52 int resultat = ((x & flag) != 0) ? 1 : 0;
53 return resultat;
54}

56//! Met le bit e a 1.
57inline void ArrOfBit::setbit(int e) const
58{
59 assert(e >= 0 && e < taille);
60 unsigned int i = (unsigned int) e;
61 unsigned int flag = 1 << (i & DRAPEAUX_INT);
62 data[i >> SIZE_OF_INT_BITS] |= flag;
63}

65//! Renvoie la valeur du bit e, puis met le bit e a 1.
66inline int ArrOfBit::testsetbit(int e) const
67{
68 assert(e >= 0 && e < taille);
69 unsigned int i = (unsigned int) e;
70 unsigned int flag = 1 << (i & DRAPEAUX_INT);
71 int index = i >> SIZE_OF_INT_BITS;
72 unsigned int old = data[index];
73 data[index] = old | flag;
74 return ((old & flag) != 0) ? 1 : 0;
75}

77//! Met le bit e a 0.

79{

81 unsigned int i = (unsigned int) e;

83 data[i >> SIZE_OF_INT_BITS] &= ~flag;
84}
