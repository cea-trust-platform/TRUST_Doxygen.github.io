51//! Classe Terme_Boussinesq_base<br>Cette classe represente le terme de gravite qui figure dans l'equation<br>de la dynamique divisee par la masse volumique de reference.<br>On est dans le cadre de l'hypothese de Boussinesq:la masse volumique est<br>supposee constante et egale a sa valeur de reference sauf dans le terme<br>des forces de volumes ou on prend en compte une petite variation de la<br>masse volumique en fonction d'un ou de plusieurs scalaires transportes<br>par l'ecoulement (la temperature et/ou une ou plusieurs concentrations).<br>Cas particulier de Terme_Boussinesq_base pour la temperature<br>Le terme de gravite a pour expression: beta*(T-T0) ou beta<br>represente la dilatabilite de la masse volumique et T0 une valeur<br>de reference pour la temperature<br>Cas particulier de Terme_Boussinesq_base pour une concentration<br>Le terme de gravite a pour expression : beta*(C-C0)<br>ou (beta[0]*(C[0]-C0[0]) + .....+ beta[i]*(C[i]-C0[i])) dans<br>le cas d'un vecteur de concentrations<br>beta represente la variation de la masse volumique en fonction<br>de la concentration du constituant et C0 une valeur de reference<br>pour la concentration

107inline void Terme_Boussinesq_base::check() const
108{
109// Pas de verification autre qu'au premier pas de temps, si verification pas desactivee et uniquement pour la temperature
110 if (equation_scalaire().probleme().schema_temps().nb_pas_dt()>0 || equation_scalaire().probleme().reprise_effectuee() || verif_==0 || !sub_type(Convection_Diffusion_Temperature,equation_scalaire())) return;
111
112// Nouveau : on verifie que moyenne(T)==T0 au demarrage du calcul uniquement
113 const double T0 = Scalaire0(0);
114 double moyenne_T = mp_moyenne_vect(equation_scalaire().inconnue().valeurs());
115 if (inf_ou_egal(moyenne_T,T0-10) || sup_ou_egal(moyenne_T,T0+10))
116 {
117 Cerr << "New criteria in TRUST for the Boussinesq source term definition:" << finl;
118 Cerr << "To avoid an incorrect choice for T0 value" << finl;
119 Cerr << "the initial average temperature on the domain " << equation_scalaire().probleme().domaine().le_nom() << finl;
120 Cerr << "should be between +/-10 degrees around T0 value." << finl;
121 Cerr << "The initial average temperature is : " << moyenne_T << finl;
122 Cerr << "T0 value is : " << T0 << finl;
123 Cerr << "So, you need to change T0 or the initial temperature field to respect this criteria defined by default." << finl;
124 Cerr << "If you want to overcome this criteria, which it is not recommended," << finl;
125 Cerr << "you can specify, into the Boussinesq source term definition, the option:" << finl;
126 Cerr << "verif_boussinesq 0" << finl;
127 Process::exit();
128 }
129}

140// Methode de calcul de la valeur sur une face encadree par elem1 et elem2 d'un champ uniforme ou non a plusieurs composantes
141inline double valeur(const DoubleTab& valeurs_champ, int elem1, int elem2, const int compo)
142{
143 if (valeurs_champ.dimension(0)==1)
144 return valeurs_champ(0,compo); //!< Champ uniforme
145 else
146 {
147 if (elem2<0) elem2 = elem1; //!< face frontiere
148 if (valeurs_champ.nb_dim()==1)
149 return 0.5*(valeurs_champ(elem1)+valeurs_champ(elem2));
150 else
151 return 0.5*(valeurs_champ(elem1,compo)+valeurs_champ(elem2,compo));
152 }
153}
