48// Entree : Uk-1 ; Pk-1
49// Sortie Uk ; Pk
50// k designe une iteration
51void Solveur_U_P::iterer_NS(Equation_base& eqn,DoubleTab& current,DoubleTab& pression,double dt,Matrice_Morse& matrice_inut,double seuil_resol,DoubleTrav& secmem,int nb_ite,int& converge, int& ok)
52{
53 if (eqn.probleme().is_dilatable())
54 {
55 Cerr<<" Solveur_U_P cannot be used with a quasi-compressible fluid."<<finl;
56 Cerr<<__FILE__<<(int)__LINE__<<" non code" <<finl;
57 Process::exit();
58 }
59
60 Parametre_implicite& param = get_and_set_parametre_implicite(eqn);
61 SolveurSys& le_solveur_ = param.solveur();
62
63 Navier_Stokes_std& eqnNS = ref_cast(Navier_Stokes_std,eqn);
64
65 /* MD_Vector (vitesse, pression) */
66 MD_Vector_composite mds;
67 mds.add_part(current.get_md_vector(), current.line_size());
68 mds.add_part(pression.get_md_vector(), pression.line_size());
69 MD_Vector md_UP;
70 md_UP.copy(mds);
71
72 DoubleTab Inconnues,residu;
73 MD_Vector_tools::creer_tableau_distribue(md_UP, Inconnues);
74 MD_Vector_tools::creer_tableau_distribue(md_UP, residu);
75
76 DoubleTab_parts residu_parts(residu);
77 DoubleTab_parts Inconnues_parts(Inconnues);
78
79 Inconnues_parts[0] = current, Inconnues_parts[1] = pression;
80
81 Matrice_Bloc Matrice_global(2,2) ; //matrice M.(du, dp) = (Navier-Stokes, divergence)
82
83 /* ligne Navier-Stokes : blocs N-S, bloc gradient */
84 Matrice_global.get_bloc(0,0).typer("Matrice_Morse");
85 Matrice_Morse& matrice=ref_cast(Matrice_Morse, Matrice_global.get_bloc(0,0).valeur());
86 Matrice_global.get_bloc(0,1).typer("Matrice_Morse");
87 Matrice_Morse& mat_grad=ref_cast(Matrice_Morse, Matrice_global.get_bloc(0,1).valeur());
88
89 if (eqnNS.has_interface_blocs()) /* si interface_blocs : on peut remplir les deux d'un coup */
90 {
91 eqnNS.dimensionner_blocs({{ "vitesse", &matrice }, { "pression", &mat_grad }});
92 eqnNS.assembler_blocs_avec_inertie({{ "vitesse", &matrice }, { "pression", &mat_grad }}, residu_parts[0]);
93 }
94 else /* sinon : moins elegant */
95 {
96 Operateur_Grad& gradient = eqnNS.operateur_gradient();
97
98 eqnNS.dimensionner_matrice(matrice);
99 eqnNS.assembler_avec_inertie(matrice,current,residu_parts[0]);
100 gradient.valeur().dimensionner( mat_grad);
101 gradient.valeur().contribuer_a_avec(pression, mat_grad);
102 mat_grad.get_set_coeff()*=-1;
103 /* pour repasser en increments */
104 residu_parts[0]*=-1;
105 matrice.ajouter_multvect(current, residu_parts[0]);
106 gradient.valeur().ajouter(pression,residu_parts[0]);
107 residu_parts[0]*=-1;
108 }
109
110 /* doit-on fixer P(elem 0) = 0 ? */
111 int has_P_ref=0;
112 const Conds_lim& cls = eqnNS.zone_Cl_dis().les_conditions_limites();
113 for (int n_bord=0; n_bord < cls.size(); n_bord++)
114 if (sub_type(Neumann_sortie_libre,cls[n_bord].valeur())) has_P_ref=1;
115
116 /* ligne de masse : (div, 0) */
117 Operateur_Div_base& divergence = eqnNS.operateur_divergence().valeur();
118 Matrice_global.get_bloc(1,0).typer("Matrice_Morse");
119 Matrice_Morse& mat_div_v = ref_cast(Matrice_Morse, Matrice_global.get_bloc(1,0).valeur());
120 if (divergence.has_interface_blocs()) /* si interface_blocs : direct */
121 {
122 Matrice_global.get_bloc(1,1).typer("Matrice_Morse");
123 Matrice_Morse& mat_div_p = ref_cast(Matrice_Morse, Matrice_global.get_bloc(1,1).valeur());
124 divergence.dimensionner_blocs({ { "vitesse", &mat_div_v } , { "pression", &mat_div_p}});
125 divergence.ajouter_blocs({ { "vitesse", &mat_div_v } , { "pression", &mat_div_p}}, residu_parts[1]);
126 if (!has_P_ref && !Process::me()) mat_div_p(0, 0) += 1; //revient a imposer P(0) = 0
127 }
128 else /* sinon : l'operateur remplit mat_div_v, on construit a la main mat_div_p = 0 */
129 {
130 divergence.dimensionner(mat_div_v);
131 divergence.contribuer_a_avec(current, mat_div_v);
132 divergence.calculer(current, residu_parts[1]);
133 Matrice_global.get_bloc(1,1).typer("Matrice_Diagonale");
134 Matrice_Diagonale& mat_div_p = ref_cast(Matrice_Diagonale,Matrice_global.get_bloc(1,1).valeur());
135 mat_div_p.dimensionner(mat_div_v.nb_lignes());
136 if (!has_P_ref && !Process::me()) mat_div_p.coeff(0, 0) += 1; //revient a imposer P(0) = 0
137 }
138
139 le_solveur_.valeur().reinit();
140 le_solveur_.valeur().resoudre_systeme(Matrice_global,residu,Inconnues);
141
142// Calcul de Uk = U*_k + U'k
143 current += Inconnues_parts[0];
144 pression += Inconnues_parts[1];
145// current.echange_espace_virtuel();
146 Debog::verifier("Solveur_U_P::iterer_NS current",current);
147 eqn.solv_masse().corriger_solution(current, current); //PolyMAC_P0 : mise en coherence de ve avec vf
148 eqnNS.assembleur_pression().modifier_solution(pression);
149
150 if (1)
151 {
152// DoubleTrav secmem(current);
153 divergence.calculer(current, secmem);
154 Cerr<<" apresdiv "<<mp_max_abs_vect(secmem)<<finl;;
155 }
156}
