47void Interpolation_IBM_elem_fluid::discretise(const Discretisation_base& dis, Zone_dis_base& la_zone_EF)
48{
49 int nb_comp = Objet_U::dimension;
50 Noms units(nb_comp);
51 Noms c_nam(nb_comp);
52
53 dis.discretiser_champ("champ_sommets",la_zone_EF,"fluid_elems","none",1,0., fluid_elems_);
54 fluid_elems_.valeur().affecter(fluid_elems_lu_);
55 if ((&corresp_elems_lu_)->non_nul())
56 {
57 dis.discretiser_champ("champ_elem",la_zone_EF,"corresp_elems","none",1,0., corresp_elems_);
58 corresp_elems_.valeur().affecter(corresp_elems_lu_);
59 }
60 dis.discretiser_champ("vitesse",la_zone_EF,vectoriel,c_nam,units,nb_comp,0.,fluid_points_);
61 fluid_points_.valeur().affecter(fluid_points_lu_);
62 dis.discretiser_champ("vitesse",la_zone_EF,vectoriel,c_nam,units,nb_comp,0.,solid_points_);
63 solid_points_.valeur().affecter(solid_points_lu_);
64 computeFluidElems(la_zone_EF);
65}

67void Interpolation_IBM_elem_fluid::computeFluidElems(Zone_dis_base& la_zone_EF)
68{
69 double eps = 1e-12;
70 int nb_som = la_zone_EF.nb_som();
71 int nb_som_tot = la_zone_EF.nb_som_tot();
72 int nb_elem = la_zone_EF.nb_elem();
73 int nb_elem_tot = la_zone_EF.nb_elem_tot();
74 const DoubleTab& coordsDom = la_zone_EF.zone().domaine().coord_sommets();
75// const IntTab& elems = la_zone_EF.zone().les_elems();
76
77 DoubleTab& elems_fluid_ref = fluid_elems_.valeur().valeurs();
78 DoubleTab& fluid_points_ref = fluid_points_.valeur().valeurs();
79 DoubleTab& solid_points_ref = solid_points_.valeur().valeurs();
80
81// La numerotation des elements peut avoir changee entre l'etape pre_pro et
82// l'etape calcul (cas du pre_pro Salome)
83// On utilise un champ d'etiquette pour fluid_elems_ (par exemple le no elem Salome)
84// et un champ d'element reprenant ces etiquettes
85 if ((&corresp_elems_)->non_nul())
86 {
87 DoubleTab& corresp_elems_ref = corresp_elems_.valeur().valeurs();
88 int nb_tag_max = -1;
89 for (int i = 0 ; i < nb_elem_tot ; i++)
90 {
91 int int_cor_elem = lrint(corresp_elems_ref(i));
92 if (int_cor_elem > nb_tag_max) nb_tag_max = int_cor_elem;
93 }
94 int dimtag = nb_tag_max+1;
95 DoubleTrav elems_fluid_trust(dimtag);
96 elems_fluid_trust = -1.e+10;
97 for (int i = 0 ; i < nb_elem_tot ; i++)
98 {
99 int indextag = lrint(corresp_elems_ref(i)) ;
100// Cerr << indextag << ", " << dimtag << ", " << nb_elem << ", " << nb_elem_tot << ", " << nb_som << ", " << nb_som_tot << finl;
101// abort();
102 if (indextag < dimtag && indextag >= 0)
103 {
104 elems_fluid_trust(indextag) = i*1.0;
105 }
106 else
107 {
108 Cerr<<"erreur in computeFluidElems : elem corresp_elems_ref(elem) = "<<i<<" "<<corresp_elems_ref(i)<<" ; indextag = "<<indextag<<" < 0 ou >= "<<dimtag<<finl;
109 exit();
110 }
111 }
112 for (int i = 0 ; i < nb_som ; i++)
113 {
114 if (elems_fluid_ref(i) >= 0.0)
115 {
116 int indexr = lrint(elems_fluid_ref(i));
117 if (indexr < dimtag && indexr >= 0)
118 {
119 if (elems_fluid_trust(indexr) >= 0.)
120 {
121 elems_fluid_ref(i) = elems_fluid_trust(indexr);
122 }
123 else
124 {
125 double x = fluid_points_ref(i,0);
126 double y = fluid_points_ref(i,1);
127 double z = fluid_points_ref(i,2);
128 double xs = solid_points_ref(i,0);
129 double ys = solid_points_ref(i,1);
130 double zs = solid_points_ref(i,2);
131 double d = (x-xs)*(x-xs)+(y-ys)*(y-ys)+(z-zs)*(z-zs);
132 if (d < eps)
133// A ne pas considerer si points fluide et solide sont confondus (erreur prepro)
134 {
135 elems_fluid_ref(i) = elems_fluid_trust(indexr);
136 }
137 else
138 {
139 Cerr << __FILE__ << (int)__LINE__ << "Interpolation_IBM_elem_fluid::computeFluidElems : ERROR : joint width too low?" << finl;
140 Cerr<<"node elems_fluid_ref(node) index = "<<i<<" "<<elems_fluid_ref(i)<<" "<<indexr<<finl;
141 Cerr<<"elems_fluid_trust(index) < 0. = "<<elems_fluid_trust(indexr)<<finl;
142 Cerr<<"nb som som_tot elem elem_tot = "<<nb_som<<" "<<nb_som_tot<<" "<<nb_elem<<" "<<nb_elem_tot<<finl;
143 Cerr<<"coords_point(node) : x y z = "<<coordsDom(i,0)<<" "<<coordsDom(i,1)<<" "<<coordsDom(i,2)<<finl;
144 Cerr<<"fluid_points(node) : xf yf zf = "<<x<<" "<<y<<" "<<z<<finl;
145 Cerr<<"solid_points(node) : xs ys zs = "<<xs<<" "<<ys<<" "<<zs<<finl;
146 int elem_found = la_zone_EF.zone().chercher_elements(x,y,z);
147 Cerr<<"chercher_elements(xf,yf,zf) = "<<elem_found<<finl;
148 exit();
149 }
150 }
151 }
152 else
153 {
154 Cerr << __FILE__ << (int)__LINE__ << "Interpolation_IBM_elem_fluid::computeFluidElems : ERROR : joint width too low?" << finl;
155 Cerr<<"node elems_fluid_ref(node) = "<<i<<" "<<elems_fluid_ref(i)<<finl;
156 Cerr<<"index = "<<indexr<<" < 0 ou >= "<<dimtag<<finl;
157 exit();
158 }
159 }
160 }
161 }
