62//! Calcule la moyenne volumique de la grandeur P0 donnee
63/** \return DoubleTab\&<br>
64Signification: rho discretise par face<br>
65Contraintes: */
66double EOS_Tools_VDF::moyenne_vol(const DoubleTab& tab) const
67{
68 int nb_elem=la_zone->nb_elem();
69 const DoubleVect& volumes = la_zone->volumes();
70 assert(tab.dimension(0)==nb_elem);
71 ArrOfDouble sum(2);
72 sum = 0;
73 for (int elem=0 ; elem<nb_elem ; elem++)
74 {
75 double v = volumes(elem);
76 sum[0] += v;
77 sum[1] += v*tab(elem);
78 }
79 mp_sum_for_each_item(sum);
80 return sum[1]/sum[0];
81}

84{

86 Debog::verifier("tab_rhoP0",tab_rhoP0);

184//! Calcule le second membre de l'equation de continuite :<br>div(rhoU) = W = -dZ/dT avec Z=rho
185void EOS_Tools_VDF::secmembre_divU_Z(DoubleTab& tab_W) const
186{
187 double dt = le_fluide().vitesse()->equation().schema_temps().pas_de_temps();
188 int elem,nb_elem = la_zone->nb_elem(),nb_faces = la_zone->nb_faces();
189 DoubleVect tab_dZ(nb_elem);
190 DoubleTab tab_gradZ(nb_faces);
191 const DoubleTab& tab_rhonP0 = le_fluide().loi_etat()->rho_n();
192 const DoubleTab& tab_rhonp1P0 = le_fluide().loi_etat()->rho_np1();
193 Debog::verifier("divU tab_rhonP0",tab_rhonP0);
194 Debog::verifier("divU tab_rhonp1P0",tab_rhonp1P0);
195 const DoubleVect& volumes = la_zone->volumes();
196
197 for (elem=0 ; elem<nb_elem ; elem++)
198 tab_dZ(elem) = (tab_rhonp1P0(elem)-tab_rhonP0(elem))/dt;
199
200 double tmp;
201 for (elem=0 ; elem<nb_elem ; elem++)
202 {
203 tmp = tab_dZ(elem) ;
204 tab_W(elem) = -tmp * volumes(elem);
205 }
206}
