204DoubleTab& Op_Dift_EF_Q1::ajouter_new(const DoubleTab& tab_inconnue, DoubleTab& resu) const
205{
206 Cerr<<"NEW"<<finl;
207 remplir_nu(nu_);
208// const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
209// const Zone_EF& zone_EF = la_zone_EF.valeur();
210
211 DoubleVect diffu_turb(diffusivite_turbulente()->valeurs());
212 DoubleTab diffu(nu_);
213
214 const int N = resu.line_size();
215 ArrOfInt marqueur_neuman;
216 ArrOfInt marqueur_paroi=0;
217 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
218 if(N > 1)
219 {
220 remplir_marqueur_sommet_neumann( marqueur_neuman,zone_ef,la_zcl_EF.valeur(),transpose_partout_ );
221 remplir_marqueur_elem_CL_paroi( marqueur_paroi,zone_ef,la_zcl_EF.valeur() );
222 }
223
224 const DoubleVect& volumes= zone_ef.volumes();
225
226 const DoubleTab& bij=zone_ef.Bij();
227 const DoubleTab& bij_thilde=zone_ef.Bij_thilde();
228 int nb_elem_tot=zone_ef.zone().nb_elem_tot();
229 int nb_som_elem=zone_ef.zone().nb_som_elem();
230 const IntTab& elems=zone_ef.zone().les_elems() ;
231
232 for (int elem=0; elem<nb_elem_tot; elem++)
233 if (elem_contribue(elem)&&(!marqueur_paroi(elem)))
234 {
235 double pond=1./volumes(elem);
236 assert(N == dimension || N == 1);
237
238 for (int i1=0; i1<nb_som_elem; i1++)
239 {
240 int glob=elems(elem,i1);
241 int transpose = (marqueur_neuman[glob] == 1 || N == 1) ? 0 : transpose_;
242 for (int i2=0; i2<nb_som_elem; i2++)
243 {
244 int glob2=elems(elem,i2);
245 for (int n = 0; n < N; n++)
246 for (int d = 0; d < dimension; d++)
247 resu(glob, n) -= bij(elem, i1, d) * (bij_thilde(elem, i2, d) * tab_inconnue(glob2, n) + transpose * bij_thilde(elem, i2, n) * tab_inconnue(glob2, d)) * (diffu[elem]+diffu_turb[elem]) * pond;
248 }
249 }
250 }
251// Debog::verifier(" Op_Dift_EF_Q1::ajouter, resu 0 ",resu);
252// Journal()<<max(resu)<<" "<<min(resu)<<finl;
253
254
255// on ajoute la contribution des bords
256 ajouter_bords(tab_inconnue,resu);
257 return resu;

427void Op_Dift_EF_Q1::ajouter_bords(const DoubleTab& tab_inconnue,DoubleTab& resu, int contrib_interne ) const
428{
429// a mettre dans calculer_flux_bord....
430
431 const Zone_Cl_EF& zone_Cl_EF = la_zcl_EF.valeur();
432 const Zone_EF& zone_EF = la_zone_EF.valeur();
433 flux_bords_=0.;
434// const DoubleTab& tab_inconnue=equation().inconnue().valeurs();
435// on parcourt toutes les faces de bord et on calcule lambda*gradT
436 const Zone_EF& zone_ef=ref_cast(Zone_EF,equation().zone_dis().valeur());
437 const IntTab& face_voisins=zone_ef.face_voisins();
438 const DoubleTab& bij=zone_ef.Bij();
439 int nb_som_elem=zone_ef.zone().nb_som_elem();
440 const IntTab& elems=zone_ef.zone().les_elems() ;
441 const DoubleTab& face_normales=zone_ef.face_normales();
442 const DoubleVect& volumes_thilde= zone_ef.volumes_thilde();
443 const DoubleVect& volumes= zone_ef.volumes();
444
445 const IntTab& face_sommets=zone_ef.face_sommets();
446 int nb_som_face=zone_ef.nb_som_face();
447 int nb_faces_elem = zone_ef.zone().nb_faces_elem();
448 const IntTab& elem_faces = zone_ef.elem_faces();
449 int nb_som_free = nb_som_elem-nb_som_face;
450
451 const int N = resu.line_size();
452
453 DoubleVect diffu_turb(diffusivite_turbulente()->valeurs());
454 DoubleTab diffu(nu_);
455
456 if (N > 1)
457 {
458 for (int n_bord=0; n_bord<zone_ef.nb_front_Cl(); n_bord++)
459 {
460
461 const Cond_lim& la_cl = zone_Cl_EF.les_conditions_limites(n_bord);
462 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
463 int nb_faces_bord = le_bord.nb_faces();
464 int num1=0;
465 int num2=nb_faces_bord;
466
467 if (sub_type(Dirichlet_paroi_fixe,la_cl.valeur()) ||
468 sub_type(Dirichlet_paroi_defilante,la_cl.valeur()) )
469 {
470 if (le_modele_turbulence.valeur().utiliser_loi_paroi())
471 {
472 DoubleVect n(dimension);
473 DoubleTrav Tgrad(dimension,dimension);
474 ArrOfDouble effort_face(dimension);
475 ArrOfDouble effort_elem(dimension);
476 ArrOfInt som_CL(nb_som_face);
477
478// Boucle sur les faces de la CL
479 for (int ind_face=num1; ind_face<num2; ind_face++)
480 {
481 int num_face = le_bord.num_face(ind_face);
482 for (int isom=0; isom<nb_som_face; isom++)
483 {
484 int glob2=face_sommets(num_face,isom);
485 som_CL(isom) = glob2;
486 }
487 int elem=face_voisins(num_face,0);
488 double pond=volumes_thilde(elem)/volumes(elem);
489 for (int a=0; a<dimension; a++) n[a]=face_normales(num_face,a);
490 n/=norme_array(n);
491
492// Calcul du gradient gradU via tau
493// Au bord, n toujours oriente vers l'exterieur
494 for (int nc=0; nc<dimension; nc++)
495 for (int nc2=0; nc2<dimension; nc2++)
496 Tgrad(nc,nc2)=tau_tan_(num_face,nc)*n[nc2];
497
498// Boucle sur les faces de l'element
499// (Sigma_e somme_face n = 0 => -Sigma_e somme_face_cl n = + Sigma_e somme_face_non_cl n)
500 effort_elem = 0.;
501 for (int i=0; i<nb_faces_elem; i++)
502 {
503 int face_i=elem_faces(elem,i);
504// On calcule la contribution de la face :
505 effort_face = 0.;
506 double ori=1.;
