70Entree& Terme_Source_Canal_perio::lire_donnees(Entree& is)
71{
72 Motcle accolade_ouverte("{");
73 Motcle accolade_fermee("}");
74 Motcle motlu;
75// Valeurs par defaut
76 u_etoile = 0.;
77 h = 1.;
78 coeff = 10.;
79 velocity_weighting_ = 0;
80// FIN valeurs par defaut
81
82 int verif = 0;
83 Motcles les_mots(7);
84
85 Motcle mot_lu;
86 {
87 les_mots[0]="direction_ecoulement"; //!< Obsolete
88 les_mots[1]="u_etoile";
89 les_mots[2]="coeff";
90 les_mots[3]="h";
91 les_mots[4]="bord";
92 les_mots[5]="velocity_weighting"; //!< Energy source term algorithm 0 | 1
93 les_mots[6]="debit_impose";
94 }
95 Motcle acc_ouverte("{");
96 Motcle acc_fermee("}");
97 is >> mot_lu;
98 if(mot_lu != acc_ouverte)
99 {
100 Cerr << "We were expecting a { instead of " << mot_lu << finl;
101 }
102 is >> mot_lu;
103 dir_source_.resize(dimension);
104 dir_source_=0;
105 while(mot_lu != acc_fermee)
106 {
107 int rang=les_mots.search(mot_lu);
108 switch(rang)
109 {
110 case 0 :
111 Cerr <<"The direction_ecoulement option is obsolete, you must now use the bord option to specify the boundary where periodicity is applied."<<finl;
112 exit();
113 break;
114 case 1 :
115 is >> u_etoile;
116 Cerr << "Friction Velocity : " << u_etoile << finl;
117 Cerr<<"L'option u_etoile est obsolete, supprimer la du jeu de donnees"<<finl;
118 exit();
119 break;
120 case 2 :
121 is >> coeff;
122 Cerr << "Damping Coefficient is now set : " << coeff << finl;
123 break;
124 case 3 :
125 is >> h;
126 Cerr << "Channel Half-height : " << h << finl;
127 break;
128 case 4 :
129 is >> bord_periodique_;
130 Cerr << "Periodic flow according the " << bord_periodique_ << " boundary." << finl;
131 verif = 1;
132 break;
133 case 6 :
134 is >> debit_impose_;
135 is_debit_impose_=1;
136 break;
137 case 5 :
138 is >> velocity_weighting_;
139 if (velocity_weighting_!=0 && velocity_weighting_!=1)
140 {
141 Cerr << "velocity_weighting value should be 0 or 1." << finl;
142 exit();
143 }
144 if (!sub_type(Convection_Diffusion_std,equation()))
145 {
146 Cerr << "velocity_weighting option is available only for a Canal_perio source term in the energy equation." << finl;
147 exit();
148 }
149 break;
150 default :
151 {
152 Cerr << mot_lu << " is not expected by Terme_Source_Canal_perio" << finl;
153 Cerr << "The possible keywords are : " << les_mots << finl;
154 exit();
155 }
156 }
157 is >> mot_lu;
158 }
159
160 if (verif != 1)
161 {
162 Cerr << "User did not specify flow direction." << finl;
163 Cerr << "trust will now stop !" << finl;
164 exit();
165 }
166 return is;

169void Terme_Source_Canal_perio::completer()
170{
171 Source_base::completer();
172 if (sub_type(Convection_Diffusion_std,equation()))
173 {
174 set_fichier("Canal_perio");
175 set_description("Energy source term = Integral(P*dv) [W]");
176 }
177 int nb_bords = equation().zone_dis().valeur().nb_front_Cl();
178 for (int n_bord=0; n_bord<nb_bords; n_bord++)
179 {
180 const Cond_lim& la_cl = equation().zone_Cl_dis().valeur().les_conditions_limites(n_bord);
181 if (sub_type(Periodique,la_cl.valeur()))
182 {
183 const Periodique& perio = ref_cast(Periodique,la_cl.valeur());
184 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
185 if ( bord_periodique_ == le_bord.le_nom() ) //!< Le bord periodique est specifie
186 {
187 if (perio.est_periodique_selon_un_axe())
188 {
189// Cas ou le bord est periodique selon un axe, alors on fixe la direction de l'ecoulement (VDF)
190 direction_ecoulement_ = perio.direction_periodicite();
191 }
192 else
193 {
194// Cas general ou le bord periodique n'est pas oriente selon un axe:
195 dir_source_ = perio.direction_perio();
196 dir_source_ *= -1;
197// On adimensionnalise:
198 double norme = norme_array(dir_source_);
199 dir_source_ /= norme;
200 }
201// On recupere la surface
202 surface_bord_ = 0.5 * le_bord.frontiere().get_aire();
203 return;
204 }
205 }
206 }
207 Cerr << "********************************************************" << finl;
208 Cerr << "Error for the definition of the Canal_perio source term." << finl;
209 if (bord_periodique_=="")
210 {
211 Cerr << "It seems the channel is not parallel to an axis. Direction_ecoulement keyword is useless in this case." << finl;
212 Cerr << "So try to use the following syntax to specify the name of the periodic boundary:" << finl;
213 Cerr << "Canal_perio { bord name }" << finl;
214 }
215 else
216 {
217 Cerr << bord_periodique_ << " is not a boundary with a periodic boundary condition." << finl;
218 }
219 exit();

223{

228

285// Reopen:
286 fichier_reprise.ouvrir(filename);

288// Read up to the time:
289 int time_found = 0;
290 while (!fichier_reprise.eof() && !time_found)
291 {
292 double temps_ecrit;
293 fichier_reprise >> temps_ecrit;
294 fichier_reprise >> debnm1_;
295 fichier_reprise >> debit_ref_;

297// std::getline(fichier_reprise.get_ifstream(), line);

315double Terme_Source_Canal_perio::compute_heat_flux() const
316{
317// On recupere flux_bords operateur de diffusion
318 const Operateur_base& op_base = equation().operateur(0).l_op_base(); //!< Diffusion operator
319 assert(sub_type(Operateur_Diff_base,op_base)); //!< Check
320 const DoubleTab& flux_bords = op_base.flux_bords();
321// If flux_bords is not build (diffusion_implicit algorithm for example, we calculate it):
322 if (flux_bords.size()==0)
323 {
324 DoubleTab dummy(equation().inconnue().valeurs());
325 equation().operateur(0).ajouter(equation().inconnue(), dummy);
326 }
327// Loop on boundaries to evaluate total heat flux:
328 double heat_flux=0;
329 int nb_bords = equation().zone_dis().valeur().nb_front_Cl();
330 for (int n_bord=0; n_bord<nb_bords; n_bord++)
331 {
332 const Cond_lim& la_cl = equation().zone_Cl_dis().valeur().les_conditions_limites(n_bord);
333 if (sub_type(Neumann_paroi,la_cl.valeur()))
334 {
335// Loop on boundary faces with imposed flux condition (Neumann)
336 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
337 int ndeb = frontiere_dis.num_premiere_face();
338 int nfin = ndeb + frontiere_dis.nb_faces();
339 for (int num_face=ndeb; num_face<nfin; num_face++)
340 heat_flux += flux_bords(num_face,0);
341 }
342 }
343 heat_flux=mp_sum(heat_flux);
344 return heat_flux;
345}

347ArrOfDouble Terme_Source_Canal_perio::source_convection_diffusion(double debit_e) const
348{
349// Compute heat_flux:
350 double heat_flux = compute_heat_flux();
351
352 const Zone_VF& zone_vf = ref_cast(Zone_VF,equation().zone_dis().valeur());
353 const double volume = zone_vf.zone().volume_total();
354 int size = zone_vf.nb_faces();
355 ArrOfDouble s(size);
356 if (velocity_weighting_) //!< It seems this algorithm do not imply dT/dt -> 0
357 {
358// Compute source term for Energy
359// Expression from TU Delft Master Thesis
360// Source = -u*Sum(imposed_heat_flux)/(Volume*Ubulk)
361// Ubulk=FlowRate/Area(PeriodicBoundary)
362// Loop on the faces
363 const DoubleTab& vitesse = ref_cast(Convection_Diffusion_std,equation()).vitesse_transportante().valeurs();
364 for (int num_face=0; num_face<size; num_face++)
365 {
366 double velocity = 0;
367 if (direction_ecoulement_>=0) //!< Ecoulement selon un axe
368 velocity = vitesse(num_face,direction_ecoulement_);
369 else //!< Cas general
370 for (int i=0; i<dimension; i++)
371 velocity += vitesse(num_face,i) * dir_source_[i];
372 s[num_face]=-velocity*heat_flux/(volume*debit_e/surface_bord_);
373 }
374 }
375 else
376 {
377// Compute source term with
378// Source = -Sum(imposed_heat_flux)/Volume
379// Loop on the faces
380 for (int num_face=0; num_face<size; num_face++)
381 s[num_face]=-heat_flux/volume;
382 }
383 return s;
384}

389ArrOfDouble Terme_Source_Canal_perio::source() const
390{
391 double tps = equation().schema_temps().temps_courant();
392 double tps_init = equation().schema_temps().temps_init();
393 double dt = equation().schema_temps().pas_de_temps();
394 int premiere_ecriture = (!equation().probleme().reprise_effectuee() && deb_==0 ? 1 : 0);
395 Nom ext_nom_source_ = equation().probleme().le_nom();
396 ext_nom_source_ += "_" ;
397 ext_nom_source_ += bord_periodique_ ;
398
399// Pourquoi ?
400 if (est_different(dernier_temps_calc_,tps))
401 {
402// Compte flow rate:
403 double debit_e= 0.;
404 calculer_debit(debit_e);
405
406 if (je_suis_maitre())
407 {
408 if (sub_type(Convection_Diffusion_std,equation()))
409 {
410// Write nothing if equation is Energy
411 }
412 else
413 write_flow_rate(ext_nom_source_, debit_e);
414 }
415 if (sub_type(Convection_Diffusion_std,equation()))
416 {
417 if (equation().probleme().is_dilatable())
418 {
419 Cerr << "Source term not validated yet for Quasi Compressible formulation." << finl;
420 Cerr << "Contact TRUST support." << finl;
421 exit();
422 }
423 if(equation().schema_temps().pas_de_temps_locaux().size()>0)
424 {
425 Cerr << "Source term not validated yet for steady option for the Convection_Diffusion_std equation." << finl;
426 Cerr << "Contact TRUST support." << finl;
427 exit();
428 }
429 return source_convection_diffusion(debit_e);
430 }
431 else if (sub_type(Navier_Stokes_std,equation()))
432 {
433// Compute source term for Navier Stokes
434
435// Master process sends values to all processes:
436 envoyer_broadcast(debit_ref_, 0);
437 envoyer_broadcast(debnm1_, 0);
438 envoyer_broadcast(source_, 0);
439
440// On conserve le test (avec dt_min) car appel a t=0 (dt=0) au cours de preparer calculer pour loi de paroi TBLE
441 double dt_min = equation().schema_temps().pas_temps_min();
442 const DoubleVect& dt_locaux = equation().schema_temps().pas_de_temps_locaux();
443 double si = 0;
444 if (sup_ou_egal(dt,dt_min))
445 {
446// si = [ 2*(Q(0)-Q(t(n))) - (Q(0)-Q(t(n-1))) ] / [ coeff * dt * aire(bord) ]
447 if( dt_locaux.size()>0)
448 {
449 Cerr << "Periodic BC plus source term not validated yet for steady option." << finl;
450 Cerr << "Contact TRUST support." << finl;
451 exit();
452
453// du au fait que ce calcul implique une division par dt, dans le cas ou le dt est variable par face,
454// calculer_debit(debit_e) renvoye [ 2*(Q(0)-Q(t(n))) - (Q(0)-Q(t(n-1))) ]/dt_locaux
455 si = debit_e/(coeff*surface_bord_);
456 }
457 else
458 {
459 si = (2.*(debit_ref_-debit_e)-(debit_ref_-debnm1_))/(coeff*dt*surface_bord_);
460 debnm1_ = debit_e;
461 }
462 }
463
464 source_ += si;
465 dernier_temps_calc_ = tps;
466
467 if (je_suis_maitre())
468 {
469// Write the pressure gradient file if equation is Navier Stokes
470 Nom filename(nom_du_cas());
471 filename+="_Pressure_Gradient_";
472 filename+= ext_nom_source_ ;
473 if (!pressure_gradient_file_.is_open())
474 {
475 pressure_gradient_file_.ouvrir(filename, (premiere_ecriture?ios::out:ios::app));
476 pressure_gradient_file_.setf(ios::scientific);
477 }
478 if ((tps <= (tps_init+dt)) && premiere_ecriture)
479 {
480 if (equation().probleme().is_dilatable()==1)
481 pressure_gradient_file_ << "# Time t gradP(t) gradP(t)-gradP(t-dt) in [kg.s-2.m-2] if SI units used" << finl;
482 else
483 pressure_gradient_file_ << "# Time t gradP(t) gradP(t)-gradP(t-dt) in [m.s-2] if SI units used" << finl;
484 }
485 pressure_gradient_file_ << tps+dt << " " << source_ << " " << si << finl;
486
487// Write the restart file:
488 filename=nom_du_cas();
489 filename+="_Channel_Flow_Rate_repr_";
490 filename+=ext_nom_source_;
491 if (!restart_file_.is_open())
492 {
493 restart_file_.ouvrir(filename, (premiere_ecriture?ios::out:ios::app));
494 restart_file_.setf(ios::scientific);
495 }
496 if ((tps <= (tps_init+dt)) && premiere_ecriture)
497 {
498 if (equation().probleme().is_dilatable()==1)
499 restart_file_ << "# Time t Flow rate Q(t) Flow rate Q(0) in [kg.s-1] gradP(t) in [kg.s-2.m-2] if SI units used" << finl;
500 else
501 restart_file_ << "# Time t Flow rate Q(t) Flow rate Q(0) in [m3.s-1] gradP(t) in [m.s-2] if SI units used" << finl;
502 }
503 restart_file_ << tps+dt << " " << debit_e << " " << debit_ref_ << " " << source_ << finl;
504 }
505 }
506 else
507 {
508 Cerr << "You can't use canal_perio on the equation " << equation().que_suis_je() << finl;
509 exit();
510 }
511 }
512 if (sub_type(Convection_Diffusion_std,equation()))
513 {
514 Cerr << "Error! Contact TRUST support." << finl;
515 exit();
516 }
517
518// Essayer de virer direction_ecoulement_ (attention assert en VDF et ecarts possibles)
519 ArrOfDouble s;
520 s.resize(dimension);
521 s = 0.;
522 if (direction_ecoulement_>=0) //!< Ecoulement selon un axe
523 s[direction_ecoulement_] = source_;
524 else //!< Cas general
525 for (int i=0; i<dimension; i++)
526 s[i] = source_ * dir_source_[i];
527 return s;
528}
