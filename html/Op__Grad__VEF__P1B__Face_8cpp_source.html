55static int chercher_arete(int elem, int somi, int somj,
56 const IntTab& elem_aretes,
57 const IntTab& aretes_som)
58{
59 if(somi>somj)
60 {
61 int k=somi;
62 somi=somj;
63 somj=k;
64 }
65 for(int i_arete=0; i_arete<6; i_arete++)
66 {
67 int arete=elem_aretes(elem, i_arete);
68 int som1=aretes_som(arete,0);
69 if(somi==som1)
70 {
71 int som2=aretes_som(arete,1);
72 if(somj==som2)
73 return arete;
74 }
75 }
76 return -1;
77}

211DoubleTab& Op_Grad_VEF_P1B_Face::
212modifier_grad_pour_Cl(DoubleTab& grad ) const
213{
214 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
215 la_zone_vef.valeur());
216 const DoubleTab& face_normales = zone_VEF.face_normales();
217 const Zone_Cl_VEF& zone_Cl_VEF=la_zcl_vef.valeur();
218 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
219 int nb_bords =les_cl.size();
220 for (int n_bord=0; n_bord<nb_bords; n_bord++)
221 {
222 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
223 {
224 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
225 int num1 = le_bord.num_premiere_face();
226 int num2 = num1 + le_bord.nb_faces();
227 if (sub_type(Periodique,la_cl.valeur()))
228 {
229// periodicite
230 const Periodique& la_cl_perio = ref_cast(Periodique,
231 la_cl.valeur());
232 ArrOfInt fait(le_bord.nb_faces());
233 for (int face=num1; face<num2; face++)
234 {
235 if(!fait[face-num1])
236 {
237 int face_associee=num1+la_cl_perio.face_associee(face-num1);
238 fait[face-num1]=(fait[face_associee-num1]=1);
239 for (int comp=0; comp<dimension; comp++)
240 grad(face_associee,comp) =
241 (grad(face,comp)+= grad(face_associee,comp));
242 }
243 }
244 }
245 else
246 {
247 if(sub_type(Symetrie,la_cl.valeur()))
248 {
249 for (int face=num1; face<num2; face++)
250 {
251 double psc=0;
252 double norm=0;
253 int comp;
254 for (comp=0; comp<dimension; comp++)
255 {
256 psc += grad(face,comp)*face_normales(face,comp);
257 norm += face_normales(face,comp)*face_normales(face,comp);
258 }
259// psc/=norm; // Fixed bug: Arithmetic exception
260 if (std::fabs(norm)>=DMINFLOAT) psc/=norm;
261 for (comp=0; comp<dimension; comp++)
262 grad(face,comp)-=psc*face_normales(face,comp);
263 }
264 }
265 else if(sub_type(Dirichlet,la_cl.valeur())||sub_type(Dirichlet_homogene,la_cl.valeur()))
266 {
267 for (int face=num1; face<num2; face++)
268 {
269 for (int comp=0; comp<dimension; comp++)
270 grad(face,comp)=0;
271 }
272 }
273 }
274 }
275 }
276 return grad;

278DoubleTab& Op_Grad_VEF_P1B_Face::
279ajouter_elem(const DoubleTab& pre,
280 DoubleTab& grad) const
281{
282 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
283 la_zone_vef.valeur());
284 assert(zone_VEF.get_alphaE());
285 const Zone& zone = zone_VEF.zone();
286 const DoubleTab& face_normales = zone_VEF.face_normales();
287 const DoubleVect& porosite_face=zone_VEF.porosite_face();
288 const IntTab& elem_faces=zone_VEF.elem_faces();
289 const IntTab& face_voisins=zone_VEF.face_voisins();
290 int nfe=zone.nb_faces_elem();
291 int nb_elem_tot=zone.nb_elem_tot();
292// Si pas de support P1, on impose Neumann sur P0
293 if (zone_VEF.get_alphaS()==0)
294 {
295 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
296 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
297 int nb_bords =les_cl.size();
298 for (int n_bord=0; n_bord<nb_bords; n_bord++)
299 {
300 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
301 if (sub_type(Neumann_sortie_libre,la_cl.valeur()))
302 {
303 const Neumann_sortie_libre& la_sortie_libre = ref_cast(Neumann_sortie_libre,la_cl.valeur());
304 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
305 int num1 = le_bord.num_premiere_face();
306 int num2 = num1 + le_bord.nb_faces();
307 for (int face=num1; face<num2; face++)
308 {
309 double P_imp = la_sortie_libre.flux_impose(face-num1);
310 double diff = P_imp;
311 for (int comp=0; comp<dimension; comp++)
312 grad(face,comp) += diff*face_normales(face,comp)*porosite_face(face);
313 }
314 }
315 }
316 }
317 int elem,indice,face,comp;
318 ArrOfDouble sigma(dimension);
319 for(elem=0; elem<nb_elem_tot; elem++)
320 {
321 for(indice=0; indice<nfe; indice++)
322 {
323 double pe=pre(elem);
324 face = elem_faces(elem,indice);
325 double signe=1;
326 if(elem!=face_voisins(face,0)) signe=-1;
327 for(comp=0; comp<dimension; comp++)
328 grad(face, comp) -= pe*signe*face_normales(face,comp)*porosite_face(face);
329 }
330 }
331 return grad;

333DoubleTab& Op_Grad_VEF_P1B_Face::
334ajouter_som(const DoubleTab& pre,
335 DoubleTab& grad) const
336{
337 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
338 la_zone_vef.valeur());
339 assert(zone_VEF.get_alphaS());
340 const Zone& zone = zone_VEF.zone();
341 const Domaine& dom=zone.domaine();
342 const DoubleTab& face_normales = zone_VEF.face_normales();
343 const DoubleVect& porosite_face=zone_VEF.porosite_face();
344 const IntTab& som_elem=zone.les_elems();
345 const IntTab& elem_faces=zone_VEF.elem_faces();
346 const IntTab& face_voisins=zone_VEF.face_voisins();
347 int nfe=zone.nb_faces_elem();
348 int nps=zone_VEF.numero_premier_sommet();
349 int nb_elem_tot=zone.nb_elem_tot();
350 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
351
352 ArrOfDouble sigma(dimension);
353 for(int elem=0; elem<nb_elem_tot; elem++)
354 {
355 double coeff_som=calculer_coef_som(elem, zone_Cl_VEF, zone_VEF);
356 for(int indice=0; indice<nfe; indice++)
357 {
358 int som = nps+dom.get_renum_som_perio(som_elem(elem,indice));
359 double ps=pre(som);
360 int face = elem_faces(elem,indice);
361 double signe=1;
362 if(elem!=face_voisins(face,0))
363 signe=-1;
364 for(int comp=0; comp<dimension; comp++)
365 sigma[comp]=signe*face_normales(face,comp);
366
367 for(int indice2=0; indice2<nfe; indice2++)
368 {
369 int face2 = elem_faces(elem,indice2);
370 for(int comp=0; comp<dimension; comp++)
371 grad(face2,comp)-=coeff_som*ps*sigma[comp]*porosite_face(face2);
372 }
373 }
374 }
375
376 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
377 const IntTab& face_sommets = zone_VEF.face_sommets();
378 int nb_bords = les_cl.size();
379 for (int n_bord=0; n_bord<nb_bords; n_bord++)
380 {
381 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
382 if(sub_type(Neumann_sortie_libre,la_cl.valeur()))
383 {
384 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
385 const Neumann_sortie_libre& sortie_libre = ref_cast(Neumann_sortie_libre,la_cl.valeur());
386 int num1 = le_bord.num_premiere_face();
387 int num2 = num1 + le_bord.nb_faces();
388 for (int face=num1; face<num2; face++)
389 {
390 const double P_imp = sortie_libre.flux_impose(face-num1);
391 for(int indice=0; indice<(nfe-1); indice++)
392 {
393 int som=nps+dom.get_renum_som_perio(face_sommets(face,indice));
394 for (int comp=0; comp<dimension; comp++)
395 grad(face,comp)-=
396 1./dimension*face_normales(face,comp)*(pre(som)-P_imp)
397 *porosite_face(face);
398 }
399 } //fin du if sur "Neumann_sortie_libre"
400 }
401 }
402 return grad;

404DoubleTab& Op_Grad_VEF_P1B_Face::
405ajouter_aretes(const DoubleTab& pre,
406 DoubleTab& grad) const
407{
408 const Zone_VEF_PreP1b& zone_VEF =
409 ref_cast(Zone_VEF_PreP1b, la_zone_vef.valeur());
410 assert(zone_VEF.get_alphaA());
411 const Zone& zone = zone_VEF.zone();
412// const Domaine& dom=zone.domaine();
413 const DoubleTab& face_normales = zone_VEF.face_normales();
414 const DoubleVect& porosite_face=zone_VEF.porosite_face();
415 const IntTab& som_elem=zone.les_elems();
416 const IntTab& elem_faces=zone_VEF.elem_faces();
417 const IntTab& face_voisins=zone_VEF.face_voisins();
418 const IntTab& aretes_som=zone_VEF.zone().aretes_som();
419 const IntTab& elem_aretes=zone_VEF.zone().elem_aretes();
420 const ArrOfInt& renum_arete_perio=zone_VEF.get_renum_arete_perio();
421 const ArrOfInt& ok_arete=zone_VEF.get_ok_arete();
422 int nfe=zone.nb_faces_elem();
423 int nb_elem_tot=zone.nb_elem_tot();
424// int nps=zone_VEF.numero_premier_sommet();
425 int npa=zone_VEF.numero_premiere_arete();
426 int indice;
427 ArrOfDouble sigma(dimension);
428 for(int elem=0; elem<nb_elem_tot; elem++)
429 {
430 for(indice=0; indice<nfe; indice++)
431 {
432// som = nps+dom.get_renum_som_perio(som_elem(elem,indice));
433 int face = elem_faces(elem,indice);
434 double signe=1;
435 if(elem!=face_voisins(face,0))
436 signe=-1;
437 for(int comp=0; comp<dimension; comp++)
438 sigma[comp]=signe*face_normales(face,comp);
439 }
440 for(int isom=0; isom<3; isom++)
441 {
442 int somi = som_elem(elem,isom);
443 int facei = elem_faces(elem,isom);
444 double signei=1.;
445 if (face_voisins(facei,0) != elem)
446 signei=-1.;
447 for(int jsom=isom+1; jsom<4; jsom++)
448 {
449 int somj = som_elem(elem,jsom);
450 int facej = elem_faces(elem,jsom);
451 double signej=1.;
452 if (face_voisins(facej,0) != elem)
453 signej=-1.;
454 int arete = chercher_arete(elem, somi, somj,
455 elem_aretes, aretes_som);
456 arete=renum_arete_perio[arete];
457 if(ok_arete[arete])
458 {
459 double pa=pre(npa+arete);
460 int niinij=0;
461 for(int ksom=0; ksom<4; ksom++)
462 {
463 int face2=elem_faces(elem,ksom);
464 if((ksom==isom)||(ksom==jsom))
465 {
466 if(niinij==ksom)
467 niinij++;
468 for(int comp=0; comp<3; comp++)
469 grad(face2,comp)+=porosite_face(face2)*1./15.*
470 (signei*face_normales(facei,comp)+
471 signej*face_normales(facej,comp))*pa;
472 }
473 else
474 {
475 if(niinij==ksom)
476 niinij++;
477 while((niinij==jsom)||(niinij==isom))
478 niinij++;
479 assert(niinij<4);
480 int facek = elem_faces(elem,niinij);
481 double signek=1.;
482 if (face_voisins(facek,0) != elem)
483 signek=-1.;
484 for(int comp=0; comp<3; comp++)
485 grad(face2,comp)+=porosite_face(face2)*2./15.*
486 signek*face_normales(facek,comp)*pa;
487 niinij=ksom;
488 }
489 }
490 }
491 }
492 }
493 }
494 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
495 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
496 const IntTab& face_sommets = zone_VEF.face_sommets();
497 int nb_bords =les_cl.size();
498 for (int n_bord=0; n_bord<nb_bords; n_bord++)
499 {
500 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
501 {
502 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
503 int num1 = le_bord.num_premiere_face();
504 int num2 = num1 + le_bord.nb_faces();
505 if(sub_type(Neumann_sortie_libre,la_cl.valeur()))
506 {
507 for (int face=num1; face<num2; face++)
508 {
509 int elem=face_voisins(face,0);
510 for(int i=0; i<2; i++)
511 {
512 int somi=face_sommets(face,i);
513 for(int j=i+1; j<3; j++)
514 {
515 int somj=face_sommets(face,j);
516 int arete=chercher_arete(elem, somi, somj,
517 elem_aretes, aretes_som);
518 arete=renum_arete_perio[arete];
519 for (int comp=0; comp<dimension; comp++)
520 grad(face,comp)-= porosite_face(face)*0*
521 face_normales(face,comp)*pre(npa+arete);
522 }
523 }
524 }
525 }
526 }
527 }
528 return grad;

531double Op_Grad_VEF_P1B_Face::calculer_coef_som(int elem, const Zone_Cl_VEF& zcl,
532 const Zone_VEF_PreP1b& zone_VEF)
533{
534 double coef_std=1./(Objet_U::dimension*(Objet_U::dimension+1));
535 if (zone_VEF.get_modif_div_face_dirichlet()==0)
536 return coef_std;
537 int type_elem=0;
538 int rang_elem=zone_VEF.rang_elem_non_std()(elem);
539 if (rang_elem!=-1)
540 type_elem=zcl.type_elem_Cl(rang_elem);
541 double coef;
542 if (type_elem==0)
543 return coef_std;
544 if (Objet_U::dimension==2)
545 {
546 switch(type_elem)
547 {
548 case 0:
549 coef=3;
550 break;
551 case 1:
552 case 2:
553 case 4:
554 coef=2;
555 break;
556 case 3:
557 case 5:
558 case 6:
559 coef=1;
560 break;
561 default:
562 coef=-1;
563 Process::exit();
564 break;
565 }
566 }
567 else if (Objet_U::dimension==3)
568 {
569 switch(type_elem)
570 {
571 case 0:
572 coef=4;
573 break;
574 case 1:
575 case 2:
576 case 4:
577 case 8:
578 coef=3;
579 break;
580 case 3:
581 case 5:
582 case 6:
583 case 9:
584 case 10:
585 case 12:
586 coef=2;
587 break;
588 case 7:
589 case 11:
590 case 13:
591 case 14:
592 coef=1;
593 break;
594 default:
595 coef=-1;
596 Process::exit();
597 break;
598 }
599 }
600 else
601 {
602 coef=-1;
603 Process::exit();
604 }
605 return 1./(Objet_U::dimension*coef);

632int Op_Grad_VEF_P1B_Face::impr(Sortie& os) const
633{
634 const int impr_mom=la_zone_vef->zone().Moments_a_imprimer();
635 const int impr_sum=(la_zone_vef->zone().Bords_a_imprimer_sum().est_vide() ? 0:1);
636 const int impr_bord=(la_zone_vef->zone().Bords_a_imprimer().est_vide() ? 0:1);
637 const Schema_Temps_base& sch = equation().probleme().schema_temps();
638 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
639 const Zone_VEF_PreP1b& zone_VEF = zone_Vef();
640 const IntTab& face_voisins = zone_VEF.face_voisins();
641 const DoubleTab& face_normales = zone_VEF.face_normales();
642 const IntTab& som_elem=la_zone_vef->zone().les_elems();
643 const Equation_base& eqn = equation();
644 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,eqn);
645 const Champ_P1_isoP1Bulle& la_pression_P1B = ref_cast(Champ_P1_isoP1Bulle,eqn_hydr.pression_pa().valeur());
646// Si on filtre:
647 la_pression_P1B.filtrage(zone_VEF,la_pression_P1B);
648 const DoubleVect& pression_P1B = la_pression_P1B.champ_filtre();
649 Debog::verifier("la pression filtree dans Op_Grad_VEF_P1B_Face::impr=",pression_P1B);
650 double pres_som, pres_arete, pres_tot ;
651 double n0, n1, n2 ;
652 const int nb_faces = zone_VEF.nb_faces_tot();
653 int nse=zone_VEF.zone().nb_som_elem();
654 int nps=zone_VEF.numero_premier_sommet();
655 int npa=zone_VEF.numero_premiere_arete();
656 int elem ;
657 int face;
658 int som_opp, som, som_glob;
659 DoubleTab xgr(nb_faces,dimension);
660 xgr=0.;
661// DoubleVect moment(dimension);
662 if (impr_mom)
663 {
664 const DoubleTab& xgrav = zone_VEF.xv();
665 const ArrOfDouble& c_grav=la_zone_vef->zone().cg_moments();
666 for (int num_face=0; num_face <nb_faces; num_face++)
667 for (int i=0; i<dimension; i++)
668 xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
669 }
670 flux_bords_.resize(zone_VEF.nb_faces_bord(),dimension);
671 flux_bords_ = 0.;
672
673 DoubleTrav tab_flux_bords(3,zone_VEF.nb_front_Cl(),dimension);
674 tab_flux_bords=0.;
675 /*
676 fluxx_s -> flux_bords2(0,num_cl,0)
677 fluxy_s -> flux_bords2(0,num_cl,1)
678 fluxz_s -> flux_bords2(0,num_cl,2)
679 fluxx_sum_s -> flux_bords2(1,num_cl,0)
680 fluxy_sum_s -> flux_bords2(1,num_cl,1)
681 fluxz_sum_s -> flux_bords2(1,num_cl,2)
682 moment(0) -> flux_bords2(2,num_cl,0)
683 moment(1) -> flux_bords2(2,num_cl,1)
684 moment(2) -> flux_bords2(2,num_cl,2)
685 */
686 double coeff_P1 = 1./dimension;
687 double coeff_Pa = 1./dimension;
688 int nb_bord = zone_VEF.nb_front_Cl();
689 for (int n_bord=0; n_bord<nb_bord; n_bord++)
690 {
691 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
692 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
693 int impr_boundary = (zone_VEF.zone().Bords_a_imprimer_sum().contient(le_bord.le_nom()) ? 1 : 0);
694 int ndeb = le_bord.num_premiere_face();
695 int nfin = ndeb + le_bord.nb_faces();
696 for (face=ndeb; face<nfin; face++)
697 {
698 int voisin = 0;
699 elem = face_voisins(face,voisin);
700 if (zone_VEF.get_num_fac_loc(face, voisin)==-1)
701 {
702 voisin = 1;
703 elem = face_voisins(face,voisin);
704 }
705 int signe=zone_VEF.oriente_normale(face,elem);
706 som_opp = la_zone_vef->get_num_fac_loc(face, voisin);
707 som_opp = som_elem(elem, som_opp);
708 pres_tot = 0. ;
709
710// Contribution de la pression P0
711 if (zone_VEF.get_alphaE())
712 pres_tot += pression_P1B(elem);
713
714// Contribution de la pression P1
715 if (zone_VEF.get_alphaS())
716 {
717 pres_som = 0.;
718 for(som=0; som<nse; som++)
719 {
720 som_glob = som_elem(elem, som);
721 if (som_glob != som_opp)
722 pres_som += pression_P1B(nps+som_glob);
723 }
724 pres_tot += coeff_P1 * pres_som;
725 }
726
727// Contribution de la pression Pa
728// GF descativation car bilan incorret
729 if (zone_VEF.get_alphaA()*0!=0)
730 {
731 pres_arete = 0.;
732 const IntTab& elem_aretes=la_zone_vef->zone().elem_aretes();
733 const IntTab& aretes_som=la_zone_vef->zone().aretes_som();
734 for(int arete=0; arete<6; arete++)
735 {
736 int arete_glob = elem_aretes(elem, arete);
737 if (aretes_som(arete_glob,0)!=som_opp && aretes_som(arete_glob,1)!=som_opp)
738 pres_arete += pression_P1B(npa+arete_glob);
739 }
740 pres_tot += coeff_Pa * pres_arete;
741 }
742
743// Calcul de la resultante et du couple de pression
744 if (dimension == 2)
745 {
746 n0 = signe*face_normales(face,0) ;
747 n1 = signe*face_normales(face,1) ;
748
749 flux_bords_(face,0) = pres_tot*n0 ;
750 flux_bords_(face,1) = pres_tot*n1 ;
751
752 tab_flux_bords(0,n_bord,0) += flux_bords_(face,0) ;
753 tab_flux_bords(0,n_bord,1) += flux_bords_(face,1) ;
754
755// Calcul du moment exerce par le fluide sur le bord (OM/\F)
756 if (impr_mom)
757 tab_flux_bords(2,n_bord,0)+=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
758 if (impr_boundary)
759 {
760 tab_flux_bords(1,n_bord,0) += flux_bords_(face,0) ;
761 tab_flux_bords(1,n_bord,1) += flux_bords_(face,1) ;
762 }
763 }
764 else if (dimension == 3)
765 {
766 n0 = signe*face_normales(face,0) ;
767 n1 = signe*face_normales(face,1) ;
768 n2 = signe*face_normales(face,2) ;
769
770 flux_bords_(face,0) = pres_tot*n0 ;
771 flux_bords_(face,1) = pres_tot*n1 ;
772 flux_bords_(face,2) = pres_tot*n2 ;
773
774 tab_flux_bords(0,n_bord,0) += flux_bords_(face,0) ;
775 tab_flux_bords(0,n_bord,1) += flux_bords_(face,1) ;
776 tab_flux_bords(0,n_bord,2) += flux_bords_(face,2) ;
777
778 if (impr_mom)
779 {
780// Calcul du moment exerce par le fluide sur le bord (OM/\F)
781 tab_flux_bords(2,n_bord,0)+=flux_bords_(face,2)*xgr(face,1)-flux_bords_(face,1)*xgr(face,2);
782 tab_flux_bords(2,n_bord,1)+=flux_bords_(face,0)*xgr(face,2)-flux_bords_(face,2)*xgr(face,0);
783 tab_flux_bords(2,n_bord,2)+=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
784 }
785 if (impr_boundary)
786 {
787 tab_flux_bords(1,n_bord,0) += flux_bords_(face,0) ;
788 tab_flux_bords(1,n_bord,1) += flux_bords_(face,1) ;
789 tab_flux_bords(1,n_bord,2) += flux_bords_(face,2) ;
790 }
791 }
792 }
793 } //!< fin for n_bord
794
795// On somme les contributions de chaque processeur
796 mp_sum_for_each_item(tab_flux_bords);
797
798 if (je_suis_maitre())
799 {
800// Open files if needed
801 SFichier Flux_grad;
802 ouvrir_fichier(Flux_grad,"",1);
803 SFichier Flux_grad_moment;
804 ouvrir_fichier(Flux_grad_moment,"moment",impr_mom);
805 SFichier Flux_grad_sum;
806 ouvrir_fichier(Flux_grad_sum,"sum",impr_sum);
807
808// Write time
809 Flux_grad.add_col(sch.temps_courant());
810 if (impr_sum) Flux_grad_sum.add_col(sch.temps_courant());
811 if (impr_mom) Flux_grad_moment.add_col(sch.temps_courant());
812
813// Write flux on boundaries
814 for (int n_bord=0; n_bord<nb_bord; n_bord++)
815 {
816 if (dimension == 2)
817 {
818 Flux_grad.add_col(tab_flux_bords(0,n_bord,0));
819 Flux_grad.add_col(tab_flux_bords(0,n_bord,1));
820 if (impr_sum)
821 {
822 Flux_grad_sum.add_col(tab_flux_bords(1,n_bord,0));
823 Flux_grad_sum.add_col(tab_flux_bords(1,n_bord,1));
824 }
825 if (impr_mom) Flux_grad_moment.add_col(tab_flux_bords(2,n_bord,0));
826 }
827 else if (dimension == 3)
828 {
829 Flux_grad.add_col(tab_flux_bords(0,n_bord,0));
830 Flux_grad.add_col(tab_flux_bords(0,n_bord,1));
831 Flux_grad.add_col(tab_flux_bords(0,n_bord,2));
832 if (impr_sum)
833 {
834 Flux_grad_sum.add_col(tab_flux_bords(1,n_bord,0));
835 Flux_grad_sum.add_col(tab_flux_bords(1,n_bord,1));
836 Flux_grad_sum.add_col(tab_flux_bords(1,n_bord,2));
837 }
838 if (impr_mom)
839 {
840 Flux_grad_moment.add_col(tab_flux_bords(2,n_bord,0));
841 Flux_grad_moment.add_col(tab_flux_bords(2,n_bord,1));
842 Flux_grad_moment.add_col(tab_flux_bords(2,n_bord,2));
843 }
844 }
845 }
846 Flux_grad << finl;
847 if (impr_sum) Flux_grad_sum << finl;
848 if (impr_mom) Flux_grad_moment << finl;
849 }
850
851 const LIST(Nom)& Liste_Bords_a_imprimer = zone_VEF.zone().Bords_a_imprimer();
852 if (!Liste_Bords_a_imprimer.est_vide())
853 {
854 EcrFicPartage Flux_grad_face;
855 ouvrir_fichier_partage(Flux_grad_face,"",impr_bord);
856 for (int n_bord=0; n_bord<nb_bord; n_bord++)
857 {
858 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
859 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
860 int ndeb = le_bord.num_premiere_face();
861 int nfin = ndeb + le_bord.nb_faces();
862 if (zone_VEF.zone().Bords_a_imprimer().contient(le_bord.le_nom()))
863 {
864 if (je_suis_maitre())
865 {
866 Flux_grad_face << "# Force par face sur " << le_bord.le_nom() << " au temps ";
867 sch.imprimer_temps_courant(Flux_grad_face);
868 Flux_grad_face << " : " << finl;
869 }
870 for (face=ndeb; face<nfin; face++)
871 {
872 Flux_grad_face << "# Face a x= " << zone_VEF.xv(face,0) << " y= " << zone_VEF.xv(face,1);
873 if (dimension==3) Flux_grad_face << " z= " << zone_VEF.xv(face,2);
874 Flux_grad_face << " : Fx= " << flux_bords_(face,0) << " Fy= " << flux_bords_(face,1);
875 if (dimension==3) Flux_grad_face << " Fz= " << flux_bords_(face,2);
876 Flux_grad_face << finl;
877 }
878 Flux_grad_face.syncfile();
879 }
880 }
881 }
882 return 1;
883}
