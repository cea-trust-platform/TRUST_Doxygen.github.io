97// Finish building the champ_front_calc_interne
98void Echange_interne_global_impose::completer()
99{
100 Nom nom_pb(zone_Cl_dis().equation().probleme().le_nom());
101 Nom nom_bord(frontiere_dis().le_nom());
102 Champ_front_calc_interne& t_ext = ref_cast(Champ_front_calc_interne, T_ext().valeur());
103
104 t_ext.creer(nom_pb, nom_bord, "temperature");
105
106 Echange_global_impose::completer();
107
108 const Front_VF& fvf = ref_cast(Front_VF, frontiere_dis());
109 const Zone_VF& zvf = ref_cast(Zone_VF, zone_Cl_dis().zone_dis().valeur());
110
111 frontiere_dis().frontiere().creer_tableau_faces(surface_gap_, Array_base::NOCOPY_NOINIT);
112
113
114 if (zvf.zone().domaine().Axi1d())
115 {
116 const DomaineAxi1d& domax = ref_cast(DomaineAxi1d,zvf.zone().domaine());
117 const IntTab& face_voisins = zvf.face_voisins();
118 const DoubleTab& xv = zvf.xv();
119 const IntTab& fmap = t_ext.face_map();
120 for (int i=0; i<fvf.nb_faces(); i++)
121 {
122 int iopp = fmap(i);
123 int face = fvf.num_face(i);
124 int face_opp = fvf.num_face(iopp);
125 int elem = face_voisins(face,0)==-1 ? face_voisins(face,1) : face_voisins(face,0);
126 double x0 = domax.origine_repere(elem,0);
127 double y0 = domax.origine_repere(elem,1);
128 double x1 = xv(face,0);
129 double y1 = xv(face,1);
130 double x2 = xv(face_opp,0);
131 double y2 = xv(face_opp,1);
132 double r1 = sqrt((x1-x0)*(x1-x0)+(y1-y0)*(y1-y0));
133 double r2 = sqrt((x2-x0)*(x2-x0)+(y2-y0)*(y2-y0));
134 surface_gap_(i) = M_PI*(r1+r2);
135 }
136 }
137 else
138 {
139 for (int i=0; i<fvf.nb_faces(); i++)
140 {
141 int face = fvf.num_face(i);
142 surface_gap_(i) = zvf.surface(face);
143 }
144 }
145}
