51Entree& Solv_Gen::readOn(Entree& is )
52{
53 Motcle accolade_ouverte("{");
54 Motcle accolade_fermee("}");
55 Motcles les_parametres(9);
56 {
57 les_parametres[0] = "seuil";
58 les_parametres[1] = "impr";
59 les_parametres[2] = "solv_elem";
60 les_parametres[3] = "precond";
61 les_parametres[4] = "save_matrice|save_matrix";
62 les_parametres[5] = "quiet";
63 les_parametres[6] = "nb_it_max";
64 les_parametres[7] = "force";
65 les_parametres[8] = "return";
66 }
67 int rang;
68
69 Motcle motlu;
70 is >> motlu;
71 if (motlu != accolade_ouverte)
72 {
73 Cerr << "Error when reading the parameters for the generic solver " << finl;
74 Cerr << "One expected : " << accolade_ouverte << finl;
75 exit();
76 }
77 is >> motlu;
78 save_matrice_=0;
79 while (motlu != accolade_fermee)
80 {
81 rang = les_parametres.search(motlu);
82 switch(rang)
83 {
84 case 0:
85 {
86 is >> seuil_;
87 break;
88 }
89 case 1:
90 {
91 fixer_limpr(1);
92 break;
93 }
94 case 2:
95 {
96 is >> le_solveur_elem_;
97 break;
98 }
99 case 3:
100 {
101 is >> le_precond_;
102 break;
103 }
104 case 4:
105 {
106 save_matrice_=1;
107 break;
108 }
109 case 5:
110 {
111 fixer_limpr(-1);
112 break;
113 }
114 case 6:
115 {
116 is>>nb_it_max_;
117 nb_it_max_flag = 1;
118 break;
119 }
120 case 7:
121 {
122 force_ = 1;
123 break;
124 }
125 case 8:
126 {
127 return_on_error_ = 1;
128 break;
129 }
130 default :
131 {
132 Cerr << "Error when reading the parameters of the generic solver " << finl;
133 Cerr << "One expected " << les_parametres << " instead of " << motlu << finl;
134 exit();
135 }
136 }
137 is >> motlu;
138 }
139 if ((nb_it_max_flag ==0) && (force_==1) )
140 {
141 Cerr << "Error while reading the parameters of the generic solver " << finl;
142 Cerr << "When you choose to 'force' negative ipar[5], you should specify nb_it_max in your datafile" << finl;
143 exit();
144 }
145 return is;
146}

148int Solv_Gen::resoudre_systeme(const Matrice_Base& la_matrice, const DoubleVect& secmem, DoubleVect& solution)
149{
150
151// On definit tout nos vecteurs et leur dimension respective et nos constantes
152
153// n : la taille du systeme lineaire
154 int n = secmem.size();
155 if(solution.size() != n)
156 {
157 if( solution.size() ==0)
158 solution.resize(n);
159 else
160 {
161 Cerr << "Solv_Gen::resoudre_systeme : wrong dimension of the vectors" << finl;
162 return(-1);
163 }
164 }
165// Cerr<<"La matrice : "<<la_matrice.que_suis_je()<<finl;
166 if(sub_type(Matrice_Bloc,la_matrice))
167 {
168 const Matrice_Bloc& MB = ref_cast(Matrice_Bloc,la_matrice);
169
170 const Matrice& MB00 = MB.get_bloc(0,0);
171 if(sub_type(Matrice_Morse_Sym,MB00.valeur()))
172 {
173 const Matrice_Morse_Sym& M = ref_cast(Matrice_Morse_Sym,MB00.valeur());
174 return solve(la_matrice,M,secmem,solution);
175 }
176 else if(sub_type(Matrice_Morse,MB00.valeur()))
177 {
178 const Matrice_Morse& mat = ref_cast(Matrice_Morse,MB00.valeur());
179 return solve(la_matrice,mat,secmem,solution);
180 }
181 else
182 {
183 Cerr<<"ERROR SolvGen : one is not able to treat the case of a matrix"<<finl;
184 Cerr<<"of type : "<<MB00.que_suis_je()<<finl;
185 exit();
186 return -1;
187 }
188 }
189 else if(sub_type(Matrice_Morse_Sym,la_matrice))
190 {
191 const Matrice_Morse_Sym& M = ref_cast(Matrice_Morse_Sym,la_matrice);
192 return solve(M,M,secmem,solution);
193 }
194 else if(sub_type(Matrice_Morse,la_matrice))
195 {
196 const Matrice_Morse& mat = ref_cast(Matrice_Morse,la_matrice);
197 return solve(mat,mat,secmem,solution);
198 }
199 else
200 {
201 Cerr<<"ERROR SolvGen : one is not able to treat the case of a matrix"<<finl;
202 Cerr<<"of type : "<<la_matrice.que_suis_je()<<finl;
203 exit();
204 return -1;
205 }

208int Solv_Gen::solve(const Matrice_Base& matrice, const Matrice_Base& mat_loc, const DoubleVect& secmem, DoubleVect& solution)
209{
210// n : la taille du systeme lineaire
211 int n = secmem.size();
212 int ntot = secmem.size_array();
213 int niter=0;
214// definition de ipar et fpar
215 ArrOfInt ipar(16);
216 ArrOfDouble fpar(16);
217
218
219// definition de w : vecteur de travail de taille 9*n
220 ArrOfDouble w;
221 le_solveur_elem_.dimensionne_wks(ntot,w);
222
223// Initialisation
224 ipar[0] = 0; //on def tjrs cela pour commencer le calcul
225 if(le_precond_.non_nul())
226 {
227 if(sub_type(ILU,le_precond_.valeur()))
228 {
229 ILU& pilu = ref_cast(ILU,le_precond_.valeur());
230 ipar[1] = pilu.type_precond();
231 }
232 else
233 ipar[1]=1;
234 }
235 else
236 ipar[1]=0;
237 ipar[2] = 1; //!< pour le test de convergence
238 ipar[2] = 2;
239 ipar[3] = w.size_array();; //taille maximale de w
240 ipar[4] = 10;
241 int nmax_min = 100;
242 int nmax = std::max(Process::mp_sum(n), nmax_min);
243 ipar[5] = nmax; //!< nb max de produit matrice vect si nb negatif on s arrete a la convergence Si gros calcul (Process::mp_sum(n)>2147483647), specifier nb_it_max et imposer ipar[5] = -1 on le fait desormais en specifiant le mot cle 'force' dans le jdd
244//
245//
246//
247 if (force_ ==1) ipar[5] = -1;
248
249 fpar[0] = seuil_; //!< tolerance relative
250 fpar[0]= 1e-50 ; //!< GF les autres solveurs n'ont pas de tol relative
251 fpar[1] = seuil_; //!< tolerance absolue
252 fpar[10] = 0; //!< initialisation du compteur
253
254// solution = 1.0;
255 le_solveur_elem_.iteration(ntot,secmem,solution,ipar,fpar,w);
256 solution.echange_espace_virtuel();
257
258 DoubleTab W7, W8;
259
260 {
261// fait pointer v sur la sous partie qui commence a ipar[7]-1, de taille ntot:
262 W7.ref_array(w, ipar[7]-1, ntot);
263// Remarque TUTU: resize autorise car il ne change pas la taille et necessaire
264// pour initialiser line_size() du tableau W7:
265 W7.resize(ntot/secmem.line_size(), secmem.line_size());
266 W7.set_md_vector(secmem.get_md_vector());
267 W7.echange_espace_virtuel();
268 }
269
270 int ret = ipar[0];
271
272 while(ret != 0 && niter < nb_it_max_ )
273 {
274 niter++;
275 W7.ref_array(w, ipar[7]-1, ntot);
276 W7.resize(ntot/secmem.line_size(), secmem.line_size()); //idem remarque TUTU
277 W7.set_md_vector(secmem.get_md_vector());
278 W8.ref_array(w, ipar[8]-1, ntot);
279 W8.resize(ntot/secmem.line_size(), secmem.line_size()); //idem remarque TUTU
280 W8.set_md_vector(secmem.get_md_vector());
281 if (ret != 4 && ret != 6)
282 W7.echange_espace_virtuel();
283
284 if(ret==1)
285 {
286// on doit calculer le produit A*u
287// int ii;
288 matrice.multvect(W7, W8);
289 le_solveur_elem_.iteration(ntot,secmem,solution,ipar,fpar,w);
290 solution.echange_espace_virtuel();
291 }
292 else if(ret==2)
293 {
294 exit();
295// on doit calculer le produit At*u
296// matrice.multvectT(u, s);
297// int ii;
298 matrice.multvectT(W7, W8);
299 le_solveur_elem_.iteration(ntot,secmem,solution,ipar,fpar,w);
300 solution.echange_espace_virtuel();
301 }
302 else if((ret == 3) || (ret == 5))
303 {
304// On preconditionne
305// int ii;
306 if(sub_type(ILU,le_precond_.valeur()))
307 le_precond_.preconditionner(mat_loc, W7, W8);
308 else
309 le_precond_.preconditionner(mat_loc,secmem,solution);
310// le_precond_.preconditionner(matrice,secmem,solution);
311 le_solveur_elem_.iteration(ntot,secmem,solution,ipar,fpar,w);
312 solution.echange_espace_virtuel();
313 }
314 else if((ret == 4) || (ret == 6))
315 {
316 exit();
317// On preconditionne
318 if(sub_type(ILU,le_precond_.valeur()))
319 le_precond_.preconditionner(mat_loc, W7, W8);
320 else
321 le_precond_.preconditionner(mat_loc,secmem,solution);
322
323 le_solveur_elem_.iteration(ntot,secmem,solution,ipar,fpar,w);
324 solution.echange_espace_virtuel();
325 }
326 else if(ret == 10)
327 {
328 Cout<<"Case 10"<<finl;
329 Cout<<"The predefined stop test is retained"<<finl;
330 le_solveur_elem_.iteration(ntot,secmem,solution,ipar,fpar,w);
331 solution.echange_espace_virtuel();
332 }
333 else if(ret > 0)
334 {
335 Cerr<<"Case > 0"<<finl;
336 Cerr<<"That code does not exist "<<finl;
337 exit();
338 return(-1);
339 }
340 else if(ret < 0)
341 {
342 Cerr<<"ERROR in SolvGen"<<finl;
343 Cerr<<" The solver terminates with code number "<<ret<<finl;
344 if(ret==-1)
345 {
346 Cerr<<"Stop since the iteration number "<<finl;
347 Cerr<<"is greater than the preset limit"<<finl;
348 }
349 else if(ret==-2)
350 {
351 Cerr<<"return due to insufficient work space"<<finl;
352 }
353 else if(ret==-3)
354 {
355 Cerr<<"return due to anticipated break-down / divide by zero"<<finl;
356 Cerr<<"ipar[11] : "<<ipar[11]<<finl;
357 Cerr<<"w : "<<w<<finl;
358 Cerr<<"ind = fpar[1O] : "<<fpar[10]<<finl;
359 }
360 else if(ret==-4)
361 {
362 Cerr<<"the values of fpar(1) and fpar(2) are both <= 0, "<<finl;
363 Cerr<<"the valid ranges are 0 <= fpar(1) < 1, 0 <= fpar(2), "<<finl;
364 Cerr<<"and they can not be zero at the same time"<<finl;
365 }
366 else if(ret==-9)
367 {
368 Cerr<<"while trying to detect a break-down, "<<finl;
369 Cerr<<"an abnormal number is detected"<<finl;
370 }
371 else if(ret==-10)
372 {
373 Cerr<<"return due to some non-numerical reasons, "<<finl;
374 Cerr<<"e.g. invalid floating-point numbers etc."<<finl;
375 }
376 if (!return_on_error_) exit();
377 return(-1);
378 }
379 ret = ipar[0];
380 if(limpr()==1)
381 {
382// Cerr<<"++ Cas : "<<ret<<finl;
383// Cerr<<"fpar(5) -- current residual norm (if available) : "<<fpar[4]<<finl;
384// Cerr<<"fpar(6) -- current residual/error norm : "<<fpar[5]<<finl;
385 Cout<< fpar[5]<<" ";
386 if (niter%10==0) Cout<<finl;
387 }
388 }
389
390 if (je_suis_maitre())
391 {
392// Cerr<<"--------------------------------------------------"<<finl;
393 if (limpr()>-1)
394 {
395 if (niter == nb_it_max_)
396 {
397 Cout<<finl<<"Maximum number of iteration nb_it_max for SolvGen = "<<nb_it_max_ << " reached..." <<finl;
398 if (nb_it_max_flag == 0) Cout<<"Advice: You can modify nb_it_max by setting it in your datafile..." <<finl;
399 }
400// Cout<<finl<<"End of the resolution by SolvGen after "<<niter <<" iterations, current residual/error norm "<<fpar[5]<<" initial residual/error norm "<<fpar[2]<<finl;
401 double residu_relatif = (fpar[2] > 0 ? fpar[5] / fpar[2] : fpar[5]);
402 Cout << finl << "Final residue: " << fpar[5] << " ( " << residu_relatif << " )"<<finl;
403 }
404// Cerr<<"fpar(3) -- initial residual/error norm : "<<fpar[2]<<finl;
405// Cerr<<"fpar(4) -- target residual/error norm : "<<fpar[3]<<finl;
406// Cerr<<"fpar(5) -- current residual norm (if available) : "<<fpar[4]<<finl;
407// Cerr<<"fpar(6) -- current residual/error norm : "<<fpar[5]<<finl;
408// Cerr<<"fpar(7) -- convergence rate : "<<fpar[6]<<finl;
409 }
410
411 if (0)
412 {
413 DoubleVect tmp(solution);
414 matrice.multvect(solution,tmp);
415 int ii;
416 /*
417 Process::Journal()<<" A = "<<matrice<<finl;
418 Process::Journal()<<" ipar = "<<finl;
419 for(ii=0;ii<ipar.size_array();ii++)
420 Process::Journal()<<ii<<": "<<ipar[ii]<<finl;
421 Process::Journal()<<" w = "<<finl;
422 for(ii=0;ii<w.size_array();ii++)
423 Process::Journal()<<ii<<": "<<w[ii]<<finl;
424 Process::Journal()<<" x = "<<solution<<finl;
425 // solution.ecrit(Process::Journal());
426 Process::Journal()<<" A*x = "<<tmp<<finl;
427 Process::Journal()<<" b = "<<secmem<<finl;
428 tmp.ecrit(Process::Journal());
429 */
430 for(ii=0; ii<solution.size(); ii++)
431 tmp(ii)-=secmem(ii);
432 tmp.echange_espace_virtuel();
433 double err=0.;
434 for(ii=0; ii<solution.size(); ii++)
435 err+=(tmp(ii)*tmp(ii));
436 err = Process::mp_sum(err);
437 Cerr<<"ERROR: "<<sqrt(err)<<finl;
438 }
439 return niter;
440}
