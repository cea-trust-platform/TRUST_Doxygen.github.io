35DoubleTab& Source_Generique_P0_Elem::ajouter(DoubleTab& resu) const
36{
37 Champ espace_stockage;
38 const Champ_base& champ_calc = ch_source_->get_champ(espace_stockage);
39 const DoubleTab& valeurs_calc = champ_calc.valeurs();
40
41 int nb_elem = la_zone->nb_elem();
42 const DoubleVect& vol = la_zone->volumes();
43 const DoubleVect& poro_vol = la_zone->porosite_elem();
44
45 for (int elem = 0; elem<nb_elem; elem++)
46 resu(elem) += valeurs_calc(elem)*vol(elem)*poro_vol(elem);
47
48 resu.echange_espace_virtuel();
49 return resu;
50}
