29Entree& Diametre_bulles_constant::readOn(Entree& is)
30{
31 Param param(que_suis_je());
32 param.ajouter("diametre", &d_bulle_, Param::REQUIRED);
33 param.lire_avec_accolades_depuis(is);
34
35 Pb_Multiphase& pb = ref_cast(Pb_Multiphase, pb_.valeur());
36 int N = pb.nb_phases();
37 const Discret_Thyd& dis=ref_cast(Discret_Thyd,pb.discretisation());
38 Noms noms(N), unites(N);
39 noms[0] = "diametre_bulles";
40 unites[0] = "m";
41 Motcle typeChamp = "champ_elem" ;
42 const Zone_dis& z = ref_cast(Zone_dis, pb.domaine_dis().zone_dis(0));
43 dis.discretiser_champ(typeChamp, z.valeur(), scalaire, noms , unites, N, 0, diametres_);
44
45 champs_compris_.ajoute_champ(diametres_);
46
47 for (int n = 0; n < pb.nb_phases(); n++) //recherche de n_l, n_g : phase {liquide,gaz}_continu en priorite
48 if (pb.nom_phase(n).debute_par("liquide") && (n_l < 0 || pb.nom_phase(n).finit_par("continu"))) n_l = n;
49 if (n_l < 0) Process::exit(que_suis_je() + " : liquid phase not found!");
50
51 DoubleTab& tab_diametres = diametres_->valeurs();
52 for (int i = 0 ; i < tab_diametres.dimension_tot(0) ; i++)
53 for (int n = 0 ; n <N ; n++)
54 if (n!=n_l) tab_diametres(i, n) = d_bulle_;
55
56 return is;
57}
