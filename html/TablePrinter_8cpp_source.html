72void TablePrinter::PrintHorizontalLine(int updown)
73{
74 *out_stream_ << (updown > 0 ? "┌" : updown < 0 ? "└" : "├"); //!< the left bar
75
76 for (int i = 0; i < (int) column_widths_.size(); i++)
77 for (int j = 0; j < column_widths_[i]; j++)
78 *out_stream_ << (!i || j ? "─" : updown > 0 ? "┬─" : updown < 0 ? "┴─" : "┼─");
79
80 *out_stream_ << (updown > 0 ? "┐" : updown < 0 ? "┘" : "┤"); //!< the right bar
81 *out_stream_ << "\n";
82}

84void TablePrinter::PrintHeader()
85{
86 PrintHorizontalLine(1);
87 *out_stream_ << "│";
88
89 for (int i=0; i<get_num_columns(); ++i)
90 {
91
92 if(flush_left_)
93 *out_stream_ << std::left;
94 else
95 *out_stream_ << std::right;
96
97 *out_stream_ << std::setw(column_widths_.at(i)) << column_headers_.at(i).substr(0, column_widths_.at(i));
98 if (i != get_num_columns()-1)
99 {
100 *out_stream_ << separator_;
101 }
102 }
103
104 *out_stream_ << "│\n";
105 PrintHorizontalLine();
106}
