125// (*my_params) have been filled by constructor.

127// Signification:

129// Contraintes:
130// Acces:

141//! This method is called once at the beginning, before any other one of<br>the interface Problem.
142bool ProblemTrio::initialize()
143{
144 Process::exception_sur_exit=1;
145 if (((*my_params).problem_name=="default_vvvvv") || ((*my_params).data_file=="default_vvvvv"))
146 throw WrongArgument("??","Constructor","data","data shoud point to the name of a Probleme_U");
147
148#ifdef MPI_
149// exception if I don't belong to comm !
150 True_int rank_in_comm=0;
151 if ((*my_params).is_mpi!=0)
152 {
153 if (MPI_Comm_rank((*my_params).comm,&rank_in_comm)!=MPI_SUCCESS)
154 throw WrongArgument((*my_params).problem_name,"initialize","comm","This process should belong to comm");
155 if (rank_in_comm==MPI_UNDEFINED)
156 throw WrongArgument((*my_params).problem_name,"initialize","comm","This process should belong to comm");
157
158 Comm_Group_MPI::set_trio_u_world((*my_params).comm);
159
160 }
161 Comm_Group_MPI::set_must_mpi_initialize(0); //!< ???
162#endif
163 int argc=2;
164 char** argv=new char*[argc];
165 string code="TRUST wrapper";
166// Les copies sont necessaires pour se debarrasser des const...
167 argv[0]=new char[code.length()+1];
168 strcpy(argv[0],code.c_str());
169 argv[1]=new char[(*my_params).data_file.length()+1];
170 strcpy(argv[1],(*my_params).data_file.c_str());
171// pour salome
172 if (p) delete p;
173 p=NULL;
174 int res;
175// on lance avec ou sans mpi
176 res=main_TRUST(argc,argv,p,(*my_params).is_mpi);
177
178
179 delete [] argv[0];
180 delete [] argv[1];
181 delete [] argv;
182
183 Nom nom("ICoCoProblemName"),nom_pb;
184 if (nom.interprete().objet_existant(nom))
185 {
186 nom_pb=ref_cast(Nom,get_obj(nom));
187 Cout<<finl<<" ICoCoProblemName from data file "<<nom_pb<< finl;
188 }
189 else
190 {
191 nom_pb="pb";
192 Cout<<finl<<" ICoCoProblemName not found in data file, we try "<<nom_pb<< finl;
193 }
194
195 (*my_params).problem_name=nom_pb;
196 pb=&ref_cast(Probleme_U,get_obj(nom_pb));
197 if (res || !pb)
198 throw WrongArgument((*my_params).problem_name,"initialize","problem_name","No problem of that name found in data file");
199 initialize_pb(*pb);
200
201// Register signal and signal handler (SIGINT) if user presses ctrl-c during exec
202 if (!Objet_U::DEACTIVATE_SIGINT_CATCH)
203 {
204 Catch_SIGINIT sig;
205 sig.set_nom_cas_pour_signal(pb->nom_du_cas());
206 signal(SIGINT, Catch_SIGINIT::signal_callback_handler);
207 }
208
209// Print the initialization CPU statistics
210 if (!Objet_U::disable_TU)
211 {
212 statistiques().dump("Statistiques d'initialisation du calcul", 0);
213 print_statistics_analyse("Statistiques d'initialisation du calcul", 0);
214 }
215 statistiques().reset_counters();
216 statistiques().begin_count(temps_total_execution_counter_);
217 return true;
218}
219bool ProblemTrio::initialize_pb(Probleme_U& pb_to_solve)

221 if (pb==NULL)
222 pb=&pb_to_solve;
223 pb_to_solve.initialize();
224 pb_to_solve.postraiter(1);
225 return true;
226}

732// Specific to TRUST, and outside the ICoCo standard
733// Get a direct access to the unknown array of an equation, wrapped into a dummy
734// MEDDoubleField. The mesh and other field information are irrelevant, only the underlying
735// array should be considered.
736ICoCo::MEDDoubleField ProblemTrio::getDirectAccessToUnknown(const std::string& unk_name, bool isFuture)
737{
738 using namespace MEDCoupling;
739
740 if(sub_type(Probleme_base,*pb))
741 {
742 Probleme_base& pb_base = ref_cast(Probleme_base,*pb);
743 double * ptr = nullptr;
744 size_t nbTup = -1, nbCompo = -1;
745 for (int i=0; i < pb_base.nombre_d_equations(); i++)
746 {
747 Champ_Inc& ci = pb_base.equation(i).inconnue();
748 assert(ci.valeurs().nb_dim() == 2);
749 if (Motcle(ci.le_nom()) == Motcle(unk_name))
750 {
751 DoubleTab& t = isFuture ? ci.futur() : ci.valeurs() ;
752 ptr = t.addr();
753 nbTup = t.dimension_tot(0);
754 nbCompo = t.dimension_tot(1);
755 break;
756 }
