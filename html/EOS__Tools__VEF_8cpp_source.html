106//! Calcule la moyenne volumique de la grandeur P1NC donnee
107/** \return DoubleTab\&<br>
108Signification: rho discretise par face<br>
109Contraintes: */
110double EOS_Tools_VEF::moyenne_vol(const DoubleTab& tab) const
111{
112 assert(tab.line_size() == 1);
113 double x = Champ_P1NC::calculer_integrale_volumique(la_zone.valeur(), tab, FAUX_EN_PERIO);
114// La facon simple de faire serait celle-ci, mais c'est faux a cause de FAUX_EN_PERIO
115// qui compte deux fois les volumes entrelaces des faces periodiques:
116 DoubleVect un;
117 la_zone.valeur().creer_tableau_faces(un, Array_base::NOCOPY_NOINIT);
118 un = 1.;
119 double y = Champ_P1NC::calculer_integrale_volumique(la_zone.valeur(), un, FAUX_EN_PERIO);
120 return x / y;
121}

124//! Div(u) is computed on faces from Div(u) on elements
125void EOS_Tools_VEF::divu_discvit(const DoubleTab& DivVelocityElements, DoubleTab& DivVelocityFaces)
126{
127 assert_espace_virtuel_vect(DivVelocityElements);
128// Corrections pour moyenner div(u) sur les faces
129 const DoubleVect& volumes = la_zone->volumes();
130 const DoubleVect& volumes_entrelaces = la_zone->volumes_entrelaces();
131 const DoubleVect& volumes_entrelaces_Cl = ref_cast(Zone_Cl_VEF,la_zone_Cl->valeur()).volumes_entrelaces_Cl();
132
133 IntTab& face_voisins = la_zone->face_voisins();
134 int premiere_fac_std=la_zone->premiere_face_std();
135
136// remplissage de div(u) sur les faces
137 for (int face=0; face<premiere_fac_std; face++)
138 {
139 int nb_comp=0;
140 DivVelocityFaces(face)=0;
141 for (int i=0 ; i<2 ; i++)
142 {
143 int elem= face_voisins(face,i);
144 if (elem!=-1)
145 {
146 nb_comp++;
147 DivVelocityFaces(face) += DivVelocityElements(elem)*(volumes_entrelaces_Cl(face)/volumes(elem));
148 }
149 }
150 DivVelocityFaces(face) /= nb_comp;
151 }
152
153 int size = la_zone->nb_faces();
154 for (int face=premiere_fac_std; face<size; face++)
155 {
156 int nb_comp=0;
157 DivVelocityFaces(face)=0;
158 for (int i=0 ; i<2 ; i++)
159 {
160 int elem= face_voisins(face,i);
161 if (elem!=-1)
162 {
163 nb_comp++;
164 DivVelocityFaces(face) += DivVelocityElements(elem)*(volumes_entrelaces(face)/volumes(elem));
165 }
166 }
167 DivVelocityFaces(face) /= nb_comp;
168 }
169}

185//! Calcule le second membre de l'equation de continuite pour une discretisation VEF_P1B:<br>div(U) = W = dZ/dT + U.grad(Z) avec Z=ln(rho)
186void EOS_Tools_VEF::secmembre_divU_Z(DoubleTab& tab_W) const
187{
188 double dt = le_fluide().vitesse()->equation().schema_temps().pas_de_temps();
189 int face, elem, som,som_glob;
190 const IntTab& face_sommets = la_zone->face_sommets();
191 int nb_elem_tot=la_zone->nb_elem_tot();
192 int nb_som_tot=la_zone->zone().nb_som_tot();
193 int nb_faces_tot=la_zone->nb_faces_tot();
194 const Equation_base& eq=le_fluide().vitesse()->equation();
195 DoubleTab tab_dZ;
196
197// Dimensionnement de tab_dZ
198 const Navier_Stokes_std& eqns = ref_cast(Navier_Stokes_std,eq);
199 const DoubleTab& pression = eqns.pression().valeurs();
200 tab_dZ = pression;
201
202 DoubleVect tab_rhon_som(nb_som_tot);
203 DoubleVect volume_int_som(nb_som_tot);
204 DoubleVect tab_rhonp1_som(nb_som_tot);
205
206 const DoubleTab& tab_rhon = le_fluide().loi_etat()->rho_n();
207 const DoubleTab& tab_rhonp1 = le_fluide().loi_etat()->rho_np1();
208 DoubleTab tab_rhonP1_,tab_rhonp1P1_;
209 const DoubleVect& porosite_face = la_zone->porosite_face();
210 const DoubleTab& tab_rhonP1=modif_par_porosite_si_flag(tab_rhon,tab_rhonP1_,1,porosite_face);
211 const DoubleTab& tab_rhonp1P1=modif_par_porosite_si_flag(tab_rhonp1,tab_rhonp1P1_,1,porosite_face);
212 const IntTab& elem_faces = la_zone->elem_faces();
213 const DoubleVect& volumes = la_zone->volumes();
214 const DoubleVect& volumes_entrelaces = la_zone->volumes_entrelaces();
215 const Zone_VEF_PreP1b& zp1b=ref_cast(Zone_VEF_PreP1b,la_zone.valeur());
216 const DoubleVect& volumes_controle=zp1b.volume_aux_sommets();
217
218 double rn,rnp1;
219 int nfe = la_zone->zone().nb_faces_elem();
220 int nsf = la_zone->nb_som_face();
221 const Domaine& dom=la_zone->zone().domaine();
222
223// calcul de la somme des volumes entrelacees autour d'un sommet
224 volume_int_som=0.;
225 for (face=0 ; face<nb_faces_tot ; face++)
226 for (som=0 ; som<nsf ; som++)
227 {
228 som_glob=dom.get_renum_som_perio(face_sommets(face,som));
229 volume_int_som(som_glob)+=volumes_entrelaces(face);
230 }
231// discretisation de rho sur les sommets
232 tab_rhon_som=0;
233 tab_rhonp1_som=0;
234
235 double pond;
236 pond=1./nsf; //!< version_originale
237 for (face=0 ; face<nb_faces_tot ; face++)
238 {
239 for (som=0 ; som<nsf ; som++)
240 {
241 som_glob=dom.get_renum_som_perio(face_sommets(face,som));
242 pond=volumes_entrelaces(face)/volume_int_som(som_glob);
243 tab_rhon_som(som_glob) += tab_rhonP1(face)*pond;
244 tab_rhonp1_som(som_glob) += tab_rhonp1P1(face)*pond;
245 }
246 }
247
248// Corrections pour test de la moyenne de la derivee de la masse volumique
249 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tab_dZ=",tab_dZ);
250 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tab_rhonP1=",tab_rhonP1);
251 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tab_rhonp1P1=",tab_rhonp1P1);
252 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z dt=",dt);
253
254 int decal=0;
255 int p_has_elem=zp1b.get_alphaE();
256 int nb_case=nb_elem_tot*p_has_elem;
257 for (elem=0 ; elem<nb_case ; elem++)
258 {
259 rn=0;
260 rnp1=0;
261 for (face=0 ; face<nfe ; face++)
262 {
263 rn += tab_rhonP1(elem_faces(elem,face));
264 rnp1 += tab_rhonp1P1(elem_faces(elem,face));
265 }
266 tab_dZ(elem) = (rnp1-rn)/(nfe*dt);
267 }
268
269 decal+=nb_case;
270 tab_dZ.echange_espace_virtuel();
271
272 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tab_dZ=",tab_dZ);
273 int p_has_som=zp1b.get_alphaS();
274 nb_case=nb_som_tot*p_has_som;
275
276 for (som=0 ; som<nb_case ; som++)
277 tab_dZ(decal+som) = ((tab_rhonp1_som(som))-(tab_rhon_som(som)))/dt;
278
279 decal+=nb_case;
280 int p_has_arrete=zp1b.get_alphaA();
281 int nb_ar_tot=la_zone->zone().nb_aretes_tot();
282 nb_case=nb_ar_tot*p_has_arrete;
283
284 for (int ar=0 ; ar<nb_case ; ar++)
285 tab_dZ(decal+ar) =0;
286
287 assert(decal+nb_case==tab_W.size_totale());
288 tab_dZ.echange_espace_virtuel();
289 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tab_dZ=",tab_dZ);
290
291 double coefdivelem=1,coefdivsom=1;
292 decal=0;
293 nb_case=nb_elem_tot*p_has_elem;
294
295 for (elem=0 ; elem<nb_case ; elem++)
296 tab_W(elem) = -coefdivelem*tab_dZ(elem) * volumes(elem);
297
298 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tabW elem",tab_W);
299 decal+=nb_case;
300 nb_case=nb_som_tot*p_has_som;
301
302 for (som=0 ; som<nb_case ; som++)
303 tab_W(decal+som) = -coefdivsom*tab_dZ(decal+som)*volumes_controle(som)*1;
304
305 decal+=nb_case;
306 nb_case=nb_ar_tot*p_has_arrete;
307 for (int ar=0 ; ar<nb_case ; ar++)
308 {
309// pour l'instant on n'a pas calcule tab_dZ aux aretes
310 assert(tab_dZ(decal+ar)==0);
311 tab_W(decal+ar) =0;
312 }
313
314 assert(decal+nb_case==tab_W.size_totale());
315 tab_W.echange_espace_virtuel();
316 Debog::verifier("EOS_Tools_VEF::secmembre_divU_Z tab_W=",tab_W);
317}
