338//! Ouverture/creation d'un fichier d'impression d'un terme source<br>A surcharger dans les classes derivees.
339void Source_base::ouvrir_fichier(SFichier& os,const Nom& type, const int flag) const
340{
341// flag nul on n'ouvre pas le fichier
342 if (flag==0)
343 return ;
344// ToDo provisoire:
345 counters[type.getString()]++;
346 if (counters[type.getString()]>1 && type!="")
347 {
348 Cerr << "Code should be rewritten to have only one call to Source_base::ouvrir_fichier for " << type << " source and not " << counters[type.getString()] << " times." << finl;
349 Process::exit();
350 }
351 const Probleme_base& pb=equation().probleme();
352 const Schema_Temps_base& sch=pb.schema_temps();
353 const int precision = sch.precision_impr(), wcol = std::max(col_width_, sch.wcol()), gnuplot_header = sch.gnuplot_header();
354 os.set_col_width(wcol);
355
356 Nom nomfichier(out_);
357 if (type!="") nomfichier+=(Nom)"_"+type;
358 nomfichier+=".out";
359
360// On cree le fichier a la premiere impression avec l'en tete
361 if (sch.nb_impr()==1 && !pb.reprise_effectuee())
362 {
363 os.ouvrir(nomfichier);
364 SFichier& fic=os;
365 Nom espace="\t\t";
366 fic << (Nom)"# Printing of the source term "+que_suis_je()+" of the equation "+equation().que_suis_je()+" of the problem "+equation().probleme().le_nom() << finl;
367 fic << "# " << description() << finl;
368 if (!gnuplot_header) fic << "#";
369 os.set_col_width(wcol - !gnuplot_header);
370 fic.add_col("Time");
371 os.set_col_width(wcol);
372 fic << finl;
373 }
374// Sinon on l'ouvre
375 else
376 {
377 os.ouvrir(nomfichier,ios::app);
378 }
379 os.precision(precision);
380 os.setf(ios::scientific);
381}
