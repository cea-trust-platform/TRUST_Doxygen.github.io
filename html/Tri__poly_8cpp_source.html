39//! remplit le tableau face_normales dans la Zone_poly
40void Tri_poly::normale(int num_Face,DoubleTab& Face_normales,
41 const IntTab& Face_sommets,
42 const IntTab& Face_voisins,
43 const IntTab& elem_faces,
44 const Zone& zone_geom) const
45{
46 const DoubleTab& les_coords = zone_geom.domaine().coord_sommets();
47
48 double x1,y1;
49 double nx,ny;
50 int no3;
51 int f0;
52 int n0 = Face_sommets(num_Face,0);
53 int n1 = Face_sommets(num_Face,1);
54 x1 = les_coords(n0,0)-les_coords(n1,0);
55 y1 = les_coords(n0,1)-les_coords(n1,1);
56 nx = -y1;
57 ny = x1;
58
59// Orientation de la normale de elem1 vers elem2
60// pour cela recherche du sommet de elem1 qui n'est pas sur la Face
61 int elem1 = Face_voisins(num_Face,0);
62 if ( (f0 = elem_faces(elem1,0)) == num_Face )
63 f0 = elem_faces(elem1,1);
64 if ( (no3 = Face_sommets(f0,0)) != n0 && no3 != n1 )
65 ;
66 else
67 no3 = Face_sommets(f0,1);
68
69 x1 = les_coords(no3,0) - les_coords(n0,0);
70 y1 = les_coords(no3,1) - les_coords(n0,1);
71
72 if ( (nx*x1+ny*y1) > 0 )
73 {
74 Face_normales(num_Face,0) = - nx;
75 Face_normales(num_Face,1) = - ny;
76 }
77 else
78 {
79 Face_normales(num_Face,0) = nx;
80 Face_normales(num_Face,1) = ny;
81 }
82}
