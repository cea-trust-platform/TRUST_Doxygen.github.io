235int Op_Conv_PolyMAC_base::impr(Sortie& os) const
236{
237 const Zone& ma_zone=la_zone_poly_->zone();
238 const int impr_mom=ma_zone.Moments_a_imprimer();
239 const int impr_sum=(ma_zone.Bords_a_imprimer_sum().est_vide() ? 0:1);
240 const int impr_bord=(ma_zone.Bords_a_imprimer().est_vide() ? 0:1);
241 const Schema_Temps_base& sch = la_zcl_poly_->equation().probleme().schema_temps();
242 DoubleTab& tab_flux_bords= flux_bords();
243 int nb_comp = tab_flux_bords.nb_dim() > 1 ? tab_flux_bords.dimension(1) : 0;
244 DoubleVect bilan(nb_comp);
245 const int nb_faces = la_zone_poly_->nb_faces_tot();
246 DoubleTab xgr(nb_faces,dimension);
247 xgr=0.;
248 if (impr_mom)
249 {
250 const DoubleTab& xgrav = la_zone_poly_->xv();
251 const ArrOfDouble& c_grav=ma_zone.cg_moments();
252 for (int num_face=0; num_face <nb_faces; num_face++)
253 for (int i=0; i<dimension; i++)
254 xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
255 }
256 int k,face;
257 int nb_front_Cl=la_zone_poly_->nb_front_Cl();
258 DoubleTrav flux_bords2( 5, nb_front_Cl , nb_comp) ;
259 flux_bords2=0;
260 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
261 {
262 const Cond_lim& la_cl = la_zcl_poly_->les_conditions_limites(num_cl);
263 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
264 int ndeb = frontiere_dis.num_premiere_face();
265 int nfin = ndeb + frontiere_dis.nb_faces();
266 for (face=ndeb; face<nfin; face++)
267 {
268 for(k=0; k<nb_comp; k++)
269 {
270 flux_bords2(0,num_cl,k)+=tab_flux_bords(face, k);
271 if (ma_zone.Bords_a_imprimer_sum().contient(frontiere_dis.le_nom()))
272 flux_bords2(3,num_cl,k)+=tab_flux_bords(face, k);
273 } /* fin for k */
274 if (impr_mom)
275 {
276 if (dimension==2)
277 {
278 flux_bords2(4,num_cl,0)+=tab_flux_bords(face,1)*xgr(face,0)-tab_flux_bords(face,0)*xgr(face,1);
279 }
280 else
281 {
282 flux_bords2(4,num_cl,0)+=tab_flux_bords(face,2)*xgr(face,1)-tab_flux_bords(face,1)*xgr(face,2);
283 flux_bords2(4,num_cl,1)+=tab_flux_bords(face,0)*xgr(face,2)-tab_flux_bords(face,2)*xgr(face,0);
284 flux_bords2(4,num_cl,2)+=tab_flux_bords(face,1)*xgr(face,0)-tab_flux_bords(face,0)*xgr(face,1);
285 }
286 }
287 } /* fin for face */
288 }
289 mp_sum_for_each_item(flux_bords2);
290
291 if (je_suis_maitre() && nb_comp > 0)
292 {
293// SFichier Flux;
294 if (!Flux.is_open()) ouvrir_fichier(Flux,"",1);
295// SFichier Flux_moment;
296 if (!Flux_moment.is_open()) ouvrir_fichier(Flux_moment,"moment",impr_mom);
297// SFichier Flux_sum;
298 if (!Flux_sum.is_open()) ouvrir_fichier(Flux_sum,"sum",impr_sum);
299 Flux.add_col(sch.temps_courant());
300 if (impr_mom) Flux_moment.add_col(sch.temps_courant());
301 if (impr_sum) Flux_sum.add_col(sch.temps_courant());
302 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
303 {
304 for(k=0; k<nb_comp; k++)
305 {
306 Flux.add_col(flux_bords2(0,num_cl,k));
307 if (impr_sum) Flux_sum.add_col(flux_bords2(3,num_cl,k));
308 bilan(k)+=flux_bords2(0,num_cl,k);
309 }
310 if (dimension==3)
311 {
312 for (k=0; k<nb_comp; k++)
313 if (impr_mom) Flux_moment.add_col(flux_bords2(4,num_cl,k));
314 }
315 else
316 {
317 if (impr_mom) Flux_moment.add_col(flux_bords2(4,num_cl,0));
318 }
319 } /* fin for num_cl */
320 for(k=0; k<nb_comp; k++)
321 Flux.add_col(bilan(k));
322 Flux << finl;
323 if (impr_sum) Flux_sum << finl;
324 if (impr_mom) Flux_moment << finl;
325 }
326 const LIST(Nom)& Liste_Bords_a_imprimer = la_zone_poly_->zone().Bords_a_imprimer();
327 if (!Liste_Bords_a_imprimer.est_vide() && nb_comp > 0)
328 {
329 EcrFicPartage Flux_face;
330 ouvrir_fichier_partage(Flux_face,"",impr_bord);
331 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
332 {
333 const Frontiere_dis_base& la_fr = la_zcl_poly_->les_conditions_limites(num_cl).frontiere_dis();
334 const Cond_lim& la_cl = la_zcl_poly_->les_conditions_limites(num_cl);
335 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
336 int ndeb = frontiere_dis.num_premiere_face();
337 int nfin = ndeb + frontiere_dis.nb_faces();
338 if (ma_zone.Bords_a_imprimer().contient(la_fr.le_nom()))
339 {
340 if(je_suis_maitre())
341 {
342 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps ";
343 sch.imprimer_temps_courant(Flux_face);
344 Flux_face << " : " << finl;
345 }
346 for (face=ndeb; face<nfin; face++)
347 {
348 if (dimension == 2)
349 Flux_face << "# Face a x= " << la_zone_poly_->xv(face,0) << " y= " << la_zone_poly_->xv(face,1) << " : ";
350 else if (dimension == 3)
351 Flux_face << "# Face a x= " << la_zone_poly_->xv(face,0) << " y= " << la_zone_poly_->xv(face,1) << " z= " << la_zone_poly_->xv(face,2) << " : ";
352 for(k=0; k<nb_comp; k++)
353 Flux_face << tab_flux_bords(face, k) << " ";
354 Flux_face << finl;
355 }
356 Flux_face.syncfile();
357 }
358 }
359 }
360 return 1;
361}
