66//! Lit les caracteristques de la loi de parois<br>a partir d'un flot d'entree.<br>Format:<br>type_de_loi_de_paroi<br>Les valeurs possibles du type de loi de paroi sont:<br>- "loi_standard_hydr"<br>- "negligeable"<br>- "loi_VanDriest"<br>- "loi_standard_hydr_scalaire"
67/** \return Entree\&<br>
68Signification: le flot d'entree modifie<br>
69Contraintes: */
70Entree& Turbulence_paroi_scal::readOn(Entree& s)
71{
72 Cerr << "Lecture du type de loi de parois " << finl;
73// On recupere l'equation hydraulique
74 const Equation_base& eqn = mon_modele_turb_scal->equation().probleme().equation(0);
75 Motcle typ;
76 s >> typ;
77
78 const RefObjU& modele_turbulence = eqn.get_modele(TURBULENCE);
79 const Mod_turb_hyd_base& mod_turb_hydr = ref_cast(Mod_turb_hyd_base,modele_turbulence.valeur());
80 const Turbulence_paroi& loi = mod_turb_hydr.loi_paroi();
81
82 if (typ!="negligeable_scalaire")
83 if ((loi.valeur().que_suis_je()=="negligeable_VDF") || (loi.valeur().que_suis_je()=="negligeable_VEF"))
84 {
85 Cerr<<"La loi de paroi de type "<<typ<<" choisie pour le scalaire n'est pas compatible avec"<<finl;
86 Cerr<<"la loi de type "<<loi.valeur().que_suis_je()<<" choisie pour l'hydraulique"<<finl;
87 Cerr<<"Utiliser le type 'negligeable_scalaire' pour le scalaire ou utiliser une loi de paroi"<<finl;
88 Cerr<<"non negligeable pour l hydraulique"<<finl;
89 exit();
90 }
91 typ+="_";
92
93 Nom discr=eqn.discretisation().que_suis_je();
94
95// les operateurs de diffusion sont communs aux discretisations VEF et VEFP1B
96 if(discr=="VEFPreP1B") discr="VEF";
97 typ+=discr;
98
99 if (typ=="loi_analytique_scalaire_VDF")
100 {
101 Cerr<<"La loi de paroi scalaire de type loi_analytique_scalaire"<<finl;
102 Cerr<<"n est utilisable qu avec une discretisation de type VEF"<<finl;
103 exit();
104 }
105 Cerr<<"et typage :"<<typ<<finl;
106 DERIV(Turbulence_paroi_scal_base)::typer(typ);
107 valeur().associer_modele(mon_modele_turb_scal.valeur());
108 valeur().associer(eqn.zone_dis(),eqn.zone_Cl_dis());
109
110 return s;
111}
