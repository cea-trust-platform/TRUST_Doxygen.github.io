52//! Les volumes calcules dans discretiser() ne tiennent pas compte de<br>la periodicite. Pour les sommets periodiques, on fait la somme<br>de tous les volumes des sommets "identiques" au sens de la periodicite.
53void Zone_VEF_PreP1b::discretiser()
54{
55 Zone_VEF::discretiser();
56
57 zone().domaine().creer_tableau_sommets(volumes_som, Array_base::NOCOPY_NOINIT);
58
59 double coeff=1./3.;
60 if (dimension==3)
61 coeff=1./4.;
62
63 const IntTab& elements = zone().les_elems();
64 const int nb_som_elem = elements.dimension(1);
65// Boucle sur tous les elements car on ajoute des contributions aux sommets de joints:
66 const int n = nb_elem_tot();
67// On a besoin de l'espace virtuel des volumes
68 const DoubleVect& volume_elem = volumes();
69 assert_espace_virtuel_vect(volume_elem);
70
71// Annule tout le tableau car on va faire += sur des items virtuels
72// (sinon acces a des cases non initialisees)
73 operator_egal(volumes_som, 0., VECT_ALL_ITEMS);
74 for(int k=0; k<n; k++)
75 {
76 double volume = coeff * volume_elem(k);
77 for(int isom=0; isom<nb_som_elem; isom++)
78 {
79 int som = elements(k, isom);
80 volumes_som(som)+=volume;
81 }
82 }
83 volumes_som.echange_espace_virtuel();
84}

123void Zone_VEF_PreP1b::discretiser_arete()
124{
125 const Domaine& dom = zone().domaine();
126
127// Creation des aretes reelles (informations geometriques construites et stockees dans la zone)
128 zone().creer_aretes();
129 md_vector_aretes_ = zone().aretes_som().get_md_vector();
130
131// Calcul des centres de gravite des aretes xa_ stockes dans la Zone_VF
132 const IntTab& aretes_som = zone().aretes_som();
133 const int nb_aretes = aretes_som.dimension(0);
134 const DoubleTab& coord = dom.les_sommets();
135 const int dim = coord.dimension(1);
136 xa_.resize(0, dim);
137 creer_tableau_aretes(xa_, ArrOfDouble::NOCOPY_NOINIT);
138 for (int i = 0; i < nb_aretes; i++)
139 {
140 const int s0 = aretes_som(i, 0);
141 const int s1 = aretes_som(i, 1);
142 for (int j = 0; j < dim; j++)
143 xa_(i, j) = (coord(s0, j) + coord(s1, j)) * 0.5;
144 }
145 xa_.echange_espace_virtuel();
146
147 const IntTab& elem_aretes = zone().elem_aretes();
148
149// Calcul du volume des aretes
150// Creation d'un tableau initialise a zero:
151 creer_tableau_aretes(volumes_aretes);
152
153 const int nbr_elem = zone().nb_elem();
154 const int nb_aretes_elem = elem_aretes.dimension(1);
155// facteur 6 pour le calcul des volumes des aretes, est-ce correct pour autre chose qu'un tetra ?
156 assert(nb_aretes_elem == 6);
157// Essai d'une autre facon de coder: calcul des contributions aux aretes par les elements reels,
158// puis sommation des contributions des aretes partagees:
159 for (int elem = 0; elem < nbr_elem; elem++)
160 {
161 double vol = volumes(elem) / 6.0;
162 for(int j = 0; j < nb_aretes_elem; j++)
163 {
164 int arete = elem_aretes(elem, j);
165 volumes_aretes[arete] += vol;
166 }
167 }
168// Sommation des contributions des aretes joint et echange des espaces virtuels
169 MD_Vector_tools::echange_espace_virtuel(volumes_aretes, MD_Vector_tools::EV_SOMME_ECHANGE);
170}

292 int nb_aretes=zone.nb_aretes();
293 for (int I=0 ; I<nb_elem; I++)

320void Zone_VEF_PreP1b::construire_ok_arete()
321{
322 Cerr << "Build array ok_arete..." << finl;
323 const Domaine& dom=zone().domaine();
324 const IntTab& aretes_som=zone().aretes_som();
325 const int nb_som_reel=nb_som();
326
327// Connectivite sommets-aretes (pour chaque sommet, liste des aretes adjacentes)
328// B.M.: je remplace IntTab(n, 64) par une Static_Int_List et je reutilise
329// la methode de calcul des connectivites... plus econome en memoire !
330// som_aretes contient une connectivite modifiee pour les sommets periodiques
331// (les aretes sont toujours rattachees au sommet get_renum_som_perio(),
332// les aretes periodiques opposees sont remplacees par une arete [nb_som_reel,nb_som_reel]
333// fictive, et les sommets perio opposes ne sont rattaches a aucune arete)
334 const int nb_aretes = aretes_som.dimension(0);
335 Static_Int_Lists som_aretes;
336 {
337// Creation d'un tableau d'aretes ou les sommets sont remplaces par le sommet
338// periodique associe et sans les aretes periodiques opposees
339 IntTab aretes_som2;
340 aretes_som2.copy(aretes_som, Array_base::NOCOPY_NOINIT);
341
342 for (int i = 0; i < nb_aretes; i++)
343 {
344 if (renum_arete_perio[i] == i)
345 {
346 aretes_som2(i, 0) = dom.get_renum_som_perio(aretes_som(i,0));
347 aretes_som2(i, 1) = dom.get_renum_som_perio(aretes_som(i,1));
348 }
349 else
350 {
351// arete periodique opposee, non conservee, on cree une arete fictive pour
352// construire_connectivite_som_elem: le sommet d'indice nb_som_reel
353// sera connecte a toutes les aretes periodiques supprimees
354 aretes_som2(i, 0) = nb_som_reel;
355 aretes_som2(i, 1) = nb_som_reel;
356 }
357 }
358// On donne un sommet de plus (le sommet fictif connecte aux aretes supprimees)
359 construire_connectivite_som_elem(nb_som_reel+1, aretes_som2, som_aretes, 0 /* do not include virtual items */);
360 }
361
362// Creation et initialisation du tableau contenu, initialise a zero par defaut
363 ArrOfInt contenu(nb_som_reel);
364
365// Initialisation pour le parallele: contenu est mis a 2
366// pour les sommets communs recus d'un autre processeur
367 {
368 ArrOfBit flags;
369 MD_Vector_tools::get_sequential_items_flags(dom.les_sommets().get_md_vector(), flags);
370 for (int i = 0; i < nb_som_reel; i++)
371 {
372 if (!flags[i])
373 {
374// Ce sommet est recu d'un autre processeur
375 const int i2 = dom.get_renum_som_perio(i);
376 contenu[i] = 2;
377 if (i2 != i)
378 contenu[i2] = 2;
379 }
380 }
381 }
382
383// Estimation du nombre d'aretes superflues a trouver sur la zone
384// en comptant les sommets reels non periodiques dont le contenu vaut 0
385 int nombre_aretes_superflues_prevues_sur_la_zone=0;
386 for (int i=0; i<nb_som_reel; i++)
387 if (i==dom.get_renum_som_perio(i) && contenu[i]==0)
388 nombre_aretes_superflues_prevues_sur_la_zone++;
389
390 ok_arete = -1;
391
392// On boucle tant qu'il y'a des sommets avec le contenu 0
393 while (min_array(contenu)==0)
394 {
395// On cherche le premier sommet avec contenu 0 en bouclant sur les aretes
396 int Aroot=-1,Sroot=-1;
397 int S0,S1;
398 do
399 {
400 while (ok_arete(++Aroot)==0) {};
401 S0=dom.get_renum_som_perio(aretes_som(Aroot,0));
402 S1=dom.get_renum_som_perio(aretes_som(Aroot,1));
403 }
404 while (Aroot<nb_aretes && contenu[S0]!=0 && contenu[S1]!=0);
405
406 assert(Aroot<nb_aretes);
407
408 Aroot=renum_arete_perio[Aroot];
409
410 ok_arete(Aroot)=0; //!< lockee
411 if (!contenu[S0])
412 {
413 contenu[S0]=1;
414 Sroot=S0;
415 }
416 else if (!contenu[S1])
417 {
418 contenu[S1]=1;
419 Sroot=S1;
420 }
421
422// Boucle sur les sommets
423 do
424 {
425 const int nb_aretes_voisines = som_aretes.get_list_size(Sroot);
426 for(int i=0; i<nb_aretes_voisines; i++)
427 {
428 int A=renum_arete_perio[som_aretes(Sroot,i)];
429 if(ok_arete(A)==-1)
430 {
431 int S=dom.get_renum_som_perio(aretes_som(A,0));
432 if(S==Sroot)
433 {
434 S=dom.get_renum_som_perio(aretes_som(A,1));
435 }
436 if(!contenu[S])
437 {
438 ok_arete(A)=0; //!< lockee
439 contenu[S]=1;
440 }
441 else
442 {
443 ok_arete(A)=1; //!< libre
444 }
445 }
446 }
447 contenu[Sroot]=2;
448 }
449 while((Sroot=next(Sroot, contenu))!=-1);
450
451// Correction pour des tableaux ok_arete et contenu pour la periodicite
452 for(int i=0; i<nb_aretes; i++)
453 ok_arete(i)=ok_arete[renum_arete_perio[i]];
454 for(int i=0; i<nb_som_reel; i++)
455 contenu[i]=contenu[dom.get_renum_som_perio(i)];
456 }
457
458// Mise a jour des parties virtuelles du tableau ok_arete
459 ok_arete.echange_espace_virtuel();
460
461// Verification des aretes superflues
462 verifie_ok_arete(nombre_aretes_superflues_prevues_sur_la_zone);
463
464// Ecriture des aretes superflues dans un fichier nom_du_cas.ok_arete afin de le relire la fois suivante
465 Nom fichier(nom_du_cas());
466 fichier+="_";
467 fichier+=zone().domaine().le_nom()+".ok_arete";
468
469 Cerr << "Writing file " << fichier << finl;
470 EcrFicPartageBin fic_ok_arete_;
471 if (!fic_ok_arete_.ouvrir(fichier,ios::out))
472 {
473 Cerr << "Error: cannot open file " << fichier << " for writing." << finl;
474 exit();
475 }
476
477 ArrOfBit marqueurs_aretes;
478 const MD_Vector& md_aretes = md_vector_aretes();
479 MD_Vector_tools::get_sequential_items_flags(md_aretes, marqueurs_aretes);
480 const int nb_aretes_seq = md_aretes.valeur().nb_items_seq_tot();
481
482 if (Process::je_suis_maitre())
483 fic_ok_arete_ << nb_aretes_seq << finl;
484
485// Chaque processeur ecrit ses aretes non communes:
486 const int n = marqueurs_aretes.size_array();
487 for (int i = 0; i < n; i++)
488 {
489 if (marqueurs_aretes[i])
490 fic_ok_arete_ << xa_(i,0) << space << xa_(i,1) << space << xa_(i,2) << space << ok_arete(i) << finl;
491 }
492 fic_ok_arete_.syncfile();
493 fic_ok_arete_.close();
494}
