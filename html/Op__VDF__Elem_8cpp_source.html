26void Op_VDF_Elem::dimensionner(const Zone_VDF& la_zone, const Zone_Cl_VDF& la_zone_cl, Matrice_Morse& la_matrice) const
27{
28// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de la convection, de la diffusion pour le cas des elements.
29// Cette matrice a une structure de matrice morse.
30// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
31
32 const IntTab& face_voisins = la_zone.face_voisins();
33 const Conds_lim& les_cl = la_zone_cl.les_conditions_limites();
34 const DoubleTab& champ_inconnue = la_zone_cl.equation().inconnue().valeurs();
35 const int n1 = la_zone.zone().nb_elem_tot(), nb_comp = champ_inconnue.line_size();
36
37 la_matrice.dimensionner(n1*nb_comp, n1*nb_comp, 0);
38
39 IntVect& tab1 = la_matrice.get_set_tab1(), &tab2 = la_matrice.get_set_tab2();
40 DoubleVect& coeff = la_matrice.get_set_coeff();
41
42 const int ndeb = la_zone.premiere_face_int(), nfin = la_zone.nb_faces();
43 coeff = 0;
44
45 IntVect rang_voisin(n1*nb_comp);
46 rang_voisin = 1;
47
48 for (int num_face = ndeb; num_face < nfin; num_face++)
49 {
50 const int elem1 = face_voisins(num_face,0), elem2 = face_voisins(num_face,1);
51 (rang_voisin(elem2))++;
52 (rang_voisin(elem1))++;
53 }
54
55// Prise en compte des conditions de type periodicite
56 for (int i = 0; i < les_cl.size(); i++)
57 {
58 const Cond_lim& la_cl = les_cl[i];
59 if (sub_type(Periodique,la_cl.valeur()))
60 {
61 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
62 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
63 const int numdeb = la_front_dis.num_premiere_face(), nfaces = la_front_dis.nb_faces();
64 int ind_face_global;
65 IntVect fait(nfaces);
66 fait = 0;
67 for (int face = 0; face < nfaces; face++)
68 {
69 if (fait[face] == 0)
70 {
71 fait[face] = 1;
72 fait[la_cl_perio.face_associee(face)] = 1;
73 ind_face_global = face+numdeb;
74
75 const int elem1 = face_voisins(ind_face_global,0), elem2 = face_voisins(ind_face_global,1);
76
77 (rang_voisin(elem2))++;
78 (rang_voisin(elem1))++;
79 }
80 }
81 }
82 }
83
84// on balaye les elements pour dimensionner tab1 et tab2
85 tab1(0) = 1;
86 for(int i = 0; i < n1; i++)
87 for (int k = 0; k < nb_comp; k++) tab1(i*nb_comp+1+k) = rang_voisin(i) + tab1(i*nb_comp+k);
88
89 la_matrice.dimensionner(n1*nb_comp,tab1(n1*nb_comp)-1);
90
91 for (int i = 0; i < n1*nb_comp; i++)
92 {
93 tab2[tab1[i]-1] = i+1;
94 rang_voisin[i] = tab1[i];
95 }
96
97// on traite les faces internes pour les voisins
98 for (int num_face = ndeb; num_face < nfin; num_face++)
99 {
100 const int elem1 = face_voisins(num_face,0), elem2 = face_voisins(num_face,1);
101 for (int k = 0; k < nb_comp; k++)
102 {
103 tab2[rang_voisin[elem2*nb_comp+k]] = elem1*nb_comp+1+k;
104 rang_voisin[elem2*nb_comp+k]++;
105
106 tab2[rang_voisin[elem1*nb_comp+k]] = elem2*nb_comp+1+k;
107 rang_voisin[elem1*nb_comp+k]++;
108 }
109 }
110
111// on traite la condition de periodicite
112 for (int i=0; i<les_cl.size(); i++)
113 {
114 const Cond_lim& la_cl = les_cl[i];
115 if (sub_type(Periodique,la_cl.valeur()) )
116 {
117 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
118 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
119 const int numdeb = la_front_dis.num_premiere_face(), nfaces = la_front_dis.nb_faces();
120 IntVect fait(nfaces);
121 fait = 0;
122 for (int ind_face_local = 0; ind_face_local < nfaces; ind_face_local++)
123 {
124 if (fait[ind_face_local] == 0)
125 {
126 const int num_face = numdeb + ind_face_local;
127 fait[ind_face_local] = 1;
128 fait[la_cl_perio.face_associee(ind_face_local)] = 1;
129
130 const int elem1 = face_voisins(num_face,0), elem2 = face_voisins(num_face,1);
131
132 for (int k = 0; k < nb_comp; k++)
133 {
134 tab2[rang_voisin[elem2*nb_comp+k]] = elem1*nb_comp+1+k;
135 rang_voisin[elem2*nb_comp+k]++;
136
137 tab2[rang_voisin[elem1*nb_comp+k]] = elem2*nb_comp+1+k;
138 rang_voisin[elem1*nb_comp+k]++;
139 }
140 }
141 }
142 }
143 }
144}

146void Op_VDF_Elem::dimensionner_bloc_vitesse(const Zone_VDF& la_zone, const Zone_Cl_VDF& la_zone_cl, Matrice_Morse& matrice) const
147{
148 const int nb_faces = la_zone.nb_faces(), nb_faces_tot = la_zone.nb_faces_tot(), nb_elem_tot = la_zone.nb_elem_tot();
149 const IntTab& face_voisins = la_zone.face_voisins();
150 IntTab stencyl(0,2);
151 stencyl.set_smart_resize(1);
152
153 int nb_coef = 0;
154 for (int face = 0; face < nb_faces; face++)
155 for (int dir = 0; dir < 2; dir++)
156 {
157 const int elem = face_voisins(face,dir);
158 if (elem != -1)
159 {
160 stencyl.resize(nb_coef+1,2);
161 stencyl(nb_coef,0) = elem;
162 stencyl(nb_coef,1) = face;
163 nb_coef++;
164 }
165 }
166
167 tableau_trier_retirer_doublons(stencyl);
168 Matrix_tools::allocate_morse_matrix(nb_elem_tot,nb_faces_tot,stencyl,matrice);
169}
