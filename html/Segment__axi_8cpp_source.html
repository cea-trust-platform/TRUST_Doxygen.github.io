33void Segment_axi::calculer_volumes(DoubleVect& volumes) const
34{
35 const Zone& zone=ma_zone.valeur();
36 const DomaineAxi1d& dom = ref_cast(DomaineAxi1d,zone.domaine());
37
38 int S1,S2;
39
40 int size_tot = zone.nb_elem_tot();
41 assert(volumes.size_totale()==size_tot);
42
43 assert(dimension==3);
44
45 for (int num_poly=0; num_poly<size_tot; num_poly++)
46 {
47 S1 = zone.sommet_elem(num_poly,0);
48 S2 = zone.sommet_elem(num_poly,1);
49
50 double x0 = dom.origine_repere(num_poly,0);
51 double y0 = dom.origine_repere(num_poly,1);
52
53 double x1 = dom.coord(S1,0);
54 double x2 = dom.coord(S2,0);
55
56 double y1 = dom.coord(S1,1);
57 double y2 = dom.coord(S2,1);
58
59 double r1 = sqrt((x1-x0)*(x1-x0)+(y1-y0)*(y1-y0));
60 double r2 = sqrt((x2-x0)*(x2-x0)+(y2-y0)*(y2-y0));
61
62
63 volumes[num_poly]= M_PI*std::fabs(r2*r2-r1*r1);
64 }
65}
