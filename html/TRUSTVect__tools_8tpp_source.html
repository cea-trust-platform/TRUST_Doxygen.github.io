45template <typename _TYPE_>
46inline void invalidate_data(TRUSTVect<_TYPE_>& resu, Mp_vect_options opt)
47{
48 _TYPE_ invalid = (std::is_same<_TYPE_,double>::value) ? -987654.321 : INT_MAX;
49
50 const MD_Vector& md = resu.get_md_vector();
51 const int line_size = resu.line_size();
52 if (opt == VECT_ALL_ITEMS || (!md.non_nul())) return; //!< no invalid values
53 assert(opt == VECT_SEQUENTIAL_ITEMS || opt == VECT_REAL_ITEMS);
54 const TRUSTArray<int>& items_blocs = (opt == VECT_SEQUENTIAL_ITEMS) ? md.valeur().get_items_to_sum() : md.valeur().get_items_to_compute();
55 const int blocs_size = items_blocs.size_array();
56 int i = 0;
57 for (int blocs_idx = 0; blocs_idx < blocs_size; blocs_idx += 2) //!< process data until beginning of next bloc, or end of array
58 {
59 const int bloc_end = line_size * items_blocs[blocs_idx];
60 _TYPE_ *ptr = resu.addr() + i;
61 for (; i < bloc_end; i++) *(ptr++) = invalid;
62 i = items_blocs[blocs_idx+1] * line_size;
63 }
64 const int bloc_end = resu.size_array(); //!< Process until end of vector
65 _TYPE_ *ptr = resu.addr() + i;
66 for (; i < bloc_end; i++) *(ptr++) = invalid;
67}
68#endif /* NDEBUG */

86template <typename _TYPE_, TYPE_OPERATION_VECT _TYPE_OP_ >
87inline _TYPE_ neutral_value()
88{
89 static constexpr bool IS_MAX = ((_TYPE_OP_ == TYPE_OPERATION_VECT::IMAX_) || (_TYPE_OP_ == TYPE_OPERATION_VECT::MAX_)), IS_MAX_ABS = (_TYPE_OP_ == TYPE_OPERATION_VECT::MAX_ABS_);

92 _TYPE_ neutral_val;

94 if (std::is_same<_TYPE_, double>::value) neutral_val = neutral_value_double_(IS_MAX);
95 else neutral_val = neutral_value_int_(IS_MAX);

100template <typename _TYPE_,typename _TYPE_RETURN_, TYPE_OPERATION_VECT _TYPE_OP_ >
101inline _TYPE_RETURN_ local_extrema_vect_generic(const TRUSTVect<_TYPE_>& vx, Mp_vect_options opt)
102{
103 static constexpr bool IS_IMAX = (_TYPE_OP_ == TYPE_OPERATION_VECT::IMAX_), IS_IMIN = (_TYPE_OP_ == TYPE_OPERATION_VECT::IMIN_), IS_MAX = (_TYPE_OP_ == TYPE_OPERATION_VECT::MAX_),
104 IS_MIN = (_TYPE_OP_ == TYPE_OPERATION_VECT::MIN_), IS_MAX_ABS = (_TYPE_OP_ == TYPE_OPERATION_VECT::MAX_ABS_), IS_MIN_ABS = (_TYPE_OP_ == TYPE_OPERATION_VECT::MIN_ABS_);
105
106 _TYPE_ min_max_val = neutral_value<_TYPE_,_TYPE_OP_>(); //!< _TYPE_ et pas _TYPE_RETURN_ desole ...
107 _TYPE_RETURN_ i_min_max = -1 ; //!< seulement pour IMAX_ et IMIN_
108
109// Master vect donne la structure de reference, les autres vecteurs doivent avoir la meme structure.
110 const TRUSTVect<_TYPE_>& master_vect = vx;
111 const int line_size = master_vect.line_size(), vect_size_tot = master_vect.size_totale();
112 const MD_Vector& md = master_vect.get_md_vector();
113 assert(vx.line_size() == line_size);
114 assert(vx.size_totale() == vect_size_tot); //!< this test is necessary if md is null
115 assert(vx.get_md_vector() == md);
116// Determine blocs of data to process, depending on " opt"
117 int nblocs_left = 1, one_bloc[2];
118 const int *bloc_ptr;
119 if (opt != VECT_ALL_ITEMS && md.non_nul())
120 {
121 assert(opt == VECT_SEQUENTIAL_ITEMS || opt == VECT_REAL_ITEMS);
122 const TRUSTArray<int>& items_blocs = (opt == VECT_SEQUENTIAL_ITEMS) ? md.valeur().get_items_to_sum() : md.valeur().get_items_to_compute();
123 assert(items_blocs.size_array() % 2 == 0);
124 nblocs_left = items_blocs.size_array() >> 1;
125 bloc_ptr = items_blocs.addr();
126 }
127 else if (vect_size_tot > 0)
128 {
129// attention, si vect_size_tot est nul, line_size a le droit d'etre nul
130// Compute all data, in the vector (including virtual data), build a big bloc:
131 nblocs_left = 1;
132 bloc_ptr = one_bloc;
133 one_bloc[0] = 0;
134 one_bloc[1] = vect_size_tot / line_size;
135 }
136 else //!< raccourci pour les tableaux vides (evite le cas particulier line_size == 0)
137 return (IS_IMAX || IS_IMIN) ? i_min_max : min_max_val;
138
139 const _TYPE_ *x_base = vx.addr();
140 for (; nblocs_left; nblocs_left--)
141 {
142// Get index of next bloc start:
143 const int begin_bloc = (*(bloc_ptr++)) * line_size, end_bloc = (*(bloc_ptr++)) * line_size;
144 assert(begin_bloc >= 0 && end_bloc <= vect_size_tot && end_bloc >= begin_bloc);
145 const _TYPE_ *x_ptr = x_base + begin_bloc;
146 int count = end_bloc - begin_bloc;
147 for (; count; count--)
148 {
149 const _TYPE_ x = *x_ptr;
150 if ((IS_IMAX && x > min_max_val) || (IS_IMIN && x < min_max_val))
151 {
152 i_min_max = x_ptr - x_base;
153 min_max_val = x;
154 }
155
156 if ((IS_MAX && x > min_max_val) || (IS_MIN && x < min_max_val)) min_max_val = x;
157
158 if (IS_MAX_ABS || IS_MIN_ABS)
159 {
160 _TYPE_ xx;
161 xx = std::is_same<_TYPE_, double>::value ? std::fabs(x) : std::abs(x);
162 if ((IS_MAX_ABS && xx > min_max_val) || (IS_MIN_ABS && xx < min_max_val)) min_max_val = xx;
163 }
164
165 x_ptr++;
166 }
167 }
168 return (IS_IMAX || IS_IMIN) ? i_min_max : min_max_val;
169}

609 one_bloc[1] = vect_size_tot / line_size;
610 }
611 else //!< raccourci pour les tableaux vides (evite le cas particulier line_size == 0)
612 return;
