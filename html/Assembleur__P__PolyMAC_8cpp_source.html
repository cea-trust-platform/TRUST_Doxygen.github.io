63int Assembleur_P_PolyMAC::assembler_mat(Matrice& la_matrice,const DoubleVect& diag,int incr_pression,int resoudre_en_u)
64{
65 set_resoudre_increment_pression(incr_pression);
66 set_resoudre_en_u(resoudre_en_u);
67 Cerr << "Assemblage de la matrice de pression ... " ;
68 statistiques().begin_count(assemblage_sys_counter_);
69 la_matrice.typer("Matrice_Morse");
70 Matrice_Morse& mat = ref_cast(Matrice_Morse, la_matrice.valeur());
71
72 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, la_zone_PolyMAC.valeur());
73 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, mon_equation->inconnue().valeur());
74 const IntTab& e_f = zone.elem_faces(), &fcl = ch.fcl();
75 const DoubleVect& pe = zone.porosite_elem();
76 int i, j, e, f, fb, ne = zone.nb_elem(), ne_tot = zone.nb_elem_tot(), nf = zone.nb_faces(), nf_tot = zone.nb_faces_tot();
77
78 DoubleTrav w2; //matrice W2 (de Zone_PolyMAC) par element
79 w2.set_smart_resize(1);
80
81 /* 1. stencil de la matrice en pression : seulement au premier passage */
82 if (!stencil_done) /* premier passage: calcul */
83 {
84 IntTrav stencil(0, 2);
85 stencil.set_smart_resize(1);
86 for (e = 0; e < ne; e++)
87 for (stencil.append_line(e, e), i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++) /* blocs "elem-elem" et "elem-face" */
88 stencil.append_line(e, ne_tot + f); //toutes les faces (sauf bord de Neumann)
89 for (e = 0; e < ne_tot; e++)
90 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(1); i++) /* blocs "face-elem" et "face-face" */
91 if (fcl(f = e_f(e, i), 0) == 1 && f < nf) stencil.append_line(ne_tot + f, ne_tot + f); //Neumann : ligne "dpf = 0"
92 else if (f < nf)
93 for (stencil.append_line(ne_tot + f, e), j = 0; j < w2.dimension(1); j++) /* sinon : ligne sum w2_{ff'} (pf' - pe) */
94 if (w2(i, j, 0)) stencil.append_line(ne_tot + f, ne_tot + e_f(e, j));
95
96 tableau_trier_retirer_doublons(stencil);
97 Matrix_tools::allocate_morse_matrix(ne_tot + nf_tot, ne_tot + nf_tot, stencil, mat);
98 tab1.ref_array(mat.get_set_tab1()), tab2.ref_array(mat.get_set_tab2());
99 stencil_done = 1;
100 }
101 else /* passages suivants : recyclage */
102 {
103 mat.get_set_tab1().ref_array(tab1);
104 mat.get_set_tab2().ref_array(tab2);
105 mat.get_set_coeff().resize(tab2.size()), mat.get_set_coeff() = 0;
106 mat.set_nb_columns(ne_tot + nf_tot);
107 }
108
109 /* 2. coefficients */
110 for (e = 0; e < ne_tot; e++)
111 {
112 zone.W2(NULL, e, w2); //calcul de W2
113 double m_ee = 0, m_fe, m_ef; //coefficients (elem, elem), (elem, face) et (face, elem)
114 for (i = 0; i < w2.dimension(0); i++, m_ee += m_ef)
115 {
116 for (m_ef = 0, m_fe = 0, f = e_f(e, i), j = 0; j < w2.dimension(1); j++)
117 if (w2(i, j, 0))
118 {
119 fb = e_f(e, j);
120 if (f < zone.nb_faces() && fcl(f, 0) != 1) mat(ne_tot + f, ne_tot + fb) += pe(e) * w2(i, j, 0); //interne ou Dirichlet
121 else if (f < zone.nb_faces() && i == j) mat(ne_tot + f, ne_tot + fb) = 1; //f Neumann : ligne dpf = 0
122 m_ef += pe(e) * w2(i, j, 0), m_fe += pe(e) * w2(i, j, 0); //accumulation dans m_ef, m_fe
123 }
124 if (e < zone.nb_elem()) mat(e, ne_tot + f) -= m_ef;
125 if (f < zone.nb_faces() && fcl(f, 0) != 1) mat(ne_tot + f, e) -= m_fe; //si f non Neumann : coef (face, elem)
126 }
127 if (e < zone.nb_elem()) mat(e, e) += m_ee; //coeff (elem, elem)
128 }
129
130// en l'absence de CLs en pression, on ajoute P(0) = 0 sur le process 0
131 has_P_ref=0;
132 for (int n_bord=0; n_bord<la_zone_PolyMAC->nb_front_Cl(); n_bord++)
133 if (sub_type(Neumann_sortie_libre, la_zone_Cl_PolyMAC->les_conditions_limites(n_bord).valeur()) )
134 has_P_ref=1;
135 if (!has_P_ref && !Process::me()) mat(0, 0) *= 2;
136
137 statistiques().end_count(assemblage_sys_counter_);
138 Cerr << statistiques().last_time(assemblage_sys_counter_) << " s" << finl;
139 return 1;
140}

179void Assembleur_P_PolyMAC::dimensionner_continuite(matrices_t matrices, int aux_only) const
180{
181 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
182 int i, j, e, f, fb, n, N = equation().inconnue().valeurs().line_size(), m, M = equation().get_champ("pression").valeurs().line_size(),
183 ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot();
184 const IntTab& fcl = ref_cast(Champ_Face_PolyMAC, mon_equation->inconnue().valeur()).fcl(), &e_f = zone.elem_faces();
185 IntTrav sten_a(0, 2), sten_p(0, 2), sten_v(0, 2);
186 DoubleTrav w2;
187 sten_a.set_smart_resize(1), sten_p.set_smart_resize(1), sten_v.set_smart_resize(1), w2.set_smart_resize(1);
188 /* equations sum alpha_k = 1 */
189 if (!aux_only)
190 for (e = 0; e < zone.nb_elem(); e++)
191 for (n = 0; n < N; n++) sten_a.append_line(e, N * e + n);
192 /* equations sur les p_f : continuite du gradient si interne, p = p_f si Neumann, sum_k alpha_k v_k = sum_k alpha_k v_k,imp si Dirichlet */
193 for (e = 0; e < zone.nb_elem_tot(); e++)
194 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(0); i++)
195 if ((f = e_f(e, i)) >= zone.nb_faces()) continue; //faces virtuelles
196 else if (!fcl(f, 0))
197 for (sten_p.append_line(!aux_only * ne_tot + f, e), j = 0; j < w2.dimension(1); j++) //face interne
198 {
199 if (w2(i, j, 0) && fcl(fb = e_f(e, j), 0) != 1)
200 for (m = 0; m < M; m++)
201 sten_p.append_line(M * (!aux_only * ne_tot + f) + m, M * (ne_tot + fb) + m);
202 }
203 else if (fcl(f, 0) == 1)
204 for (m = 0; m < M; m++) sten_p.append_line(M * (!aux_only * ne_tot + f) + m, M * (ne_tot + f) + m); //Neumann
205 else for (n = 0, m = 0; n < N; n++, m += (M > 1)) sten_v.append_line(M * (!aux_only * ne_tot + f) + m, N * f + n); //Dirichlet
206
207 tableau_trier_retirer_doublons(sten_v), tableau_trier_retirer_doublons(sten_p);
208 if (!aux_only) Matrix_tools::allocate_morse_matrix(ne_tot + nf_tot, N * ne_tot, sten_a, *matrices.at("alpha"));
209 Matrix_tools::allocate_morse_matrix(M * (!aux_only * ne_tot + nf_tot), M * (ne_tot + nf_tot), sten_p, *matrices.at("pression"));
210 Matrix_tools::allocate_morse_matrix(M * (!aux_only * ne_tot + nf_tot), equation().inconnue()->valeurs().size_totale(), sten_v, *matrices.at("vitesse"));
211}

213void Assembleur_P_PolyMAC::assembler_continuite(matrices_t matrices, DoubleTab& secmem, int aux_only) const
214{
215 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
216 const Pb_Multiphase* pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
217 const Conds_lim& cls = la_zone_Cl_PolyMAC->les_conditions_limites();
218 const DoubleTab *alpha = pbm ? &pbm->eq_masse.inconnue().valeurs() : NULL, &press = equation().probleme().get_champ("pression").valeurs(),
219 &vit = equation().inconnue().valeurs(), *alpha_rho = pbm ? &pbm->eq_masse.champ_conserve().passe() : NULL, &nf = zone.face_normales();
220 const IntTab& fcl = ref_cast(Champ_Face_PolyMAC, mon_equation->inconnue().valeur()).fcl(), &e_f = zone.elem_faces();
221 const DoubleVect& ve = zone.volumes(), &pe = zone.porosite_elem(), &fs = zone.face_surfaces(), &vf = zone.volumes_entrelaces();
222 int i, j, e, f, fb, n, N = vit.line_size(), m, M = press.line_size(), ne_tot = zone.nb_elem_tot(), d, D = dimension;
223 Matrice_Morse *mat_a = alpha ? matrices.at("alpha") : NULL, &mat_p = *matrices.at("pression"), &mat_v = *matrices.at("vitesse");
224 DoubleTrav w2, fac(N);
225 double ar_tot, acc;
226 w2.set_smart_resize(1);
227 secmem = 0, fac = 1;
228
229 /* equations sum alpha_k = 1 */
230 /* second membre : on multiplie par porosite * volume pour que le systeme en P soit symetrique en cartesien */
231 if (!aux_only)
232 for (e = 0; e < zone.nb_elem(); e++)
233 for (secmem(e) = -pe(e) * ve(e), n = 0; n < N; n++) secmem(e) += pe(e) * ve(e) * (*alpha)(e, n);
234 /* matrice */
235 if (!aux_only)
236 for (e = 0; e < zone.nb_elem(); e++)
237 for (n = 0; n < N; n++) (*mat_a)(e, N * e + n) = -pe(e) * ve(e);
238
239 /* equations sur les p_f : continuite du gradient si interne, p = p_f si Neumann, sum_k alpha_k v_k = sum_k alpha_k v_k,imp si Dirichlet */
240 for (mat_p.get_set_coeff() = 0, mat_v.get_set_coeff() = 0, e = 0; e < ne_tot; e++)
241 for (zone.W2(NULL, e, w2), i = 0; i < w2.dimension(0); i++)
242 if ((f = e_f(e, i)) >= zone.nb_faces()) continue; //faces virtuelles
243 else if (!fcl(f, 0)) //face interne
244 {
245 for (acc = 0, j = 0; j < w2.dimension(1); acc+= pe(e) * vf(f) * w2(i, j, 0), j++)
246 for (m = 0; m < M; m++) //second membre
247 secmem(!aux_only * ne_tot + f, m) -= pe(e) * vf(f) * w2(i, j, 0) * (press(ne_tot + e_f(e, j), m) - press(e, m));
248 for (m = 0; m < M; m++) mat_p(M * (!aux_only * ne_tot + f) + m, M * e + m) -= acc;
249 for (j = 0; j < w2.dimension(1); j++) //matrice (sauf bords de Meumann)
250 if (w2(i, j, 0) && fcl(fb = e_f(e, j), 0) != 1)
251 for (m = 0; m <M; m++)
252 mat_p(M * (!aux_only * ne_tot + f) + m, M * (ne_tot + fb) + m) += pe(e) * vf(f) * w2(i, j, 0);
253 }
254 else if (fcl(f, 0) == 1) //Neumann -> egalites p_f = p_imp
255 {
256 for (m = 0; m < M; m++) secmem(M * (!aux_only * ne_tot + f) + m) = fs(f) * (ref_cast(Neumann, cls[fcl(f, 1)].valeur()).flux_impose(fcl(f, 2), m) - press(ne_tot + f, m));
257 for (m = 0; m < M; m++) mat_p(M * (!aux_only * ne_tot + f) + m, M * (ne_tot + f) + m) = fs(f);
258 }
259 else //Dirichlet -> egalite flux_tot_imp - flux_tot = 0
260 {
261 if (M == 1 && N > 1) //une pression, plusieurs vitesses -> on ne peut imposer qu'une ponderation : on choisit celle ocrrespondant aux flux de masse total
262 {
263 for (ar_tot = 0, n = 0; n < N; n++) ar_tot += (*alpha_rho)(e, n);
264 for (n = 0; n < N; n++) fac(n) = (*alpha_rho)(e, n) / ar_tot;
265 }
266 else if (M != N) abort(); //sinon, il faut autant de pressions que de vitesses
267
268 for (n = 0, m = 0; n < N; n++, m += (M > 1)) secmem(!aux_only * ne_tot + f, m) += vf(f) * fac(n) * vit(f, n);
269 if (fcl(f, 0) == 3)
270 for (d = 0; d < D; d++)
271 for (n = 0, m = 0; n < N; n++, m += (M > 1)) //contrib de la valeur imposee: Dirichlet non homogene seulement
272 secmem(!aux_only * ne_tot + f, m) -= vf(f) * fac(n) * nf(f, d) / fs(f) * ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), N * d + n);
273 for (n = 0, m = 0; n < N; n++, m += (M > 1)) mat_v(M * (!aux_only * ne_tot + f) + m, N * f + n) -= vf(f) * fac(n);
274 }
275}
