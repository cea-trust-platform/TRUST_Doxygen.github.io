85 EVAL_TYPE& associer_(const Zone_dis& zone_dis, const Zone_Cl_dis& zone_cl_dis)

88 const Zone_Cl_VDF& zclvdf = ref_cast(Zone_Cl_VDF,zone_cl_dis.valeur());
89 iter_()->associer(zvdf,zclvdf,static_cast<OP_TYPE&>(*this)); //!< Et ouiiiiiiiii
90 EVAL_TYPE& eval_conv = static_cast<EVAL_TYPE&> (iter_()->evaluateur()); //!< Mais ouiiiiiiiiiiii
91 eval_conv.associer_zones(zvdf, zclvdf );
92
93 if (is_QUICK || is_CENTRE4)
94 if ( Process::nproc()>1 && zvdf.zone().nb_joints() && zvdf.zone().joint(0).epaisseur()<2)
95 {
96 Cerr << "Overlapping width (given by larg_joint option) of " << zvdf.zone().joint(0).epaisseur() << finl;
97 Cerr << "is not enough for Quick scheme in VDF parallel calculation." << finl;
98 Cerr << "Please, partition your mesh with an overlapping width of 2 with larg_joint option." << finl;
99 Process::exit();
100 }
101 return eval_conv;
102 }
