32void Tetraedriser_homogene_compact::decoupe(Zone& zone, Faces& faces, IntTab& new_soms_old_elems,IntTab& fait_sommet, int nface) const
33{
34 IntTab& sommets=faces.les_sommets();
35 int nb_faces=sommets.dimension(0);
36 assert(sommets.dimension(1)==4);
37 IntTab nouveaux(4*nb_faces, 3);
38 faces.voisins().resize(4*nb_faces, 2);
39 faces.voisins()=-1;
40 IntVect indice(5);
41
42 for(int i=0; i<nb_faces; i++)
43 {
44 indice(0) = sommets(i,0);
45 indice(1) = sommets(i,1);
46 indice(2) = sommets(i,2);
47 indice(3) = sommets(i,3);
48
49 int i0=indice(0);
50 int i1=indice(1);
51 int i2=indice(2);
52 int i3=indice(3);
53 int i4=-1;
54
55// on remet un peu d'ordre dans les indices
56
57 int permut,ind;
58
59 do
60 {
61 permut=0;
62 for(int j=1; j<4; j++)
63 {
64 if( indice[j]<indice[j-1] )
65 {
66 ind=indice[j-1];
67 indice[j-1]=indice[j];
68 indice[j]=ind;
69 permut++;
70 }
71 }
72 }
73 while(permut);
74
75
76 for (int ii=0 ; ii<nface ; ii++)
77 {
78 if ( fait_sommet(ii,0)==indice[0] && fait_sommet(ii,1)==indice[1] && fait_sommet(ii,2)==indice[2] )
79 {
80 i4=fait_sommet(ii,3);
81 break;
82 }
83 }
84
85 if(indice(4)<0)
86 {
87// Cerr << indice(0) << " " << indice(1) << " " << indice(2) << finl;
88 Cerr << " FALSE " << finl;
89 exit();
90 }
91 assert(indice(4)>=0);
92
93 nouveaux(i,0) = i0;
94 nouveaux(i,1) = i1;
95 nouveaux(i,2) = i4;
96
97 nouveaux(nb_faces+i,0) = i1;
98 nouveaux(nb_faces+i,1) = i3;
99 nouveaux(nb_faces+i,2) = i4;
100
101 nouveaux(2*nb_faces+i,0) = i3;
102 nouveaux(2*nb_faces+i,1) = i2;
103 nouveaux(2*nb_faces+i,2) = i4;
104
105 nouveaux(3*nb_faces+i,0) = i2;
106 nouveaux(3*nb_faces+i,1) = i0;
107 nouveaux(3*nb_faces+i,2) = i4;
108
109 }
110 sommets.ref(nouveaux);
111}

116int Tetraedriser_homogene_compact::creer_sommet (
117 Domaine& dom,
118 Zone& zone,
119 DoubleTab& new_soms,
120 IntTab& elem_traite,
121 IntTab& new_soms_old_elems,
122 int NbSom, IntTab& sommets,
123 int& compteur,
124 int oldnbsom, int& nbnewsoms,IntTab& fait_sommet, int& nface ) const
125{
126
127 int _out=-1;
128 if(NbSom==4) //!< On verifie que le sommet rattache a une face n'a pas ete cree ulterieurement
129 {
130 int permut,som;
131
132 do
133 {
134 permut=0;
135 for(int j=1; j<4; j++)
136 {
137 if( sommets[j]<sommets[j-1] )
138 {
139 som=sommets[j-1];
140 sommets[j-1]=sommets[j];
141 sommets[j]=som;
142 permut++;
143 }
144 }
145 }
146 while(permut);
147
148
149
150 for (int i=0 ; i<nface ; i++)
151 if (fait_sommet(i,0)==sommets[0] && fait_sommet(i,1)==sommets[1] && fait_sommet(i,2)==sommets[2]) return fait_sommet(i,3);
152 }
153
154 const DoubleTab& coord_sommets=dom.coord_sommets();
155
156 double x=0,y=0,z=0;
157
158 for (int ii=0 ; ii<NbSom ; ii++)
159 {
160 x += coord_sommets(sommets(ii),0);
161 y += coord_sommets(sommets(ii),1);
162 z += coord_sommets(sommets(ii),2);
163 }
164 x /= NbSom;
165 y /= NbSom;
166 z /= NbSom;
167
168
169
170 _out = oldnbsom + nbnewsoms;
171 new_soms(compteur, 0) = x;
172 new_soms(compteur, 1) = y;
173 new_soms(compteur, 2) = z;
174 nbnewsoms++;
175 compteur++;
176
177 if(NbSom==4)
178 {
179 fait_sommet(nface, 0)=sommets[0];
180 fait_sommet(nface, 1)=sommets[1];
181 fait_sommet(nface, 2)=sommets[2];
182 fait_sommet(nface, 3)=_out;
183// Cerr << sommets(0) << " " << sommets(1) << " " << sommets(2) << " " << _out << finl;
184 nface++;
185 }
186
187 return _out;
188}
