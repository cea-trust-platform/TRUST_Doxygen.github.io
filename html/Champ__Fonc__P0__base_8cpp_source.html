75double Champ_Fonc_P0_base::moyenne(int ncomp) const
76{
77 const Zone_VF& zvf = la_zone_VF.valeur();
78 const DoubleVect& porosite = zvf.porosite_elem();
79 const DoubleVect& volumes = zvf.volumes();
80 const DoubleTab& val = valeurs();
81
82 double moy = 0, sum_vol=0;
83
84 for (int e = 0; e < zvf.nb_elem(); e++)
85 {
86 const double coef = porosite(e) * volumes(e);
87 moy += val(e, ncomp) * coef;
88 sum_vol += coef;
89 }
90 moy = mp_sum(moy);
91 sum_vol = mp_sum(sum_vol);
92 moy /= sum_vol;
93 return moy;
94}

101DoubleVect Champ_Fonc_P0_base::moyenne() const
102{
103 const Zone_VF& zvf = la_zone_VF.valeur();
104 const DoubleVect& porosite = zvf.porosite_elem();
105 const DoubleVect& volumes = zvf.volumes();
106 const DoubleTab& val = valeurs();
107
108 const int nb_compo = nb_comp();
109 DoubleVect moy(nb_compo);
110 moy = 0;
111 double sum_vol=0;
112
113 for (int e = 0; e < zvf.nb_elem(); e++)
114 {
115 const double coef = porosite(e) * volumes(e);
116 for (int k = 0; k < nb_compo; k++)
117 moy[k] += val(e, k) * coef;
118 sum_vol += coef;
119 }
120
121 moy /= sum_vol;
122 return moy;
123}
