40DoubleTab& Champ_som_lu::valeur_aux_elems(const DoubleTab& positions, const IntVect& les_polys, DoubleTab& val) const
41{
42 if (val.nb_dim() == 2)
43 {
44 assert(val.dimension(0) == les_polys.size_array());
45 assert(val.line_size() == nb_compo_);
46 }
47 else erreur_champ_(__func__);
48
49 const DoubleTab& coord = mon_domaine->coord_sommets();
50 const Zone& zone = mon_domaine->zone(0);
51 const DoubleTab& ch = valeurs();
52 int le_poly;
53
54 for (int rang_poly = 0; rang_poly < les_polys.size_array(); rang_poly++)
55 {
56 le_poly = les_polys(rang_poly);
57 if (le_poly == -1)
58 for (int ncomp = 0; ncomp < nb_compo_; ncomp++) val(rang_poly, ncomp) = 0;
59 else
60 for (int ncomp = 0; ncomp < nb_compo_; ncomp++)
61 {
62 int som1 = zone.sommet_elem(le_poly, 0);
63 int som2 = zone.sommet_elem(le_poly, 1);
64 int som3 = zone.sommet_elem(le_poly, 2);
65 int som4 = zone.sommet_elem(le_poly, 3);
66 int som5 = zone.sommet_elem(le_poly, 4);
67 int som6 = zone.sommet_elem(le_poly, 5);
68 int som7 = zone.sommet_elem(le_poly, 6);
69 int som8 = zone.sommet_elem(le_poly, 7);
70
71 double xmin = coord(som1, 0);
72 double xmax = coord(som2, 0);
73 double ymin = coord(som1, 1);
74 double ymax = coord(som3, 1);
75 double zmin = coord(som1, 2);
76 double zmax = coord(som5, 2);
77
78 double hx = xmax - xmin;
79 double hy = ymax - ymin;
80 double hh = zmax - zmin;
81
82 val(rang_poly, ncomp) = ch(som1, ncomp) * (xmax - positions(rang_poly, 0)) * (ymax - positions(rang_poly, 1)) * (zmax - positions(rang_poly, 2));
83 val(rang_poly, ncomp) += ch(som2, ncomp) * (positions(rang_poly, 0) - xmin) * (ymax - positions(rang_poly, 1)) * (zmax - positions(rang_poly, 2));
84 val(rang_poly, ncomp) += ch(som3, ncomp) * (xmax - positions(rang_poly, 0)) * (positions(rang_poly, 1) - ymin) * (zmax - positions(rang_poly, 2));
85 val(rang_poly, ncomp) += ch(som4, ncomp) * (positions(rang_poly, 0) - xmin) * (positions(rang_poly, 1) - ymin) * (zmax - positions(rang_poly, 2));
86 val(rang_poly, ncomp) += ch(som5, ncomp) * (xmax - positions(rang_poly, 0)) * (ymax - positions(rang_poly, 1)) * (positions(rang_poly, 2) - zmin);
87 val(rang_poly, ncomp) += ch(som6, ncomp) * (positions(rang_poly, 0) - xmin) * (ymax - positions(rang_poly, 1)) * (positions(rang_poly, 2) - zmin);
88 val(rang_poly, ncomp) += ch(som7, ncomp) * (xmax - positions(rang_poly, 0)) * (positions(rang_poly, 1) - ymin) * (positions(rang_poly, 2) - zmin);
89 val(rang_poly, ncomp) += ch(som8, ncomp) * (positions(rang_poly, 0) - xmin) * (positions(rang_poly, 1) - ymin) * (positions(rang_poly, 2) - zmin);
90
91 val(rang_poly, ncomp) /= (hx * hy * hh);
92 }
