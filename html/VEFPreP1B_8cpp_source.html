29// XD vefprep1b vef vefprep1b -1 Finite element volume discretization (P1NC/P1-bubble element). Since the 1.5.5 version, several new discretizations are available thanks to the optional keyword Read. By default, the VEFPreP1B keyword is equivalent to the former VEFPreP1B formulation (v1.5.4 and sooner). P0P1 (if used with the strong formulation for imposed pressure boundary) is equivalent to VEFPreP1B but the convergence is slower. VEFPreP1B dis is equivalent to VEFPreP1B dis Read dis { P0 P1 Changement_de_base_P1Bulle 1 Cl_pression_sommet_faible 0 }

99//! Discretisation d'un champ pour le VEFP1B en fonction d'une directive<br>de discretisation. La directive est un Motcle comme "pression",<br>"divergence_vitesse" (cree un champ de type P1_isoP1Bulle).<br>Cette methode determine le type du champ a creer en fonction du type d'element<br>et de la directive de discretisation. Elle determine ensuite le nombre de ddl<br>et fixe l'ensemble des parametres du champ (type, nb_compo, nb_ddl, nb_pas_dt,<br>nom(s), unite(s), nature du champ et attribue un temps) et associe la Zone_dis au champ.<br>Voir le code pour avoir la correspondance entre les directives et<br>le type de champ cree.
100void VEFPreP1B::discretiser_champ(
101 const Motcle& directive, const Zone_dis_base& z,
102 Nature_du_champ nature,
103 const Noms& noms, const Noms& unites,
104 int nb_comp, int nb_pas_dt, double temps,
105 Champ_Inc& champ, const Nom& sous_type) const
106{
107 Motcles motcles(2);
108 motcles[0] = "pression"; //!< Choix standard pour la pression
109 motcles[1] = "divergence_vitesse"; //!< Le type de champ obtenu en calculant div v
110
111 Nom type;
112 int default_nb_comp = 0; //!< Valeur par defaut du nombre de composantes
113 int rang = motcles.search(directive);
114 switch(rang)
115 {
116 case 0:
117 type = "Champ_P1_isoP1Bulle";
118 default_nb_comp = 1;
119 break;
120 case 1:
121 type = "Champ_P1_isoP1Bulle";
122 default_nb_comp = 1;
123 break;
124 default:
125 assert(rang < 0);
126 break;
127 }
128
129 if (directive == demande_description)
130 Cerr << "VEFPreP1B : " << motcles;
131
132// Si on n'a pas compris la directive (ou si c'est une demande_description)
133// alors on appelle l'ancetre :
134 if (rang < 0)
135 {
136 VEF_discretisation::discretiser_champ(directive, z,
137 nature, noms, unites,
138 nb_comp, nb_pas_dt, temps, champ);
139 return;
140 }
141
142 const int nb_ddl = -1; //!< c'est le descripteur p1b qui donnera le chiffre
143 if (nb_comp < 0)
144 nb_comp = default_nb_comp;
145 assert(nb_comp > 0);
146 creer_champ(champ, z, type, noms[0], unites[0], nb_comp, nb_ddl, nb_pas_dt, temps,
147 directive, que_suis_je());
148
149 if (nature == multi_scalaire)
150 {
151 Cerr<<"There is no field of type Champ_Inc with P1Bulle discretization"<<finl;
152 Cerr<<"and a multi_scalaire nature."<<finl;
153 exit();
154 }
155
156}

189//! Idem que VEF_discretisation::discretiser_champ(... , Champ_Inc)<br>Traitement commun aux champ_fonc et champ_don.<br>Cette methode est privee (passage d'un Objet_U pas propre vu<br>de l'exterieur ...)

198// suivant le type de l'objet champ.

200 Champ_Don * champ_don = 0;
201 if (sub_type(Champ_Fonc, champ))
202 champ_fonc = & ref_cast(Champ_Fonc, champ);
203 else
204 champ_don = & ref_cast(Champ_Don, champ);
205
206 Motcles motcles(2);
207 motcles[0] = "pression"; //!< Choix standard pour la pression
208 motcles[1] = "divergence_vitesse"; //!< Le type de champ obtenu en calculant div v
209
210 Nom type;
211 int default_nb_comp = 0; //!< Valeur par defaut du nombre de composantes
212 int rang = motcles.search(directive);
213 switch(rang)
214 {
215 case 0:
216 type = "Champ_Fonc_P1_isoP1Bulle";
217 default_nb_comp = 1;
218 break;
219 case 1:
220 type = "Champ_Fonc_P1_isoP1Bulle";
221 default_nb_comp = 1;
222 break;
223 default:
224 assert(rang < 0);
225 break;
226 }
227
228 if (directive == demande_description)
229 Cerr << "VEFPreP1B : " << motcles;
230
231// Si on n'a pas compris la directive (ou si c'est une demande_description)
232// alors on appelle l'ancetre :
233 if (rang < 0)
234 {
235 if (champ_fonc)
236 VEF_discretisation::discretiser_champ(directive, z, nature, noms, unites,
237 nb_comp, temps, *champ_fonc);
238 else
239 VEF_discretisation::discretiser_champ(directive, z, nature, noms, unites,
240 nb_comp, temps, *champ_don);
241 return;
242 }
243
244 int nb_ddl = -1; //!< c'est le descripteur p1b qui donnera le chiffre
245
246 if (nb_comp < 0)
247 nb_comp = default_nb_comp;
248 assert(nb_comp > 0);
249 if (champ_fonc)
250 creer_champ(*champ_fonc, z, type, noms[0], unites[0], nb_comp, nb_ddl, temps,
251 directive, que_suis_je());
252 else
253 creer_champ(*champ_don, z, type, noms[0], unites[0], nb_comp, nb_ddl, temps,
254 directive, que_suis_je());
255
256 if (nature == multi_scalaire)
257 {
258 Cerr<<"There is no field of type Champ_Fonc or Champ_Don with P1Bulle discretization"<<finl;
259 Cerr<<"and a multi_scalaire nature."<<finl;
