41double Echange_contact_Colburn_VDF::calculer_coefficient_echange(int i)
42{
43 double Re,Pr;
44 Re = std::fabs(getQh()*getDh()/getMu(i));
45 Pr = getMu(i)*getCp()/getLambda(i);
46 return 0.023*pow(Re,0.8)*pow(Pr,1./3.)*getLambda(i)/getDh();
47}
