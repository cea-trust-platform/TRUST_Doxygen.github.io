47void Refine_Mesh::apply(void)
48{
49 Cerr << "Refining domain " << domaine().le_nom() << finl;
50
51 check_dimension();
52 check_nb_zones();
53 check_cell_type();
54
55 if ( Objet_U::dimension == 2 )
56 {
57 apply_2D();
58 }
59 else
60 {
61 apply_3D();
62 }
63
64 Cerr << "Refinement... OK" << finl;
65}

67void Refine_Mesh::check_dimension(void) const
68{
69 if ( ! ( (Objet_U::dimension == 2) || (Objet_U::dimension == 3) ) )
70 {
71 Cerr << "Error in 'Refine_Mesh::check_dimension()':" << finl;
72 Cerr << " Invalid dimension: " << Objet_U::dimension << finl;
73 Cerr << " Refine_Mesh can only deal 2D or 3D domains" << finl;
74 Process::exit();
75 }

104{
105 Domaine& domain = domaine();

112 build_edges_2D(nodes_of_edges, edges_of_cells);

241void Refine_Mesh::build_edges(IntTab& nodes_of_edges, IntTab& edges_of_cells, int nb_edges_per_cell, int nb_nodes_per_edge, const IntTab& local_nodes_of_edges) const
242{
243 const IntTab& cells = domaine().zone(0).les_elems();
244
245 const int nb_cells = cells.dimension(0);
246
247 int size = nb_cells * nb_edges_per_cell;
248 IntTab nodes_of_edges_tmp(size,nb_nodes_per_edge);
249 edges_of_cells.resize(nb_cells, nb_edges_per_cell);
250 ArrOfInt global_nodes_of_edge(nb_nodes_per_edge);
251
252 for (int c=0; c<nb_cells; ++c)
253 {
254 for (int e=0; e<nb_edges_per_cell; ++e)
255 {
256 const int idx = c * nb_edges_per_cell + e;
257
258 for (int n=0; n<nb_nodes_per_edge; ++n)
259 {
260 global_nodes_of_edge[n] = cells(c,local_nodes_of_edges(e,n));
261 }
262
263 nodes_of_edges_tmp(idx,0) = (global_nodes_of_edge[0]<global_nodes_of_edge[1]) ? global_nodes_of_edge[0] : global_nodes_of_edge[1];
264 nodes_of_edges_tmp(idx,1) = (global_nodes_of_edge[0]<global_nodes_of_edge[1]) ? global_nodes_of_edge[1] : global_nodes_of_edge[0];
265
266 edges_of_cells(c,e) = idx;
267 }
268 }
269
270 ArrOfInt renum;
271 ArrOfInt renum_inverse;
272 calculer_renum_sans_doublons(nodes_of_edges_tmp,renum,renum_inverse);
273
274 const int nb_edges = renum_inverse.size_array();
275 nodes_of_edges.resize(nb_edges,nb_nodes_per_edge);
276 for (int e=0; e<nb_edges; ++e)
277 {
278 for (int n=0; n<nb_nodes_per_edge; ++n)
279 {
280 nodes_of_edges(e,n) = nodes_of_edges_tmp(renum_inverse[e],n);
281 }
282 }
283
284 for (int c=0; c<nb_cells; ++c)
285 {
286 for (int e=0; e<nb_edges_per_cell; ++e)
287 {
288 edges_of_cells(c,e) = renum[edges_of_cells(c,e)];
289 }
290 }
291}

293void Refine_Mesh::build_new_nodes(DoubleTab& new_nodes, const IntTab& nodes_of_edges) const
294{
295 const DoubleTab& nodes = domaine().les_sommets();
296
297 const int nb_nodes = nodes.dimension(0);
298 const int nb_edges = nodes_of_edges.dimension(0);
299
300 new_nodes.resize(nb_nodes+nb_edges,Objet_U::dimension);
301 for (int n=0; n<nb_nodes; ++n)
302 {
303 for (int i=0; i<Objet_U::dimension; ++i)
304 {
305 new_nodes(n,i) = nodes(n,i);
306 }
307 }
308
309 for (int e=0; e<nb_edges; ++e)
310 {
311 for (int i=0; i<Objet_U::dimension; ++i)
312 {
313 new_nodes(e+nb_nodes,i) = 0.5 * (nodes(nodes_of_edges(e,0),i) + nodes(nodes_of_edges(e,1),i));
314 }
315 }

318void Refine_Mesh::build_new_cells_2D(IntTab& new_cells, const IntTab& edges_of_cells) const
319{
320 const DoubleTab& nodes = domaine().les_sommets();
321 const IntTab& cells = domaine().zone(0).les_elems();
322
323 const int nb_cells = cells.dimension(0);
324 const int nb_nodes = nodes.dimension(0);
325
326 new_cells.resize(4*nb_cells,3);
327
328 int n0, n1, n2, n01, n02, n12;
329 for (int c=0; c<nb_cells; ++c)
330 {
331 n0 = cells(c,0);
332 n1 = cells(c,1);
333 n2 = cells(c,2);
334 n01 = nb_nodes + edges_of_cells(c,0);
335 n02 = nb_nodes + edges_of_cells(c,1);
336 n12 = nb_nodes + edges_of_cells(c,2);
337
338 new_cells(4*c+0,0) = n0;
339 new_cells(4*c+0,1) = n01;
340 new_cells(4*c+0,2) = n02;
341 new_cells(4*c+1,0) = n01;
342 new_cells(4*c+1,1) = n1;
343 new_cells(4*c+1,2) = n12;
344 new_cells(4*c+2,0) = n02;
345 new_cells(4*c+2,1) = n12;
346 new_cells(4*c+2,2) = n2;
347 new_cells(4*c+3,0) = n12;
348 new_cells(4*c+3,1) = n02;
349 new_cells(4*c+3,2) = n01;
350 }

413void Refine_Mesh::build_new_sub_zones_descriptions(Noms& sub_zones_descriptions) const
414{
415 const int nb_sub_zones = domaine().nb_ss_zones();
416 sub_zones_descriptions.dimensionner(nb_sub_zones);
417
418 const int nb_new_cells_per_old_cell = (Objet_U::dimension == 2) ? 4 : 8;
419
420 for (int i=0; i<nb_sub_zones; ++i)
421 {
422 const Sous_Zone& sub_zone = domaine().ss_zone(i);
423 const int old_nb_cells_in_sub_zone = sub_zone.nb_elem_tot();
424 const int new_nb_cells_in_sub_zone = old_nb_cells_in_sub_zone * nb_new_cells_per_old_cell;
425 IntVect new_sub_zone_cells(new_nb_cells_in_sub_zone);
426 int idx = 0;
427 for (int c=0; c<old_nb_cells_in_sub_zone; ++c)
428 {
429 const int old_cell = sub_zone[c];
430 for (int j=0; j<nb_new_cells_per_old_cell; ++j)
431 {
432 new_sub_zone_cells[idx] = old_cell*nb_new_cells_per_old_cell+j;
433 ++idx;
434 }
435 }
436
437 SChaine os;
438 os << " { Liste " << new_sub_zone_cells << " } ";
439 sub_zones_descriptions[i] = Nom(os.get_str());
440 }
441}

443void Refine_Mesh::build_incidence_from_node_to_edges(int nb_nodes,
444 const IntTab& nodes_of_edges,
445 Static_Int_Lists& incidence) const
446{
447 const int nb_edges = nodes_of_edges.dimension(0);
448 const int nb_nodes_per_edge = nodes_of_edges.dimension(1);
449
450 ArrOfInt nb_incident_edges(nb_nodes);
451
452 for (int edge=0; edge<nb_edges; ++edge)
453 {
454 for (int n=0; n<nb_nodes_per_edge; ++n)
455 {
456 const int node = nodes_of_edges(edge,n);
457 ++nb_incident_edges[node];
458 }
459 }
460
461 incidence.set_list_sizes(nb_incident_edges);
462
463 nb_incident_edges = 0;
464 for (int edge=0; edge<nb_edges; ++edge)
465 {
466 for (int n=0; n<nb_nodes_per_edge; ++n)
467 {
468 const int node = nodes_of_edges(edge,n);
469 const int index = nb_incident_edges[node];
470 incidence.set_value(node, index, edge);
471 ++nb_incident_edges[node];
472 }
473 }
474
475 incidence.trier_liste(-1);
476}

600void Refine_Mesh::build_new_frontier_faces_2D(IntTab& new_nodes_of_frontier_faces,
601 IntTab& new_cells_of_frontier_faces,
602 const Static_Int_Lists& incidence_from_node_to_edges,
603 const Faces& old_frontier_faces) const
604{
605 const int nb_nodes = domaine().les_sommets().dimension(0);
606

626 new_cells_of_frontier_faces.resize(new_nb_faces,2);
627 new_cells_of_frontier_faces = -1;

630void Refine_Mesh::build_new_frontier_faces_3D(IntTab& new_nodes_of_frontier_faces,
631 IntTab& new_cells_of_frontier_faces,
632 const Static_Int_Lists& incidence_from_node_to_edges,
633 const Faces& old_frontier_faces) const
634{
635 const int nb_nodes = domaine().les_sommets().dimension(0);
636
637 const int old_nb_faces = old_frontier_faces.nb_faces();

640 const IntTab old_nodes_of_frontier_faces = old_frontier_faces.les_sommets();
641
642 new_nodes_of_frontier_faces.resize(new_nb_faces,3);
643 for (int f=0; f<old_nb_faces; ++f)
644 {
645 const int node0 = old_nodes_of_frontier_faces(f,0);
646 const int node1 = old_nodes_of_frontier_faces(f,1);
647 const int node2 = old_nodes_of_frontier_faces(f,2);
648
649 const int edge0 = find_edge(incidence_from_node_to_edges,node1,node2);
650 const int edge1 = find_edge(incidence_from_node_to_edges,node2,node0);
651 const int edge2 = find_edge(incidence_from_node_to_edges,node0,node1);
652
653 new_nodes_of_frontier_faces(4*f+0,0) = node0;
654 new_nodes_of_frontier_faces(4*f+0,1) = nb_nodes + edge2;
655 new_nodes_of_frontier_faces(4*f+0,2) = nb_nodes + edge1;
656
657 new_nodes_of_frontier_faces(4*f+1,0) = nb_nodes + edge2;
658 new_nodes_of_frontier_faces(4*f+1,1) = node1;
659 new_nodes_of_frontier_faces(4*f+1,2) = nb_nodes + edge0;
660
661 new_nodes_of_frontier_faces(4*f+2,0) = nb_nodes + edge1;
662 new_nodes_of_frontier_faces(4*f+2,1) = nb_nodes + edge0;
663 new_nodes_of_frontier_faces(4*f+2,2) = node2;
664
665 new_nodes_of_frontier_faces(4*f+3,0) = nb_nodes +edge0;
666 new_nodes_of_frontier_faces(4*f+3,1) = nb_nodes +edge1;
667 new_nodes_of_frontier_faces(4*f+3,2) = nb_nodes +edge2;
668 }
669
670 new_cells_of_frontier_faces.resize(new_nb_faces,2);
671 new_cells_of_frontier_faces = -1;

674int Refine_Mesh::find_edge(const Static_Int_Lists& incidence_from_node_to_edges, int node0, int node1) const
675{
676 ArrOfInt edges0;
677 incidence_from_node_to_edges.copy_list_to_array(node0, edges0);
678
679 ArrOfInt edges1;
680 incidence_from_node_to_edges.copy_list_to_array(node1, edges1);
681
682 array_calculer_intersection(edges0,edges1);
683
684 assert( edges0.size_array() == 1 );
685 return edges0[0];
686}
