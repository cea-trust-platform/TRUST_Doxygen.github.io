50Entree& Pb_Multiphase::lire_equations(Entree& is, Motcle& mot)
51{
52 bool already_read;
53
54 is >> mot;
55 if (mot == "correlations") lire_correlations(is), already_read = false;
56 else already_read = true;
57
58// Particular treatment in cas of Homogeneous Equilibrium Mechanical
59// calculation using the Pb_HEM class
60// We enforce a interfacial flux correlation with constant coefficient
61 if (que_suis_je() == "Pb_HEM")
62 {
63// enforce_interfacial_flux_correlation();
64 if (has_correlation("flux_interfacial"))
65 {
66 for (auto &&corr : correlations)
67 {
68 if (corr.second.valeur().que_suis_je() == "Flux_interfacial_Coef_Constant")
69 {
70 Cout << "Flux_interfacial_Coef_Constant is already defined." << finl;
71 }
72 else
73 {
74 Cerr << "An interfacial flux correlation is already defined in the data file, but from a different kind that coef_constant." << finl;
75 Cerr << "For a Pb_HEM, the interfacial flux correlation must be of the kind coef_constant. Please either remove the line as it will be automatically added or modify the expression." << finl;
76 Process::exit();
77 }
78 }
79 }
80 else
81 {
82// The coefficient 1e10 is enforced. If the user wants to
83// modify it, we invite her to add the line in the data
84// file.
85 Cout << "Pb_HEM: we add a interfacial flux correlation with constant coefficient." << finl;
86 Nom corr_FICC ("{ flux_interfacial coef_constant { ");
87 for (int ii = 0; ii < nb_phases(); ii++)
88 {
89 corr_FICC += nom_phase(ii);
90 corr_FICC += " 1e10 ";
91 }
92 corr_FICC += "} }";
93 Cout << "Expression added: " << corr_FICC << finl;
94 Cout << "Please add this line to the data file directly if you want to modify the enforced coefficient 1e10." << finl;
95
96 EChaine corr_flux_inter_coef_const(corr_FICC);
97 lire_correlations(corr_flux_inter_coef_const);
98 }
99 }
100// End of Pb_HEM special treatment
101
102 Cerr << "Reading of the equations" << finl;
103 for(int i = 0; i < nombre_d_equations(); i++, already_read = false)
104 {
105 if (!already_read) is >> mot;
106 is >> getset_equation_by_name(mot);
107 }
108
109 /* lecture d'equations optionnelles */
110 Noms noms_eq, noms_eq_maj; //noms de toutes les equations possibles!
111 Type_info::les_sous_types(Nom("Equation_base"), noms_eq);
112 for (int i = 0; i < noms_eq.size(); i++) noms_eq_maj.add(Motcle(noms_eq(i))); //ha ha ha
113 for (is >> mot; noms_eq_maj.rang(mot) >= 0; is >> mot)
114 {
115 eq_opt.add(Equation()); //une autre equation optionelle
116 eq_opt.dernier().valeur().typer(mot); //on lui donne le bon type
117 Equation_base& eq = eq_opt.dernier().valeur().valeur();
118// memes associations que pour les autres equations : probleme, milieu, schema en temps
119 eq.associer_pb_base(*this);
120 eq.associer_milieu_base(milieu());
121 eq.associer_sch_tps_base(le_schema_en_temps);
122 eq.associer_zone_dis(domaine_dis().zone_dis(0));
123 eq.discretiser(); //a faire avant de lire l'equation
124 is >> eq; //et c'est parti!
125 eq.associer_milieu_equation(); //remontee vers le milieu
126 }
127 return is;
128}

130Entree& Pb_Multiphase::lire_correlations(Entree& is)
131{
132 Motcle mot;
133 is >> mot;
134 if (mot != "{") Cerr << "correlations : { expected instead of " << mot << finl, Process::exit();
135 for (is >> mot; mot != "}"; is >> mot)
136 if (correlations.count(mot.getString())) Process::exit(que_suis_je() + " : a correlation already exists for " + mot + " !");
137 else correlations[mot.getString()].typer_lire(*this, mot, is);
138 return is;
139}
