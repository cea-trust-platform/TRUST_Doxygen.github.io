41//! ecriture d'un tableau sequentiel (idem que TRUSTVect::printOn() on ne sait pas quoi faire de pertinent pour un tableau distribue).
42 Sortie& printOn(Sortie& os) const override
43 {
44 assert(verifie_LINE_SIZE());
45 if (TRUSTVect<_TYPE_>::nproc() > 1 && TRUSTVect<_TYPE_>::get_md_vector().non_nul())
46 {
47 Cerr << "Error in TRUSTTab::printOn: try to print a parallel vector" << finl;
48 Process::exit();
49 }
50 os << nb_dim_ << finl;
51 assert(dimensions_[0] == dimension_tot_0_);
52 if (nb_dim_ > 0) os.put(dimensions_, nb_dim_, nb_dim_);
53
54 const int sz = TRUSTVect<_TYPE_>::size_array();
55 os << sz << finl;
56 const int l_size = TRUSTVect<_TYPE_>::line_size();
57 const _TYPE_ *data = TRUSTVect<_TYPE_>::addr();
58
59 if (sz > 0) os.put(data, sz, l_size);
60 return os;
61 }

63// Precondition: le md_vector_ doit etre nul.
64//! lecture d'un tableau sequentiel
65 Entree& readOn(Entree& is) override
66 {
67 if (TRUSTVect<_TYPE_>::get_md_vector().non_nul())
68 {
69// Que veut-on faire si on lit dans un vecteur ayant deja une structure parallele ?
70 Cerr << "Error in TRUSTTab::readOn: vector has a parallel structure" << finl;
71 Process::exit();
72 }
73
74 is >> nb_dim_;
75 if (nb_dim_ < 1 || nb_dim_ > MAXDIM_TAB)
76 {
77 Cerr << "Error in TRUSTTab::readOn: wrong nb_dim_ = " << nb_dim_ << finl;
78 Process::exit();
79 }
80 is.get(dimensions_, nb_dim_);
81 if (dimensions_[0] < 0)
82 {
83 Cerr << "Error in TRUSTTab::readOn: wrong dimension(0) = " << dimensions_[0] << finl;
84 Process::exit();
85 }
86 int l_size = 1;
87 for (int i = 1; i < nb_dim_; i++)
88 {
89 if (dimensions_[i] < 0)
90 {
91 Cerr << "Error in TRUSTTab::readOn: wrong dimension(" << i << ") = " << dimensions_[i] << finl;
92 Process::exit();
93 }
94 l_size *= dimensions_[i];
95 }
96 dimension_tot_0_ = dimensions_[0];
97 TRUSTVect<_TYPE_>::readOn(is);
98 TRUSTVect<_TYPE_>::set_line_size_(l_size);
99 if (dimension_tot_0_ * l_size != TRUSTVect<_TYPE_>::size_array())
100 {
101 Cerr << "Error in TRUSTTab::readOn: wrong size_array " << TRUSTVect<_TYPE_>::size_array() << ", expected " << dimension_tot_0_ * l_size << finl;
102 Process::exit();
103 }
104 assert(verifie_LINE_SIZE());
105 return is;
106 }

239private:
240 static constexpr int MAXDIM_TAB = 4;
241// Nombre de dimensions du tableau (nb_dim_>=1)
242 int nb_dim_;
243
244// Dimensions "reelles" (dimensions_[0] * line_size() = size_reelle()) : line_size() est egal au produit des dimensions_[i] pour 1 <= i < nb_dim_
245 int dimensions_[MAXDIM_TAB];
246
247// Dimension totale (nombre de lignes du tableau) = nb lignes reeles + nb lignes virtuelles
248// Les dimensions dimension_tot(i>=1) sont implicitement egales a dimension(i)
249 int dimension_tot_0_;
250
251 inline void verifie_MAXDIM_TAB() const
252 {
253 if (MAXDIM_TAB != 4)
254 {
255 Cerr << "Update code for MAXDIM_TAB for CHECK_LINE_SIZE" << finl;
256 throw;
257 }
258 }
259
260 inline bool verifie_LINE_SIZE() const
261 {
262 return ((TRUSTVect<_TYPE_>::line_size() == ((nb_dim_ > 1) ? dimensions_[1] : 1) * ((nb_dim_ > 2) ? dimensions_[2] : 1) * ((nb_dim_ > 3) ? dimensions_[3] : 1))
263 && (TRUSTVect<_TYPE_>::line_size() * dimension_tot_0_ == TRUSTVect<_TYPE_>::size_array()));
264 }
265
266 inline void init_dimensions(int * tab)
267 {
268#ifndef NDEBUG
269 for (int i = 0; i < MAXDIM_TAB; i++) tab[i] = -1;
270#endif
271 }
