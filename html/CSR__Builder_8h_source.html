19//! This class is a helper to build CSR stored matrices
20class CSR_Builder
21{
22public:
23 CSR_Builder();
24 initialize(int nb_lines);
25 inline void add_coeff(int line, int column, double coeff);
26 build_matrix(Matrice_Bloc&);
27protected:
28 int nb_lines_;
29// List data contains groups of 2 integers like this:
30// column, next_index, column, next_index, ...
31// column is the parameter passed to add_coeff
32// next_index is the group number of the next coefficient on the same line
33// or -1 if this is the last coefficient.
34// if not -1, the next group is
35// column = list_data_[2*next_index]
36// next_index = list_data_[2*next_index+1]
37// the coefficient is at coeff_data_[next_index]
38// The first coefficient on line "l" is at list_head[l]
39 ArrOfInt list_head_;
40// list_tail_[l] = -1 if the line "l" is empty, otherwise, gives the index
41// of the last "next_index" on the line (we have list_data_[2*tail+1]=-1)
42 ArrOfInt list_tail_;
43 ArrOfInt list_data_;
44 ArrOfDouble coeff_data_;
45};

47inline void CSR_Builder::add_coeff(int line, int column, double coeff)
48{
49 assert(line >= 0 && line < nb_lines_);
50
51// We put the new coefficient here:
52 const int n = coeff_data_.size_array();
53 assert(list_data_.size_array() == n*2); //!< mixing calls with and without "coeff" is forbidden
54 list_data_.resize_array(n*2+2);
55 list_data_[n*2] = column;
56 list_data_[n*2+1] = -1; //!< new end of list
57 coeff_data_.append_array(coeff);
58
59 const int tail = list_tail_[line];
60 if (tail < 0)
61 {
62// this line is empty, initialize list_head_
63 list_head_[line] = n;
64 }
65 else
66 {
67// update the last next_index of the line:
68 list_data_[2*tail+1] = n;
69 }
70 list_tail_[line] = n;
71}
