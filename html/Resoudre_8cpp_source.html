69//! Task of the interpretor:<br>Solve a problem (name read from the input stream)
70Entree& Resoudre::interpreter(Entree& is)
71{
72 Nom problem_name;
73 is >> problem_name;
74 Probleme_U& pb=ref_cast(Probleme_U,objet(problem_name));
75
76 Nom string("=====================================================");
77 Cerr << string << finl;
78 Cerr << "Initialization of the problem " << pb.le_nom() << " ("<<que_suis_je()<<") in progress ... " << finl;
79 Cerr << string << finl;
80// Initialize the problem
81 pb.initialize();
82
83// Register signal and signal handler (SIGINT) if user presses ctrl-c during exec
84 if (!Objet_U::DEACTIVATE_SIGINT_CATCH)
85 {
86 Catch_SIGINIT sig;
87 sig.set_nom_cas_pour_signal(pb.nom_du_cas());
88 signal(SIGINT, Catch_SIGINIT::signal_callback_handler);
89 }
90
91 Cerr << string << finl;
92 Cerr << "Solving of the problem " << pb.le_nom() << " ("<<que_suis_je()<<") in progress ... " << finl;
93 Cerr << string << finl;
94// Run the problem
95 bool ok = pb.run();
96
97// Terminate the problem
98 pb.terminate();
99 Cerr << string << finl;
100 Cerr << "End of solving of the problem " << pb.le_nom() << " ("<<que_suis_je()<<")" << finl;
101 Cerr << string << finl;
102 if (!ok)
103 {
104 Cerr << "Error while solving!" << finl;
105 exit();
106 }
107
108 return is;
109}
