36DoubleTab& Init_par_partie::valeur_aux(const DoubleTab& positions, DoubleTab& tab_valeurs) const
37{
38 assert(tab_valeurs.size_totale() * dimension / nbcomp == positions.size_totale());
39
40 for (int i = 0; i < tab_valeurs.dimension(0); i++)
41 {
42 if (nbcomp == 1)
43 {
44 if (positions(i, 0) > R_loc(0)) tab_valeurs(i) = val_don(0);
45 else tab_valeurs(i) = val_don(1);
46 }
47 else if (nbcomp == 2)
48 {
49 if (positions(i, 0) > R_loc(0)) tab_valeurs(i, 0) = val_don(0);
50 else tab_valeurs(i, 0) = val_don(1);
51
52 if (positions(i, 1) > R_loc(1)) tab_valeurs(i, 1) = val_don(2);
53 else tab_valeurs(i, 1) = val_don(3);
54 }
55 else if (nbcomp == 3)
56 {
57 if (positions(i, 0) > R_loc(0)) tab_valeurs(i, 0) = val_don(0);
58 else tab_valeurs(i, 0) = val_don(1);
59
60 if (positions(i, 1) > R_loc(1)) tab_valeurs(i, 1) = val_don(2);
61 else tab_valeurs(i, 1) = val_don(3);
62
63 if (positions(i, 2) > R_loc(2)) tab_valeurs(i, 2) = val_don(4);
64 else tab_valeurs(i, 2) = val_don(5);
65 }
66 else Cerr << " error nbcomp " << finl;
67 }
68
69 return tab_valeurs;
