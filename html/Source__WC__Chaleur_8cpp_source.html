56//! Ajoute le terme source weakly compressible<br>d P_tot / d t = del P / del t + u.grad(P_tot)
57/** \return Entree\&<br>
58Signification:<br>
59Contraintes: */
60DoubleTab& Source_WC_Chaleur::ajouter_(DoubleTab& resu) const
61{
62 double dt_ = mon_equation->schema_temps().temps_courant() - mon_equation->schema_temps().temps_precedent();
63 if (dt_ <= 0.) return resu; //!< On calcul pas ce terme source si dt<=0
64
65 Fluide_Weakly_Compressible& FWC = ref_cast_non_const(Fluide_Weakly_Compressible,le_fluide.valeur());
66 DoubleTab& Ptot = FWC.pression_th_tab(), Ptot_n = FWC.pression_thn_tab(); //!< present & passe
67
68 Ptot.echange_espace_virtuel();
69
70 DoubleTab UgradP(Ptot); //!< champ elem en vdf et face en vef
71 UgradP = 0.;
72
73// Only if non uniform and if the flag use_grad_pression_eos is 1
74 if (FWC.use_total_hydro_pressure() && FWC.use_grad_pression_eos()) compute_interpolate_gradP(UgradP,Ptot);
75
76 assert (resu.dimension(0) == UgradP.dimension(0));
77 assert (resu.size() == UgradP.size());
78
79 for (int i=0 ; i< resu.dimension(0); i++)
80 {
81 double dpth = ( Ptot(i,0) - Ptot_n(i,0) ) / dt_ + UgradP(i,0);
82 resu(i) += dpth * volumes(i)*porosites(i) ;
83 }
84 return resu;
85}

87const DoubleTab& Source_WC_Chaleur::correct_grad_boundary(const Zone_VF& zone, DoubleTab& grad_Ptot) const
88{
89// We dont have a CL for Ptot => we apply explicitly to have a null gradient on the boundary ...
90 const Zone_Cl_dis& zone_cl = mon_equation->zone_Cl_dis();
91 for (int n_bord=0; n_bord<zone.nb_front_Cl(); n_bord++)
92 {
93 const Cond_lim& la_cl = zone_cl.les_conditions_limites(n_bord);
94 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
95// recuperer face et remplace gradient par 0
96 const int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
97 for (int num_face=ndeb; num_face<nfin; num_face++) grad_Ptot(num_face,0) = 0.;
98 }
99 return grad_Ptot;
100}
