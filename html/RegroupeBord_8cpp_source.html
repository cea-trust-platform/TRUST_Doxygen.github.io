106void RegroupeBord::regroupe_bord(Domaine& dom, Nom nom,const LIST(Nom)& nlistbord)
107{
108
109 Zone& zone=dom.zone(0);
110
111 int nbfr=zone.nb_front_Cl();
112 int nbfacestot=0;
113 int prem_b=-1;
114
115 for (int b=0; b<nbfr; b++)
116 {
117 Frontiere& org=zone.frontiere(b);
118
119 if (nlistbord.contient(org.le_nom()))
120 {
121 if (prem_b==-1) prem_b=b;
122// Frontiere& org=zone.frontiere(b);
123 nbfacestot+=org.nb_faces();
124 }
125 }
126 Frontiere& newb=zone.frontiere(prem_b);
127
128
129 Faces& newfaces=newb.faces();
130// newfaces.typer(zone.frontiere(0).faces().type_face());
131 int nbsom=newfaces.nb_som_faces();
132 int nbfacestot0=newfaces.nb_faces();
133 newfaces.dimensionner(nbfacestot);
134 IntTab& newsommet=newfaces.les_sommets();
135 nbfacestot=nbfacestot0;
136 for (int b=0; b<nbfr; b++)
137 {
138 if (b!=prem_b)
139 {
140 Frontiere& org=zone.frontiere(b);
141 Nom nombord=org.le_nom();
142
143 if (nlistbord.contient(nombord))
144 {
145 const IntTab& sommets=org.faces().les_sommets();
146 int nbface=org.nb_faces();
147 for (int fa=0; fa<nbface; fa++)
148 for (int s=0; s<nbsom; s++)
149 newsommet(fa+nbfacestot,s)=sommets(fa,s);
150 nbfacestot+=org.nb_faces();
151 }
152 }
153 }
154 zone.frontiere(prem_b).nommer("bord_a_conserver_coute_que_coute");
155 Nom bordcons =zone.frontiere(prem_b).le_nom();
156
157 for (int b=0; b<nbfr; b++)
158 {
159// la recup des bords et des raccords est dans la boucle
160// pour pouvoir supprimer ...
161 Bords& listbord=zone.faces_bord();
162 Raccords& listrac=zone.faces_raccord();
163// const Nom& nombord=nlistbord[b];
164 Frontiere& org=zone.frontiere(b);
165 const Nom& nombord=org.le_nom();
166
167 if (nlistbord.contient(nombord))
168 if (nombord!=bordcons)
169 {
170
171 int num_b=zone.rang_frontiere(nombord);
172 if (num_b<listbord.size())
173 listbord.suppr(zone.bord(nombord));
174 else
175 listrac.suppr(zone.raccord(nombord));
176
177// on supprime un bord
178 nbfr--;
179 b--; //!< on veut verifie que le suivant n'est pas dans la liste
180
181 }
182
183 }
184 newb.nommer(nom);
185 zone.fixer_premieres_faces_frontiere();
186}
