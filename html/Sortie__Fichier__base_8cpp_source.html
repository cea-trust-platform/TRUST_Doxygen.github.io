71void Sortie_Fichier_base::set_toFlush()
72{
73 char* theValue = getenv("TRIOU_FLUSHFILES");
74// On fixe une valeur a 0 par defaut pour toFlush_
75// Elle peut etre changee par la variable d'environnement
76// On bufferize sur DEC pour accelerer les ecritures .lml, .lata, ...
77#ifdef DECalpha
78 toFlush_ = 0;
79#else
80 toFlush_ = 1;
81#endif
82// Les acces disques ont ete optimises depuis
83// donc par defaut on flushe a nouveau car
84// sinon les sondes pas ecrites...
85// Si lent, mettre export TRIOU_FLUSHFILES=0 dans son sub_file
86 toFlush_ = 1;
87 if (theValue != NULL)
88 {
89 toFlush_=atoi(theValue);
90 }
91}

93void Sortie_Fichier_base::set_buffer()
94{
95 if (!toFlush_)
96 {
97 char* theValue = getenv("TRIOU_BUFFSIZE");
98// On fixe une valeur a 3000000 par defaut pour buffSize
99// Elle peut etre changee par la variable d'environnement
100 int buffSize = 3000000;
101 if (theValue != NULL)
102 {
103 buffSize = atoi(theValue);
104 }
105 if (buffSize>=0)
106 {
107 size_t internalBuffSize = buffSize;
108 internalBuff_ = new char[internalBuffSize];
109 ofstream_->rdbuf()->pubsetbuf(internalBuff_,internalBuffSize);
110 toFlush_ = 0;
111 }
112 }
113}

155int Sortie_Fichier_base::ouvrir(const char* name,IOS_OPEN_MODE mode)
156{
157#ifdef ver_file
158 if (Process::me()!=0)
159 {
160
161 char* marq=strrchr(name,'_');
162 int error=0;
163 if (!marq)
164 error=1;
165 else
166 {
167 Nom test(name);
168 test.prefix(marq);
169 Nom suf(name);
170 marq=strrchr(name,'.');
171 if (!marq)
172 error=1;
173 else
174 {
175 int l=strlen(name);
176 int deb=l-strlen(marq)+1;
177 suf=suf.substr(deb,l);
178 test+=suf;
179 test=test.nom_me(Process::me());
180 if (test!=name) error=1;
181 }
182
183
184 }
185 if (error)
186 {
187 Cerr<<Process::me()<<name<<" strange :"<<__FILE__<<(int)__LINE__<<finl;
188 Cerr<<name<<finl;
189 ::abort();
190 }
191
192
193 }
194#endif
195 if (++counters[name]%1000==0) Cerr << "Warning, file " << name << " has been opened/closed " << counters[name] << " times..." << finl;
196 if(ofstream_)
197 delete ofstream_;
198 IOS_OPEN_MODE ios_mod=mode;
199 int new_bin=0;
200 if (bin_)
201 {
202
203 if (ios_mod==ios::out)
204 {
205 new_bin=1;
206 }
207 else
208 assert((ios_mod==ios::app)||(ios_mod==(ios::app|ios::out)));
209 ios_mod=ios_mod|ios::binary;
210 }
211 ofstream_ = new ofstream(name,ios_mod);
212 set_toFlush();
213 set_buffer();
214 set_ostream(ofstream_);
215 if (!ofstream_->good())
216 {
217 Cerr << "Error when opening the file " << name << finl;
218 Cerr << "Check if this file can be opened." << finl;
219 Process::exit();
220 }
221
222 if (new_bin)
223 {
224 Nom marq("INT64");
225#ifdef INT_is_64_
226 (*this)<<marq;
227#endif
228 }
229 return 1;
230}
