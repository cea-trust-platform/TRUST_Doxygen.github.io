21enum class Champ_Divers_Type { INUTILE , UNIFORME , CANAL };
22
23template <Champ_Divers_Type _TYPE_>
24class TRUSTChamp_Divers_generique : public Champ_Don_base
25{
26public:
27 Champ_base& affecter(const Champ_base& ch)
28 {
29 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME), IS_CANAL = (_TYPE_ == Champ_Divers_Type::CANAL);
30 if (IS_UNIFORME) return Champ_Don_base::affecter(ch);
31 if (IS_CANAL) Champ_base::affecter_erreur();
32
33 return *this;
34 }

36 DoubleVect& valeur_a(const DoubleVect&, DoubleVect& tab_valeurs) const override
37 {
38 assert (_TYPE_ != Champ_Divers_Type::CANAL);
39 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME);
40 if (IS_UNIFORME)
41 {
42 for (int j = 0; j < nb_comp(); j++) tab_valeurs(j) = valeurs_(0, j);
43 return tab_valeurs;
44 }
45
46 return not_implemented_champ_<DoubleVect&>(__func__); /* Cas INUTILE : methode sur-chargee pour CANAL */
47 }
48
49 double valeur_a_compo(const DoubleVect& x, int ncomp) const override
50 {
51 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME), IS_CANAL = (_TYPE_ == Champ_Divers_Type::CANAL);
52 if (IS_UNIFORME) return valeurs_(0,ncomp);
53 if (IS_CANAL) return Champ_Don_base::valeur_a_compo(x,ncomp);
54
55 return not_implemented_champ_<double>(__func__); /* INUTILE */
56 }
57
58 DoubleVect& valeur_a_elem(const DoubleVect& positions, DoubleVect& tab_valeurs, int) const override
59 {
60 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME), IS_CANAL = (_TYPE_ == Champ_Divers_Type::CANAL);
61 if (IS_UNIFORME)
62 {
63 for (int j = 0; j < nb_comp(); j++) tab_valeurs(j) = valeurs_(0, j);
64 return tab_valeurs;
65 }
66
67 if (IS_CANAL) return valeur_a(positions, tab_valeurs); //!< from VTABLE
68
69 return not_implemented_champ_<DoubleVect&>(__func__); /* INUTILE */
70 }
71
72 double valeur_a_elem_compo(const DoubleVect&, int, int ncomp) const override
73 {
74 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME);
75 return IS_UNIFORME ? valeurs_(0,ncomp) : not_implemented_champ_<double>(__func__); /* methode sur-chargee pour CANAL */
76 }

78 DoubleTab& valeur_aux(const DoubleTab& , DoubleTab& tab_valeurs) const override
79 {
80 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME);
81 if (IS_UNIFORME)
82 {
83 for (int i = 0; i < tab_valeurs.dimension_tot(0); i++) //!< GF dimension_tot pour que la ligne soit valide pour les champs P1B
84 for (int j = 0; j < tab_valeurs.line_size(); j++)
85 tab_valeurs(i, j) = valeurs_(0, j);
86
87 return tab_valeurs;
88 }
89
90 return not_implemented_champ_<DoubleTab&>(__func__); //!< voir classes filles ...
91 }
92
93 DoubleVect& valeur_aux_compo(const DoubleTab&, DoubleVect& tab_valeurs, int ncomp) const override
94 {
95 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME);
96 if (IS_UNIFORME) return tab_valeurs = valeurs_(0, ncomp);
97
98 return not_implemented_champ_<DoubleVect&>(__func__); //!< BOOM
99 }
100
101 DoubleTab& valeur_aux_elems(const DoubleTab& positions, const IntVect&, DoubleTab& tab_valeurs) const override
102 {
103 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME);
104 if (IS_UNIFORME)
105 {
106 for (int i = 0; i < tab_valeurs.dimension(0); i++)
107 for (int j = 0; j < tab_valeurs.line_size(); j++)
108 tab_valeurs(i, j) = valeurs_(0, j);
109
110 return tab_valeurs;
111 }
112
113 return valeur_aux(positions, tab_valeurs); //!< from VTABLE
114 }
115
116 DoubleVect& valeur_aux_elems_compo(const DoubleTab& positions, const IntVect&, DoubleVect& tab_valeurs, int ncomp) const override
117 {
118 static constexpr bool IS_UNIFORME = (_TYPE_ == Champ_Divers_Type::UNIFORME);
119 if (IS_UNIFORME) return tab_valeurs = valeurs_(0, ncomp);
120
121 return valeur_aux_compo(positions, tab_valeurs, ncomp); //!< from VTABLE
122 }
123};
