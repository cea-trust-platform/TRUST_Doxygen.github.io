195int Navier_Stokes_std::lire_motcle_non_standard(const Motcle& mot, Entree& is)
196{
197 if (mot=="diffusion")
198 {
199 Cerr << "Reading and typing of the diffusion operator : " << finl;
200 terme_diffusif.associer_diffusivite(diffusivite_pour_transport());
201 is >> terme_diffusif;
202// le champ pour le dt_stab est le meme que celui de l'operateur
203 terme_diffusif.associer_diffusivite_pour_pas_de_temps(diffusivite_pour_pas_de_temps());
204 return 1;
205 }
206 else if (mot=="convection")
207 {
208 Cerr << "Reading and typing of the convection operator : " << finl;
209 const Champ_base& vitesse_transportante = vitesse_pour_transport();
210 terme_convectif.associer_vitesse(vitesse_transportante);
211 is >> terme_convectif;
212 return 1;
213 }
214 else if (mot=="solveur_pression")
215 {
216 Cerr << "Reading and typing of pressure solver : " << finl;
217 is >> solveur_pression_;
218 Cerr<<"Pressure solver type : "<<solveur_pression_.valeur().que_suis_je()<< finl;
219 solveur_pression_.nommer("solveur_pression");
220 return 1;
221 }
222 else if (mot=="dt_projection")
223 {
224 Cerr << "Reading projection time step " << finl;
225 is >> dt_projection;
226 is >> seuil_projection;
227 return 1;
228 }
229 else if (mot=="Traitement_particulier")
230 {
231 Cerr << "Reading and typing of Traitement_particulier considered : " << finl;
232 Nom type="Traitement_particulier_NS_";
233 Motcle motbidon;
234 Motcle accouverte = "{" , accfermee = "}" ;
235 is >> motbidon;
236 if (motbidon == accouverte)
237 {
238 Motcle le_cas;
239 is >> le_cas;
240 if (le_cas == accfermee)
241 le_cas ="";
242 else
243 {
244 type+= le_cas;
245 type+= "_";
246 }
247 Nom discr=discretisation().que_suis_je();
248 if (discr == "VEFPreP1B")
249 discr = "VEF";
250 type+=discr;
251 Cerr << type << finl;
252 le_traitement_particulier.typer(type);
253 le_traitement_particulier.associer_eqn(*this);
254 le_traitement_particulier.lire(is);
255 }
256 else
257 {
258 Cerr << "Error while reading Traitement_particulier for Navier_Stokes_standard equation";
259 Cerr << "A { was expected." << finl;
260 exit();
261 }
262 return 1;
263 }
264 else if (mot=="Erreur_max_DivU")
265 {
266 Cerr << "Reading DivU maximun" << finl;
267 is >> max_div_U;
268 is >> seuil_projection;
269 return 1;
270 }
271 else if (mot=="seuil_divU")
272 {
273 Cerr << "Reading the threshold value for the velocity divergence " << finl;
274 is >> seuil_divU;
275 is >> raison_seuil_divU;
276 return 1;
277 }
278 else if (mot=="solveur_bar")
279 {
280 Motcle motlu;
281 Motcle accouverte = "{" , accfermee = "}" ;
282 Nom type_solv("");
283 int acc=0;
284 int ok=0;
285 while (acc!=0 || !ok)
286 {
287 is >> motlu;
288 if (motlu==accouverte)
289 {
290 ok=1;
291 acc++;
292 };
293 if (motlu==accfermee) acc--;
294 type_solv+=" ";
295 type_solv+=motlu;
296 }
297 if (je_suis_maitre())
298 {
299 SFichier s("solveur.bar");
300 s<<type_solv<<finl;
301 s.close();
302 }
303 return 1;
304 }
305 else if (mot=="methode_calcul_pression_initiale")
306 {
307 Motcle methode;
308 is >> methode;
309 Motcles compris(4);
310 compris[0]="avec_les_cl";
311 compris[1]="avec_sources";
312 compris[2]="avec_sources_et_operateurs";
313 compris[3]="sans_rien";
314 methode_calcul_pression_initiale_=compris.rang(methode);
315 if (methode_calcul_pression_initiale_<0)
316 {
317 Cerr<<methode<<" is not understood."<<finl;
318 Cerr<<" Allowed keywords are :"<<compris<<finl;
319 exit();
320 }
321 return 1;
322 }
323 else
324 return Equation_base::lire_motcle_non_standard(mot,is);
325 return 1;
326}

901//
902//
903//! Resolution de la pression, inconnue implicitee de Navier Stokes
904DoubleTab& Navier_Stokes_std::corriger_derivee_impl(DoubleTab& derivee)
905{
906// We want to solve:
907// dU/dt + M-1 Bt Cp = M-1(F - BtP)
908// B dU/dt = 0
909// with F explicit terms: sum(operators)+sources
910// In: derivee = M-1(F - BtP(n))
911// Out: derivee = M-1(F - BtP(n+1)), P(n+1)=P(n)+Cp
912
913// In case of ALE:
914// Out: derivee = M-1(F + ALEconvectiveTerm - BtP(n+1)) = derivee_out, P(n+1)=P(n)+Cp
915// Cp is calculated from the following equation:
916// (BJ_{n}M-1Bt)Cp=B(J_{n}U_{n}/timestep+J_{n}derivee_out-J_{n+1}U_ALE), J-Jacobian, _{n}- at time n
917
918 DoubleTab& tab_pression=la_pression.valeurs();
919 DoubleTab& gradP=gradient_P.valeurs();
920 DoubleTrav secmemP(tab_pression);
921 DoubleTrav deriveeALE(derivee);
922
923 double timestep=probleme().schema_temps().pas_de_temps();
924
925// can be used for methods like ALE
926 renewing_jacobians( derivee );
927
928 if (div_u_nul_et_non_dsurdt_divu_)
929 {
930// on veut div u =0 et non d/dt (div u)=0 pour eviter de cumuler les erreurs
931// cela ne marche qu'avec les schema type euler_explicite
932 DoubleTab derivee2(derivee);
933 const double dt=schema_temps().pas_de_temps();
934 derivee2*=dt;
935 derivee2+=la_vitesse.passe();
936 derivee2/=dt;
937 if( !sub_type(Op_Conv_ALE, terme_convectif.valeur()) ) //No ALE method
938 {
939 divergence.calculer(derivee2, secmemP); //!< Div(M-1(F - BtP))
940 }
941 }
942 else
943 {
944 if( !sub_type(Op_Conv_ALE, terme_convectif.valeur()) ) //No ALE method
945 {
946 divergence.calculer(derivee, secmemP); //!< Div(M-1(F - BtP))
947 }
948 }
949
950 if( !sub_type(Op_Conv_ALE, terme_convectif.valeur()) ) //No ALE method
951 {
952 secmemP *= -1; //!< car div =-B Correction du second membre d'apres les conditions aux limites :
953//
954 assembleur_pression_.modifier_secmem(secmemP);
955 }
956
957// Set print of the linear system solve according to dt_impr:
958 solveur_pression_->fixer_schema_temps_limpr(schema_temps().limpr());
959
960 const bool is_ALE = ( sub_type(Op_Conv_ALE, terme_convectif.valeur()) );
961
962 if (assembleur_pression_.valeur().get_resoudre_increment_pression())
963 {
964 if( is_ALE )
965 {
966// we don't want to have domaine_ale object here
967 div_ale_derivative( deriveeALE, timestep, derivee, secmemP );
968 }
969// Solve B M-1 Bt Cp = M-1(F - BtP)
970 DoubleTrav Cp(tab_pression);
971 solveur_pression_.resoudre_systeme(matrice_pression_.valeur(), secmemP, Cp);
972
973// P(n+1) = P(n) + Cp
974 tab_pression += Cp;
975 assembleur_pression_.modifier_solution(tab_pression);
976
977// M-1 Bt P(n+1)
978 solveur_masse.appliquer(gradP);
979 derivee += gradP; //!< M-1 F
980 }
981 else
982 {
983// Solve B M-1 Bt P(n+1) = B M-1 F
984 solveur_pression_.resoudre_systeme(matrice_pression_.valeur(), secmemP, tab_pression);
985 assembleur_pression_.modifier_solution(tab_pression);
986// It is not done anymore cause:
987// Iterative solvers are less accurate
988// Time converges in O(sqrt(dt)) and not O(dt)
989// See: http://www.sciencedirect.com/science/article/pii/S0021999108004518
990 }
991
992// (BM) gradient operator requires updated virtual space in source vector
993// Calculate Bt P(n+1)
994 tab_pression.echange_espace_virtuel();
995 gradient.calculer(tab_pression, gradP);
996
997// gradP = Bt P(n+1) is kept and
998// M-1Bt P(n+1) is calculated:
999 DoubleTrav Mmoins1gradP(gradP);
1000 Mmoins1gradP = gradP;
1001 solveur_masse.appliquer(Mmoins1gradP);
1002
1003// dU/dt = M-1(F-Bt P(n+1))
1004 derivee -= Mmoins1gradP;
1005
1006 return derivee;
1007}

1030void Navier_Stokes_std::projeter()
1031{
1032 if (probleme().is_dilatable() && probleme().reprise_effectuee())
1033 {
1034 Cerr<<"WARNING: Quasi compressible model --> no projection (except the first time step)."<<finl;
1035 }
1036 else
1037 {
1038 DoubleTab& tab_vitesse=la_vitesse.valeurs();
1039 tab_vitesse.echange_espace_virtuel();
1040 la_pression.valeurs().echange_espace_virtuel();
1041 DoubleTrav secmem(la_pression.valeurs());
1042// Cela ne sert a rien d'initialiser lagrange avec la pression
1043// voir ca penalise le calcul en p1B et CL p<>0
1044// On prend un DoubleTrav au lieu d'un DoubleTab pour avoir lagrange=0
1045 DoubleTrav lagrange(la_pression.valeurs());
1046 DoubleTrav gradP(gradient_P.valeurs());
1047
1048 double normal_seuil=0;
1049
1050// M u + eBt l = M v
1051// B u = 0
1052// => e B(M-1)Bt l = Bv
1053//
1054 if ((i_source_pdf_ != -1) && (correction_vitesse_projection_initiale_==1))
1055 {
1056 Cerr<<"(IBM) Immersed Interface: modified initial velocity for initial projection."<<finl;

1058 src.correct_vitesse(champ_coeff_pdf_som_, tab_vitesse);
1059 tab_vitesse.echange_espace_virtuel();
1060 }
1061 divergence.calculer(tab_vitesse, secmem);
1062// Desormais on calcule le pas de temps avant la projection
1063// Avant, on avait dt=dt_min au debut du calcul
1064 double dt = std::max(le_schema_en_temps->pas_temps_min(),calculer_pas_de_temps());
1065 dt = std::min(dt, le_schema_en_temps->pas_temps_max());
1066
1067 secmem*=(-1./dt);
1068 secmem.echange_espace_virtuel();
1069
1070 double bilan=mp_norme_vect(secmem);
1071 Cout << "------------- Projection -----------------" << finl;
1072 Cout << "--------------------------------------------" << finl;
1073 Cout << "Bilan de masse avant projection : " << bilan << finl;
1074
1075 if( sub_type(solv_iteratif,solveur_pression_.valeur()) )
1076 {
1077 solv_iteratif& solv_iter=ref_cast(solv_iteratif,solveur_pression_.valeur());
1078 normal_seuil=solv_iter.get_seuil();
1079 solv_iter.set_seuil(seuil_projection);
1080 }
1081
1082// Correction du second membre d'apres les conditions aux limites :
1083 solveur_pression_.resoudre_systeme(matrice_pression_.valeur(),secmem,lagrange);
1084 assembleur_pression_.modifier_solution(lagrange);
1085 lagrange.echange_espace_virtuel();
1086
1087// M-1 Bt l
1088 gradient->multvect(lagrange, gradP);
1089 if ((i_source_pdf_ != -1) && (correction_vitesse_projection_initiale_==1))
1090 {
1091 Cerr<<"(IBM) Immersed Interface: modified velocity corrector for initial projection."<<finl;
1092 gradP /= champ_coeff_pdf_som_;
1093 }
1094 gradP.echange_espace_virtuel();
1095
1096 solveur_masse.appliquer(gradP);
1097 gradP.echange_espace_virtuel();
1098
1099 if (tab_vitesse.dimension_tot(0) == gradP.dimension_tot(0))
1100 tab_vitesse.ajoute(-dt,gradP);
1101 else
1102 {
1103 DoubleTab_parts partv(tab_vitesse);
1104 partv[0].ajoute(-dt,gradP);
1105 }
1106 tab_vitesse.echange_espace_virtuel();
1107 solveur_masse.corriger_solution(tab_vitesse, tab_vitesse);
1108
1109 Debog::verifier("Navier_Stokes_std::projeter, vitesse", tab_vitesse);
1110
1111// Verif ...
1112 divergence.calculer(tab_vitesse, secmem);
1113 secmem.echange_espace_virtuel();
1114
1115 bilan=mp_norme_vect(secmem);
1116 Cout << "Bilan de masse apres projection : " << bilan << finl;
1117 Cout << "------------- Projection OK---------------" << finl;
1118 Cout << "--------------------------------------------" << finl;
1119
1120 if( sub_type(solv_iteratif,solveur_pression_.valeur()) )
1121 {
1122 solv_iteratif& solv_iter=ref_cast(solv_iteratif,solveur_pression_.valeur());
1123 solv_iter.set_seuil(normal_seuil);
1124 }
1125 divergence.calculer(la_vitesse.valeurs(),divergence_U.valeurs());
1126 divergence_U.valeurs()=0.; //!< on remet les bons flux bords
1127
1128 }
1129 projection_initiale = 0;
1130}

2182void Navier_Stokes_std::uzawa(const DoubleTab& secmem,
2183 const Matrice_Base& A,
2184 SolveurSys& solveur,
2185 DoubleTab& U,
2186 DoubleTab& P)
2187{
2188// A U + Bt P = secmem
2189// B U = G
2190// On part de la pression courante et
2191// secmem = inertie + conv + sources + cl diff
2192// On part de P0 et U0 verifiant les C.L. et BU0=G
2193
2194// AU + Bt Cp = secmem
2195// BU = 0
2196
2197// On ecrit un GC sur B(A-1)Bt Cp = B(A-1)(secmem)
2198
2199 DoubleTrav Cu(U);
2200 DoubleTrav grad(U);
2201 DoubleTrav grad0(U);
2202 DoubleTrav resu(P);
2203 DoubleTrav residu(P);
2204 DoubleTrav Cp(P);
2205 double dold,dnew,alfa;
2206 double seuil=seuil_uzawa;
2207
2208// Cu = A(-1) secmem
2209 Cerr << "Begining Uzawa, secmem norm value : " << mp_norme_vect(secmem) << finl;
2210 P=0.;
2211 gradient->multvect(P, grad0);
2212 solveur.nommer("uzawa_solver");
2213 solveur.resoudre_systeme(A, secmem, U);
2214 solv_masse().corriger_solution(U,Cu); //!< pour les C.L. de Dirichlet!
2215
2216// residu=BCu
2217 divergence->multvect(U, resu);
2218
2219 residu.copy(resu);
2220 residu*=-1.;
2221
2222// Cp = -residu;
2223 Cp = residu;
2224 Cp*=-1;
2225 Cp.echange_espace_virtuel();
2226
2227// Carre de la norme
2228 dold = mp_norme_vect(residu);
2229 dold = dold * dold;
2230 dnew = dold;
2231
2232 double s=0;
2233 int niter=0;
2234 int nmax=Cp.size();
2235 Cerr << "Uzawa, initial residue : " << dnew << finl;
2236// seuil=std::max(seuil, dnew*1.e-12);
2237 while ( ( dnew > seuil ) && (niter++ < nmax) )
2238 {
2239 gradient->multvect(Cp, grad);
2240 grad-=grad0;
2241 grad*=-1;
2242 solveur.resoudre_systeme(A, grad, Cu);
2243 solv_masse().corriger_solution(Cu,U); //!< pour les C.L. de Dirichlet!
2244 divergence->multvect(Cu, resu);
2245 resu*=-1;
2246
2247 s = mp_prodscal(resu, Cp);
2248 alfa = dold/(s);
2249 P.ajoute(alfa,Cp);
2250 residu.ajoute(alfa,resu);
2251 U.ajoute(alfa,Cu);
2252 dnew = mp_norme_vect(residu);
2253 dnew = dnew * dnew;
2254 assert(dnew >= 0);
2255 Cp *= (dnew/dold);
2256 Cp -= residu;
2257 dold = dnew;
2258// Cerr << "Uzawa, Residu apres "
2259// << niter << " iterations = " << dnew << finl;
2260 }
2261
2262 if(dnew > seuil)
2263 {
2264 Cerr << "######## Uzawa, No convergence after : " << niter << " iterations\n";
2265 Cerr << "######## Uzawa, Residue : "<< dnew << "\n";
2266 exit();
2267 }
2268
2269 else if ((je_suis_maitre()))
2270 {
2271 Cerr << finl << "Uzawa, convergence reached after " << niter << " iterations" << finl;
2272 }
2273 {
2274// On verifie :
2275
2276 DoubleTrav R(resu);
2277 divergence->multvect(U, R);
2278 Cerr << "Ending Uzawa : mass residue : " << mp_norme_vect(R) <<finl;
2279
2280 gradient->multvect(P, grad);
2281 grad-=grad0;
2282 DoubleTrav F(secmem);
2283 DoubleTrav UU(U);
2284 F=secmem;
2285 F-=grad;
2286
2287 solveur.resoudre_systeme(A, F, UU);
2288 UU-=U;
2289 Cerr << "Ending Uzawa : Qdm residue : " << mp_norme_vect(UU)<<finl;
2290 }
