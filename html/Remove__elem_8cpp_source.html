89void Remove_elem::recreer_faces(Zone& zone, Faces& faces, IntTab& som_face) const
90{
91 IntTab& sommets=faces.les_sommets();
92 int nb_faces=sommets.dimension(0);
93 int nbs = (dimension==2) ? 2 : 4 ; //!< nombre de sommets par face
94 IntTab faces_recreees(nb_faces,nbs);
95
96 int ii=0;
97
98 for(int i=0; i<nb_faces; i++)
99 {
100 ArrOfInt ind(4);
101 ind[0]=sommets(i,0);
102 ind[1]=sommets(i,1);
103 ind[2]=(dimension==3) ? sommets(i,2) : -1;
104 ind[3]=(dimension==3) ? sommets(i,3) : -1;
105 ind.ordonne_array(); //!< ordonnancement des indices pour eviter les mauvaises surprises de numerotations d'indices
106
107 int trouve=0;
108
109 int j=0;
110 for( ; j<som_face.dimension(2); j++)
111 {
112 if(som_face(ind[3],0,j)==ind[2] && som_face(ind[3],1,j)==ind[1] && som_face(ind[3],2,j)==ind[0])
113 {
114 trouve=1;
115 break;
116 }
117 }
118
119 if(trouve==1) //!< on "supprime" la face en reinitialisant a -1 les indices
120 {
121 som_face(ind[3],0,j)=-1 ;
122 som_face(ind[3],1,j)=-1 ;
123 som_face(ind[3],2,j)=-1 ;
124 }
125 else
126 {
127 faces_recreees(ii,0) = sommets(i,0);
128 faces_recreees(ii,1) = sommets(i,1);
129 if(dimension==3)
130 {
131 faces_recreees(ii,2) = sommets(i,2);
132 faces_recreees(ii,3) = sommets(i,3);
133 }
134 ii++;
135 }
136 }
137
138 faces_recreees.resize(ii,nbs);
139
140 sommets.reset();
141 sommets.ref(faces_recreees);
142}

144void Remove_elem::creer_faces(Zone& zone, Faces& faces, IntTab& som_face) const
145{
146 faces.dimensionner(1);
147 IntTab& sommets=faces.les_sommets();
148 int nbsom=domaine().les_sommets().dimension(0);
149 int nbs = (dimension==2) ? 2 : 4 ; //!< nombre de sommets par face
150 IntTab faces_recreees(1,nbs);
151
152 int ii=0;
153
154 for(int i=0; i<nbsom; i++)
155 {
156 for(int j=0 ; j<som_face.dimension(2); j++)
157 {
158 if(som_face(i,0,j)!=-1)
159 {
160 faces_recreees.resize(ii+1,nbs);
161 faces_recreees(ii,0) = i;
162 faces_recreees(ii,1) = som_face(i,0,j);
163 if(dimension==3)
164 {
165 faces_recreees(ii,2) = som_face(i,1,j);
166 faces_recreees(ii,3) = som_face(i,2,j);
167 }
168 ii++;
169 }
170 }
171 }
172
173 faces.dimensionner(ii);
174 sommets.ref(faces_recreees);
175}

177void Remove_elem::remplir_liste(IntTab& som_face, int ind1, int ind2, int ind3, int ind4) const
178{
179 ArrOfInt ind(4);
180 ind[0]=ind1;
181 ind[1]=ind2;
182 ind[2]=ind3;
183 ind[3]=ind4;
184 ind.ordonne_array(); //!< ordonnancement des indices pour eviter les mauvaises surprises de numerotations d'indices (rencontrees prealablement)
185
186 int trouve=0;
187
188 int j=0;
189 for( ; j<som_face.dimension(2); j++)
190 {
191 if(som_face(ind[3],0,j)==ind[2] && som_face(ind[3],1,j)==ind[1] && som_face(ind[3],2,j)==ind[0])
192 {
193 trouve=1;
194 break;
195 }
196 }
197
198 if(trouve==1) //!< on "supprime" la face en reinitialisant a -1 les indices concernes
199 {
200 som_face(ind[3],0,j)=-1 ;
201 som_face(ind[3],1,j)=-1 ;
202 som_face(ind[3],2,j)=-1 ;
203 }
204 else //!< on "ajoute" la face en affectant les indices aux premieres cases "libres" (=-1)
205 {
206 for(j=0; j<som_face.dimension(2); j++)
207 {
208 if(som_face(ind[3],0,j)==-1) break;
209 }
210 som_face(ind[3],0,j)=ind[2] ;
211 som_face(ind[3],1,j)=ind[1] ;
212 som_face(ind[3],2,j)=ind[0] ;
213 }

216void Remove_elem::Remove_elem_(Zone& zone)
217{
218 if (zone.type_elem()->que_suis_je() == "Rectangle" || zone.type_elem()->que_suis_je() == "Hexaedre")
219 {
220
221 IntTab& les_elems=zone.les_elems();
222 int oldsz=les_elems.dimension(0);
223 ArrOfInt marq_remove(oldsz);
224 int nbsom=domaine().les_sommets().dimension(0);
225
226 int nbs = (dimension==2) ? 4 : 8 ; //!< nombre de sommets par element
227 IntTab new_elems(oldsz,nbs);
228
229 int nbfacesom = (dimension==2) ? 4 : 4*3 ; //!< nbre de faces connectees a un sommet
230 IntTab som_face(nbsom,3,nbfacesom);
231 som_face=-1;
232
233 if(f_ok)
234 {
235 DoubleTab xg(oldsz, dimension);
236 zone.type_elem()->calculer_centres_gravite(xg);
237 for(int i=0; i<oldsz; i++)
238 {
239 f.setVar(0,xg(i,0));
240 f.setVar(1,xg(i,1));
241 if(dimension==3) f.setVar(2,xg(i,2));
242// if(f.eval()) listelem.add(i);
243 if((int)(f.eval()+0.5)) marq_remove[i]=1; //listelem.add(i); //!< pour etre conforme a ce qui est fait dans DecoupeBord
244 }
245 }
246 else
247 {
248 for (int i=0; i<listelem.size(); i++)
249 {
250 marq_remove[listelem[i]]=1;
251 }
252
253 }
254
255 int j=0;
256 Cerr << "-> " << listelem.size() << " elements will be removed from the domain " << domaine().le_nom() << finl;
257 /* if (listelem.size()==0)
258 {
259 Cerr << "May be an error when applying Remove_elem : no elements found." << finl;
260 Process::exit();
261 } */
262 for(int i=0; i<oldsz; i++)
263 {
264 if(marq_remove[i]==0)
265 {
266 for(int k=0; k<nbs; k++) new_elems(j,k) = les_elems(i,k);
267 j++;
268 }
269 else
270 {
271 if(dimension==2)
272 {
273 int i0 = les_elems(i,0);
274 int i1 = les_elems(i,1);
275 int i2 = les_elems(i,2);
276 int i3 = les_elems(i,3);
277
278 remplir_liste(som_face,i0,i1,-1,-1) ;
279 remplir_liste(som_face,i0,i2,-1,-1) ;
280 remplir_liste(som_face,i1,i3,-1,-1) ;
281 remplir_liste(som_face,i2,i3,-1,-1) ;
282 }
283 else
284 {
285 int i0 = les_elems(i,0);
286 int i1 = les_elems(i,1);
287 int i2 = les_elems(i,2);
288 int i3 = les_elems(i,3);
289 int i4 = les_elems(i,4);
290 int i5 = les_elems(i,5);
291 int i6 = les_elems(i,6);
292 int i7 = les_elems(i,7);
293
294 remplir_liste(som_face,i0,i1,i2,i3) ;
295 remplir_liste(som_face,i0,i1,i4,i5) ;
296 remplir_liste(som_face,i0,i4,i2,i6) ;
297 remplir_liste(som_face,i1,i5,i3,i7) ;
298 remplir_liste(som_face,i2,i3,i6,i7) ;
299 remplir_liste(som_face,i4,i5,i6,i7) ;
300 }
301 }
302 }
303
304 new_elems.resize(j,nbs);
305 les_elems.ref(new_elems);
306
307// Reconstruction de l'octree
308 zone.invalide_octree();
309 zone.construit_octree();
310 zone.reordonner();
311
312 {
313 Cerr << " Regeneration of boundaries" << finl;
314 LIST_CURSEUR(Bord) curseur(zone.faces_bord());;;
315 while(curseur)
316 {
317 Faces& les_faces=curseur->faces();
318 if(dimension==2) les_faces.typer(Faces::segment_2D);
319 else les_faces.typer(Faces::quadrangle_3D);
320 recreer_faces(zone, les_faces, som_face);
321 ++curseur;
322 }
323 Cerr << " addition of a new boundary issued from removed elements" << finl;
324 Bord& new_bord=zone.faces_bord().add(Bord());
325 new_bord.nommer("newBord");
326 if(dimension==2) new_bord.typer_faces(Faces::segment_2D);
327 else new_bord.typer_faces(Faces::quadrangle_3D);
328 Faces& les_faces=new_bord.faces();
329 creer_faces(zone, les_faces, som_face);
330 }
331
332 {
333// Les Faces internes
334 Cerr << "Regeneration of internal faces" << finl;
335 LIST_CURSEUR(Faces_Interne) curseur(zone.faces_int());
336 while(curseur)
337 {
338 Faces& les_faces=curseur->faces();
339 if(dimension==2) les_faces.typer(Faces::segment_2D);
340 else les_faces.typer(Faces::quadrangle_3D);
341 recreer_faces(zone, les_faces, som_face);
342 ++curseur;
343 }
344 }
345
346 Cerr<<"END of Remove_elem..."<<finl;
347 Cerr<<" 1 NbElem="<<zone.les_elems().dimension(0)<<" NbNod="<<zone.nb_som()<<finl;
348 }
349
350 else
351
352 {
353 Cerr << "We do not yet know how to Remove_elem the "
354 << zone.type_elem()->que_suis_je() <<"s"<<finl;
355 exit();
356 }
357}
