39void Source_Fluide_Dilatable_VEF_Proto::ajouter_impl(const Equation_base& eqn,const DoubleVect& g,
40 const int dimension, const double rho_m,
41 const DoubleTab& tab_rho, DoubleTab& resu) const
42{
43 const int nb_faces = la_zone->nb_faces(), premiere_face_interne = la_zone->premiere_face_int();
44 const DoubleVect& volumes_entrelaces = la_zone->volumes_entrelaces();
45 const DoubleVect& porosite_face = la_zone->porosite_face();
46 const IntTab& face_voisins = la_zone->face_voisins();
47 const DoubleTab& face_normales = la_zone->face_normales(), xp = la_zone->xp(), xv = la_zone->xv();
48
49 if (eqn.discretisation().que_suis_je()=="VEF")
50 {
51// Boucle faces bord
52 for (int num_cl=0 ; num_cl<la_zone->nb_front_Cl() ; num_cl++)
53 {
54 const Cond_lim& la_cl = la_zone_Cl->les_conditions_limites(num_cl);
55 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
56 int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
57 if (sub_type(Neumann_sortie_libre,la_cl.valeur()))
58 {
59 for (int face=ndeb ; face<nfin ; face++)
60 {
61 int elem1 = face_voisins(face,0);
62 if (elem1==-1) elem1 = face_voisins(face,1);
63
64 for (int comp=0 ; comp< Objet_U::dimension ; comp++)
65 {
66 double delta_coord = (xv(face,comp) - xp(elem1,comp));
67 for (int i=0 ; i<dimension ; i++)
68 resu(face,i) += (tab_rho(face)-rho_m)*delta_coord*face_normales(face,i)*g(comp);
69 }
70 }
71 }
72 else if (sub_type(Periodique,la_cl.valeur()))
73 {
74 for (int face=premiere_face_interne ; face<nb_faces; face++)
75 {
76 int elem1 = face_voisins(face,0), elem2 = face_voisins(face,1);
77 for (int comp=0 ; comp<dimension ; comp++)
78 {
79 double delta_coord = (xp(elem2,comp) - xp(elem1,comp));
80 for (int i=0 ; i<dimension ; i++)
81 resu(face,i) += (tab_rho(face)-rho_m)*delta_coord*face_normales(face,i)*g(comp);
82 }
83 }
84 }
85 }
86// Boucle faces internes
87 for (int face=premiere_face_interne ; face<nb_faces; face++)
88 {
89 int elem1 = face_voisins(face,0), elem2 = face_voisins(face,1);
90 for (int comp=0 ; comp<dimension ; comp++)
91 {
92 double delta_coord = (xp(elem2,comp) - xp(elem1,comp));
93 for (int i=0 ; i<dimension ; i++)
94 resu(face,i) += (tab_rho(face)-rho_m)*delta_coord*face_normales(face,i)*g(comp);
95 }
96 }
97 }
98 else if (eqn.discretisation().que_suis_je()=="VEFPreP1B")
99 {
100// Boucle faces bord
101 for (int num_cl=0 ; num_cl<la_zone->nb_front_Cl() ; num_cl++)
102 {
103 const Cond_lim& la_cl = la_zone_Cl->les_conditions_limites(num_cl);
104 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
105 int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
106 if (sub_type(Neumann_sortie_libre,la_cl.valeur())||sub_type(Symetrie,la_cl.valeur())||sub_type(Periodique,la_cl.valeur()))
107 {
108 for (int face=ndeb ; face<nfin ; face++)
109 for (int comp=0 ; comp<dimension ; comp++)
110 resu(face,comp) += (tab_rho(face)-rho_m)*volumes_entrelaces(face)*porosite_face(face)*g(comp);
111 }
112 }
113// Boucle faces internes
114 for (int face=premiere_face_interne ; face<nb_faces; face++)
115 for (int comp=0 ; comp<dimension ; comp++)
116 resu(face,comp) += (tab_rho(face)-rho_m)*volumes_entrelaces(face)*porosite_face(face)*g(comp);
117 }
118 else
119 {
120 Cerr<<"La discretisation "<<eqn.discretisation().que_suis_je()<<" n'est pas reconnue dans Source_Fluide_Dilatable_VEF_Proto"<<finl;
121 Process::exit();
122 }
123}
