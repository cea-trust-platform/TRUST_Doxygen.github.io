78inline Type_Cl_VDF Iterateur_VDF_base::type_cl(const Cond_lim& la_cl) const
79{
80 Type_Cl_VDF retour=nouvelle_Cl_VDF;
81 if ( sub_type(Symetrie,la_cl.valeur()) ) retour=symetrie;
82 else if ( sub_type(Neumann_sortie_libre, la_cl.valeur()) ) retour=sortie_libre;
83 else if ( sub_type(Dirichlet_entree_fluide, la_cl.valeur()) ) retour=entree_fluide;
84 else if ( sub_type(Dirichlet_paroi_fixe, la_cl.valeur()) ) retour=paroi_fixe;
85 else if ( sub_type(Dirichlet_paroi_defilante, la_cl.valeur()) ) retour=paroi_defilante;
86 else if ( sub_type(Neumann_paroi_adiabatique, la_cl.valeur()) ) retour=paroi_adiabatique;
87 else if ( sub_type(Neumann_paroi, la_cl.valeur()) ) retour=paroi;
88 else if ( sub_type(Echange_externe_impose, la_cl.valeur()) ) retour=echange_externe_impose;
89 else if ( sub_type(Echange_global_impose, la_cl.valeur()) ) retour=echange_global_impose;
90 else if ( sub_type(Periodique, la_cl.valeur()) ) retour=periodique;
91 return retour;
92}
