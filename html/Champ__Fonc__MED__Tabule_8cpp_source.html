35void Champ_Fonc_MED_Tabule::mettre_a_jour(double le_temps)
36{
37
38// temps1_ temps2_ tel que temps [temps1_ , temps2_[
39 if (est_egal(le_temps,temps_calc_))
40 return;
41
42 if ((temps2_<-1e8)||(le_temps<=temps_calc_))
43 {
44// initialisation de -1e9 a T[0] on met la meme chose
45 lire(temps_sauv_[0]);
46 tab1_=valeurs();
47 tab2_=valeurs();
48 temps2_=temps_sauv_[0];
49 }
50 if (le_temps>=temps2_)
51 {
52 int i,nbt=temps_sauv_.size_array();
53 int trouve=0;
54 for (i=0; i<nbt-1; i++)
55 {
56 if ((le_temps>=temps_sauv_[i])&&(le_temps<=temps_sauv_[i+1]))
57 {
58 trouve=1;
59 temps1_=temps_sauv_[i];
60 lire(temps1_);
61 tab1_=valeurs();
62 temps2_=temps_sauv_[i+1];
63 lire(temps2_);
64 tab2_=valeurs();
65 }
66 }
67 if (trouve==0)
68 {
69 temps1_=temps_sauv_[nbt-1];
70 assert(le_temps>=temps_sauv_[nbt-1]);
71 lire(temps1_);
72 tab1_=valeurs();
73 tab2_=tab1_;
74 temps2_=DMAXFLOAT;
75 }
76 }
77
78 if ((le_temps>=temps1_) && (le_temps<temps2_))
79 {
80 valeurs()=tab2_;
81 valeurs()-=tab1_;
82 valeurs()*=(le_temps-temps1_)/(temps2_-temps1_);
83 valeurs()+=tab1_;
84 temps_calc_=le_temps;
85// Cerr<<temps_calc_<<" YYYYYY "<<valeurs()[0]<<" "<< tab2_[0]<< " "<<tab1_[0]<<finl;
86 }
87 else
88 {
89 Cerr<<"internal error"<<finl;
90 exit();
91 }
92
93}
