39Entree& Terme_Source_Solide_SWIFT_VDF::readOn(Entree& is )
40{
41 int i,compteur = 0;
42 Motcle motlu;
43 Cerr << " ######### Warning ! #########" << finl;
44 Cerr << " For the moment, this forcing term is only available" << finl;
45 Cerr << " for recovering identical solid meshes." << finl;
46 Cerr << " If otherwise, it will produce a beautiful core-dump..." << finl;
47
48 for(i=0 ; i<3 ; i++)
49 {
50 is >> motlu;
51
52 if(motlu == "TAU")
53 {
54 compteur++;
55 is >> tau;
56 if(tau==0)
57 {
58 Cerr << "TAU can not be equal to 0." << finl;
59 exit();
60 }
61 Cerr << "Characteristic time TAU is : " << tau << finl;
62 }
63 else if(motlu == "pb_swift")
64 {
65 compteur++;
66 Nom swift;
67 is >> swift;
68 pb_swift=ref_cast(Probleme_base, Interprete::objet(swift));
69 Cerr << "pb_swift is: " << pb_swift.le_nom() << finl;
70 }
71 else if(motlu == "pb_corse")
72 {
73 compteur++;
74 Nom corse;
75 is >> corse;
76 pb_corse=ref_cast(Probleme_base, Interprete::objet(corse));
77 Cerr << "pb_coarse is: " << pb_corse.le_nom() << finl;
78 }
79 else
80 {
81 Cerr << "An argument of Terme_Source_Solide_SWIFT_VDF is not correct !" << finl;
82 exit();
83 }
84 }
85 if(compteur != 3)
86 {
87 Cerr << "The forcing term for the mean solid temperature field with the SWIFT method needs 3 arguments" << finl;
88 Cerr << "Syntax : " << finl;
89 Cerr << "Solide_SWIFT SWIFT pb_sol_SWIFT COARSE pb_sol_coarse TAU tau " << finl;
90 exit();
91 }
92
93
94// On recupere les equations de conduction des deux domaines (SWIFT et grossier)
95 int flag=0;
96 for(i=0; i<pb_swift->nombre_d_equations(); i++)
97 {
98 if(sub_type(Conduction,pb_swift->equation(i)))
99 {
100 eq_swift = ref_cast(Conduction,pb_swift->equation(i));
101 flag=1;
102 }
103 }
104 if (flag==0)
105 {
106 Cerr << "Error : The Conduction equation was not found...!" << finl;
107 Cerr << "for the SWIFT solid forcing term (in pb_swift)." << finl;
108 exit();
109 }
110
111 flag=0;
112 for(i=0; i<pb_corse->nombre_d_equations(); i++)
113 {
114 if(sub_type(Conduction,pb_corse->equation(i)))
115 {
116 eq_corse = ref_cast(Conduction,pb_corse->equation(i));
117 flag=1;
118 }
119 }
120 if (flag==0)
121 {
122 Cerr << "Error : The Conduction equation was not found...!" << finl;
123 Cerr << "for the SWIFT solid forcing term (in pb_coarse)." << finl;
124 exit();
125 }
126
127 init_calcul_moyenne(eq_swift.valeur(),Y_swift,corresp_swift,compt_swift);
128 init_calcul_moyenne(eq_corse.valeur(),Y_corse,corresp_corse,compt_corse);
129
130// Calcul du tableau de correspondance entre un element de Tmoy_swift et l'element de Tmoy_corse correspondant
131// pour calculer le terme de forcage.
132 correspondance_SWIFT_coarse();
133
134 return is;
135
136}

165void Terme_Source_Solide_SWIFT_VDF::init_calcul_moyenne(const Conduction& my_eqn, DoubleVect& Y, IntVect& corresp, IntVect& compt)
166{
167 int num_elem,j,indic,trouve;
168 double y;
169
170 const Zone_dis_base& zdisbase = my_eqn.inconnue().zone_dis_base();
171 const Zone_VDF& zone_VDF = ref_cast(Zone_VDF, zdisbase);
172 const DoubleTab& xp = zone_VDF.xp();
173
174 int nb_elems = zone_VDF.zone().nb_elem();
175 int nb_faces_x = zone_VDF.nb_faces_X();
176 int nb_faces_z = zone_VDF.nb_faces_Z();
177
178 int Nx = nb_elems/(nb_faces_x-nb_elems);
179 int Nz = nb_elems/(nb_faces_z-nb_elems);
180 int Ny = nb_elems/Nx/Nz;
181
182 Y.resize(Ny);
183 compt.resize(Ny);
184 corresp.resize(nb_elems);
185
186 Y = -100.;
187 compt = 0;
188 corresp = -1;
189
190 j=0;
191 indic = 0;
192 for (num_elem=0; num_elem<nb_elems; num_elem++)
193 {
194 y = xp(num_elem,1);
195 trouve = 0;
196 for (j=0; j<indic+1; j++)
197 {
198 if(std::fabs(y-Y[j])<=1e-8)
199 {
200 corresp[num_elem] = j;
201 compt[j] ++;
202 j=indic+1;
203 trouve = 1;
204 break;
205 }
206 }
207 if (trouve==0)
208 {
209 corresp[num_elem] = indic;
210 Y[indic]=y;
211 compt[indic] ++;
212 indic++;
213 }
214 }
215}

226void Terme_Source_Solide_SWIFT_VDF::correspondance_SWIFT_coarse( )
227{
228 int t_S = Y_swift.size();
229 int t_C = Y_corse.size();
230 corresp_SC.resize(t_S);
231
232 for(int j=0; j<t_S; j++)
233 {
234 int burk=1;
235 for(int i=0; i<t_C; i++)
236 {
237 if(std::fabs(Y_swift[j]-Y_corse[i])<=1e-8)
238 {
239 burk=0;
240 corresp_SC[j]=i;
241 }
242 }
243 if(burk)
244 {
245 Cerr << "## ACHTUNG !!! On n'a pas trouve l'equivalence entre" << finl;
246 Cerr << "## les deux domaines au niveau des Y !" << finl;
247 }
248 }
249}

260void Terme_Source_Solide_SWIFT_VDF::calcul_moyenne(const Conduction& my_eqn, DoubleVect& T_moy, const IntVect& corresp, const IntVect& compt) const
261{
262 const Zone_dis_base& zdisbase = my_eqn.inconnue().zone_dis_base();
263 const Zone_VDF& zone_VDF = ref_cast(Zone_VDF, zdisbase);
264 const DoubleTab& Temp = my_eqn.inconnue().valeurs();
265
266 int nb_elems = zone_VDF.zone().nb_elem();
267 int nb_faces_x = zone_VDF.nb_faces_X();
268 int nb_faces_z = zone_VDF.nb_faces_Z();
269
270 int Nx = nb_elems/(nb_faces_x-nb_elems);
271 int Nz = nb_elems/(nb_faces_z-nb_elems);
272 int Ny = nb_elems/Nx/Nz;
273
274
275 int j=0;
276 T_moy.resize(Ny);
277 T_moy = 0.;
278
279 for (j=0; j<nb_elems; j++) T_moy[corresp[j]] += Temp[j];
280 for (j=0; j<Ny; j++) T_moy[j] /= compt[j];
281}
