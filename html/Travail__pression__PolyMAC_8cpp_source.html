38void Travail_pression_PolyMAC::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
39{
40 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, equation().zone_dis().valeur());
41 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins();
42 const DoubleTab& inco = equation().inconnue().valeurs();
43 int i, j, e, eb, ne = zone.nb_elem(), f,n, N = inco.line_size(), m, M = ref_cast(Pb_Multiphase, equation().probleme()).eq_qdm.pression().valeurs().line_size();
44
45 for (auto &&n_m : matrices)
46 if (n_m.first == "pression" || (n_m.first == "alpha" && !semi_impl.count("alpha")) || n_m.first == "vitesse")
47 {
48 Matrice_Morse& mat = *n_m.second, mat2;
49 IntTrav sten(0, 2);
50 sten.set_smart_resize(1);
51 if (n_m.first == "pression") /* pression : dependance locale, implicite */
52 for (e = 0; e < ne; e++)
53 for (n = 0, m = 0; n < N; n++, m += (M > 1)) sten.append_line(N * e + n, M * e + m);
54 else if (n_m.first == "vitesse")
55 for (e = 0; e < ne; e++)
56 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
57 for (n = 0; n < N; n++) sten.append_line(N * e + n, N * f + n);
58 else for (e = 0; e < ne; e++)
59 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
60 for (j = 0; j < 2 && (eb = f_e(f, j)) >= 0; j++)
61 for (n = 0; n < N; n++) sten.append_line(N * e + n, N * eb + n);
62 tableau_trier_retirer_doublons(sten);
63 Matrix_tools::allocate_morse_matrix(inco.size_totale(), equation().probleme().get_champ(n_m.first).valeurs().size_totale(), sten, mat2);
64 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
65 }
66}

68void Travail_pression_PolyMAC::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
69{
70 const Pb_Multiphase& pbm = ref_cast(Pb_Multiphase, equation().probleme());
71 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, equation().zone_dis().valeur());
72 const DoubleVect& pe = zone.porosite_elem(), &pf = zone.porosite_face(), &fs = zone.face_surfaces(), &ve = zone.volumes();
73 const Champ_Inc_base& ch_a = pbm.eq_masse.inconnue().valeur(), &ch_v = pbm.eq_qdm.inconnue().valeur(), &ch_p = pbm.eq_qdm.pression().valeur();
74 /* trois tableaux de alpha : present / passe et champ convecte (peut etre semi-implicite) */
75 const DoubleTab& alpha = ch_a.valeurs(), &c_alpha = semi_impl.count("alpha") ? semi_impl.at("alpha") : alpha, &p_alpha = ch_a.passe(), &press = ch_p.valeurs(), &vit = ch_v.valeurs();
76 const IntTab& fcl = ref_cast(Champ_Elem_PolyMAC, ch_a).fcl(), &f_e = zone.face_voisins();
77 DoubleTab b_alpha = ch_a.valeur_aux_bords();
78 Matrice_Morse *Mp = matrices.count("pression") ? matrices.at("pression") : NULL,
79 *Ma = matrices.count("alpha") && !semi_impl.count("alpha") ? matrices.at("alpha") : NULL,
80 *Mv = matrices.count("vitesse") ? matrices.at("vitesse") : NULL;
81
82 int i, j, e, eb, f, n, N = alpha.line_size(), m, M = press.line_size();
83 const Op_Conv_EF_Stab_PolyMAC_Elem *op_conv = sub_type(Op_Conv_EF_Stab_PolyMAC_Elem, equation().operateur(1).l_op_base()) ? &ref_cast(Op_Conv_EF_Stab_PolyMAC_Elem, equation().operateur(1).l_op_base()) : NULL;
84 double dt = equation().schema_temps().pas_de_temps(), alp = op_conv ? op_conv->alpha : 1; /* meme decentrement que l'operateur de convection si il existe, amont sinon */
85
86// partie -p d alpha_k / dt et ses derivees
87 for (e = 0; e < zone.nb_elem(); e++)
88 {
89 for (n = 0, m = 0; n < N; n++, m += (M > 1)) secmem(e, n) -= pe(e) * ve(e) * press(e, m) * (alpha(e, n) - p_alpha(e, n)) / dt;
90 if (Mp)
91 for (n = 0, m = 0; n < N; n++, m += (M > 1)) (*Mp)(N * e + n, M * e + m) += pe(e) * ve(e) * (alpha(e, n) - p_alpha(e, n)) / dt;
92 if (Ma)
93 for (n = 0, m = 0; n < N; n++, m += (M > 1)) (*Ma)(N * e + n, N * e + n) += pe(e) * ve(e) * press(e, m) / dt;
94 }
95
96// partie -p div (alpha_k v_k)
97 DoubleTrav dv_flux(N), dc_flux(2, N); //derivees du flux convectif a la face par rapport a la vitesse / au champ convecte amont / aval
98 /* convection aux faces internes (fcl(f, 0) == 0), de Neumann_val_ext ou de Dirichlet */
99 for (f = 0; f < zone.nb_faces(); f++)
100 if (!fcl(f, 0) || (fcl(f, 0) > 4 && fcl(f, 0) < 7))
101 {
102 for (dv_flux = 0, dc_flux = 0, i = 0; i < 2; i++)
103 for (e = f_e(f, i), n = 0; n < N; n++)
104 {
105 double v = vit(f, n) ? vit(f, n) : DBL_MIN, fac = pf(f) * fs(f) * (1. + (v * (i ? -1 : 1) > 0 ? 1. : -1) * alp) / 2;
106 dv_flux(n) += fac * (e >= 0 ? c_alpha(e, n) : b_alpha(f, n)); //f est reelle -> indice trivial dans b_alpha
107 dc_flux(i, n) = e >= 0 ? fac * vit(f, n) : 0;
108 }
109
110// second membre
111 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
112 if (e < zone.nb_elem())
113 for (n = 0, m = 0; n < N; n++, m += (M > 1))
114 secmem(e, n) -= (i ? -1 : 1) * press(e, m) * dv_flux(n) * vit(f, n);
115// derivees : vitesse
116 if (Mv)
117 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
118 if (e < zone.nb_elem())
119 for (n = 0, m = 0; n < N; n++, m += (M > 1))
120 (*Mv)(N * e + n, N * f + n) += (i ? -1 : 1) * press(e, m) * dv_flux(n);
121// derivees : pression
122 if (Mp)
123 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
124 if (e < zone.nb_elem())
125 for (n = 0, m = 0; n < N; n++, m += (M > 1))
126 (*Mp)(N * e + n, M * e + m) += (i ? -1 : 1) * dv_flux(n) * vit(f, n);
127// derivees : alpha
128 if (Ma)
129 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
130 if (e < zone.nb_elem())
131 for (j = 0; j < 2 && (eb = f_e(f, j)) >= 0; j++)
132 for (n = 0, m = 0; n < N; n++, m += (M > 1)) (*Ma)(N * e + n, N * eb + n) += (i ? -1 : 1) * press(e, m) * dc_flux(j, n);
133 }
