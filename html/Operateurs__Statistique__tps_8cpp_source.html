83int Operateurs_Statistique_tps::sauvegarder(Sortie& os) const
84{
85// en mode ecriture special seul le maitre ecrit l'entete
86 int a_faire,special;
87 EcritureLectureSpecial::is_ecriture_special(special,a_faire);
88
89 if (a_faire)
90 {
91 Nom mon_ident(que_suis_je());
92 mon_ident += mon_post->probleme().domaine().le_nom();
93 double temps = mon_post->probleme().schema_temps().temps_courant();
94 mon_ident += Nom(temps,"%e");
95 os << mon_ident << finl;
96 os << que_suis_je() << finl;
97 os << size() << finl;
98 os << Nom(tstat_deb_,"%e") << finl;
99 os << Nom(tstat_dernier_calcul_,"%e") << finl ;
100 }
101 CONST_LIST_CURSEUR(Operateur_Statistique_tps) curseur=*this;
102 int bytes=0;
103 while(curseur)
104 {
105 bytes += curseur->sauvegarder(os);
106 ++curseur;
107 }
108 if (a_faire) os.flush();
109 return bytes;
110}

113int Operateurs_Statistique_tps::reprendre(Entree& is)
114{
115// Cerr << "Operateurs_Statistique_tps::reprendre" << finl;
116 if (mon_post.non_nul())
117 {
118 Nom bidon;
119 is >> bidon;
120 if (bidon=="fin")
121 {
122// Ce test evite un beau segmentation fault a la lecture
123// du deuxieme bidon lors d'une sauvegarde/reprise au format binaire
124 Cerr << "The end of the restarting file is reached." << finl;
125 Cerr << "This file does not contain statistics." << finl;
126 Cerr << "The restarting time tinit must therefore be lower" << finl;
127 Cerr << "than the statistics begining time t_deb." << finl;
128 exit();
129 }
130 is >> bidon;
131 int n;
132 is >> n;
133 double tstat_deb_sauv,temps_derniere_mise_a_jour_stats;
134 is >> tstat_deb_sauv;
135 is >> temps_derniere_mise_a_jour_stats;
136// Cerr << "temps_derniere_mise_a_jour_stats" << temps_derniere_mise_a_jour_stats << finl;
137 double tinit = mon_post->probleme().schema_temps().temps_courant();
138
139// Plusieurs cas possibles:
140 if (inf_strict(tinit,temps_derniere_mise_a_jour_stats,1.e-5))
141 {
142 Cerr << "The initial time " << tinit << " of the restarting calculation is lower" << finl;
143 Cerr << "than the last statistics update time " << temps_derniere_mise_a_jour_stats << finl;
144 Cerr << "One is not able to treat this case." << finl;
145 exit();
146 }
147 else if (!est_egal(tstat_deb_sauv,tstat_deb_))
148 {
149// t_deb est modifie : on refait une statistique sans reprendre dans certains cas
150 if (inf_strict(tstat_deb_,tinit,1.e-5))
151 {
152 Cerr << "t_deb has been modified to carry out a new statistics calculation without restarting" << finl;
153 Cerr << "but t_deb is lower than tinit : it is not possible." << finl;
154 exit();
155 }
156 Cerr << "Statistics are not restarted and therefore the statistics calculation" << finl;
157 Cerr << "will restart a t_deb =" << tstat_deb_ << finl;
158 LIST_CURSEUR(Operateur_Statistique_tps) curseur=*this;
159 Nom bidon2;
160 double dbidon;
161 DoubleTab tab_bidon;
162 while(curseur) //!< On saute les champs
163 {
164 is >> bidon2 >> bidon2;
165 is >> dbidon;
166 tab_bidon.reset(); //!< sinon erreur sur la taille dans lit()
167 tab_bidon.jump(is);
168 ++curseur;
169 }
170 }
171 else //!< tinit=>temps_derniere_mise_a_jour_stats : on fait la reprise
172 {
173 Nom bidon2;
174 LIST_CURSEUR(Operateur_Statistique_tps) curseur=*this;
175 while(curseur)
176 {
177 is >> bidon2 >> bidon2; //!< On saute l'identificateur et le type des champs
178 curseur->reprendre(is);
179 ++curseur;
180 }
181// On modifie l'attribut tstat_deb_ et l'attribut t_debut_ des champs
182// pour tenir compte de la reprise
183 tstat_deb_ = tstat_deb_sauv;
184 curseur=*this;
185 while(curseur)
186 {
187 curseur->fixer_tstat_deb(tstat_deb_,temps_derniere_mise_a_jour_stats);
188 ++curseur;
189 }
190 }
191 }
192 else //!< lecture pour sauter le bloc
193 {
194 int n;
195 is >> n;
196 double tstat_deb_sauv,temps_derniere_mise_a_jour_stats;
197 is >> tstat_deb_sauv;
198 is >> temps_derniere_mise_a_jour_stats;
199 Nom bidon;
200 double dbidon;
201 DoubleTab tab_bidon;
202 for (int i=0; i<n; i++)
203 {
204 is >> bidon >> bidon;
205 is >> dbidon;
206 tab_bidon.reset(); //!< sinon erreur sur la taille dans lit()
207 tab_bidon.jump(is);
208 }
209 }
210 return 1;
211}
