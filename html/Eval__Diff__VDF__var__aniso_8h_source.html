27//! class Eval_Diff_VDF_var_aniso<br>Cette classe represente un evaluateur de flux diffusif<br>avec un coefficient de diffusivite qui n'est pas constant en espace.
28/** \sa Evaluateur_VDF */
29class Eval_Diff_VDF_var_aniso : public Eval_Diff_VDF
30{
31public:
32 inline void associer(const Champ_base& ) override;
33 inline void mettre_a_jour() override;
34 inline const Champ_base& get_diffusivite() const override;
35
36// Methods used by the flux computation in template class:
37 inline double nu_1_impl(int i, int compo) const { return dt_diffusivite(i,compo); }
38 inline double nu_2_impl(int i, int compo) const { return dt_diffusivite(i,compo); }
39 inline double compute_heq_impl(double d0, int i, double d1, int j, int compo) const
40 {
41 return 1./(d0/dt_diffusivite(i,compo) + d1/dt_diffusivite(j,compo));
42 }
43
44// Methods used in F5 (var face aniso)
45 inline double nu_1_impl_face(int i, int j, int compo) const
46 {
47 return 0.5*(dt_diffusivite(i,compo)+dt_diffusivite(j,compo));
48 }
49
50 inline double nu_2_impl_face(int i, int j, int k, int l, int compo) const
51 {
52 return 0.25*(dt_diffusivite(i,compo)+dt_diffusivite(j,compo)+
53 dt_diffusivite(k,compo)+dt_diffusivite(l,compo));
54 }
55
56protected:
57 REF(Champ_base) diffusivite_;
58 DoubleTab dt_diffusivite;
59};

76{
77// ONLY AVAILABLE FOR TRUST : conduction problem
78// TODO : generalize ME
79 if (diffu.le_nom() == "conductivite" && diffu.nb_comp() != Objet_U::dimension)

85 }
