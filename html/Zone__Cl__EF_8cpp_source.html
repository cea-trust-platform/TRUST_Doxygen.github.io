110void construit_connectivite_sommet(int type_cl,Static_Int_Lists& som_face_bord,const Conds_lim& les_conditions_limites_,const Zone_EF& zone_EF)
111{
112 if (type_cl!=1) Process::exit();
113 int compt=0;
114 int nb_faces_tot=zone_EF.nb_faces_tot();
115 ArrOfInt face_bords(nb_faces_tot);
116
117 for(int i=0; i<les_conditions_limites_.size(); i++)
118 {
119 const Cond_lim_base& la_cl=les_conditions_limites_[i].valeur();
120 const Front_VF& le_bord= zone_EF.front_VF(i);
121 int num2 = le_bord.nb_faces_tot();
122
123 if ( (sub_type(Symetrie,la_cl)))
124 {
125 for (int ind_face=0; ind_face<num2; ind_face++)
126 {
127 int face=le_bord.num_face(ind_face);
128 face_bords[compt++]=face;
129 }
130 }
131 }
132 face_bords.resize_array(compt);
133 ArrOfInt is_sommet_sur_bord(zone_EF.nb_som_tot());
134 const IntTab& face_sommets=zone_EF.face_sommets();
135 int nb_som_face=face_sommets.dimension(1);
136 for (int fac=0; fac<compt; fac++)
137 {
138 int face=face_bords[fac];
139 for (int som=0; som<nb_som_face; som++)
140 {
141 int sommet=face_sommets(face,som);
142 is_sommet_sur_bord[sommet]++;
143 }
144 }
145
146// On exploite is_sommet_sur_bord pour fixer la taille des List que va repertorier som_face_bord
147// -som_face_bord contient nb_som_tot List
148// -chaque List est dimensionnee par le nombre de faces de bord (portant une CL de Dirichlet)
149// connectees au sommet considere
150// -une List donnee pour un sommet repertorie le numero des faces qui lui sont connectees
151
152 som_face_bord.set_list_sizes(is_sommet_sur_bord);
153 is_sommet_sur_bord=0;
154 for (int fac=0; fac<compt; fac++)
155 {
156 int face=face_bords[fac];
157 for (int som=0; som<nb_som_face; som++)
158 {
159 int sommet=face_sommets(face,som);
160 int n=(is_sommet_sur_bord[sommet])++;
161 som_face_bord.set_value(sommet,n,face);
162
163 }
164 }

171 const Zone& z = la_zone_EF.zone();
172

175 int nb_som_tot=z.nb_som_tot();
176 type_sommet_.resize_array(z.nb_som_tot());
177 type_sommet_=-1;
178 IntTab titi(nb_som_tot);
179
180 for(int i=0; i<les_conditions_limites_.size(); i++)
181 {
182 Cond_lim_base& la_cl=les_conditions_limites_[i].valeur();
183 const Front_VF& le_bord= la_zone_EF.front_VF(i);
184 int num2 = le_bord.nb_faces_tot();
185
186 if ( (sub_type(Dirichlet,la_cl))|| (sub_type(Dirichlet_homogene,la_cl)) )
187 {
188 for (int ind_face=0; ind_face<num2; ind_face++)
189 {
190 int face=le_bord.num_face(ind_face);
191 for (int s=0; s<nb_som_face; s++)
192 {
193 int som=faces_sommets(face,s);
194// Si on a Dirichlet_paroi_fixe_iso_Genepi2 on ne prend pas en compte le 0 pour la moyenne
195 if (!sub_type(Dirichlet_paroi_fixe_iso_Genepi2,la_cl))
196 titi(som)++;
197 if ((type_sommet_[som]!=1)&& (type_sommet_[som]!=3))
198 type_sommet_[som]=2;
199 else
200 type_sommet_[som]=3;
201 }
202 }
203 }
204 else if ( (sub_type(Symetrie,la_cl)))
205 {
206 for (int ind_face=0; ind_face<num2; ind_face++)
207 {
208 int face=le_bord.num_face(ind_face);
209 for (int s=0; s<nb_som_face; s++)
210 {
211 int som=faces_sommets(face,s);
212 if (type_sommet_[som]<=1)
213 type_sommet_[som]=1;
214 else
215 type_sommet_[som]=3;
216 }
217 }
218 }
219 else if ( (sub_type(Neumann,la_cl))|| (sub_type(Neumann_homogene,la_cl)) )
220 {
221 for (int ind_face=0; ind_face<num2; ind_face++)
222 {
223 int face=le_bord.num_face(ind_face);
224 for (int s=0; s<nb_som_face; s++)
225 {
226 int som=faces_sommets(face,s);
227 if (type_sommet_[som]<0)
228 type_sommet_[som]=0;
229 }
230 }
231 }
232 else
233 {
234 Cerr<<__FILE__<<":" <<(int)__LINE__<<" non code pour cette cl "<<la_cl.que_suis_je()<<finl;
235 }
236
237 }
238// si on a un dirichet on stocke 2*nb participant + 1 si sommet aussi de sym
239 for (int som=0; som<nb_som_tot; som++)
240 {
241 if (type_sommet_[som]>1)
242 {
243// assert(titi(som)>0);
244 type_sommet_[som]+=2*(titi(som));
245 }
246 }
247// On cree la connectivite sommet -> face de bord symetrie
248 if (equation().inconnue().valeur().nature_du_champ()==vectoriel)
249 {
250 equation().probleme().discretisation().discretiser_champ("VITESSE",la_zone_EF,"normales_nodales","1",dimension,0.,normales_symetrie_);
251 equation().probleme().discretisation().discretiser_champ("CHAMP_SOMMETS",la_zone_EF,"normales_nodales_bis","1",dimension,0., normales_symetrie_bis_);
252 Static_Int_Lists sommet_face_symetrie;
253 int type_cl=1;
254 construit_connectivite_sommet(type_cl,sommet_face_symetrie,les_conditions_limites_,la_zone_EF);
255// sommet_face_symetrie contient le nombre de face de symetrie associe a chaque sommet
256 const DoubleTab& face_normales = la_zone_EF.face_normales();
257 ArrOfDouble n(dimension),t1(dimension),t2(dimension);
258 for (int som=0; som<nb_som_tot; som++)
259 {
260 int nbf= sommet_face_symetrie.get_list_size(som);
261 if (nbf>0)
262// if ( type_sommet_(som)==1)
263 {
264 n=0;
265 t1=0;
266 t2=0;
267// on determine la normale sommet
268// int nbf= sommet_face_symetrie.get_list_size(som);
269 for (int f=0; f<nbf; f++)
270 {
271 int face=sommet_face_symetrie(som,f);
272 for (int d=0; d<dimension; d++)
273 n[d]+=face_normales(face,d);
274 }
275 n/=nbf;
276
277 double norm_n=norme_array(n);
278 n/=norm_n;
279 for (int d=0; d<dimension; d++)
280 normales_symetrie_.valeur().valeurs()(som,d)=n[d];
281// Cerr<<som<<" on doit annuler une premiere direction "<<n(0) << " " <<n(1)<<" "<<n(dimension==3?2:1)<<finl;
282
283 for (int f=0; f<nbf; f++)
284 {
285 int face=sommet_face_symetrie(som,f);
286 double prod=0;
287 for (int d=0; d<dimension; d++)
288 prod+=face_normales(face,d)*n[d];
289
290 double s=0;
291// double v = 0;
292
293 for (int d=0; d<dimension; d++)
294 {
295 t1[d]=face_normales(face,d)-n[d]*prod;
296 s+=face_normales(face,d)*face_normales(face,d);
297// v+=t1[d]*n[d];
298 }
299
300// Cerr<<" vv"<< v<<" "<<norme_array(t1)<<" "<<norm_n<<finl;
301 if (norme_array(t1)>(1e-4*sqrt(s)))
302 {
303
304// facilite le debugage
305 if (std::fabs(min_array(t1))>max_array(t1))
306 t1*=-1;
307 t1/=norme_array(t1);
308
309
310// Cerr<<som<<" on doit annuler une deuxieme direction "<<t1(0) << " " <<t1(1)<<" "<<t1(dimension==3?2:1)<<finl;
311 f=nbf;
312 for (int d=0; d<dimension; d++)
313 normales_symetrie_bis_.valeur().valeurs()(som,d)=t1[d];
314// assert(v==0);
315 }
316
317 }
318 for (int f=0; f<nbf; f++)
319 {
320 int face=sommet_face_symetrie(som,f);
321 double prod=0,prod1=0,s=0;
322 for (int d=0; d<dimension; d++)
323 {
324 prod+=face_normales(face,d)*n[d];
325
326 prod1+=face_normales(face,d)*t1[d];
327 s+=face_normales(face,d)*face_normales(face,d);
328
329 }
330 for (int d=0; d<dimension; d++)
331 t2[d]=face_normales(face,d)-n[d]*prod-t1[d]*prod1;
332
333 if (norme_array(t2)>(1e-4*sqrt(s)))
334 {
335// facilite le debugage
336 if (std::fabs(min_array(t2))>max_array(t2))
337 t2*=-1;
338 t2/=norme_array(t2);
339 Cerr<<face<<" "<<nbf<<" sommet "<<som<<" "<<norme_array(t2)/s<<" on doit annuler une troiseme direction"<<t2[0] << " " <<t2[1]<<" "<<t2[2]<<finl;
340 Cerr<<som<<" "<<t1[0] << " " <<t1[1]<<" "<<t1[2]<<finl;
341 Cerr<<som<<" "<<n[0] << " " <<n[1]<<" "<<n[2]<<finl;
342 f=nbf;
343 if (!normales_symetrie_ter_.non_nul())
344 equation().probleme().discretisation().discretiser_champ("CHAMP_SOMMETS",la_zone_EF,"normales_nodales_bis","1",dimension,0., normales_symetrie_ter_);
345 for (int d=0; d<dimension; d++)
346 normales_symetrie_ter_.valeur().valeurs()(som,d)=t2[d];
347// exit();
348 }
349 }
350 }
351 }
352 normales_symetrie_.valeur().valeurs().echange_espace_virtuel();
353 normales_symetrie_bis_.valeur().valeurs().echange_espace_virtuel();
354// exit();
355 }

484// Description :
485// On transforme la_matrice et le secmem pour avoir un secmem normal aux bords , plus la matrice pour assurer que la solution sera correcte
486void Zone_Cl_EF::imposer_symetrie_matrice_secmem(Matrice_Morse& la_matrice, DoubleTab& secmem) const
487{
488// return;
489 assert(secmem.dimension(1)==dimension);
490 const Zone& z = zone_dis().zone();
491 int nb_som=z.nb_som();
492 assert(secmem.dimension(0)==nb_som);
493 int nb_comp=secmem.dimension(1);
494 const DoubleTab& n =normales_symetrie_.valeur().valeurs();
495 const DoubleTab& n_bis =normales_symetrie_bis_.valeur().valeurs();
496 ArrOfDouble normale(dimension);
497
498 const IntVect& tab1=la_matrice.get_tab1();
499 const IntVect& tab2=la_matrice.get_tab2();
500
501 const DoubleTab& champ_inconnue = equation().inconnue().valeurs();
502 int dirmax=2;
503 if (normales_symetrie_ter_.non_nul()) dirmax=3;
504 for (int som=0; som<nb_som; som++)
505 if ( type_sommet_[som]==1)
506 {
507 for (int dir=0; dir<dirmax; dir++)
508 {
509 const DoubleTab& nn=(dir==0?n:(dir==1?n_bis:normales_symetrie_ter_.valeur().valeurs()));
510 for (int d=0; d<dimension; d++) normale[d]=nn(som,d);
511// On commence par recalculer secmem=secmem-A *present pour pouvoir modifier A (on en profite pour projeter)
512 int nb_coeff_ligne=tab1[som*nb_comp+1] - tab1[som*nb_comp];
513 for (int k=0; k<nb_coeff_ligne; k++)
514 {
515 for (int comp=0; comp<nb_comp; comp++)
516 {
517 int j=tab2[tab1[som*nb_comp+comp]-1+k]-1;
518// assert(j!=(som*nb_comp+comp));
519// if ((j>=(som*nb_comp))&&(j<(som*nb_comp+nb_comp)))
520
521 const double coef_ij=la_matrice(som*nb_comp+comp,j);
522 int som2=j/nb_comp;
523 int comp2=j-som2*nb_comp;
524 secmem(som,comp)-=coef_ij*champ_inconnue(som2,comp2);
525 }
526 }
527 double somme_b=0;
528
529 for (int comp=0; comp<nb_comp; comp++)
530 somme_b+=secmem(som,comp)*normale[comp];
531// Cerr<<som<<" sommet " <<somme_b<<" "<<secmem(som,0)<<" "<<secmem(som,1)<<finl;
532// on retire secmem.n n
533 for (int comp=0; comp<nb_comp; comp++)
534 secmem(som,comp)-=somme_b*normale[comp];
535
536 if (1)
537 {
538// on doit remettre la meme diagonale partout on prend la moyenne
539 double ref=0;
540 for (int comp=0; comp<nb_comp; comp++)
541 {
542
543 int j0=som*nb_comp+comp;
544 ref+=la_matrice.coef(j0,j0);
545 }
546 ref/=nb_comp;
547
548 for (int comp=0; comp<nb_comp; comp++)
549 {
550 int j0=som*nb_comp+comp;
551 double rap=ref/la_matrice.coef(j0,j0);
552// Cerr<<dir<<" "<<som <<" "<<comp<<" rpp "<<rap <<finl;
553 for (int k=0; k<nb_coeff_ligne; k++)
554 {
555
556 int j=tab2[tab1[j0]-1+k]-1;
557 la_matrice(j0,j)*=rap;
558 }
559 assert(est_egal(la_matrice(j0,j0),ref));
560 }
561 }
562 if (1)
563 {
564// on annule tous les coef extra diagonaux du bloc
565// sur les lignes i pour lesquelles normale(d) !=0
566// c.a.d dasb(normale(d)>1e-5)
567 for (int k=0; k<nb_coeff_ligne; k++)
568 {
569
570 for (int comp=0; comp<nb_comp; comp++)
