49//! Lecture d'un champ input dans un flot d'entree<br>Factorisation des readOn des differents Champ_input<br>Format:<br>{ nb_comp nombre_de_composantes_du_champ<br>nom nom_du_champ<br>probleme nom_du_probleme_base_associe<br>[sous_zone nom_de_la_sous_zone_associee] }
50void Champ_Input_Proto::read(Entree& is)
51{
52 Nom nom_champ;
53 Nom nom_pb;
54 Nom nom_sous_zone;
55 int nbc=-1;
56
57// Lire nom_champ et nom_pb
58 Motcle accolade_ouverte("{");
59 Motcle accolade_fermee("}");
60 Motcle motlu;
61 is >> motlu;
62 if (motlu != accolade_ouverte)
63 {
64 Cerr << "It was expected { to start to read the Champ_Input" << finl;
65 Process::exit();
66 }
67 is >> motlu;
68 while(1)
69 {
70 if (motlu == "nb_comp")
71 {
72 is >> nbc;
73 set_nb_comp(nbc);
74 }
75 else if (motlu == "nom")
76 {
77 is >> nom_champ;
78 }
79 else if (motlu == "initial_value")
80 {
81 is >> initial_value_;
82 }
83 else if (motlu=="probleme")
84 {
85 is >> nom_pb;
86 }
87 else if (motlu=="sous_zone")
88 {
89 is >> nom_sous_zone;
90 }
91 else if (motlu == accolade_fermee)
92 break;
93 else
94 {
95 Cerr << "Error in Ch_front_input_uniforme::readOn" << finl;
96 Cerr << "We expected a } instead of " << motlu << finl;
97 Process::exit();
98 }
99 is >> motlu;
100 }
101 if (nom_champ=="??" || nom_pb=="??" || nbc==-1)
102 {
103 Cerr << "Error in Ch_front_input_uniforme::readOn" << finl;
104 Cerr << "It must be define the name, problem and number of components" << motlu << finl;
105 Process::exit();
106 }
107// Nommer le champ
108 set_name(nom_champ);
109
110// Retrouver le probleme et s'inscrire dans sa liste
111 mon_pb=ref_cast(Probleme_base,Interprete::objet(nom_pb));
112
113 ma_sous_zone.reset();
114 if (nom_sous_zone!="??")
115 {
116 if (!sous_zone_ok)
117 {
118 Cerr << "Error in Ch_front_input_uniforme::readOn" << finl;
119 Cerr << "The sous_zones are not managed by this field" << finl;
120 Process::exit();
121 }
122 ma_sous_zone=mon_pb->domaine().ss_zone(nom_sous_zone);
123 }
124}

126void Champ_Input_Proto::setValueOnTab(const TrioField& afield, DoubleTab& tab)
127{
128 if (afield._type==0)
129 {
130 if ((afield._nb_elems!=tab.dimension(0))&&(afield._nb_elems!=tab.dimension_tot(0)))

134 }
135 }
136 else
137 {
138 if (afield._nbnodes!=tab.dimension(0))
139 throw WrongArgument(mon_pb.le_nom().getChar(),"setInputField","afield","should have the same _nb_elems as returned by getInputFieldTemplate");

143 }
144// order is c1 c2 c3 c1 c2 c3 c1 c2 c3
145 memcpy(tab.addr(),afield._field,tab.size()*sizeof(double));
146 tab.echange_espace_virtuel();
