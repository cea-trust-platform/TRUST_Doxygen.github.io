32//! Joint_Items contient les informations de distribution<br>parallele d'un item geometrique particulier avec une zone<br>voisine particuliere (item = sommet, element, face, etc..)<br>Ces structures sont initialisees dans Scatter.<br>Elles sont ensuite utilisees par exemple pour creer<br>un tableau distribue indexe par les indices des items<br>geometriques.
33//
34
35class Joint_Items
36{
37public:
38 Joint_Items();
39 void reset();
40
41// Pour utiliser ces accesseurs, il faut avoir initialise
42// les structures au prealable avec set_xxx.
43 int nb_items_reels() const;
44 const ArrOfInt& items_communs() const;
45 const ArrOfInt& items_distants() const;
46 int nb_items_virtuels() const;
47 const IntTab& renum_items_communs() const;
48
49// Methodes d'initialisation des structures
50 void set_nb_items_reels(int n);
51 ArrOfInt& set_items_communs();
52 ArrOfInt& set_items_distants();
53 void set_nb_items_virtuels(int n);
54 IntTab& set_renum_items_communs();
55
56private:
57// Nombre d'items reels (permet de construire un tableau distribue)
58// uniquement avec les infos du joint.
59 int nb_items_reels_;
60
61// Liste des items communs avec la zone voisine (la liste est
62// classee dans le meme ordre sur la zone locale et sur la zone
63// voisine => items_communs[i] sur joint_j de zone_k represente la meme
64// entite geometrique que items_communs[i] sur joint_k de zone_j)
65 ArrOfInt items_communs_;
66
67// Liste des items distants a envoyer a la zone voisine
68// (l'ordre des items dans cette liste determine l'ordre d'apparition
69// de ces items dans l'espace virtuel du voisin)
70 ArrOfInt items_distants_;
71
72// Nombre d'items virtuels recus de la zone voisine.
73// on a "nb_items_virtuels_ sur joint_j de zone_k"
74// = "items_distants.size_array() sur joint_k de zone_j"
75 int nb_items_virtuels_;
76
77// Correspondance entre l'indice local d'un item commun et l'indice
78// du meme item sur la zone voisine:
79// colonne 0 = indice sur la zone voisine,
80// colonne 1 = indice sur la zone locale
81// dimension(0) est egal a items_communs.size_array()
82// L'ordre des items dans le tableau n'est pas specifie
83 IntTab renum_items_communs_;
84
85// Qu'est ce qui a ete initialise ?
86 int flags_init_;
87};
