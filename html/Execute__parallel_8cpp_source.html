52//! Cree une partition des nproc processeurs du calcul pour interpreter<br>N jeux de donnees differents. La syntaxe du jeu de donnees est la suivante<br>Execute_parallel {<br>liste_cas N cas1 cas2 cas3 ...<br>[ nb_procs N nproc1 nproc2 nproc3 ... ]<br>}<br>"cas1" est le nom du cas (on lit le fichier cas1.data sur disque)<br>nproc1 est le nombre de processeurs a utiliser pour ce cas<br>Par defaut, on utilise 1 processeur pour chaque cas<br>Les sorties Cerr et Cout sont redirigees vers le journal du processeur<br>maitre de chaque cas.
53Entree& Execute_parallel::interpreter(Entree& is)
54{
55 Cerr << "Execute_parallel::interpreter to run several cases:" << finl;
56
57 Noms liste_cas;
58 ArrOfInt nb_procs;
59
60 int disable_journal = 0;
61
62 Param param(que_suis_je());
63 param.ajouter("liste_cas", &liste_cas, Param::REQUIRED);
64 param.ajouter("nb_procs", &nb_procs);
65 param.ajouter_flag("disable_journal", &disable_journal);
66 param.lire_avec_accolades_depuis(is);
67// Si on n'a pas donne nb_procs, on suppose que ca vaut 1
68 const int n_calculs = liste_cas.size();
69 if (nb_procs.size_array() == 0)
70 {
71 Cerr << "Nb_procs not given, we assume that calculations are sequential."
72 << finl;
73 nb_procs.resize_array(n_calculs);
74 nb_procs = 1;
75 }
76// Verification de la taille des tableaux
77 if (nb_procs.size_array() != n_calculs)
78 {
79 Cerr << "Error : nb_procs array must have " << n_calculs
80 << " values." << finl;
81 barrier();
82 exit();
83 }
84 if (n_calculs == 0)
85 return is;
86
87// Verification du contenu :
88 if (min_array(nb_procs) < 1)
89 {
90 Cerr << "Error : processor numbers must be >= 1" << finl;
91 barrier();
92 exit();
93 }
94 int count = 0;
95 for (int i = 0; i < n_calculs; i++)
96 count += nb_procs[i];
97 if (count > nproc())
98 {
99 Cerr << "Error : computations require " << count << " processors." << finl;
100 Cerr << "but only " << nproc() << " processors has been asked." << finl;
101 barrier();
102 exit();
103 }
104
105// Creation des N groupes de processeurs
106// (les groupes sont detruits quand le VECT est detruit)
107 VECT(DERIV(Comm_Group)) groupes(n_calculs);
108 count = 0;
109 Nom log_courant("");
110 for (int i = 0; i < n_calculs; i++)
111 {
112 const int n = nb_procs[i];
113 Nom log("");
114 Nom log1(Objet_U::nom_du_cas());
115 log1+="_";
116 char s[20];
117 sprintf(s, "%05d", (True_int)count);
118 log1+=s;
119 log1+=".log";
120 if (n==1)
121 {
122 log+=log1;
123 log+=" file";
124 }
125 else
126 {
127 Nom log2(Objet_U::nom_du_cas());
128 log2+="_";
129 char s2[20];
130 sprintf(s2, "%05d", (True_int)(count+n-1));
131 log2+=s2;
132 log2+=".log";
133 log="log files from ";
134 log+=log1;
135 log+=" to ";
136 log+=log2;
137 }
138 Cerr << "Error and standard outputs are redirected into " << log << " for case " << liste_cas[i] << finl;
139// On stocke dans log_courant car reutilise plus loin
140 if (Process::me()>=count && Process::me()<=count+n-1)
141 log_courant=log;
142
143 ArrOfInt tab(n);
144 for (int j = 0; j < n; j++)
145 tab[j] = count++;
146 PE_Groups::create_group(tab, groupes[i]);
147 }
148// On lancer une exception si un calcul s'arrete donc
149// on change le comportement de Process par defaut (MPI_Abort)
150 Process::exception_sur_exit=1;
151 Cerr << n_calculs << " cases are running..." << finl;
152// Chaque processeur entre dans son groupe et interprete le jeu de donnees
153 Nom ancien_nom_du_cas(nom_du_cas());
154 const int old_journal_level = get_journal_level();
155 for (int i = 0; i < n_calculs; i++)
156 {
157 if (PE_Groups::enter_group(groupes[i].valeur()))
158 {
159 set_Cerr_to_journal(1);
160 if (disable_journal)
161 change_journal_level(0);
162 Nom nom_fichier(liste_cas[i]);
163 get_set_nom_du_cas() = nom_fichier;
164 Journal(1) << "Execute_parallel: Entering subgroup " << i
165 << " to run case " << nom_fichier << finl;
166
167 nom_fichier += ".data";
168 {
169// Ouverture du fichier (on cree l'objet LecFicDiffuse a l'interieur
170// de l'accolade pour detruire le fichier avant de sortir du groupe)
171 LecFicDiffuse_JDD data_file(nom_fichier);
172 data_file.set_check_types(1);
173// On cree un nouvel interprete. A la fin de la lecture du cas
174// les objets seront detruits.
175 Interprete_bloc interp;
176// On utilise les exceptions pour des calculs qui s'arretent
177 int ok=1;
178 try
179 {
180 interp.interpreter_bloc(data_file,
181 Interprete_bloc::FIN /* on attend FIN a la fin du fichier */,
182 0 /* verifie_sans_interpreter=0 */);
183 }
184 catch (TriouError err)
185 {
186 assert(err.get_pe()==Process::me());
187 ok=0;
188 }
189 set_Cerr_to_journal(0);
190 if (ok)
191 Cerr << "Case " << liste_cas[i] << " has finished. See " << log_courant << finl;
192 else
193 Cerr << "!!! Case " << liste_cas[i] << " has failed. See " << log_courant << " !!!" << finl;
194 }
195 Journal(1) << "Execute_parallel: Exiting subgroup " << i << finl;
196 PE_Groups::exit_group();
197 }
198 }
199// on revient au comportement standard de trio
200 Process::exception_sur_exit=0;
201 change_journal_level(old_journal_level);
202// On attend que tous les processeurs aient fini d'executer leur calcul.
203 barrier();
204 get_set_nom_du_cas() = ancien_nom_du_cas;
205 Cerr << finl << "End of Execute_parallel::interpreter" << finl;
206 return is;
207}
