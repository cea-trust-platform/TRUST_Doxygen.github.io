42void Pb_Dilatable_base::associer_sch_tps_base(const Schema_Temps_base& sch)
43{
44 if (!sub_type(Schema_Euler_explicite,sch) && !sub_type(Schema_Euler_Implicite,sch)
45 && !sub_type(RRK2,sch) && !sub_type(RK2,sch) && !sub_type(RK3,sch) && !sub_type(RK4,sch))
46 {
47 Cerr << "TRUST can't solve a " << que_suis_je() << " with a " << sch.que_suis_je() << " time scheme." << finl;
48 Process::exit();
49 }
50 if ( sub_type(Schema_Euler_Implicite,sch) && coupled_==1 )
51 {
52 Cerr << finl;
53 Cerr << "Coupled problem with unique Euler implicit time scheme with " << que_suis_je() << "fluid are not allowed!" << finl;
54 Cerr << "Choose another time scheme or set a time scheme for each of your problems." << finl;
55 Process::exit();
56 }
57 Pb_Fluide_base::associer_sch_tps_base(sch);
58}

100bool Pb_Dilatable_base::iterateTimeStep(bool& converged)
101{
102 Debog::set_nom_pb_actuel(le_nom());
103 Schema_Temps_base& sch=schema_temps();
104 double temps_present=sch.temps_courant();
105 double temps_futur=temps_present+sch.pas_de_temps();
106
107// 1. Solve all the equations except the first one (Navier Stokes, solved later at //6)
108 for (int i=1; i<nombre_d_equations(); i++)
109 {
110 sch.faire_un_pas_de_temps_eqn_base(equation(i));
111 statistiques().begin_count(mettre_a_jour_counter_);
112 equation(i).milieu().mettre_a_jour(temps_futur);
113 equation(i).inconnue().mettre_a_jour(temps_futur);
114 statistiques().end_count(mettre_a_jour_counter_);
115 }
116
117 statistiques().begin_count(mettre_a_jour_counter_);
118
119// 2. Compute temperature-dependent coefficients
120 le_fluide_->calculer_coeff_T();
121
122// 3. Compute volumic mass (update Cp)
123 le_fluide_->calculer_masse_volumique();
124
125// 4. Solve EDO equation (pressure) if needed (ie. QC)
126 solve_pressure_thermo();
127
128// 5. Compute volumic mass
129 le_fluide_->calculer_masse_volumique();
130 statistiques().end_count(mettre_a_jour_counter_);
131
132// 6. Solve Navier Stokes equation
133 sch.faire_un_pas_de_temps_eqn_base(equation(0));
134 statistiques().begin_count(mettre_a_jour_counter_);
135 equation(0).milieu().mettre_a_jour(temps_futur);
136 equation(0).inconnue().mettre_a_jour(temps_futur);
137 statistiques().end_count(mettre_a_jour_counter_);
138
139// Update pressure fields (total/thermo/hydro) if necessary
140 update_pressure_fields(temps_futur);
141
142// on recule les inconnues (le pb mettra a jour les equations)
143 for (int i=0; i<nombre_d_equations(); i++)
144 equation(i).inconnue().reculer();
145
146// Calculs coeffs echange sur l'instant sur lequel doivent agir les operateurs.
147 double tps=schema_temps().temps_defaut();
148 for(int i=0; i<nombre_d_equations(); i++)
149 equation(i).zone_Cl_dis()->calculer_coeffs_echange(tps);
150
151 converged=true;
152 return true;
153}
