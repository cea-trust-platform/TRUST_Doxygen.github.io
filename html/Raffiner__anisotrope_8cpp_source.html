31Entree& Raffiner_anisotrope::interpreter_(Entree& is)
32{
33 if (dimension == 2)
34 {
35 Cerr << "Refinement of each element into 3 new triangles... " <<finl;
36 }
37 else if (dimension == 3)
38 {
39 Cerr << "Refinement of each element into 4 new tetrahedra... " <<finl;
40 }
41 else
42 {
43 Cerr << "we can not refine in dimension " << dimension <<finl;
44 exit();
45 }
46 associer_domaine(is);
47 Scatter::uninit_sequential_domain(domaine());
48 for(int i=0; i<domaine().nb_zones() ; i++)
49
50 {
51 Raffiner_anisotrope_(domaine().zone(i));
52 }
53 Scatter::init_sequential_domain(domaine());
54 Cerr << "Refinement... OK" << finl;
55 return is;
56}

58void Raffiner_anisotrope::Raffiner_anisotrope_(Zone& zone)
59{
60 if ((zone.type_elem()->que_suis_je() == "Triangle")
61 ||(zone.type_elem()->que_suis_je() == "Tetraedre"))
62 {
63 Domaine& dom=zone.domaine();
64 IntTab& les_elems=zone.les_elems();
65 int oldsz=les_elems.dimension(0);
66 DoubleTab& sommets=dom.les_sommets();
67 int cpt=sommets.dimension(0);
68 {
69 DoubleTab sommets_ajoutes(oldsz, dimension);
70 zone.type_elem()->calculer_centres_gravite(sommets_ajoutes);
71 sommets.resize(cpt+oldsz, dimension);
72 for(int i=0; i<oldsz; i++)
73 for(int j=0; j<dimension; j++)
74 sommets(cpt+i,j)=sommets_ajoutes(i,j);
75 }
76
77 zone.typer(zone.type_elem()->que_suis_je());
78 IntTab new_elems((dimension+1)*oldsz, dimension+1);
79
80 if(dimension==2)
81 {
82 for(int i=0; i< oldsz; i++)
83 {
84 new_elems(i , 0) = i+cpt;
85 new_elems(i , 1) = les_elems(i,0);
86 new_elems(i , 2) = les_elems(i,1);
87
88 new_elems(i+oldsz, 0) = i+cpt;
89 new_elems(i+oldsz, 1) = les_elems(i,0);
90 new_elems(i+oldsz, 2) = les_elems(i,2);
91
92 mettre_a_jour_sous_zone(zone,i,i+oldsz,1);
93
94 new_elems(i+2*oldsz, 0) = i+cpt;
95 new_elems(i+2*oldsz, 1) = les_elems(i,1);
96 new_elems(i+2*oldsz, 2) = les_elems(i,2);
97
98 mettre_a_jour_sous_zone(zone,i,i+2*oldsz,1);
99 }
100 }
101 else
102 {
103 for(int i=0; i< oldsz; i++)
104 {
105 new_elems(i , 0) = i+cpt;
106 new_elems(i , 1) = les_elems(i,0);
107 new_elems(i , 2) = les_elems(i,1);
108 new_elems(i , 3) = les_elems(i,2);
109
110 new_elems(i+oldsz, 0) = i+cpt;
111 new_elems(i+oldsz, 1) = les_elems(i,0);
112 new_elems(i+oldsz, 2) = les_elems(i,1);
113 new_elems(i+oldsz, 3) = les_elems(i,3);
114
115 mettre_a_jour_sous_zone(zone,i,i+oldsz,1);
116
117 new_elems(i+2*oldsz, 0) = i+cpt;
118 new_elems(i+2*oldsz, 1) = les_elems(i,0);
119 new_elems(i+2*oldsz, 2) = les_elems(i,2);
120 new_elems(i+2*oldsz, 3) = les_elems(i,3);
121
122 mettre_a_jour_sous_zone(zone,i,i+2*oldsz,1);
123
124 new_elems(i+3*oldsz, 0) = i+cpt;
125 new_elems(i+3*oldsz, 1) = les_elems(i,1);
126 new_elems(i+3*oldsz, 2) = les_elems(i,2);
127 new_elems(i+3*oldsz, 3) = les_elems(i,3);
128
129 mettre_a_jour_sous_zone(zone,i,i+3*oldsz,1);
130 }
131 }
132
133 les_elems.ref(new_elems);
134
135// Reconstruction de l'octree
136 if (dimension == 2)
137 Cerr << "We have split the triangles ..." << finl;
138 else
139 Cerr << "We have split the tetrahedra ..." << finl;
140 zone.invalide_octree();
141
142 Cerr<<"END of Raffiner_anisotrope..."<<finl;
143 Cerr<<" 1 NbElem="<<zone.les_elems().dimension(0)<<" NbNod="<<zone.nb_som()<<finl;
144 }
145 else
146 {
147 Cerr << "We do not yet know how to Raffiner_anisotrope the "
148 << zone.type_elem()->que_suis_je() <<"s"<<finl;
149 exit();
150 }
151}
