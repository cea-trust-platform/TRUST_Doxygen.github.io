771 zone_VEF.oriente_normale(facej,elem);
772 coeffij*=mijK/volume;

776 int tmp=ii;

810 int som=zone_VEF.face_sommets(face,isom);
811 som=dom.get_renum_som_perio(som);
812 diag[som]=1.e1*diag_ref;
813 }
814 }
815 }

817 mat.dimensionner(nb_som, nnz+nb_som) ;
818 mat.remplir(voisins, coeffs, diag) ;
819 mat.compacte() ;

833 DoubleVect solution(nb_som);
