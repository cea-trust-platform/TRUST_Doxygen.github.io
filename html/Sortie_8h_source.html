122//! Methode de bas niveau pour ecrire un tableau d'ints ou reels dans le stream.<br>Dans l'implementation de la classe de base, on ecrit dans ostream_.<br>En binaire on utilise ostream::write(), en ascii ostream::operato<<()<br>En ascii, on revient a la ligne chaque fois qu'on a ecrit "nb_col" valeurs et a la fin du tableau.<br>Valeur de retour : ostream_->good()
123template<typename _TYPE_>
124int Sortie::put_template(const _TYPE_ *ob, int n, int nb_col)
125{
126 assert(n >= 0);
127 if (bin_)
128 {
129 std::streamsize sz = sizeof(_TYPE_);
130 sz *= n;
131// Overflow checking :
132 assert(sz / (std::streamsize) sizeof(_TYPE_) == (std::streamsize) n);
133 ostream_->write((const char*) ob, sz);
134 }
135 else
136 {
137 int j = nb_col;
138 for (int i = 0; i < n; i++)
139 {
140 (*ostream_) << (ob[i]) << (' ');
141 j--;
142 if (j <= 0)
143 {
144 (*ostream_) << (endl);
145 j = nb_col;
146 }
147 }
148// Si on n'a pas fini pas un retour a la ligne, en ajouter un
149 if (j != nb_col && n > 0) (*ostream_) << (endl);
150
151 ostream_->flush();
152 }
153 return ostream_->good();
154}

159//! Methode de bas niveau pour ecrire un int ou flottant dans le stream.<br>Dans l'implementation de la classe de base, on ecrit dans ostream_.<br>En binaire on utilise ostream::write(), en ascii ostream::operato<<()
160template<typename _TYPE_>
161Sortie& Sortie::operator_template(const _TYPE_ &ob)
162{
163 if (bin_)
164 ostream_->write((char*) &ob, sizeof(_TYPE_)); /* Ecriture en mode binaire */
165 else
166 (*ostream_) << ob; /* Ecriture avec conversion en ascii */
167 return *this;
168}
