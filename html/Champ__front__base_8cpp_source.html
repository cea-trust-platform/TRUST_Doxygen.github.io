171//! Renvoie le vecteur des valeurs du champ pour la face donnee.<br>Par defaut pour les champs fonc, on suppose que le tableau des<br>valeurs porte nb_faces * nb_compo_ valeurs.<br>Exemple de cas particulier: champ_front_uniforme::valeurs_face<br>Parametre: num_face<br>Signification: l'indice d'une face sur la frontiere<br>0 <= num_face < frontiere_dis().frontiere().nb_faces()<br>Parametre: val<br>Signification: On resize ce tableau et on le remplit.
172void Champ_front_base::valeurs_face(int num_face, DoubleVect& val) const
173{
174#ifndef NDEBUG
175 const int nb_faces = frontiere_dis().frontiere().nb_faces();
176// Si plantage ici, c'est que le tableau des valeurs ne contient pas une
177// valeur pour chaque face, il faut reimplementer la methode dans la
178// classe derivee...
179 assert(num_face >= 0 && num_face < nb_faces);
180 assert(valeurs().dimension(0) == nb_faces);
181#endif
182 const int n = nb_compo_;
183 val.resize(n);
184 const DoubleTab& valeurs_a_copier=valeurs();
185 for (int i = 0; i < n; i++)
186 val[i] = valeurs_a_copier(num_face, i);
187}
