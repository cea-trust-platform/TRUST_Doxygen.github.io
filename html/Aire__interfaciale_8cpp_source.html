48Entree& Aire_interfaciale::readOn(Entree& is)
49{
50 assert(l_inco_ch.non_nul());
51 assert(le_fluide.non_nul());
52 Convection_Diffusion_std::readOn(is);
53
54 terme_convectif.set_fichier("Convection_interfacial_area");
55 terme_convectif.set_description((Nom)"interfacial area transfer rate=Integral(-A*u*ndS) [kg] if SI units used");
56
57 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, probleme()) ? &ref_cast(Pb_Multiphase, probleme()) : NULL;
58
59 if (!pbm || pbm->nb_phases() == 1) Process::exit(que_suis_je() + " : not needed for single-phase flow!");
60 for (int n = 0; n < pbm->nb_phases(); n++) //recherche de n_l, n_g : phase {liquide,gaz}_continu en priorite
61 if (pbm->nom_phase(n).debute_par("liquide") && (n_l < 0 || pbm->nom_phase(n).finit_par("continu"))) n_l = n;
62 if (n_l < 0) Process::exit(que_suis_je() + " : liquid phase not found!");
63
64 if (pbm->has_correlation("diametre_bulles")) Process::exit(que_suis_je() + " : the interfacial area equation sets bubble diameter, there cannot be an exterior correlation !");
65
66 return is;
67}

75void Aire_interfaciale::discretiser()
76{
77 int nb_valeurs_temp = schema_temps().nb_valeurs_temporelles();
78 double temps = schema_temps().temps_courant();
79 const Discret_Thyd& dis=ref_cast(Discret_Thyd, discretisation());
80 int N = ref_cast(Pb_Multiphase, probleme()).nb_phases();
81
82 Cerr << "Interfacial area discretization" << finl;
83// On utilise temperature pour la directive car discretisation identique
84 dis.discretiser_champ("temperature",zone_dis(),"interfacial_area","m-1", N,nb_valeurs_temp,temps,l_inco_ch);//une aire interfaciale par phase
85 l_inco_ch.valeur().fixer_nature_du_champ(multi_scalaire);
86 l_inco_ch.valeur().fixer_nom_compo(0, Nom("tau"));
87 champs_compris_.ajoute_champ(l_inco_ch);
88
89 Cerr << "Bubble diameter discretization" << finl;
90// On utilise temperature pour la directive car discretisation identique
91 Noms noms(N), unites(N);
92 noms[0] = "diametre_bulles";
93 unites[0] = "m";
94 Motcle typeChamp = "champ_elem" ;
95 const Zone_dis& z = ref_cast(Zone_dis, probleme().domaine_dis().zone_dis(0));
96 dis.discretiser_champ(typeChamp, z.valeur(), multi_scalaire, noms , unites, N, 0, diametre_bulles);
97
98 champs_compris_.ajoute_champ(diametre_bulles);
99
100 Equation_base::discretiser();
101 Cerr << "Echelle_temporelle_turbulente::discretiser() ok" << finl;
102}

142void Aire_interfaciale::mettre_a_jour(double temps)
143{
144 Convection_Diffusion_std::mettre_a_jour(temps);
145
146 int i, n, N = ref_cast(Pb_Multiphase, probleme()).nb_phases();
147
148 const DoubleTab& alpha = probleme().get_champ("alpha").passe(), &a_i = inconnue().passe();
149 DoubleTab& d_b = diametre_bulles.valeurs();
150
151 diametre_bulles.mettre_a_jour(temps);
152
153 for (n = 0; n < N; n++)
154 for (i = 0; i < d_b.dimension_tot(0); i++)
155 if (n != n_l)
156 d_b(i, n) = std::max(1.e-8, ((a_i(i, n)>1.e-6) ? 6 * alpha(i, n)/a_i(i, n) : 0));
157
158 for (i = 0; i < d_b.dimension_tot(0); i++) d_b(i, n_l) = 0;
159}
