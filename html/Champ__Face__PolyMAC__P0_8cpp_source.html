90Champ_base& Champ_Face_PolyMAC_P0::affecter_(const Champ_base& ch)
91{
92 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0,ref_zone_vf_.valeur());
93 const DoubleVect& fs = zone.face_surfaces();
94 const DoubleTab& nf = zone.face_normales(), &xv = zone.xv();
95 DoubleTab& val = valeurs(), eval;
96
97 if (sub_type(Champ_Fonc_reprise, ch))
98 {
99 for (int num_face=0; num_face<zone.nb_faces(); num_face++)
100 val(num_face) = ch.valeurs()[num_face];
101 }
102 else
103 {
104 int f, n, N = val.line_size(), d, D = dimension, unif = sub_type(Champ_Uniforme, ch);
105
106 if (unif) eval = ch.valeurs();
107 else eval.resize(val.dimension_tot(0), N * D), ch.valeur_aux(xv,eval);
108 for (f = 0; f < zone.nb_faces_tot(); f++)
109 for (d = 0; d < D; d++)
110 for (n = 0; n < N; n++)
111 val(f, n) += eval(unif ? 0 : f, N * d + n) * nf(f, d) / fs(f);
112 update_ve(val);
113// copie dans toutes les cases
114 valeurs().echange_espace_virtuel();
115 for(int i=1; i<les_valeurs->nb_cases(); i++) les_valeurs[i].valeurs() = valeurs();
116 }
117 return *this;

121void Champ_Face_PolyMAC_P0::update_ve(DoubleTab& val) const
122{
123 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0,zone_vf());
124 if (valeurs().get_md_vector() != zone.mdv_ch_face) return; //pas de variables auxiliaires -> rien a faire
125 const DoubleVect& fs = zone.face_surfaces(), &ve = zone.volumes(), &pf = zone.porosite_face(), &pe = zone.porosite_elem();
126 const DoubleTab& xp = zone.xp(), &xv = zone.xv();
127 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins();
128 int e, f, j, d, D = dimension, n, N = val.line_size(), ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot();
129 double fac;
130 for (e = 0; e < ne_tot; e++)
131 {
132 for (d = 0; d < D; d++)
133 for (n = 0; n < N; n++) val(nf_tot + D * e + d, n) = 0;
134 for (j = 0; j < e_f.dimension(1) && (f = e_f(e, j)) >= 0; j++)
135 for (fac = (e == f_e(f, 0) ? 1 : -1) * pf(f) * fs(f) / (pe(e) * ve(e)), d = 0; d < D; d++)
136 for (n = 0; n < N; n++) val(nf_tot + D * e + d, n) += fac * (xv(f, d) - xp(e, d)) * val(f, n);
137 }

247void Champ_Face_PolyMAC_P0::update_ve2(DoubleTab& val, int incr) const
248{
249 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0,zone_vf());
250 if (valeurs().get_md_vector() != zone.mdv_ch_face) return; //pas de variables auxiliaires -> on sort
251 const Conds_lim& cls = zone_Cl_dis().les_conditions_limites();
252 int i, j, e, ed, d, D = dimension, n, N = val.line_size(), nf_tot = zone.nb_faces_tot();
253 init_ve2(), init_fcl();
254
255 for (e = 0, ed = 0, i = nf_tot; e < zone.nb_elem(); e++)
256 for (d = 0; d < D; d++, ed++, i++)
257 for (n = 0; n < N; n++)
258 {
259 /* partie "interne" */
260 for (val(i, n) = 0, j = ve2d(ed, 0); j < ve2d(ed + 1, 0); j++)
261 val(i, n) += ve2c(j) * val(ve2j(j), n);
262
263 /* partie "faces de bord de Dirichlet" (sauf si on fait des increments) */
264 if (!incr)
265 for (j = ve2d(ed, 1); j < ve2d(ed + 1, 1); j++)
266 if (sub_type(Dirichlet, cls[fcl_(ve2bj(j, 0), 1)].valeur()))
267 val(i, n) += ve2bc(j) * ref_cast(Dirichlet, cls[fcl_(ve2bj(j, 0), 1)].valeur()).val_imp(fcl_(ve2bj(j, 0), 2), N * ve2bj(j, 1) + n);
268 }
269
270 val.echange_espace_virtuel();
271}
