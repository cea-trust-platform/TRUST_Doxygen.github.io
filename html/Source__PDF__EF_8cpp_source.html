224 DoubleTrav marqueur=equation().probleme().get_champ("vitesse").valeurs();
225 vitesse_imposee = 0.;

229 {
230 double norm = sqrt(rotation(num_elem,0)*rotation(num_elem,0)
231 +rotation(num_elem,1)*rotation(num_elem,1)
232 +rotation(num_elem,2)*rotation(num_elem,2)
233 +rotation(num_elem,3)*rotation(num_elem,3)
234 +rotation(num_elem,4)*rotation(num_elem,4)
235 +rotation(num_elem,5)*rotation(num_elem,5)
236 +rotation(num_elem,6)*rotation(num_elem,6)
237 +rotation(num_elem,7)*rotation(num_elem,7)
238 +rotation(num_elem,8)*rotation(num_elem,8));
239 if (norm > 1e-6)
240 {
241 for (int i=0; i<nb_som_elem; i++)
242 {
243 int s = elems(num_elem,i);
244 for (int c=0; c<3; c++)
245 {
246 for (int k=0; k<3; k++)
247 {
248 vitesse_imposee(s,c) += rotation(num_elem,3*c+k)*vitesse(s,k);
249// Cerr << "k = " << k << ", c = " << c << ", 3*c+k = " << 3*c+k << ", rotation = " << rotation(num_elem,3*c+k) << finl;

259 {
260 for (int c=0; c<3; c++)

275DoubleVect Source_PDF_EF::diag_coeff_elem(ArrOfDouble& vitesse_elem, const DoubleTab& rotation, int num_elem) const
276{
277 assert(Objet_U::dimension==3);
278 ArrOfDouble tuvw(dimension);
279
280 tuvw = get_tuvw_local();
281
282 ArrOfDouble sum_dir_loc(dimension) ;
283 for (int i=0; i<dimension; i++)
284 {
285 sum_dir_loc[i]=0.;
286 for (int k=0; k<dimension; k++)
287 sum_dir_loc[i]+=rotation(num_elem,3*i+k);
288 }
289
290 DoubleVect diag_coef(dimension);
291 for (int comp=0; comp<dimension; comp++)
292 {
293 diag_coef(comp) = 0. ;
294 for (int k=0; k<dimension; k++)
295 diag_coef(comp)+=tuvw[k]*sum_dir_loc[k]*rotation(num_elem,3*k+comp);
296 }
297
298 return diag_coef ;
299}

301DoubleTab Source_PDF_EF::compute_coeff_elem() const
302{
303 const Zone_EF& zone_EF = la_zone_EF.valeur();
304 const IntTab& elems= zone_EF.zone().les_elems() ;
305 int nb_som_elem=zone_EF.zone().nb_som_elem();
306 int nb_elems=zone_EF.zone().nb_elem_tot();
307 ArrOfDouble vitesse_elem(dimension);
308 const DoubleTab& rotation=champ_rotation_.valeurs();
309 const DoubleTab& aire = champ_aire_.valeurs();
310
311 int dim_esp = Objet_U::dimension ;
312 DoubleTab coeff(nb_elems, dim_esp);
313
314 const double dt_ref = equation().probleme().schema_temps().pas_de_temps();
315 const double dt_min = equation().probleme().schema_temps().pas_temps_min();
316 double dt = std::max(dt_ref,dt_min);
317 const DoubleTab& rho_m=champ_rho_.valeurs();
318 if (equation().probleme().schema_temps().temps_courant()==0)
319 {
320 dt = 1.0;
321 }
322// Cerr<<"dt pour compute_coeff_elem : "<< dt <<finl;
323
324 double val_coeff ;
325 DoubleVect val_diag_coef(dim_esp);
326 for (int num_elem=0; num_elem<nb_elems; num_elem++)
327 {
328 if (aire(num_elem)<=0.)
329 {
330 for (int comp=0; comp<dim_esp; comp++) coeff(num_elem, comp) = 1. ;
331 }
332 else
333 {
334 const DoubleTab& vitesse=equation().inconnue().valeurs();
335 vitesse_elem=0;
336 for (int s=0; s<nb_som_elem; s++)
337 {
338 int som_glob=elems(num_elem,s);
339 for (int comp=0; comp<dimension; comp++)
340 vitesse_elem[comp]+=vitesse(som_glob,comp);
341 }
342 val_coeff = fonct_coeff(rho_m(num_elem), aire(num_elem), dt) ;
343 val_diag_coef = diag_coeff_elem(vitesse_elem, rotation, num_elem) ;
344 for (int comp=0; comp<dim_esp; comp++)
345 coeff(num_elem, comp) = 1.+ val_coeff * val_diag_coef(comp) * dt / rho_m(num_elem) ;
346 }
347 }
348 return coeff;
349}

351DoubleTab Source_PDF_EF::compute_coeff_matrice_pression() const
352{
353 const Zone_EF& zone_EF = la_zone_EF.valeur();
354 const IntTab& elems= zone_EF.zone().les_elems();
355 int nb_som_elem=zone_EF.zone().nb_som_elem();
356 int nb_elems=zone_EF.zone().nb_elem_tot();
357 int nb_som_tot=zone_EF.zone().nb_som_tot();
358 ArrOfDouble vitesse_elem(dimension);
359 const DoubleTab& rotation=champ_rotation_.valeurs();
360 const DoubleTab& aire = champ_aire_.valeurs();
361
362 int dim_esp = Objet_U::dimension ;
363 const DoubleTab& vitesse=equation().inconnue().valeurs();
364 DoubleTab coeff(vitesse);
365 IntTab contrib(nb_som_tot, dim_esp);
366 contrib = 0;
367 coeff = 0.;
368
369 const double dt_ref = equation().probleme().schema_temps().pas_de_temps();
370 const double dt_min = equation().probleme().schema_temps().pas_temps_min();
371 double dt = std::max(dt_ref,dt_min);
372 const DoubleTab& rho_m=champ_rho_.valeurs();
373 if (equation().probleme().schema_temps().temps_courant()==0)
374 {
375 dt = 1.0;
376 }
377// Cerr<<"dt pour compute_coeff_matrice_pression : "<< dt <<finl;
378
379 double val_coeff ;
380 DoubleVect val_diag_coef(dim_esp), coeff_el(dim_esp) ;
381 for (int num_elem=0; num_elem<nb_elems; num_elem++)
382 {
383 if (aire(num_elem)<=0.)
384 {
385 for (int comp=0; comp<dim_esp; comp++) coeff_el(comp) = 0. ;
386 }
387 else
388 {
389 vitesse_elem=0.;
390 for (int s=0; s<nb_som_elem; s++)
391 {
392 int som_glob=elems(num_elem,s);
393 for (int comp=0; comp<dimension; comp++)
394 vitesse_elem[comp]+=vitesse(som_glob,comp);
395 }
396 val_coeff = fonct_coeff(rho_m(num_elem), aire(num_elem), dt) ;
397 val_diag_coef = diag_coeff_elem(vitesse_elem, rotation, num_elem) ;
398 for (int comp=0; comp<dim_esp; comp++)
399 coeff_el(comp) = val_coeff * val_diag_coef(comp) * dt / rho_m(num_elem) ;
400 }
401 for (int i = 0; i<nb_som_elem; i++)
402 {
403 int s = elems(num_elem,i);
404 for (int comp=0; comp<dim_esp; comp++)
405 {
406 if (coeff_el(comp) > 0.)
407 {
408 coeff(s,comp)+= coeff_el(comp);
409 contrib(s, comp) += 1;
410 }
411 }
412 }
413 }
414
415 double val = 0.;
416 for (int s = 0; s<nb_som_tot; s++)
417 {
418 for (int comp=0; comp<dim_esp; comp++)
419 {
420 val = 1.;
421 if (contrib(s,comp) != 0)
422 {
423 val += coeff(s,comp)/contrib(s,comp);
424 }
425 coeff(s,comp) = val;
426 }
427 }
428 return coeff;
429}

448DoubleTab Source_PDF_EF::compute_pond(const DoubleTab& rho_m, const DoubleTab& aire, const DoubleVect& volume_thilde, int& nb_som_elem, int& nb_elems) const
449{
450 DoubleTab pond = rho_m;
451 double inv_nb_som = 1. / nb_som_elem;
452
453 const double dt_ref = equation().probleme().schema_temps().pas_de_temps();
454 const double dt_min = equation().probleme().schema_temps().pas_temps_min();
455 double dt = std::max(dt_ref,dt_min);
456 if (equation().probleme().schema_temps().temps_courant()==0)
457 {
458 dt = 1.0;
459 }
460
461 for (int num_elem=0; num_elem<nb_elems; num_elem++)
462 {
463 pond(num_elem) = fonct_coeff(rho_m(num_elem), aire(num_elem), dt) ;
464 pond(num_elem) *= volume_thilde(num_elem)*inv_nb_som*inv_nb_som;
465 }
466
467 return pond;
468}

500DoubleTab& Source_PDF_EF::ajouter_(const DoubleTab& vitesse, DoubleTab& resu, const int i_traitement_special) const
501{
502 /* i_traitement_special = 0 => traitement classique; coefficient (1/eta) */

507 const IntTab& elems= zone_EF.zone().les_elems() ;
508 int nb_som_elem=zone_EF.zone().nb_som_elem();
509 int nb_elems=zone_EF.zone().nb_elem_tot();

512 ArrOfDouble tuvw(dimension);

521
522 for (int num_elem=0; num_elem<nb_elems; num_elem++)
523 {
524 if (aire(num_elem)>0.)
525 {
526 if (i_traitement_special == 0)
527 {
528 tuvw[0] = 1.0 / modele_lu_.eta_;
529 tuvw[1] = 1.0 / modele_lu_.eta_;
530 tuvw[2] = 1.0 / modele_lu_.eta_;
531 }
532 else if (i_traitement_special == 1) //terme temps en rho v
533 {
534 tuvw[0] = 1.0;
535 tuvw[1] = 1.0;
536 tuvw[2] = 1.0;

541 tuvw[1] = 1.0 / rho_m(num_elem);
