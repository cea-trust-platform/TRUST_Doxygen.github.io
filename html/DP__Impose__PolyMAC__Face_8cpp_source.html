58void DP_Impose_PolyMAC_Face::remplir_num_faces(Entree& s)
59{
60 const Domaine& le_domaine = equation().probleme().domaine();
61 const Zone_Poly_base& zone_poly = ref_cast(Zone_Poly_base,equation().zone_dis().valeur());
62 int taille_bloc = zone_poly.nb_elem();
63 num_faces.resize(taille_bloc);
64 lire_surfaces(s,le_domaine,zone_poly,num_faces, sgn);
65// int nfac_tot = mp_sum(num_faces.size());
66 int nfac_max = (int)mp_max(num_faces.size()); //!< not to count several (number of processes) times the same face
67
68 if(je_suis_maitre() && nfac_max == 0)
69 {
70 Cerr << "Error when defining the surface plane for the singular porosity :" << finl;
71 Cerr << "No mesh faces has been found for the surface plane." << finl;
72 Cerr << "Check the coordinate of the surface plane which should match mesh coordinates." << finl;
73 exit();
74 }
75
76 DoubleTrav S;
77 zone_poly.creer_tableau_faces(S);
78 for (int i = 0; i < num_faces.size(); i++) S(num_faces(i)) = zone_poly.face_surfaces(num_faces(i));
79 surf = mp_somme_vect(S);
80}

82void DP_Impose_PolyMAC_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
83{
84 const Zone_Poly_base& zone_poly = ref_cast(Zone_Poly_base,equation().zone_dis().valeur());
85 const DoubleVect& pf = zone_poly.porosite_face(), &fs = zone_poly.face_surfaces();
86 const DoubleTab& vit = equation().inconnue().valeurs();
87 const std::string& nom_inco = equation().inconnue().le_nom().getString();
88 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
89
90// valeurs du champ de DP
91 DoubleTrav xvf(num_faces.size(), dimension), DP(num_faces.size(), 3);
92 for (int i = 0; i < num_faces.size(); i++)
93 for (int j = 0; j < dimension; j++) xvf(i, j) = zone_poly.xv()(num_faces(i), j);
94 DP_.valeur().valeur_aux(xvf, DP);
95
96 double rho = equation().milieu().masse_volumique()(0, 0), fac_rho = equation().probleme().is_dilatable() ? 1.0 : 1.0 / rho;
97
98 for (int i = 0, f; i < num_faces.size(); i++)
99 if ((f = num_faces(i)) < zone_poly.nb_faces())
100 {
101 secmem(f) += fs(f) * pf(f) * sgn(i) * (DP(i, 0) + DP(i, 1) * (surf * sgn(i) * vit(f) - DP(i, 2))) * fac_rho;
102 if (mat) (*mat)(f, f) -= fs(f) * pf(f) * DP(i, 1) * surf * fac_rho;
103 }
104
105 bilan().resize(4); //DP dDP/dQ Q Q0
106 bilan()(0) = Process::mp_max(num_faces.size() ? DP(0, 0) : -DBL_MAX);
107 bilan()(1) = Process::mp_max(num_faces.size() ? DP(0, 1) / rho : -DBL_MAX);
108 bilan()(3) = Process::mp_max(num_faces.size() ? DP(0, 2) * rho : -DBL_MAX);
109 if (Process::me()) bilan() = 0; //pour eviter un sommage en sortie
110}
