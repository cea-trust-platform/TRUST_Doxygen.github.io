89 &temp = ref_cast(Pb_Multiphase, equation().probleme()).eq_energie.inconnue().passe(),
90 &rho = equation().milieu().masse_volumique().passe(),
91 &mu = ref_cast(Fluide_base, equation().milieu()).viscosite_dynamique().passe();
92 const Milieu_composite& milc = ref_cast(Milieu_composite, equation().milieu());
93
94 DoubleTab const * d_bulles = (equation().probleme().has_champ("diametre_bulles")) ? &equation().probleme().get_champ("diametre_bulles").valeurs() : NULL ;
95
96 int e, f, c, i, j, k, l, n, N = inco.line_size(), Np = press.line_size(), d, D = dimension, nf_tot = zone.nb_faces_tot(),
97 cR = (rho.dimension_tot(0) == 1), cM = (mu.dimension_tot(0) == 1), exp_res = 2;
98 DoubleTrav a_l(N), p_l(N), T_l(N), rho_l(N), mu_l(N), sigma_l(N,N), dv(N, N), ddv(N, N, 4), ddv_c(4), d_bulles_l(N), coeff(N, N, 2); //arguments pour coeff
99 double dv_min = 0.1, dh, a_res = 1e-2;
100 const Frottement_interfacial_base& correlation_fi = ref_cast(Frottement_interfacial_base, correlation_.valeur());
101
102 /* faces */
103 for (f = 0; f < zone.nb_faces(); f++)
104 if (fcl(f, 0) < 2)
105 {
106 for (a_l = 0, p_l = 0, T_l = 0, rho_l = 0, mu_l = 0, dh = 0, sigma_l = 0, dv = dv_min, ddv = 0, d_bulles_l=0, c = 0; c < 2 && (e = f_e(f, c)) >= 0; c++)
107 {
108 for (n = 0; n < N; n++) a_l(n) += vfd(f, c) / vf(f) * alpha(e, n);
109 for (n = 0; n < N; n++) p_l(n) += vfd(f, c) / vf(f) * press(e, n * (Np > 1));
110 for (n = 0; n < N; n++) T_l(n) += vfd(f, c) / vf(f) * temp(e, n);
111 for (n = 0; n < N; n++) rho_l(n) += vfd(f, c) / vf(f) * rho(!cR * e, n);
112 for (n = 0; n < N; n++) mu_l(n) += vfd(f, c) / vf(f) * mu(!cM * e, n);
113 for (n = 0; n < N; n++)
114 for (k = 0; k < N; k++)
115 if (milc.has_interface(n,k))
116 {
117 Interface_base& sat = milc.get_interface(n, k);
118 sigma_l(n,k) += vfd(f, c) / vf(f) * sat.sigma_(temp(e,n),press(e,n * (Np > 1)));
119 }
120
121 for (n = 0; n < N; n++) dh += vfd(f, c) / vf(f) * alpha(e, n) * dh_e(e);
122 for (k = 0; k < N; k++)
123 for (l = 0; l < N; l++)
124 {
125 double dv_c = ch.v_norm(pvit, pvit, e, f, k, l, NULL, &ddv_c(0));
126 if (dv_c > dv(k, l))
127 for (dv(k, l) = dv_c, i = 0; i < 4; i++) ddv(k, l, i) = ddv_c(i);
128 }
129 for (n=0; n<N; n++) d_bulles_l(n) += (d_bulles) ? vfd(f, c) / vf(f) * (*d_bulles)(e,n) : 0;
130 }
131 correlation_fi.coefficient(a_l, p_l, T_l, rho_l, mu_l, sigma_l, dh, dv, d_bulles_l, coeff);
132 for (k = 0; k < N; k++)
133 for (l = 0; l < N; l++)
134 for (j = 0; j < 2; j++)
135 coeff(k, l, j) *= 1 + (a_l(k) > 1e-8 ? std::pow(a_l(k) / a_res, -exp_res) : 0) + (a_l(l) > 1e-8 ? std::pow(a_l(l) / a_res, -exp_res) : 0);
136
137 /* contributions : on prend le max entre les deux cotes */
138 for (k = 0; k < N; k++)
139 for (l = 0; l < N; l++)
140 if (k != l)
141 {
142 double fac = pf(f) * vf(f);
143 /* on essaie d'impliciter coeff sans ralentir la convergence en en faisant un developpement limite autour de pvit (dans la direction d'interet seulement) */
144 secmem(f, k) -= fac * (coeff(k, l, 0) * (inco(f, k) - inco(f, l)) + coeff(k, l, 1) * ddv(k, l, 3) * (pvit(f, k) - pvit(f, l)) * ((inco(f, k) - inco(f, l)) - (pvit(f, k) - pvit(f, l))));
145 if (mat)
146 for (j = 0; j < 2; j++) (*mat)(N * f + k, N * f + (j ? l : k)) += fac * (j ? -1 : 1) * (coeff(k, l, 0) + coeff(k, l, 1) * ddv(k, l, 3) * (pvit(f, k) - pvit(f, l)));
147 }
148 }
149
150 /* elements */
151 for (e = 0; e < zone.nb_elem_tot(); e++)
152 {
153 /* arguments de coeff */
154 for (n = 0; n < N; n++) a_l(n) = alpha(e, n);
155 for (n = 0; n < N; n++) p_l(n) = press(e, n * (Np > 1));
156 for (n = 0; n < N; n++) T_l(n) = temp(e, n);
157 for (n = 0; n < N; n++) rho_l(n) = rho(!cR * e, n);
158 for (n = 0; n < N; n++) mu_l(n) = mu(!cM * e, n);
159 for (n = 0; n < N; n++)
160 {
161 for (k = 0; k < N; k++)
162 if(milc.has_interface(n, k))
163 {
164 Interface_base& sat = milc.get_interface(n, k);
165 sigma_l(n,k) = sat.sigma_(temp(e,n), press(e,n * (Np > 1)));
166 }
167 }
168 for (n=0; n<N; n++) d_bulles_l(n) = (d_bulles) ? (*d_bulles)(e,n) : 0;
169
170 for (k = 0; k < N; k++)
171 for (l = 0; l < N; l++) dv(k, l) = std::max(ch.v_norm(pvit, pvit, e, -1, k, l, NULL, &ddv(k, l, 0)), dv_min);
172 correlation_fi.coefficient(a_l, p_l, T_l, rho_l, mu_l, sigma_l, dh_e(e), dv, d_bulles_l, coeff);
173 for (k = 0; k < N; k++)
174 for (l = 0; l < N; l++) coeff(k, l, 1) *= (dv(k, l) > dv_min); //pas de derivee si dv < dv_min
175 for (k = 0; k < N; k++)
176 for (l = 0; l < N; l++)
177 for (j = 0; j < 2; j++)
178 coeff(k, l, j) *= 1 + (a_l(k) > 1e-8 ? std::pow(a_l(k) / a_res, -exp_res) : 0) + (a_l(l) > 1e-8 ? std::pow(a_l(l) / a_res, -exp_res) : 0);
179
180 for (d = 0, i = nf_tot + D * e; d < D; d++, i++)
181 for (k = 0; k < N; k++)
182 for (l = 0; l < N; l++)
183 if (k != l)
184 {
185 double fac = pe(e) * ve(e);
186 /* on essaie d'impliciter coeff sans ralentir la convergence en en faisant un developpement limite autour de pvit (dans la direction d'interet seulement) */
