76// Fonction de conversion Petsc ->CSR
77PetscErrorCode Solv_AMGX::petscToCSR(Mat& A, Vec& lhs_petsc, Vec& rhs_petsc)
78{
79 PetscFunctionBeginUser;
80 PetscBool done;
81 MatType type;
82
83// Get the Mat type
84 PetscErrorCode ierr = MatGetType(A, &type);
85 CHKERRQ(ierr);
86
87// Check whether the Mat type is supported
88 if (std::strcmp(type, MATSEQAIJ) == 0) //!< sequential AIJ
89 {
90// Make localA point to the same memory space as A does
91 localA = A;
92 }
93 else if (std::strcmp(type, MATMPIAIJ) == 0)
94 {
95// Get local matrix from redistributed matrix
96 if (localA!=NULL) MatDestroy(&localA); //!< Suite accroissement memoire !
97 ierr = MatMPIAIJGetLocalMat(A, MAT_INITIAL_MATRIX, &localA);
98 CHKERRQ(ierr);
99 }
100 else
101 {
102 SETERRQ1(PETSC_COMM_WORLD, PETSC_ERR_ARG_WRONG,"Mat type %s is not supported!\n", type);
103 }
104
105// Get row and column indices in compressed row format
106 ierr = MatGetRowIJ(localA, 0, PETSC_FALSE, PETSC_FALSE, &nRowsLocal, &rowOffsets, &colIndices, &done);
107 if (done==PETSC_FALSE) Process::abort();
108 CHKERRQ(ierr);
109
110 ierr = MatSeqAIJGetArray(localA, &values);
111 CHKERRQ(ierr);
112
113// Get pointers to the raw data of local vectors
114 ierr = VecGetArray(lhs_petsc, &lhs);
115 CHKERRQ(ierr);
116 ierr = VecGetArray(rhs_petsc, &rhs);
117 CHKERRQ(ierr);
118
119// Calculate the number of rows in nRowsGlobal
120 ierr = MPI_Allreduce(&nRowsLocal, &nRowsGlobal, 1, MPI_INT, MPI_SUM, PETSC_COMM_WORLD);
121 CHKERRQ(ierr);
122
123// Store the number of non-zeros
124 nNz = rowOffsets[nRowsLocal];
125 PetscFunctionReturn(0);
126}

136// Check and return true if new stencil
137bool Solv_AMGX::check_stencil(const Matrice_Morse& mat_morse)
138{
139 std::clock_t start = std::clock();
140// Parcours de la matrice_morse (qui peut contenir des 0 et qui n'est pas triee par colonnes croissantes)
141// si matrice sur le GPU deja construite (qui est sans 0 et qui est triee par colonnes croissantes):
142 const ArrOfInt& tab1 = mat_morse.get_tab1();
143 const ArrOfInt& tab2 = mat_morse.get_tab2();
144 const ArrOfDouble& coeff = mat_morse.get_coeff();
145 const ArrOfInt& renum_array = renum_;
146 int new_stencil = 0, RowLocal = 0;
147 Journal() << "Provisoire: nb_rows_=" << nb_rows_ << " nb_rows_tot_=" << nb_rows_tot_ << finl;
148 for (int i = 0; i < tab1.size_array() - 1; i++)
149 {
150 if (items_to_keep_[i])
151 {
152 int nnz_row = 0;
153 for (int k = tab1(i) - 1; k < tab1(i + 1) - 1; k++)
154 if (coeff(k) != 0) nnz_row++;
155 if (nnz_row != rowOffsets[RowLocal + 1] - rowOffsets[RowLocal])
156 {
157 Journal() << "Provisoire: Number of non-zero on GPU will change from " << rowOffsets[RowLocal + 1] - rowOffsets[RowLocal] << " to " << nnz_row << " on row " << RowLocal << finl;
158 new_stencil = 1;
159 break;
160 }
161 else
162 {
163 for (int k = tab1(i) - 1; k < tab1(i + 1) - 1; k++)
164 {
165 if (coeff(k) != 0)
166 {
167 bool found = false;
168 int col = renum_array[tab2(k) - 1];
169// Boucle pour voir si le coeff est sur le GPU:
170 int RowGlobal = decalage_local_global_+RowLocal;
171 for (int kk = rowOffsets[RowLocal]; kk < rowOffsets[RowLocal + 1]; kk++)
172 {
173 if (colIndices[kk] == col)
174 {
175 values[kk] = coeff(k); //!< On met a jour le coefficient
176 found = true;
177 break;
178 }
179 }
180 if (!found)
181 {
182 Journal() << "Provisoire: mat_morse(" << RowGlobal << "," << col << ")!=0 new on GPU " << finl;
183 new_stencil = 1;
184 break;
185 }
186 }
187 }
188 }
189 RowLocal++;
190 }
191 }
192 new_stencil = mp_max(new_stencil);
193 Cout << "[AmgX] Time to check stencil: " << (std::clock() - start) / (double) CLOCKS_PER_SEC << finl;
194 return new_stencil;
195}

197// Resolution
198int Solv_AMGX::solve(ArrOfDouble& residu)
199{
200// Calcul du seuil avant resolution pour eviter des soucis:
201 Vec ResidualPetsc_;
202 VecDuplicate(SolutionPetsc_, &ResidualPetsc_);
203 MatResidual(MatricePetsc_, SecondMembrePetsc_, SolutionPetsc_, ResidualPetsc_);
204 VecNorm(ResidualPetsc_, NORM_2, &residu(0));
205 VecDestroy(&ResidualPetsc_);
206 if (residu(0)==0) return 0;
207 else if (residu(0) < seuil_)
208 {
209// Crash d'AmgX (non reproduit sur les cas poisson d'AmgXWrapper), peut etre a cause d'une creation differente
210// des vecteurs PETSc (via DM) par rapport a la notre (VecCreate), si deja converge a cause d'un seuil absolu trop haut (nbiter=0)
211// Contournement en calculant le residu avant le solve et on ne resout pas si inferieur au seuil:
212 Cerr << "[AmgX] The residual seems to small to be solved on GPU: ||Ax-b||="<<residu(0)<<"<"<<seuil_<< finl;
213 Cerr << "Please, try to use a relative tolerance, with rtol option or lower the absolute tolerance atol." << finl;
214 return 0;
215// Process::exit();
216 }
217 std::clock_t start = std::clock();
218 SolveurAmgX_.solve(lhs, rhs, nRowsLocal);
219 Cout << "[AmgX] Time to solve system on GPU: " << ( std::clock() - start ) / (double) CLOCKS_PER_SEC << finl;
220 return nbiter(residu);
221}
