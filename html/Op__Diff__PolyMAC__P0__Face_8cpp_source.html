64double Op_Diff_PolyMAC_P0_Face::calculer_dt_stab() const
65{
66 const Zone_PolyMAC_P0& zone = la_zone_poly_.valeur();
67 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins(), &fcl = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur()).fcl();
68 const DoubleTab& nf = zone.face_normales(), &vfd = zone.volumes_entrelaces_dir(),
69 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL,
70 *a_r = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.champ_conserve().passe() : (has_champ_masse_volumique() ? &get_champ_masse_volumique().valeurs() : NULL); /* produit alpha * rho */
71 const DoubleVect& pe = zone.porosite_elem(), &pf = zone.porosite_face(), &vf = zone.volumes_entrelaces(), &ve = zone.volumes();
72 update_nu();
73
74 int i, e, f, n, N = equation().inconnue().valeurs().line_size();
75 double dt = 1e10;
76 DoubleTrav flux(N), vol(N);
77 for (e = 0; e < zone.nb_elem(); e++)
78 {
79 for (flux = 0, vol = pe(e) * ve(e), i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
80 for (n = 0; n < N; n++)
81 {
82 flux(n) += zone.nu_dot(&nu_, e, n, &nf(f, 0), &nf(f, 0)) / vf(f);
83 if (fcl(f, 0) < 2) vol(n) = std::min(vol(n), pf(f) * vf(f) / vfd(f, e != f_e(f, 0)) * vf(f)); //cf. Op_Conv_EF_Stab_PolyMAC_P0_Face.cpp
84 }
85 for (n = 0; n < N; n++)
86 if ((!alp || (*alp)(e, n) > 0.25) && flux(n)) /* sous 0.5e-6, on suppose que l'evanescence fait le job */
87 dt = std::min(dt, vol(n) * (a_r ? (*a_r)(e, n) : 1) / flux(n));
88 }
89 return Process::mp_min(dt);
90}

92void Op_Diff_PolyMAC_P0_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
93{

95 const Zone_PolyMAC_P0& zone = la_zone_poly_.valeur();
96 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces(), &fcl = ch.fcl();
97 const DoubleTab& nf = zone.face_normales();
98 const DoubleVect& fs = zone.face_surfaces();
99
100 const std::string& nom_inco = ch.le_nom().getString();
101 if (!matrices.count(nom_inco) || semi_impl.count(nom_inco)) return; //semi-implicite ou pas de bloc diagonal -> rien a faire
102 Matrice_Morse& mat = *matrices.at(nom_inco), mat2;
103
104 int i, j, k, i_f, e, e_s, f, fb, fc, f_s, n, N = ch.valeurs().line_size(), ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), d, D = dimension, c;
105
106 IntTrav stencil(0, 2), tpfa(0, N);
107 stencil.set_smart_resize(1), zone.creer_tableau_faces(tpfa);
108
109 /* stencils du flux : ceux (reduits) de update_nu si nu constant ou scalaire, ceux (complets) de la zone sinon */
110 update_phif(!nu_constant_); //si nu variable, stencil complet
111 Cerr << "Op_Diff_PolyMAC_P0_Face::dimensionner() : ";
112
113 for (f = 0; f < zone.nb_faces(); f++)
114 if (fcl(f, 0) < 2)
115 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++) /* op. aux faces : contrib de l'elem e */
116 {
117// recherche de i_f : indice de f dans l'element e
118 for (i_f = -1, j = 0; j < e_f.dimension(1) && (fb = e_f(e, j)) >= 0; j++)
119 if (fb == f) i_f = j;
120// contribution de la diffusion a la face fb
121 for (j = 0; j < e_f.dimension(1) && (fb = e_f(e, j)) >= 0; j++)
122 {
123 for (c = (e != f_e(fb, 0)), k = phif_d(fb); k < phif_d(fb + 1); k++)
124 {
125 if ((e_s = phif_e(k)) >= ne_tot) continue; //bord -> pas de terme de matrice
126 if ((fc = zone.equiv(fb, c, i_f)) >= 0 && fcl(f_s = e_s == e ? f : fc, 0) < 2) /* amont/aval si equivalence : operateur entre faces */
127 for (n = 0; n < N; n++) stencil.append_line(N * f + n, N * f_s + n);
128 /* sinon : elem -> face, avec un traitement particulier de e_s == e pour eviter les modes en echiquier dans la diffusion */
129 for (d = 0; d < D; d++)
130 if (std::fabs(nf(f, d)) > 1e-6 * fs(f))
131 for (n = 0; n < N; n++)
132 stencil.append_line(N * f + n, N * (nf_tot + D * e_s + d) + n);
133 if (e_s == e)
134 for (n = 0; n < N; n++) stencil.append_line(N * f + n, N * f + n);
135 }
136
137 }
138 }
139
140 for (e = 0; e < ne_tot; e++)
141 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++) /* aux elements : on doit traiter aussi les elems virtuels */
142 for (j = phif_d(f); j < phif_d(f + 1); j++)
143 if ((e_s = phif_e(j)) < ne_tot) //contrib d'un element
144 for (d = 0; d < D; d++)
145 for (n = 0; n < N; n++) stencil.append_line(N * (nf_tot + D * e + d) + n, N * (nf_tot + D * e_s + d) + n);
146
147 for (tpfa = 1, f = 0; f < zone.nb_faces(); f++)
148 for (i = phif_d(f); i < phif_d(f + 1); i++)
149 if ((e_s = phif_e(i)) < ne_tot && e_s != f_e(f, 0) && e_s != f_e(f, 1))
150 for (n = 0; n < N; n++)
151 if (phif_c(i, n)) tpfa(f, n) = 0;
152
153 tableau_trier_retirer_doublons(stencil);
154 Cerr << "width " << Process::mp_sum(stencil.dimension(0)) * 1. / (N * (nf_tot + D * ne_tot))
155 << " " << mp_somme_vect(tpfa) * 100. / (N * zone.md_vector_faces().valeur().nb_items_seq_tot()) << "% TPFA " << finl;
156 Matrix_tools::allocate_morse_matrix(N * (nf_tot + ne_tot * D), N * (nf_tot + ne_tot * D), stencil, mat2);
157 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
158}

162void Op_Diff_PolyMAC_P0_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
163{
164 const std::string& nom_inco = equation().inconnue().le_nom().getString();
165 Matrice_Morse *mat = matrices.count(nom_inco) && !semi_impl.count(nom_inco) ? matrices[nom_inco] : NULL; //facultatif
166 const DoubleTab& inco = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : le_champ_inco.non_nul() ? le_champ_inco->valeurs() : equation().inconnue().valeurs();
167 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
168 const Zone_PolyMAC_P0& zone = la_zone_poly_.valeur();
169 const Conds_lim& cls = la_zcl_poly_->les_conditions_limites();
170 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces(), &fcl = ch.fcl();
171 const DoubleVect& fs = zone.face_surfaces(), &vf = zone.volumes_entrelaces(), &ve = zone.volumes(), &pf = porosite_f, &pe = porosite_e;
172 const DoubleTab& nf = zone.face_normales(), &xp = zone.xp(), &xv = zone.xv(), &vfd = zone.volumes_entrelaces_dir();
173 int i, j, i_f, c, e_s, f_s, k, e, f, fb, fc, n, N = inco.line_size(), d, D = dimension, ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), sgn;
174
175 update_phif();
176
177 DoubleTrav coeff(N), fac(N);
178 for (f = 0; f < zone.nb_faces(); f++)
179 if (fcl(f, 0) < 2)
180 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++) /* op. aux faces : contrib de l'elem e */
181 {
182// recherche de i_f : indice de f dans l'element e
183 for (i_f = -1, j = 0; i_f < 0 && j < e_f.dimension(1) && (fb = e_f(e, j)) >= 0; j++)
184 if (fb == f) i_f = j;
185// contribution de la diffusion a la face fb
186 for (j = 0; j < e_f.dimension(1) && (fb = e_f(e, j)) >= 0; j++)
187 {
188 int tpfa = 1;
189 for (k = phif_d(fb); k < phif_d(fb + 1); k++)
190 if ((e_s = phif_e(k)) < ne_tot && e_s != f_e(fb, 0) && e_s != f_e(fb, 1)) tpfa = 0;
191 double df_sur_d = tpfa && fcl(fb, 0) ? std::max(std::fabs(zone.dot(&xv(fb, 0), &nf(fb, 0), &xv(f, 0)) / zone.dot(&xv(fb, 0), &nf(fb, 0), &xp(e, 0))) , 1.) : 1;
192 for (c = (e != f_e(fb, 0)), n = 0; n < N; n++) coeff(n) = vfd(f, i) / ve(e) * fs(fb) * (c ? 1 : -1) / df_sur_d; /* prefacteur diff elem -> diff face */
193 for (k = phif_d(fb); k < phif_d(fb + 1); k++)
194 {
195 for (n = 0; n < N; n++) fac(n) = coeff(n) * phif_c(k, n);
196 if ((e_s = phif_e(k)) >= ne_tot) /* contrib. d'un bord : seul Dirichlet fait quelque chose */
197 {
198 if (fcl(f_s = e_s - ne_tot, 0) == 3 && sub_type(Dirichlet, cls[fcl(f_s, 1)].valeur()))
199 for (d = 0; d < D; d++)
200 for (n = 0; n < N; n++)
201 secmem(f, n) -= fac(n) * ref_cast(Dirichlet, cls[fcl(f_s, 1)].valeur()).val_imp(fcl(f_s, 2), N * d + n) * nf(f, d) / fs(f);
202 }
203 else if (tpfa && (fc = zone.equiv(fb, c, i_f)) >= 0) /* amont/aval si equivalence : operateur entre faces */
204 {
205 f_s = e_s == e ? f : fc, sgn = zone.dot(&nf(f_s, 0), &nf(f, 0)) > 0 ? 1 : -1; //sgn = 1 si f et f_s ont la meme orientation par rapport a e / e_s
206 for (n = 0; n < N; n++) secmem(f, n) -= fac(n) * sgn * pf(f_s) / pe(e_s) * inco(f_s, n);
207 if (mat && fcl(f_s, 0) < 2)
208 for (n = 0; n < N; n++) (*mat)(N * f + n, N * f_s + n) += fac(n) * sgn * pf(f_s) / pe(e_s);
209 }
210 else /* sinon : elem -> face, avec un traitement particulier de e_s == e pour eviter les modes en echiquier dans la diffusion */
211 {
212 double f_eps = e_s != e ? 0 : tpfa && fcl(fb, 0) ? 1 : std::min(eps, 1000 * std::pow(vf(f) / fs(f), 2));
213 for (d = 0; d < D; d++)
214 for (n = 0; n < N; n++)
215 secmem(f, n) -= fac(n) * (1 - f_eps) * inco(nf_tot + D * e_s + d, n) * nf(f, d) / fs(f);
216 if (mat)
217 for (d = 0; d < D; d++)
218 if (std::fabs(nf(f, d)) > 1e-6 * fs(f))
219 for (n = 0; n < N; n++)
220 (*mat)(N * f + n, N * (nf_tot + D * e_s + d) + n) += fac(n) * (1 - f_eps) * nf(f, d) / fs(f);
221 if (!f_eps) continue;
222 for (n = 0; n < N; n++) secmem(f, n) -= fac(n) * f_eps * inco(f, n);
223 if (mat)
224 for (n = 0; n < N; n++) (*mat)(N * f + n, N * f + n) += fac(n) * f_eps;
225 }
226 }
227 }
228 }
229
230 for (e = 0; e < ne_tot; e++)
231 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++) /* aux elements : on doit traiter aussi les elems virtuels */
232 for (c = (e != f_e(f, 0)), j = phif_d(f); j < phif_d(f + 1); j++)
233 {
234 for (n = 0; n < N; n++) coeff(n) = fs(f) * (c ? 1 : -1) * phif_c(j, n);
235 if ((e_s = phif_e(j)) < ne_tot) //contrib d'un element
236 {
237 for (d = 0; d < D; d++)
238 for (n = 0; n < N; n++) secmem(nf_tot + D * e + d, n) -= coeff(n) * inco(nf_tot + D * e_s + d, n);
239 if (mat)
240 for (d = 0; d < D; d++)
241 for (n = 0; n < N; n++) (*mat)(N * (nf_tot + D * e + d) + n, N * (nf_tot + D * e_s + d) + n) += coeff(n);
242 }
243 else if (fcl(f_s = e_s - ne_tot, 0) == 3 && sub_type(Dirichlet, cls[fcl(f_s, 1)].valeur())) //contrib d'un bord : seul Dirichlet contribue
244 for (d = 0; d < D; d++)
245 for (n = 0; n < N; n++) secmem(nf_tot + D * e + d, n) -= coeff(n) * ref_cast(Dirichlet, cls[fcl(f_s, 1)].valeur()).val_imp(fcl(f_s, 2), N * d + n);
246 }
