49void MorseSymToMorse(const Matrice_Morse_Sym& MS, Matrice_Morse& M)
50{
51 M = MS;
52 Matrice_Morse mattmp(MS);
53 int i, ordre;
54 ordre = M.ordre();
55 M.transpose(mattmp);
56 for (i = 0; i < ordre; i++)
57 M(i, i) = 0.;
58 M = mattmp + M;
59}

62void ILU::prepare_(const Matrice_Base& la_matrice, const DoubleVect& v)
63{
64 if (get_status() != READY)
65 {
66 if(sub_type(Matrice_Morse_Sym,la_matrice))
67 {
68 Cerr<<"ILU ERROR : one is not able to carry out a ILU preconditioning"
69 <<" on a symetric matrix"<<finl;
70 exit();
71 }
72 else if(sub_type(Matrice_Morse,la_matrice))
73 {
74 const Matrice_Morse& m = ref_cast(Matrice_Morse,la_matrice);
75 factorisation(m, v);
76 }
77 else if(sub_type(Matrice_Bloc,la_matrice))
78 {
79 const Matrice_Bloc& matrice = ref_cast(Matrice_Bloc,la_matrice);
80 const Matrice& bloc0 = matrice.get_bloc(0,0);
81 if(sub_type(Matrice_Morse_Sym,bloc0.valeur()))
82 {
83 /*
84 Cerr<<"ILU ERROR : one is not able to carry out a ILU preconditioning"
85 <<" on a symetric matrix"<<finl;
86 exit();
87 */
88 const Matrice_Morse_Sym ms=ref_cast(Matrice_Morse_Sym,bloc0.valeur());
89 Matrice_Morse m2;
90 MorseSymToMorse(ms,m2);
91 factorisation(m2,v);
92 }
93 else if(sub_type(Matrice_Morse,bloc0.valeur()))
94 {
95 const Matrice_Morse& m = ref_cast(Matrice_Morse,bloc0.valeur());
96 factorisation(m, v);
97 }
98 }
99 else
100 {
101 Cerr<<"ILU : Warning : preconditioning can be carry out only for linear systems based on Matrice_Morse or Matrice_Bloc type matrixes"<<finl;
102 exit();
103 }
104 }
105 Precond_base::prepare_(la_matrice, v);
106}

183int ILU::factorisation(const Matrice_Morse& mat,const DoubleVect& secmem)
184{
185 if (nproc()==1)
186 {
187 return factoriser(mat,secmem);
188 }
189 if(1)
190 {
191// dimensionnenment de la matrice MLOC
192 int n_reel = secmem.size();
193// int n_tot = secmem.size_array();
194 MLOC.dimensionner(n_reel,0);
195 int ii,jj;
196 int cpt=0;
197 const IntVect& tab1_tot = mat.get_tab1();
198 const IntVect& tab2_tot = mat.get_tab2();
199// const DoubleVect& coeff_tot = mat.get_set_coeff();
200 IntVect& tab1_reel = MLOC.get_set_tab1();
201 IntVect& tab2_reel = MLOC.get_set_tab2();
202// DoubleVect& coeff_reel = MLOC.get_set_coeff();
203 tab1_reel(0)=1;
204 for(ii=0; ii<n_reel; ii++)
205 {
206 int dl = tab1_tot(ii);
207 int fl = tab1_tot(ii+1);
208 for(jj=dl; jj<fl; jj++)
209 {
210 if(tab2_tot(jj-1)<=n_reel)
211 cpt++;
212 }
213 tab1_reel(ii+1)=cpt+1;
214 }
215 MLOC.dimensionner(n_reel,tab1_reel(n_reel)-1);
216 cpt=0;
217 for(ii=0; ii<n_reel; ii++)
218 {
219 int dl = tab1_tot(ii);
220 int fl = tab1_tot(ii+1);
221 for(jj=dl; jj<fl; jj++)
222 {
223 if(tab2_tot(jj-1)<=n_reel)
224 {
225 tab2_reel(cpt++)=tab2_tot(jj-1);
226// coeff_reel(cpt++)=coeff_tot(jj-1);
227 }
228 }
229 }
230 }
231// copie des coeffs
232 {
233 int ii,jj;
234 int cpt=0;
235 int n_reel = secmem.size();
236
237 const IntVect& tab1_tot = mat.get_tab1();
238 const IntVect& tab2_tot = mat.get_tab2();
239 const DoubleVect& coeff_tot = mat.get_coeff();
240// IntVect& tab1_reel = MLOC.get_set_tab1();
241// IntVect& tab2_reel = MLOC.get_set_tab2();
242 DoubleVect& coeff_reel = MLOC.get_set_coeff();
243
244 for(ii=0; ii<n_reel; ii++)
245 {
246 int dl = tab1_tot(ii);
247 int fl = tab1_tot(ii+1);
248 for(jj=dl; jj<fl; jj++)
249 {
250 if(tab2_tot(jj-1)<=n_reel)
251 {
252// tab2_reel(cpt)=tab2_tot(jj-1);
253 coeff_reel(cpt++)=coeff_tot(jj-1);
254 }
255 }
256 }
257 }
258
259 return factoriser(MLOC,secmem);
260}
