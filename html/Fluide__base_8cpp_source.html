114void Fluide_base::discretiser(const Probleme_base& pb, const Discretisation_base& dis)
115{
116 const Zone_dis_base& zone_dis=pb.equation(0).zone_dis();
117// mu rho nu revoir
118 double temps=pb.schema_temps().temps_courant();
119 if (mu.non_nul())
120 if (sub_type(Champ_Fonc_MED,mu.valeur()))
121 {
122 Cerr<<" on convertit le champ_fonc_med en champ_don"<<finl;
123 Champ_Don mu_prov;
124 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,mu_prov);
125 mu_prov.affecter_(mu.valeur());
126 mu.detach();
127 nu.detach();
128 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,mu);
129
130 mu.valeur().valeurs()=mu_prov.valeur().valeurs();
131
132 }
133 if (!mu.non_nul())
134 {
135 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,mu);
136 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,nu);
137 }
138 if (mu.non_nul())
139 {
140 dis.nommer_completer_champ_physique(zone_dis,"viscosite_dynamique","kg/m/s",mu.valeur(),pb);
141 champs_compris_.ajoute_champ(mu.valeur());
142 }
143 if (sub_type(Champ_Fonc_Tabule,mu.valeur()))
144 {
145 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,nu);
146 }
147 if (!nu.non_nul())
148 dis.discretiser_champ("champ_elem",zone_dis,"neant","neant",1,temps,nu);
149
150 if (nu.non_nul())
151 {
152 dis.nommer_completer_champ_physique(zone_dis,"viscosite_cinematique","m2/s",nu.valeur(),pb);
153 champs_compris_.ajoute_champ(nu.valeur());
154 }
155 if (beta_co.non_nul())
156 {
157 dis.nommer_completer_champ_physique(zone_dis,"dilatabilite_solutale",".",beta_co.valeur(),pb);
158 champs_compris_.ajoute_champ(beta_co.valeur());
159 }
160
161 Milieu_base::discretiser(pb,dis);
162}

184void Fluide_base::verifier_coherence_champs(int& err,Nom& msg)
185{
186 msg="";
187 if (rho.non_nul())
188 {
189 if (mp_min_vect(rho.valeurs()) <= 0)
190 {
191 msg += "The density rho is not striclty positive. \n";
192 err = 1;
193 }
194 }
195 else
196 {
197 msg += "The density rho has not been specified. \n";
198 err = 1;
199 }
200 if ( (Cp.non_nul()) && ( (lambda.non_nul()) && (beta_th.non_nul()) ) ) //!< Fluide anisotherme
201 {
202// if (mp_min_vect(Cp.valeurs()) <= 0)
203// {
204// msg += "The heat capacity Cp is not striclty positive. \n";
205// err = 1;
206// }
207// if (mp_min_vect(lambda.valeurs()) <= 0)
208// {
209// msg += "The conductivity lambda is not positive. \n";
210// err = 1;
211// }
212 if (sub_type(Champ_Uniforme,Cp.valeur()))
213 {
214 if (Cp(0,0) <= 0)
215 {
216 msg += "The heat capacity Cp is not striclty positive. \n";
217 err = 1;
218 }
219 }
220 else
221 {
222 msg += "The heat capacity Cp is not of type Champ_Uniforme. \n";
223 err = 1;
224 }
225 if (sub_type(Champ_Uniforme,lambda.valeur()))

228 {
229 msg += "The conductivity lambda is not positive. \n";

231 }
232 }
233
234 }
235 if ( ( (Cp.non_nul()) || (beta_th.non_nul()) ) && (!lambda.non_nul()) )
236 {
237 msg += " Physical properties for an anisotherm case : \n";
238 msg += "the conductivity lambda has not been specified. \n";
239 err = 2;
240 }
241 if ( ( (lambda.non_nul()) || (beta_th.non_nul()) ) && (!Cp.non_nul()) )
242 {
243 msg += " Physical properties for an anisotherm case : \n";
244 msg += "the heat capacity Cp has not been specified. \n";
245 err = 2;
246 }
247 if ( ( (lambda.non_nul()) || (Cp.non_nul()) ) && (!beta_th.non_nul()) )
248 {
249 msg += " Physical properties for an anisotherm case : \n";
250 msg += "the thermal expansion coefficient beta_th has not been specified. \n";
251 err = 2;
252 }
253
254// Test de la coherence des proprietees radiatives du fluide incompressible
255// (pour un milieu semi transparent
256 if ( (coeff_absorption_.non_nul()) && (!indice_refraction_.non_nul()) )
257 {
258 msg += " Physical properties for semi tranparent radiation case : \n";
259 msg += "Refraction index has not been specfied while it has been done for absorption coefficient. \n";

264 msg += " Physical properties for semi tranparent radiation case : \n";

280 Milieu_base::verifier_coherence_champs(err,msg);

311void Fluide_base::calculer_nu()
312{
313 const DoubleTab& tabmu = mu.valeurs();
314 const DoubleTab& tabrho = rho.valeurs();
315 DoubleTab& tabnu = nu.valeurs();
316
317 int cRho = sub_type(Champ_Uniforme, rho.valeur()), cMu = sub_type(Champ_Uniforme, mu.valeur()),
318 i, j, n, Nl = tabnu.dimension_tot(0), N = tabnu.line_size();
319 /* valeurs : mu / rho */
320 for (i = j = 0; i < Nl; i++)
321 for (n = 0; n < N; n++, j++) tabnu.addr()[j] = tabmu.addr()[cMu ? n : j] / tabrho.addr()[cRho ? n : j];

352//! Effectue une mise a jour en temps du milieu,<br>et donc de ses parametres caracteristiques.<br>Les champs uniformes sont recalcules pour le<br>nouveau temps specifie, les autres sont mis a<br>par un appel a CLASSE_DU_CHAMP::mettre_a_jour(double temps).
353void Fluide_base::mettre_a_jour(double temps)
354{
355 Milieu_base::mettre_a_jour(temps);
356 if (beta_co.non_nul())
357 beta_co.mettre_a_jour(temps);
358 mu.mettre_a_jour(temps);
359 calculer_nu();
360 nu.valeur().changer_temps(temps);
361 nu->valeurs().echange_espace_virtuel();
362 if (e_int_auto_) e_int.mettre_a_jour(temps);
363
364// Mise a jour des proprietes radiatives du fluide incompressible
365// (Pour un fluide incompressible semi transparent).
366
367 if (coeff_absorption_.non_nul() && indice_refraction_.non_nul())
368 {
369 coeff_absorption_.mettre_a_jour(temps);
370 indice_refraction_.mettre_a_jour(temps);
371
372// Mise a jour de longueur_rayo
373 longueur_rayo_.mettre_a_jour(temps);
374
375 if (sub_type(Champ_Uniforme,kappa().valeur()))
376 {
377 longueur_rayo()->valeurs()(0,0)=1/(3*kappa()(0,0));
378 longueur_rayo()->valeurs().echange_espace_virtuel();
379 }
380 else
381 {
382 DoubleTab& l_rayo = longueur_rayo_.valeurs();
383 const DoubleTab& K = kappa().valeurs();
384 for (int i=0; i<kappa().nb_valeurs_nodales(); i++)
385 l_rayo[i] = 1/(3*K[i]);
386 l_rayo.echange_espace_virtuel();
387 }
388 }
389}

490void Fluide_base::calculer_e_int(const Objet_U& obj, DoubleTab& val, DoubleTab& bval, tabs_t& deriv)
491{
492 const Fluide_base& fl = ref_cast(Fluide_base, obj);
493 const Champ_base& ch_T = fl.equation("temperature").inconnue().valeur(), &ch_Cp = fl.capacite_calorifique();
494 const DoubleTab& T = ch_T.valeurs(), &Cp = ch_Cp.valeurs();
495 int i, zero = 0, Ni = T.dimension_tot(0), Nb = bval.dimension_tot(0), n, N = fl.id_composite >= 0 ? 1 : Cp.dimension(1),
496 n0 = std::max(fl.id_composite, zero), cCp = Cp.dimension_tot(0) == 1;
497
498 for (i = 0; i < Ni; i++)
499 for (n = 0; n < N; n++) val(i, n) = fl.h0_ + Cp(!cCp * i, n) * (T(i, n0 + n) - fl.T0_);
500 DoubleTab bT = ch_T.valeur_aux_bords(), bCp;
501 if (ch_Cp.a_une_zone_dis_base()) bCp = ch_Cp.valeur_aux_bords();
502 else bCp.resize(bval.dimension_tot(0), N), ch_Cp.valeur_aux(ref_cast(Zone_VF, ch_T.zone_dis_base()).xv_bord() , bCp);
503 for (i = 0; i < Nb; i++)
504 for (n = 0; n < N; n++) bval(i, n) = fl.h0_ + bCp(i, n) * (bT(i, n0 + n) - fl.T0_);
505
506 DoubleTab& der_T = deriv[ch_T.le_nom().getString()];
507 for (der_T.resize(Ni, N), i = 0; i < Ni; i++)
508 for (n = 0; n < N; n++) der_T(i, n) = Cp(!cCp * i, n);
509}
