53DoubleTab& Perte_Charge_VEF::ajouter(DoubleTab& resu) const
54{
55 const Zone_VEF& zvef=la_zone_VEF.valeur();
56 const Champ_Don& nu=le_fluide->viscosite_cinematique(); //!< viscosite cinematique
57 const DoubleTab& xv=zvef.xv() ; //!< centres de gravite des faces
58 const DoubleTab& vit=la_vitesse->valeurs();
59// Sinon segfault a l'initialisation de ssz quand il n'y a pas de sous-zone !
60 const Sous_zone_VF& ssz = sous_zone ? la_sous_zone_dis.valeur() : Sous_zone_VF();
61
62// Parametres pour perte_charge()
63 DoubleVect u(dimension);
64 DoubleVect pos(dimension);
65 DoubleVect v_valeur(dimension);
66// Optimisations pour les cas ou nu ou diam_hydr sont constants
67 bool nu_constant=sub_type(Champ_Uniforme,nu.valeur())?true:false;
68
69 bool dh_constant=sub_type(Champ_Uniforme,diam_hydr.valeur())?true:false;
70
71// Le temps actuel
72 double t=equation().schema_temps().temps_courant();
73
74// Nombre de faces a traiter.
75 int max_faces = sous_zone ? ssz.les_faces().size() : zvef.nb_faces();
76
77 for (int face=0; face<max_faces; face++)
78 {
79// indice de la face dans la zone_VEF
80 int la_face = sous_zone ? ssz.les_faces()[face] : face;
81
82 if (la_face<zvef.nb_faces())
83 {
84// Recup la vitesse a la face, et en calcule le module
85 double norme_u=0;
86 for (int dim=0; dim<dimension; dim++)
87 {
88 u[dim]=vit(la_face,dim);
89 norme_u+=u[dim]*u[dim];
90 }
91 norme_u=sqrt(norme_u) ;
92
93// Calcul de la position
94 for (int i=0; i<dimension; i++)
95 pos[i]=xv(la_face,i);
96
97// Calcul de nu
98 double nu_valeur = nu_constant ? nu(0,0) : nu->valeur_a_compo(pos,0);
99
100// Calcul du diametre hydraulique
101 double dh_valeur = dh_constant ? diam_hydr(0,0) : diam_hydr->valeur_a_compo(pos,0);
102
103// Calcul du reynolds
104 double reynolds=norme_u*dh_valeur/nu_valeur;
105// Lambda est souvent indetermine pour Re->0
106 if (reynolds < 1.e-10)
107 reynolds=1e-10;
108
109// Calcul du volume d'integration
110 double volume=sous_zone?
111 ssz.volumes_entrelaces(face) :
112 zvef.volumes_entrelaces(la_face);
113 volume*=zvef.porosite_face(la_face);
114
115// Calcul du resultat final (ouf)
116 double coeff_ortho,coeff_long,u_l;
117 coeffs_perte_charge(u,pos,t,norme_u,dh_valeur,nu_valeur,reynolds,coeff_ortho,coeff_long,u_l,v_valeur);
118 for (int dim=0; dim<dimension; dim++)
119 {
120// La perte de charge vaut -coeff_long*u_l*v_valeur[dim] -coeff_ortho(u[dim] -u_v*v_valeur[dim])
121// soit -coeff_ortho* u[dim] - (coeff_long-coeff_ortho)* u_l*v_valeur[dim]
122 resu(la_face,dim)-=(coeff_ortho* u[dim] + (coeff_long-coeff_ortho)* u_l*v_valeur[dim])*volume;
123 }
124 }
125 }
126 return resu;
127}

130void Perte_Charge_VEF::contribuer_a_avec(const DoubleTab& inco, Matrice_Morse& matrice) const
131{
132 if (implicite_==0) return; //!< La perte de charge n'est pas implicitee, on quitte
133
134// Raccourcis
135 const Champ_Don& nu=le_fluide->viscosite_cinematique(); //!< viscosite cinematique
136 const DoubleTab& xv=la_zone_VEF->xv() ; //!< centres de gravite des faces
137 const DoubleTab& vit=la_vitesse->valeurs();
138// Sinon segfault a l'initialisation de ssz quand il n'y a pas de sous-zone !
139 const Sous_zone_VF& ssz=sous_zone?la_sous_zone_dis.valeur():Sous_zone_VF();
140 const Zone_VEF& zvef=la_zone_VEF.valeur();
141
142// Parametres pour perte_charge()
143 DoubleVect u(dimension);
144 double norme_u;
145 double dh_valeur;
146 double reynolds;
147 DoubleVect pos(dimension);
148 DoubleVect v_valeur(dimension);
149// Optimisations pour les cas ou nu ou diam_hydr sont constants
150 bool nu_constant=sub_type(Champ_Uniforme,nu.valeur())?true:false;
151 double nu_valeur;
152 bool dh_constant=sub_type(Champ_Uniforme,diam_hydr.valeur())?true:false;
153
154// Le temps actuel
155 double t=equation().schema_temps().temps_courant();
156
157// Nombre de faces a traiter.
158 int max_faces=sous_zone?
159 ssz.les_faces().size() :
160
161 zvef.nb_faces();
162
163// To ensure that nu_valeur will not be zero
164// and not have a division by zero for the calculation of Reynolds
165 assert_espace_virtuel_vect(nu->valeurs());
166
167 for (int face=0; face<max_faces; face++)
168 {
169
170// indice de la face dans la zone_VEF
171 int la_face=sous_zone?
172 ssz.les_faces()[face] :
173 face;
174
175// Fixed bug by GF: si la face est une face virtuelle on passe, car nu mal calcule et coef(no,no) invalide
176 if (la_face>=zvef.nb_faces())
177 continue;
178
179// Recup la vitesse a la face, et en calcule le module
180 norme_u=0;
181 for (int dim=0; dim<dimension; dim++)
182 {
183 u[dim]=vit(la_face,dim);
184 norme_u+=u[dim]*u[dim];
185 }
186 norme_u=sqrt(norme_u) ;
187
188// Calcul de la position
189 for (int i=0; i<dimension; i++)
190 pos[i]=xv(la_face,i);
191
192// Calcul de nu
193 if (!nu_constant)
194 {
195 nu_valeur=nu->valeur_a_compo(pos,0);
196 }
197 else nu_valeur=nu(0,0);
198
199// Calcul du diametre hydraulique
200 if (!dh_constant)
201 {
202 dh_valeur=diam_hydr->valeur_a_compo(pos,0);
203 }
204 else dh_valeur=diam_hydr(0,0);
205
206// Calcul du reynolds
207 reynolds=norme_u*dh_valeur/nu_valeur;
208// Lambda est souvent indetermine pour Re->0
209 if (reynolds < 1.e-10)
210 reynolds=1e-10;
211
212// Calcul du volume d'integration
213 double volume=sous_zone?
214 ssz.volumes_entrelaces(face) :
215 zvef.volumes_entrelaces(la_face);
216 volume*=zvef.porosite_face(la_face);
217
218// Calcul du resultat final (ouf)
219 double coeff_ortho,coeff_long,u_l;
220 coeffs_perte_charge(u,pos,t,norme_u,dh_valeur,nu_valeur,reynolds,coeff_ortho,coeff_long,u_l,v_valeur);
221 for (int dim=0; dim<dimension; dim++)
222 {
223 int n0=la_face*dimension+dim;
224// matrice.coef(n0,n0) += (coeff_ortho + (coeff_long-coeff_ortho)* v_valeur[dim]*v_valeur[dim]); // Fixed bug
225 matrice.coef(n0,n0) += (coeff_ortho + (coeff_long-coeff_ortho)* v_valeur[dim]*v_valeur[dim])*volume;
226// La perte de charge vaut -coeff_long*u_l*v_valeur[dim] -coeff_ortho(u[dim] -u_v*v_valeur[dim])
227// soit -coeff_ortho* u[dim] - (coeff_long-coeff_orho)* u_l*v_valeur[dim]
228// resu(la_face,dim)-=(coeff_ortho* u[dim] + (coeff_long-coeff_orho)* u_l*v_valeur[dim]);
229 }
230 }
231}

240void Perte_Charge_VEF::completer()
241{
242 Source_base::completer();
243
244 if (sous_zone)
245 {
246 sous_zone=false;
247 const Sous_Zone& la_sous_zone=equation().probleme().domaine().ss_zone(nom_sous_zone);
248 const Domaine_dis& le_domaine_dis=la_zone_VEF->domaine_dis();
249 for (int ssz=0; ssz<le_domaine_dis.nombre_de_sous_zones_dis(); ssz++)
250 {
251 if (le_domaine_dis.sous_zone_dis(ssz)->sous_zone().est_egal_a(la_sous_zone))
252 {
253 sous_zone=true;
254 la_sous_zone_dis=ref_cast(Sous_zone_VF,le_domaine_dis.sous_zone_dis(ssz).valeur());
255 }
256 }
257
258 if(!sous_zone)
259 {
260 Cerr << "On ne trouve pas la sous_zone discretisee associee a " << nom_sous_zone << finl;
261 exit();
262 }
263 }
264}
