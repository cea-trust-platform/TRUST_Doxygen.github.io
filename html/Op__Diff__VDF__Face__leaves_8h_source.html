102{
103 double db_diffusivite = 0;
104 int element, compteur = 0;
105
106 if ((element=face_voisins(i,0)) != -1)
107 {
108 db_diffusivite += diffusivite_->valeurs()(element);
109 compteur++;
110 }
111 if ((element=face_voisins(i,1)) != -1)
112 {
113 db_diffusivite += diffusivite_->valeurs()(element);
114 compteur++;
115 }
116 if ((element=face_voisins(j,0)) != -1)
117 {
118 db_diffusivite += diffusivite_->valeurs()(element);
119 compteur++;
120 }
121 if ((element=face_voisins(j,1)) != -1)
122 {
123 db_diffusivite += diffusivite_->valeurs()(element);
124 compteur++;
125 }
126 db_diffusivite /= compteur;
127 return db_diffusivite;
128}
