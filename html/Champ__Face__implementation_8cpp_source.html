23DoubleTab& Champ_Face_implementation::valeur_aux_elems(const DoubleTab& positions, const IntVect& les_polys, DoubleTab& val) const
24{
25 const Champ_base& cha=le_champ();
26 int nb_compo_=cha.nb_comp();
27 if (val.nb_dim() == 2)
28 {
29 assert((val.dimension(0) == les_polys.size())||(val.dimension_tot(0) == les_polys.size()));
30 assert(val.dimension(1) == nb_compo_);
31 }
32 else
33 {
34 Cerr << "Erreur TRUST dans Champ_Face_implementation::valeur_aux_elems()" << finl;
35 Cerr << "Le DoubleTab val n'a pas 2 entrees" << finl;
36 Process::exit();
37 }
38
39 const Zone_VDF& zone_VDF = zone_vdf();
40// const Zone& zone_geom = zone_VDF.zone();
41 const IntTab& face_sommets = zone_VDF.face_sommets();
42 const IntTab& elem_faces = zone_VDF.elem_faces();
43 double psi,val1,val2;
44 int som0,som1;
45 int le_poly;
46
47 const DoubleTab& ch = cha.valeurs();
48 const Domaine& dom=zone().domaine();
49
50 if (nb_compo_ == 1)
51 {
52 for(int rang_poly=0; rang_poly<les_polys.size(); rang_poly++)
53 {
54 le_poly=les_polys(rang_poly);
55 if (le_poly == -1)
56 val(rang_poly, 0) = 0;
57 else
58 for(int dir=0; dir<Objet_U::dimension; dir++)
59 {
60 int face1 = elem_faces(le_poly,dir);
61 int face2 = elem_faces(le_poly,Objet_U::dimension+dir);
62 val1 = ch(face1);
63 val2 = ch(face2);
64
65 som0 = face_sommets(face1,0);
66 som1 = face_sommets(face2,0);
67
68 psi = ( positions(rang_poly,dir) - dom.coord(som0,dir) )
69 / ( dom.coord(som1,dir) - dom.coord(som0,dir) ) ;
70 if (est_egal(psi,0) || est_egal(psi,1))
71 val(rang_poly,0) = interpolation(val1,val2,psi);
72 }
73 }
74 }
75 else //!< (nb_compo_ != 1) */
76 {
77 for(int rang_poly=0; rang_poly<les_polys.size(); rang_poly++)
78 {
79 le_poly=les_polys(rang_poly);
80 if (le_poly == -1)
81 for(int ncomp=0; ncomp<nb_compo_; ncomp++)
82 val(rang_poly,ncomp) = 0;
83 else
84 for(int ncomp=0; ncomp<nb_compo_; ncomp++)
85 {
86
87 val1 = ch(elem_faces(le_poly,ncomp));
88 val2 = ch(elem_faces(le_poly,Objet_U::dimension+ncomp));
89
90 som0 = face_sommets(elem_faces(le_poly,ncomp),0);
91 som1 = face_sommets(elem_faces(le_poly,Objet_U::dimension+ncomp),0);
92
93 psi = ( positions(rang_poly,ncomp) - dom.coord(som0,ncomp) )
94 / ( dom.coord(som1,ncomp) - dom.coord(som0,ncomp) ) ;
95 val(rang_poly,ncomp) = interpolation(val1,val2,psi);
96 }
97 }
98 }
99 return val;
100}

102double Champ_Face_implementation::interpolation(const double val1, const double val2, const double psi) const
103{
104 double epsilon=1.e-12;
105 if (std::fabs(psi) < epsilon)
106 return val1 ;
107 else if (std::fabs(1.-psi) < epsilon)
108 return val2 ;
109 else
110 return val1 + psi * (val2-val1) ;
111}

262

264 {
265 Cerr<<"Champ_Face_implementation::valeur_aux_sommets"<<finl;
266 Cerr <<"A scalar field cannot be of Champ_Face type." << finl;
267 Process::exit();
268 }
269 else //!< (nb_compo_ != 1)
270 {
271 DoubleVect position(Objet_U::dimension);
272 for (num_elem=0; num_elem<nb_elem_tot; num_elem++)
273 for (j=0; j<nb_som_elem; j++)
274 {

290 for (ncomp=0; ncomp<nb_compo_; ncomp++)

372int Champ_Face_implementation::imprime_Face(Sortie& os, int ncomp) const
373{
374 const Champ_base& champ=le_champ();
375 int ni,nj,nk=-1;
376 IntVect m(3);
377 int i,j,k;
378 int np,elem;
379 int cmax=7;
380 DoubleVect xi,yj,zk;
381 DoubleTab Grille;
382// Lecture de xi,yj,zk dans un fichier .xiyjzk
383 Nom nomfic(Objet_U::nom_du_cas());
384 nomfic+=".xiyjzk";
385 LecFicDiffuse ficijk(nomfic);
386 ficijk >> xi;
387 ni=xi.size();
388 ficijk >> yj;
389 nj=yj.size();
390 if (Objet_U::dimension==3)
391 {
392 ficijk >> zk;
393 nk=zk.size();
394 }
395 m=1;
396 m(ncomp)=0;
397 if (Objet_U::dimension==3)
398 {
399// Grille ordonnee sur les faces des elements
400 ni-=m(0);
401 nj-=m(1);
402 nk-=m(2);
403 np=ni*nj*nk;
404 Grille.resize(np,Objet_U::dimension);
405 for(k=0; k<nk; k++)
406 for(i=0; i<ni; i++)
407 for(j=0; j<nj; j++)
408 {
409 elem=j+nj*(i+k*ni);
410 if (ncomp==0)
411 Grille(elem,0)=xi(i);
412 else
413 Grille(elem,0)=0.5*(xi(i)+xi(i+1));
414 if (ncomp==1)
415 Grille(elem,1)=yj(j);
416 else
417 Grille(elem,1)=0.5*(yj(j)+yj(j+1));
418 if (ncomp==2)
419 Grille(elem,2)=zk(k);
420 else
421 Grille(elem,2)=0.5*(zk(k)+zk(k+1));
422 }
423 DoubleVect valeurs(np);
424 champ.valeur_aux_compo(Grille, valeurs, ncomp);
425 for(k=0; k<nk; k++)
426 {
427 os << finl;
428 os << "Coupe a K= " << k << finl;
429 int n1=0,n2=0;
430 while (n2<ni)
431 {
432 n1=n2;
433 n2=std::min(ni,n2+cmax);
434 os << finl;
435 os << "I= ";
436 for(i=n1; i<n2; i++)
437 os << i << " ";
438 os << finl;
439 for(j=nj-1; j>-1; j--)
440 {
441 os << "J= " << j << " ";
442 for(i=n1; i<n2; i++)
443 {
444 elem=j+nj*(i+k*ni);
445 os << valeurs(elem) << " ";
446 }
447 os << finl;
448 }
449 }
450 }
451 }
452 else if (Objet_U::dimension==2)
453 {
454 ni-=m(0);
455 nj-=m(1);
456 np=ni*nj;
457 Grille.resize(np,Objet_U::dimension);
458 for(i=0; i<ni; i++)
459 for(j=0; j<nj; j++)
460 {
461 elem=j+nj*i;
462 if (ncomp==0)
463 Grille(elem,0)=xi(i);
464 else
465 Grille(elem,0)=0.5*(xi(i)+xi(i+1));
466 if (ncomp==1)
467 Grille(elem,1)=yj(j);
468 else
469 Grille(elem,1)=0.5*(yj(j)+yj(j+1));
470 }
471 DoubleVect valeurs(np);
472 champ.valeur_aux_compo(Grille, valeurs, ncomp);
473 int n1=0,n2=0;
474 while (n2<ni)
475 {
476 n1=n2;
477 n2=std::min(ni,n2+cmax);
478 os << finl;
479 os << "I= ";
480 for(i=n1; i<n2; i++)
481 os << i << " ";
482 os << finl;
483 for(j=nj-1; j>-1; j--)
484 {
485 os << "J= " << j << " ";
486 for(i=n1; i<n2; i++)
487 {
488 elem=j+nj*i;
489 os << valeurs(elem) << " ";
490 }
491 os << finl;
492 }
493 }
494 }
495 return 1;
496}

498DoubleTab& Champ_Face_implementation::trace(const Frontiere_dis_base& fr, const DoubleTab& y, DoubleTab& x,int distant) const
499{
500 assert(distant==0);
501 const Front_VF& fr_vf=ref_cast(Front_VF, fr);
502 const Zone_VDF& zvdf=zone_vdf();
503 const IntVect& ori = zvdf.orientation();
504 const IntTab& face_voisins = zvdf.face_voisins();
505 const IntTab& elem_faces = zvdf.elem_faces();
506 int elem1,elem2;
507 int face,i,f1,f2,f3,f4;
508 int nb_faces = fr_vf.nb_faces();
509 if (x.dimension(0)!=nb_faces)
510 {
511 Cerr << "The number of faces " << nb_faces << " on the remote boundary " << fr.le_nom() << finl;
512 Cerr << "does not match the number of faces " << x.dimension(0) << " on the local boundary." << finl;
513 Cerr << "Please, check if the boundary condition is not applied on wrong boundaries." << finl;
514 Process::exit();
515 }
516// assert(x.dimension(1)==Objet_U::dimension);
517
518 if (x.dimension(1) == 1)
519 {
520 for (i=0; i<fr_vf.nb_faces(); i++)
521 {
522 face=fr_vf.num_premiere_face()+i;
523 x(i,0)=y(face);
524 }
525 return x;
526 }
527
528 for (i=0; i<fr_vf.nb_faces(); i++)
529 {
530 face=fr_vf.num_premiere_face()+i;
531 x(i,ori[face])=y(face);
532 }
533
534 for (i=0; i<fr_vf.nb_faces(); i++)
535 {
536 face=fr_vf.num_premiere_face()+i;
537 elem1 = face_voisins(face,0);
538 if (elem1 != -1)
539 {
540 if (Objet_U::dimension == 2)
541 {
542 if (ori[face] == 0)
543 {
544 f1 = elem_faces(elem1,1);
545 f2 = elem_faces(elem1,3);

547 }
548 else
549 {
550 f1 = elem_faces(elem1,0);
551 f2 = elem_faces(elem1,2);
552 x(i,0)= 0.5*(y[f1] + y[f2]);
553 }
554 }
555 else if (Objet_U::dimension == 3)
556 {
557 if (ori[face] == 0)
558 {
559 f1 = elem_faces(elem1,1);
560 f2 = elem_faces(elem1,4);
561 f3 = elem_faces(elem1,2);
562 f4 = elem_faces(elem1,5);
563 x(i,1)= 0.5*(y[f1] + y[f2]);
564 x(i,2)= 0.5*(y[f3] + y[f4]);
565 }
566 else if (ori[face] == 1)
567 {
568 f1 = elem_faces(elem1,0);
569 f2 = elem_faces(elem1,3);
570 f3 = elem_faces(elem1,2);
571 f4 = elem_faces(elem1,5);
572 x(i,0)= 0.5*(y[f1] + y[f2]);
573 x(i,2)= 0.5*(y[f3] + y[f4]);
574 }
575 else //!< ori[face] = 2
576 {
577 f1 = elem_faces(elem1,0);
578 f2 = elem_faces(elem1,3);
579 f3 = elem_faces(elem1,1);
580 f4 = elem_faces(elem1,4);
581 x(i,0)= 0.5*(y[f1] + y[f2]);
582 x(i,1)= 0.5*(y[f3] + y[f4]);
583 }
584 }
585 }
586 else //!< elem1 = -1
587 {
588 elem2 = face_voisins(face,1);
589 if (Objet_U::dimension == 2)
590 {
591 if (ori[face] == 0)
592 {
593 f1 = elem_faces(elem2,1);
594 f2 = elem_faces(elem2,3);
595 x(i,1)= 0.5*(y[f1] + y[f2]);
596 }
597 else
598 {

600 f2 = elem_faces(elem2,2);
601 x(i,0)= 0.5*(y[f1] + y[f2]);

603 }
604 else if (Objet_U::dimension == 3)
605 {
606 if (ori[face] == 0)
607 {
608 f1 = elem_faces(elem2,1);
609 f2 = elem_faces(elem2,4);
610 f3 = elem_faces(elem2,2);
611 f4 = elem_faces(elem2,5);
612 x(i,1)= 0.5*(y[f1] + y[f2]);

614 }
615 else if (ori[face] == 1)
616 {
617 f1 = elem_faces(elem2,0);

619 f3 = elem_faces(elem2,2);
620 f4 = elem_faces(elem2,5);
