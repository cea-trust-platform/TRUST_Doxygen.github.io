52void Terme_Puissance_Thermique_Echange_Impose_Elem_PolyMAC::mettre_a_jour(double temps)
53{
54 const Zone_VF& zone = la_zone_PolyMAC.valeur();
55 const DoubleVect& volumes = zone.volumes();
56 const DoubleTab& himp = himp_.valeur().valeurs();
57 const DoubleTab& Text = Text_.valeur().valeurs();
58 const DoubleTab& T = equation().inconnue().valeurs();
59 int nb_elem = la_zone_PolyMAC.valeur().nb_elem(), c_h = himp.dimension(0) == 1, c_T = Text.dimension(0) == 1, n, N = T.line_size();
60
61 bilan().resize(N), bilan() = 0;
62
63 for (int e = 0; e < nb_elem; e++)
64 for (n = 0; n < N; n++)
65 bilan()(n) += himp(!c_h * e, n) * volumes(e) * (Text(!c_T * e, n) - T(e, n));
66
67 himp_.mettre_a_jour(temps);
68 Text_.mettre_a_jour(temps);
69}

88{
89 const Zone_VF& zone = la_zone_PolyMAC.valeur();
90 const DoubleVect& volumes = zone.volumes();

94 int nb_elem=la_zone_PolyMAC.valeur().nb_elem(), c_h = himp.dimension(0) == 1, c_T = Text.dimension(0) == 1, n, N = T.line_size();
95 const std::string& nom_inco = equation().inconnue().le_nom().getString();
96 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
97
98 for (int e = 0; e < nb_elem; e++)
99 for (n = 0; n < N; n++)
100 {
101 secmem(e, n) -= volumes(e) * himp(!c_h * e, n) * (T(e, n) - Text(!c_T * e, n));
