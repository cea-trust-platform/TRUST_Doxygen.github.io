57// XD champ_fonc_med field_base champ_fonc_med 0 Field to read a data field in a MED-format file .med at a specified time. It is very useful, for example, to resume a calculation with a new or refined geometry. The field post-processed on the new geometry at med format is used as initial condition for the resume.
58// XD attr use_existing_domain chaine(into=["use_existing_domain"]) use_existing_domain 1 not_set
59// XD attr last_time chaine(into=["last_time"]) last_time 1 to use the last time of the MED file instead of the specified time.
60// XD attr option decoup option 1 Keyword for a partition file
61// XD attr filename chaine filename 0 Name of the .med file.
62// XD attr domain_name chaine domain_name 0 Name of the domain.
63// XD attr field_name chaine field_name 0 Name of the problem unknown.
64// XD attr location chaine(into=["som","elem"]) location 0 To indicate where the field has been post-processed.
65// XD attr time floattant time 0 Time of the field in the .med file.

68// XD attr key chaine(into=["decoup"]) key 0 Name of for a partition file

75 s>>chaine_lue;
76 int use_existing_domain=0, field_size, nom_decoup_lu = 0;
77 last_time_only_=0;
78 if (chaine_lue=="use_existing_domain")
79 {
80 use_existing_domain=1;
81 s>>chaine_lue;
82 }
83 if (chaine_lue=="last_time")
84 {
85 last_time_only_=1;
86 s>>chaine_lue;
87 }
88 if (chaine_lue=="decoup")
89 {
90 s>>nom_decoup;
91 s>>chaine_lue;
92 if (use_medcoupling_)
93 {
94 Cerr << "Error: Decoup option is only available for the moment with Champ_Fonc_MEDFile keyword only." << finl;
95 Process::exit();

104 nom_fichier_med_ = chaine_lue;
105 lire_nom_med(nom_dom,s);
106 if (!nom_decoup_lu) nom_decoup = Nom("decoup/") + nom_dom + ".txt"; //valeur par defaut de nom_decoup
107 lire_nom_med(nom_champ,s);
108 s>>localisation;
109 s>>un_temps;
110 traite_nom_fichier_med(nom_fichier_med_);
111// #ifdef MED_
112// test_version(nom_fichier_med_);
113// #endif

117// La lecture de fichiers multiples .med fonctionne: voir cas test Champ_fonc_MED_Parallele
118// Un test est fait plus loin pour bien verifier que les partitionnement se recouvrent pour cela
119 int multiple_med = 0;
120 Nom tmp(nom_fichier_med_);
121 tmp.prefix("0001.med");
122 if (tmp!=nom_fichier_med_ && Process::nproc()>1)
123 {
124 multiple_med = 1;
125 /*
126 // Ajout d'un test bloquant pour empecher la lecture de fichiers 000n.med
127 Cerr << "Error, Champ_Fonc_MED can't read, for the moment, partitioned MED files." << finl;
128 Cerr << "Try to build a single MED file by:" << finl;
129 Cerr << "A) Using LATA format for the previous calculation." << finl;
130 Cerr << "B) Convert the LATA files into a single MED file with Lata_to_MED keyword." << finl;
131 Cerr << "C) Use this single MED file with Champ_Fonc_MED keyword." << finl;
132 exit(); */
133 }
134 multiple_med = mp_max(multiple_med);
135 nommer(nom_champ);
136
137 bool domain_exist=( interprete().objet_existant(nom_dom) && sub_type(Domaine, interprete().objet(nom_dom)) );
138 if (use_existing_domain)
139 {
140 if (!domain_exist)
141 {
142 Cerr << "The domain " << nom_dom << " does not exist !!!! In Champ_Fonc_MED " << nom_fichier_med_ << " " << nom_dom
143 << " " << nom_champ << " " << localisation << " " << un_temps << finl;
144 exit();
145 }
146 else
147 {
148// use_existing_domain utilisable en parallele uniquement si le process 0 gere tout le domaine ou si decoup specifie:
149 const Domaine& le_domaine=ref_cast(Domaine, interprete().objet(nom_dom));
150 if (Process::nproc()>1 && mp_max(le_domaine.nb_som()>0) != 0 && use_medcoupling_==1)
151 {
152 Cerr << "Warning, you can't use use_existing_domain on a partitionned domain like " << nom_dom << finl;
153 Cerr << "It is not parallelized yet... So we use MED mesh, which is not optimal." << finl;
154 Cerr << "Try suppress use_existing_domain option." << finl;
155// Process::exit();
156 use_existing_domain = 0;
157 }
158 }
159 }
160 if (domain_exist && use_existing_domain)
161 {
162 Cerr<<nom_dom<<" was not read into the med file because it already exists." <<finl;
163
164 const Domaine& le_domaine=ref_cast(Domaine, interprete().objet(nom_dom));
165
166#ifndef NDEBUG
167// on va verifier que l'on a le meme domaine en mode debug car lent
168// Ce test ne peut etre fait qu'en sequentiel car en parallele, dom=domaine_complet, un_dom=domaine_local
169 if (nproc()==1)
170 {
171 LireMED liremed;
172 dom_med_.nommer(nom_dom);
173// ne pas initialiser le nom du domaine va conduire a ne pas creer les fichiers sous zones
174 Nom nom_dom_trio_non_nomme;
175 liremed.lire_geom(nom_fichier_med_,dom_med_,nom_dom,nom_dom_trio_non_nomme);
176
177 DoubleTab diff_som(dom_med_.les_sommets());
178 IntTab diff_elem(dom_med_.zone(0).les_elems());
179 diff_som.set_md_vector(le_domaine.les_sommets().get_md_vector());
180 diff_elem.set_md_vector(le_domaine.zone(0).les_elems().get_md_vector());
181 diff_som-=le_domaine.les_sommets();
182 diff_elem-=le_domaine.zone(0).les_elems();
183 /*
184 diff_som.set_md_vector(un_dom.les_sommets().get_md_vector());
185 diff_som-=un_dom.les_sommets();
186 diff_som.set_md_vector(MD_Vector());
187 diff_elem.set_md_vector(un_dom.zone(0).les_elems().get_md_vector());
188 diff_elem-=un_dom.zone(0).les_elems();
189 diff_elem.set_md_vector(MD_Vector());
190 */
191 double err_som0 = max_abs_array(diff_som);
192 int err_elem0 = max_abs_array(diff_elem);
193
194
195// Le domaine un_dom a peut etre ete reordonne, on refait le test en reordonnant le domaine dom .....
196 dom_med_.reordonner();
197 dom_med_.les_sommets().set_md_vector(le_domaine.les_sommets().get_md_vector());
198 dom_med_.les_sommets()-=le_domaine.les_sommets();
199 dom_med_.les_sommets().set_md_vector(MD_Vector());
200 dom_med_.zone(0).les_elems().set_md_vector(le_domaine.zone(0).les_elems().get_md_vector());
201 dom_med_.zone(0).les_elems()-=le_domaine.zone(0).les_elems();
202 dom_med_.zone(0).les_elems().set_md_vector(MD_Vector());
203 double err_som = max_abs_array(dom_med_.les_sommets());
204 int err_elem = max_abs_array(dom_med_.zone(0).les_elems());
205
206 if ((err_som>1e-5 || err_elem>0) && (err_som0>1e-5 || err_elem0>0))
207 {
208 Cerr<<"Error the domain in the file "<<nom_fichier_med_<<" and domain "<<nom_dom<<" are not the same (coords,conn)."<<finl;
209 exit();
210 }
211
212// Reset:
213 dom_med_=Domaine();
214 }
215#endif
216 field_size = creer(nom_fichier_med_,le_domaine,localisation,temps_sauv_);
217 }
218 else
219 {
220 if (domain_exist && !use_existing_domain)
221 {
222 Cerr<<"INFO: You can use Champ_fonc_med use_existing_domain "<< nom_fichier_med_<<" "<<nom_dom<<" "<<nom_champ<<" "<<localisation<<" "<<un_temps<<" to optimize."<<finl;
223 }
224 LireMED liremed;
225 dom_med_.nommer(nom_dom);
226// ne pas initialiser le nom du domaine va conduire a ne pas creer les fichiers sous zones
227 Nom nom_dom_trio_non_nomme;
228// Remplit dom:
229 liremed.lire_geom(nom_fichier_med_,dom_med_,nom_dom,nom_dom_trio_non_nomme);
230 if (multiple_med)
231 {
232// On verifie que l'on a bien des recouvrements identiques (verification imparfaite sur les BoundingBox)
233 DoubleTab BB1 = dom_med_.getBoundingBox();
234 const Domaine& dom_calcul = ref_cast(Domaine, interprete().objet(nom_dom));
235 DoubleTab BB2 = dom_calcul.getBoundingBox();
236 for (int i=0; i<dimension; i++)
237 for (int j=0; j<2; j++)
238 if (est_different(BB1(i,j), BB2(i,j)))
239 {
240 Cerr << "Error, Champ_Fonc_MED can't read the partitionned MED files." << finl;
241 Cerr << "Cause if seems the MED partition is different than the " << dom_calcul.le_nom() << " domain partition." << finl;
242 Process::exit();
243 }
244 Cerr << "Ok MED partition matches the domain partition so reading multiple MED files..." << finl;
245 }
246// MODIF ELI LAUCOIN (06/03/2012) :
247// j'ajoute l'attribut temps_sauv_
248 field_size = creer(nom_fichier_med_,dom_med_,localisation,temps_sauv_);
249 }
250 if (last_time_only_)
251 {
252 un_temps=temps_sauv_[temps_sauv_.size_array()-1];
253 Cout << "The resumption time is "<<un_temps<<finl;
254 }
255// FIN MODIF ELI LAUCOIN (06/03/2012)
256 /* si on est en parallele : creation du filtre */
257 if (Process::nproc() > 1 && field_size != le_champ().valeurs().dimension(0))
258 {
259 ArrOfInt dec;
260 filter.set_smart_resize(1);
261 EFichier fdec;
262 fdec.ouvrir(nom_decoup);
263// Cas ou le maillage du fichier .med suit la numerotation du maillage initial (necessite le fichier du decoupage pour retrouver la numerotation)
264 if (fdec.good())
265 {
266 fdec >> dec;
267 for (int i = 0; i < dec.size_array(); i++)
268 if (dec[i] == Process::me()) filter.append_array(i + 1);//les indices commencent a 1
269 if (field_size != dec.size_array())
270 Cerr << "Champ_Fonc_MED on parallel domain : inconsistency between 'decoup' file and field!" << finl, Process::exit();
271 }
272 else //!< Cas ou le maillage .med suit la numerotation du maillage decoupe
273 {
274 int nb_item = le_champ().valeurs().dimension(0);
275 int first_item = mppartial_sum(nb_item);
276 for (int i=0; i<nb_item; i++)
277 filter.append_array(first_item + i + 1);//les indices commencent a 1
278 }
279 filter.set_smart_resize(0), filter.resize(filter.size_array());
280 Cerr << "Creating a filter to access efficiently values in " << nom_fichier_med_ << finl;
281 if (filter.size_array() != le_champ().valeurs().dimension(0))
282 Cerr << "Champ_Fonc_MED on parallel domain : inconsistency between filter and domain!" << finl, Process::exit();
283 }
284 else if (field_size != le_champ().valeurs().dimension(0))
285 Cerr << "Champ_Fonc_MED on existing domain : inconsistency between domain file and field!" << finl, Process::exit();
286
287 le_champ().nommer(nom_champ);
288 le_champ().corriger_unite_nom_compo();
289 mettre_a_jour(un_temps);
290
291 return s ;

306void Champ_Fonc_MED::lire(double t, int given_it)
307{
308 if (zonebidon_inst.nb_elem()==0) //!< Cas d'une zone vide
309 {
310// Mise a jour:
311 Champ_Fonc_base::mettre_a_jour(t);
312 le_champ().Champ_Fonc_base::mettre_a_jour(t);
313 return;
314 }
315#ifdef MED_
316#ifdef MEDCOUPLING_
317 if (use_medcoupling_)
318 {
319// Read a field
320 std::string meshName = mon_dom.le_nom().getString();
321 std::string fieldName = nom_champ_dans_fichier_med_.getString();
322 std::string fileName = nom_fichier_med_.getString();
323// Etude des conditions pour chercher ou nom un champ dans le fichier MED:
324 bool search_field = true;
325 nb_dt = temps_sauv_.size_array();
326 if (nb_dt>0)
327 {
328// nb_dt nombre de pas de temps dans le fichier MED
329// ndt taille du tableau temps_sauv contenant les pas de temps du champ dans le fichier MED
330// tmax dernier temps du tableau temps_sauv ?
331// dt ?
332// t temps courant pour lequel on veut remplir le champ
333// last_time_only_ specifie dans le jeu de donnees ou non
334 double tmax = temps_sauv_[nb_dt - 1];
335 double dt = temps_sauv_[nb_dt - 1];
336 if (((nb_dt == 1) && (!est_egal(dt, t))) || ((last_time_only_ == 1) && (!est_egal(tmax, t))))
337 {
338 Cout << "We assume that the field " << fieldName << " is stationary." << finl;
339 search_field = false;
340 }
341 }
342 if (search_field)
343 {
344 if (temps_sauv_.size_array()==0)
345 temps_sauv_ = lire_temps_champ(fileName, fieldName);
346 unsigned int nn = temps_sauv_.size_array();
347 if (given_it == -1)
348 {
349 for (given_it = 0; given_it < (int) nn; given_it++)
350 if (est_egal(temps_sauv_[given_it], t)) break;
351 if (given_it == (int)nn)
352 {
353 Cerr << "Error. Time " << t << " not found in the times list of the " << fileName << " file" << finl;
354 for (unsigned int n=0; n<nn; n++)
355 Cerr << temps_sauv_[n] << finl;
356 Process::exit();
357 }
358 }
359 int iteration = time_steps_[given_it].first;
360 int order = time_steps_[given_it].second;
361
362// Only one MCAuto below to avoid double deletion:
363 MCAuto<MEDCouplingField> ffield = lire_champ(fileName, meshName, fieldName, iteration, order);
364 Cerr << " at time " << t << " ... " << finl;
365 MEDCouplingFieldDouble * field = dynamic_cast<MEDCouplingFieldDouble *>((MEDCouplingField *)ffield);
366 if (field == 0)
367 {
368 Cerr << "ERROR reading MED field! Not a MEDCouplingFieldDouble!!" << finl;
369 Process::exit(-1);
370 }
371 const double *field_values = field->getArray()->begin();
372 assert(field->getNumberOfTuplesExpected() == le_champ().valeurs().dimension(0));
373 assert((int) field->getNumberOfComponents() ==
374 (le_champ().valeurs().nb_dim() == 1 ? 1 : le_champ().valeurs().dimension(1)));
375 memcpy(le_champ().valeurs().addr(), field_values,
376 le_champ().valeurs().size_array() * sizeof(double));
377 }
378 }
379 else
380#endif
381 {
382 med_idt fid = MEDfileOpen(nom_fichier_med_, MED_ACC_RDONLY);
383 if (fid < 0)
384 {
385 Cerr << "Problem while opening the file " << nom_fichier_med_ << finl;
386 exit();
387 }
388
389 Char_ptr pflname;
390 pflname.allocate(MED_NAME_SIZE);
391 if (given_it != -1)
392 {
393 Cerr<<"Searching for the "<<given_it+1<< " iteration of the field "<<le_nom()<<finl;
394 }
395
396 Char_ptr dt_unit;
397 dt_unit.allocate(MED_SNAME_SIZE);
398// on cherche le numero du pas de temps
399 med_int numdt, numo;
400 double dt = -2;
401 int ret, size = 0;
402 Char_ptr maa_ass;
403 maa_ass.allocate(MED_NAME_SIZE);
404// Cerr<<"nb_dt"<<nb_dt<<finl;
405 Nom nom_dom(mon_dom.le_nom());
406 int i2 = 0;
407 Nom& le_nom_du_champ = nom_champ_dans_fichier_med_;
408 med_field_type typcha;
409// Nom meshname; dimensionne_nom_taille(meshname,MED_NAME_SIZE);
410 Char_ptr dtunit;
411 dtunit.allocate(MED_SNAME_SIZE);
412 med_bool localmesh;
413 med_int nbofcstp;
414 int ncomp = MEDfieldnComponentByName(fid, le_nom_du_champ);
415 Char_ptr comp;
416 comp.allocate(MED_SNAME_SIZE * ncomp);
417 Char_ptr unit;
418 unit.allocate(MED_SNAME_SIZE * ncomp);
419 ret = MEDfieldInfoByName(fid, le_nom_du_champ, maa_ass, &localmesh, &typcha, comp, unit, dtunit, &nbofcstp);
420 int nb_dt_tot = nbofcstp;
421 double tmax = -1;
422 for (int j = 0; j < nb_dt_tot; j++)
423 {
424 med_int meshnumdt, meshnumit;
425 MEDfieldComputingStepMeshInfo(fid, le_nom_du_champ, j + 1, &numdt, &numo, &dt, &meshnumdt, &meshnumit);
426 tmax = dt;
427 }
428 for (int j = 0; j < nb_dt_tot; j++)
429 {
430 if (i2 >= nb_dt) exit();
431
432 med_int meshnumdt, meshnumit;
433 MEDfieldComputingStepMeshInfo(fid, le_nom_du_champ, j + 1, &numdt, &numo, &dt, &meshnumdt, &meshnumit);
434 Nom Nmaa_ass(maa_ass);
435 if (est_egal(dt, t))
436 {
437 size = MEDfieldnValue(fid, le_nom_du_champ, numdt, numo, type_ent, type_geo);
438 if ((size != 0) && (nom_dom == Nmaa_ass))
439 break;
440 }
441 if (Nmaa_ass == nom_dom)
442 {
443 if (given_it == i2)
444 {
445 t = dt;
446 given_it = j;
447 break;
448 }
449 i2++;
450 }
451 }
452// dt=t;
453// numdt=1;
454// numo=MED_NONOR;
455// Cerr << "Provisoire temps=" << temps_sauv_ << finl;
456// Cerr << "Provisoire temps= " << nb_dt << " " << dt << " " << t << " " << tmax << " " << last_time_only_ << finl;
457 if (((nb_dt == 1) && (!est_egal(dt, t))) || ((last_time_only_ == 1) && (!est_egal(tmax, t))))
458 {
459 Cout << "We assume that the field " << le_nom_du_champ << " is stationary." << finl;
460 }
461 else
462 {
463 if (!est_egal(dt, t))
464 {
465 Cerr.precision(12);
466 Cerr << "Error: we did not find the time " << t << finl;
467 Cerr << "Available times for " << le_nom_du_champ << " domain " << nom_dom << " are:" << finl;
468 for (int j = 0; j < nb_dt; j++)
469 {
470 med_int meshnumdt, meshnumit;
471 MEDfieldComputingStepMeshInfo(fid, le_nom_du_champ, j + 1, &numdt, &numo, &dt, &meshnumdt, &meshnumit);
472 Cerr << dt << finl;
473 }
474 exit();
475 }
476 Cerr << "Reading the time step " << (int) numdt << " of field " << le_nom_du_champ << " time " << t
477 << " type ent " << (int) type_ent << finl;
478 if (filter.size_array() > 0) //lecture sur domaine parallele -> avec un filtre
479 {
480 Cerr << "Read with filter..." << finl;
481 med_filter flt = MED_FILTER_INIT;
482 if(MEDfilterEntityCr(fid, size, 1, le_champ().valeurs().nb_dim() > 1 ? le_champ().valeurs().dimension(1) : 1, MED_ALL_CONSTITUENT,
483 MED_FULL_INTERLACE, MED_COMPACT_STMODE, MED_ALLENTITIES_PROFILE, filter.size_array(), filter.addr(), &flt) < 0)
484 Cerr << "Champ_Fonc_MED on parallel domain : error at filter creation!" << finl, Process::exit();
485 ret = MEDfieldValueAdvancedRd(fid,le_nom_du_champ,numdt,numo,type_ent,type_geo,&flt,(unsigned char*)le_champ().valeurs().addr());
486 }
487 else if (le_champ().valeurs().dimension(0) > 0)//lecture complete
488 ret = MEDfieldValueRd(fid, le_nom_du_champ, numdt, numo, type_ent, type_geo, MED_FULL_INTERLACE,
489 MED_ALL_CONSTITUENT, (unsigned char *) le_champ().valeurs().addr());
490 if (ret < 0)
491 {
492 Cerr << "Problem while reading field " << le_nom_du_champ << " name of domain " << mon_dom.le_nom() << finl;
493 exit();
494 }
495 MEDfileClose(fid);
496 }
497 }
498// Mise a jour:
499 Champ_Fonc_base::mettre_a_jour(t);
500 le_champ().Champ_Fonc_base::mettre_a_jour(t);
501#else
502 med_non_installe();
503#endif
504}

627
628// corriger_unite_nom_compo();

630// le_champ().corriger_unite_nom_compo();

632#else
633 med_non_installe();
634 return 0;
635#endif
