93/** \return Entree\&<br>
94Signification: le flot d'entree modifie<br>
95Contraintes: */
96Entree& Champ_Fonc_Tabule::readOn(Entree& is)
97{
98 int nbcomp, old_table_syntax_ = 0;
99 Nom motlu;
100 const Motcle accolade_ouverte("{"), accolade_fermee("}");
101 is >> motlu;
102 if (motlu != accolade_ouverte)
103 {
104 noms_champs_parametre_.add(motlu);
105 old_table_syntax_ = 1;
106 }
107 else
108 {
109 assert (old_table_syntax_ == 0);
110 while (true)
111 {
112 is >> motlu;
113 if (motlu == "}") break;
114 noms_pbs_.add(motlu);
115 is >> motlu;
116 noms_champs_parametre_.add(motlu);
117 }
118 }
119 const int nb_param = noms_champs_parametre_.size();
120
121 if (old_table_syntax_ && noms_pbs_.size() != 0 && nb_param != 1)
122 {
123 Cerr << "What ??? Big problem in Champ_Fonc_Tabule::readOn !!!" << finl;
124 throw;
125 }
126
127 is >> nbcomp;
128 fixer_nb_comp(nbcomp);
129 is >> motlu;
130 if (motlu == accolade_ouverte)
131 {
132 /* 1. lecture de la grille de parametres */
133 DoubleVects params;
134 for (int n = 0; n < nb_param; n++)
135 {
136 const int nb_val = lire_dimension(is, que_suis_je());
137 DoubleVect param(nb_val);
138 for (int i = 0; i < nb_val; i++) is >> param[i];
139 params.add(param);
140 }
141
142 /* 2. lecture des valeurs des parametres */
143// taille totale du tableau de valeurs
144 int size = nbcomp;
145 for (int n = 0; n < nb_param; n++) size *= params[n].size();
146
147// lecture : tout dans un tableau 1D
148 DoubleVect tab_valeurs(size);
149 for (int i = 0; i < size; i++) is >> tab_valeurs[i];
150 la_table.remplir(params, tab_valeurs);
151
152 is >> motlu;
153 if (motlu != accolade_fermee)
154 {
155 Cerr << "Error reading from an object of type Champ_Fonc_Tabule" << finl;
156 Cerr << "We expected keyword } instead of " << motlu << finl;
157 exit();
158 }
159 }
160 else if (motlu=="fonction")
161 {
162 Cerr<<"The syntax has changed..." << finl;
163 Cerr<<"The syntax is now Champ_Fonc_fonction 1 field_expression"<<finl;
164 exit();
165 Cerr<<"We read the analytic function "<<finl;
166 la_table.lire_f(is,0);
167 }
168 else
169 {
170 Cerr << "Error reading from an object of type Champ_Fonc_Tabule" << finl;
171 Cerr << "We expected keyword { or fonction instead of " << motlu << finl;
172 exit();
173 }
174 return is;
175}
