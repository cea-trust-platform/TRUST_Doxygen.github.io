63//! Lit les caracteristques de la loi de parois<br>a partir d'un flot d'entree.<br>Format:<br>type_de_loi_de_paroi<br>Les valeurs possibles du type de loi de paroi sont:<br>- "loi_standard_hydr"<br>- "negligeable"<br>- "loi_VanDriest"<br>- "loi_standard_hydr_scalaire"
64/** \return Entree\&<br>
65Signification: le flot d'entree modifie<br>
66Contraintes: */
67Entree& Turbulence_paroi::readOn(Entree& s)
68{
69 Cerr << "Lecture du type de loi de parois " << finl;
70 const Equation_base& eqn = mon_modele_turb_hyd->equation();
71 Motcle typ;
72 s >> typ;
73 if (typ == "loi_standard_hydr_scalaire" || typ == "loi_paroi_2_couches_scalaire" || typ == "negligeable_scal")
74 {
75 Cerr << finl;
76 Cerr << "Le format du jeu de donnees a change:" << finl;
77 Cerr << "Chaque modele de turbulence doit avoir sa loi de paroi specifiee." << finl;
78 Cerr << "Ainsi par exemple, loi_standard_hydr sera pour le modele de turbulence de l'equation de qdm" << finl;
79 Cerr << "et loi_standard_hydr_scalaire pour le modele de turbulence de l'equation d'energie." << typ << finl;
80 Cerr << finl;
81 exit();
82 }
83 typ+="_";
84
85 Nom discr=eqn.discretisation().que_suis_je();
86
87// les operateurs de diffusion sont communs aux discretisations VEF et VEFP1B
88 if(discr=="VEFPreP1B") discr="VEF";
89// GF: je n'en suis pas sur du tout !!!
90 if (discr=="VDF_Front_Tracking") discr="VDF";
91 if(discr=="VEFPreP1B_Front_Tracking") discr="VEF";
92 typ+=discr;
93
94 Cerr<<"et typage :"<<typ<<finl;
95 DERIV(Turbulence_paroi_base)::typer(typ);
96 valeur().associer_modele(mon_modele_turb_hyd.valeur());
97 valeur().associer(eqn.zone_dis(),eqn.zone_Cl_dis());
98 return s;
99}
