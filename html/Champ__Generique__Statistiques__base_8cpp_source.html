69int Champ_Generique_Statistiques_base::completer_post_statistiques(const Domaine& dom,const int is_axi,Format_Post_base& format)
70{
71
72 const Nom nom_post = le_nom();
73
74 Nom localisation;
75 const Entity& loc = get_localisation(0);
76 if (loc==ELEMENT)
77 localisation = "ELEM";
78 else if (loc==NODE)
79 localisation = "SOM";
80 else
81 {
82 Cerr<<"This localization is not valid for postprocessing"<<finl;
83 exit();
84 }
85
86 Champ espace_stockage_source;
87 const Champ_base& source = integrale().le_champ()->get_champ(espace_stockage_source);
88 source.completer_post_champ(dom,is_axi,localisation,nom_post,format);
89 return 1;
90
91}
