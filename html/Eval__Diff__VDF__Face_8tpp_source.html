28template <typename DERIVED_T> template<Type_Flux_Fa7 Fa7_Type, typename Type_Double> inline enable_if_t< Fa7_Type == Type_Flux_Fa7::ELEM, void>
29Eval_Diff_VDF_Face<DERIVED_T>::flux_fa7(const DoubleTab& inco, int elem, int fac1, int fac2, Type_Double& flux) const
30{
31 const int ori = orientation(fac1), ncomp = flux.size_array();
32 const double dist = dist_face(fac1,fac2,ori), surf = 0.5*(surface(fac1)*porosite(fac1)+surface(fac2)*porosite(fac2)),
33 visc_lam = nu_lam(elem), visc_turb = DERIVED_T::IS_TURB ? nu_turb(elem) : 0.;
34 for (int k = 0; k < ncomp; k++)
35 {
36 const double tau = (inco(fac2,k)-inco(fac1,k))/dist, tau_tr = ACTIVATE_TAU_TR ? tau : 0.0;
37 flux[k] = ((tau + tau_tr) * (visc_lam + visc_turb)) * surf;
38 }
39}

55template <typename DERIVED_T> template<Type_Flux_Arete Arete_Type, typename Type_Double> inline enable_if_t< Arete_Type == Type_Flux_Arete::MIXTE, void>
56Eval_Diff_VDF_Face<DERIVED_T>::flux_arete(const DoubleTab& inco, int fac1, int fac2, int fac3, int fac4, Type_Double& flux) const
57{
58 int elem[4], ori1 = orientation(fac1), ori3 = orientation(fac3);
59 elem[0] = elem_(fac3,0), elem[1] = elem_(fac3,1), elem[2] = elem_(fac4,0), elem[3] = elem_(fac4,1);
60 double visc_lam_temp = 0, visc_turb_temp = 0;
61 for (int i = 0; i < 4; i++)
62 if (elem[i] != -1)
63 {
64 visc_lam_temp += nu_lam(elem[i]);
65 visc_turb_temp += nu_turb(elem[i]);
66 }
67 visc_lam_temp /= 3.0;
68 visc_turb_temp /= 3.0;
69 const double surf = surface_(fac1,fac2), poros = porosity_(fac1,fac2),
70 visc_lam = DERIVED_T::IS_VAR ? visc_lam_temp : nu_lam(0), visc_turb = DERIVED_T::IS_TURB ? visc_turb_temp : 0.0;
71
72 for (int k = 0; k < flux.size_array(); k++)
73 if (inco(fac4,k)*inco(fac3,k) != 0)
74 {
75 const double tau = (inco(fac4,k)-inco(fac3,k))/dist_face(fac3,fac4,ori1), tau_tr = ACTIVATE_TAU_TR ? (inco(fac2,k)-inco(fac1,k))/dist_face(fac1,fac2,ori3) : 0.0;
76 flux[k] = ((tau + tau_tr) * (visc_lam + visc_turb)) * surf * poros;
77 }

114template <typename DERIVED_T> template<Type_Flux_Arete Arete_Type, typename Type_Double>
115inline enable_if_t< Arete_Type == Type_Flux_Arete::FLUIDE || Arete_Type == Type_Flux_Arete::SYMETRIE_FLUIDE || Arete_Type == Type_Flux_Arete::PAROI_FLUIDE, void>
116Eval_Diff_VDF_Face<DERIVED_T>::flux_arete(const DoubleTab& inco, int fac1, int fac2, int fac3, int signe, Type_Double& flux3, Type_Double& flux1_2) const
117{
118 assert (flux3.size_array() == flux1_2.size_array());
119 constexpr bool is_SYM_FL = (Arete_Type == Type_Flux_Arete::SYMETRIE_FLUIDE), is_PAR_FL = (Arete_Type == Type_Flux_Arete::PAROI_FLUIDE);
120 const int rang1 = (fac1-premiere_face_bord), rang2 = (fac2-premiere_face_bord), elem1 = elem_(fac3,0), elem2 = elem_(fac3,1), ori= orientation(fac3), ncomp = flux3.size_array();
121 const double surf = surface_(fac1,fac2), poros = porosity_(fac1,fac2), surfporos = surface(fac3)*porosite(fac3), tps = inconnue->temps(),
122 dist1 = dist_norm_bord(fac1), dist2 = dist_face(fac1,fac2,ori),
123 visc_lam = nu_lam_mean_2pts(elem1,elem2,ori), visc_turb = DERIVED_T::IS_TURB ? nu_mean_2pts(elem1,elem2,ori) : 0.0;
124
125 double vit_imp;
126 for (int k = 0; k < ncomp; k++)
127 {
128 if (is_SYM_FL) vit_imp = 0.5*(Champ_Face_get_val_imp_face_bord_sym(inco,tps,rang1,ori,la_zcl)+ Champ_Face_get_val_imp_face_bord_sym(inco,tps,rang2,ori,la_zcl));
129 else if (is_PAR_FL) //!< On ne sait pas qui de fac1 ou de fac2 est la face de paroi
130 {
131 if (est_egal(inco(fac1,k),0)) vit_imp = Champ_Face_get_val_imp_face_bord(tps,rang2,ori,la_zcl); //!< fac1 est la face de paroi
132 else vit_imp = Champ_Face_get_val_imp_face_bord(tps,rang1,ori,la_zcl); //!< fac2 est la face de paroi
133 }
134 else vit_imp = 0.5*(Champ_Face_get_val_imp_face_bord(tps,rang1,ori,la_zcl)+Champ_Face_get_val_imp_face_bord(tps,rang2,ori,la_zcl));
135
136// |
137// fac 3 | fac 2
138// --------
139// | fac 1
140// |
141
142// fac3 est la face interne et fac1 et fac2 sont au bord
143 const double tau_3 = signe*(vit_imp-inco(fac3,k))/dist1, tau_12 = (inco(fac2,k)-inco(fac1,k))/dist2, tau_tr_3 = ACTIVATE_TAU_TR ? tau_12 : 0.0, tau_tr_12 = ACTIVATE_TAU_TR ? tau_3 : 0.0;
144 flux3[k] = ((tau_3 + tau_tr_3) * (visc_lam + visc_turb)) * surf * poros;
145 flux1_2[k] = ((tau_12 + tau_tr_12) * (visc_lam + visc_turb)) * surfporos;
146 }
