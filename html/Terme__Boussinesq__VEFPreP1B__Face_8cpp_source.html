43DoubleTab& Terme_Boussinesq_VEFPreP1B_Face::ajouter(DoubleTab& resu) const
44{
45 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b, la_zone_VEF.valeur());
46// Si seulement support P0 on appelle en VEF
47 if (zone_VEF.get_alphaE() && !zone_VEF.get_alphaS() && !zone_VEF.get_alphaA())
48 return Terme_Boussinesq_VEF_Face::ajouter(resu);
49
50 const DoubleVect& volumes = zone_VEF.volumes();
51 const DoubleVect& porosite_surf = zone_VEF.porosite_face();
52 const Champ_Inc& le_scalaire = equation_scalaire().inconnue();
53 const DoubleVect& g = gravite().valeurs();
54 const Zone_Cl_VEF& zone_Cl_VEF = la_zone_Cl_VEF.valeur();
55 const IntTab& elem_sommets = zone_VEF.zone().les_elems();
56 const IntTab& elem_faces = zone_VEF.elem_faces();
57 const DoubleTab& coord_sommets=zone_VEF.zone().domaine().les_sommets();
58
59// Verifie la validite de T0:
60 check();
61
62 int nbpts=-1; //!< nombre de points d'integration
63
64 if(dimension==2)
65 {
66 nbpts=3; //!< ordre 2
67 }
68 else if(dimension==3)
69 {
70 nbpts=4; //!< ordre 2
71 }
72 else
73 assert(0);
74
75// On remplit les Poids et les coord_bary :
76 ArrOfDouble Poids(nbpts);
77 DoubleTab coord_bary(nbpts,dimension+1); //!< lambda_i des points
78 if(dimension==2)
79 {
80 double tiers=0.333333333333333;
81 Poids[0]=tiers;
82 Poids[1]=Poids[2]=Poids[0];
83
84 coord_bary(0,0)=0.5;
85 coord_bary(0,1)=0.;
86 coord_bary(0,2)=0.5;
87
88 coord_bary(1,0)=0.;
89 coord_bary(1,1)=0.5;
90 coord_bary(1,2)=0.5;
91
92 coord_bary(2,0)=0.5;
93 coord_bary(2,1)=0.5;
94 coord_bary(2,2)=0.;
95 }
96 else if(dimension==3)
97 {
98 Poids[0]=0.25;
99 Poids[1]=Poids[2]=Poids[3]=Poids[0];
100
101 double a = 0.5854101966249685;
102 double b = 0.1381966011250105;
103
104 coord_bary(0,0)=a;
105 coord_bary(0,1)=b;
106 coord_bary(0,2)=b;
107 coord_bary(0,3)=b;
108
109 coord_bary(1,0)=b;
110 coord_bary(1,1)=a;
111 coord_bary(1,2)=b;
112 coord_bary(1,3)=b;
113
114 coord_bary(2,1)=b;
115 coord_bary(2,0)=b;
116 coord_bary(2,2)=a;
117 coord_bary(2,3)=b;
118
119 coord_bary(3,2)=b;
120 coord_bary(3,0)=b;
121 coord_bary(3,1)=b;
122 coord_bary(3,3)=a;
123 }
124 else
125 assert(0);
126
127 int nb_comp = le_scalaire.valeur().nb_comp(); //!< Vaut 1 si temperature, nb_constituents si concentration
128 IntVect les_polygones(nbpts);
129 DoubleTab les_positions(nbpts,dimension);
130 DoubleTab valeurs_scalaire(nbpts,nb_comp);
131 DoubleTab valeurs_beta(nbpts,nb_comp);
132 DoubleVect valeurs_Psi(nbpts);
133 ArrOfDouble somme(dimension);
134 ArrOfDouble a0(dimension),a0a1(dimension),a0a2(dimension),a0a3(dimension);
135
136// Extension possible des volumes de controle:
137 int modif_traitement_diri=( sub_type(Zone_VEF_PreP1b,zone_VEF) ? ref_cast(Zone_VEF_PreP1b,zone_VEF).get_modif_div_face_dirichlet() : 0);
138 modif_traitement_diri = 0; //!< Forcee a 0 car ne marche pas d'apres essais Ulrich&Thomas
139 int nb_face_diri=0;
140 ArrOfInt indice_diri(dimension+1);
141
142// Boucle sur les elements:
143 int nb_elem_tot=zone_VEF.nb_elem_tot();
144 for (int elem=0; elem<nb_elem_tot; elem++)
145 {
146 if (modif_traitement_diri)
147 calculer_coef_som(elem,nb_face_diri,indice_diri,zone_Cl_VEF,zone_VEF);
148
149 double volume=volumes(elem);
150 for (int i=0; i<nbpts; i++)
151 les_polygones(i)=elem;
152
153// On remplit la matrice de changement d'element :
154 const int som_glob = elem_sommets(elem,0);
155 for (int dim=0; dim<dimension; dim++)
156 a0[dim]=coord_sommets(som_glob,dim);
157
158 const int som_glob1 = elem_sommets(elem,1);
159 for (int dim=0; dim<dimension; dim++)
160 a0a1[dim]=coord_sommets(som_glob1,dim)-a0[dim];
161
162 const int som_glob2 = elem_sommets(elem,2);
163 for (int dim=0; dim<dimension; dim++)
164 a0a2[dim]=coord_sommets(som_glob2,dim)-a0[dim];
165
166 if(dimension == 3)
167 {
168 const int som_glob3 = elem_sommets(elem,3);
169 for (int dim=0; dim<dimension; dim++)
170 a0a3[dim]=coord_sommets(som_glob3,dim)-a0[dim];
171 }
172
173// On remplit les_positions :
174 if(dimension == 2)
175 {
176 for (int pt=0; pt<nbpts; pt++)
177 {
178 for (int dim=0; dim<dimension; dim++)
179 les_positions(pt,dim)=a0[dim]
180 +coord_bary(pt,1)* a0a1[dim]
181 +coord_bary(pt,2)* a0a2[dim];
182 }
183 }
184 else if(dimension == 3)
185 {
186 for (int pt=0; pt<nbpts; pt++)
187 {
188 for (int dim=0; dim<dimension; dim++)
189 les_positions(pt,dim)=a0[dim]
190 +coord_bary(pt,1)* a0a1[dim]
191 +coord_bary(pt,2)* a0a2[dim]
192 +coord_bary(pt,3)* a0a3[dim];
193 }
194 }
195 else
196 assert(0);
197
198// Calcul du terme source aux points d'integration :
199 le_scalaire.valeur().valeur_aux_elems(les_positions,les_polygones,valeurs_scalaire);
200 beta().valeur().valeur_aux_elems(les_positions,les_polygones,valeurs_beta);
201
202// Boucle sur les faces de l'element:
203 for(int face=0; face<=dimension; face++)
204 {
205 int num_face=elem_faces(elem, face);
206
207// Calcul des Psi aux points d'integration :
208 for (int pt=0; pt<nbpts; pt++)
209// valeurs_Psi(pt)=(1-dimension*coord_bary(pt,face))*porosite_surf(face); // Fixed bug
210 valeurs_Psi(pt)=(1-dimension*coord_bary(pt,face))*porosite_surf(num_face);
211
212// Integration sur les nbpts points:
213 for (int pt=0; pt<nbpts; pt++)
214 {
215 for (int dim=0; dim<dimension; dim++)
216 {
217 double contrib=0;
218 for (int comp=0; comp<nb_comp; comp++)
219 contrib+=Poids[pt]*volume*(Scalaire0(comp)-valeurs_scalaire(pt,comp))*valeurs_beta(pt,comp)*g(dim)*valeurs_Psi(pt);
220 resu(num_face,dim)+=contrib;
221 somme[dim]+=contrib;
222 if (modif_traitement_diri)
223 {
224 for (int fdiri=0; fdiri<nb_face_diri; fdiri++)
225 {
226 int indice=indice_diri[fdiri];
227 int facel = elem_faces(elem,indice);
228 if (num_face==facel)
229 {
230 resu(facel,dim)-=contrib;
231 double contrib2=contrib/(dimension+1-nb_face_diri);
232 for (int f2=0; f2<dimension+1; f2++)
233 {
234 int face2=elem_faces(elem,f2);
235 resu(face2,dim)+=contrib2;
236 }
237 }
238 }
239 }
240 }
241 }
242 }
243 }
244 {
245// modif pour periodique
246 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
247 {
248 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
249 if (sub_type(Periodique,la_cl.valeur()))
250 {
251 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
252 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
253 int nb_faces_bord=le_bord.nb_faces();
254 IntVect fait(nb_faces_bord);
255 fait = 0;
256 for (int ind_face=0; ind_face<nb_faces_bord; ind_face++)
257 {
258 if (fait(ind_face) == 0)
259 {
260 int ind_face_associee = la_cl_perio.face_associee(ind_face);
261 fait(ind_face) = 1;
262 fait(ind_face_associee) = 1;
263 int face = le_bord.num_face(ind_face);
264 int face_associee = le_bord.num_face(ind_face_associee);
265 for (int dim=0; dim<dimension; dim++)
266 resu(face, dim)=(resu(face_associee, dim)+=resu(face, dim));
267 }//!< if fait
268 }//!< for face
269 }//!< sub_type Perio
270 }
271 }
272
273// Verification:
274 if (0)
275 {
276 Cout <<"Integrale : (";
277 for (int dim=0; dim<dimension; ++dim)
278 {
279 Cout<<somme[dim];
280 if (dim<(dimension-1))
281 Cout<<",";
282 }
283 Cout<<")"<<finl;
284 Process::exit();
285 }
286 return resu;
287}
