248//! Renvoie la valeur du flux impose sur la i-eme composante<br>du champ representant le flux a la frontiere.<br>Le champ a la frontiere est considere constant sur tous<br>les elements de la frontiere.<br>La valeur du flux impose a la frontiere est egale<br>a la valeur du champ (considere constant) a la frontiere divise par d_rho.
249/** \return double<br>
250Signification: la valeur imposee sur la composante du champ specifiee<br>
251Contraintes: */
252double Sortie_libre_pression_moyenne_imposee::flux_impose(int i) const
253{
254 const Milieu_base& mil=ma_zone_cl_dis->equation().milieu();
255 const Champ_base& rho=mil.masse_volumique().valeur();
256 double rho_;
257 assert(!est_egal(d_rho,-123.));
258 if (d_rho==-1)
259 {
260 rho_ = rho(i);
261 }
262 else
263 {
264 rho_ = d_rho;
265 }
266
267 if (le_champ_front.valeurs().size()==1)
268 return le_champ_front(0,0)/rho_;
269 else if (le_champ_front.valeurs().dimension(1)==1)
270 {
271 return le_champ_front(i,0)/rho_;
272 }
273 else
274 Cerr << "Neumann::flux_impose erreur" << finl;
275 exit();
276 return 0.;
277}
