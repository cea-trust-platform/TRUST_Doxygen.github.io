31void Flux_interfacial_Ranz_Marshall::completer()
32{
33 Pb_Multiphase *pbm = sub_type(Pb_Multiphase, pb_.valeur()) ? &ref_cast(Pb_Multiphase, pb_.valeur()) : NULL;
34
35 if (!pbm || pbm->nb_phases() == 1) Process::exit(que_suis_je() + " : not needed for single-phase flow!");
36 for (int n = 0; n < pbm->nb_phases(); n++) //recherche de n_l, n_g : phase {liquide,gaz}_continu en priorite
37 if (pbm->nom_phase(n).debute_par("liquide") && (n_l < 0 || pbm->nom_phase(n).finit_par("continu"))) n_l = n;
38
39 if (n_l < 0) Process::exit(que_suis_je() + " : liquid phase not found!");
40
41 if (!pbm->has_champ("diametre_bulles")) Process::exit("Flux_interfacial_Ranz_Mashall : bubble diameter needed !");
42}

44void Flux_interfacial_Ranz_Marshall::coeffs(const double dh, const double *alpha, const double *T, const double p, const double *nv,
45 const double *lambda, const double *mu, const double *rho, const double *Cp, int e,
46 DoubleTab& hi, DoubleTab& dT_hi, DoubleTab& da_hi, DoubleTab& dp_hi) const
47{
48 int k, N = hi.dimension(0);
49 const DoubleTab& d_bulles = pb_->get_champ("diametre_bulles").valeurs() ;
50 for (k = 0; k < N; k++)
51 if (k != n_l)
52 {
53 double Re_b = rho[n_l] * nv[N * n_l + k] * d_bulles(e,k)/mu[n_l];
54 double Pr = mu[n_l] * Cp[n_l]/lambda[n_l];
55 double Nu = 2 + 0.6* std::pow(Re_b, .5)*std::pow(Pr, .3);
56 hi(n_l, k) = Nu * lambda[n_l] / d_bulles(e,k) * 6 * std::max(alpha[k], 1e-3) / d_bulles(e,k) ; //!< std::max() pour que le flux interfacial sont non nul
57 da_hi(n_l, k, k) = alpha[k] > 1e-3 ? Nu * lambda[n_l] * 6. / (d_bulles(e,k)*d_bulles(e, k)) : 0;
58 hi(k, n_l) = 1e8;
59 }
60}
