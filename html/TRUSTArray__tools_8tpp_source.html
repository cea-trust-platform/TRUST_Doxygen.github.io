20//! Renvoie 1 si les tableaux "v" et "a" sont de la meme taille et contiennent les memes valeurs au sens strict, sinon renvoie 0.<br>Le test est !(v[i]!=a[i])
21template <typename _TYPE_>
22int operator==(const TRUSTArray<_TYPE_>& v, const TRUSTArray<_TYPE_>& a)
23{
24 const int n = v.size_array();
25 const int na = a.size_array();
26 int resu = 1;
27 if (n != na)
28 {
29 resu = 0;
30 }
31 else
32 {
33 const _TYPE_ * vv = v.addr();
34 const _TYPE_ * av = a.addr();
35 for (int i = 0; i < n; i++)
36 {
37 if (av[i] != vv[i])
38 {
39 resu = 0;
40 break;
41 }
42 }
43 }
44 return resu;
45}

159//! Produit scalaire de deux "array" (dotproduct_array remplace operator* car le nom indique clairement que l'on fait un produit scalaire non distribue)
160template <typename _TYPE_>
161_TYPE_ dotproduct_array(const TRUSTArray<_TYPE_>& dx, const TRUSTArray<_TYPE_>& dy)
162{
163 const int size = dx.size_array();
164 assert(size == dy.size_array());
165 _TYPE_ resultat = 0.;
166 if (size > 0)
167 {
168 integer n = size;
169 integer incx = 1;
170 resultat = F77NAME(DDOT)(&n, dx.addr(), &incx, dy.addr(), &incx);
171 }
172 return resultat;
173}
