81void Champ_front_calc_interne::completer()
82{
83#ifdef MEDCOUPLING_
84 using namespace MEDCoupling;
85
86 Champ_front_calc::completer();
87
88// Get barycenters of the concerned faces:
89 const Front_VF& fvf = ref_cast(Front_VF, front_dis());
90 const Zone_VF& zvf = ref_cast(Zone_VF, fvf.zone_dis());
91 const DoubleTab& xv = zvf.xv();
92 int nbfaces = fvf.nb_faces();
93
94 if (nbfaces % 2)
95 {
96 Cerr << "ERROR : Champ_front_calc_interne::completer() - number of faces of the internal boundary is not even!"<< finl;
97 Process::exit(1);
98 }
99 face_map_.resize(nbfaces,Array_base::NOCOPY_NOINIT);
100
101 if (nbfaces>0)
102 {
103 const int ndim = xv.dimension(1);
104 MCAuto<DataArrayDouble> coo(DataArrayDouble::New());
105 coo->alloc(nbfaces, ndim);
106
107 for (int i = 0; i < nbfaces; i++)
108 {
109 int num_f = fvf.num_face(i);
110
111 for (int d = 0; d < ndim; d++)
112 coo->setIJSilent(i, d, xv(num_f, d));
113 }
114
115 DataArrayInt *cP, *cIP;
116 coo->findCommonTuples(Objet_U::precision_geom, 0, cP, cIP);
117 int nb_faces_dup = cP->getNumberOfTuples();
118 if (nb_faces_dup == nbfaces)
119 {
120// Faces dupliquees -> singularite au niveau du gap
121 MCAuto<DataArrayInt> dsi(cIP->deltaShiftIndex());
122
123 if (!dsi->isUniform(2))
124 {
125 Cerr << "ERROR : Problem on internal boundary. Some face(s) not duplicated." << finl;
126 Process::exit(1);
127 }
128
129 const int *p(cP->begin());
130 for (int i = 0; i < nbfaces / 2; i++, p += 2)
131 {
132 face_map_(*p) = *(p + 1);
133 face_map_(*(p + 1)) = *p;
134 }
135 }
136 else
137 {
138// Gap de dimension non-nulle -> on cherche le point
139// d'en face le plus proche
140 const DoubleVect& face_surfaces = zvf.face_surfaces();
141 MCAuto<DataArrayDouble> fnorm(DataArrayDouble::New());
142 fnorm->alloc(nbfaces, 1);
143 fnorm->fillWithZero();
144
145 double* pfnorm = fnorm->getPointer();
146
147 for (int i = 0; i < nbfaces; i++)
148 {
149 int num_f = fvf.num_face(i);
150
151 for (int d = 0; d < ndim; d++)
152 {
153 pfnorm[i]+=zvf.face_normales(num_f,d);
154 coo->setIJSilent(i, d, xv(num_f, d));
155 }
156 pfnorm[i]/=face_surfaces(num_f);
157 }
158
159 double eps = 1e-10;
160
161 MCAuto<DataArrayInt> Idpos = fnorm->findIdsInRange(1-eps,1+eps);
162 MCAuto<DataArrayInt> Idneg = fnorm->findIdsInRange(-1-eps,-1+eps);
163
164 int n1 = Idpos->getNumberOfTuples();
165 int n2 = Idneg->getNumberOfTuples();
166
167 if ((n1!=n2) || (n1+n2)!=nbfaces)
168 {
169 Cerr << "ERROR : Problem on internal boundary. Cannot associate pairs of faces." << finl;
170 Process::exit(1);
171 }
172 MCAuto<DataArrayDouble> coo_pos = coo->selectByTupleId(*Idpos);
173 MCAuto<DataArrayDouble> coo_neg = coo->selectByTupleId(*Idneg);
174
175 const int* pIdpos = Idpos->getConstPointer();
176 const int* pIdneg = Idneg->getConstPointer();
177
178 MCAuto<DataArrayInt> idx = coo_pos->findClosestTupleId(coo_neg);
179
180 const int* pidx = idx->getConstPointer();
181
182 for (int i=0; i<nbfaces/2; i++)
183 {
184 int j2 = pIdneg[i];
185 int k = pidx[i];
186 int j1 = pIdpos[k];
187
188 face_map_(j1) = j2;
189 face_map_(j2) = j1;
190 }
191 }
192 cIP->decrRef();
193 cP->decrRef();
194 }
195#endif
196}
