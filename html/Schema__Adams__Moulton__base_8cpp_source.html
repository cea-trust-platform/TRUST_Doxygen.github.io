84void Schema_Adams_Moulton_base::add_multi_timestep_data(const Equation_base& eqn, Matrice_Base& mat_morse, DoubleTab& secmem) const
85{
86 Equation_base& eqn_bis = ref_cast_non_const(Equation_base,eqn);
87 double time_step = pas_de_temps();
88 double effective_time_step = -1.;
89 double scaling = -1.;
90
91// Remark for "pen" interger : we do not penalize matrices and right members even in the case
92// of Dirichlet or Symetrie boundary conditions
93 int pen = 0;
94 int offset = 0;
95 int i = 0;
96
97 effective_time_step = coefficients()[nb_valeurs_passees()+1] * time_step;
98 scaling = 1./coefficients()[nb_valeurs_passees()+1];
99
100// Modification due to pressure gradient
101 DoubleTab correction(eqn.inconnue().valeurs());
102
103 correction = 0.;
104 eqn_bis.corriger_derivee_expl(correction);
105 correction *= (scaling-1.);
106 secmem += correction;
107
108// Modification due to present time and past times
109 DoubleTab dudt(eqn.inconnue().passe(1)); //due to Initialiser_Champs() offset
110 DoubleTab tmp(dudt);
111
112 for (i=0; i<nb_valeurs_passees()+1; ++i)
113 {
114 offset = nb_valeurs_passees()-i; //present time if offset==0 else past times
115 tmp = eqn.derivee_en_temps().passe(offset);
116 tmp *= coefficients()[i] * time_step;
117 dudt += tmp;
118 }
119
120 eqn.solv_masse().ajouter_masse(effective_time_step,secmem,dudt,pen);
121 eqn.solv_masse().ajouter_masse(effective_time_step,mat_morse,pen);
122}

124// To get du/dt table without pressure gradient and influences of past and present times
125void Schema_Adams_Moulton_base::modifier_second_membre_full_impl(const Equation_base& eqn, DoubleTab& secmem)
126{
127 Equation_base& eqn_bis = ref_cast_non_const(Equation_base,eqn);
128
129 DoubleTab correction(secmem);
130 correction=0.;
131 eqn_bis.corriger_derivee_expl(correction);
132 eqn.solv_masse().appliquer(correction);
133 secmem-=correction; //on enleve le gradient de pression
134
135 if (nb_pas_dt()>nb_pas_dt_seuil())
136 {
137 int i = 0;
138 int offset = 0;
139 double scaling = 1./coefficients()[nb_valeurs_passees()+1];
140
141 for (i=0; i<nb_valeurs_passees()+1; ++i )
142 {
143 offset = nb_valeurs_passees() - i;
144 correction = eqn.derivee_en_temps().passe(offset);//present time if i==nb_valeurs_passees() else past times
145 correction *= coefficients()[i];
146 secmem -= correction;
147 }
148
149 secmem *= scaling;
150 }
151}
