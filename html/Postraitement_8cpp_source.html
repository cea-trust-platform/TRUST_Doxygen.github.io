48// Si forcer != 0, on postraite quel que soit le temps courant,
49// sinon on postraite si l'intervalle entre les postraitements demandes
50// est ecoule.
51void Postraitement::postraiter(int forcer)
52{
53 if (forcer)
54 {
55 if (sondes_demande_) postraiter_sondes();
56 if (besoin_postraiter_champs()) postraiter_champs();
57 if (tableaux_demande_) postraiter_tableaux();
58 }
59 else
60 {
61 if (sondes_demande_) traiter_sondes();
62 if (besoin_postraiter_champs()) traiter_champs();
63 if (tableaux_demande_) traiter_tableaux();
64 }
65 dernier_temps=mon_probleme->schema_temps().temps_courant();
66
67// Cas des statistiques en serie, il faut traiter APRES le postraitement
68 if (lserie_)
69 {
70 if (sup_ou_egal(dernier_temps-tstat_deb_,dt_integr_serie_))
71 {
72 tstat_deb_ = dernier_temps;
73 tstat_dernier_calcul_ = tstat_deb_;
74 tstat_fin_ = tstat_deb_ + dt_integr_serie_;
75 LIST_CURSEUR(Champ_Generique) curseur2 = champs_post_complet_ ;
76 while(curseur2)
77 {
78 if (sub_type(Champ_Gen_de_Champs_Gen,curseur2.valeur().valeur()))
79 {
80 Champ_Gen_de_Champs_Gen& champ_post = ref_cast(Champ_Gen_de_Champs_Gen,curseur2.valeur().valeur());
81 champ_post.fixer_serie(tstat_deb_,tstat_fin_);
82 }
83 ++curseur2;
84 }
85 }
86 }

89void Postraitement::mettre_a_jour(double temps)
90{
91// Mise a jour des operateurs statistiques portes par les Champ_Generique_Statistiques
92 LIST_CURSEUR(Champ_Generique) curseur2 = champs_post_complet_ ;
93 while(curseur2)
94 {
95 curseur2.valeur()->mettre_a_jour(temps);
96 ++curseur2;
97 }
98
99 if ( inf_ou_egal(tstat_deb_,temps) && inf_ou_egal(temps,tstat_fin_) )
100 tstat_dernier_calcul_ = temps;//!< Il y a eu mise a jour effective des integrales
101 if ( sup_ou_egal(temps, tstat_fin_)) //!< si reprise du calcul apres la fin des stats
102 {
103 double dt=probleme().schema_temps().pas_de_temps(); //!< si dt cst !!!
104 if (sup_strict(dt,0)) //!< pour eviter une division par 0
105 tstat_dernier_calcul_ = floor(tstat_fin_/dt)*dt;
106 }
107}

261/** \return Entree\&<br>
262Signification: le flot d'entree modifie<br>
263Contraintes: */
264Entree& Postraitement::readOn(Entree& s )
265{
266 Cerr << "Reading guidelines of postprocessing" << finl;
267 assert(mon_probleme.non_nul());
268
269 const Nom& le_nom_du_post =le_nom();
270 Nom vide;
271 if ((le_nom_du_post!="neant")&&(le_nom_du_post!=vide))
272 {
273 nom_fich_+="_";
274 nom_fich_+=le_nom_du_post;
275 }
276
277 Probleme_base& le_pb = mon_probleme.valeur();
278 le_domaine=le_pb.domaine();
279
280//
281// Creation d'alias pour certains champs
282//
283 Noms liste_noms;
284 const Probleme_base& Pb = probleme();
285 Pb.get_noms_champs_postraitables(liste_noms);
286// On ajoute temperature_physique aux champs definis
287 if ( liste_noms.rang("temperature")!=-1 && !comprend_champ_post("temperature_physique") )
288 {
289 Motcle disc(probleme().discretisation().que_suis_je());
290 if ((disc=="VEF")||(disc=="VEFPreP1b"))
291 {
292 Nom in("{ temperature_physique ");
293 in+="Tparoi_VEF { source refChamp { Pb_champ ";
294 in+=probleme().le_nom();
295 in+=" temperature } } }";
296 Cerr<<" Building of temperature_physique "<<in<<finl;
297 EChaine IN(in);
298 lire_champs_operateurs(IN);
299 }
300 }
301
302 Postraitement_base::readOn(s);
303
304 if (Motcle(format)=="MED") format="med";
305 if (Motcle(format)=="MEDFILE") format="medfile";
306 if (Motcle(format)=="MED_MAJOR") format="med_major";
307 if ((Motcle(format)!="LML")
308 && (Motcle(format)!="MESHTV")
309 && (Motcle(format)!="MED")
310 && (Motcle(format)!="MEDFILE")
311 && (Motcle(format)!="MED_MAJOR")
312 && (Motcle(format)!="LATA")
313 && (Motcle(format)!="LATA_V1")
314 && (Motcle(format)!="LATA_V2")
315 && (Motcle(format)!="XYZ"))
316 {
317 Cerr<<"The postprocessing format " << format << " is not recognized"<<finl;
318 Cerr<<"The recognized formats are lml, meshtv, med, medfile, med_major, xyz, lata, lata_V1 and lata_V2"<<finl;
319 exit();
320 }
321// Changement pour la 1.6.4, le format LATA par defaut est LATA_V2
322 if (Motcle(format)=="LATA") format="lata_V2";
323 if (Motcle(format)=="LATA_V1") format="lata";
324
325 nom_fich_+=".";
326 nom_fich_+=format;
327
328// Les sondes sont completees (en effet, si les sondes
329// ont des champs statistiques, on n'a besoin d'avoir
330// lu le bloc statistiques ET le bloc sondes)
331
332 les_sondes_.completer();
333
334// On type l objet Format_Post
335
336 Nom type_format = "Format_Post_";
337 type_format += format;
338 if (Motcle(format)=="LATA")
339 type_format +="_V1";
340
341 format_post.typer_direct(type_format);
342
343 Nom base_name(nom_fich_);
344 base_name.prefix(format);
345 base_name.prefix(".");
346// format_post->initialize_by_default(base_name);
347 format_post->initialize(base_name,binaire,option_para);
348
349// Le test de verification a ete simplifie entre la v1.5.1 et la v1.5.2
350// On simplifie donc la methode test_coherence
351 format_post->test_coherence(champs_demande_,stat_demande_,dt_post_ch_,dt_post_stat_);
352
353 return s;
354}
