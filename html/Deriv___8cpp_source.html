45Entree& Deriv_::readOn(Entree& is)
46{
47// Efface l'objet existant :
48 detach();
49// static pour le pas creer un Objet_U a chaque fois.
50 static Nom nom_type;
51 is >> nom_type;
52 Objet_U * objet = (Objet_U*) 0;
53 if (nom_type != "vide")
54 {
55 objet = typer(nom_type);
56 if (! objet)
57 {
58 assert(0);
59 exit();
60 }
61 }
62// Appel a la methode virtuelle : met a jour le pointeur_
63 set_O_U_Ptr(objet);
64// Lecture de l'objet derive
65 if (objet)
66 is >> (*objet);
67 return is;
68}

84Objet_U * Deriv_::typer(const char* type)
85{
86// Type_info du type demande
87 const Type_info * type_info = Type_info::type_info_from_name(type);
88// Type de base du DERIV:
89 const Type_info& type_base = get_info_ptr();
90 if ( get_O_U_Ptr())
91 {
92// Cerr<<"on avait deja typer ???"<<finl;
93 detach();
94 }
95 Objet_U * instance = 0;
96
97 if (type_info == 0)
98 {
99 const Synonyme_info* syn_info= Synonyme_info::synonyme_info_from_name(type);
100 if (syn_info!=0)
101 {
102 return typer(syn_info->org_name_());
103 }
104 else
105 {
106 Cerr << "Error in Deriv_::typer_(const char* const type)\n";
107 Cerr << type << " is not a type." << finl;
108 Cerr << "Type required : derived from " << type_base.name() << finl;
109 Cerr << finl;
110 Cerr << type << " is not a recognized keyword." << finl;
111 Cerr << "Check your data set." << finl;
112 Nom nompb = type;
113 if (nompb.find("TURBULENT")!=-1 )
114 Cerr << "Since TRUST V1.8.0, turbulence models are in TrioCFD and not anymore in TRUST.\nTry using TrioCFD executable or contact trust support." << finl;
115 exit();
116 }
117 }
118 if (! type_info->instanciable())
119 {
120 Cerr << "Error in Deriv_::typer_(const char* const type)\n ";
121 Cerr << type << " is not instanciable." << finl;
122 assert(0);
123 exit();
124 }
125 if (! type_info->has_base(&type_base))
126 {
127 Cerr << "Error in Deriv_::typer_(const char* const type)\n ";
128 Cerr << type << " is not a subtype of " << type_base.name() << finl;
129 }
130 else
131 {
132// Cree une instance du type decrit dans type_info
133 instance = type_info->instance();
134 }
135 set_O_U_Ptr(instance);
136 return instance;
137}
