98/** \return Entree\&<br>
99Signification: *this<br>
100Contraintes: */
101int LecFicDistribue::ouvrir(const char* name, IOS_OPEN_MODE mode)
102{
103 Nom nom_fic(name);
104 if(Process::nproc()>1)
105 nom_fic=nom_fic.nom_me(Process::me());
106
107#ifdef FILESYSTEM_NON_GLOBAL
108 Nom localisation;
109 if(Process::je_suis_maitre())
110 localisation = pwd();
111 envoyer_broadcast(localisation, 0);
112
113 nom_fic = localisation + "/" + nom_fic;
114#endif
115
116 if (!EFichier::ouvrir((const char*)nom_fic,mode))
117 {
118 Cerr << "File " << nom_fic << " not found or could not be opened." << finl;
119 return 0;
120 }
121 else
122 {
123 Process::Journal() << "File " << nom_fic << " is opened." << finl;
124 return 1;
125 }
126}
