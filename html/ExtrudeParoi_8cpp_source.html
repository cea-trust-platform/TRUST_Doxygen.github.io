58void calcul_normal2(const ArrOfDouble& pt0, const ArrOfDouble& pt1, const ArrOfDouble& pt2, ArrOfDouble& normal)
59{
60 normal[0]=(pt1[1]-pt0[1])*(pt2[2]-pt0[2])-((pt2[1]-pt0[1])*(pt1[2]-pt0[2]));
61 normal[1]=(pt1[2]-pt0[2])*(pt2[0]-pt0[0])-((pt2[2]-pt0[2])*(pt1[0]-pt0[0]));
62 normal[2]=(pt1[0]-pt0[0])*(pt2[1]-pt0[1])-((pt2[0]-pt0[0])*(pt1[1]-pt0[1]));
63}

67 calcul_normal2(pt0, pt1, pt2, normal);
68 normal/=norme_array(normal);

72void calcul_tab_norme(DoubleTab& tab)
73{
74 for(int i=0; i<tab.dimension(0); i++)
75 {
76 double norm = sqrt(tab(i,0)*tab(i,0)+tab(i,1)*tab(i,1)+tab(i,2)*tab(i,2));
77 norm += DMINFLOAT;
78 tab(i,0)/=norm;
79 tab(i,1)/=norm;
80 tab(i,2)/=norm;
81 }
82}
