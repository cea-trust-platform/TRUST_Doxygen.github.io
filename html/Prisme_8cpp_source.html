151// a coder :
152// est-ce que le prisme de numero element contient le point de
153// coordonnees pos ?
154// adaptation de tetraedre contient
155 assert(pos.size_array()==3);
156 const Zone& zone=ma_zone.valeur();
157 const Domaine& dom=zone.domaine();
158 double prod1,prod2,xn,yn,zn;
159 int som0, som1, som2, som3,som4,som5;
160
161// On regarde tout d'abord si le pintr cherche n'est pas un des
162// sommets du triangle
163 som0 = zone.sommet_elem(ielem,0);
164 som1 = zone.sommet_elem(ielem,1);
165 som2 = zone.sommet_elem(ielem,2);
166 som3 = zone.sommet_elem(ielem,3);
167 som4 = zone.sommet_elem(ielem,4);
168 som5 = zone.sommet_elem(ielem,5);
169 if( ( est_egal(dom.coord(som0,0),pos[0]) && est_egal(dom.coord(som0,1),pos[1]) && est_egal(dom.coord(som0,2),pos[2]) )
170 || (est_egal(dom.coord(som1,0),pos[0]) && est_egal(dom.coord(som1,1),pos[1]) && est_egal(dom.coord(som1,2),pos[2]))
171 || (est_egal(dom.coord(som2,0),pos[0]) && est_egal(dom.coord(som2,1),pos[1]) && est_egal(dom.coord(som2,2),pos[2]))
172 || (est_egal(dom.coord(som3,0),pos[0]) && est_egal(dom.coord(som3,1),pos[1]) && est_egal(dom.coord(som3,2),pos[2]))
173 || (est_egal(dom.coord(som4,0),pos[0]) && est_egal(dom.coord(som4,1),pos[1]) && est_egal(dom.coord(som4,2),pos[2]))
174 || (est_egal(dom.coord(som5,0),pos[0]) && est_egal(dom.coord(som5,1),pos[1]) && est_egal(dom.coord(som5,2),pos[2]))
175 )
176 return 1;
177
178 for (int j=0; j<5; j++)
179 {
180 switch(j)
181 {
182 case 0 :
183 som0 = zone.sommet_elem(ielem,0);
184 som1 = zone.sommet_elem(ielem,1);
185 som2 = zone.sommet_elem(ielem,3);
186 som3 = zone.sommet_elem(ielem,2);
187 break;
188 case 1 :
189 som0 = zone.sommet_elem(ielem,0);
190 som1 = zone.sommet_elem(ielem,2);
191 som2 = zone.sommet_elem(ielem,3);
192 som3 = zone.sommet_elem(ielem,4);

201 som0 = zone.sommet_elem(ielem,0);
202 som1 = zone.sommet_elem(ielem,1);
203 som2 = zone.sommet_elem(ielem,2);
204 som3 = zone.sommet_elem(ielem,3);

214// Algorithme : le sommet 3 et le point M doivent pour j=0 a 3 du meme cote
215// que le plan formes par les points som0,som1,som2.
216// calcul de la normale au plan som0,som1,som2 :
217 xn = (dom.coord(som1,1)-dom.coord(som0,1))*(dom.coord(som2,2)-dom.coord(som0,2))
218 - (dom.coord(som1,2)-dom.coord(som0,2))*(dom.coord(som2,1)-dom.coord(som0,1));
219 yn = (dom.coord(som1,2)-dom.coord(som0,2))*(dom.coord(som2,0)-dom.coord(som0,0))
220 - (dom.coord(som1,0)-dom.coord(som0,0))*(dom.coord(som2,2)-dom.coord(som0,2));
221 zn = (dom.coord(som1,0)-dom.coord(som0,0))*(dom.coord(som2,1)-dom.coord(som0,1))
222 - (dom.coord(som1,1)-dom.coord(som0,1))*(dom.coord(som2,0)-dom.coord(som0,0));
223 prod1 = xn * ( dom.coord(som3,0) - dom.coord(som0,0) )
224 + yn * ( dom.coord(som3,1) - dom.coord(som0,1) )
225 + zn * ( dom.coord(som3,2) - dom.coord(som0,2) );
226 prod2 = xn * ( pos[0] - dom.coord(som0,0) )
227 + yn * ( pos[1] - dom.coord(som0,1) )
228 + zn * ( pos[2] - dom.coord(som0,2) );
229// Si le point est sur le plan (prod2 quasi nul) : on ne peut pas conclure...
230 if (prod1*prod2 < 0 && std::fabs(prod2)>std::fabs(prod1)*Objet_U::precision_geom) return 0;
231 }
232 return 1;

284//! NE FAIT RIEN: A CODER<br>Calcule les volumes des elements de la zone associee.
285void Prisme::calculer_volumes(DoubleVect& volumes) const
286{
287 const Zone& zone=ma_zone.valeur();
288 const IntTab& elem=zone.les_elems();
289 const DoubleTab& coord=zone.domaine().coord_sommets();
290 int size_tot = zone.nb_elem_tot();
291 assert(volumes.size_totale()==size_tot);
292 for (int num_poly=0; num_poly<size_tot; num_poly++)
293 {
294 int i1=elem(num_poly,0);
295 int i2=elem(num_poly,1);
296 int i3=elem(num_poly,2);
297 int i4=elem(num_poly,3);
298 int i5=elem(num_poly,4);
299 int i6=elem(num_poly,5);
300
301// recuperer de MEDMEM_Formulae.hxx
302 double a1 = (coord(i2,0)-coord(i3,0))/2.0, a2 = (coord(i2,1)-coord(i3,1))/2.0, a3 = (coord(i2,2)-coord(i3,2))/2.0;
303 double b1 = (coord(i5,0)-coord(i6,0))/2.0, b2 = (coord(i5,1)-coord(i6,1))/2.0, b3 = (coord(i5,2)-coord(i6,2))/2.0;
304 double c1 = (coord(i4,0)-coord(i1,0))/2.0, c2 = (coord(i4,1)-coord(i1,1))/2.0, c3 = (coord(i4,2)-coord(i1,2))/2.0;
305 double d1 = (coord(i5,0)-coord(i2,0))/2.0, d2 = (coord(i5,1)-coord(i2,1))/2.0, d3 = (coord(i5,2)-coord(i2,2))/2.0;
306 double e1 = (coord(i6,0)-coord(i3,0))/2.0, e2 = (coord(i6,1)-coord(i3,1))/2.0, e3 = (coord(i6,2)-coord(i3,2))/2.0;
307 double f1 = (coord(i1,0)-coord(i3,0))/2.0, f2 = (coord(i1,1)-coord(i3,1))/2.0, f3 = (coord(i1,2)-coord(i3,2))/2.0;
308 double h1 = (coord(i4,0)-coord(i6,0))/2.0, h2 = (coord(i4,1)-coord(i6,1))/2.0, h3 = (coord(i4,2)-coord(i6,2))/2.0;
309
310 double A = a1*c2*f3 - a1*c3*f2 - a2*c1*f3 + a2*c3*f1 +
311 a3*c1*f2 - a3*c2*f1;
312 double B = b1*c2*h3 - b1*c3*h2 - b2*c1*h3 + b2*c3*h1 +
313 b3*c1*h2 - b3*c2*h1;
314 double C = (a1*c2*h3 + b1*c2*f3) - (a1*c3*h2 + b1*c3*f2) -
315 (a2*c1*h3 + b2*c1*f3) + (a2*c3*h1 + b2*c3*f1) +
316 (a3*c1*h2 + b3*c1*f2) - (a3*c2*h1 + b3*c2*f1);
317 double D = a1*d2*f3 - a1*d3*f2 - a2*d1*f3 + a2*d3*f1 +
318 a3*d1*f2 - a3*d2*f1;
319 double E = b1*d2*h3 - b1*d3*h2 - b2*d1*h3 + b2*d3*h1 +
320 b3*d1*h2 - b3*d2*h1;
321 double F = (a1*d2*h3 + b1*d2*f3) - (a1*d3*h2 + b1*d3*f2) -
322 (a2*d1*h3 + b2*d1*f3) + (a2*d3*h1 + b2*d3*f1) +
323 (a3*d1*h2 + b3*d1*f2) - (a3*d2*h1 + b3*d2*f1);
324 double G = a1*e2*f3 - a1*e3*f2 - a2*e1*f3 + a2*e3*f1 +
325 a3*e1*f2 - a3*e2*f1;
326 double H = b1*e2*h3 - b1*e3*h2 - b2*e1*h3 + b2*e3*h1 +
327 b3*e1*h2 - b3*e2*h1;
328 double P = (a1*e2*h3 + b1*e2*f3) - (a1*e3*h2 + b1*e3*f2) -
329 (a2*e1*h3 + b2*e1*f3) + (a2*e3*h1 + b2*e3*f1) +
330 (a3*e1*h2 + b3*e1*f2) - (a3*e2*h1 + b3*e2*f1);
331 volumes[num_poly]=std::fabs(-2.0*(2.0*(A + B + D + E + G + H) + C + F + P)/9.0);
332
333 }
