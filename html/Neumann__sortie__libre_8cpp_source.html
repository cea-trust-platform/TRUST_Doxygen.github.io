57//! Type le_champ_front en "Champ_front_uniforme".<br>Lit les valeurs du champ exterieur si les conditions<br>aux limites sont specifiees: "T_ext", "C_ext", "Y_ext", "K_Eps_ext" ou "F_M_ext"<br>Produit une erreur sinon.

60Contraintes: */
61Entree& Neumann_sortie_libre::readOn(Entree& s )
62{
63// le_champ_front.typer("Champ_front_uniforme");
64 Motcle motlu;
65 Motcles les_motcles(13);
66 {
67 les_motcles[0] = "T_ext";
68 les_motcles[1] = "C_ext";
69 les_motcles[2] = "K_Eps_ext";
70 les_motcles[3] = "F_M_ext";
71 les_motcles[4] = "Fluctu_Temperature_ext";
72 les_motcles[5] = "Flux_Chaleur_Turb_ext";
73 les_motcles[6] = "V2_ext";
74 les_motcles[7] = "Y_ext";
75 les_motcles[8] = "A_ext";
76 les_motcles[9] = "k_ext";
77 les_motcles[10] = "tau_ext";
78 les_motcles[11] = "omega_ext";
79 les_motcles[12] = "k_WIT_ext";
80 les_motcles[12] = "a_i_ext";
81 }
82 s >> motlu;
83 int rang = les_motcles.search(motlu);
84 if (rang >= 0) s >> le_champ_ext;
85 else
86 {
87 Cerr << "Erreur a la lecture de la condition aux limites de type: " << finl;
88 Cerr << que_suis_je() << finl;
89 Cerr << "On attendait " << les_motcles << " a la place de " << motlu << finl;
90 exit();
91 }
92
93 le_champ_front = le_champ_ext;
94
95 return s;
96}

209//! Renvoie un booleen indiquant la compatibilite des conditions<br>aux limites avec l'equation specifiee en parametre.<br>Des CL de type Neumann_sortie_libre sont compatibles<br>avec une equation dont le domaine est la Thermique, le Transport_Keps,<br>la Concentration, la Fraction_massique ou bien indetermine.
210/** \return int<br>
211Signification: valeur booleenne,<br>
2121 si les CL sont compatibles avec l'equation<br>
2130 sinon<br>
214Contraintes: */
215int Neumann_sortie_libre::compatible_avec_eqn(const Equation_base& eqn) const
216{
217 Motcle dom_app=eqn.domaine_application();
218
219 Motcle KEPS ="Transport_Keps";
220 Motcle KEPS_V2 ="Transport_Keps_V2";
221 Motcle K_Eps_Bas_Re ="Transport_Keps_Bas_Re";
222 Motcle K_Eps_Rea ="Transport_Keps_Rea";
223 Motcle Thermique ="Thermique";
224 Motcle Thermique_H ="Thermique_H";
225 Motcle Concentration ="Concentration";
226 Motcle Fraction_massique ="Fraction_massique";
227 Motcle Fraction_volumique ="Fraction_volumique";
228 Motcle V2 ="Transport_V2";
229 Motcle indetermine ="indetermine";
230 Motcle turbulence = "Turbulence";
231 Motcle aire_interfaciale = "Interfacial_area";
232
233 if ( (dom_app==KEPS) || (dom_app==K_Eps_Bas_Re) || (dom_app==K_Eps_Rea) || (dom_app==indetermine) ||
234 (dom_app==Thermique) || (dom_app==Concentration) ||
235 (dom_app==Thermique_H) || (dom_app==V2) || (dom_app==KEPS_V2) ||
236 (dom_app==Fraction_massique) || (dom_app==Fraction_volumique) || (dom_app == turbulence) || (dom_app == aire_interfaciale))
237 return 1;
238 else
239 {
240 err_pas_compatible(eqn);
241 return 0;
242 }
