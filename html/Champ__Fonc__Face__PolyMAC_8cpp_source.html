52int Champ_Fonc_Face_PolyMAC::fixer_nb_valeurs_nodales(int n)
53{
54
55// j'utilise le meme genre de code que dans Champ_Fonc_P0_base
56// sauf que je recupere le nombre de faces au lieu du nombre d'elements
57//
58// je suis tout de meme etonne du code utilise dans
59// Champ_Fonc_P0_base::fixer_nb_valeurs_nodales()
60// pour recuperer la zone discrete...
61
62 const Champ_Fonc_base& self = ref_cast(Champ_Fonc_base, *this);
63 const Zone_VF& la_zone_vf = ref_cast(Zone_VF,self.zone_dis_base());
64
65 assert(n == la_zone_vf.nb_faces());
66
67 const MD_Vector& md = la_zone_vf.md_vector_faces();
68// Probleme: nb_comp vaut 2 mais on ne veut qu'une dimension !!!

77Champ_base& Champ_Fonc_Face_PolyMAC::affecter_(const Champ_base& ch)
78{
79 const DoubleTab& v = ch.valeurs();
80 DoubleTab& val = valeurs();
81 const Zone_PolyMAC& zone_PolyMAC = ref_cast( Zone_PolyMAC,ref_zone_vf_.valeur());
82 int nb_faces = zone_PolyMAC.nb_faces();
83 const DoubleVect& surface = zone_PolyMAC.face_surfaces();
84 const DoubleTab& normales = zone_PolyMAC.face_normales();
85
86
87 if (sub_type(Champ_Uniforme,ch))
88 {
89 for (int num_face=0; num_face<nb_faces; num_face++)
90 {
91 double vn=0;
92 for (int dir=0; dir<dimension; dir++)
93 vn+=v(0,dir)*normales(num_face,dir);
94
95 vn/=surface(num_face);
96 val(num_face) = vn;
97 }
98 }
99 else if (nb_compo_ == dimension)
100 {
101// int ndeb_int = zone_PolyMAC.premiere_face_int();
102// const IntTab& face_voisins = zone_PolyMAC.face_voisins();
103 const DoubleTab& xv=zone_PolyMAC.xv();
104 DoubleTab eval(val.dimension_tot(0),dimension);
105 ch.valeur_aux(xv,eval);
106 for (int num_face=0; num_face<nb_faces; num_face++)
107 {
108 double vn=0;
109 for (int dir=0; dir<dimension; dir++)
110 vn+=eval(num_face,dir)*normales(num_face,dir);
111
112
113 vn/=surface(num_face);
114 val(num_face) = vn;
115 }
116 }
117 else if (nb_compo_ == 1)
118 {
119// int ndeb_int = zone_PolyMAC.premiere_face_int();
120// const IntTab& face_voisins = zone_PolyMAC.face_voisins();
121 const DoubleTab& xv=zone_PolyMAC.xv();
122 ch.valeur_aux(xv,val);
123 }
124 else abort();
125 return *this;
126}

141DoubleTab& Champ_Fonc_Face_PolyMAC::valeur_aux_elems(const DoubleTab& positions,
142 const IntVect& les_polys,
143 DoubleTab& val) const
144{
145 const Champ_base& cha=le_champ();
146 int nb_compo=cha.nb_comp();
147 if (val.nb_dim() == 1)
148 {
149 assert((val.dimension(0) == les_polys.size())||(val.dimension_tot(0) == les_polys.size()));
150 assert(nb_compo == 1);
151 }
152 else if (val.nb_dim() == 2)
153 {
154 assert((val.dimension(0) == les_polys.size())||(val.dimension_tot(0) == les_polys.size()));
155 assert(val.dimension(1) == nb_compo);
156 }
157 else
158 {
159 Cerr << "Erreur TRUST dans Champ_Fonc_Face_PolyMAC::valeur_aux_elems()" << finl;
160 Cerr << "Le DoubleTab val a plus de 2 entrees" << finl;
161 Process::exit();
162 }
163
164 const Zone_PolyMAC& zone_VF = ref_cast(Zone_PolyMAC,zone_vf());
165// const Zone& zone_geom = zone_VDF.zone();
166 const DoubleTab& normales = zone_VF.face_normales();
167// const DoubleVect& surfaces = zone_VF.face_surfaces();
168 const IntTab& elem_faces = zone_VF.elem_faces();
169
170 const DoubleTab& ch = cha.valeurs();
171
172 if (nb_compo == 1)
173 {
174 Cerr<<"Champ_Fonc_Face_PolyMAC::valeur_aux_elems"<<finl;
175 Cerr <<"A scalar field cannot be of Champ_Face type." << finl;
176 Process::exit();
177 }
178 else //!< (nb_compo != 1)
179 {
180 ArrOfDouble vale(nb_compo),s(nb_compo);
181 for(int rang_poly=0; rang_poly<les_polys.size(); rang_poly++)
182 {
183 int le_poly=les_polys(rang_poly);
184 if (le_poly == -1)
185 for(int ncomp=0; ncomp<nb_compo; ncomp++)
186 val(rang_poly,ncomp) = 0;
187 else
188 {
189 vale=0;
190 s=0;
191 int nb_faces_elem_max=elem_faces.dimension(1);
192 int nf;
193
194 for (nf=0; nf<nb_faces_elem_max; nf++)
195 {
196 int face=elem_faces(le_poly,nf);
197 if(face<0)
198 break;
199 for(int ncomp=0; ncomp<nb_compo; ncomp++)
200 {
201// vale(ncomp)+=ch(face)*normales(face,ncomp);
202 vale[ncomp]+=ch(face)*std::fabs(normales(face,ncomp));
203 s[ncomp]+=std::fabs(normales(face,ncomp));
204 }
205 }
206
207
208 for(int ncomp=0; ncomp<nb_compo; ncomp++)
209 val(rang_poly,ncomp)=vale[ncomp]/s[ncomp];
210 }
211 }
212 }
213 return val;
214}
