36// Evaluation du champ sur le domaine par defaut et les sous zones specifiees
37// -Pour chacune des sous zones et pour le domaine complet on calcul CONTRIB_loc:
38// loc designe une localisation (sous zone i ou domaine par defaut)
39// cas VDF : CONTRIB_loc = somme de volume(elem)*porosite(elem) pour les elements de loc
40// cas VEF : CONTRIB_loc = somme de vol_entrelaces(face)*por_face(face) pour les faces de loc
41// -On retranche CONTRIB_loc de chaque sous zone a celle du domaine complet
42// -Estimation de la puissance aux elements pour chaque localisation par la relation :
43// P(elem,compo) = P_lue/CONTRIB_loc
44//
45void Champ_val_tot_sur_vol_base::evaluer(const Zone_dis_base& zdis,const Zone_Cl_dis_base& zcldis)
46{
47 const int nb_elem = zdis.nb_elem();
48 DoubleVect vol_glob_pond;
49 vol_glob_pond = eval_contrib_loc(zdis,zcldis,vol_glob_pond);
50
51 for(int elem=0; elem<nb_elem; elem++)
52 for (int k=0; k<nb_compo_; k++)
53 valeurs_(elem,k) /= vol_glob_pond(0);
54
55 int cpt=1;
56 LIST_CURSEUR(REF(Sous_Zone)) curseur1 = les_sous_zones;
57 while(curseur1)
58 {
59 const Sous_Zone& sz = curseur1.valeur().valeur();
60 int size_sz = sz.nb_elem_tot();
61 int el;
62 for (int elem=0; elem<size_sz; elem++)
63 {
64 el = sz(elem);
65 for (int k=0; k<nb_compo_; k++)
66 {
67 valeurs_(el,k) *= vol_glob_pond(0);
68 valeurs_(el,k) /= vol_glob_pond(cpt);
69 }
70 }
71
72 cpt++;
73 ++curseur1;
74 }
75
76 valeurs().echange_espace_virtuel();
77}
