42// Description
43// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de
44// la convection, de la diffusion pour le cas des faces.
45// Cette matrice a une structure de matrice morse.
46// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
47
48void Op_EF_base::dimensionner(const Zone_EF& la_zone,
49 const Zone_Cl_EF& la_zone_cl,
50 Matrice_Morse& la_matrice) const
51{
52
53 int matrice_stocke=1;
54 if (matrice_stocke)
55 {
56// if (matrice_sto_.nb_colonnes()) PL: non faux en //, nb_colonnes() peut etre nul localement
57 if (Process::mp_sum(matrice_sto_.nb_colonnes()))
58 {
59 la_matrice=matrice_sto_;
60 return;
61 }
62
63 }
64// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de
65// la convection, de la diffusion pour le cas des faces.
66// Cette matrice a une structure de matrice morse.
67// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
68// Pour ce faire il faut chercher les sommets voisins du sommet considere.
69
70 int nb_som=la_zone.nb_som();
71 int nfin = la_zone.nb_som_tot();
72// essai
73 nb_som=nfin;
74
75 const DoubleTab& champ_inconnue = la_zone_cl.equation().inconnue().valeurs();
76 int nb_comp = champ_inconnue.line_size();
77
78// on ne s'occupe pas dans un premier temps des composantes
79
80 const IntTab& elems=la_zone.zone().les_elems();
81// int nb_elem=la_zone.zone().nb_elem();
82 int nb_elem_tot=la_zone.zone().nb_elem_tot();
83 int nb_som_elem=la_zone.zone().nb_som_elem();
84 int nb_som_face=la_zone.nb_som_face();
85
86// Computation of the number of non void coeffs:
87 int extra_nb_coeff = 0;
88// Specific BC that will require extra terms in the matrix:
89 for(int i=0; i<la_zone_cl.nb_cond_lim(); i++)
90 {
91 const Cond_lim_base& ccl = la_zone_cl.les_conditions_limites(i).valeur();
92 if(sub_type(Echange_interne_global_impose, ccl))
93 {
94 if (nb_som_face != 1)
95 {
96 Cerr << "Boundary condition 'Echange_interne_global_impose' or 'Echange_interne_parfait' is not validated for EF and mesh_dim > 1 (i.e. sth other than a wire mesh)!!" << finl;
97 Process::exit(-1);
98 }
99 const Echange_interne_global_impose& cl = ref_cast(Echange_interne_global_impose, ccl);
100 const Champ_front_calc_interne& ch = ref_cast(Champ_front_calc_interne, cl.T_ext().valeur());
101 extra_nb_coeff += ch.face_map().size_array();
102 if(sub_type(Echange_interne_global_parfait, ccl))
103 extra_nb_coeff += ch.face_map().size_array() / 2;
104 }
105 }
106 int nb_coeff=(int)(nb_elem_tot*(nb_som_elem*(nb_som_elem)))+extra_nb_coeff;
107 la_matrice.dimensionner(nb_som*nb_comp,nfin*nb_comp,0);
108 IntTab Indice(nb_coeff,2),p(1,2);
109 int tot=0;
110 for (int elem=0; elem<nb_elem_tot; elem++)
111 {
112 for (int i1=0; i1<nb_som_elem; i1++)
113 {
114 int glob=elems(elem,i1);
115 for (int i2=0; i2<nb_som_elem; i2++)
116 {
117 int glob2=elems(elem,i2);
118 if (glob<nb_som)
119 {
120 Indice(tot,0)=glob;
121 Indice(tot,1)=glob2;
122 tot++;
123 }
124 }
125 }
126 }
127// Specific BC that will require extra terms in the matrix:
128 for(int i=0; i<la_zone_cl.nb_cond_lim(); i++)
129 {
130 const Cond_lim_base& ccl = la_zone_cl.les_conditions_limites(i).valeur();
131 if(sub_type(Echange_interne_global_impose, ccl))
132 {
133 const Echange_interne_global_impose& cl = ref_cast(Echange_interne_global_impose, ccl);
134 bool is_parfait = sub_type(Echange_interne_global_parfait, ccl);
135 const Champ_front_calc_interne& ch = ref_cast(Champ_front_calc_interne, cl.T_ext().valeur());
136 const IntTab& mp = ch.face_map(); //!< indices of (local) *faces*
137 const Front_VF& le_bord = ref_cast(Front_VF,cl.frontiere_dis());
138 int nfacedeb = le_bord.num_premiere_face();
139 int nfacefin = nfacedeb + le_bord.nb_faces();
140 std::vector<bool> hit(le_bord.nb_faces());
141 std::fill(hit.begin(), hit.end(), false);
142 for(int face_recto=nfacedeb; face_recto < nfacefin; face_recto++)
143 {
144 int face_verso = mp(face_recto-nfacedeb)+nfacedeb;
145 int face_plus_2 = 0;
146
147 if (is_parfait)
148 {
149// element attached to the face
150 int elem1 = la_zone.face_voisins(face_verso, 0);
151 int elem = (elem1 != -1) ? elem1 : la_zone.face_voisins(face_verso, 1);
152// other face of the element (i.e. face "after" opp_face)
153 int f1 = la_zone.elem_faces(elem, 0);
154 face_plus_2 = f1 != face_verso ? f1 : la_zone.elem_faces(elem, 1); //!< 2 faces per elem max - 1D
155 }
156
157// all vertices of recto face are put in correspondance with all vertices of verso face:
158 for(int som_idx=0; som_idx < nb_som_face; som_idx++)
159 {
160 int som_recto = la_zone.face_sommets(face_recto, som_idx);
161 for(int som_idx2=0; som_idx2 < nb_som_face; som_idx2++)
162 {
163 int som_verso = la_zone.face_sommets(face_verso, som_idx2);
164// tri-diagonal structure extended through the wall
165// actually this puts too many coeffs in case of is_parfait, but I can't be bothered ...
166 Indice(tot, 0) = som_recto;
167 Indice(tot, 1) = som_verso;
168 tot++;
169
170 if (is_parfait) //!< more complicated: recto face takes (i-1), i and ("next face after opp_face") verso face will just serve to connect recto and verso
171//
172 {
173 int som_plus_2 = la_zone.face_sommets(face_plus_2, som_idx2);
174 if(!hit[face_recto-nfacedeb]) //!< first inner face
175 {
176 Indice(tot, 0) = som_recto;
177 Indice(tot, 1) = som_plus_2;
178 tot++;
179 }
180 else //!< opposite inner face
181 {
182// Nothing to do! no extra tri-band coeff.
183 }
184 }
185 }
186 }
187 hit[face_recto-nfacedeb] = true;
188 hit[face_verso-nfacedeb] = true;
189 }
190 }
191 }
192
193 Indice.resize(tot,2);
194
195 tableau_trier_retirer_doublons(Indice);
196 if (nb_comp==1)
197 la_matrice.dimensionner(Indice);
198 else
199 {
200 int ntot=Indice.dimension(0);
201// on cree pour chaque coeff d'origine un bloc nb_comp*nb_comp
202 IntTab Indice_v(ntot*nb_comp*nb_comp,2);
203 int newcas=0;
204 for (int cas=0; cas<ntot; cas++)
205 {
206 for (int i=0; i<nb_comp; i++)
207 for (int j=0; j<nb_comp; j++)
208 {
209 int newi=Indice(cas,0)*nb_comp+i;
210 int newj=Indice(cas,1)*nb_comp+j;
211 Indice_v(newcas,0)=newi;
212 Indice_v(newcas,1)=newj;
213 newcas++;
214
215 }
216 }
217 tableau_trier_retirer_doublons(Indice_v);
218 la_matrice.dimensionner(Indice_v);
219// Cerr<<Indice<<" nnnnnnnnn "<<Indice_v<<finl;
220// abort();
221 }
222
223 assert(mp_min_vect(la_matrice.get_tab2())>-1);
224
225 if (matrice_stocke)
226 {
227 ref_cast_non_const(Matrice_Morse,matrice_sto_)=la_matrice;
228 }
229//
230 if (0)
231 {
232 Cerr<<la_matrice.get_tab1()<<finl;
233 Cerr<<la_matrice.get_tab2()<<finl;
234 la_matrice.get_set_coeff() = 1;
235 Cerr<<"ici "<<tot <<" "<< nb_coeff<<finl;
236 Cerr<<"Indice"<<Indice<<finl;
237 la_matrice.imprimer_formatte(Cerr);
238
239 Process::exit();
240 }
241
242}

248void Op_EF_base::modifier_pour_Cl(const Zone_EF& la_zone,
249 const Zone_Cl_EF& la_zone_cl,
250 Matrice_Morse& la_matrice, DoubleTab& secmem) const
251{
252// Cerr<<__PRETTY_FUNCTION__<<" ne fait rien "<<finl; return;
253 Cerr<<__PRETTY_FUNCTION__<<" a revoir "<<finl;
254
255// Dimensionnement de la matrice qui devra recevoir les coefficients provenant de
256// la convection, de la diffusion pour le cas des faces.
257// Cette matrice a une structure de matrice morse.
258// Nous commencons par calculer les tailles des tableaux tab1 et tab2.
259
260 int k, nbvois;
261// int nfin = la_zone.nb_faces();
262 const Conds_lim& les_cl = la_zone_cl.les_conditions_limites();
263
264 const IntVect& tab1=la_matrice.get_tab1();
265 DoubleVect& coeff = la_matrice.get_set_coeff();
266
267// determination de la taille du champ inconnue.
268// Cerr << "dans modifier cl " << finl;
269 const DoubleTab& champ_inconnue = la_zone_cl.equation().inconnue().valeurs();
270 int nb_comp = champ_inconnue.line_size();
271
272 const IntTab& faces_sommets=la_zone.face_sommets();
273 int nb_som_face=faces_sommets.dimension(1);
274
275 int nb_som=secmem.dimension(0);
276// Cerr << "nb_comp " << nb_comp << finl;
277 for (int i=0; i<les_cl.size(); i++)
278 {
279 const Cond_lim& la_cl = les_cl[i];
280 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
281 int nfaces = le_bord.nb_faces_tot();
282 if (sub_type(Dirichlet,la_cl.valeur()))
283 {
284 const Dirichlet& la_cl_Dirichlet = ref_cast(Dirichlet,la_cl.valeur());
285 for (int ind_face=0; ind_face < nfaces; ind_face++)
286 for (int s=0; s<nb_som_face; s++)
287 {
288 int face=le_bord.num_face(ind_face);
289 int som=faces_sommets(face,s);
290 if (som<nb_som)
291 for (int comp=0; comp<nb_comp; comp++)
292 {
293 int m=som*nb_comp+comp;
294 int idiag = tab1[som*nb_comp+comp]-1;
295 nbvois = tab1[som*nb_comp+1+comp] - tab1[som*nb_comp+comp];
296 for (k=0; k < nbvois; k++)
297 {
298 coeff[idiag+k]=0;
299 }
300 la_matrice.coef(m,m)=1;
301 assert(la_matrice.coef(som*nb_comp+comp,som*nb_comp+comp)==1);
302// pour les voisins
303
304
305// pour le second membre
306 secmem(som,comp)= la_cl_Dirichlet.val_imp(ind_face,comp);
307 secmem(som,comp)= champ_inconnue(som,comp);
308 }
309 }
310 }
311 if (sub_type(Dirichlet_homogene,la_cl.valeur()))
312 {
313 const Dirichlet_homogene& la_cl_Dirichlet = ref_cast(Dirichlet_homogene,la_cl.valeur());
314 for (int ind_face=0; ind_face < nfaces; ind_face++)
315 for (int s=0; s<nb_som_face; s++)
316 {
317 int face=le_bord.num_face(ind_face);
318 int som=faces_sommets(face,s);
319
320 for (int comp=0; comp<nb_comp; comp++)
321 {
322 int m=som*nb_comp+comp;
323 int idiag = tab1[som*nb_comp+comp]-1;
324 nbvois = tab1[som*nb_comp+1+comp] - tab1[som*nb_comp+comp];
325 for (k=0; k < nbvois; k++)
326 {
327 coeff[idiag+k]=0;
328 }
329 la_matrice.coef(m,m)=1;
330 assert(la_matrice.coef(som*nb_comp+comp,som*nb_comp+comp)==1);
331// pour les voisins
332
333
334// pour le second membre
335 secmem(som,comp)= 0;
336 if (la_cl_Dirichlet.val_imp(ind_face,comp)!=0)
337 {
338 Cerr<<"impossible non ??? "<<finl;
339 Process::exit();
340 }
341 }
342 }
343 }
344
345 if (sub_type(Symetrie,la_cl.valeur()))
346 {
347// on ne fait rien ici on fait ensuite
348 }
349 }
350
351// On modifie pour la symetrie
352
353 if (la_zone_cl.equation().inconnue().valeur().nature_du_champ()==vectoriel)
354 {
355 la_zone_cl.imposer_symetrie_matrice_secmem(la_matrice,secmem);
356 }
357
358 /*
359 la_matrice.imprimer(Cerr);
360 la_matrice.imprimer_formatte(Cerr);
361 Cerr<<"scd "<<secmem<<finl;
362 */
363// exit();
364// Prise en compte des conditions de type periodicite
365// Cerr << "fin de modifier" << finl;
366}

368// Description multiplie le flux bordpar rho cp ou rho si necessaire
369void Op_EF_base::modifier_flux( const Operateur_base& op) const
370{
371 controle_modifier_flux_=1;
372 const Zone_EF& la_zone_EF=ref_cast(Zone_EF,op.equation().zone_dis().valeur());
373 DoubleTab& flux_bords_=op.flux_bords();
374 int nb_compo=flux_bords_.dimension(1);
375 const Probleme_base& pb=op.equation().probleme();
376
377// On multiplie par rho si Navier Stokes incompressible
378 Nom nom_eqn=op.equation().que_suis_je();
379 if (nom_eqn.debute_par("Navier_Stokes") && pb.milieu().que_suis_je()=="Fluide_Incompressible")
380 {
381 const Champ_base& rho = op.equation().milieu().masse_volumique().valeur();
382 if (sub_type(Champ_Uniforme,rho))
383 {
384 double coef = rho(0,0);
385 int nb_faces_bord=la_zone_EF.nb_faces_bord();
386 for (int face=0; face<nb_faces_bord; face++)
387 for(int k=0; k<nb_compo; k++)
388 flux_bords_(face,k) *= coef;
389 }
390 }
391
392}

394// Description
395// Impression des flux d'un operateur EF aux faces (ie: diffusion, convection)
396//
397int Op_EF_base::impr(Sortie& os, const Operateur_base& op) const
398{
399 const Zone_EF& la_zone_EF=ref_cast(Zone_EF,op.equation().zone_dis().valeur());
400 DoubleTab& flux_bords_=op.flux_bords();
401 if (flux_bords_.nb_dim()!=2)
402 {
403 Cout << "L'impression des flux n'est pas codee pour l'operateur " << op.que_suis_je() << finl;
404 return 1;
405 }
406 if (controle_modifier_flux_==0)
407 {
408 Cerr<<op.que_suis_je()<<" appelle Op_EF_base::impr sans avoir appeler Op_EF_base::modifier_flux, on arrete tout "<<finl;
409 Process::exit();
410 }
411
412 int nb_compo=flux_bords_.dimension(1);
413 const Probleme_base& pb=op.equation().probleme();
414 const Schema_Temps_base& sch=pb.schema_temps();
415// On n'imprime les moments que si demande et si on traite l'operateur de diffusion de la vitesse
416 int impr_mom=0;
417 if (la_zone_EF.zone().Moments_a_imprimer() && sub_type(Operateur_Diff_base,op) && op.equation().inconnue().le_nom()=="vitesse")
418 impr_mom=1;
419
420 const int impr_sum=(la_zone_EF.zone().Bords_a_imprimer_sum().est_vide() ? 0:1);
421 const int impr_bord=(la_zone_EF.zone().Bords_a_imprimer().est_vide() ? 0:1);
422 int flag=0;
423 if (Process::je_suis_maitre()) flag=1;
424// SFichier Flux;
425 if (!Flux.is_open()) op.ouvrir_fichier( Flux,"",flag);
426// SFichier Flux_moment;
427 if (!Flux_moment.is_open()) op.ouvrir_fichier(Flux_moment,"moment",impr_mom&&flag);
428// SFichier Flux_sum;
429 if (!Flux_sum.is_open()) op.ouvrir_fichier(Flux_sum,"sum",impr_sum&&flag);
430 EcrFicPartage Flux_face;
431 op.ouvrir_fichier_partage(Flux_face,"",impr_bord);
432
433
434// Impression du temps
435 double temps=sch.temps_courant();
436 if(Process::je_suis_maitre())
437 {
438 Flux.add_col(temps);
439 if (impr_mom) Flux_moment.add_col(temps);
440 if (impr_sum) Flux_sum.add_col(temps);
441 }
442
443// Calcul des moments
444 const int nb_faces = la_zone_EF.nb_faces_tot();
445 DoubleTab xgr(nb_faces,Objet_U::dimension);
446 xgr=0.;
447 DoubleVect moment(nb_compo);
448 moment=0;
449 if (impr_mom)
450 {
451 const DoubleTab& xgrav = la_zone_EF.xv();
452 const ArrOfDouble& c_grav=la_zone_EF.zone().cg_moments();
453 for (int num_face=0; num_face <nb_faces; num_face++)
454 for (int i=0; i<Objet_U::dimension; i++)
455 xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
456 }
457
458// On parcours les frontieres pour sommer les flux par frontiere dans le tableau flux_bord
459 DoubleVect flux_bord(nb_compo);
460 DoubleVect bilan(nb_compo);
461 bilan = 0;
462 for (int num_cl=0; num_cl<la_zone_EF.nb_front_Cl(); num_cl++)
463 {
464 flux_bord=0;
465 const Frontiere_dis_base& la_fr = op.equation().zone_Cl_dis().les_conditions_limites(num_cl).frontiere_dis();
466 const Cond_lim& la_cl = op.equation().zone_Cl_dis().les_conditions_limites(num_cl);
467 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
468 int ndeb = frontiere_dis.num_premiere_face();
469 int nfin = ndeb + frontiere_dis.nb_faces();
470 for (int face=ndeb; face<nfin; face++)
471 {
472 for(int k=0; k<nb_compo; k++)
473 flux_bord(k)+=flux_bords_(face, k);
474
475 if (impr_mom)
476 {
477// Calcul du moment exerce par le fluide sur le bord (OM/\F)
478 if (Objet_U::dimension==2)
479 moment(0)+=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
480 else
481 {
482 moment(0)+=flux_bords_(face,2)*xgr(face,1)-flux_bords_(face,1)*xgr(face,2);
483 moment(1)+=flux_bords_(face,0)*xgr(face,2)-flux_bords_(face,2)*xgr(face,0);
484 moment(2)+=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
485 }
486 }
487 }
488
489// On somme les contributions de chaque processeur
490 for(int k=0; k<nb_compo; k++)
491 {
492 flux_bord(k)=Process::mp_sum(flux_bord(k));
493 if (impr_mom) moment(k)=Process::mp_sum(moment(k));
494 }
495
496// Ecriture dans les fichiers
497 if (Process::je_suis_maitre())
498 {
499 for(int k=0; k<nb_compo; k++)
500 {
501 Flux.add_col(flux_bord(k));
502 if (impr_mom) Flux_moment.add_col(moment(k));
503 if (la_zone_EF.zone().Bords_a_imprimer_sum().contient(la_fr.le_nom())) Flux_sum.add_col(flux_bord(k));
504
505// On somme les flux de toutes les frontieres pour mettre dans le tableau bilan
506 bilan(k)+=flux_bord(k);
507 }
508 }
509 }
510
511// On imprime les bilans et on va a la ligne
512 if(Process::je_suis_maitre())
513 {
514 for(int k=0; k<nb_compo; k++)
515 Flux.add_col(bilan(k));
516
517 Flux << finl;
518 if (impr_mom) Flux_moment << finl;
519 if (impr_sum) Flux_sum << finl;
520 }
521
522// Impression sur chaque face si demande
523 for (int num_cl=0; num_cl<la_zone_EF.nb_front_Cl(); num_cl++)
524 {
525 const Frontiere_dis_base& la_fr = op.equation().zone_Cl_dis().les_conditions_limites(num_cl).frontiere_dis();
526 const Cond_lim& la_cl = op.equation().zone_Cl_dis().les_conditions_limites(num_cl);
527 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
528 int ndeb = frontiere_dis.num_premiere_face();
529 int nfin = ndeb + frontiere_dis.nb_faces();
530// Impression sur chaque face
531 if (la_zone_EF.zone().Bords_a_imprimer().contient(la_fr.le_nom()))
532 {
533 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps " << temps << " : " << finl;
534 const DoubleTab& xv=la_zone_EF.xv();
535 for (int face=ndeb; face<nfin; face++)
536 {
537 if (Objet_U::dimension==2)
538 Flux_face << "# Face a x= " << xv(face,0) << " y= " << xv(face,1) << " flux=" ;
539 else if (Objet_U::dimension==3)
540 Flux_face << "# Face a x= " << xv(face,0) << " y= " << xv(face,1) << " z= " << xv(face,2) << " flux=" ;
