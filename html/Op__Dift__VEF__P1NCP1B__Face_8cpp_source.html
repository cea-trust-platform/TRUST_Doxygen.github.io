68double Op_Dift_VEF_P1NCP1B_Face::
69calculer_dt_stab() const
70{
71// Calcul de dt_stab
72// La diffusivite est constante par elements donc
73// il faut calculer dt_diff pour chaque element et
74// dt_stab=Min(dt_diff (K) = h(K)*h(K)/(2*dimension*diffu2_(K)))
75// ou diffu2_ est la somme des 2 diffusivite laminaire et turbulente
76
77 int num_face;
78 double vol,surf;
79 double surf_max;
80 double dt_stab=1.e30;
81 double coef;
82 const Zone_VEF_PreP1b& la_zone_VEF = la_zone_vef.valeur();
83 const DoubleVect& volumes = la_zone_VEF.volumes();
84 const IntTab& elem_faces = la_zone_VEF.elem_faces();
85 const DoubleTab& face_normales = la_zone_VEF.face_normales();
86 const Zone& la_zone= la_zone_VEF.zone();
87// const int nb_elem = la_zone.nb_elem_tot();
88// const DoubleTab& xp=la_zone_VEF.xp();
89 int nb_faces_elem = la_zone.nb_faces_elem();
90 double diffu = (diffusivite_.valeur())(0,0);
91 const DoubleVect& diffu_turb=diffusivite_turbulente()->valeurs();
92 double diffu2_;
93// diffu_turb=1.;
94// for( int j=0;j<nb_elem;j++)
95// diffu_turb[j]=1.-0.9*xp(j,1);
96 int la_zone_nb_elem=la_zone.nb_elem();
97 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
98 {
99 surf_max = 1.e-30;
100 for (int i=0; i<nb_faces_elem; i++)
101 {
102 num_face = elem_faces(num_elem,i);
103 surf = face_normales(num_face,0)*face_normales(num_face,0);
104 for (int j=1; j<dimension; j++)
105 surf += face_normales(num_face,j)*face_normales(num_face,j);
106 surf_max = (surf > surf_max)? surf : surf_max;
107 }
108 vol = volumes(num_elem);
109 vol *=vol/surf_max;
110 diffu2_ = diffu + diffu_turb[num_elem];
111 coef=vol/(2.*dimension*(diffu2_+DMINFLOAT));
112 dt_stab = (coef < dt_stab )? coef : dt_stab;
113 }
114 return dt_stab;
115}

117DoubleTab& Op_Dift_VEF_P1NCP1B_Face::
118calculer_gradient_elem(const DoubleTab& vit,
119 DoubleTab& grad) const
120{
121 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
122 (la_zone_vef.valeur()));
123
124 const Zone& zone = zone_VEF.zone();
125 const DoubleTab& face_normales = zone_VEF.face_normales();
126 const IntTab& elem_faces=zone_VEF.elem_faces();
127 const IntTab& face_voisins=zone_VEF.face_voisins();
128 int nfe=zone.nb_faces_elem();
129 int nb_elem_tot=zone.nb_elem_tot();
130 int elem,indice,face,compi, compj;
131 double signe;
132
133 for(elem=0; elem<nb_elem_tot; elem++)
134 {
135 for(indice=0; indice<nfe; indice++)
136 {
137 face = elem_faces(elem,indice);
138 signe=1;
139 if(elem!=face_voisins(face,0))
140 signe=-1;
141 for(compi=0; compi<dimension; compi++)
142 for(compj=0; compj<dimension; compj++)
143 grad(elem,compi,compj)+=signe*vit(face, compi)*
144 face_normales(face,compj);
145 }
146 }
147 return grad;

150DoubleTab& Op_Dift_VEF_P1NCP1B_Face::
151calculer_gradient_som(const DoubleTab& vit,
152 DoubleTab& grad) const
153{
154
155 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
156 (la_zone_vef.valeur()));
157 const Zone& zone = zone_VEF.zone();
158 const Domaine& dom=zone.domaine();
159 const DoubleTab& face_normales = zone_VEF.face_normales();
160 const IntTab& som_elem=zone.les_elems();
161 const IntTab& elem_faces=zone_VEF.elem_faces();
162 const IntTab& face_voisins=zone_VEF.face_voisins();
163
164 double mijK, miiK, miK=1./(dimension+1)/(dimension+1);
165 int nfe=zone.nb_faces_elem();
166 int nb_elem_tot=zone.nb_elem_tot();
167 int nps=zone_VEF.numero_premier_sommet();
168 int nb_som=grad.dimension(0)-nb_elem_tot;
169 int nb_som_elem = zone.nb_som_elem();
170 const IntTab& elem_som = zone.les_elems();
171 DoubleTab secmem(nb_som, dimension, dimension);
172
173 if(Objet_U::dimension==2)
174 {
175 mijK=1./12.;
176 miiK= 1./6.;
177 }
178 else
179 {
180 mijK=1./18.;
181 miiK= 1./12.;
182 }
183
184 if ( masse.nb_lignes() <2 )
185 {
186 Cerr<<"On remplit la matrice de masse"<<finl;
187 (ref_cast_non_const(DoubleTab,savgrad)).resize(nb_som,dimension,dimension);
188 Matrice_Morse_Sym& mat=ref_cast_non_const(Matrice_Morse_Sym, masse);
189 int rang;
190 int nnz=0;
191 IntLists voisins(nb_som);
192 DoubleLists coeffs(nb_som);
193 DoubleVect diag(nb_som);
194 int elem;
195 for (elem=0; elem<nb_elem_tot; elem++)
196 {
197 double vol=zone_VEF.volumes(elem);
198 double coeff_ij=mijK*vol-miK*vol;
199 double coeff_ii=miiK*vol-miK*vol;
200 for (int isom=0; isom<nb_som_elem; isom++)
201 {
202 int som=elem_som(elem,isom);
203 int ii=dom.get_renum_som_perio(som);
204 diag[ii]+=coeff_ii;
205 for (int jsom=isom+1; jsom<nb_som_elem; jsom++)
206 {
207 int sombis=elem_som(elem,jsom);
208 int j=dom.get_renum_som_perio(sombis);
209 int i=ii;
210 if(i>j)
211 {
212 int kl=j;
213 j=i;
214 i=kl;
215 }
216 rang=voisins[i].rang(j);
217 if(rang==-1)
218 {
219 voisins[i].add(j);
220 coeffs[i].add(coeff_ij);
221 nnz++;
222 }
223 else
224 {
225 coeffs[i][rang]+=coeff_ij;
226 }
227 }
228 }
229 }
230
231
232 {
233 for (int i=0; i<nb_som; i++)
234 if(diag(i)==0)
235 diag(i)=1.;
236 }
237
238 mat.dimensionner(nb_som, nnz+nb_som) ;
239 mat.remplir(voisins, coeffs, diag) ;
240 mat.compacte() ;
241 mat.set_est_definie(1);
242 }
243 int elem,indice,face,compi,compj,som;
244 ArrOfDouble sigma(dimension);
245 static double coeff_som=1./(dimension)/(dimension+1);
246 double signe;
247
248 for(elem=0; elem<nb_elem_tot; elem++)
249 {
250 sigma = 0;
251 for(indice=0; indice<nfe; indice++)
252 {
253 face = elem_faces(elem,indice);
254 for(int comp=0; comp<dimension; comp++)
255 sigma[comp]+=vit(face, comp);
256 }
257 for(indice=0; indice<nfe; indice++)
258 {
259 som = dom.get_renum_som_perio(som_elem(elem,indice));
260 face = elem_faces(elem,indice);
261 signe=1;
262 if(elem!=face_voisins(face,0))
263 signe=-1;
264 for(compi=0; compi<dimension; compi++)
265 for(compj=0; compj<dimension; compj++)
266 secmem(som, compi, compj)+=coeff_som*signe*
267 sigma[compi]*face_normales(face,compj);
268 }
269 }
270// Cerr << "Avant CL secmem = " << secmem << finl;
271 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
272 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
273 const IntTab& face_sommets = zone_VEF.face_sommets();
274 int nb_bords =les_cl.size();
275 for (int n_bord=0; n_bord<nb_bords; n_bord++)
276 {
277 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
278 {
279 if (!sub_type(Periodique,la_cl.valeur()))
280 {
281 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
282 int num1 = le_bord.num_premiere_face();
283 int num2 = num1 + le_bord.nb_faces();
284 for (face=num1; face<num2; face++)
285 {
286 for(indice=0; indice<(nfe-1); indice++)
287 {
288 som=dom.get_renum_som_perio(face_sommets(face,indice));
289 for (compi=0; compi<dimension; compi++)
290 for (compj=0; compj<dimension; compj++)
291 secmem(som, compi, compj) += 1./dimension*
292 vit(face,compi) * face_normales(face,compj) ;
293 }
294 }
295 }
296 }
297 }
298// Cerr << "Apres CL secmem = " << secmem << finl;
299 double coeff=1./(dimension+1.);
300 {
301 DoubleVect secmemij;
302 DoubleVect gradij;
303 dom.creer_tableau_sommets(secmemij);
304 dom.creer_tableau_sommets(gradij);
305 DoubleTab& sgrad = ref_cast_non_const(DoubleTab, savgrad);
306
307 for (compi=0; compi<dimension; compi++)
308 for (compj=0; compj<dimension; compj++)
309 {
310 int i;
311 for(i=0; i<nb_som; i++)
312 {
313 int soml=dom.get_renum_som_perio(i);
314 secmemij(soml)=secmem(soml,compi,compj);
315 gradij(soml)=sgrad(soml,compi,compj);
316 }
317 for(elem=0; elem<nb_elem_tot; elem++)
318 for(indice=0; indice<nfe; indice++)
319 {
320 som = dom.get_renum_som_perio(som_elem(elem,indice));
321 secmemij(som)-=coeff*grad(elem,compi,compj);
322 }
323 (ref_cast_non_const(SolveurSys,solveur)).resoudre_systeme(masse, secmemij, gradij);
324 for(i=0; i<nb_som; i++)
325 {
326 int soml=dom.get_renum_som_perio(i);
327 double x = gradij(soml);
328 grad(nps+soml,compi,compj)=x;
329 sgrad(soml,compi,compj)=x;
330 }
331 }
332 }
333 for (elem=0; elem<nb_elem_tot; elem++)
334 {
335 double vol=zone_VEF.volumes(elem);
336 for (compi=0; compi<dimension; compi++)
337 for (compj=0; compj<dimension; compj++)
338 grad(elem,compi,compj)/=vol;
339 for(indice=0; indice<nfe; indice++)
340 {
341 som = nps+dom.get_renum_som_perio(som_elem(elem,indice));
342 for (compi=0; compi<dimension; compi++)
343 for (compj=0; compj<dimension; compj++)
344 grad(elem,compi,compj)-=coeff*grad(som,compi,compj);
345 }
346 }
347
348 DoubleTab tmp(grad);
349 int ntot=tmp.dimension(0);
350 for(int ind=0; ind<ntot; ind++)
351 for (compi=0; compi<dimension; compi++)
352 for (compj=0; compj<dimension; compj++)
353 tmp(ind,compi, compj)=0.5*(grad(ind,compi, compj)+grad(ind,compj, compi));
354 grad=tmp;
355// Cout << "grad = " << grad << finl;
356 return grad;
357}

359DoubleTab& Op_Dift_VEF_P1NCP1B_Face::
360corriger_div_pour_Cl(DoubleTab& div ) const
361{
362 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
363 la_zone_vef.valeur());
364 const DoubleTab& face_normales = zone_VEF.face_normales();
365 const Zone_Cl_VEF& zone_Cl_VEF=la_zcl_vef.valeur();
366 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
367 int nb_bords =les_cl.size();
368 for (int n_bord=0; n_bord<nb_bords; n_bord++)
369 {
370 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
371 {
372 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
373 int num1 = le_bord.num_premiere_face();
374 int num2 = num1 + le_bord.nb_faces();
375 if (sub_type(Periodique,la_cl.valeur()))
376 {
377// periodicite
378 const Periodique& la_cl_perio = ref_cast(Periodique,
379 la_cl.valeur());
380 ArrOfInt fait(le_bord.nb_faces());
381 for (int face=num1; face<num2; face++)
382 {
383 if(!fait[face-num1])
384 {
385 int face_associee=num1+la_cl_perio.face_associee(face-num1);
386 fait[face-num1]=(fait[face_associee-num1]=1);
387 for (int comp=0; comp<dimension; comp++)
388 {
389 div(face,comp)+= div(face_associee,comp);
390 div(face_associee,comp)=div(face,comp);
391 }
392 }
393 }
394 }
395 else
396 {
397 if(sub_type(Symetrie,la_cl.valeur()))
398 {
399 for (int face=num1; face<num2; face++)
400 {
401 double psc=0;
402 double norme=0;
403 int comp;
404 for (comp=0; comp<dimension; comp++)
405 psc += div(face,comp)*face_normales(face,comp);
406 for (comp=0; comp<dimension; comp++)
407 norme += face_normales(face,comp)*face_normales(face,comp);
408
409// psc/=norme; // Fixed bug: Arithmetic exception
410 if (std::fabs(norme)>=DMINFLOAT) psc/=norme;
411 for (comp=0; comp<dimension; comp++)
412 div(face,comp)-=psc*face_normales(face,comp);
413 }
414 }
415 }
416 }
417 }
418// Cout << "div = " << div << finl;
419 return div;
420}

422DoubleTab& Op_Dift_VEF_P1NCP1B_Face::
423calculer_divergence_elem(double nu, const DoubleTab& nu_turb,
424 const DoubleTab& grad,
425 DoubleTab& div) const
426{
427 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
428 la_zone_vef.valeur());
429 const Zone& zone = zone_VEF.zone();
430 const DoubleTab& face_normales = zone_VEF.face_normales();
431 const IntTab& elem_faces=zone_VEF.elem_faces();
432 const IntTab& face_voisins=zone_VEF.face_voisins();
433 int nfe=zone.nb_faces_elem();
434 int nb_elem_tot=zone.nb_elem_tot();
435
436 int elem,indice,face,compi, compj;
437 ArrOfDouble sigma(dimension);
438 for(elem=0; elem<nb_elem_tot; elem++)
439 {
440 double nu_tot=nu+nu_turb(elem);
441 for(indice=0; indice<nfe; indice++)
442 {
443 face = elem_faces(elem,indice);
444 double signe=1;
445 if(elem!=face_voisins(face,0))
446 signe=-1;
447 for(compi=0; compi<dimension; compi++)
448 for(compj=0; compj<dimension; compj++)
449 div(face, compi)-=nu_tot*grad(elem, compi, compj)
450 *signe*face_normales(face,compj);
451 }
452 }
453// Cerr << "AP elem div =" << div << finl;
454 return div;
455}
456DoubleTab& Op_Dift_VEF_P1NCP1B_Face::
457calculer_divergence_som(double nu, const DoubleTab& nu_turb,
458 const DoubleTab& grad,
459 DoubleTab& div) const
460{
461 const Zone_VEF_PreP1b& zone_VEF = ref_cast(Zone_VEF_PreP1b,
462 la_zone_vef.valeur());
463 const Zone& zone = zone_VEF.zone();
464 const Domaine& dom=zone.domaine();
465 const DoubleTab& face_normales = zone_VEF.face_normales();
466 const IntTab& som_elem=zone.les_elems();
467 const IntTab& elem_faces=zone_VEF.elem_faces();
468 const IntTab& face_voisins=zone_VEF.face_voisins();
469 int nfe=zone.nb_faces_elem();
470 int nb_elem_tot=zone.nb_elem_tot();
471 int nps=zone_VEF.numero_premier_sommet();
472 int elem,indice,indice2,face,compi,compj,som;
473 ArrOfDouble sigma(dimension);
474 static double coeff_som=1./(dimension)/(dimension+1);
475 for(elem=0; elem<nb_elem_tot; elem++)
476 {
477 double nu_tot=nu+nu_turb(elem);
478 for(indice=0; indice<nfe; indice++)
479 {
480 som = nps+dom.get_renum_som_perio(som_elem(elem,indice));
481 face = elem_faces(elem,indice);
482 double signe=1;
483 if(elem!=face_voisins(face,0))
484 signe=-1;
485 for(int comp=0; comp<dimension; comp++)
486 {
487 sigma[comp]=signe*face_normales(face,comp);
488 }
489 for(indice2=0; indice2<nfe; indice2++)
490 for(compi=0; compi<dimension; compi++)
491 for(compj=0; compj<dimension; compj++)
492 div(elem_faces(elem,indice2),compi)-=coeff_som*nu_tot*
493 grad(som,compi,compj)*sigma[compj];
494 }
495 }
496// Cerr << "div AV CL : " << div << finl;
497 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
498 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
499 const IntTab& face_sommets = zone_VEF.face_sommets();
500 int nb_bords =les_cl.size();
501 for (int n_bord=0; n_bord<nb_bords; n_bord++)
502 {
503 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
504 {
505 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
506 int num1 = le_bord.num_premiere_face();
507 int num2 = num1 + le_bord.nb_faces();
508 if(sub_type(Neumann_sortie_libre,la_cl.valeur()))
509 for (int face2=num1; face2<num2; face2++)
510 {
511 double nu_tot=nu+nu_turb(face_voisins(face2,0));
512 for(indice=0; indice<(nfe-1); indice++)
513 {
514 som=nps+dom.get_renum_som_perio(face_sommets(face2,indice));
515 for (compi=0; compi<dimension; compi++)
516 for (compj=0; compj<dimension; compj++)
517 div(face2,compi)-= nu_tot/dimension*
518 face_normales(face2,compj)*grad(som,compi,compj);
519 }
520 }
521 }
522 }
523// Cerr << "div AP CL : " << div << finl;
524 return div;
525}

529DoubleTab& Op_Dift_VEF_P1NCP1B_Face::ajouter(const DoubleTab& inconnue,
530 DoubleTab& resu) const
531{
532// const Zone_Cl_VEFP1B& zone_Cl_VEF = la_zcl_vef.valeur();
533 const Zone_VEF_PreP1b& zone_VEF = la_zone_vef.valeur();
534 const Zone& zone = zone_VEF.zone();
535 const Domaine& dom=zone.domaine();
536 int nb_elem_tot=zone.nb_elem_tot();
537 int nb_som_tot=dom.nb_som_tot();
538
539 const DoubleTab& nu_turb=diffusivite_turbulente()->valeurs();
540 double nu=(diffusivite_.valeur())(0,0);
541
542 DoubleTab gradient(nb_elem_tot+nb_som_tot, dimension, dimension);
543 gradient = 0.;
544 calculer_gradient_elem(inconnue, gradient);
545 calculer_gradient_som(inconnue, gradient);
546 calculer_divergence_elem(nu, nu_turb, gradient, resu);
547 calculer_divergence_som(nu, nu_turb, gradient, resu);
548 corriger_div_pour_Cl(resu);
549 resu.echange_espace_virtuel();
550 modifier_flux(*this);
551 return resu;
552}
