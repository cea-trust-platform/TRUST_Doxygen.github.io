62Entree& Terme_Source_Canal_RANS_LES_VEF_Face::readOn(Entree& is )
63{
64 int compteur = 0;
65 Motcle mot_lu;
66 Motcle acc_ouverte("{");
67 Motcle acc_fermee("}");
68
69 u_target = 0;
70 f_start = 0.;
71 moyenne = 0;//valeur par defaut
72// 0 => moyenne spatiale
73// 1 => moyenne temporelle glissante (moyenne en alpha)
74// 2 => moyenne temporelle
75// 3 => moyenne temporelle t=0
76
77 Motcles les_mots(11);
78 {
79 les_mots[0]="alpha_tau"; //coeff de relaxation du terme source
80 les_mots[1]="Ly"; //Hauteur du canal plan (utile pour la moyenne spatiale)
81 les_mots[2]="t_moy_start"; //temps de demarrage de la moyenne
82 les_mots[3]="f_start"; //temps a partir duquel le terme source commence a etre active (pondere)
83 les_mots[4]="f_tot"; //temps a partir duquel le terme source est totalement applique
84 les_mots[5]="type_moyenne";//!< 0: moyenne spatiale 1: moyenne temp. glissante 2: moyenne temporelle
85 les_mots[6]="dir"; //!< direction de forcage de l'ecoulement
86 les_mots[7]="u_tau"; //!< vitesse de frottement theorique
87 les_mots[8]="nu"; //!< viscosite moleculaire
88 les_mots[9]="rayon"; //!< rayon du cylindre
89 les_mots[10]="u_target"; //!< cle pour champ target : 1=stockage 2=lecture
90 }
91 is >> mot_lu;
92 if(mot_lu != acc_ouverte)
93 {
94 Cerr << "On attendait { a la place de " << mot_lu
95 << " lors de la lecture des parametres de la loi de paroi " << finl;
96 }
97 is >> mot_lu;
98 while(mot_lu != acc_fermee)
99 {
100 int rang=les_mots.search(mot_lu);
101 switch(rang)
102 {
103 case 0 :
104 is >> alpha_tau;
105 Cerr << "alpha_tau = " << alpha_tau << finl;
106 compteur++;
107 break;
108 case 1 :
109 is >> Ly;
110 Cerr << "Ly = "<< Ly << finl;
111 compteur++;
112 break;
113 case 2 :
114 is >> t_moy_start;
115 Cerr << " t_moy_start = "<< t_moy_start << finl;
116 compteur++;
117 break;
118 case 3 :
119 is >> f_start;
120 Cerr << " f_start = "<< f_start << finl;
121 compteur++;
122 break;
123 case 4 :
124 is >> f_tot;
125 Cerr << " f_tot = "<< f_tot << finl;
126 compteur++;
127 break;
128 case 5 :
129 is >> moyenne;
130 Cerr << "type_moyenne : " << moyenne << finl;
131 if(moyenne==0)
132 {
133 Cerr << " Pertinence de la moyenne spatiale en VEF ? -> ARRET (non code)" << finl;
134 exit();
135 }
136 if(moyenne==1)
137 {
138 Cerr << " YOUNES, YOUNES !!! !" << finl;
139 Cerr << " YOUNES EST PARTI !!!" << finl;
140 exit();
141 }
142 compteur++;
143 break;
144 case 6 :
145 is >> dir_nom;
146 if ( (dir_nom=="-X") || (dir_nom=="+X") ) dir=0;
147 else if( (dir_nom=="-Y") || (dir_nom=="+Y") ) dir=1;
148 else if( (dir_nom=="-Z") || (dir_nom=="+Z") ) dir=2;
149 else
150 {
151 Cerr << " ERROR when reading direction for forcing term " << finl;
152 Cerr << " 'dir' parameter has to be selected among following list : +X, -X, +Y, -Y, +Z or -Z " << finl;
153 exit();
154 }
155 Cerr << " forcage suivant l'axe : " << dir_nom << " -> composante : " << dir <<finl;
156 compteur++;
157 break;
158 case 7 :
159 is >> u_tau;
160 Cerr << "vitesse de frottement : " << u_tau << finl;
161 compteur++;
162 break;
163 case 8 :
164 is >> nu;
165 Cerr << "viscosite moleculaire : " << nu << finl;
166 compteur++;
167 break;
168 case 9 :
169 is >> rayon;
170 Cerr << "rayon : " << rayon << finl;
171 compteur++;
172 break;
173 case 10 :
174 is >> u_target;
175 Cerr << " u_target: " << u_target << finl;
176 compteur++;
177 break;
178 default :
179 {
180 Cerr << mot_lu << " n'est pas un mot compris par Source_Canal_RANS_LES_VEF_P1NC" << finl;
181 Cerr << "Les mots compris sont : " << les_mots << finl;
182 exit();
183 }
184 }
185 is >> mot_lu;
186 }
187 Cerr << compteur << " motcles ont ete lus dans le readOn de Source_Canal_RANS_LES_VEF_P1NC" << finl;
188
189 init();
190
191 return is;
