93Entree& Traitement_particulier_NS_canal::lire(Entree& is)
94{
95 if(!sub_type(Navier_Stokes_std,mon_equation.valeur()))
96 {
97 Cerr << " Traitement_particulier_canal has to be called from a Navier_Stokes problem " << finl;
98 exit();
99 }
100
101 Motcle accouverte = "{" , accfermee = "}" ;
102 Motcle motbidon, motlu;
103 is >> motbidon ;
104 if (motbidon == accouverte)
105 {
106 Motcles les_mots(7);
107 les_mots[0] = "dt_impr_moy_spat";
108 les_mots[1] = "dt_impr_moy_temp";
109 les_mots[2] = "debut_stat";
110 les_mots[3] = "fin_stat";
111 les_mots[4] = "reprise";
112 les_mots[5] = "pulsation_w";
113 les_mots[6] = "nb_points_par_phase";
114
115 is >> motlu;
116 while(motlu != accfermee)
117 {
118 int rang=les_mots.search(motlu);
119 switch(rang)
120 {
121 case 0 :
122 {
123 is >> dt_impr_moy_spat; //!< intervalle de temps de sorties des moyennes spatiales
124 Cerr << "Spatial averages are printed for : dt_impr_moy_spat = " << dt_impr_moy_spat << finl;
125 break;
126 }
127 case 1 :
128 {
129 is >> dt_impr_moy_temp; //!< intervalle de temps de sorties des moyennes temporelles
130 Cerr << "Temporal averages are printed for : dt_impr_moy_temp = " << dt_impr_moy_temp << finl;
131 break;
132 }
133 case 2 :
134 {
135 is >> temps_deb; //!< temps de debut des moyennes temporelles
136 Cerr << "Temporal averages start at : temps_deb = " << temps_deb << finl;
137 break;
138 }
139 case 3 :
140 {
141 is >> temps_fin; //!< temps de debut des moyennes temporelles
142 Cerr << "Temporal averages finish at : temps_fin = " << temps_fin << finl;
143 break;
144 }
145 case 4 :
146 {
147 oui_repr = 1;
148 is >> fich_repr ; //!< indication du nom du fichier de reprise des stats
149 Cerr << "The time statistics file is : " << fich_repr << finl;
150 break;
151 }
152 case 5 :
153 {
154 oui_pulse = 1;
155 is >> w ; //!< pulsation
156 freq = w/(2.*3.141592653); //!< frequence associee
157 break;
158 }
159 case 6 :
160 {
161 oui_pulse = 1;
162 is >> Nphase ; //!< nombre de points pour decrire une phase
163 break;
164 }
165 default :
166 {
167 Cerr << motlu << " is not a keyword for Traitement_particulier_NS_canal " << finl;
168 Cerr << "Since the TRUST v1.5, the syntax of Canal option has changed." << finl;
169 Cerr << "Check the reference manual." << finl;
170 exit();
171 }
172 }
173 is >> motlu;
174 }
175 is >> motlu;
176 if (motlu != accfermee)
177 {
178 Cerr << "Error while reading in Traitement_particulier_NS_canal" << finl;;
179 Cerr << "We were expecting a }" << finl;
180 exit();
181 }
182 }
183 else
184 {
185 Cerr << "Error while reading in Traitement_particulier_NS_canal" << finl;
186 Cerr << "We were expecting a {" << finl;
187 exit();
188 }
189
190 return is;
191}

411void Traitement_particulier_NS_canal::post_traitement_particulier()
412{
413
414// Calcul des Moyennes spatiales
415//
416
417 DoubleTrav val_moy(Ny,Nval);
418
419 val_moy=0.;
420
421 calculer_moyenne_spatiale_vitesse_rho_mu(val_moy);
422
423 if (oui_profil_nu_t != 0)
424 calculer_moyenne_spatiale_nut(val_moy);
425
426 if (oui_profil_Temp != 0)
427 calculer_moyenne_spatiale_Temp(val_moy);
428
429
430// Echange des donnees entre processeurs
431//
432
433 DoubleVect Y_p(Y);
434 DoubleVect compt_p(compt);
435 DoubleTab val_moy_p(val_moy);
436
437 const int NN = Y_tot.size();
438
439 envoyer(Y_p,Process::me(),0,Process::me());
440 envoyer(compt_p,Process::me(),0,Process::me());
441 envoyer(val_moy_p,Process::me(),0,Process::me());
442
443 if(je_suis_maitre())
444 {
445 compt_tot=0;
446 val_moy_tot=0.;
447 int j;
448 for(int p=0; p<Process::nproc(); p++)
449 {
450 recevoir(Y_p,p,0,p);
451 recevoir(compt_p,p,0,p);
452 recevoir(val_moy_p,p,0,p);
453
454 int Y_p_size = Y_p.size();
455 for (int i=0; i<Y_p_size; i++)
456 {
457 for (j=0; j<NN; j++)
458 if(est_egal(Y_tot[j],Y_p[i])) break;
459
460 compt_tot(j) += compt_p(i);
461
462 for (int k=0; k<Nval; k++)
463 val_moy_tot(j,k,0) += val_moy_p(i,k);
464 }
465 }
466
467 for (int i=0; i<NN; i++)
468 for (int k=0; k<Nval; k++)
469 val_moy_tot(i,k,0) /= compt_tot[i];
470 }
471
472
473// Calcul et ecriture des grandeurs parietales
474//
475
476 calcul_reynolds_tau();
477
478// Calcul des Moyennes temporelles
479//
480
481 double tps = mon_equation->inconnue().temps();
482
483 if(je_suis_maitre())
484 {
485 if ((tps>=temps_deb)&&(tps<=temps_fin))
486 {
487 static int init_stat_temps = 0;
488
489 double dt_v = mon_equation->schema_temps().pas_de_temps();
490 int NN2 = Y_tot.size();
491
492 if(init_stat_temps==0 && oui_repr!=1) //!< sinon, les valeurs de val_moy_temp ont ete lues a partir de reprendre_stat()
493 {
494 temps_deb = tps-dt_v;
495
496 val_moy_temp.resize(NN2,Nval,1);
497 val_moy_temp=0.;
498
499 init_stat_temps++;
500 }
501
502 for (int j=0; j<Nval; j++)
503 for (int i=0; i<NN2; i++)
504 {
505 if ((j!=3) && (j!=4) && (j!=5) && (j!=6) && (j!=7) && (j!=8) && (j!=14) && (j!=15) && (j!=16) && (j!=17))
506 val_moy_temp(i,j,0)+= dt_v*val_moy_tot(i,j,0);
507
508 if (j==3)
509 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,3,0)-val_moy_tot(i,0,0)*val_moy_tot(i,0,0));
510 if (j==4)
511 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,4,0)-val_moy_tot(i,1,0)*val_moy_tot(i,1,0));
512 if (j==5)
513 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,5,0)-val_moy_tot(i,2,0)*val_moy_tot(i,2,0));
514
515 if (j==6)
516 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,6,0)-val_moy_tot(i,0,0)*val_moy_tot(i,1,0));
517 if (j==7)
518 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,7,0)-val_moy_tot(i,0,0)*val_moy_tot(i,2,0));
519 if (j==8)
520 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,8,0)-val_moy_tot(i,1,0)*val_moy_tot(i,2,0));
521
522
523 if (j==14)
524// val_moy_temp(i,j,0)+= dt_v*val_moy_tot(i,13,0)*val_moy_tot(i,13,0);
525 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,14,0)-val_moy_tot(i,13,0)*val_moy_tot(i,13,0));
526 if (j==15)
527 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,15,0)-val_moy_tot(i,0,0)*val_moy_tot(i,13,0));
528 if (j==16)
529 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,16,0)-val_moy_tot(i,1,0)*val_moy_tot(i,13,0));
530 if (j==17)
531 val_moy_temp(i,j,0)+= dt_v*(val_moy_tot(i,17,0)-val_moy_tot(i,2,0)*val_moy_tot(i,13,0));
532
533 }
534 }
535 }
536
537// Calcul des Moyennes de phases
538//
539
540 if(je_suis_maitre() && oui_pulse==1)
541 {
542 if ((tps>=temps_deb)&&(tps<=temps_fin))
543 {
544 double dt_v = mon_equation->schema_temps().pas_de_temps();
545
546 if ( (cos(w*tps) > cos(w*(tps+dt_v))) && (cos(w*tps) > cos(w*(tps-dt_v))) ) //!< debut d'une periode
547 {
548 debut_phase=tps;
549 ind_phase=1;
550 }
551
552 if(ind_phase==1)
553 {
554 for(int k=0; k<Nphase; k++)
555 {
556 double tps_k=debut_phase+k/(Nphase*freq);
557
558 if( (tps > tps_k-0.5*dt_v) && (tps < tps_k+0.5*dt_v) ) //!< recherche de la k-phase correspondant
559 {
560 for (int j=0; j<Nval; j++)
561 for (int i=0; i<NN; i++)
562 val_moy_phase(i,j,k) += val_moy_tot(i,j,0);
563
564 Nb_ech_phase(k)++;
565
566 if(k==Nphase-1) ind_phase=2; //!< marqueur pour ecriture de moyennes de phase (voir plus bas)
567 }
568 }
569 }
570 }
571 }
572
573// Ecriture des Moyennes spatiales et temporelles
574//
575
576 if(je_suis_maitre())
577 {
578 double stationnaire_atteint=mon_equation->schema_temps().stationnaire_atteint();
579 double tps_passe=mon_equation->schema_temps().temps_courant();
580 double nb_pas_dt_max=mon_equation->schema_temps().nb_pas_dt_max();
581 double nb_pas_dt=mon_equation->schema_temps().nb_pas_dt();
582 double temps_max=mon_equation->schema_temps().temps_max();
583 int impr_inst;
584 if (dt_impr_moy_spat<=(tps-tps_passe))
585 impr_inst=1;
586 else
587 {
588// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
589 double i, j, epsilon = 1.e-8;
590 modf(tps/dt_impr_moy_spat + epsilon, &i);
591 modf(tps_passe/dt_impr_moy_spat + epsilon, &j);
592 impr_inst=(i>j);
593 }
594 int impr_stat;
595 if (dt_impr_moy_temp<=(tps-tps_passe))
596 impr_stat=1;
597 else
598 {
599// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
600 double i, j, epsilon = 1.e-8;
601 modf(tps/dt_impr_moy_temp + epsilon, &i);
602 modf(tps_passe/dt_impr_moy_temp + epsilon, &j);
603 impr_stat=(i>j);
604 }
605
606// sauvegarde periodique des moyennes spatiales
607
608 if ((nb_pas_dt+1<=1) || impr_inst || (temps_max <= tps) || (nb_pas_dt_max <= nb_pas_dt+1) || stationnaire_atteint)
609 {
610 double dt = 1.;
611
612 Nom fichier1 = "Moyennes_spatiales_vitesse_rho_mu_";
613 Nom fichier2 = "Moyennes_spatiales_nut_";
614 Nom fichier3 = "Moyennes_spatiales_Temp_";
615 Nom temps = Nom(tps);
616
617 fichier1 += temps;
618 fichier2 += temps;
619 fichier3 += temps;
620
