89//! Impose les conditions aux limites a la valeur temporelle "temps" du<br>Champ_Inc
90void Zone_Cl_PolyMAC::imposer_cond_lim(Champ_Inc& ch, double temps)
91{
92
93 Champ_Inc_base& ch_base=ch.valeur();
94 DoubleTab& ch_tab = ch_base.valeurs(temps);
95 int n, N = ch_tab.line_size();
96 if (sub_type(Champ_Inc_P0_base,ch_base))
97 ;
98 else if(ch_base.nature_du_champ()==scalaire)
99 ;
100 else if (sub_type(Champ_Face_PolyMAC,ch_base))
101 {
102 Champ_Face_PolyMAC& ch_face = ref_cast(Champ_Face_PolyMAC, ch_base);
103 const Zone_VF& ma_zone_VF = ch_face.zone_vf();
104 int ndeb,nfin, num_face;
105
106 for(int i=0; i<nb_cond_lim(); i++)
107 {
108 const Cond_lim_base& la_cl = les_conditions_limites(i).valeur();
109 if (sub_type(Periodique,la_cl)) abort(); //nope
110 else if( sub_type(Symetrie,la_cl) )
111 {
112 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
113 ndeb = le_bord.num_premiere_face();
114 nfin = ndeb + le_bord.nb_faces();
115 for (num_face=ndeb; num_face<nfin; num_face++)
116 for (n = 0; n < N; n++)
117 ch_tab(num_face, n) = 0;
118 }
119 else if ( sub_type(Dirichlet_entree_fluide,la_cl) )
120 {
121 const Dirichlet_entree_fluide& la_cl_diri = ref_cast(Dirichlet_entree_fluide,la_cl);
122 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
123 ndeb = le_bord.num_premiere_face();
124 nfin = ndeb + le_bord.nb_faces();
125
126 for (num_face=ndeb; num_face<nfin; num_face++)
127 for (n = 0; n < N; n++)
128 {
129// WEC : optimisable (pour chaque face recherche le bon temps !)
130// vn
131 double vn=0;
132 for (int d=0; d<dimension; d++)
133 vn+=ma_zone_VF.face_normales(num_face,d)*la_cl_diri.val_imp_au_temps(temps,num_face-ndeb, N * d + n);
134 vn/=ma_zone_VF.face_surfaces(num_face);
135 ch_tab(num_face, n) = vn;
136 }
137 }
138 else if ( sub_type(Dirichlet_paroi_fixe,la_cl) )
139 {
140 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
141 ndeb = le_bord.num_premiere_face();
142 nfin = ndeb + le_bord.nb_faces();
143 for (num_face=ndeb; num_face<nfin; num_face++)
144 for (n = 0; n < N; n++)
145 ch_tab(num_face, n) = 0;
146 }
147 else if ( sub_type(Dirichlet_paroi_defilante,la_cl) )
148 {
149 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
150 ndeb = le_bord.num_premiere_face();
151 nfin = ndeb + le_bord.nb_faces();
152 for (num_face=ndeb; num_face<nfin; num_face++)
153 for (n = 0; n < N; n++)
154 ch_tab(num_face, n) = 0;
155 }
156 }
157 modif_perio_fait_ = 1;
158 }
159 else
160 {
161 Cerr << "Le type de Champ_Inc " << ch->que_suis_je() << " n'est pas prevu en PolyMAC\n";
162 exit();
163 }
164 ch_tab.echange_espace_virtuel();
165 Debog::verifier("Zone_Cl_PolyMAC::imposer_cond_lim ch_tab",ch_tab);
166}
