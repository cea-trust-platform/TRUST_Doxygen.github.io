81inline double Op_Dift_Stab_VEF_Face::aij_extradiag(const int elem, const int facei, const int facej, const int dim, const int dim2, const double nu_elem) const
82{
83 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
84
85 const IntTab& face_voisins=zone_VEF.face_voisins();
86
87 const DoubleTab& face_normales=zone_VEF.face_normales();
88 const DoubleVect& volumes=zone_VEF.volumes();
89
90 double volume=0.;
91 double signei=1.;
92 double signej=1.;
93 double aij=0.;
94
95// Debut de l'algo
96 assert(dim<equation().inconnue().valeurs().dimension(1));
97 assert(dim2<equation().inconnue().valeurs().dimension(1));
98 assert(dim<dim2);
99
100 volume=1./volumes(elem);
101 volume*=nu_elem;
102
103 if (face_voisins(facei,0)!=elem) signei=-1.;
104 if (face_voisins(facej,0)!=elem) signej=-1.;
105
106 aij=face_normales(facei,dim2)*face_normales(facej,dim);
107 aij*=signei*signej;
108 aij*=volume;
109
110 return aij;
111}
