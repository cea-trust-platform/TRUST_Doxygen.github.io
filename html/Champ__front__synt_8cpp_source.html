91Entree& Champ_front_synt::readOn(Entree& is)
92{
93 timeScale = 0.;
94 lenghtScale = 0.;
95 nbModes = 0;
96 turbKinEn = 0.;
97 turbDissRate = 0.;
98 KeOverKmin = 0.;
99 ratioCutoffWavenumber = 0;
100
101 int dim;
102 dim=lire_dimension(is,que_suis_je());
103 if( dim != 3)
104 {
105 Cerr << "Error the dimension must be equal to 3" << finl;
106 exit();
107 }
108 Motcle motlu;
109 Motcles les_mots(9);
110 les_mots[0]="moyenne";
111 les_mots[1]="lenghtScale";
112 les_mots[2]="nbModes";
113 les_mots[3]="turbKinEn";
114 les_mots[4]="turbDissRate";
115 les_mots[5]="KeOverKmin";
116 les_mots[6]="timeScale";
117 les_mots[7]="ratioCutoffWavenumber";
118 les_mots[8]="dir_fluct";
119
120 is >> motlu;
121 if (motlu != "{")
122 {
123 Cerr << "Error while reading Champ_front_synt" << finl;
124 Cerr << "We expected a { instead of " << motlu << finl;
125 exit();
126 }
127 int cpt = 0;
128 is >> motlu;
129 while (motlu!="}")
130 {
131 int rang=les_mots.search(motlu);
132 switch(rang)
133 {
134 case 0:
135 {
136 cpt++;
137 moyenne.resize(dim);
138 fixer_nb_comp(dim);
139 for(int i=0; i<dim; i++)
140 is >> moyenne(i);
141 break;
142 }
143 case 1:
144 {
145 cpt++;
146 is >> lenghtScale;
147 break;
148 }
149 case 2:
150 {
151 cpt++;
152 is >> nbModes;
153 break;
154 }
155 case 3:
156 {
157 cpt++;
158 is >> turbKinEn;
159 break;
160 }
161 case 4:
162 {
163 cpt++;
164 is >> turbDissRate;
165 break;
166 }
167 case 5:
168 {
169 cpt++;
170 is >> KeOverKmin;
171 break;
172 }
173 case 6:
174 {
175 cpt++;
176 is >> timeScale;
177 break;
178 }
179 case 7:
180 {
181 cpt++;
182 is >> ratioCutoffWavenumber;
183 break;
184 }
185 case 8:
186 {
187 cpt++;
188 dir_fluct.resize(dim);
189 fixer_nb_comp(dim);
190 for(int i=0; i<dim; i++)
191 {
192 is >> dir_fluct(i);
193 }
194 break;
195 }
196 default :
197 {
198 if (motlu == "p")
199 {
200 Cerr << "Error while reading Champ_front_synt:" << finl;
201 Cerr << " Parameter " << motlu << " has been renamed to KeOverKmin since TRUST v1.9.0"<< finl;
202 Cerr << " Update your datafile."<< finl;
203 }
204 else
205 {
206 Cerr << "Error while reading Champ_front_synt:" << finl;
207 Cerr << " " << motlu << "is not understood."<< finl;
208 Cerr << " We are expecting a parameter among " << les_mots << finl;
209 }
210 exit();
211 }
212 }
213 is >> motlu;
214 }
215 if(cpt != 9)
216 {
217 Cerr << "Error while reading Champ_front_synt: wrong number of parameters" << finl;
218 Cerr << "You should specify all these parameters: " << les_mots << finl;
219 exit();
220 }
221 if( lenghtScale == 0 || nbModes == 0 || turbKinEn == 0 || turbDissRate == 0 || KeOverKmin == 0 || timeScale == 0 || ratioCutoffWavenumber == 0 )
222 {
223 Cerr << "Error while reading Champ_front_synt" << finl;
224 Cerr << "There is at least one parameter among: timeScale, lenghtScale, nbModes, turbKinEn, turbDissRate and ratioCutoffWavenumber set to 0" << finl;
225 exit();
226 }
227
228 for(int i=0; i<dim; i++)
229 {
230 Cerr << dir_fluct(i) << " ";
231 }
232 Cerr << finl;
233
234
235 return is;
236}

321 double kappa_eta = pow((turbDissRate/(visc*visc*visc)),0.25); //!< nombre d'onde de Kolmogorov

362 ///!< creation vecteur onde en coordonnee cartesienne ///
363 kappa_x(i) = sin(tetha(i))*cos(phi(i));
364 kappa_y(i) = sin(tetha(i))*sin(phi(i));
365 kappa_z(i) = cos(tetha(i));

367 ///!< creation de la direction orthogonal au vecteur onde ///
368 sigma_x(i) = cos(phi(i))*cos(tetha(i))*cos(alpha(i)) - sin(phi(i))*sin(alpha(i));

396 double y_center = centreGrav(i,1);

399 for(int m = 0; m<nbModes; m++)
400 {
401 double kx = kappa_x(m) * kappa_center(m);
402 double ky = kappa_y(m) * kappa_center(m);
403 double kz = kappa_z(m) * kappa_center(m);
404
405 double arg = kx*x_center + ky*y_center + kz*z_center + psi(m);
406
407 double tfunk = cos(arg);
408
409 double karman_spectrum = amp/kappa_e * (2.*turbKinEn/3.) * pow((kappa_center(m)/kappa_e),4)/pow(1+pow(kappa_center(m)/kappa_e, 2),17.0/6.0) * exp(-2*(pow(kappa_center(m)/kappa_eta, 2)));

411
412 turb(0) += 2*amplitude*tfunk*sigma_x(m);
413 turb(1) += 2*amplitude*tfunk*sigma_y(m);
414 turb(2) += 2*amplitude*tfunk*sigma_z(m);

416// Cerr << "Kappa = " << kappa_center(m) << finl;
417// Cerr << "Kappa, Spectre = " << kappa_center(m) << " , " << karman_spectrum << finl;
418// Cerr << "Amplitude = " << amplitude << finl;

426 double a = exp(-dt/timeScale);
427 double b = sqrt(1-a*a);
428
429 tab(i,0) = moyenne(0) + dir_fluct(0) * (a * (tab_avant(i,0) - moyenne(0)) + b * turb(0));

434// Cerr << "Uprime, uprime = " << tab(i,0) << " , " << turb(0) << finl;
435// Cerr << "face, Uprime, uprime = " << i << " , " << tab(i,0) << " , " << turb(0) << finl;
