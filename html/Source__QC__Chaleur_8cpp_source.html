56//! Ajoute les termes sources
57/** \return Entree\&<br>
58Signification:<br>
59Contraintes: */
60DoubleTab& Source_QC_Chaleur::ajouter_(DoubleTab& resu) const
61{
62 double dt_= mon_equation->schema_temps().temps_courant() - mon_equation->schema_temps().temps_precedent();
63
64 if (dt_<=0) return resu; //!< On calcul pas ce terme source si dt<=0
65
66 /*
67 * The source term corresponds to :
68 * d P_tot / d t = del P / del t + u.grad(P_tot)
69 * Here grad(P_tot) = 0 (uniform in space)
70 */
71
72 int i, nsom = resu.dimension(0);
73 const Fluide_Quasi_Compressible& FQC = ref_cast(Fluide_Quasi_Compressible,le_fluide.valeur());
74 double Pth = FQC.pression_th(), Pthn = FQC.pression_thn();
75 double dpth = ( Pth - Pthn ) / dt_;
76 for (i=0 ; i<nsom ; i++) resu(i) += dpth * volumes(i) * porosites(i);
77
78 return resu;
79}
