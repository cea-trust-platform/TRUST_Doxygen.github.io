52// Description
53// By default initializes to zero.
54int Ch_front_input_uniforme::initialiser(double temps, const Champ_Inc_base& inco)
55{
56 if (!Champ_front_instationnaire_base::initialiser(temps,inco))
57 return 0;
58 for(int i=0; i<les_valeurs->nb_cases(); i++)
59 {
60 DoubleTab& val=les_valeurs[i].valeurs();
61 int nbcomp=val.dimension(1);
62 if (initial_value_.size_array()==0)
63 val=0;
64 else
65 for (int c=0; c<nbcomp; c++)
66 val(0,c)=initial_value_[c];
67 }
68 return 1;
69}
