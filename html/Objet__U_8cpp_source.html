37// Fonction utilitaire pour intercepter la creation d'un objet
38// ATTENTION: depuis gcc3, le compilo genere plusieurs routines pour
39// chaque constructeur: au moins Objet_U::Objet_U() et Objet_U::Objet_U$base()
40// Cela rend difficile l'utilisatino de breakpoints dans ces routines.

51//! Constructeur par defaut : attribue un numero d'identifiant unique<br>a l'objet (object_id_), et enregistre l'objet en "memoire"<br>en lui donnant un numero _num_obj_. L'object_id_ et tres<br>pratique pour debugger...
52Objet_U::Objet_U() : Process(),
53 object_id_(static_obj_counter_++)
54{
55 int obj_id = object_id_;
56 if (obj_id == object_id_to_track)
57 {
58 object_trap(obj_id);
59 }
60 Memoire& memoire = Memoire::Instance();
61 _num_obj_ = memoire.add(this);
62}
