43int Op_Grad_VDF_Face::impr(Sortie& os) const
44{
45 const int impr_mom=la_zone_vdf->zone().Moments_a_imprimer();
46 const int impr_sum=(la_zone_vdf->zone().Bords_a_imprimer_sum().est_vide() ? 0:1);
47 const int impr_bord=(la_zone_vdf->zone().Bords_a_imprimer().est_vide() ? 0:1);
48 const Schema_Temps_base& sch = equation().probleme().schema_temps();
49 const Zone_VDF& zvdf = la_zone_vdf.valeur();
50 const Zone_Cl_VDF& zclvdf = la_zcl_vdf.valeur();
51 const DoubleVect& face_surfaces = zvdf.face_surfaces();
52 const Equation_base& eqn = equation();
53 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,eqn);
54 const Champ_P0_VDF& la_pression_P0 = ref_cast(Champ_P0_VDF,eqn_hydr.pression_pa().valeur());
55 const DoubleTab& pression_P0 = la_pression_P0.valeurs();
56 int elem0, elem1, face, ori;
57 double n0;
58 const int nb_faces = zvdf.nb_faces_tot();
59 DoubleTab xgr(nb_faces,dimension);
60 xgr=0.;
61 if (impr_mom)
62 {
63 const DoubleTab& xgrav = zvdf.xv();
64 const ArrOfDouble& c_grav=zvdf.zone().cg_moments();
65 for (int num_face=0; num_face <nb_faces; num_face++)
66 for (int i=0; i<dimension; i++) xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
67 }
68
69 flux_bords_.resize(zvdf.nb_faces_bord(),dimension);
70 flux_bords_ = 0.;
71// flux_bords contains the sum of flux on each boundary:
72 DoubleTrav tab_flux_bords(3,zvdf.nb_front_Cl(),3);
73 tab_flux_bords=0.;
74 /* flux_bord(k) -> flux_bords2(0,num_cl,k)
75 flux_bord_perio1(k) -> flux_bords2(1,num_cl,k)
76 flux_bord_perio2(k) -> flux_bords2(2,num_cl,k)
77 moment(k) -> flux_bords2(3,num_cl,k) */
78 int nb_bord = zvdf.nb_front_Cl();
79 for (int n_bord=0; n_bord<nb_bord; n_bord++)
80 {
81 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
82 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
83 int impr_boundary = (zvdf.zone().Bords_a_imprimer_sum().contient(le_bord.le_nom()) ? 1 : 0);
84 int ndeb = le_bord.num_premiere_face();
85 int nfin = ndeb + le_bord.nb_faces();
86
87 for (face=ndeb; face<nfin; face++)
88 {
89 elem0 = face_voisins(face,0);
90 ori = orientation(face);
91 n0 = face_surfaces(face)*porosite_surf(face);
92 if (ori == 0)
93 {
94 if (elem0 != -1) flux_bords_(face,0) = (pression_P0(elem0))*n0 ;
95 else
96 {
97 elem1 = face_voisins(face,1);
98 flux_bords_(face,0) = -(pression_P0(elem1))*n0 ;
99 }
100 tab_flux_bords(0, n_bord, 0) += flux_bords_(face,0) ;
101 }
102 else if (ori == 1)
103 {
104 if (elem0 != -1) flux_bords_(face,1) = (pression_P0(elem0))*n0 ;
105 else
106 {
107 elem1 = face_voisins(face,1);
108 flux_bords_(face,1) = -(pression_P0(elem1))*n0 ;
109 }
110 tab_flux_bords(0, n_bord, 1) += flux_bords_(face,1) ;
111 }
112 else if (ori == 2)
113 {
114 if (elem0 != -1) flux_bords_(face,2) = (pression_P0(elem0))*n0 ;
115 else
116 {
117 elem1 = face_voisins(face,1);
118 flux_bords_(face,2) = -(pression_P0(elem1))*n0 ;
119 }
120 tab_flux_bords(0, n_bord, 2) += flux_bords_(face,2) ;
121 }
122
123 if (dimension == 2)
124 {
125 if (impr_mom)
126 tab_flux_bords(2, n_bord, 2) +=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
127 if (impr_boundary)
128 {
129 tab_flux_bords(1, n_bord, 0) += flux_bords_(face,0) ;
130 tab_flux_bords(1, n_bord, 1) += flux_bords_(face,1) ;
131 }
132 }
133 else if (dimension == 3)
134 {
135 if (impr_mom)
136 {
137 tab_flux_bords(2, n_bord, 0) +=flux_bords_(face,2)*xgr(face,1)-flux_bords_(face,1)*xgr(face,2);
138 tab_flux_bords(2, n_bord, 1) +=flux_bords_(face,0)*xgr(face,2)-flux_bords_(face,2)*xgr(face,0);
139 tab_flux_bords(2, n_bord, 2) +=flux_bords_(face,1)*xgr(face,0)-flux_bords_(face,0)*xgr(face,1);
140 }
141 if (impr_boundary)
142 {
143 tab_flux_bords(1, n_bord, 0) += flux_bords_(face,0) ;
144 tab_flux_bords(1, n_bord, 1) += flux_bords_(face,1) ;
145 tab_flux_bords(1, n_bord, 2) += flux_bords_(face,2) ;
146 }
147 }
148 } //!< fin for face
149 } //!< fin for n_bord
150
151// Sum on all process:
152 mp_sum_for_each_item(tab_flux_bords);
153
154// Write the boundary fluxes:
155 if (je_suis_maitre())
156 {
157// SFichier Flux_grad;
158 if (!Flux_grad.is_open()) ouvrir_fichier(Flux_grad,"",1);
159// SFichier Flux_grad_moment;
160 if (!Flux_grad_moment.is_open()) ouvrir_fichier(Flux_grad_moment,"moment",impr_mom);
161// SFichier Flux_grad_sum;
162 if (!Flux_grad_sum.is_open()) ouvrir_fichier(Flux_grad_sum,"sum",impr_sum);
163 Flux_grad.add_col(sch.temps_courant());
164 if (impr_mom) Flux_grad_moment.add_col(sch.temps_courant());
165 if (impr_sum) Flux_grad_sum.add_col(sch.temps_courant());
166 for (int n_bord=0; n_bord<nb_bord; n_bord++)
167 {
168 if (dimension == 2)
169 {
170 Flux_grad.add_col(tab_flux_bords(0, n_bord, 0));
171 Flux_grad.add_col(tab_flux_bords(0, n_bord, 1));
172 if (impr_sum)
173 {
174 Flux_grad_sum.add_col(tab_flux_bords(1, n_bord, 0));
175 Flux_grad_sum.add_col(tab_flux_bords(1, n_bord, 1));
176 }
177 if (impr_mom) Flux_grad_moment.add_col(tab_flux_bords(2, n_bord, 2));
178 }
179
180 if (dimension == 3)
181 {
182 Flux_grad.add_col(tab_flux_bords(0, n_bord, 0));
183 Flux_grad.add_col(tab_flux_bords(0, n_bord, 1));
184 Flux_grad.add_col(tab_flux_bords(0, n_bord, 2));
185 if (impr_sum)
186 {
187 Flux_grad_sum.add_col(tab_flux_bords(1, n_bord, 0));
188 Flux_grad_sum.add_col(tab_flux_bords(1, n_bord, 1));
189 Flux_grad_sum.add_col(tab_flux_bords(1, n_bord, 2));
190 }
191 if (impr_mom)
192 {
193 Flux_grad_moment.add_col(tab_flux_bords(2, n_bord, 0));
194 Flux_grad_moment.add_col(tab_flux_bords(2, n_bord, 1));
195 Flux_grad_moment.add_col(tab_flux_bords(2, n_bord, 2));
196 }
197 }
198 }
199 Flux_grad << finl;
200 if (impr_sum) Flux_grad_sum << finl;
201 if (impr_mom) Flux_grad_moment << finl;
202 }
203
204 const LIST(Nom)& Liste_Bords_a_imprimer = zvdf.zone().Bords_a_imprimer();
205 if (!Liste_Bords_a_imprimer.est_vide())
206 {
207 EcrFicPartage Flux_grad_face;
208 ouvrir_fichier_partage(Flux_grad_face,"",impr_bord);
209 for (int n_bord=0; n_bord<nb_bord; n_bord++)
210 {
211 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
212 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
213 int ndeb = le_bord.num_premiere_face();
214 int nfin = ndeb + le_bord.nb_faces();
215 if (zvdf.zone().Bords_a_imprimer().contient(le_bord.le_nom()))
216 {
217 if (je_suis_maitre())
218 {
219 Flux_grad_face << "# Force par face sur " << le_bord.le_nom() << " au temps ";
220 sch.imprimer_temps_courant(Flux_grad_face);
221 Flux_grad_face << " : " << finl;
222 }
223 for (face=ndeb; face<nfin; face++)
224 {
225 Flux_grad_face << "# Face a x= " << zvdf.xv(face,0) << " y= " << zvdf.xv(face,1);
226 if (dimension==3) Flux_grad_face << " z= " << zvdf.xv(face,2);
227 Flux_grad_face << " : Fx= " << flux_bords_(face, 0) << " Fy= " << flux_bords_(face, 1);
228 if (dimension==3) Flux_grad_face << " Fz= " << flux_bords_(face, 2);
229 Flux_grad_face << finl;
230 }
231 Flux_grad_face.syncfile();
232 }
233 }
234 }
235 return 1;
236}

238void Op_Grad_VDF_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
239{
240 Matrice_Morse *mat_p = matrices.count("pression") ? matrices.at("pression") : NULL, mat2_p;
241
242 const Zone_VDF& zvdf = la_zone_vdf.valeur();
243 IntTab stencil_p(0, 2);
244 stencil_p.set_smart_resize(1);
245
246 int e;
247 for (int f = 0; f < zvdf.nb_faces(); f++)
248 for (int i = 0; i < 2; i++)
249 if ((e = zvdf.face_voisins(f, i)) >= 0) stencil_p.append_line(f, e);
250 tableau_trier_retirer_doublons(stencil_p);
251 if (mat_p)
252 {
253 Matrix_tools::allocate_morse_matrix(zvdf.nb_faces_tot(), zvdf.nb_elem_tot(), stencil_p, mat2_p);
254 mat_p->nb_colonnes() ? *mat_p += mat2_p : *mat_p = mat2_p;
255 }

260void Op_Grad_VDF_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
261{
262 statistiques().begin_count(gradient_counter_);
263 Matrice_Morse *mat = matrices.count("pression") ? matrices.at("pression") : NULL;
264 const DoubleTab& inco = semi_impl.count("pression") ? semi_impl.at("pression") : (le_champ_inco.non_nul() ? le_champ_inco->valeurs() : ref_cast(Navier_Stokes_std, equation()).pression().valeurs());
265
266 assert_espace_virtuel_vect(inco);
267
268 const Zone_VDF& zvdf = la_zone_vdf.valeur();
269 const Zone_Cl_VDF& zclvdf = la_zcl_vdf.valeur();
270 const DoubleVect& face_surfaces = zvdf.face_surfaces();
271
272 double coef;
273 int n0, n1;
274
275// Boucle sur les bords pour traiter les conditions aux limites
276 for (int n_bord = 0; n_bord < zvdf.nb_front_Cl(); n_bord++)
277 {
278 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
279 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
280 const int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
281
282 if ( sub_type(Neumann_sortie_libre,la_cl.valeur()) )
283 {
284 const Neumann_sortie_libre& la_cl_typee = ref_cast(Neumann_sortie_libre, la_cl.valeur());
285 for (int num_face = ndeb; num_face < nfin; num_face++)
286 {
287 const double P_imp = la_cl_typee.flux_impose(num_face-ndeb);
288
289 n0 = face_voisins(num_face,0);
290 if (n0 != -1)
291 {
292 coef = face_surfaces(num_face)*porosite_surf(num_face) * Option_VDF::coeff_P_neumann;
293 if(mat) (*mat)(num_face, n0) += coef;
294 secmem(num_face) += (coef*(P_imp - inco(n0)));
295 }
296 else
297 {
298 n1 = face_voisins(num_face,1);
299 coef = face_surfaces(num_face)*porosite_surf(num_face) * Option_VDF::coeff_P_neumann;
300 if(mat) (*mat)(num_face, n1) -= coef;
301 secmem(num_face) += (coef*(inco(n1) - P_imp));
302 }
303 }
304 }
305 else if (sub_type(Periodique,la_cl.valeur())) //!< Correction periodicite
306 {
307 for (int num_face = ndeb; num_face < nfin; num_face++)
308 {
309 n0 = face_voisins(num_face,0), n1 = face_voisins(num_face,1);
310 coef = face_surfaces(num_face)*porosite_surf(num_face);
311 secmem(num_face) += (coef*(inco(n1) - inco(n0)));
312 }
313 }
314 else if (sub_type(Symetrie,la_cl.valeur())) { /* Do nothing */ }
315 else if ( (sub_type(Dirichlet,la_cl.valeur())) || (sub_type(Dirichlet_homogene,la_cl.valeur())) ) { /* Do nothing */ }
316 }
317
318// Boucle sur les faces internes
319 for (int num_face = zvdf.premiere_face_int(); num_face < zvdf.nb_faces(); num_face++)
320 {
321 n0 = face_voisins(num_face,0), n1 = face_voisins(num_face,1);
322 coef = face_surfaces(num_face)*porosite_surf(num_face);
323 if(mat) (*mat)(num_face, n0) += coef;
324 if(mat) (*mat)(num_face, n1) -= coef;
325 secmem(num_face) += coef*(inco(n1)-inco(n0));
326
327 }
328 secmem.echange_espace_virtuel();
329 statistiques().end_count(gradient_counter_);
330
331}
