44void Op_Evanescence_Homogene_PolyMAC_Elem::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
45{
46 const Champ_Elem_PolyMAC& ch = ref_cast(Champ_Elem_PolyMAC, equation().inconnue().valeur());
47 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, equation().zone_dis().valeur());
48 const DoubleTab& inco = ch.valeurs();
49
50 /* on doit pouvoir ajouter / soustraire les equations entre composantes */
51 int i, j, e, n, N = inco.line_size();
52 if (N == 1) return; //pas d'evanescence en simple phase!
53 for (auto &&n_m : matrices)
54 if (n_m.second->nb_colonnes())
55 {
56 Matrice_Morse& mat = *n_m.second, mat2;
57 IntTrav sten(0, 2);
58 sten.set_smart_resize(1);
59
60 std::set<int> idx;
61 for (e = 0; e < zone.nb_elem(); e++, idx.clear())
62 {
63 for (i = N * e, n = 0; n < N; n++, i++)
64 for (j = mat.get_tab1()(i) - 1; j < mat.get_tab1()(i + 1) - 1; j++)
65 idx.insert(mat.get_tab2()(j) - 1);
66 for (i = N * e, n = 0; n < N; n++, i++)
67 for (auto &&c : idx) sten.append_line(i, c);
68 }
69 Matrix_tools::allocate_morse_matrix(mat.nb_lignes(), mat.nb_colonnes(), sten, mat2);
70 mat = mat2; //pour forcer l'ordre des coefficients dans la matrice (accelere les operations ligne a ligne)
71 }
72}

74void Op_Evanescence_Homogene_PolyMAC_Elem::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
75{
76 const Milieu_composite& milc = ref_cast(Milieu_composite, equation().milieu());
77 const Champ_Elem_PolyMAC& ch = ref_cast(Champ_Elem_PolyMAC, equation().inconnue().valeur());
78 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, equation().zone_dis().valeur());
79 const Pb_Multiphase& pb = ref_cast(Pb_Multiphase, equation().probleme());
80 const DoubleTab& inco = ch.valeurs(), &alpha = pb.eq_masse.inconnue().valeurs(),
81 &rho = equation().milieu().masse_volumique().valeurs(),
82 &p = pb.eq_qdm.pression().valeurs();
83 const SETS *sch = sub_type(Schema_Implicite_base, pb.eq_qdm.schema_temps()) && sub_type(SETS, ref_cast(Schema_Implicite_base, pb.eq_qdm.schema_temps()).solveur().valeur())
84 ? &ref_cast(SETS, ref_cast(Schema_Implicite_base, pb.eq_qdm.schema_temps()).solveur().valeur()) : NULL;
85
86 int e, i, j, k, n, N = inco.line_size(), m, M = p.line_size(), is_m = ch.le_nom() == "alpha", cR = (rho.dimension_tot(0) == 1),
87 iter = sch ? sch->iteration : 0, p_degen = is_m && sch ? sch->p_degen : 0;
88 if (N == 1 || p_degen || (is_m && !iter)) return; //pas d'evanescence en simple phase ou si p est degenere
89
90 double a_eps = alpha_res_, a_eps_min = alpha_res_min_, a_m, a_max; //seuil de declenchement du traitement de l'evanescence
91
92 /* recherche de phases evanescentes et traitement des seconds membres */
93 IntTrav maj(inco.dimension_tot(0)); //maj(i) : phase majoritaire de la ligne i
94 DoubleTrav coeff(inco.dimension_tot(0), inco.line_size(), 2); //coeff(i, n, 0/1) : coeff a appliquer a l'equation existante / a l'eq. "inco = v_maj", leurs derivees en alpha
95 Matrice_Morse& mat_diag = *matrices.at(ch.le_nom().getString());
96 for (e = 0; e < zone.nb_elem(); e++)
97 {
98 /* phase majoritaire */
99 for (a_max = 0, k = -1, n = 0; n < N; n++)
100 if ((a_m = alpha(e, n)) > a_max) k = n, a_max = a_m;
101 if (k >= 0) maj(e) = k;
102 else abort();
103
104 /* coeff d'evanescence, second membre */
105 for (n = 0, m = 0; n < N; n++, m += (M > 1))
106 if (n != k && (a_m = alpha(e, n)) < a_eps)
107 {
108 double val = is_m ? 0 : milc.has_saturation(n, k) ? milc.get_saturation(n, k).Tsat(p(e, m)) : inco(e, k); //valeur a laquelle on veut ramener inco(e, n)
109 coeff(e, n, 0) = (a_eps == a_eps_min ? (a_m < a_eps) : std::min(std::max((a_eps - a_m) / (a_eps - a_eps_min), 0.), 1.));
110 coeff(e, n, 1) = mat_diag(N * e + k, N * e + k) * coeff(e, n, 0);
111 double flux = coeff(e, n, 0) * secmem(e, n) + coeff(e, n, 1) * (inco(e, n) - val);
112 secmem(e, k) += (p_degen ? rho(!cR * e, k) : 1) * flux, secmem(e, n) -= (p_degen ? rho(!cR * e, n) : 1) * flux;
113 }
114 }
115
116 /* lignes de matrices */
117 for (auto &&n_m : matrices)
118 if (n_m.second->nb_colonnes())
119 {
120 int diag = (n_m.first == ch.le_nom().getString()), press = (n_m.first == "pression"); //est-on sur le bloc diagonal, sur le bloc pression?
121 Matrice_Morse& mat = *n_m.second;
122 for (e = 0; e < zone.nb_elem(); e++)
123 for (n = 0, m = 0; n < N; n++, m += (M > 1))
124 if (coeff(e, n, 0))
125 {
126 k = maj(e); //phase majoritaire
127 double dval = is_m ? 0 : milc.has_saturation(n, k) ? (press ? milc.get_saturation(n, k).dP_Tsat(p(e, m)) : 0) : diag; //derivee de val (nulle si on n'est pas sur le bon bloc)
128 int cval = is_m ? -1 : milc.has_saturation(n, k) ? M * e + m : N * e + k; //indice de colonne associe a cette derivee
129 for (i = mat.get_tab1()(N * e + n) - 1, j = mat.get_tab1()(N * e + k) - 1; i < mat.get_tab1()(N * e + n + 1) - 1; i++, j++)
130 {
131 assert(mat.get_tab2()(j) == mat.get_tab2()(i));
132 int c = mat.get_tab2()(i) - 1; //indice de colonne (commun aux deux lignes grace au dimensionner_blocs())
133 double dflux = -coeff(e, n, 0) * mat.get_set_coeff()(i) + coeff(e, n, 1) * (diag * (c == N * e + n) - dval * (c == cval)); //derivee de flux en secmem, inco, val
134 mat.get_set_coeff()(i) += (p_degen ? rho(!cR * e, n) : 1) * dflux;
135 mat.get_set_coeff()(j) -= (p_degen ? rho(!cR * e, k) : 1) * dflux;
136 }
137 }
138 }
139 if (!is_m && equation().schema_temps().temps_courant() > 0.68)
140 e++;
141}
