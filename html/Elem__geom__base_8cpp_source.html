135//! Cree les faces de l'element geometrique specifie de la zone<br>en precisant le type de face a creer.
136void Elem_geom_base::creer_faces_elem(Faces& les_faces ,
137 int num_elem,
138 Type_Face type) const
139{
140 int type_id=0;
141 for(; ((type_id<nb_type_face())
142 &&(type!=type_face(type_id))); type_id++)
143 ;
144 const IntTab& les_Polys = ma_zone->les_elems();
145 assert(les_Polys.dimension_tot(0) > num_elem);
146 int face, i;
147 les_faces.dimensionner(nb_faces(type_id));
148 les_faces.associer_zone(ma_zone.valeur());
149
150 for(face=0; face<nb_faces(type_id); face++)
151 {
152 int face_id=num_face(face, type);
153 for (i=0; i<nb_som_face(type_id); i++)
154 les_faces.sommet(face_id,i) = les_Polys(num_elem,face_sommet(face_id, i));
155 les_faces.completer(face_id, num_elem);
156 }
157}

175//! Calcule les centres de gravites de tous les elements<br>de la zone associee a l'element goemetrique.
176void Elem_geom_base::calculer_centres_gravite(DoubleTab& xp) const
177{
178 const IntTab& les_Polys = ma_zone->les_elems();
179 const Domaine& le_domaine = ma_zone->domaine();
180 int nb_elem;
181 if(xp.dimension(0)==0)
182 {
183 nb_elem = ma_zone->nb_elem_tot();
184 xp.resize(nb_elem,dimension);
185 }
186 else
187 nb_elem=xp.dimension(0);
188
189 int num_som;
190
191 xp=0;
192 for (int num_elem=0; num_elem<nb_elem; num_elem++)
193 {
194 int nb_som_reel=nb_som();
195 while (les_Polys(num_elem,nb_som_reel-1)==-1) nb_som_reel--;
196 for(int s=0; s<nb_som_reel; s++)
197 {
198 num_som = les_Polys(num_elem,s);
199 for(int i=0; i<dimension; i++)
200 xp(num_elem,i) += le_domaine.coord(num_som,i)/nb_som_reel;
201 }
202 }
203}
