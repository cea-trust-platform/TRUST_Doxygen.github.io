94double Op_Conv_EF_Stab_PolyMAC_Elem::calculer_dt_stab() const
95{
96 double dt = 1e10;
97 const Zone_Poly_base& zone = la_zone_poly_.valeur();
98 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face(), &ve = zone.volumes(), &pe = zone.porosite_elem();
99 const DoubleTab& vit = vitesse_->valeurs(),
100 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
101 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins();
102 int i, e, f, n, N = std::min(vit.line_size(), equation().inconnue().valeurs().line_size());
103 DoubleTrav flux(N); //somme des flux pf * |f| * vf
104
105 for (e = 0; e < zone.nb_elem(); e++)
106 {
107 for (flux = 0, i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
108 for (n = 0; n < N; n++)
109 flux(n) += pf(f) * fs(f) * std::max((e == f_e(f, 1) ? 1 : -1) * vit(f, n), 0.); //seul le flux entrant dans e compte
110
111 for (n = 0; n < N; n++)
112 if ((!alp || (*alp)(e, n) > 1e-3) && flux(n)) dt = std::min(dt, pe(e) * ve(e) / flux(n));
113 }
114
115 return Process::mp_min(dt);
116}

154void Op_Conv_EF_Stab_PolyMAC_Elem::ajouter_blocs(matrices_t mats, DoubleTab& secmem, const tabs_t& semi_impl) const
155{
156 const Zone_Poly_base& zone = la_zone_poly_.valeur();
157 const IntTab& f_e = zone.face_voisins(), &fcl = ref_cast(Champ_Elem_PolyMAC, equation().inconnue().valeur()).fcl(), &fcl_v = ref_cast(Champ_Face_PolyMAC, vitesse_.valeur()).fcl();
158 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face();
159 const Champ_Inc_base& cc = le_champ_inco.non_nul() ? le_champ_inco->valeur() : equation().champ_convecte();
160 const std::string& nom_cc = cc.le_nom().getString();
161 const DoubleTab& vit = vitesse_->valeurs(), &vcc = semi_impl.count(nom_cc) ? semi_impl.at(nom_cc) : cc.valeurs(), bcc = cc.valeur_aux_bords();
162 int i, j, e, eb, f, n, m, N = vcc.line_size(), Mv = vit.line_size(), M;
163
164 Matrice_Morse *m_vit = mats.count("vitesse") ? mats.at("vitesse") : NULL;
165 std::vector<std::tuple<const DoubleTab *, Matrice_Morse *, int>> d_cc; //liste des derivees de cc a renseigner : couples (derivee de cc, matrice, nb de compos de la variable)
166 if (!semi_impl.count(nom_cc))
167 for (auto &i_m : mats)
168 if (cc.derivees().count(i_m.first))
169 d_cc.push_back(std::make_tuple(&cc.derivees().at(i_m.first), i_m.second, equation().probleme().get_champ(i_m.first.c_str()).valeurs().line_size()));
170
171 DoubleTrav dv_flux(N), dc_flux(2, N); //derivees du flux convectif a la face par rapport a la vitesse / au champ convecte amont / aval
172
173 /* convection aux faces internes (fcl(f, 0) == 0), de Neumann_val_ext ou de Dirichlet */
174 for (f = 0; f < zone.nb_faces(); f++)
175 if (!fcl(f, 0) || (fcl(f, 0) > 4 && fcl(f, 0) < 7))
176 {
177 for (dv_flux = 0, dc_flux = 0, i = 0; i < 2; i++)
178 for (e = f_e(f, i), n = 0, m = 0; n < N; n++, m += (Mv > 1))
179 {
180 double v = vit(f, m) ? vit(f, m) : DBL_MIN, fac = pf(f) * fs(f) * (1. + (v * (i ? -1 : 1) > 0 ? 1. : -1) * alpha) / 2;
181 dv_flux(n) += fac * (e >= 0 ? vcc(e, n) : bcc(f, n)); //f est reelle -> indice trivial dans bcc
182 dc_flux(i, n) = e >= 0 ? fac * vit(f, m) : 0;
183 }
184
185// second membre
186 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
187 if (e < zone.nb_elem())
188 for (n = 0, m = 0; n < N; n++, m += (Mv > 1))
189 secmem(e, n) -= (i ? -1 : 1) * dv_flux(n) * vit(f, m);
190// derivees : vitesse
191 if (m_vit && fcl_v(f, 0) < 2)
192 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
193 if (e < zone.nb_elem())
194 for (n = 0, m = 0; n < N; n++, m += (Mv > 1))
195 (*m_vit)(N * e + n, Mv * f + m) += (i ? -1 : 1) * dv_flux(n);
196// derivees : champ convecte
197 for (auto &&d_m_i : d_cc)
198 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
199 if (e < zone.nb_elem())
200 for (j = 0; j < 2 && (eb = f_e(f, j)) >= 0; j++)
201 for (n = 0, m = 0, M = std::get<2>(d_m_i); n < N; n++, m += (M > 1))
202 (*std::get<1>(d_m_i))(N * e + n, M * eb + m) += (i ? -1 : 1) * dc_flux(j, n) * (*std::get<0>(d_m_i))(eb, m);
203 }

227void Op_Conv_EF_Stab_PolyMAC_Elem::mettre_a_jour(double temps)
228{
229 Op_Conv_PolyMAC_base::mettre_a_jour(temps);
230 const Zone_Poly_base& zone = la_zone_poly_.valeur();
231 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces();
232 const Champ_Inc_base& cc = le_champ_inco.non_nul() ? le_champ_inco.valeur() : equation().champ_convecte();
233 const DoubleVect& pf = zone.porosite_face(), &pe = zone.porosite_elem(), &fs = zone.face_surfaces(), &ve = zone.volumes();
234 const DoubleTab& vit = vitesse_->valeurs(), &vcc = cc.valeurs(), bcc = cc.valeur_aux_bords(), &xv = zone.xv(), &xp = zone.xp();
235 DoubleTab balp;
236 if (vd_phases_.size()) balp = equation().inconnue().valeur().valeur_aux_bords();
237
238
239 int i, e, f, d, D = dimension, n, m, N = vcc.line_size(), M = vit.line_size();
240 DoubleTrav cc_f(N); //valeur du champ convecte aux faces
241 /* flux aux bords */
242 for (f = 0; f < zone.premiere_face_int(); f++)
243 {
244 for (cc_f = 0, i = 0; i < 2; i++)
245 for (e = f_e(f, i), n = 0, m = 0; n < N; n++, m += (M > 1))
246 cc_f(n) += (1. + (vit(f, m) * (i ? -1 : 1) >= 0 ? 1. : -1.) * alpha) / 2 * (e >= 0 ? vcc(e, n) : bcc(f, n));
247
248 for (n = 0, m = 0; n < N; n++, m += (M > 1)) flux_bords_(f, n) = pf(f) * fs(f) * vit(f, m) * cc_f(n);
249 }
250
251 if (cc_phases_.size())
252 for (n = 0, m = 0; n < N; n++, m += (M > 1))
253 if (cc_phases_[n].non_nul()) /* mise a jour des champs de debit */
254 {
255 Champ_Face_PolyMAC& c_ph = ref_cast(Champ_Face_PolyMAC, cc_phases_[n].valeur());
256 DoubleTab& v_ph = c_ph.valeurs();
257 for (f = 0; f < zone.nb_faces(); v_ph(f) *= vit(f, m) * pf(f), f++)
258 for (v_ph(f) = 0, i = 0; i < 2; i++) v_ph(f) += (1. + (vit(f, m) * (i ? -1 : 1) >= 0 ? 1. : -1.) * alpha) / 2 * ((e = f_e(f, i)) >= 0 ? vcc(e, n) : bcc(f, n));
259 c_ph.changer_temps(temps);
260 }
261
262 if (vd_phases_.size())
263 for (n = 0, m = 0; n < N; n++, m += (M > 1))
264 if (vd_phases_[n].non_nul()) /* mise a jour des champs de vitesse debitante */
265 {
266 const DoubleTab& alp = equation().inconnue().valeurs();
267 Champ_Face_PolyMAC& c_ph = ref_cast(Champ_Face_PolyMAC, vd_phases_[n].valeur());
268 DoubleTab& v_ph = c_ph.valeurs();
269 /* on remplit la partie aux faces, puis on demande au champ d'interpoler aux elements */
270 for (f = 0; f < zone.nb_faces(); v_ph(f) *= vit(f, m) * pf(f), f++)
271 for (v_ph(f) = 0, i = 0; i < 2; i++) v_ph(f) += (1. + (vit(f, m) * (i ? -1 : 1) >= 0 ? 1. : -1.) * alpha) / 2 * ((e = f_e(f, i)) >= 0 ? alp(e, n) : balp(f, n));
272 c_ph.changer_temps(temps);
273 }
274
275 DoubleTrav G(N), v(N, D);
276 double Gt;
277 if (x_phases_.size())
278 for (e = 0; e < zone.nb_elem(); e++) //titre : aux elements
279 {
280 for (v = 0, i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
281 for (n = 0; n < N; n++)
282 for (d = 0; d < D; d++)
283 v(n, d) += fs(f) * pf(f) * (xv(f, d) - xp(e, d)) * (e == f_e(f, 0) ? 1 : -1) * vit(f, n) / (pe(e) * ve(e));
284 for (Gt = 0, n = 0; n < N; Gt += G(n), n++) G(n) = vcc(e, n) * sqrt(zone.dot(&v(n, 0), &v(n, 0)));
285 for (n = 0; n < N; n++)
286 if (x_phases_[n].non_nul()) x_phases_[n]->valeurs()(e) = Gt ? G(n) / Gt : 0;
287 }
288 if (x_phases_.size())
289 for (n = 0; n < N; n++)
290 if (x_phases_[n].non_nul()) x_phases_[n]->changer_temps(temps);
291}
