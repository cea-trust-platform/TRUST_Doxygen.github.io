59//! Fonction outil pour Terme_Source_Acceleration_VDF_Face::ajouter<br>Ajout des contributions d'une liste contigue de faces du terme source de translation:<br>s_face = terme_source * rho<br>resu += integrale (s_face) sur le volume de controle de la vitesse.<br>On traite les cas suivants:<br>rho = reference nulle (=> rho = 1.) sinon rho != nul<br>faces de bord => sortie libre<br>periodicite<br>faces_internes
60static void TSAVDF_ajouter_liste_faces(const int premiere_face, const int derniere_face,
61 const DoubleVect& volumes_entrelaces,
62 const DoubleVect& volumes_elements,
63 const DoubleVect& porosite_surf,
64 const IntVect& orientation,
65 const IntTab& face_voisins,
66 const REF(Champ_base) & ref_rho,
67 const DoubleTab& terme_source,
68 DoubleTab& s_face,
69 DoubleTab& resu)
70{
71 int num_face;
72// Pointeur constant sur tableau constant.
73// Pointeur nul si ref_rho_ est une reference nulle.
74 const DoubleTab * const rho_elem =
75 (ref_rho.non_nul()) ? &(ref_rho->valeurs()) : 0;
76
77 for (num_face=premiere_face; num_face<derniere_face; num_face++)
78 {
79 const double vol = volumes_entrelaces(num_face)*porosite_surf(num_face);
80 const int ncomp = orientation(num_face);
81 const double src = terme_source(num_face, ncomp);
82
83 double rho = 1.;
84
85// Calcul d'un rho moyen sur le volume de controle de la vitesse
86 if (rho_elem)
87 {
88 const int elem0 = face_voisins(num_face,0);
89 const int elem1 = face_voisins(num_face,1);
90 double rho0 = 0, rho1 = 0, vol0 = 0, vol1 = 0;
91 if (elem0 >= 0)
92 {
93 rho0 = (*rho_elem)(elem0);
94 vol0 = volumes_elements(elem0);
95 }
96 if (elem1 >= 0)
97 {
98 rho1 = (*rho_elem)(elem1);
99 vol1 = volumes_elements(elem1);
100 }
101 rho = (rho0 * vol0 + rho1 * vol1) / (vol0 + vol1);
102 }
103
104 double a = src * rho;
105 s_face(num_face) = a;
106// Integrale sur le volume de controle :
107 resu(num_face) += a * vol;
108 }
109}

115//! Ajoute le terme (la_source_ * rho * volume_entrelace) au champ resu.<br>On suppose que resu est discretise comme la vitesse en VDF.<br>Effet de bord:<br>On met (la_source_ * rho) dans terme_source_post_
116void Terme_Source_Acceleration_VDF_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
117{
118 const Zone_VDF& zone_VDF = la_zone_VDF_.valeur();
119 const Zone_Cl_VDF& zone_Cl_VDF = la_zone_Cl_VDF_.valeur();
120 const IntTab& face_voisins = zone_VDF.face_voisins();
121 const IntVect& orientation = zone_VDF.orientation();
122 const DoubleVect& porosite_surf = zone_VDF.porosite_face();
123 const DoubleVect& volumes_entrelaces = zone_VDF.volumes_entrelaces();
124
125 DoubleTab& s_face = get_set_terme_source_post().valeur().valeurs();
126 s_face = 0.;
127
128// Calcul de la_source_ en fonction des champs d'acceleration et de la
129// vitesse du fluide.
130 const int dim = Objet_U::dimension;
131 const int nb_faces = secmem.dimension(0);
132 DoubleTab acceleration_aux_faces(nb_faces, dim);
133 calculer_la_source(acceleration_aux_faces);
134
135// Boucle sur les conditions limites pour traiter les faces de bord
136
137 for (int n_bord=0; n_bord<zone_VDF.nb_front_Cl(); n_bord++)
138 {
139// pour chaque Condition Limite on regarde son type
140// Si face de Dirichlet ou de Symetrie on ne fait rien
141// Si face de Neumann on calcule la contribution au terme source
142 const Cond_lim& la_cl = zone_Cl_VDF.les_conditions_limites(n_bord);
143 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
144 const int ndeb = le_bord.num_premiere_face();
145 const int nfin = ndeb + le_bord.nb_faces();
146 TSAVDF_ajouter_liste_faces(ndeb, nfin,
147 volumes_entrelaces,
148 la_zone_VDF_.valeur().volumes(),
149 porosite_surf,
150 orientation,
151 face_voisins,
152 ref_rho_,
153 acceleration_aux_faces,
154 s_face,
155 secmem);
156 }
157
158
159// Boucle sur les faces internes
160 {
161 const int ndeb = zone_VDF.premiere_face_int();
162 const int nfin = zone_VDF.nb_faces();
163 TSAVDF_ajouter_liste_faces(ndeb, nfin,
164 volumes_entrelaces,
165 la_zone_VDF_.valeur().volumes(),
166 porosite_surf,
167 orientation,
168 face_voisins,
169 ref_rho_,
170 acceleration_aux_faces,
171 s_face,
172 secmem);
173 }
174 {
175// Force la periodicite
176 for (int n_bord=0; n_bord<zone_VDF.nb_front_Cl(); n_bord++)
177 {
178 const Cond_lim& la_cl = zone_Cl_VDF.les_conditions_limites(n_bord);
179 if (sub_type(Periodique,la_cl.valeur()))
180 {
181 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
182 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
183 int nb_faces_bord=le_bord.nb_faces();
184 ArrOfInt fait(nb_faces_bord);
185 fait = 0;
186 for (int ind_face=0; ind_face<nb_faces_bord; ind_face++)
187 {
188 if (fait[ind_face] == 0)
189 {
190 int ind_face_associee = la_cl_perio.face_associee(ind_face);
191 fait[ind_face] = 1;
192 fait[ind_face_associee] = 1;
193 int face = le_bord.num_face(ind_face);
194 int face_associee = le_bord.num_face(ind_face_associee);
195 double val = 0.5*(secmem(face_associee)+secmem(face));
196 secmem(face)=secmem(face_associee) = val;
197 }//!< if fait
198 }//!< for face
199 }//!< sub_type Perio
200 }
201 }
202}

213//! Calcul des trois composantes du champ de vitesse fluide au centre<br>de chaque face. Le resultat est stocke dans v_faces_stockage et on renvoie<br>une reference au tableau. Pas d'espace virtuel dans le tableau.<br>La composante normale a la face est deja connue: c'est la valeur discrete.<br>Les autres composantes sont calculees en faisant la moyenne des vitesse<br>des faces des elements voisins qui ont la bonne orientation, ponderee<br>par le volume de l'element.<br>Traitement des bords : on prend la vitesse de l'element voisin (a priori<br>pas bon pour les conditions aux limites de vitesse imposee au bord,<br>mais le vpoint est corrige ensuite pour imposer la vitesse).
214const DoubleTab& Terme_Source_Acceleration_VDF_Face::calculer_vitesse_faces(
215 DoubleTab& v_faces_stockage) const
216{
217 const Zone_VDF& zone_VDF = la_zone_VDF_.valeur();
218 const IntVect& orientation = zone_VDF.orientation();
219 const IntTab& faces_voisins = zone_VDF.face_voisins();
220 const DoubleVect& volumes = zone_VDF.volumes(); //!< volumes des elements
221 const IntTab& elem_faces = zone_VDF.elem_faces();
222 const DoubleTab& v_faces = get_eq_hydraulique().inconnue()->valeurs();
223 const int dim = Objet_U::dimension;
224 const int nb_faces = v_faces.dimension(0);
225 v_faces_stockage.resize(nb_faces, dim);
226 int i_face;
227 ArrOfDouble composante_vitesse(3);
228 for (i_face = 0; i_face < nb_faces; i_face++)
229 {
230 const int orientation_face = orientation(i_face);
231 composante_vitesse=0;
232 int composante;
233
234// Numeros des deux elements voisins de la face (-1 si face de bord)
235 int elem[2];
236 elem[0] = faces_voisins(i_face, 0);
237 elem[1] = faces_voisins(i_face, 1);
238
239// Volumes de ces deux elements (0. si pas de voisin)
240 double volume_elem[2] = {0., 0.};
241 if (elem[0] >= 0)
242 volume_elem[0] = volumes(elem[0]);
243 if (elem[1] >= 0)
244 volume_elem[1] = volumes(elem[1]);
245
246 const double i_volume_total = 1. / (volume_elem[0] + volume_elem[1]);
247
248 for (composante = 0; composante < dim; composante++)
249 {
250 if (composante == orientation_face)
251 {
252 composante_vitesse[composante] = v_faces(i_face);
253 }
254 else
255 {
256// Calcul de la moyenne des vitesses sur les faces voisines
257// qui ont la bonne orientation:
258 composante_vitesse[composante] = 0.;
259 int i_elem;
260 for (i_elem = 0; i_elem < 2; i_elem++)
261 {
262 if (elem[i_elem] >= 0)
263 {
264 const int element = elem[i_elem];
265 const int face1 = elem_faces(element, composante);
266 const int face2 = elem_faces(element, composante + dim);
267 const double v1 = v_faces(face1);
268 const double v2 = v_faces(face2);
269 const double p = volume_elem[i_elem]; //!< Ponderation par le volume
270 composante_vitesse[composante] += (v1 + v2) * 0.5 * p;
271 }
272 }
273 composante_vitesse[composante] *= i_volume_total;
274 }
275 }
276 for (composante = 0; composante < dim; composante++)
277 {
278 v_faces_stockage(i_face, composante) = composante_vitesse[composante];
279 }
280 }
281 return v_faces_stockage;
282}
