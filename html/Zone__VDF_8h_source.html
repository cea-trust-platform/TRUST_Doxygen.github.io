60//! class Zone_VDF<br>Classe instanciable qui derive de Zone_VF.<br>Cette classe contient les informations geometriques que demande<br>la methode des Volumes Differences Finis.<br>La classe porte un certain nombre d'informations concernant les faces<br>L'ensemble des faces est numerote comme suit:<br>- les faces qui sont sur une Zone_joint apparaissent en premier<br>(dans l'ordre du vecteur les_joints)<br>- les faces qui sont sur une Zone_bord apparaissent ensuite<br>(dans l'ordre du vecteur les_bords)<br>- les faces internes apparaissent ensuite<br>A chaque face on fait correspondre un int qui indique son orientation.<br>On suppose qu'a l'interieur de chaque famille de faces (bord,joint,interne)<br>on trouve:<br>- le bloc des faces d'equation x = cte (faces d'orientation 0)<br>- le bloc des faces d'equation y = cte (faces d'orientation 1)<br>- le bloc des faces d'equation z = cte (faces d'orientation 2)<br>Pour le bloc des faces de bord on conserve les sous-blocs constitues par<br>les faces d'un meme bord.<br>On n'a pas besoin d'une numerotation particuliere des elements.<br>On a introduit la notion d'arete pour le calcul des flux diffusifs et<br>convectifs dans la conservation de la quantite de mouvement.<br>Le tableau Qdm contient la connectivite arete/faces. Dans le tableau<br>Qdm les aretes apparaissent dans l'ordre suivant:<br>- bloc des aretes joints<br>- bloc des aretes bord<br>- bloc des aretes mixtes<br>- bloc des aretes internes<br>A l'interieur de chaque bloc les aretes apparaissent dans l'ordre<br>suivant: aretes XY, aretes XZ et aretes YZ.
61//

285// Fonction de calcul utilisable uniquement en coordonnees cylindriques
286// de la distance entre les centres de 2 faces dans la direction k
287
288//
289inline double Zone_VDF::dist_face_axi(int fac1, int fac2, int k) const
290{
291 if (k != 1)
292 return xv_(fac2,k) - xv_(fac1,k);
293 else
294 {
295 double d_teta = xv_(fac2,1) - xv_(fac1,1);
296 if (d_teta < 0)
297 d_teta += 2.0*M_PI;
298 return d_teta*xv_(fac1,0);
299 }
300}

302// Fonction de calcul utilisable uniquement en coordonnees cartesiennes
303// de la distance normale pour une face interne
304// Pour une face interne distance normale = distance entre les centres des
305// deux mailles voisines.

308inline double Zone_VDF::dist_norm(int num_face) const
309{

312 int k = orientation_[num_face];
313 return (xp_(n2,k) - xp_(n1,k));

316// Fonction de calcul utilisable uniquement en coordonnees cylindriques
317// de la distance normale pour une face interne
318
319//
320inline double Zone_VDF::dist_norm_axi(int num_face) const
321{
322 int n1 = face_voisins_(num_face,0);
323 int n2 = face_voisins_(num_face,1);
324 int k = orientation_[num_face];
325 double dist;
326 if (k != 1)
327 dist = xp_(n2,k) - xp_(n1,k);
328 else
329 {
330 double d_teta = xp_(n2,1) - xp_(n1,1);
331 if (d_teta < 0)
332 d_teta += 2.0*M_PI;
333 dist = d_teta*xp_(n1,0);
334 }
335 return dist;
336}

338// Fonction de calcul utilisable uniquement en coordonnees cartesiennes
339// de la distance normale pour une face de bord
340// Pour une face de bord distance normale est egale a la distance entre le
341// centre de la maille voisine et le bord.
342
343//
344inline double Zone_VDF::dist_norm_bord(int num_face) const
345{
346 int n1 = face_voisins_(num_face,0);
347 int n2 = face_voisins_(num_face,1);
348 assert(num_face<nb_faces_bord() || n1==-1 || n2==-1); //!< Verifie que num_face est bien une face de bord reelle ou virtuelle
349 int k = orientation_[num_face];
350 if (n1!=-1)
351 return (xv_(num_face,k) - xp_(n1,k));
352 else
353 return (xp_(n2,k) - xv_(num_face,k));
354}

393inline double Zone_VDF::distance_face(int n1, int n2, int k) const
394{
395 double dist,d_teta;
396 assert ( (orientation_[n1]==orientation_[n2]) );
397 if ( (axi!=1) || (k!=1) )
398 dist = xv_(n2,k) - xv_(n1,k);
399 else
400 {
401 d_teta = xv_(n2,1) - xv_(n1,1);
402 if (d_teta < 0)
403 d_teta += 2.0*M_PI;
404 dist = d_teta*xv_(n1,0);
405 }
406 return dist;
407}

409// Fonction de calcul de la distance normale pour une face quelconque
410// utilisable en coordonnees cylindriques et cartesiennes
411
412//
413inline double Zone_VDF::distance_normale(int num_face) const
414{
415
416 double dist,d_teta;
417 int n1 = face_voisins_(num_face,0);
418 int n2 = face_voisins_(num_face,1);
419 int k = orientation_[num_face];
420 if ((n1!=-1) && (n2!=-1))
421 {
422 if ( (k!=1) || (axi!=1) )
423 dist = xp_(n2,k) - xp_(n1,k);
424 else
425 {
426 d_teta = xp_(n2,1) - xp_(n1,1);
427 if (d_teta < 0)
428 d_teta += 2.0*M_PI;
429 dist = d_teta*xp_(n1,0);
430 }
431 }
432 else if (n1!=-1)
433 {
434 if ( (k!=1) || (axi!=1) )
435 dist = (xv_(num_face,k) - xp_(n1,k));
436 else
437 {
438 d_teta = xv_(num_face,1) - xp_(n1,1);
439 if (d_teta < 0)
440 d_teta += 2.0*M_PI;
441 dist = d_teta*xp_(n1,0);
442 }
443 }
444 else
445 {
446 if ( (k!=1) || (axi!=1) )
447 dist = (xp_(n2,k) - xv_(num_face,k));
448 else
449 {
450 d_teta = xp_(n2,1) - xv_(num_face,1);
451 if (d_teta < 0)
452 d_teta += 2.0*M_PI;
453 dist = d_teta*xp_(n2,0);
454 }
455 }
456 return dist;
457}

591//
592inline int Zone_VDF::face_amont_conj(int num_face, int k, int i) const
593{
594 int ori = orientation(num_face);
595 int elem = face_voisins_(num_face,1);
596 int face_conj=-2,face,elem_bis=-2;
597
598 if(elem != -1)
599 {
600 face = elem_faces_(elem, k+i*dimension);
601 elem_bis = face_voisins_(face,i);
602 if (elem_bis != -1)
603 face_conj = elem_faces_(elem_bis, ori);
604 else
605 face_conj = -1;
606 }
607 if ((elem==-1) || (elem_bis==-1))
608 {
609 elem = face_voisins_(num_face,0);
610 if(elem != -1)
611 {
612 face = elem_faces_(elem, k+i*dimension);
613 elem_bis = face_voisins_(face,i);
614 if (elem_bis != -1)
615 face_conj = elem_faces_(elem_bis, ori+dimension);
616 else
617 face_conj = -1;
618 }
619 }
620 assert(face_conj!=-2);
621 return face_conj;
622}

640 if (elem == -1)
641 {
642 elem = face_voisins_(num_face,0);
643 if(elem != -1)
644 {
645 int face = elem_faces_(elem, k+i*dimension);
646 int elem_bis = face_voisins_(face,i);
647 if (elem_bis != -1)
648 elem = elem_faces_(elem_bis, ori+dimension);

651 }
652 }
