22//! calcul de la norme Linfini de la matrice<br>Propriete: on note |x| la norme Linfini de x (vecteur ou matrice)<br>On a |m * x| <= |m| * |x|<br>En pratique: c'est le max sur j de la somme sur i de std::fabs(m(i,j))
23inline double Matrice33::norme_Linfini()
24{
25 double x = std::fabs(m[0][0]) + std::fabs(m[0][1]) + std::fabs(m[0][2]);
26 double y = std::fabs(m[1][0]) + std::fabs(m[1][1]) + std::fabs(m[1][2]);
27 double z = std::fabs(m[2][0]) + std::fabs(m[2][1]) + std::fabs(m[2][2]);
28 double resu = ((x > y) ? x : y);
29 resu = ((resu > z) ? resu : z);
30 return resu;
31}

33//! produit avec de la matrice avec le vecteur x.
34inline void Matrice33::produit(const Matrice33& m, const Vecteur3& x, Vecteur3& y)
35{
36 y.v[0] = m.m[0][0] * x.v[0] + m.m[0][1] * x.v[1] + m.m[0][2] * x.v[2];
37 y.v[1] = m.m[1][0] * x.v[0] + m.m[1][1] * x.v[1] + m.m[1][2] * x.v[2];
38 y.v[2] = m.m[2][0] * x.v[0] + m.m[2][1] * x.v[1] + m.m[2][2] * x.v[2];
39}

41inline void Vecteur3::produit_vectoriel(const Vecteur3& x, const Vecteur3& y, Vecteur3& z)
42{
43 z.v[0] = x.v[1] * y.v[2] - x.v[2] * y.v[1];
44 z.v[1] = x.v[2] * y.v[0] - x.v[0] * y.v[2];
45 z.v[2] = x.v[0] * y.v[1] - x.v[1] * y.v[0];
46}

48inline double Vecteur3::produit_scalaire(const Vecteur3& x, const Vecteur3& y)
49{

68//! calcul de l'inverse.<br>Si le determinant de "matrice" est nul, exit() si exit_on_error (valeur par defaut)<br>sinon on ne remplit pas matrice_inv et on renvoie 0.<br>Valeur de retour: determinant de la "matrice" (pas de l'inverse !)
69inline double Matrice33::inverse(const Matrice33& matrice, Matrice33& matrice_inv, int exit_on_error)
70{
71 const double a00 = matrice.m[0][0];
72 const double a01 = matrice.m[0][1];
73 const double a02 = matrice.m[0][2];
74 const double a10 = matrice.m[1][0];
75 const double a11 = matrice.m[1][1];
76 const double a12 = matrice.m[1][2];
77 const double a20 = matrice.m[2][0];
78 const double a21 = matrice.m[2][1];
79 const double a22 = matrice.m[2][2];
80// calcul de valeurs temporaires pour optimisation
81 const double t4 = a00*a11;
82 const double t6 = a00*a12;
83 const double t8 = a01*a10;
84 const double t10 = a02*a10;
85 const double t12 = a01*a20;
86 const double t14 = a02*a20;
87 const double t = t4*a22-t6*a21-t8*a22+t10*a21+t12*a12-t14*a11;
88 if (t==0.)
89 {
90 if (exit_on_error)
91 {
92 Cerr << "Error in Matrice33::inverse: determinant is null" << finl;
93 Process::exit();
94 }
95// To avoid the compiler to complain about "might be non initialized":
96 for (int i = 0; i < 3; i++)
97 for (int j = 0; j < 3; j++)
98 matrice_inv.m[i][j] = 0.;
99 return 0.;
100 }
101 const double t17 = 1/(t);
102
103// calcul de la matrice inverse
104 matrice_inv.m[0][0] = (a11*a22-a12*a21)*t17;
105 matrice_inv.m[0][1] = -(a01*a22-a02*a21)*t17;
106 matrice_inv.m[0][2] = -(-a01*a12+a02*a11)*t17;
107 matrice_inv.m[1][0] = (-a10*a22+a12*a20)*t17;
108 matrice_inv.m[1][1] = (a00*a22-t14)*t17;
109 matrice_inv.m[1][2] = -(t6-t10)*t17;
110 matrice_inv.m[2][0] = -(-a10*a21+a11*a20)*t17;
111 matrice_inv.m[2][1] = -(a00*a21-t12)*t17;
112 matrice_inv.m[2][2] = (t4-t8)*t17;
113 return t;
114}
