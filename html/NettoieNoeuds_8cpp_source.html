96void NettoieNoeuds::nettoie(Domaine& dom)
97{
98 if (NettoiePasNoeuds==1)
99 return;
100
101// Autorise la modification de structure des tableaux sommets et elements:
102 Scatter::uninit_sequential_domain(dom);
103
104 DoubleTab& coord_sommets=dom.les_sommets();
105 Zone& zone=dom.zone(0);
106 IntTab& les_elems=zone.les_elems();
107 int nb_som_init=coord_sommets.dimension(0);
108 int nb_elem=les_elems.dimension(0);
109 int nb_som_elem=les_elems.dimension(1);
110 ArrOfInt renum_som_old2new(nb_som_init);
111
112 int compteur=0;
113
114 {
115// int i;
116 for (int i=0; i<nb_som_init ; i++)
117 renum_som_old2new[i]=-1;
118
119 for (int i=0; i<nb_elem ; i++)
120 for (int j=0; j< nb_som_elem; j++)
121 {
122// GF dans le cas ou on a des polyedres
123 if (les_elems(i,j)==-1) break;
124 int& tmp=renum_som_old2new[les_elems(i,j)];
125 assert (tmp < compteur);
126 if(tmp==-1)
127 tmp=compteur++;
128 }
129// ajout GF sinon un sommet double sur un bord devient -1
130 int nb_bords=zone.nb_front_Cl();
131 for (int ii=0; ii<nb_bords; ii++)
132 {
133 const Frontiere& front= zone.frontiere(ii);
134 const Faces& faces=front.faces();
135 const IntTab& faces_sommets=faces.les_sommets();
136 int nb_faces=faces_sommets.dimension(0);
137 int nb_som_face=faces_sommets.dimension(1);
138 for(int i=0; i<nb_faces; i++)
139 for(int j=0; j<nb_som_face; j++)
140 {
141// dans le cas ou l'on a des polygones
142 if (faces_sommets(i,j)==-1) break;
143 int& tmp=renum_som_old2new[faces_sommets(i,j)];
144 assert (tmp < compteur);
145 if(tmp==-1)
146 tmp=compteur++;
147 }
148 }
149 }
150 int nb_som_new=compteur;
151 Cerr << "Elimination of " << nb_som_init - nb_som_new << " unnecessary nodes in the domain " << dom.le_nom() << finl;
152 if(nb_som_new!=nb_som_init)
153 {
154 ArrOfInt renum_som_new2old(nb_som_new);
155 {
156 int j;
157 for (int i=0; i<nb_som_init ; i++)
158 if((j=renum_som_old2new[i])!=-1)
159 renum_som_new2old[j]=i;
160 }
161 {
162 int i, j;
163 DoubleTab new_coord(nb_som_new, dimension);
164
165 for (i=0; i<nb_som_new ; i++)
166 for (j=0; j<dimension ; j++)
167 {
168 new_coord(i,j)=coord_sommets(renum_som_new2old[i],j);
169 }
170 coord_sommets=new_coord;
171
172 for (i=0; i<nb_elem ; i++)
173 for (j=0; j<nb_som_elem ; j++)
174 {
175 int num_som=les_elems(i,j);
176 if (num_som!=-1)
177 les_elems(i,j)=renum_som_old2new[num_som];
178 else
179 les_elems(i,j)=-1;
180 }
181 }
182
183 {
184// On recupere les bords :
185 LIST_CURSEUR(Bord) curseur(zone.faces_bord());;;
186 while(curseur)
187 {
188// while(curseur)
189 Frontiere& front=curseur.valeur();
190 Faces& faces=front.faces();
191 IntTab& faces_sommets=faces.les_sommets();
192 IntTab old_faces_sommets(faces.les_sommets());
193 int nb_faces=faces_sommets.dimension(0);
194 int nb_som_face=faces_sommets.dimension(1);
195 for(int i=0; i<nb_faces; i++)
196 for(int j=0; j<nb_som_face; j++)
197 {
198 int som=old_faces_sommets(i,j);
199 if (som!=-1)
200 faces_sommets(i,j)=
201 renum_som_old2new[som];
202 else
203 faces_sommets(i,j)=som;
204 }
205 ++curseur;
206 }
207 }
208 {
209// Les Faces Internes :
210 LIST_CURSEUR(Faces_Interne) curseur(zone.faces_int());;;
211 while(curseur)
212 {
213// while(curseur)
214 Frontiere& front=curseur.valeur();
215 Faces& faces=front.faces();
216 IntTab& faces_sommets=faces.les_sommets();
217 IntTab old_faces_sommets(faces.les_sommets());
218 int nb_faces=faces_sommets.dimension(0);
219 int nb_som_face=faces_sommets.dimension(1);
220 for(int i=0; i<nb_faces; i++)
221 for(int j=0; j<nb_som_face; j++)
222 faces_sommets(i,j)=
223 renum_som_old2new[old_faces_sommets(i,j)];
224 ++curseur;
225 }
226 }
227 {
228// Les Raccords
229 LIST_CURSEUR(Raccord) curseur(zone.faces_raccord());;;
230 while(curseur)
231 {
232 Frontiere& front=curseur->valeur();
233 Faces& faces=front.faces();
234 IntTab& faces_sommets=faces.les_sommets();
235 IntTab old_faces_sommets(faces.les_sommets());
236 int nb_faces=faces_sommets.dimension(0);
237 int nb_som_face=faces_sommets.dimension(1);
238 for(int i=0; i<nb_faces; i++)
239 for(int j=0; j<nb_som_face; j++)
240 faces_sommets(i,j)=
241 renum_som_old2new[old_faces_sommets(i,j)];
242 ++curseur;
243 }
244 }
245 }
246 Scatter::init_sequential_domain(dom);
247}
