70void Masse_VDF_base::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, double dt, const tabs_t& semi_impl, int resoudre_en_increments) const
71{
72 const std::string& nom_inco = equation().inconnue().le_nom().getString();
73 const DoubleTab& passe = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : equation().inconnue().passe();
74 const DoubleTab& inco = equation().inconnue().valeurs();
75 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
76 DoubleTrav diag(inco);
77
78 const int sz = equation().inconnue().valeurs().dimension_tot(0) * diag.line_size();
79 diag = 1.;
80 appliquer(diag); //!< M-1
81 int prems = 0;
82 if (penalisation_flag_)
83 {
84 if (mat && penalisation_matrice_ == 0)
85 {
86 double penal = 0;
87 for (int i = 0; i < sz; i++)
88 penal = std::max(penal, (*mat)(i, i));
89 penal = mp_max(penal);
90 penalisation_matrice_ = (mp_max_vect(diag) / dt + penal) * 1.e3;
91 prems = 1;
92 }
93 if (penalisation_secmem_ == 0)
94 penalisation_secmem_ = mp_max_vect(diag) * 1.e3;
95 }
96 else
97 {
98 penalisation_matrice_ = 0;
99 penalisation_secmem_ = 1;
100 }
101
102 for (int i = 0; i < sz; i++)
103 {
104 if (diag.addr()[i])
105 {
106 if (mat)
107 (*mat)(i, i) += 1. / (diag.addr()[i] * dt); //!< M/dt
108 secmem.addr()[i] += 1. / (diag.addr()[i] * dt) * passe.addr()[i];
109 }
110 else
111 {
112 if (prems)
113 {
114 if (mat)
115 (*mat)(i, i) += penalisation_matrice_;
116 prems = 0;
117 }
118 if (mat)
119 (*mat)(i, i) += penalisation_matrice_ / dt;
120 secmem.addr()[i] = penalisation_secmem_ * passe.addr()[i];
121 }
122 }
123}
