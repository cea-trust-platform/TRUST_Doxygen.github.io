78void Partitionneur_Fichier_MED::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
79{
80 if (! ref_domaine_.non_nul())
81 {
82 Cerr << "Error in Partitionneur_Fichier_MED::construire_partition\n";
83 Cerr << " The domain has not been associated" << finl;
84 exit();
85 }
86 if (filename_ == "")
87 {
88 Cerr << "Error in Partitionneur_Fichier_MED::construire_partition\n";
89 Cerr << " The file name has not been initialized" << finl;
90 exit();
91 }
92 if (fieldname_ == "")
93 {
94 Cerr << "Error in Partitionneur_Fichier_MED::construire_partition\n";
95 Cerr << " The field name has not been initialized" << finl;
96 exit();
97 }
98
99 Cerr << "Reading of splitting file : " << filename_ << finl;
100
101#ifdef MEDCOUPLING_
102 using namespace MEDCoupling;
103
104 MCAuto<MEDCouplingField> ffield = MEDCoupling::ReadField("Partition.med", "Thread");
105 MEDCouplingFieldInt32 * field = dynamic_cast<MEDCouplingFieldInt32 *>((MEDCouplingField *)ffield);
106
107 if(!field)
108 Process::exit("The initialized field is not composed of integers ");
109
110 DataArrayInt32 *da = field->getArray();
111 const True_int *field_values = da->begin();
112
113 const mcIdType sz = field->getNumberOfTuplesExpected();
114 elem_part.resize(sz);
115 std::copy(field_values, field_values+sz, elem_part.addr());
116// Sanity check
117 const int nelem = ref_domaine_.valeur().zone(0).nb_elem();
118 if (nelem != sz)
119 {
120 Cerr << "Error in Partitionneur_Fichier_MED::construire_partition" << finl;
121 Cerr << " The file contains an array of " << sz << " values." << finl;
122 Cerr << " The area contains " << nelem << " elements" << finl;
123 exit();
124 }
125// Checking that the full range is covered
126 MCAuto<DataArrayInt32> das = da->buildUniqueNotSorted();
127 MCAuto<DataArrayInt32> iot(DataArrayInt32::New());
128 mcIdType dnu;
129 const mcIdType max_v = das->getMaxValue(dnu);
130 iot->alloc(max_v+1, 1);
131 iot->iota();
132 if (!iot->isEqualWithoutConsideringStr(*das))
133 {
134 Cerr << "Error in Partitionneur_Fichier_MED::construire_partition" << finl;
135 Cerr << " The file contains/lacks integer values which do not cover the range [0; n_elem[." << finl;
136 Cerr << " Are some field values missing?" << finl;
137 exit();
138 }
139#else
140 Cerr << "Partitionneur_Fichier_MED requires TRUST compiled with MEDCoupling support!" << finl;
141 exit();
142#endif
143
144
145}
