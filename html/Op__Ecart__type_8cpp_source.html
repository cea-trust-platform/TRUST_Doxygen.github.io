67 val = tab1;

69

77DoubleTab Op_Ecart_type::calculer_valeurs() const
78{
79 double dt = dt_integration();
80 if (!est_egal(dt,dt_integration_carre()))
81 {
82 Cerr << "Not implemented yet in Op_Ecart_type::calculer_valeurs()" << finl;
83 exit();
84 }
85 DoubleTab ecart_type(valeurs_carre());
86 if ( dt > 0 )
87 {
88 ecart_type *= dt; //!< sum(I^2)*dt
89 ecart_type.ajoute_carre(-1,valeurs()); //!< sum(I^2)*dt-sum(I)^2
90 ecart_type /= (dt*dt); //!< sum(I^2)/dt-(sum(I)/dt)^2)
91 ecart_type.abs(); //!< To avoid negative number ?
92 ecart_type.racine_carree(); //!< sqrt(mean(I^2)-mean(I)^2)
93 }
94 return ecart_type;
95}
