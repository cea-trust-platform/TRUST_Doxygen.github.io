48Entree& Perte_Charge_Singuliere::lire_donnees(Entree& is)
49{
50 Motcle motlu;
51 Motcle acc_ouverte("{");
52 Motcles les_motcles(4);
53 les_motcles[0] = "KX";
54 les_motcles[1] = "KY";
55 les_motcles[2] = "KZ";
56 les_motcles[3] = "K";
57 regul_ = 0;
58 is >> motlu;
59 if (motlu != acc_ouverte)
60 {
61 Cerr << "On attendait le mot cle" << acc_ouverte << " a la place de " << motlu << finl;
62 Process::exit();
63 }
64 is >> motlu;
65 if (motlu != "dir")
66 {
67 Cerr << "On attendait le mot cle dir a la place de " << motlu << finl;
68 Process::exit();
69 }
70 else
71 {
72 is >> motlu;
73 int rang = les_motcles.search(motlu);
74 if (rang == -1)
75 {
76 Cerr << "Erreur a la lecture des donnees de Perte_Charge_Singuliere" << finl;
77 Cerr << "On attendait l'un des mots cles " << les_motcles << finl;
78 Cerr << "a la place de " << motlu << finl;
79 Process::exit();
80 }
81 direction_perte_charge_ = rang == 3 ? -1 : rang;
82
83 }
84 is >> motlu;
85 if (motlu == "regul")
86 {
87 regul_ = 1;
88 Nom eps_str, deb_str;
89 Param param("regul");
90 param.ajouter("K0", &K_, Param::REQUIRED);
91 param.ajouter("deb", &deb_str, Param::REQUIRED);
92 param.ajouter("eps", &eps_str, Param::REQUIRED);
93 param.lire_avec_accolades(is);
94 deb_cible_.setNbVar(1), eps_.setNbVar(1);
95 deb_cible_.setString(deb_str), eps_.setString(eps_str);
96 deb_cible_.addVar("t"), eps_.addVar("t");
97 deb_cible_.parseString(), eps_.parseString();
98 }
99 else if (motlu == "coeff")
100 {
101 is >> motlu;
102 convert_to(motlu.getChar(), K_);
103 }
104 else
105 {
106 Cerr << "On attendait le mot cle coeff ou regul a la place de " << motlu << finl;
107 Process::exit();
108 }
109
110 Cerr << " direction_perte_charge_ " << direction_perte_charge_ << finl;
111 Cerr << " perte de charge K_ " << K_ << finl;
112
113 return is;
114}

116void Perte_Charge_Singuliere::lire_surfaces(Entree& is, const Domaine& le_domaine,
117 const Zone_dis_base& zone_dis, IntVect& les_faces, IntVect& sgn)
118{
119 const Zone_VF& zvf = ref_cast(Zone_VF,zone_dis);
120 const IntTab& elem_faces = zvf.elem_faces();
121 const DoubleTab& xv = zvf.xv();
122 int nfe = zvf.zone().nb_faces_elem();
123 IntTab face_tab; //1 for faces in the surface
124 zvf.creer_tableau_faces(face_tab);
125 Champ_Don orientation;
126
127 int algo=-1;
128 Motcle method;
129
130 Motcle motlu;
131 Motcle acc_ouverte("{");
132 Motcle acc_fermee("}");
133
134 Nom nom_ss_zone, nom_surface;
135 double position;
136
137 /* Read plan */
138 is >> motlu;
139 if (motlu != "surface")
140 {
141 Cerr << "On attendait le mot cle surface a la place de " << motlu << finl;
142 Process::exit();
143 }
144 is >> motlu;
145 if (motlu != acc_ouverte)
146 {
147 Cerr << "On attendait le mot cle" << acc_ouverte << " a la place de " << motlu << finl;
148 Process::exit();
149 }
150 is >> method;
151 if (method=="X" || method=="Y" || method=="Z")
152 {
153 /* Subzone algorithm */
154 algo=0;
155 Nom direction=method, egal;
156 if ( ((direction=="X") && (direction_perte_charge()!=0))
157 || ((direction=="Y") && (direction_perte_charge()!=1))
158 || ((direction=="Z") && (direction_perte_charge()!=2)))
159 {
160 Nom dir,sect;
161 if (direction_perte_charge() == 0)
162 {
163 dir="KX";
164 sect="X";
165 }
166 else if (direction_perte_charge() == 1)
167 {
168 dir="KY";
169 sect="Y";
170 }
171 else if (direction_perte_charge() == 2)
172 {
173 dir="KZ";
174 sect="Z";
175 }
176 Cerr << "The section specified " << direction << " is not coherent with the coefficient pressure loss direction indicated by " << dir << finl;
177 Cerr << "Here the section to specify is " << sect << "." << finl;
178 Process::exit();
179
180 }
181 is >> egal >> position;
182 is >> nom_ss_zone;
183 Cerr << " position " << direction << " " << position << finl;
184 identifiant_ = nom_ss_zone + "_" + direction + "=" + Nom(position);
185 }
186 else if (method=="Surface")
187 {
188 /* Surface algorithm */
189 algo=1;
190 /*
191 if (zvf.dimension!=3)
192 {
193 Cerr << "Error in Perte_Charge_Singuliere::lire_surfaces" << finl;
194 Cerr << "The keyword " << method << " specified for plan definition is only possible in 3D !" << finl;
195 Cerr << "You must used the method of intersection between subzone and location of plane." << finl;
196 Process::exit();
197 }
198 */
199 if (zone_dis.que_suis_je().debute_par("Zone_VDF"))
200 {
201 Cerr << "Error in Perte_Charge_Singuliere::lire_surfaces" << finl;
202 Cerr << "The keyword " << method << " specified for plan definition is only possible in VEF discretization !" << finl;
203 Cerr << "You must used the method of intersection between subzone and location of plane." << finl;
204 Process::exit();
205 }
206 is >> nom_surface;
207 Cerr << " surface " << nom_surface << finl;
208 identifiant_ = nom_surface;
209 }
210 else
211 {
212 Cerr << "Error in Perte_Charge_Singuliere::lire_surfaces" << finl;
213 Cerr << "The keyword " << method << " specified for plan definition is not coherent !" << finl;
214 Process::exit();
215 }
216 is >> motlu;
217 if (motlu == "orientation") is >> orientation, is >> motlu; //on lit un champ pour orienter la surface
218
219 if (motlu != acc_fermee)
220 {
221 Cerr << "On attendait le mot cle" << acc_fermee << " a la place de " << motlu << finl;
222 Process::exit();
223 }
224 is >> motlu;
225 if (motlu != acc_fermee)
226 {
227 Cerr << "On attendait le mot cle" << acc_fermee << " a la place de " << motlu << finl;
228 Process::exit();
229 }
230 /* Found faces */
231 int compteur=0;
232
233 if (algo==0)
234 {
235 /* Subzone algorithm */
236 const Sous_Zone& ssz = le_domaine.ss_zone(nom_ss_zone);
237 for (int poly=0; poly<ssz.nb_elem_tot(); poly++)
238 {
239 for (int j=0; j<nfe; j++)
240 {
241 int numfa = elem_faces(ssz(poly),j);
242// numfa might be negative in case of polyhedron where number of faces for an elem varies (padding)
243 if (numfa >= 0 && est_egal(xv(numfa,direction_perte_charge()),position) )
244 {
245 bool trouve=0;
246 for (int i=0; i<compteur; i++)
247 if (les_faces[i]==numfa)
248 {
249 trouve=1;
250 break;
251 }
252 if (trouve==0) les_faces[compteur++] = numfa, face_tab(numfa) = 1;
253 }
254 }
255 }
256 }
257
258 else if (algo==1)
259 {
260 /* Surface algorithm */
261 const Domaine& le_domaine2D = ref_cast(Domaine,le_domaine.interprete().objet(nom_surface));
262 if (le_domaine2D.nb_zones() == 0)
263 {
264 Cerr << "Error in Perte_Charge_Singuliere::lire_surfaces" << finl;
265 Cerr << "You must correct your data file because the " << nom_surface << " surface is not well defined in parallel !" << finl;
266 Process::exit();
267 }
268 const Zone& zone_2D = le_domaine2D.zone(0);
269 const DoubleTab& coord_sommets_2D=le_domaine2D.coord_sommets();
270
271 DoubleTab xv2D;
272 zone_2D.calculer_centres_gravite(xv2D);
273 const int nb_elem_2D = zone_2D.nb_elem_tot();
274 int nse2D = zone_2D.nb_som_elem();
275
276 const DoubleTab& coord_sommets=le_domaine.coord_sommets();
277 const IntTab& face_sommets = zvf.face_sommets();
278 const Zone& zone_vol = le_domaine.zone(0);
279 const OctreeRoot& octree_vol = zone_vol.construit_octree();
280
281// Loop on faces on the surface domain
282 Cerr << " Surface " << nom_surface << " with " << nb_elem_2D << " faces" << finl;
283 for (int ind_face=0; ind_face<nb_elem_2D; ind_face++)
284 {
285// Fill elem_list_vol by elements surrounding the center of gravity
286 ArrOfInt elem_list_vol;
287 double zv2d=0;
288 if (Objet_U::dimension>2)
289 zv2d=xv2D(ind_face,2);
290 octree_vol.rang_elems_sommet(elem_list_vol,xv2D(ind_face,0),xv2D(ind_face,1),zv2d);
291 int nb_elem_vol = elem_list_vol.size_array();
292// Loop on elements on the volume domain
293 for (int ind_elem=0; ind_elem<nb_elem_vol; ind_elem++)
294 {
295 int elem = elem_list_vol[ind_elem];
296// Loop on faces of each element on the volume domain
297 for (int j=0, numfa; j < nfe && (numfa = elem_faces(elem,j)) >= 0; j++)
298 {
299// Test if the same face between surface domain and volum domain
300 int coincide=0;
301// Loop on vertex of each face of each element on the volume domain
302 for (int k=0; k<nse2D; k++)
303 {
304 int numso = face_sommets(numfa,k);
305 double xcoord_vol=coord_sommets(numso,0);
306 double ycoord_vol=coord_sommets(numso,1);
307// double zcoord_vol=coord_sommets(numso,2);
308 double zcoord_vol=0;
309 if (Objet_U::dimension>2)
310 zcoord_vol=coord_sommets(numso,2);
311
312 for (int i=0; i<nse2D ; i++)
313 {
314 int numso2D = zone_2D.sommet_elem(ind_face,i);
315 double xcoord_2D=coord_sommets_2D(numso2D,0);
316 double ycoord_2D=coord_sommets_2D(numso2D,1);
317 double zcoord_2D=0;
318 if (Objet_U::dimension>2)
319 zcoord_2D=coord_sommets_2D(numso2D,2);
320 if ( est_egal(xcoord_vol,xcoord_2D)
321 && est_egal(ycoord_vol,ycoord_2D)
322 && est_egal(zcoord_vol,zcoord_2D))
323 {
324 coincide=coincide+1;
325 break;
326 }
327 }
328 }
329 if (coincide==nse2D)
330 {
331 bool trouve=0;
332 for (int i=0; i<compteur ; i++)
333 if (les_faces[i]==numfa)
334 {
335 trouve=1;
336 break;
337 }
338 if (trouve==0) les_faces[compteur++] = numfa, face_tab(numfa) = 1;
339 }
340 }
341 }
342 }
343 }
344
345 int faces_found=mp_somme_vect(face_tab);
346 if (faces_found==0)
347 {
348 Cerr << "Error in Perte_Charge_Singuliere::lire_surfaces" << finl;
349 Cerr << "No faces has been found !" << finl;
350 Process::exit();
351 }
352 else
353 Cerr << " " << faces_found << " faces have been found for the section." << finl;
354
355 les_faces.resize(compteur);
356 if (orientation.non_nul())
357 {
358 sgn.resize(compteur);
359 DoubleTrav xvf(compteur, Objet_U::dimension), ori(compteur, Objet_U::dimension);
360 for (int i = 0; i < compteur; i++)
361 for (int j = 0; j < Objet_U::dimension; j++) xvf(i, j) = xv(les_faces(i), j);
362 orientation.valeur().valeur_aux(xvf, ori);
363 for (int i = 0; i < compteur; i++)
364 {
365 double scal = 0;
366 for (int j = 0; j < Objet_U::dimension; j++) scal += ori(i, j) * zvf.face_normales(les_faces(i), j);
367 sgn(i) = (scal > 0 ? 1 : -1);
368 }
369 }

372double Perte_Charge_Singuliere::calculate_Q(const Equation_base& eqn, const IntVect& num_faces, const IntVect& sgn) const
373{
374 const Zone_VF& zvf = ref_cast(Zone_VF, eqn.zone_dis().valeur());
375 const DoubleTab& vit = eqn.inconnue().valeurs(),
376 &fac = sub_type(Pb_Multiphase, eqn.probleme()) ? ref_cast(Pb_Multiphase, eqn.probleme()).eq_masse.champ_conserve().passe()
377 : eqn.probleme().get_champ("masse_volumique").valeurs(); //!< get_champ pour flica5 car la masse volumique n'est pas dans le milieu...
378 const DoubleVect& pf = zvf.porosite_face(), &fs = zvf.face_surfaces();
379 const IntTab& f_e = zvf.face_voisins();
380 int cF = fac.dimension_tot(0) == 1, i, n, N = fac.line_size(), d, D = Objet_U::dimension;
381
382 DoubleTrav deb_vect;
383 zvf.creer_tableau_faces(deb_vect); //pour bien sommer les debits en parallele
384 for (i = 0; i < num_faces.size(); i++)
385 {
386 int f = num_faces(i), e = f_e(f, f_e(f, 0) < 0); //todo : evaluer fac du bon cote
387 if (vit.line_size() > N)
388 for (d = 0; d < D; d++)
389 for (n = 0; n < N; n++) deb_vect(f) += zvf.face_normales(f, d) * vit(f, N * d + n) * fac(!cF * e, n); //vecteur complet : v . nf
390 else for (n = 0; n < N; n++) deb_vect(f) += fs(f) * vit(f, n) * fac(!cF * e, n); //normale aux faces seule
391 deb_vect(f) *= pf(f) * (sgn.size() ? sgn(i) : deb_vect(f) > 0 ? 1 : -1) ; //produit par la porosite + orientation (si elle n'est pas definie, on compte tout en positif)
392 }
393 return mp_somme_vect(deb_vect);
394}

396void Perte_Charge_Singuliere::update_K(const Equation_base& eqn, double deb, DoubleVect& bilan)
397{
398 if (!regul_) return;
399 double t = eqn.probleme().schema_temps().temps_courant(), dt = eqn.probleme().schema_temps().pas_de_temps();
400 deb_cible_.setVar(0, t), eps_.setVar(0, t);
401 double deb_cible = deb_cible_.eval(), eps = eps_.eval(), f_min = std::pow(1 - eps, dt), f_max = std::pow(1 + eps, dt); //bande de variation de K
402 K_ *= std::min(std::max(std::pow(std::fabs(deb) / deb_cible, 2), f_min), f_max);
403
404// pour le fichier de suivi : seulement sur le maitre, car Source_base::imprimer() fait une somme sur les procs
405 if (!Process::me()) bilan(0) = K_, bilan(1) = deb, bilan(2) = deb_cible;
406}
