43void Op_Evanescence_Homogene_PolyMAC_P0_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
44{
45 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
46 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0, equation().zone_dis().valeur());
47 const DoubleTab& inco = ch.valeurs();
48 const IntTab& fcl = ch.fcl();
49
50 /* on doit pouvoir ajouter / soustraire les equations entre composantes */
51 int i, j, l, e, f, n, N = inco.line_size(), d, D = dimension, nf_tot = zone.nb_faces_tot();
52 if (N == 1) return; //pas d'evanescence en simple phase!
53 for (auto &&n_m : matrices)
54 if (n_m.second->nb_colonnes())
55 {
56 IntTrav sten(0, 2);
57 sten.set_smart_resize(1);
58
59 std::set<int> idx;
60 Matrice_Morse& mat = *n_m.second, mat2;
61 /* equations aux faces : celles calculees seulement */
62 for (f = 0; f < zone.nb_faces(); f++, idx.clear())
63 if (fcl(f, 0) < 2)
64 {
65 for (i = N * f, n = 0; n < N; n++, i++)
66 for (j = mat.get_tab1()(i) - 1; j < mat.get_tab1()(i + 1) - 1; j++)
67 idx.insert(mat.get_tab2()(j) - 1);
68 for (i = N * f, n = 0; n < N; n++, i++)
69 for (auto &&c : idx) sten.append_line(i, c);
70 }
71 else if (n_m.first == ch.le_nom().getString()) /* CLs a valeur imposee : diagonale */
72 for (n = 0, i = N * f; n < N; n++, i++) sten.append_line(i, i);
73
74 /* equations aux elements */
75 for (e = 0, l = nf_tot; e < zone.nb_elem_tot(); e++)
76 for (d = 0; d < D; d++, l++, idx.clear())
77 {
78 for (i = N * l, n = 0; n < N; n++, i++)
79 for (j = mat.get_tab1()(i) - 1; j < mat.get_tab1()(i + 1) - 1; j++)
80 idx.insert(mat.get_tab2()(j) - 1);
81 for (i = N * l, n = 0; n < N; n++, i++)
82 for (auto &&c : idx) sten.append_line(i, c);
83 }
84 Matrix_tools::allocate_morse_matrix(mat.nb_lignes(), mat.nb_colonnes(), sten, mat2);
85 mat = mat2; //pour forcer l'ordre des coefficients dans la matrice (accelere les operations ligne a ligne)
86 }

98 iter = sub_type(SETS, equation().schema_temps()) ? 0 * ref_cast(SETS, equation().schema_temps()).iteration : 0;
99 if (N == 1) return; //pas d'evanescence en simple phase!
100
101 double a_eps = alpha_res_, a_eps_min = alpha_res_min_, a_m, a_max; //seuil de declenchement du traitement de l'evanescence
102
103 /* recherche de phases evanescentes et traitement des seconds membres */
104 IntTrav maj(inco.dimension_tot(0)); //maj(i) : phase majoritaire de la ligne i
105 DoubleTrav coeff(inco.dimension_tot(0), inco.line_size(), 2); //coeff(i, n, 0/1) : coeff a appliquer a l'equation existante / a l'eq. "inco = v_maj"
106 Matrice_Morse& mat_diag = *matrices.at(ch.le_nom().getString());
107 for (f = 0; f < zone.nb_faces(); f++)
108 if (fcl(f, 0) < 2)
109 {
110 /* phase majoritaire : avec alpha interpole par defaut, avec alpha amont pour les ierations de SETS / ICE */
111 for (a_max = 0, k = -1, n = 0; n < N; n++)
112 {
113 if (iter) a_m = alpha(f_e(f, f_e(f, 1) >= 0 && inco(f, n) < 0), n);
114 else for (a_m = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++) a_m += vfd(f, i) / vf(f) * alpha(e, n);
115 if (a_m > a_max) k = n, a_max = a_m;
116 }
117 if (k >= 0) maj(f) = k;
118 else abort();
119
120 /* phases evanescentes : avec alpha amont. La phase majoritaire ne peut pas etre evanescente! */
121 for (n = 0; n < N; n++)
122 {
123 if (iter) a_m = alpha(f_e(f, f_e(f, 1) >= 0 && inco(f, n) < 0), n);
124 else for (a_m = 0, i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++) a_m += vfd(f, i) / vf(f) * alpha(e, n);
125 if (n != k && a_m < a_eps)
126 {
127 coeff(f, n, 1) = mat_diag(N * f + k, N * f + k) * (coeff(f, n, 0) = std::min(std::max((a_eps - a_m) / (a_eps - a_eps_min), 0.), 1.));
128 double flux = coeff(f, n, 0) * secmem(f, n) + coeff(f, n, 1) * (inco(f, n) - inco(f, k));
129 secmem(f, k) += flux, secmem(f, n) -= flux;
130 }
131 }
132
133 }
134
135 for (e = 0; e < zone.nb_elem_tot(); e++) /* elements : a faire D fois par element */
136 {
137 /* phase majoritaire : directement dans l'element */
138 for (a_max = 0, k = -1, n = 0; n < N; n++)
139 if ((a_m = alpha(e, n)) > a_max) k = n, a_max = a_m;
140 if (k >= 0)
141 for (i = nf_tot + D * e, d = 0; d < D; d++, i++) maj(i) = k;
142 else abort();
143
144 /* coeff d'evanescence */
145 for (i = nf_tot + D * e, d = 0; d < D; d++, i++)
146 for (n = 0; n < N; n++)
147 if (n != k && (a_m = alpha(e, n)) < a_eps)
148 {
149 coeff(i, n, 1) = mat_diag(N * i + k, N * i + k) * (coeff(i, n, 0) = std::min(std::max((a_eps - a_m) / (a_eps - a_eps_min), 0.), 1.));
150 double flux = coeff(i, n, 0) * secmem(i, n) + coeff(i, n, 1) * (inco(i, n) - inco(i, k));
151 secmem(i, k) += flux, secmem(i, n) -= flux;
152 }
153 }
154
155 /* lignes de matrices */
156 for (auto &&n_m : matrices)
157 if (n_m.second->nb_colonnes())
158 {
159 int diag = (n_m.first == ch.le_nom().getString()); //est-on sur le bloc diagonal?
160 Matrice_Morse& mat = *n_m.second;
161 /* faces */
162 for (f = 0; f < zone.nb_faces(); f++)
163 if (fcl(f, 0) < 2)
164 for (n = 0; n < N; n++)
165 if (coeff(f, n, 0))
166 for (k = maj(f), i = mat.get_tab1()(N * f + n) - 1, j = mat.get_tab1()(N * f + k) - 1; i < mat.get_tab1()(N * f + n + 1) - 1; i++, j++)
167 {
168 assert(mat.get_tab2()(i) == mat.get_tab2()(j));
169 int c = diag * mat.get_tab2()(i) - 1; //indice de colonne (commun aux deux lignes grace au dimensionner_blocs())
170 mat.get_set_coeff()(j) += coeff(f, n, 0) * mat.get_set_coeff()(i) - coeff(f, n, 1) * ((c == N * f + n) - (c == N * f + k));
171 mat.get_set_coeff()(i) += -coeff(f, n, 0) * mat.get_set_coeff()(i) + coeff(f, n, 1) * ((c == N * f + n) - (c == N * f + k));
172 }
173 /* elements : l est l'indice de ligne */
174 for (e = 0, l = nf_tot; e < zone.nb_elem_tot(); e++)
175 for (d = 0; d < D; d++, l++)
176 for (n = 0; n < N; n++)
177 if (coeff(l, n, 0))
178 for (k = maj(l), i = mat.get_tab1()(N * l + n) - 1, j = mat.get_tab1()(N * l + k) - 1; i < mat.get_tab1()(N * l + n + 1) - 1; i++, j++)
179 {
