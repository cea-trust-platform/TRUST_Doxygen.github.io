36void Champ_Generique_Correlation::completer(const Postraitement_base& post)
37{
38 Champ_Gen_de_Champs_Gen::completer(post);
39 const Probleme_base& Pb = get_ref_pb_base();
40
41 Noms nom = get_source(0).get_property("nom");
42 Motcle nom_champ = nom[0];
43 Noms nom2 = get_source(1).get_property("nom");
44 Motcle nom_champ2 = nom2[0];
45
46 const Champ_Generique_base& ch_integre = get_source(0);
47 const Champ_Generique_base& ch_integre2 = get_source(1);
48 const Zone_dis_base& zdis = get_ref_zone_dis_base();
49
50 Op_Correlation_.associer(zdis,ch_integre,ch_integre2,tstat_deb_,tstat_fin_);
51 const Postraitement& mon_post = ref_cast(Postraitement,post);
52
53 Motcle nom_champ_moyenne("Moyenne_");
54 nom_champ_moyenne += nom_champ;
55 nom_champ_moyenne += "_";
56 nom_champ_moyenne += Nom(tstat_deb_,"%e");
57 nom_champ_moyenne += "_";
58 nom_champ_moyenne += Nom(tstat_fin_,"%e");
59
60 Motcle nom_champ_moyenne2("Moyenne_");
61 nom_champ_moyenne2 += nom_champ2;
62 nom_champ_moyenne2 += "_";
63 nom_champ_moyenne2 += Nom(tstat_deb_,"%e");
64 nom_champ_moyenne2 += "_";
65 nom_champ_moyenne2 += Nom(tstat_fin_,"%e");
66
67 if ((mon_post.comprend_champ_post(nom_champ_moyenne)) && (mon_post.comprend_champ_post(nom_champ_moyenne2)))
68 {
69 const Champ_Generique_base& champ_stat_base = mon_post.get_champ_post(nom_champ_moyenne);
70 const Operateur_Statistique_tps_base& operateur = ref_cast(Champ_Generique_Moyenne,champ_stat_base).Operateur_Statistique();
71 const Op_Moyenne& op_moyenne = ref_cast(Op_Moyenne,operateur);
72 Op_Correlation_.associer_op_stat(op_moyenne);
73
74 const Champ_Generique_base& champ_stat_base2 = mon_post.get_champ_post(nom_champ_moyenne2);
75 const Operateur_Statistique_tps_base& operateur2 = ref_cast(Champ_Generique_Moyenne,champ_stat_base2).Operateur_Statistique();
76 const Op_Moyenne& op_moyenne2 = ref_cast(Op_Moyenne,operateur2);
77 Op_Correlation_.associer_op_stat(op_moyenne2);
78 }
79 else
80 {
81 Cerr<<"The specification of the post-processing correlation field "<<finl;
82 Cerr<<get_property("nom")[0]<<finl;
83 Cerr<<"implies to specify before the corresponding "<<finl;
84 Cerr<<"post-processing statistics average fields."<<finl;
85 Cerr<<"Please, modify your data set to specify the post-processing statistics"<<finl;
86 Cerr<<"average fields before (instead of after) the post-processing correlation field"<<finl;
87 Cerr<<get_property("nom")[0]<<finl;
88 exit();
89 }
90 Op_Correlation_.completer(Pb);
91}

93const Champ_base& Champ_Generique_Correlation::get_champ(Champ& espace_stockage) const
94{
95 int nb_comp = integrale().valeur().nb_comp();
96// Nature_du_champ nature_source = source.nature_du_champ();
97// Pas completement exact car il y a le cas de la correlation vecteur-vecteur et dans
98// ce cas c est un tenseur qui est manipule (la nature n est pas scalaire ou vectorielle)
99 Nature_du_champ nature_source = (nb_comp==1)?scalaire:vectoriel;
100 Champ_Fonc es_tmp;
101 espace_stockage = creer_espace_stockage(nature_source,nb_comp,es_tmp);
102
103 DoubleTab& tab_correlation = espace_stockage.valeurs();
104 tab_correlation = Op_Correlation_.calculer_valeurs();
105 tab_correlation.echange_espace_virtuel();
106 return espace_stockage.valeur();
107}

169void Champ_Generique_Correlation::nommer_source()
170{
171 if (nom_post_=="??")
172 {
173 Nom nom_post_source, nom_champ_source_1, nom_champ_source_2;
174 const Noms nom_1 = get_source(0).get_property("nom");
175 const Noms nom_2 = get_source(1).get_property("nom");
176 nom_champ_source_1 = nom_1[0];
177 nom_champ_source_2 = nom_2[0];
178
179 nom_post_source = "Correlation_";
180 nom_post_source += nom_champ_source_1;
181 nom_post_source += "_";
182 nom_post_source += nom_champ_source_2;
183 nommer(nom_post_source);
184 }
185}
