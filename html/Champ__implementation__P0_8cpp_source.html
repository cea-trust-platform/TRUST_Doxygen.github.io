63DoubleTab& Champ_implementation_P0::valeur_aux_elems(const DoubleTab& positions, const IntVect& polys, DoubleTab& result) const
64{
65 const Champ_base& ch_base = le_champ();
66 int nb_components = ch_base.nb_comp();
67 const DoubleTab& values = ch_base.valeurs();
68 int nb_polys = polys.size();
69
70 if (nb_polys==0)
71 return result;
72
73// TODO : FIXME
74// For FT the resize should be done in its good position and not here ...
75 if (result.nb_dim() == 1 ) result.resize(nb_polys,1);
76
77 assert(values.line_size() == nb_components);
78 assert(result.line_size() == nb_components || nb_components == 1);
79
80 for (int i=0; i<nb_polys; i++)
81 {
82 int cell = polys(i);
83 assert(cell<values.dimension_tot(0));
84
85 for (int j = 0; j < result.line_size(); j++)
86 if (cell != -1)
87 result(i, j) = values.nb_dim() == 1 ? values(cell) : values(cell, (result.line_size() == nb_components) * j); //!< Some post-processed fields can have nb_dim() == 1
88 }
89
90 return result;
91}

138DoubleTab& Champ_implementation_P0::valeur_aux_sommets_impl(DoubleTab& result) const
139{
140 const Champ_base& ch_base = le_champ();
141 int nb_components = ch_base.nb_comp();
142 const DoubleTab& values = ch_base.valeurs();
143
144 const Zone& zone = get_zone_geom();
145 int nb_cells = zone.nb_elem_tot();
146 int nb_nodes = zone.nb_som();
147 int nb_nodes_per_cell = zone.nb_som_elem();
148
149 ArrOfInt count(nb_nodes);
150
151 assert((result.dimension_tot(0) == nb_nodes)||(result.dimension(0) == nb_nodes));
152
153 result = 0;
154 count = 0;
155
156 assert(result.line_size() == nb_components);
157 assert(values.line_size() == nb_components);
158
159 for (int i=0; i<nb_cells; i++)
160 for (int j=0; j<nb_nodes_per_cell; j++)
161 {
162 int node = zone.sommet_elem(i,j);
163 if (node < nb_nodes)
164 {
165 count[node]++;
166 if ( values.nb_dim() == 1 ) //!< TODO : FIXME : champ porosite doit etre un DoubleTab
167 for (int k=0; k<nb_components; k++) result(node,k) += values(i);
168 else
169 for (int k=0; k<nb_components; k++) result(node,k) += values(i,k);
170 }
171 }
172 for (int i=0; i<nb_nodes; i++)
173 for (int j=0; j<nb_components; j++) result(i,j) /= count[i];
174
175 return result;
176}

178DoubleVect& Champ_implementation_P0::valeur_aux_sommets_compo_impl(DoubleVect& result, int ncomp) const
179{
180 const Champ_base& ch_base = le_champ();
181 const DoubleTab& values = ch_base.valeurs();
182
183 const Zone& zone = get_zone_geom();
184 int nb_cells = zone.nb_elem_tot();
185 int nb_nodes = zone.nb_som();
186 int nb_nodes_per_cell = zone.nb_som_elem();
187
188 ArrOfInt count(nb_nodes);
189// dvq ajout result=0;
190 result=0;
191 assert(ncomp>=0);
192 assert(ncomp<ch_base.nb_comp());
193 assert(result.size() == nb_nodes);
194 assert(values.line_size() == ch_base.nb_comp());
195
196 for (int i=0; i<nb_cells; i++)
197 for (int j=0; j<nb_nodes_per_cell; j++)
198 {
199 int node = zone.sommet_elem(i,j);
200 if (node < nb_nodes)
201 {
202 count[node]++;
203 result(node) += values(i,ncomp);
204 }
205 }
206 for (int i=0; i<nb_nodes; i++)
207 result(i) /= count[i];
208
209 return result;
210}
