51//
52//! on ajoute la contribution du second membre.
53void Op_Diff_VDF_base::contribuer_au_second_membre(DoubleTab& resu) const
54{
55 iter.contribuer_au_second_membre(resu);
56 if (equation().domaine_application() == Motcle("Hydraulique"))
57// On est dans le cas des equations de Navier_Stokes
58 {
59// Ajout du terme du au terme supplementaire en V/(R*R)
60// dans le cas des coordonnees axisymetriques
61 if(Objet_U::bidim_axi == 1)
62 {
63 const Zone_VDF& zvdf=iter.zone();
64 const Zone_Cl_VDF& zclvdf = iter.zone_Cl();
65 const DoubleTab& xv=zvdf.xv();
66 const IntVect& ori=zvdf.orientation();
67 const IntTab& face_voisins=zvdf.face_voisins();
68 const DoubleVect& volumes_entrelaces=zvdf.volumes_entrelaces();
69// int nb_faces=zvdf.nb_faces();
70 DoubleTrav diffu_tot(zvdf.nb_elem_tot());
71 double db_diffusivite;
72
73 if (equation().que_suis_je() == "Navier_Stokes_standard")
74 {
75 const Eval_Diff_VDF& eval=dynamic_cast<const Eval_Diff_VDF&> (iter->evaluateur());
76 const Champ_base& ch_diff=eval.get_diffusivite();
77 const DoubleTab& tab_diffusivite=ch_diff.valeurs();
78 if (tab_diffusivite.size() == 1)
79 diffu_tot = tab_diffusivite(0,0);
80 else
81 diffu_tot = tab_diffusivite;
82 }
83 else
84 {
85 Cerr << "Probleme dans Op_Diff_VDF_base::contribuer_au_second_membre avec le type de l'equation" << finl;
86 Cerr << "on n'a pas prevu d'autre cas que Navier_Stokes_std" << finl;
87 exit();
88 }
89
90// Boucle sur les conditions limites pour traiter les faces de bord
91
92 int ndeb,nfin,num_face;
93
94 for (int n_bord=0; n_bord<zvdf.nb_front_Cl(); n_bord++)
95 {
96 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
97
98 if (sub_type(Dirichlet,la_cl.valeur()))
99 {
100 const Dirichlet& la_cl_diri = ref_cast(Dirichlet,la_cl.valeur());
101 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
102 ndeb = le_bord.num_premiere_face();
103 nfin = ndeb + le_bord.nb_faces();
104 double r_2,coef;
105 for (num_face=ndeb; num_face<nfin; num_face++)
106 {
107 if (ori(num_face) == 0)
108 {
109 int elem1=face_voisins(num_face,0);
110 int elem2=face_voisins(num_face,1);
111 if(elem1==-1)
112 db_diffusivite=diffu_tot(elem2);
113 else
114 {
115 assert (elem2==-1);
116 db_diffusivite=diffu_tot(elem1);
117 }
118 r_2 = xv(num_face,0)*xv(num_face,0);
119 coef = volumes_entrelaces(num_face)/r_2;
120 resu(num_face) -= (la_cl_diri.val_imp(num_face-ndeb)*coef)*db_diffusivite;
121 }
122 }
123 }
124 }
125 }
126 }
127}

137void Op_Diff_VDF_base::init_op_ext() const
138{
139 const Zone_VDF& zvdf = iter.zone();
140 const Zone_Cl_VDF& zclvdf = iter.zone_Cl();
141 op_ext = { this };//le premier op_ext est l'operateur local
142
143 for (int n_bord=0; n_bord<zvdf.nb_front_Cl(); n_bord++)
144 {
145 const Cond_lim& la_cl = zclvdf.les_conditions_limites(n_bord);
146 if (sub_type(Echange_contact_VDF,la_cl.valeur()))
147 {
148 const Echange_contact_VDF& cl = ref_cast(Echange_contact_VDF, la_cl.valeur());
149 const Champ_front_calc& ch=ref_cast(Champ_front_calc, cl.T_autre_pb().valeur());
150 const Equation_base& o_eqn = ch.equation();
151 const Op_Diff_VDF_base *o_diff = &ref_cast(Op_Diff_VDF_base, o_eqn.operateur(0).l_op_base());
152
153 if (std::find(op_ext.begin(), op_ext.end(), o_diff) == op_ext.end()) op_ext.push_back(o_diff);
154 }
155 }
156 op_ext_init_ = 1;
157}
