55double Sortie_libre_Gradient_Pression_impose_VEF::flux_impose(int face) const
56{
57 if (le_champ_front.valeurs().size()==1)
58 {
59 double a1 = trace_pression_int[face];
60 double a2 = coeff[face];
61 double a3 = le_champ_front(0,0);
62 double Pimp = a1 + a2*a3;
63 return Pimp ;
64 }
65 else if (le_champ_front.valeurs().line_size()==1)
66 {
67 return (trace_pression_int[face] + coeff[face]*le_champ_front(face,0));
68 }
69 else
70 Cerr << "Sortie_libre_Gradient_Pression_impose_VEF::flux_impose erreur" << finl;
71 exit();
72 return 0.;
73}
