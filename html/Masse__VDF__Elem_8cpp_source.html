45DoubleTab& Masse_VDF_Elem::appliquer_impl(DoubleTab& sm) const
46{
47 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
48 const DoubleVect& volumes = zone_VDF.volumes();
49 const DoubleVect& porosite_elem = zone_VDF.porosite_elem();
50 int nb_elem = zone_VDF.nb_elem();
51 if(nb_elem==0)
52 {
53 sm.echange_espace_virtuel();
54 return sm;
55 }
56 int nb_comp = sm.size()/nb_elem;
57 int nb_dim=sm.nb_dim();
58 assert((nb_comp*nb_elem == sm.size())||(nb_dim==3));
59 if (nb_dim == 1)
60 for (int num_elem=0; num_elem<nb_elem; num_elem++)
61 sm(num_elem) /= (volumes(num_elem)*porosite_elem(num_elem));
62 else if (nb_dim == 2)
63 {
64 for (int num_elem=0; num_elem<nb_elem; num_elem++)
65 for (int k=0; k<nb_comp; k++)
66 sm(num_elem,k) /= (volumes(num_elem)*porosite_elem(num_elem));
67 }
68 else if (sm.nb_dim() == 3)
69 {
70// int d0=sm.dimension(0);
71 int d1=sm.dimension(1);
72 int d2=sm.dimension(2);
73 for (int num_elem=0; num_elem<nb_elem; num_elem++)
74 for (int k=0; k<d1; k++)
75 for (int d=0; d<d2; d++)
76 sm(num_elem,k,d) /= (volumes(num_elem)*porosite_elem(num_elem));
77 }
78 else
79 {
80 Cerr<< "Masse_VDF_Elem::appliquer ne peut pas s'appliquer a un DoubleTab a "<<sm.nb_dim()<<" dimensions"<<finl;
81 assert(0);
82 exit();
83 }
84 sm.echange_espace_virtuel();
85 return sm;
86}
