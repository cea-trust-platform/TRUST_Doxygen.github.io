38void Ecrire_Fichier::interpreter_(Entree& is, int format_binaire)
39{
40 if (nproc() > 1)
41 {
42 Cerr << "Error: keyword Ecrire_Fichier_Bin cannot be used in parallel calculation. What do you mean ?" << finl;
43 barrier();
44 exit();
45 }
46 SFichier fic;
47 Nom nom1, nom2;
48 is >> nom1 >> nom2;
49 Cerr << "Ecrire_Fichier_Bin: writing object " << nom1 << " in file " << nom2;
50 if (format_binaire)
51 Cerr << " in binary format." << finl;
52 else
53 Cerr << " in ascii format." << finl;
54 Objet_U& obj = objet(nom1);
55 fic.set_bin(format_binaire);
56 if (!fic.ouvrir(nom2))
57 {
58 Cerr << " Error opening file: " << nom2 << finl;
59 exit();
60 }
61 if (!format_binaire)
62 {
63 fic.get_ofstream().setf(ios::scientific);
64 fic.get_ofstream().precision(8);
65 }
66 fic << obj;
67 if (!fic.get_ostream().good())
68 {
69 Cerr << " An error occured while writing the file." << finl;
70 exit();
71 }
72}
