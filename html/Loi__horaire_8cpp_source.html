122// Produit Matrice*Vecteur stockes dans des tableaux
123inline void multiplie(const DoubleTab& matrice, ArrOfDouble& vecteur)
124{
125 check(matrice);
126 ArrOfDouble x(vecteur);
127 vecteur=0;
128 int dimension = vecteur.size_array();
129 for (int i=0; i<dimension; i++)
130 for (int j=0; j<dimension; j++)
131 vecteur[i] += matrice(0,i*dimension+j) * x[j];
132}

134// Produit Inverse(Matrice)*Vecteur dans des tableaux
135inline void inverse(const DoubleTab& mat, ArrOfDouble& x)
136{
137 check(mat);
138 int dimension = x.size_array();
139 if (dimension==2)
140 {
141 const double a=mat(0,0);
142 const double b=mat(0,1);
143 const double c=mat(0,2);
144 const double d=mat(0,3);
145 double det=a*d-b*c;
146 if (det==0)
147 {
148 Cerr << "Matrix of order 2 not invertible in Loi_horaire::inverse" << finl;
149 Cerr << mat << finl;
150 Process::exit();
151 }
152 double y0=(d*x[0]-b*x[1])/det;
153 double y1=(-c*x[0]+a*x[1])/det;
154 x[0]=y0;
155 x[1]=y1;
156 }
157 else if (dimension==3)
158 {
159 const double a=mat(0,0);
160 const double b=mat(0,1);
161 const double c=mat(0,2);
162 const double d=mat(0,3);
163 const double e=mat(0,4);
164 const double f=mat(0,5);
165 const double g=mat(0,6);
166 const double h=mat(0,7);
167 const double i=mat(0,8);
168
169 double det=a*(e*i-f*h)
170 -b*(d*i-f*g)
171 +c*(d*h-e*g);
172 if (det==0)
173 {
174 Cerr << "Matrix of order 3 not invertible in Loi_horaire::inverse" << finl;
175 Cerr << mat << finl;
176 Process::exit();
177 }
178 double y0=((e*i-f*h)*x[0]
179 +(c*h-b*i)*x[1]
180 +(b*f-c*e)*x[2])/det;
181 double y1=((f*g-d*i)*x[0]
182 +(a*i-c*g)*x[1]
183 +(c*d-a*f)*x[2])/det;
184 double y2=((d*h-e*g)*x[0]
185 +(b*g-a*h)*x[1]
186 +(a*e-b*d)*x[2])/det;
187 x[0]=y0;
188 x[1]=y1;
189 x[2]=y2;
190 }
191 else
192 {
193 Cerr << "Case not provided in Loi_horaire::inverse()" << finl;
194 Process::exit();
195 }
196}

198/* Renvoie un tableau contenant la position xM(t) d'un point M au temps t
199 en fonction du temps t0 et de sa position xM(t0) au temps t0.
200 Pour cela, on utilisera les relations suivantes:
201 xM(t)=xG(t)+R(t)(xM(0)-xG(0))
202 xM(t0)=xG(t0)+R(t0)(xM(0)-xG(0))
203 On a:
204 xM(0)-xG(0)=R-1(t0)(xM(t0)-xG(t0))
205 Donc:
206 xM(t) = xG(t)+R(t)R-1(t0)(xM(t0)-xG(t0))
207*/
208ArrOfDouble Loi_horaire::position(const double t, const double t0, const ArrOfDouble& xMt0)
209{
210 ArrOfDouble xMt(xMt0); //!< xM(t0)
211 position_.me_calculer(t0); //!< xG(t0)
212 xMt -= position_.valeurs(); //!< xM(t0)-xG(t0)
213 rotation_.me_calculer(t0); //!< R(t0)
214 inverse(rotation_.valeurs(),xMt); //!< R-1(t0)(xM(t0)-xG(t0))
215 rotation_.me_calculer(t); //!< R(t)
216 multiplie(rotation_.valeurs(),xMt); //!< R(t)R-1(t0)(xM(t0)-xG(t0))
217 position_.me_calculer(t); //!< xG(t)
218 xMt += position_.valeurs(); //!< xM(t) = xG(t)+R(t)R-1(t0)(xM(t0)-xG(t0))
219 return xMt;
220}
221
222/* Renvoie un tableau contenant la vitesse vM(t) d'un point M au temps t
223 en fonction de sa position xM(t) au temps t. Pour connaitre cette vitesse du point M,
224 on utilise les relations suivantes:
225 xM(t)=xG(t)+R(t)(xM(0)-xG(0))
226 vM(t)=vG(t)+R'(t)(xM(0)-xG(0))
227 On a:
228 xM(0)-xG(0)=R-1(t)(xM(t)-xG(t))
229 Donc:
230 vM(t) = vG(t)+R'(t)R-1(t)(xM(t)-xG(t))
231*/
232ArrOfDouble Loi_horaire::vitesse(const double t, const ArrOfDouble& xMt)
233{
234 ArrOfDouble vMt(xMt); //!< xM(t)
235 position_.me_calculer(t); //!< xG(t)
236 vMt -= position_.valeurs(); //!< xM(t)-xG(t)
237 rotation_.me_calculer(t); //!< R(t)
238 inverse(rotation_.valeurs(),vMt); //!< R-1(t)(xM(t)-xG(t))
239 derivee_rotation_.me_calculer(t); //!< R'(t)
240 multiplie(derivee_rotation_.valeurs(),vMt); //!< R'(t)R-1(t)(xM(t)-xG(t))
241 vitesse_.me_calculer(t); //!< vG(t)
242 vMt += vitesse_.valeurs(); //!< vM(t) = vG(t)+R'(t)R-1(t)(xM(t)-xG(t))
243 return vMt;
244}

246// Verification de la coherence des derivees vitesse et derivee_rotation
247// par rapport a position et rotation. On utilise un developpement limite d'ordre 2
248// |f(t+dt)-f(t)-dt*f'(t)|=|0.5*f''(t)*dt*dt+O(dt*dt*dt)|>|10*f''(t)*dt*dt|
249// Une erreur sur des expressions de derivee seconde nulle ne sera donc pas detectee...
250void Loi_horaire::verifier_derivee(const double t)
251{
252 if (verification_derivee_)
253 {
254 double dt=0.001*t;
255 DoubleVect err_t(position_.valeurs());
256 position_.me_calculer(t+dt); //!< xG(t+dt)
257 err_t=position_.valeurs(); //!< xG(t+dt)
258 position_.me_calculer(t); //!< xG(t)
259 err_t-=position_.valeurs(); //!< xG(t+dt)-xG(t)
260 vitesse_.me_calculer(t); //!< vG(t)
261 err_t.ajoute(-dt, vitesse_.valeurs()); //!< xG(t+dt)-xG(t)-dt*vG(t) Evaluation d'un seuil avec la derivee seconde xG'' cad vG'
262//
263 ArrOfDouble seuil_t(position_.valeurs());
264 vitesse_.me_calculer(t+dt); //!< vG(t+dt)
265 seuil_t=vitesse_.valeurs();
266 vitesse_.me_calculer(t); //!< vG(t)
267 seuil_t-=vitesse_.valeurs();
268 seuil_t*=100*dt; //!< ~vG'(t)*dt*dt
269 int n=err_t.size_array();
270 for (int i=0; i<n; i++)
271 {
272 double seuil=std::fabs(seuil_t[i]);
273 if (!est_egal(seuil,0) && std::fabs(err_t(i))>seuil)
274 {
275 if (Process::je_suis_maitre())
276 {
277 Cerr << "At time " << t << ", inconsistency in the schedule law " << le_nom() << finl;
278 Cerr << "The expression of the component " << i << " of velocity does not appear" << finl;
279 Cerr << "to be the time derivative of the position expression." << finl;
280 Cerr << "Verify the expressions of this schedule law in your data set." << finl;
281 Cerr << "If the expressions are correct after all, add the option 'verification_derivee 0'" << finl;
282 Cerr << "in the law to not to make this verification." << finl;
283 }
284 Process::exit();
285 }
286 }
287 DoubleTab err_r(rotation_.valeurs());
288 rotation_.me_calculer(t+dt); //!< R(t+dt)
289 err_r=rotation_.valeurs(); //!< R(t+dt)
290 rotation_.me_calculer(t); //!< R(t)
291 err_r-=rotation_.valeurs(); //!< R(t+dt)-R(t)
292 derivee_rotation_.me_calculer(t); //!< R'(t)
293 err_r.ajoute(-dt, derivee_rotation_.valeurs()); //!< R(t+dt)-R(t)-dt*R'(t) Evaluation d'un seuil avec la derivee seconde de R cad R''
294//
295 DoubleTab seuil_r(rotation_.valeurs());
296 derivee_rotation_.me_calculer(t+dt); //!< R'(t+dt)
297 seuil_r=derivee_rotation_.valeurs();
298 derivee_rotation_.me_calculer(t); //!< R'(t)
299 seuil_r-=derivee_rotation_.valeurs(); //!< R'(t+dt)-R'(t)
300 seuil_r*=100*dt; //!< ~R''(t)*dt*dt
301 int ni=err_r.dimension(0);
302 int nj=err_r.dimension(1);
303 for (int i=0; i<ni; i++)
304 for (int j=0; j<nj; j++)
305 {
306 double seuil=std::fabs(seuil_r(i,j));
307 if (!est_egal(seuil,0) && std::fabs(err_r(i,j))>seuil)
308 {
309 if (Process::je_suis_maitre())
310 {
311 Cerr << "At time " << t << ", inconsistency in the schedule law " << le_nom() << finl;
312 Cerr << "The expression of the component " << i*ni+j << " of derivee_rotation does not appear" << finl;
313 Cerr << "to be the time derivative of the rotation expression." << finl;
314 Cerr << "Verify the expressions of this schedule law in your data set" << finl;
315 Cerr << "or add the option 'verification_derivee 0' in this law." << finl;
316 }
317 Process::exit();
318 }
319 }

374// Tests divers de la classe
375void Loi_horaire::tester(const Schema_Temps_base& sch)
376{
377// Test de produit et d'inversion de matrice
378 for (int dim=2; dim<=3; dim++)
379 {
380 DoubleTab mat(1,dim*dim);
381 for (int i=0; i<dim; i++)
382 for (int j=0; j<dim; j++)
383 mat(0,i*dim+j)=(i>j?1+i+j:-2*j);
384 ArrOfDouble vect(dim);
385 for (int i=0; i<dim; i++)
386 vect[i]=i;
387 ArrOfDouble tmp(vect);
388 multiplie(mat,vect);
389 inverse(mat,vect);
390 for (int i=0; i<dim; i++)
391 if (!est_egal(tmp[i],vect[i]))
392 {
393 Cerr << "Loi_horaire::inverse is not validated for dim=" << dim << finl;
394 Cerr << "A=" << mat << finl;
395 Cerr << "x=" << tmp << finl;
396 Cerr << "Inv(A)Ax=" << vect << finl;
397 Cerr << "Contact TRUST support." << finl;
398 Process::exit();
399 }
400 }
401// Verification entre tinit et tmax
402 double tinit=sch.temps_init();
403 double tmax=sch.temps_max();
404// Test des derivees:
405 int n=100;
406 for (int i=1; i<n; i++)
407 {
408 double temps=tinit+(tmax-tinit)*(i+1)/n;
409// Cerr << "Verification a t=" << temps << finl;
410 verifier_derivee(temps);
411 }
412// Test de position et vitesse:
413 double t=0.5*(tinit+tmax);
414 double eps=1e-8;
415 double dt=eps*t;
416 ArrOfDouble pos(dimension);
417 pos[0]=dimension*2;
418 pos[1]=dimension*3;
419 if (dimension==3) pos[2]=-dimension;
420// Verification que position(t,t,pos)=pos
421 ArrOfDouble tmp(position(t,t,pos));
422 for (int i=0; i<dimension; i++)
423 if (!est_egal(tmp[i],pos[i]))
424 {
425 Cerr << "Loi_horaire::position is not validated for pos=position(t,t,pos):" << finl;
426 Cerr << pos << finl;
427 Cerr << tmp << finl;
428 Cerr << "Contact TRUST support." << finl;
429 Process::exit();
430 }
431// Verification que vitesse(t,pos)~(position(t+dt,t,pos)-pos)/dt;
432 tmp=position(t+dt,t,pos);
433 tmp-=pos;
434 tmp/=dt;
435 ArrOfDouble vit(vitesse(t,pos));
436 ArrOfDouble relative_error(vit);
437 relative_error-=tmp;
438 relative_error/=(norme_array(tmp)!=0?norme_array(tmp):1);
439 for (int i=0; i<dimension; i++)
440 if (!est_egal(relative_error[i],0,0.001))
441 {
442 Cerr << "Loi_horaire::position is not validated for velocity(t,pos):" << finl;
443 Cerr << "t=" << t << finl;
444 Cerr << "dt=" << dt << finl;
445 Cerr << "vit=" << vit << finl;
446 Cerr << "tmp=" << tmp << finl;
447 Cerr << "relative_error=" << relative_error << finl;
448 Cerr << "Contact TRUST support." << finl;
449 Process::exit();
450 }
451 Cerr << "Loi_horaire::tester() OK" << finl;
452}
