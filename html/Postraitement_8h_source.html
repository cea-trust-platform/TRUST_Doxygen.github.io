241inline int Postraitement::lpost(double temps_courant, double dt_post) const
242{
243 double epsilon = 1.e-8;
244 if (dt_post<=temps_courant - dernier_temps)
245 return 1;
246 else
247 {
248// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
249 double i, j;
250 modf(temps_courant/dt_post + epsilon, &i);
251 modf(dernier_temps/dt_post + epsilon, &j);
252 return ( i>j );
253 }
254}
255
256// Precondition:
257// Parametre: double temps_courant
258// Signification: le temps courant
259// Valeurs par defaut:
260// Contraintes:
261// Acces:
262// Parametre: double dt
263// Signification: le pas de temps qui vient d'etre accompli
264// Valeurs par defaut:
265// Contraintes:
266// Acces:
267// Exception:
268// Effets de bord:
269// Postcondition: la methode ne modifie pas l'objet
270//
271//
272//
273//
274//! Test de postraitement en tenant compte de l'evolution en temps<br>du champ.<br>Renvoie VRAI si le champ necessite un postraitement, vu<br>le temps courant et le pas de temps fourni.
275/** \return int<br>
276Signification: valeur booleenne, VRAI si le pas de temp et le temps<br>
277courant fournis indique qu'un postraitement est<br>
278necessaire, FAUX sinon.<br>
279Contraintes: */
280inline int Postraitement::lpost_champ(double temps_courant) const { return lpost(temps_courant, dt_post_ch_); }
