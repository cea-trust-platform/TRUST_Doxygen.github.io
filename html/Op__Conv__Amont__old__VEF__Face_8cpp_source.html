43static void convbis(const double psc,const int num1,const int num2,
44 const DoubleTab& transporte,const int ncomp,
45 DoubleTab& resu, DoubleVect& fluent)
46{
47 int comp,amont;
48 double flux;
49
50 if (psc >= 0)
51 {
52 amont = num1;
53 fluent[num2] += psc;
54 }
55 else
56 {
57 amont = num2;
58 fluent[num1] -= psc;
59 }
60
61 for (comp=0; comp<ncomp; comp++)
62 {
63 flux = transporte(amont,comp)*psc;
64 resu(num1,comp) -= flux;
65 resu(num2,comp) += flux;
66 }
67}
68static void convbisimplicite(const double psc,const int num1,const int num2,
69 const DoubleTab& transporte,const int ncomp,
70 DoubleVect& coeff, Matrice_Morse& matrice)
71{
72 int comp,k;
73 IntVect& tab1 = matrice.get_set_tab1();
74 IntVect& tab2 = matrice.get_set_tab2();
75
76 for (comp=0; comp<ncomp; comp++)
77 {
78 if (psc >=0)
79 {
80 for (k=tab1[num1*ncomp+comp]-1; k<tab1[num1*ncomp+comp+1]-1; k++)
81 {
82 if (tab2[k]-1== num1*ncomp+comp)
83 coeff(k) += psc;
84 }
85 for (k=tab1[num2*ncomp+comp]-1; k<tab1[num2*ncomp+comp+1]-1; k++)
86 {
87 if (tab2[k]-1== num1*ncomp+comp)
88 coeff(k) -= psc;
89 }
90 }
91 else
92 {
93 for (k=tab1[num1*ncomp+comp]-1; k<tab1[num1*ncomp+comp+1]-1; k++)
94 {
95 if (tab2[k]-1== num2*ncomp+comp)
96 coeff(k) += psc;
97 }
98 for (k=tab1[num2*ncomp+comp]-1; k<tab1[num2*ncomp+comp+1]-1; k++)
99 {
100 if (tab2[k]-1== num2*ncomp+comp)
101 coeff(k) -= psc;
102 }
103 }
104 }

189 }
190
191 DoubleTab tab(nb_faces_perio,ncomp_ch_transporte);
192// Boucle pour remplir tab
193 nb_faces_perio=0;
194 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
195 {
196 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
197 if (sub_type(Periodique,la_cl.valeur()))
198 {
199// const Periodique& la_cl_perio = ref_cast(Periodique, la_cl.valeur());
200 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
201 int num1 = le_bord.num_premiere_face();
202 int num2 = num1 + le_bord.nb_faces();
203 for (num_face=num1; num_face<num2; num_face++)
204 {
205 for (int comp=0; comp<ncomp_ch_transporte; comp++)
206 tab(nb_faces_perio,comp) = resu(num_face,comp);
207 nb_faces_perio++;
208 }
209 }
210 }
211
212 IntVect compteur(nsom);
213 compteur = 0;
214 vsom=0.;
215
216// Les polyedres non standard sont ranges en 2 groupes dans la Zone_VEF:
217// - polyedres bords et joints
218// - polyedres bords et non joints
219// On traite les polyedres en suivant l'ordre dans lequel ils figurent
220// dans la zone
221
222// boucle sur les polys
223 const IntTab& KEL=zone_VEF.type_elem().valeur().KEL();
224 for (poly=0; poly<nb_elem_tot; poly++)
225 {
226 rang = rang_elem_non_std(poly);
227 if (rang==-1)
228 itypcl=0;
229 else
230 itypcl=zone_Cl_VEF.type_elem_Cl(rang);
231
232// calcul des numeros des faces du polyedre
233 for (face_adj=0; face_adj<nfac; face_adj++)
234 face[face_adj]= elem_faces(poly,face_adj);
235// on conserve cette partie
236 for (j=0; j<dimension; j++)
237 {
238 vs[j] = la_vitesse(face[0],j)*porosite_face(face[0]);
239 for (i=1; i<nfac; i++)
240 vs[j]+= la_vitesse(face[i],j)*porosite_face(face[i]);
241 }
242
243// calcul de la vitesse aux sommets des polyedres
244// int ncomp;
245 if (istetra==1)
246 {
247 for (j=0; j<nsom; j++)
248 for (int ncomp=0; ncomp<Objet_U::dimension; ncomp++)
249 vsom(j,ncomp) =vs[ncomp] - Objet_U::dimension*la_vitesse(face[j],ncomp)*porosite_face(face[j]);
250 }
251 else
252 {
253// pour que cela soit valide avec les hexa
254// On va utliser les fonctions de forme implementees dans la classe Champs_P1_impl ou Champs_Q1_impl
255// int ncomp;
256 for (j=0; j<nsom; j++)
257 {
258 num_som = zone.sommet_elem(poly,j);
259 for (int ncomp=0; ncomp<dimension; ncomp++)
260 vsom(j,ncomp) = la_vitesse.valeur_a_sommet_compo(num_som,poly,ncomp);
261 }
262 }
263
264 type_elem.calcul_vc(face,vc,vs,vsom,vitesse(),itypcl,porosite_face);
265
266// Boucle sur les facettes du polyedre
267 for (fa7=0; fa7<nfa7; fa7++)
268 {
269 if (rang==-1)
270 for (i=0; i<dimension; i++)
271 cc[i] = facette_normales(poly,fa7,i);
272 else
273 for (i=0; i<dimension; i++)
274 cc[i] = normales_facettes_Cl(rang,fa7,i);
275
276// On applique le schema de convection a chaque sommet de la facette
277// On traite le ou les sommets qui sont aussi des sommets du polyedre
278 for (i=0; i<nb_som_facette-1; i++)
279 {
280 psc =0;
281 for (j=0; j<dimension; j++)
282 psc+= vsom(KEL(i+2,fa7),j)*cc[j];
283
284// Boucle sur les facettes du polyedre
285 psc /= nb_som_facette;
286 num10 = face[KEL(0,fa7)];
287 num20 = face[KEL(1,fa7)];
288// psc *= (porosite_face(num1)+porosite_face(num2))/2. ;
289
290 convbis(psc,num10,num20,transporte,ncomp_ch_transporte,resu,fluent_);
291 }
292// On traite le sommet confondu avec le centre de gravite du polyedre
293 psc=0;
294 for (j=0; j<dimension; j++)
295 psc += vc[j]*cc[j];
296 psc /= nb_som_facette;
297 num10 = face[KEL(0,fa7)];
298 num20 = face[KEL(1,fa7)];
299// psc *= (porosite_face(num1)+porosite_face(num2))/2. ;
300
301 convbis(psc,num10,num20,transporte,ncomp_ch_transporte,resu,fluent_);
302 }
303
304 } //!< fin de la boucle if(Process::nproc()==1) Process::Journal()<<"OpVEFFaAmont ap interne resu[8]="<<resu(8,0)<<","<<resu(8,1)<<finl; if((Process::nproc()==2)&&(Process::me()==0)) Process::Journal()<<"OpVEFFaAmont ap interne resu[4]="<<resu(4,0)<<","<<resu(4,1)<<finl;

306//
307//

310 int voisine;
311 nb_faces_perio = 0;
312 double diff1,diff2;

314// Dimensionnement du tableau des flux convectifs au bord du domaine
315// de calcul
316 DoubleTab& flux_b = flux_bords_;
317 flux_b.resize(zone_VEF.nb_faces_bord(),ncomp_ch_transporte);
318 flux_b = 0.;
319
320// Boucle sur les bords pour traiter les conditions aux limites
321// il y a prise en compte d'un terme de convection pour les
322// conditions aux limites de Neumann_sortie_libre seulement
323
324 for (n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
325 {
326
327 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
328
329 if (sub_type(Neumann_sortie_libre,la_cl.valeur()))
330 {
331 const Neumann_sortie_libre& la_sortie_libre = ref_cast(Neumann_sortie_libre, la_cl.valeur());
332 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
333 int num1 = le_bord.num_premiere_face();
334 int num2 = num1 + le_bord.nb_faces();
335 for (num_face=num1; num_face<num2; num_face++)
336 {
337 psc =0;
338 for (i=0; i<dimension; i++)
339 psc += la_vitesse(num_face,i)*face_normales(num_face,i)*porosite_face(num_face);
340 if (psc>0)
341 {
342 for (i=0; i<ncomp_ch_transporte; i++)
343 {
344 resu(num_face,i) -= psc*transporte(num_face,i);
345 flux_b(num_face,i) -= psc*transporte(num_face,i);
346 }
347 }
348 else
349 {
350 for (i=0; i<ncomp_ch_transporte; i++)
351 {
352 resu(num_face,i) -= psc*la_sortie_libre.val_ext(num_face-num1,i);
353 flux_b(num_face,i) -= psc*la_sortie_libre.val_ext(num_face-num1,i);
354 }
355 fluent_[num_face] -= psc;
356 }
357 }
358 }
359 else if (sub_type(Periodique,la_cl.valeur()))
360 {
361 const Periodique& la_cl_perio = ref_cast(Periodique, la_cl.valeur());
362 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
363 int num1 = le_bord.num_premiere_face(), num2 = num1 + le_bord.nb_faces();
364 IntVect fait(le_bord.nb_faces());
365 fait = 0;
366 for (num_face=num1; num_face<num2; num_face++)
367 {
368 if (fait[num_face-num1] == 0)
369 {
370 voisine = la_cl_perio.face_associee(num_face-num1) + num1;
371 for (int comp=0; comp<ncomp_ch_transporte; comp++)
372 {
373 diff1 = resu(num_face,comp)-tab(nb_faces_perio,comp);
374 diff2 = resu(voisine,comp)-tab(nb_faces_perio+voisine-num_face,comp);
375 resu(voisine,comp) += diff1;
376 resu(num_face,comp) += diff2;
377 flux_b(voisine,comp) += diff1;
378 flux_b(num_face,comp) += diff2;
379 }
380
381 fait[num_face-num1]= 1;
382 fait[voisine-num1] = 1;
383 }
384 nb_faces_perio++;
385 }
386 }
387 }
388 modifier_flux(*this);
389 return resu;
390
