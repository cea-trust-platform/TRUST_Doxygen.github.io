42void Champ_front_MED::remplir_tab(DoubleTab& tab,double temps) const
43{
44 int dim=tab.dimension(1);
45 const Frontiere_dis_base& fr_dis=frontiere_dis();
46 const Frontiere& frontiere=fr_dis.frontiere();
47 const Zone& zone=frontiere.zone();
48 const Domaine& domaine=zone.domaine();
49 const Faces& faces=frontiere.faces();
50 int nb_faces=frontiere.nb_faces();
51 double x,y,z;
52 int nbsf=faces.nb_som_faces();
53 int i,k;
54
55 for( i=0; i<nb_faces; i++)
56 {
57 x=y=z=0;
58 for( k=0; k<nbsf; k++)
59 {
60 x+=domaine.coord(faces.sommet(i,k),0);
61 if(dimension>1)
62 y+=domaine.coord(faces.sommet(i,k),1);
63 if(dimension>2)
64 z+=domaine.coord(faces.sommet(i,k),2);
65 }
66 x/=nbsf;
67 y/=nbsf;
68 z/=nbsf;
69 for( k=0; k<dim; k++)
70 {
71 tab(i,k)=valeur_au_temps_et_au_point(temps,-1,x,y,z,k);
72 }
73 }
74 tab.echange_espace_virtuel();
75}
