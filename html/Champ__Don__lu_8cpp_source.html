23Entree& Champ_Don_lu::readOn(Entree& is)
24{
25 int dim;
26 Motcle motlu;
27 Nom nom;
28 is >> nom;
29
30 Domaine& domaine = interprete_get_domaine(nom);
31 const Zone& ma_zone = domaine.zone(0);
32 int nb_elems = ma_zone.nb_elem();
33 dim = lire_dimension(is, que_suis_je());
34
35 dimensionner(nb_elems, dim);
36 domaine.creer_tableau_elements(valeurs());
37
38 is >> nom;
39 LecFicDiffuse fic(nom);
40 Cerr << "Open and read the file : " << nom << finl;
41
42 int nb_val_lues = 0;
43 Nom chaine;
44 fic >> chaine;
45 int star_ccm = 0;
46// Est ce un fichier .csv de STAR-CCM avec en tete: "champ1 unite" "champ2 unite" .... "X" "Y" "Z"
47 if (chaine.debute_par("\""))
48 {
49 star_ccm=1;
50 Cerr << "STAR-CCM csv file detected..." << finl;
51 Cerr << "Find component 1 of the field: " << chaine << finl;
52 for (int i=0; i<dim; i++)
53 {
54 fic >> chaine; //!< Unite du champ
55 fic >> chaine;
56 if (i<dim-1) Cerr << "Find component " << i+2 << " of the field: " << chaine << finl;
57 }
58 if (chaine!="\"X\"")
59 {
60 Cerr << "Error in the file. There must be one field with " << dim << " components." << finl;
61 Process::exit();
62 }
63 fic >> chaine; //!< "Y"
64 fic >> chaine; //!< "Z" Determination du nombre de lignes du fichier STAR-CCM
65//
66 LecFicDiffuse tmp(nom);
67 while (!tmp.eof())
68 {
69 nb_val_lues++;
70 tmp >> chaine;
71 }
72 nb_val_lues-=(dim*2+dimension+1);
73 nb_val_lues/=(dim+dimension);
74 }
75 else
76 {
77// Sinon nombre de lignes a lire
78 nb_val_lues=atoi(chaine);
79 Cerr << "The format of this file containing values of a field should be:" << finl;
80 Cerr << "N" << finl;
81 int nb_lines=4;
82 for (int line=1; line<=nb_lines; line++)
83 {
84 if (line==nb_lines-1) Cerr << "..." << finl;
85 else
86 {
87 Nom l;
88 if (line==nb_lines) l="N";
89 else l=(Nom)line;
90
91 Cerr << "x["<<l<<"] y["<<l<<"] ";
92 if (dimension==3) Cerr << "z["<<l<<"] ";
93 for (int k=0; k<dim; k++)
94 Cerr << "value["<<l<<","<<k<<"] ";
95 Cerr << finl;
96 }
97 }
98 }
99 Cerr << finl << "Read number of values (N)=" << nb_val_lues << finl;
100// Lecture des valeurs dans le fichier fic
101 DoubleTab& mes_val = valeurs();
102 IntTab compteur(nb_elems);
103 compteur = 0;
104 DoubleVect point(dimension);
105 DoubleVect val_lu(dim);
106 int elem2;
107 double x,y,z=0;
108 int pourcent=0,tmp;
109 for (int i=0; i<nb_val_lues; i++)
110 {
111 tmp=(i+1)*10/nb_val_lues;
112 if (tmp>pourcent)
113 {
114 pourcent=tmp;
115 Cerr<<"\rProcess "<<Process::me()<<" has read "<<pourcent*10<<"% of values"<<flush;
116 }
117
118// Lecture des coordonnees du centre de l'element et des valeurs du champs
119 if (star_ccm)
120 {
121 for (int k=0; k<dim; k++) fic >> val_lu[k];
122 for (int k=0; k<dimension; k++) fic >> point[k];
123 }
124 else
125 {
126 for (int k=0; k<dimension; k++) fic >> point[k];
127 for (int k=0; k<dim; k++) fic >> val_lu[k];
128 }
129 x=point[0];
130 y=point[1];
131 if (Objet_U::dimension==3) z=point[2];
132
133// Recherche de l'element
134 elem2=ma_zone.chercher_elements(x,y,z);
135 if ((elem2!=-1) && (elem2<nb_elems))
136 {
137 for (int k=0; k<dim; k++) mes_val(elem2,k) = val_lu[k];
138 compteur[elem2] += 1;
139 }
140 }
141 Cerr << finl;
142// Verification que tous les elements ont ete trouves
143 for (int num_elem=0; num_elem<nb_elems; num_elem++)
144 if (compteur[num_elem] == 0)
145 {
146 Cerr << "Error when using Champ_Don_lu keyword:" << finl;
147 Cerr << "There is no value found for the cell number " << num_elem << finl;
148 Cerr << "Check your mesh and/or the file " << nom << finl;
149 Process::exit();
150 }
151
152 mes_val.echange_espace_virtuel();
153
154 return is;
155}
