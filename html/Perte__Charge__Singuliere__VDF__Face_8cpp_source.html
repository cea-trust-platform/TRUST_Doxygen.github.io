62void Perte_Charge_Singuliere_VDF_Face::remplir_num_faces(Entree& s)
63{
64 const Domaine& le_domaine = equation().probleme().domaine();
65 const Zone_VDF& zone_VDF = ref_cast(Zone_VDF,equation().zone_dis().valeur());
66 int taille_bloc = zone_VDF.nb_elem();
67 num_faces.resize(taille_bloc);
68 lire_surfaces(s,le_domaine,zone_VDF,num_faces, sgn);
69 int nfac = num_faces.size();
70 int nfac_tot = mp_sum(num_faces.size());
71 if (nfac_tot==0)
72 {
73 Cerr << "Erreur a la lecture des donnees de la perte de charge singuliere :" << finl;
74 Cerr << " la surface sur laquelle vous avez defini une perte de charge" << finl;
75 Cerr << " ne contient entierement aucune des faces de la geometrie " << finl;
76 exit();
77 }
78
79 if (nfac != 0)
80 {
81 int ori = zone_VDF.orientation(num_faces[0]);
82 if (ori != direction_perte_charge())
83 {
84 Cerr << "Erreur a la lecture des donnees de la perte de charge singuliere : " << finl;
85 Cerr << " l'orientation de la surface est differente de la direction" << finl;
86 Cerr << " de la perte de charge" << finl;
87 exit();
88 }
89 }
90}

93void Perte_Charge_Singuliere_VDF_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
94{
95 const std::string& nom_inco = equation().inconnue().le_nom().getString();
96 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL, mat2;
97
98 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
99 IntTab stencil(0, 2);
100 stencil.set_smart_resize(1);
101 for (int f = 0; f < zone_VDF.nb_faces(); f++)
102 stencil.append_line(f, f);
103 tableau_trier_retirer_doublons(stencil);
104 if (mat && axi)

107 mat->nb_colonnes() ? *mat += mat2 : *mat = mat2;

112void Perte_Charge_Singuliere_VDF_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
113{
114 const std::string& nom_inco = equation().inconnue().le_nom().getString();
115 const DoubleTab& inco = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : equation().inconnue().valeurs();
116 Matrice_Morse *mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
117
118 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
119 const DoubleVect& volumes_entrelaces = zone_VDF.volumes_entrelaces();
120 const DoubleVect& porosite_surf = zone_VDF.porosite_face();
121 const DoubleTab& vit = la_vitesse->valeurs();
122 int ndeb_faces_int = zone_VDF.premiere_face_int();
123
124 int nb_faces = num_faces.size();
125 int numfa;
126 double Ck,U;
127 double Ud; //!< vitesse debitante
128
129 if (axi)
130 for (int i=0; i<nb_faces; i++)
131 {
132 numfa = num_faces[i];
133 U = vit[numfa];
134
135 if (numfa < ndeb_faces_int)
136 Ck = -0.5*K()/la_zone_VDF->dist_norm_bord_axi(numfa);
137 else
138 Ck = -0.5*K()/la_zone_VDF->dist_norm_axi(numfa);
139 secmem[numfa] += Ck*U*std::fabs(U)*volumes_entrelaces[numfa]*porosite_surf[numfa];
140 }
141 else
142 for (int i=0; i<nb_faces; i++)
143 {
144
145 numfa = num_faces[i];
146// U = vit[numfa];
147
148 Ud = vit[numfa]*porosite_surf[numfa];
149 U = inco[numfa]*porosite_surf[numfa];
150
151 if (numfa < ndeb_faces_int)
152 Ck = -0.5*K()/la_zone_VDF->dist_norm_bord(numfa);
153 else
154 Ck = -0.5*K()/la_zone_VDF->dist_norm(numfa);
155
156 secmem[numfa] += Ck*U*std::fabs(Ud)*volumes_entrelaces[numfa]*porosite_surf[numfa];
157 if(mat) (*mat)(numfa,numfa)-=Ck*porosite_surf[numfa]*std::fabs(Ud)*volumes_entrelaces[numfa]*porosite_surf[numfa];
158
159 }
160
161}
