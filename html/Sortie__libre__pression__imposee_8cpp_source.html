122void Sortie_libre_pression_imposee::completer()
123{
124 const Milieu_base& mil=ma_zone_cl_dis->equation().milieu();
125 if (sub_type(Fluide_Incompressible,mil) && ma_zone_cl_dis->equation().que_suis_je() != "QDM_Multiphase")
126 {
127 if (sub_type(Champ_Uniforme,mil.masse_volumique().valeur()))
128 {
129 const Champ_Uniforme& rho=ref_cast(Champ_Uniforme,mil.masse_volumique().valeur());

160//! 161/** \return double<br>
162Signification: la valeur imposee sur la composante du champ specifiee<br>
163Contraintes: */
164double Sortie_libre_pression_imposee::flux_impose(int i) const
165{
166 const Milieu_base& mil=ma_zone_cl_dis->equation().milieu();
167 double rho_;
168 assert(!est_egal(d_rho,-123.));
169 if (d_rho==-1)
170 {
171 const Champ_base& rho=mil.masse_volumique().valeur();
172 rho_ = rho(i);
173 }
174 else
175 {
176 rho_ = d_rho;
177 }
178
179 if (le_champ_front.valeurs().size()==1)
180 return le_champ_front(0,0)/rho_;
181 else if (le_champ_front.valeurs().dimension(1)==1)
182 return le_champ_front(i,0)/rho_;
183 else
184 Cerr << "Neumann::flux_impose erreur" << finl;
185 exit();
186 return 0.;
187}
