39//! Ecrit le champ sous la forme IJK
40int Champ_Fonc_P0_VDF::imprime(Sortie& os, int ncomp) const
41{
42// valeur_au_ijk(xi,yj,zk,valeurs,ncomp);
43// principe : creer un tableau points avec selon si champ_elem ou champ_faces
44// une grille construite a partie de xi,yj,zk lu dans un fichier
45// sub_type(Champ_P0, *this) : grille construite a partir de (xi+xi+1)/2,etc...
46// Sinon : grille construite a partir de xi,xi+1,etc...
47// Boucle sur ni,nj,nk
48// K=
49// I=1 I=2 I=3 I=4
50// J=10 Vij
51// J=9
52// ...
53 int ni,nj,nk=-1;
54 int ii,jj,k;
55 int np,elem;
56 int cmax=7;
57 DoubleVect xi,yj,zk;
58 DoubleTab Grille;
59// Lecture de xi,yj,zk dans un fichier .xiyjzk
60 Nom nomfic(nom_du_cas());
61 nomfic+=".xiyjzk";
62 LecFicDiffuse ficijk(nomfic);
63 ficijk >> xi;
64 ni=xi.size();
65 ficijk >> yj;
66 nj=yj.size();
67 if (dimension==3)
68 {
69 ficijk >> zk;
70 nk=zk.size();
71 }
72 if (ncomp == 1)
73 {
74 if (dimension==3)
75 {
76// Grille ordonnee sur les centres des elements
77 np=(ni-1)*(nj-1)*(nk-1);
78 Grille.resize(np,dimension);
79 for(k=0; k<nk-1; k++)
80 for(ii=0; ii<ni-1; ii++)
81 for(jj=0; jj<nj-1; jj++)
82 {
83 elem=jj+(nj-1)*(ii+k*(ni-1));
84 Grille(elem,0)=0.5*(xi(ii)+xi(ii+1));
85 Grille(elem,1)=0.5*(yj(jj)+yj(jj+1));
86 Grille(elem,2)=0.5*(zk(k)+zk(k+1));
87 }
88 DoubleTab tab_valeurs(np, nb_compo_);
89 valeur_aux(Grille, tab_valeurs);
90 for(k=0; k<nk-1; k++)
91 {
92 os << finl;
93 os << "Coupe a K= " << k << finl;
94 int n1=0,n2=0;
95 while (n2<ni-1)
96 {
97 n1=n2;
98 n2=std::min(ni-1,n2+cmax);
99 os << finl;
100 os << "I= ";
101 for(int i=n1; i<n2; i++)
102 os << i << " ";
103 os << finl;
104 for(int j=nj-2; j>-1; j--)
105 {
106 os << "J= " << j << " ";
107 for(int i=n1; i<n2; i++)
108 {
109 elem=j+(nj-1)*(i+k*(ni-1));
110 os << tab_valeurs(elem,0) << " ";
111 }
112 os << finl;
113 }
114 }
115 }
116 }
117 else if (dimension==2)
118 {
119 np=(ni-1)*(nj-1);
120 Grille.resize(np,dimension);
121 for(ii=0; ii<ni-1; ii++)
122 for(jj=0; jj<nj-1; jj++)
123 {
124 elem=jj+(nj-1)*ii;
125 Grille(elem,0)=0.5*(xi(ii)+xi(ii+1));
126 Grille(elem,1)=0.5*(yj(jj)+yj(jj+1));
127 }
128 DoubleTab tab_valeurs(np, nb_compo_);
129 valeur_aux(Grille, tab_valeurs);
130 int n1=0,n2=0;
131 while (n2<ni-1)
132 {
133 n1=n2;
134 n2=std::min(ni-1,n2+cmax);
135 os << finl;
136 os << "I= ";
137 for(int i=n1; i<n2; i++)
138 os << i << " ";
139 os << finl;
140 for(int j=nj-2; j>-1; j--)
141 {
142 os << "J= " << j << " ";
143 for(int i=n1; i<n2; i++)
144 {
145 elem=j+(nj-1)*i;
146 os << tab_valeurs(elem,0) << " ";
147 }
148 os << finl;
149 }
150 }
151 }
152 }
153 else
154 {
155 Cerr << "Champ_P0_implementation::imprime_P0_VDF avec nb_compo_>1 non code." << finl;
156 exit();
157 }
158 return 1;
159}
