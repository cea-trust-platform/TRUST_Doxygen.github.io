35//! Lit une liste de sondes a partir d'un flot d'entree<br>Format: { [LIRE UNE SONDE AUTANT DE FOIS QUE NECESSAIRE] }
36/** \return Entree\&<br>
37Signification: le flot d'entree modifie<br>
38Contraintes: */
39Entree& Sondes_Int::readOn(Entree& s )
40{
41 assert(mon_post.non_nul());
42
43 Motcle motlu;
44 Motcle accolade_ouverte("{");
45 Motcle accolade_fermee("}");
46
47 s >> motlu;
48
49 if (motlu != accolade_ouverte)
50 {
51 Cerr << "Error while reading the probes in the postprocessing" << finl;
52 Cerr << "We expected { to start to read the probes" << finl;
53 exit();
54 }
55 s >> motlu;
56
57 if (motlu == accolade_fermee)
58 {
59 Cerr << "Error while reading the probes in the postprocessing" << finl;
60 Cerr << "You have not defined any probe" << finl;
61 exit();
62 }
63 while (motlu != accolade_fermee)
64 {
65 Sonde_Int une_sonde(motlu);
66 une_sonde.associer_post(mon_post.valeur());
67 s >> une_sonde;
68 add(une_sonde);
69 s >> motlu;
70 }
71 Cerr << "End of reading probes " << finl;
72 return s;
73}
