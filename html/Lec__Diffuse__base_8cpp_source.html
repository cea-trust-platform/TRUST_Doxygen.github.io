60int Lec_Diffuse_base::get(char *buf, int bufsize)
61{
62 int l = -1;
63 if (Process::je_suis_maitre())
64 {
65 Entree& is = get_entree_master();
66 assert(is.get_error_action() == ERROR_CONTINUE);
67 int ok = is.get(buf, bufsize);
68 if (ok)
69 l = strlen(buf) + 1;
70 }
71 else if (!diffuse_)
72 {
73 Cerr << "Lec_Diffuse_base::get(...) can't be used with diffuse_=0 on non master process." << finl;
74 Process::exit();
75 }
76 if (diffuse_)
77 {
78 envoyer_broadcast(l, 0);
79 if (l > 0)
80 envoyer_broadcast_array(buf, l, 0);
81 }
82 return error_handle(l < 0);
83}
