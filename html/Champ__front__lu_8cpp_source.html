111int Champ_front_lu::initialiser(double temps, const Champ_Inc_base& inco)
112{
113
114 if (!Ch_front_var_stationnaire::initialiser(temps,inco))
115 return 0;
116
117 int dim=nb_comp();
118 Domaine& domaine=mon_domaine.valeur();
119 const Frontiere_dis_base& fr_dis=frontiere_dis();
120 const Frontiere& frontiere=fr_dis.frontiere();
121 const int nb_faces=frontiere.nb_faces();
122 const int ndeb=frontiere.num_premiere_face();
123 DoubleTab& les_val=valeurs();
124
125 LecFicDiffuse fic;
126 int bin=is_bin(nom_fic);
127 fic.set_bin(bin);
128 fic.ouvrir(nom_fic);
129 Nom format;
130 if (bin)
131 fic >> format;
132 else
133 format="ascii";
134
135 Cerr << "Reading of the " << format << " file : " << nom_fic << finl;
136
137 DoubleVect point(dimension);
138 DoubleVect val_lu(dim);
139 int nb_val_lues;
140 fic >> nb_val_lues;
141
142// Verification (il se peut que le fichier contienne PLUS de faces que la frontiere, exemple en periodique)
143 int nb_faces_tot = mp_sum(nb_faces);
144 if (nb_val_lues<nb_faces_tot)
145 {
146 Cerr << "This file contains " << nb_val_lues << " faces which is not enough compare to" << finl;
147 Cerr << "the faces number (" << nb_faces_tot << ") of the boundary " << frontiere.le_nom() << finl;
148 exit();
149 }
150
151// PQ 20/06/06 : pour permettre une recherche "rapide" des faces
152// on s'appuie sur ma_zone.chercher_elements(x,y,z)
153// (methode inspiree de EcritureLectureSpecial::lecture_special)
154//
155// Restriction --> champs aux faces
156
157 const Zone& ma_zone = domaine.zone(0);
158 int nb_elems = ma_zone.nb_elem();
159 int nbfacelem=ma_zone.nb_faces_elem();
160 const Zone_VF& zvf = ref_cast(Zone_VF, zone_dis());
161 const IntTab& elem_faces=zvf.elem_faces();
162 const DoubleTab& xv=zvf.xv();
163 IntTab compteur(nb_faces);
164 compteur=0;
165
166 int k;
167 double x,y,z=0;
168 int pourcent=0,tmp;
169
170 for (int p=0; p<nb_val_lues; p++)
171 {
172 tmp=(p*10)/nb_val_lues;
173 if (tmp>pourcent)
174 {
175 pourcent=tmp;
176 Cerr<<"\rProcess "<<Process::me()<<" has read "<<pourcent*10<<"% of values"<<flush;
177 }
178 for (k=0; k<dimension; k++)
179 fic >> point[k];
180 for (k=0; k<dim; k++)
181 fic >>val_lu[k];
182
183 x=point[0];
184 y=point[1];
185 if (Objet_U::dimension==3) z=point[2];
186 int Case=-1;
187// Premier algorithme (le plus rapide) cherchant l'element
188// contenant le centre de gravite de la face
189 int num_elem=ma_zone.chercher_elements(x,y,z);
190 if ((num_elem!=-1) && (num_elem<nb_elems))
191 {
192// On cherche la face de l'element
193 for (int fac=0; fac<nbfacelem; fac++)
194 {
195 int face=elem_faces(num_elem,fac);
196 int ok=1;
197 for (int dir=0; dir<dimension; dir++)
198 {
199 ok=ok&&(est_egal(point[dir],xv(face,dir)));
200 }
201 if (ok)
202 {
203 Case=face;
204 break;
205 }
206 }
207 }
208 else
209 {
210// Deuxieme algorithme (plus lent) utilise dans le cas ou le point se trouve a l'exterieur du domaine
211// Dans ce cas la, on parcours les faces de la frontiere
212 for (int face=0; face<nb_faces; face++)
213 if (compteur[face] == 0)
214 {
215 int ok=1;
216 for (int dir=0; dir<dimension; dir++)
217 ok=ok&&(est_egal(point[dir],xv(ndeb+face,dir)));
218 if (ok)
219 {
220 Case=ndeb+face;
221 break;
222 }
223 }
224 }
225// Face trouvee
226 if (Case!=-1)
227 {
228 int face_loc=Case-ndeb;
229 if ((face_loc<0) || (face_loc>=nb_faces))
230 {
231 Cerr << " The face " << ndeb << " found doesn't belong to the boundary sought " << finl;
232 exit();
233 }
234 compteur[face_loc] += 1;
235 for (k=0; k<dim; k++)
236 les_val(face_loc,k) = val_lu[k];
237 }
238 }
239 Cerr<<"\rProcess "<<Process::me()<<" has read 100% of values"<<finl;
240
241// On verifie que toutes les faces de la frontiere ont ete affectees
242 int err=0;
243 for (int face=0; face<nb_faces; face++)
244 if (compteur[face] != 1)
245 {
246 Cerr.precision(12);
247 if (compteur[face] == 0)
248 Cerr << "No value found for the face " << face << "/" << nb_faces-1 << " of the boundary " << frontiere.le_nom() << " on the processor " << Process::me() << ".";
249 else
250 Cerr << "Several values found for the face " << face << "/" << nb_faces-1 << " of the boundary " << frontiere.le_nom() << " on the processor " << Process::me() << ".";
251 Cerr << " Coordinates: " << xv(ndeb+face,0) << " " << xv(ndeb+face,1);
252 if (dimension==3) Cerr << " " << xv(ndeb+face,2);
253 Cerr << finl;
254 err=1;
255 }
256 if (Process::mp_sum(err))
257 {
258 erreur_chfront(nom_fic, dim, mon_domaine.le_nom());
259 }
260 les_val.echange_espace_virtuel();
261 return 1;
262}
