59Entree& Rotation::interpreter_(Entree& is)
60{
61 associer_domaine(is);
62 verifie_type_elem();
63
64 Noms les_fcts;
65 les_fcts.dimensionner(dimension);
66
67 int dir_axe,dir1,dir2;
68 double coor1,coor2,angle;
69 Nom axe_direction,nomc1,nomc2,nomang;
70 Nom var1,var2,var_axe;
71 double pi = M_PI;
72 is>>axe_direction;
73//
74 dir_axe=-1;
75 if (Motcle(axe_direction)=="X")
76 dir_axe=0;
77 else if (Motcle(axe_direction)=="Y")
78 dir_axe=1;
79 else if (Motcle(axe_direction)=="Z")
80 dir_axe=2;
81 else
82 {
83 Cerr<<"The direction of the rotation axis must be specified by X, Y or Z"<<finl;
84 exit();
85 }
86
87 is>>coor1>>coor2>>angle;
88 angle = (2*pi)*(angle/360.);
89 nomc1 = Nom(coor1);
90 nomc2 = Nom(coor2);
91 nomang = Nom(angle);
92
93 dir1=-1;
94 dir2=-1;
95 if (dimension==3)
96 {
97 if (dir_axe==0)
98 {
99 var1="(y-";
100 var2="(z-";
101 var_axe="x";
102 dir1=1;
103 dir2=2;
104 }
105 else if (dir_axe==1)
106 {
107 var1="(z-";
108 var2="(x-";
109 var_axe="y";
110 dir1=2;
111 dir2=0;
112 }
113 else if (dir_axe==2)
114 {
115 var1="(x-";
116 var2="(y-";
117 var_axe="z";
118 dir1=0;
119 dir2=1;
120 }
121
122 les_fcts[dir_axe] = var_axe;
123 }
124 else
125 {
126 var1="(x-";
127 var2="(y-";
128 dir1=0;
129 dir2=1;
130 }
131
132 les_fcts[dir1] = var1;
133 les_fcts[dir1] += "(";
134 les_fcts[dir1] += nomc1;
135 les_fcts[dir1] += ")";
136 les_fcts[dir1] += ")*cos(";
137 les_fcts[dir1] += nomang;
138 les_fcts[dir1] += ")-";
139 les_fcts[dir1] += var2;
140 les_fcts[dir1] += "(";
141 les_fcts[dir1] += nomc2;
142 les_fcts[dir1] += ")";
143 les_fcts[dir1] += ")*sin(";
144 les_fcts[dir1] += nomang;
145 les_fcts[dir1] += ")";
146 les_fcts[dir1] += "+(";
147 les_fcts[dir1] += nomc1;
148 les_fcts[dir1] += ")";
149
150 les_fcts[dir2] = var1;
151 les_fcts[dir2] += "(";
152 les_fcts[dir2] += nomc1;
153 les_fcts[dir2] += ")";
154 les_fcts[dir2] += ")*sin(";
155 les_fcts[dir2] += nomang;
156 les_fcts[dir2] += ")+";
157 les_fcts[dir2] += var2;
158 les_fcts[dir2] += "(";
159 les_fcts[dir2] += nomc2;
160 les_fcts[dir2] += ")";
161 les_fcts[dir2] += ")*cos(";
162 les_fcts[dir2] += nomang;
163 les_fcts[dir2] += ")";
164 les_fcts[dir2] += "+(";
165 les_fcts[dir2] += nomc2;
166 les_fcts[dir2] += ")";
167
168 transformation_complete(les_fcts);
169 return is;
