60Entree& Terme_Source_Canal_RANS_LES_VDF_Face::readOn(Entree& is )
61{
62 int compteur = 0;
63 Motcle mot_lu;
64 Motcle acc_ouverte("{");
65 Motcle acc_fermee("}");
66
67 nom_pb_rans="non_couple";
68 t_av=-1;
69 f_start = 0.;
70 moyenne = 0;//valeur par defaut
71// 0 => moyenne spatiale
72// 1 => moyenne temporelle glissante (moyenne en alpha)
73// 2 => moyenne temporelle commence a t=f_start-t_av
74// 3 => moyenne temporelle commence a t=0 (mettre valeur positive a t_av dans .data)
75
76 Motcles les_mots(6);
77 {
78 les_mots[0]="alpha_tau"; //coeff de relaxation du terme source
79 les_mots[1]="Ly"; //Hauteur du canal plan (utile pour la moyenne spatiale)
80 les_mots[2]="f_start"; //temps a partir duquel le terme source est active
81 les_mots[3]="t_av"; //temps de prise de moyenne (moyenne temporelle glissante)
82 les_mots[4]="type_moyenne"; //temps de prise de moyenne (moyenne temporelle glissante)
83 les_mots[5]="nom_pb_rans";
84 }
85 is >> mot_lu;
86 if(mot_lu != acc_ouverte)
87 {
88 Cerr << "On attendait { a la place de " << mot_lu
89 << " lors de la lecture des parametres de la loi de paroi " << finl;
90 }
91 is >> mot_lu;
92 while(mot_lu != acc_fermee)
93 {
94 int rang=les_mots.search(mot_lu);
95 switch(rang)
96 {
97 case 0 :
98 is >> alpha_tau;
99 Cerr << "alpha_tau = " << alpha_tau << finl;
100 compteur++;
101 break;
102 case 1 :
103 is >> Ly;
104 Cerr << "Ly = "<< Ly << finl;
105 compteur++;
106 break;
107 case 2 :
108 is >> f_start;
109 Cerr << "f_start = "<< f_start << finl;
110 compteur++;
111 break;
112 case 3 :
113 is >> t_av;
114 Cerr << "t_av = " << t_av << finl;
115 Cerr << "type_moyenne : " << moyenne << finl;
116 if((moyenne==1)&&(t_av<=0))
117 {
118 Cerr << "La periode en temps de la moyenne temporelle glissante est non precisee !"
119 << finl;
120 exit();
121 }
122 compteur++;
123 break;
124 case 4 :
125 is >> moyenne;
126 Cerr << "type_moyenne : " << moyenne << finl;
127 if((moyenne==1)&&(t_av<=0))
128 {
129 Cerr << "La periode en temps de la moyenne temporelle glissante est non precisee !"
130 << finl;
131 exit();
132 }
133 compteur++;
134 break;
135 case 5 :
136 is >> nom_pb_rans;
137
138 compteur++;
139 break;
140 default :
141 {
142 Cerr << mot_lu << " n'est pas un mot compris" << finl;
143 Cerr << "Les mots compris sont : " << les_mots << finl;
144 exit();
145 }
146 }
147 is >> mot_lu;
148 }
149 Cerr << "nom_pb_rans = " << nom_pb_rans << finl;
150 Cerr << compteur << " motcles ont ete lus dans le readOn" << finl;
151
152 init();
153
154 return is;

267void Terme_Source_Canal_RANS_LES_VDF_Face::init_calcul_moyenne_spat()
268{
269 Cerr << "Debut initialisation des tableaux de moyenne spatiale..." << finl;
270
271// On va initialiser les differents parametres membres de la classe
272// utiles au calcul des differentes moyennes
273// Initialisation de : Yu,Yv,Yw + compt_x,compt_y,compt_z
274// + corresp_u,corresp_v,corresp_w
275 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
276 const Zone_VDF& zone_VDF=ref_cast(Zone_VDF, zdisbase);
277
278 const DoubleTab& xv = zone_VDF.xv();
279
280 int nb_faces = zone_VDF.nb_faces();
281 int nb_elems = zone_VDF.zone().nb_elem();
282 int nb_faces_y = zone_VDF.nb_faces_Y();
283
284 int Ny; //Nombre de maille en y + 1 (=nombre de faces)
285 Ny = nb_elems/(nb_faces_y-nb_elems);
286 Nxy = Ny;
287 Nyy = Ny+1;
288 Nzy = Ny;
289
290 const IntVect& orientation = zone_VDF.orientation();
291
292 int num_face,j,ori,indic,indicv,indicw,trouve;
293 double y;
294 j=0;
295 indic = 0;
296 indicv = 0;
297 indicw = 0;
298
299// dimensionnement aux valeurs rentrees dans le jeu de donnees
300 Yu.resize(Nxy);
301 Yv.resize(Nyy);
302 Yw.resize(Nzy);
303
304 compt_x.resize(Nxy);
305 compt_y.resize(Nyy);
306 compt_z.resize(Nzy);
307
308// champ_spat.resize(nb_faces);
309
310// champ_spat_x.resize(Nxy);
311// champ_spat_y.resize(Nyy);
312// champ_spat_z.resize(Nzy);
313
314// champ_spat_x_2.resize(Nxy);
315// champ_spat_y_2.resize(Nyy);
316// champ_spat_z_2.resize(Nzy);
317
318 corresp_u.resize(nb_faces);
319 corresp_v.resize(nb_faces);
320 corresp_w.resize(nb_faces);
321
322// initialisation
323 Yu = -100.;
324 Yv = -100.;
325 Yw = -100.;
326
327 compt_x = 0;
328 compt_y = 0;
329 compt_z = 0;
330
331 corresp_u = -1;
332 corresp_v = -1;
333 corresp_w = -1;
334
335// remplissage des tableaux ci-dessus
336
337// Pour le calcul de u, v, w sur les plans d hmogeneite
338 for (num_face=0; num_face<nb_faces; num_face++)
339 {
340 ori = orientation(num_face);
341 switch(ori)
342 {
343 case 0 :
344 {
345 y = xv(num_face,1);
346 trouve = 0;
347 for (j=0; j<indic+1; j++)
348 {
349 if(y==Yu[j])
350 {
351 corresp_u[num_face] = j;
352 compt_x[j] ++;
353 j=indic+1;
354 trouve = 1;
355 break;
356 }
357 }
358 if (trouve==0)
359 {
360 corresp_u[num_face] = indic;
361 Yu[indic]=y;
362 compt_x[indic] ++;
363 indic++;
364 }
365 break;
366 }
367 case 1 :
368 {
369 y = xv(num_face,1);
370 trouve = 0;
371 for (j=0; j<indicv+1; j++)
372 {
373 if(y==Yv[j])
374 {
375 corresp_v[num_face] = j;
376 compt_y[j] ++;
377 j = indicv+1;
378 trouve = 1;
379 break;
380 }
381 }
382 if (trouve==0)
383 {
384 corresp_v[num_face] = indicv;
385 Yv[indicv]=y;
386 compt_y[indicv] ++;
387 indicv++;
388 }
389 break;
390 }
391 case 2 :
392 {
393 y = xv(num_face,1);
394 trouve = 0;
395 for (j=0; j<indicw+1; j++)
396 {
397 if(y==Yw[j])
398 {
399 corresp_w[num_face] = j;
400 compt_z[j] ++;
401 j = indicw+1;
402 trouve = 1;
403 break;
404 }
405 }
406 if (trouve==0)
407 {
408 Yw[indicw]=y;
409 corresp_w[num_face] = indicw;
410 compt_z[indicw] ++;
411 indicw++;
412 }
413 break;
414 }
415 default :
416 {
417 Cerr << "Cas de figure impossible!!" << finl;
418 exit();
419 break;
420 }
421 }
422 }
423 Nxy = indic; //!< nombre de y pour umoy
424 Nyy = indicv;
425 Nzy = indicw;
426
427 Cerr << "Initialisation des tableaux de moyenne spatiale : OK" << finl;
428
429}

431DoubleTab Terme_Source_Canal_RANS_LES_VDF_Face::norme_vit(void) const
432{
433 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
434 int nb_elems = zone_VDF.nb_elem();
435 int nb_faces = zone_VDF.nb_faces();
436 const IntTab& face_voisins = zone_VDF.face_voisins();
437 const IntTab& elem_faces = zone_VDF.elem_faces();
438 DoubleVect norme_vit_elem(nb_elems);
439 norme_vit_elem.resize(nb_faces);
440 double vit1, vit2, vit3;
441 DoubleTab norme(nb_faces);
442
443 for(int num_elem = 0 ; num_elem<nb_elems ; num_elem++)
444 {
445// WARNING 3D ONLY !!!!!!
446 int face1 = elem_faces(num_elem,0);
447 int face2 = elem_faces(num_elem,3);
448 int face3 = elem_faces(num_elem,1);
449 int face4 = elem_faces(num_elem,4);
450 int face5 = elem_faces(num_elem,2);
451 int face6 = elem_faces(num_elem,5);
452
453// Cerr << "face1 = "<<face1<<finl;
454// Cerr << "face2 = "<<face2<<finl;
455// Cerr << "U_RANS(face1) = "<< U_RANS(face1) <<finl;
456// Cerr << "U_RANS(face2) = "<< U_RANS(face2) <<finl;
457
458
459 vit1 = 0.5*(U_RANS(face1)+U_RANS(face2));
460 vit2 = 0.5*(U_RANS(face3)+U_RANS(face4));
461 vit3 = 0.5*(U_RANS(face5)+U_RANS(face6));
462
463 norme_vit_elem(num_elem) = sqrt(vit1*vit1 + vit2*vit2 + vit3*vit3);
464 }
465
466// Redistribution de la norme sur les faces
467
468 for(int num_face = 0 ; num_face<nb_faces ; num_face++)
469 {
470 int elem0 = face_voisins(num_face,0);
471 int elem1 = face_voisins(num_face,1);
472
473 if(elem0==-1)
474 norme(num_face) = norme_vit_elem(elem1);
475 else if(elem1==-1)
476 norme(num_face) = norme_vit_elem(elem0);
477 else
478 norme(num_face) = 0.5*(norme_vit_elem(elem0)+norme_vit_elem(elem1));
479 }
480
481 return norme;
482}

484void Terme_Source_Canal_RANS_LES_VDF_Face::moy_spat(DoubleVect& champ, DoubleVect& champ_spat_x,
485 DoubleVect& champ_spat_y, DoubleVect& champ_spat_z,
486 DoubleVect& champ_spat_x_2, DoubleVect& champ_spat_y_2,
487 DoubleVect& champ_spat_z_2)
488{
489 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
490 const IntVect& orientation = zone_VDF.orientation();
491 int nb_faces = zone_VDF.nb_faces();
492
493 int j=-1;
494
495 compt_x=0;
496 compt_y=0;
497 compt_z=0;
498
499 for (int num_face=0; num_face<nb_faces; num_face++)
500 {
501 int ori = orientation[num_face];
502
503 if(ori==0)
504 {
505 j=corresp_u[num_face];
506 champ_spat_x[j] += champ(num_face);
507 champ_spat_x_2[j] += champ(num_face)*champ(num_face);
508 compt_x[j]++;
509 }
510
511 if(ori==1)
512 {
513 j=corresp_v[num_face];
514 champ_spat_y[j] += champ(num_face);
515 champ_spat_y_2[j] += champ(num_face)*champ(num_face);
516 compt_y[j]++;
517 }
518
519 if(ori==2)
520 {
521 j=corresp_w[num_face];
522 champ_spat_z[j] += champ(num_face);
523 champ_spat_z_2[j] += champ(num_face)*champ(num_face);
524 compt_z[j]++;
525 }
526 }
527
528 for(j=0; j<Nxy; j++)
529 {
530 champ_spat_x[j] = champ_spat_x[j]/compt_x[j];
531 champ_spat_x_2[j] = champ_spat_x_2[j]/compt_x[j];
532 }
533
534 for(j=0; j<Nyy; j++)
535 {
536 champ_spat_y[j] = champ_spat_y[j]/compt_y[j];
537 champ_spat_y_2[j] = champ_spat_y_2[j]/compt_y[j];
538 }
539
540 for(j=0; j<Nzy; j++)
541 {
542 champ_spat_z[j] = champ_spat_z[j]/compt_z[j];
543 champ_spat_z_2[j] = champ_spat_z_2[j]/compt_z[j];
544 }
545}
