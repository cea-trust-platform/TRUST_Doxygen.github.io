72int Convection_Diffusion_std::lire_motcle_non_standard(const Motcle& mot, Entree& is)
73{
74 if (mot=="diffusion")
75 {
76 Cerr << "Reading and typing of the diffusion operator : " << finl;
77 terme_diffusif.associer_diffusivite(diffusivite_pour_transport());
78 is >> terme_diffusif;
79 terme_diffusif.associer_diffusivite_pour_pas_de_temps(diffusivite_pour_pas_de_temps());
80 return 1;
81 }
82 else if (mot=="convection")
83 {
84 Cerr << "Reading and typing of the convection operator : " << finl;
85 const Champ_base& ch_vitesse_transportante = vitesse_pour_transport();
86 associer_vitesse(ch_vitesse_transportante);
87 terme_convectif.associer_vitesse(ch_vitesse_transportante);
88 is >> terme_convectif;
89 return 1;
90 }
91 else
92 return Equation_base::lire_motcle_non_standard(mot,is);
93 return 1;
94}
