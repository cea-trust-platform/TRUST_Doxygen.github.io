48void Champ_front_debit_massique::update_coeff(double temps)
49{
50 int i, fb, n, N = coeff_.line_size(), cR = sub_type(Champ_Uniforme, *ch_rho);
51 DoubleTab rho_bord = cR ? ch_rho->valeurs() : ch_rho->valeur_aux_bords(); /* si rho uniforme, on ne peut pas appeler valeur_aux_bords() */
52 const Zone_VF& zone = ref_cast(Zone_VF, zone_dis());
53 const Front_VF& le_bord= ref_cast(Front_VF,frontiere_dis());
54 for(i = 0; i < le_bord.nb_faces_tot(); i++)
55 for (fb = zone.fbord(le_bord.num_face(i)), n = 0; n < N; ++n)
56 coeff_(i, n) = 1. / rho_bord(!cR * fb, n);
57}
