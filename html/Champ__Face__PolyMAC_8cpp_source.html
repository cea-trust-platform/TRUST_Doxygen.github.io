120Champ_base& Champ_Face_PolyMAC::affecter_(const Champ_base& ch)
121{
122 const DoubleTab& v = ch.valeurs();
123 DoubleTab_parts parts(valeurs());
124 DoubleTab& val = parts[0]; //partie vitesses
125 const Zone_PolyMAC& zone = ref_cast( Zone_PolyMAC,ref_zone_vf_.valeur());
126 int nb_faces = zone.nb_faces();
127 const DoubleVect& surface = zone.face_surfaces();
128 const DoubleTab& normales = zone.face_normales();
129
130 if (sub_type(Champ_Uniforme,ch))
131 {
132 for (int num_face=0; num_face<nb_faces; num_face++)
133 {
134 double vn=0;
135 for (int dir=0; dir<dimension; dir++)
136 vn+=v(0,dir)*normales(num_face,dir);
137
138 vn/=surface(num_face);
139 val(num_face) = vn;
140 }
141 }
142 else if (sub_type(Champ_Fonc_reprise, ch))
143 {
144 for (int num_face=0; num_face<nb_faces; num_face++)
145 val(num_face) = ch.valeurs()[num_face];
146 }
147 else
148 {
149// int ndeb_int = zone.premiere_face_int();
150// const IntTab& face_voisins = zone.face_voisins();
151 const DoubleTab& xv=zone.xv();
152 DoubleTab eval(val.dimension_tot(0),dimension);
153 ch.valeur_aux(xv,eval);
154 for (int num_face=0; num_face<nb_faces; num_face++)
155 {
156 double vn=0;
157 for (int dir=0; dir<dimension; dir++)
158 vn+=eval(num_face,dir)*normales(num_face,dir);
159
160
161 vn/=surface(num_face);
162 val(num_face) = vn;
163 }
164 }
165 return *this;
166}

182void Champ_Face_PolyMAC::init_fcl() const
183{
184 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,zone_vf());
185 const Conds_lim& cls = zone_Cl_dis().les_conditions_limites();
186 int i, f, n;
187
188 fcl_.resize(zone.nb_faces_tot(), 3);
189 for (n = 0; n < cls.size(); n++)
190 {
191 const Front_VF& fvf = ref_cast(Front_VF, cls[n].frontiere_dis());
192 int idx = sub_type(Neumann, cls[n].valeur())
193 + 2 * sub_type(Navier, cls[n].valeur())
194 + 3 * sub_type(Dirichlet, cls[n].valeur()) + 3 * sub_type(Neumann_homogene, cls[n].valeur())
195 + 4 * sub_type(Dirichlet_homogene, cls[n].valeur());
196 if (!idx) Cerr << "Champ_Face_PolyMAC : CL non codee rencontree!" << finl, Process::exit();
197 for (i = 0; i < fvf.nb_faces_tot(); i++)
198 f = fvf.num_face(i), fcl_(f, 0) = idx, fcl_(f, 1) = n, fcl_(f, 2) = i;
199 }
200 fcl_init_ = 1;
201}

203/* vitesse aux elements */
204void Champ_Face_PolyMAC::interp_ve(const DoubleTab& inco, DoubleTab& val, bool is_vit) const
205{
206 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,zone_vf());
207 const DoubleTab& xv = zone.xv(), &xp = zone.xp();
208 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face(), &pe = zone.porosite_elem(), &ve = zone.volumes();
209 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins();
210 int e, f, j, r;
211
212 val = 0;
213 for (e = 0; e < val.dimension(0); e++)
214 for (j = 0; j < e_f.dimension(1) && (f = e_f(e, j)) >= 0; j++)
215 {
216 const double coef = is_vit ? pf(f) / pe(e) : 1.0;
217 for (r = 0; r < dimension; r++) val(e, r) += fs(f) / ve(e) * (xv(f, r) - xp(e, r)) * (e == f_e(f, 0) ? 1 : -1) * inco(f) * coef;
218 }
219}

221/* vitesse aux elements sur une liste d'elements */
222void Champ_Face_PolyMAC::interp_ve(const DoubleTab& inco, const IntVect& les_polys, DoubleTab& val, bool is_vit) const
223{
224 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,zone_vf());
225 const DoubleTab& xv = zone.xv(), &xp = zone.xp();
226 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face(), &pe = zone.porosite_elem(), &ve = zone.volumes();
227 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins();
228 int e, f, j, d, D = dimension, n, N = inco.line_size();
229 assert(ve.line_size() == N * D);
230
231 for (int poly = 0; poly < les_polys.size(); poly++)
232 if ((e = les_polys(poly)) != -1)
233 {
234 for (n = 0; n < N * D; n++) val(e, n) = 0;
235 for (j = 0; j < e_f.dimension(1) && (f = e_f(e, j)) >= 0; j++)
236 {
237 const double coef = is_vit ? pf(f) / pe(e) : 1.0;
238 for (d = 0; d < D; d++)
239 for (n = 0; n < N; n++) val(e, N * d + n) += fs(f) / ve(e) * (xv(f, d) - xp(e, d)) * (e == f_e(f, 0) ? 1 : -1) * inco(f, n) * coef;
240 }
241 }

244DoubleTab& Champ_Face_PolyMAC::valeur_aux_elems(const DoubleTab& positions, const IntVect& les_polys, DoubleTab& val) const
245{
246 const Champ_base& cha=le_champ();
247 int nb_compo=cha.nb_comp(), N = cha.valeurs().line_size(), D = dimension;
248 assert(val.line_size() == nb_compo * N);
249// XXX : TODO Check this assert (positions and not val)
250 assert((positions.dimension(0) == les_polys.size())||(positions.dimension_tot(0) == les_polys.size()));
251// assert((val.dimension(0) == les_polys.size())||(val.dimension_tot(0) == les_polys.size()));
252
253
254 if (val.nb_dim() > 2)
255 {
256 Cerr << "Erreur TRUST dans Champ_Face_implementation::valeur_aux_elems()" << finl;
257 Cerr << "Le DoubleTab val a plus de 2 entrees" << finl;
258 Process::exit();
259 }
260
261 if (nb_compo == 1)
262 {
263 Cerr<<"Champ_Face_implementation::valeur_aux_elems"<<finl;
264 Cerr <<"A scalar field cannot be of Champ_Face type." << finl;
265 Process::exit();
266 }
267
268// on interpole ve sur tous les elements, puis on se restreint a les_polys
269 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,zone_vf());
270 DoubleTrav ve(0, N * D);
271 zone.zone().domaine().creer_tableau_elements(ve);
272 bool is_vit = cha.le_nom().debute_par("vitesse") && !cha.le_nom().debute_par("vitesse_debitante");
273 interp_ve(cha.valeurs(), ve, is_vit);
274 for (int p = 0; p < les_polys.size(); p++)
275 for (int r = 0, e = les_polys(p); e < zone.nb_elem() && r < N * D; r++) val(p, r) = (e==-1) ? 0. : ve(e, r);
276 return val;
277}

309DoubleTab& Champ_Face_PolyMAC::valeur_aux_faces(DoubleTab& val) const
310{
311 const Champ_base& cha=le_champ();
312 int nb_compo=cha.nb_comp(), n, N = cha.valeurs().line_size(), d, D = dimension;
313
314 if (nb_compo == 1)
315 {
316 Cerr<<"Champ_Face_PolyMAC::valeur_aux_faces"<<finl;
317 Cerr <<"A scalar field cannot be of Champ_Face type." << finl;
318 Process::exit();
319 }
320
321 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,zone_vf());
322 val.resize(zone.nb_faces(), N * D);
323
324 for (int f = 0; f < zone.nb_faces(); f++)
325 for (d = 0; d < D; d++)
326 for (n = 0; n < N; n++) val(f, N * d + n) = cha.valeurs()(f, n) * zone.face_normales(f, d) / zone.face_surfaces(f);
327 return val;
328}

334
335DoubleTab& Champ_Face_PolyMAC::trace(const Frontiere_dis_base& fr, DoubleTab& x, double t, int distant) const
336{
337 assert(distant==0);
338 const bool vectoriel = (le_champ().nb_comp() > 1);
339 const DoubleTab& val = valeurs(t);
340 int n, N = val.line_size(), d, D = dimension, dim = vectoriel ? D : 1;
341 const Front_VF& fr_vf = ref_cast(Front_VF, fr);
342 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC,zone_vf());
343 const IntTab& face_voisins = zone.face_voisins();
344 DoubleTrav ve(0, N * D);
345 if (vectoriel)
346 {
347 zone.zone().domaine().creer_tableau_elements(ve);
348 interp_ve(val, ve);
349 }
350
351 for (int i = 0; i < fr_vf.nb_faces(); i++)
352 {
353 const int face = fr_vf.num_premiere_face() + i;
354 for (int dir = 0; dir < 2; dir++)
355 {
356 const int elem = face_voisins(face, dir);
357 if (elem != -1)
358 {
359 for (d = 0; d < dim; d++)
360 for (n = 0; n < N; n++) x(i, N * d + n) = vectoriel ? ve(elem, N * d + n) : val(face, n);
361 }
362 }
363 }
364
365 return x;
366}
