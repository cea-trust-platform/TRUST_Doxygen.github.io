79void Source_Generique_base::completer()
80{
81 Source_base::completer();
82
83 Postraitement post_bidon;
84 post_bidon.associer_nom_et_pb_base("", equation().probleme());
85 ch_source_->completer(post_bidon);
86 Entity loc = ch_source_->get_localisation();
87 if (((loc == ELEMENT) && (localisation_source() != "elem"))
88 || ((loc == FACE) && (localisation_source() != "faces")))
89 {
90 Nom nom_loc;
91 if (loc == ELEMENT)
92 nom_loc = "elem";
93 else if (loc == FACE)
94 nom_loc = "faces";
95 Cerr << "----------------------------------------------------------------" << finl;
96 Cerr << "Error during TRUST calculation :" << finl;
97 Cerr << "----------------------------------------------------------------" << finl;
98 Cerr << "The generic source field associated to the " << que_suis_je() << " source term " << finl;
99 Cerr << "of the equation " << equation().que_suis_je() << " will be evaluated at localisation " << nom_loc
100 << finl;
101 Cerr << "while the source term itself must be computed at location " << localisation_source() << finl;
102 Cerr << "Please contact TRUST support." << finl;
103 exit();
104 }
