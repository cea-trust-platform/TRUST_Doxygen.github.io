38{
39 statistiques().begin_count(diffusion_counter_);
40 const std::string& nom_inco = equation().inconnue().le_nom().getString();
41 Matrice_Morse* mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
42 const DoubleTab& inco = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : equation().inconnue().valeur().valeurs();
43
44 assert_invalide_items_non_calcules(secmem, 0.);
45 if(mat) iter.ajouter_contribution(inco, *mat);
46 iter.ajouter(inco,secmem);
47
48 if (equation().domaine_application() == Motcle("Hydraulique"))
49// On est dans le cas des equations de Navier_Stokes
50 {
51// Ajout du terme supplementaire en V/(R*R) dans le cas des coordonnees axisymetriques
52 if(Objet_U::bidim_axi == 1)
53 {
54 const Zone_VDF& zvdf=iter.zone();
55 const DoubleTab& xv=zvdf.xv();
56 const IntVect& ori=zvdf.orientation();
57 const IntTab& face_voisins=zvdf.face_voisins();
58 const DoubleVect& volumes_entrelaces=zvdf.volumes_entrelaces();
59 int face, nb_faces=zvdf.nb_faces();//, cst;
60 DoubleTrav diffu_tot(zvdf.nb_elem_tot());
61 double db_diffusivite;
62 Nom nom_eq=equation().que_suis_je();
63 if ((nom_eq == "Navier_Stokes_standard")||(nom_eq == "Navier_Stokes_QC")||(nom_eq == "Navier_Stokes_FT_Disc"))
64 {
65 const Eval_Diff_VDF& eval=dynamic_cast<const Eval_Diff_VDF&> (iter.evaluateur());
66 const Champ_base& ch_diff=eval.get_diffusivite();
67 const DoubleTab& tab_diffusivite=ch_diff.valeurs();
68 if (tab_diffusivite.size() == 1)
69 diffu_tot = tab_diffusivite(0,0);
70 else
71 diffu_tot = tab_diffusivite;
72
73 for(face=0; face<nb_faces; face++)
74 if(ori(face)==0)
75 {
76 int elem1=face_voisins(face,0);
77 int elem2=face_voisins(face,1);
78 if(elem1==-1)
79 db_diffusivite=diffu_tot(elem2);
80 else if (elem2==-1)
81 db_diffusivite=diffu_tot(elem1);
82 else
83 db_diffusivite=0.5*(diffu_tot(elem2)+diffu_tot(elem1));
84 double r= xv(face,0);
85 if(r>=1.e-24)
86 {
87 if(mat) (*mat)(face,face)+=db_diffusivite*volumes_entrelaces(face)/(r*r);
88 secmem(face) -=inco(face)*db_diffusivite*volumes_entrelaces(face)/(r*r);

90 }
91 }
92 }
93 else if (equation().que_suis_je() == "Navier_Stokes_Interface_avec_trans_masse" ||
94 equation().que_suis_je() == "Navier_Stokes_Interface_sans_trans_masse" ||

114double Op_Diff_VDF_Face_base::calculer_dt_stab(const Zone_VDF& zone_VDF) const
115{
116// Calcul du pas de temps de stabilite :
117//
118//
119// - La diffusivite n'est pas uniforme donc:
120//
121// dt_stab = Min (1/(2*diffusivite(elem)*coeff(elem))
122//
123// avec :
124// coeff = 1/(dx*dx) + 1/(dy*dy) + 1/(dz*dz)
125//
126// i decrivant l'ensemble des elements du maillage
127//
128// Rq: en hydraulique on cherche le Max sur les elements du maillage
129// initial (comme en thermique) et non le Max sur les volumes de Qdm.
130 double coef=0;
131 const Champ_base& ch_diffu = has_champ_masse_volumique() ? diffusivite() : diffusivite_pour_pas_de_temps();
132 const DoubleTab& diffu = ch_diffu.valeurs();
133 const double Cdiffu = sub_type(Champ_Uniforme, ch_diffu);
134
135// Si la diffusivite est variable, ce doit etre un champ aux elements.
136 assert(Cdiffu || diffu.size() == zone_VDF.nb_elem());
137 const int nb_elem = zone_VDF.nb_elem();
138 for (int elem = 0; elem < nb_elem; elem++)
139 {
140 double diflo = 0.;
141 for (int i = 0; i < dimension; i++)
142 {
143 const double h = zone_VDF.dim_elem(elem, i);
144 diflo += 1. / (h * h);
145 }
146 const int k = Cdiffu ? 0 : elem;
147 double alpha = diffu(k, 0);
148 for (int ncomp = 1; ncomp < diffu.dimension(1); ncomp++)
149 alpha = std::max(alpha, diffu(k, ncomp));
150
151 diflo *= alpha;
152 if (has_champ_masse_volumique())
153 {
154 const DoubleTab& rho = get_champ_masse_volumique().valeurs();
155 diflo/= rho(elem);
156 }
157 coef = std::max(coef,diflo);
158 }
159
160 double dt_stab = (coef==0 ? DMAXFLOAT : 0.5/coef);
161 return Process::mp_min(dt_stab);
162}
