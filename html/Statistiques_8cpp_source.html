91#ifdef GetTimeOfDay_or_rusage
92// Sur un PC a 1.5Ghz, le temps d'execution de get_time est de
93// 1 micro-seconde.
94// getrusage n'est pas utilisable pour le grain fin: la precision
95// n'est que de 0.01 s.
96static const int MILLION = 1000000;
97class Time
98{
99public:
100 Time()
101 {
102 time_.tv_sec = 0;
103 time_.tv_usec = 0;
104 }
105 static const char * const description;
106 void get_time()
107 {
108#ifdef TIME_IS_gettimeofday
109 gettimeofday(& time_, 0);
110#else
111 struct rusage ru;
112// Sous Linux, RUSAGE_BOTH ne fonctionne pas
113 getrusage(RUSAGE_SELF, &ru);
114 time_ = ru.ru_utime; //!< Temps utilisateur ecoule (non systeme)
115#endif
116 }
117// Calcule :
118// time_begin = time_end - time_begin
119// time_ += time_end - time_begin
120 void add_time(Time& time_begin, const Time& time_end)
121 {
122 time_begin.time_.tv_sec = time_end.time_.tv_sec - time_begin.time_.tv_sec;
123 time_begin.time_.tv_usec = time_end.time_.tv_usec - time_begin.time_.tv_usec;
124 if (time_.tv_usec < 0)
125 {
126 time_begin.time_.tv_usec += MILLION;
127 time_begin.time_.tv_sec--;
128 }
129 time_.tv_sec += time_begin.time_.tv_sec;
130 time_.tv_usec += time_begin.time_.tv_usec;
131 if (time_.tv_usec >= MILLION)
132 {
133 time_.tv_usec -= MILLION;
134 time_.tv_sec++;
135 }
136 }
137 void add(Time& time)
138 {
139 time_.tv_sec += time.time_.tv_sec;
140 time_.tv_usec += time.time_.tv_usec;
141 if (time_.tv_usec >= MILLION)
142 {
143 time_.tv_usec -= MILLION;
144 time_.tv_sec++;
145 }
146 }
147 void minus(Time& time)
148 {
149 time_.tv_sec -= time.time_.tv_sec;
150 time_.tv_usec -= time.time_.tv_usec;
151 if (time_.tv_usec < 0)
152 {
153 time_.tv_usec += MILLION;
154 time_.tv_sec--;
155 }
156 }
157 double second() const
158 {
159 return (double) time_.tv_sec + 0.000001 * time_.tv_usec;
160 }
161
162private:
163 struct timeval time_;
164};

176// get_time s'execute en quelques cycles (beaucoup plus fin que
177// gettimeofday). Marche uniquement sur plateforme Intel.
178class Time
179{
180public:
181 Time() : time_(0);
182 { }
183 Time(const Time& t) : time_(t.time_)
184 { }
185 static const char * const description;
186 void get_time()
187 {
188// On recupere le clock counter du processeur
189// (int 64 bits incremente de 1 a chaque instruction)
190 __asm__ __volatile__("rdtsc\n\t" : "=A" (time_) );
191 }
192 void add_time(Time& time_begin, const Time& time_end)
193 {
194 time_begin.time_ = time_end.time_ - time_begin.time_;
195 time_ += time_begin.time_;
196 }
197 double second()
198 {
199 if (seconds_per_hardware_clock == 0.)
200 calibrate();
201 return (double) time_ * seconds_per_hardware_clock;
202 }
203 void calibrate();
204private:
205 static double seconds_per_hardware_clock;
206 unsigned long long time_; //!< Entier 64 bits
207};

214void Time::calibrate()
215{
216 Cerr << "Time::calibrate() for 1 second " << finl;
217 timeval t_begin, now;
218 Time ticks_begin, ticks_now;
219
220 gettimeofday(& t_begin, 0);
221 ticks_begin.get_time();
222 unsigned int delta;
223 do
224 {
225 gettimeofday(& now, 0);
226 ticks_now.get_time();
227 delta = now.tv_usec - t_begin.tv_usec;
228 delta += (now.tv_sec - t_begin.tv_sec) * 1000000;
229 }
230 while (delta < 1000000);
231 unsigned long long delta_ticks = ticks_now.time_ - ticks_begin.time_;
232 seconds_per_hardware_clock = delta * 0.000001 / (double) delta_ticks;
233
234 Cerr << " Result : hardware_clocks_per_second = ";
235 Cerr << 1. / seconds_per_hardware_clock << finl;
236}

257// Je mets une limite en dur parce que:
258// - un nombre trop eleve de compteurs est sans doute le signe d'un bug
259// (creation repetee du meme compteur)
260// - dans un premier temps, ca suffit et c'est plus simple
261static const int MAXCOUNTERS = 1000;
262
263class Stat_Internals
264{
265public:
266 Stat_Internals();
267 int nb_counters;
268 int nb_comm_counters; //nombre de compteurs de communication
269
270// Tous les compteurs sont crees, meme si le niveau est inferieur,
271// cela permet d'activer un niveau superieur au cours du calcul.
272// A la fin, on imprime les resultats correspondant au niveau le plus
273// eleve atteint au cours du calcul.
274 int max_debug_level_during_session;
275
276 Time time_begin[MAXCOUNTERS];
277
278// Cumul des temps cpu pour chaque compteur
279 Time counter_time[MAXCOUNTERS];
280// Le compteur est-il en cours de comptage ?
281 int counter_running[MAXCOUNTERS];
282// Nombre de comptages pour chaque compteur
283 unsigned long long counter_nb[MAXCOUNTERS];
284// Cumul des quantites pour chaque compteur
285 unsigned long long counter_quantity[MAXCOUNTERS];
286// Description des compteurs
287 const char * description[MAXCOUNTERS];
288// Famille de chaque compteur
289 const char * family[MAXCOUNTERS];
290// Niveau de debug du compteur (on n'a pas acces aux Stat_Counter_Id lors
291// de l'impression des statistiques, et on veut pouvoir afficher le niveau
292// quand meme).
293 int counter_level[MAXCOUNTERS];
294
295 int change_level_forbidden;
296 int debug_level_before;
297
298// mapping between counter description and counter id
299 std::map<const char*, int> description_to_id;
300
301// is the counter a communication counter ?
302 int counter_comm[MAXCOUNTERS];
303// est-ce qu'on traque les communications pour ce compteur ?
304 bool comm_zones_on[MAXCOUNTERS];
305// if the counter is a communication, give its index inside the communication_tracking_info array (-1 otherwise)
306 int index_in_communication_tracking_info[MAXCOUNTERS];
307
308 /* tableau stockant les temps de communication de chaque zone du code souhaitee
309 * 1ere dimension : les differents types de communication
310 * 2eme dimension : le temps passe dans chaque zone de communication qu'on a definie pour chaque compteur
311 * ==> quand j different de 0 : l'element (i,j) du tableau contient le temps passe dans la zone j par la communication i
312 * ==> quand j = 0 : l'element (i,j) contient le temps total (dans tout le code) consomme par la communication i
313 */
314 Stat_Results** communication_tracking_info;
315
316
317 /*
318 * Ce tableau stocke et calcule a la fin de chaque iteration:
319 * 1: le cumul du temps passe dans chaque compteur sans compter la derniere iteration
320 * 2: la moyenne du temps passe dans chaque compteur durant une iteration
321 * 3: le temps minimal passe durant une iteration pour chaque compteur
322 * 4: le temps maximal passe durant une iteration pour chaque compteur
323 * 5: la variance du temps passe dans chaque compteur durant une iteration
324 */
325 double counters_avg_min_max_var_per_step[MAXCOUNTERS][5];
326};
327
328Stat_Internals::Stat_Internals() :
329 nb_counters(0),
330 nb_comm_counters(0),
331 max_debug_level_during_session(0),
332 change_level_forbidden(0),
333 debug_level_before(0),
334 communication_tracking_info(0)
335{
336 for (int i = 0; i < MAXCOUNTERS; i++)
337 {
338 counter_running[i] = 0;
339 counter_nb[i] = 0;
340 counter_quantity[i] = 0;
341 description[i] = 0;
342 family[i] = 0;
343 counter_level[i] = 0;
344 counter_comm[i] = 0;
345 comm_zones_on[i] = false;
346 index_in_communication_tracking_info[i] = -1;
347
348 for(int j = 0; j < 5; j++)
349 {
350 if(j==2) //temps minimum ecoule lors d'une iteration : on l'initialise par une valeur arbitrairement grande
351 counters_avg_min_max_var_per_step[i][j] = INITIAL_MIN;
352 else
353 counters_avg_min_max_var_per_step[i][j] = 0.0;
354 }
355 }
356}

396Stat_Counter_Id Statistiques::new_counter(int level,
397 const char * const description,
398 const char * const family,
399 int comm)
400{
401 Stat_Internals& si = *stat_internals;
402 assert(description); //!< Pointeur non nul ?

416 Cerr << "Statistiques::new_counter : description or family : string too long"

418 Cerr << description << finl;
419 assert(0);
420 Process::exit();
421 }
422
423 if (level < 1)
424 level = 1;
425
426 const int id = si.nb_counters;
427 Stat_Counter_Id new_id(id, level);
428 si.counter_level[id] = level;
429 si.description[id] = description;
430 si.family[id] = family;
431 si.nb_counters++;
432 si.description_to_id[description] = id;
433
434 si.counter_comm[id] = comm;
435 if (comm)
436 {
437 si.index_in_communication_tracking_info[id] = si.nb_comm_counters;
438 si.nb_comm_counters++;
439 }
440
441 return new_id;
442}

444void Statistiques::begin_count_(const int id)
445{
446 Stat_Internals& si = *stat_internals;
447 assert(id < si.nb_counters);
448
449 if(si.counter_running[id])
450 {
451 Process::Journal() << "WARNING : counter " << si.description[id]
452 << " has been started while it's already running !" << finl;
453 return;
454 }
455
456
457// demarrage des compteurs seulement apres les 3 premieres iterations car elles peuvent fausser les statistiques
458// (on demarre le compteur du temps total sans attendre que les 3 premiers pas de temps soient ecoules)
459 bool is_temps_total = strcmp(si.description[id], "Temps total") == 0;
460 if ( (JUMP_3_FIRST_STEPS && (is_temps_total || three_first_steps_elapsed_ ) )
461 || !JUMP_3_FIRST_STEPS)
462 {
463 si.time_begin[id].get_time();
464 si.counter_running[id] = 1;
465#ifdef PETSCKSP_H
466 Nom info(si.description[id]);
467 info+="\n";
468 PetscInfo(0,info);
469#endif
470#ifdef VTRACE
471// Level 1 only to avoid MPI calls
472 if (si.counter_level[id]==1) VT_USER_START(si.description[id]);
473#endif
474 }
475
476}

478void Statistiques::end_count_(const int id, int quantity, int count)
479{
480 Stat_Internals& si = *stat_internals;
481 assert(id < si.nb_counters);
482
483// arret des compteurs seulement apres les 3 premieres iterations (car on les demarre seulement apres 3 iterations)
484 bool is_temps_total = strcmp(si.description[id], "Temps total") == 0;
485 if ( (JUMP_3_FIRST_STEPS && (is_temps_total || three_first_steps_elapsed_ ) )
486 || !JUMP_3_FIRST_STEPS)
487 {
488 if (si.counter_running[id])
489 {
490 Time time_end;
491 time_end.get_time();
492 si.counter_time[id].add_time(si.time_begin[id], time_end);
493 si.counter_running[id] = 0;
494 si.counter_nb[id] += count;
495 si.counter_quantity[id] += quantity;
496#ifdef VTRACE
497// Level 1 only to avoid MPI calls
498 if (si.counter_level[id]==1) VT_USER_END(si.description[id]);
499#endif
500 }
501 }
502
503
504}

524 }
525}

614 avg_nb, s2,
615 avg_quantity, s3,
616 avg_avg_time_ps, avg_var_time_ps);
617
618 perfs_globales << tampon << finl;
619 }
620 }

624{
625 assert(message); //!< message non nul !
626
627 Stat_Internals& si = *stat_internals;
628
629 char tampon[BUFLEN+150];
630
631 stop_counters();
632
633 SChaine perfs;
634 SChaine perfs_globales;
635 if (Process::nproc() > 1)
636 {
637 perfs.precision(3);
638 perfs << "Statistics execution [PE " << Process::me() << "] : " << message << finl;
639 perfs << " the time was measured by the following method :" << finl;
640 perfs << " " << Time::description << finl;
641
642
643 sprintf(tampon, "%60s %10s %10s %10s %-23s %10s ",
644 "Counter (level)", "time(s)", "count", "quantity", "avg time per step", "(min,max,std dev)" );
645 perfs << tampon << finl;
646 }
647
648 if (Process::je_suis_maitre())
649 {
650 sprintf(tampon,
651 "%60s %10s %-23s %10s %-23s %10s %-23s %10s %-23s",
652 "Counter (level)",
653 "avg time", "(min,max)",
654 "avg count", "(min,max)",
655 "avg qty", "(min,max)",
656 "avg time per step", "(std dev)");
657 perfs_globales << "Statistics overall execution : " << message << finl;
658 perfs_globales << " the time was measured by the following method :" << finl;
659 perfs_globales << " " << Time::description << finl;
660 perfs_globales << tampon << finl;
661 }
662
663// Verification : on a le meme nombre de compteurs sur tous les procs ?
664 int skip_globals = 0;
665 {
666 int min_nb = (int) Process::mp_min(si.nb_counters);
667 int max_nb = (int) Process::mp_max(si.nb_counters);
668 if (min_nb != max_nb)
669 {

671 {
672 perfs_globales << "Unable to collect statistics :" << finl
673 << " there is not the same number of counters on all"
674 " processors.";
675 }
676 skip_globals = 1;
677 }
678 }
679
680// Temps de reference pour les pourcentages
681 double temps_total_max = 0.;
682
683// Compute Divers (time not counted by other counters in time step loop):
684 if (si.counter_nb[2]>0)
685 {
686 if (strcmp(si.description[2],"Resoudre (timestep loop)"))
687 {
688 Cerr << "Error in Statistiques::dump(...)" << finl;
689 Process::exit();
690 }
691 int id=-1;

811 quantity += si.counter_quantity[i];

814 avg_time_per_step += si.counters_avg_min_max_var_per_step[i][1];

816

830 }

834 perfs << finl;

845 file << perfs_globales.get_str();
846 file << perfs.get_str();
847 file.syncfile();
848 }

851 restart_counters();
852}

1304void Statistiques::compute_avg_min_max_var_per_step(int tstep)
1305{
1306
1307 Stat_Internals& si = *stat_internals;
1308
1309 int step = JUMP_3_FIRST_STEPS ? tstep - 3 : tstep;
1310 double counter_time_step;
1311
1312 for (int i = 0; i < si.nb_counters; i++)
1313 {
1314// suivant que l'option GET_COMM_DETAILS est activee ou non,
1315// les informations de temps des compteurs de communication sont a jour uniquement dans le tableau communication_tracking_times
1316 if ( (GET_COMM_DETAILS && !si.counter_comm[i]) || !GET_COMM_DETAILS )
1317 {
1318 counter_time_step = si.counter_time[i].second() - si.counters_avg_min_max_var_per_step[i][0]; //temps passe par le compteur i durant l'iteration tstep
1319 si.counters_avg_min_max_var_per_step[i][0] = si.counter_time[i].second();
1320 }
1321 else
1322 {
1323 if(si.communication_tracking_info == 0)
1324 {
1325 Cerr << "compute_avg_min_max_var_per_step: error ! communication_tracking_times hasn't been allocated yet... (disable the option GET_COMM_DETAILS in file Statistiques.h)" <<finl;
1326 Process::exit();
1327 }
1328
1329// correspondance entre l'identifiant du compteur de communication et son indice dans le tableau communication_tracking_times
1330 int index = si.index_in_communication_tracking_info[i] ;
1331 counter_time_step = si.communication_tracking_info[index][0].time - si.counters_avg_min_max_var_per_step[i][0]; //temps passe par le compteur i durant l'iteration tstep
1332 si.counters_avg_min_max_var_per_step[i][0] = si.communication_tracking_info[index][0].time;
1333
1334
1335 }
1336
1337 double old_moy = si.counters_avg_min_max_var_per_step[i][1];
1338// moyenne courante
1339 si.counters_avg_min_max_var_per_step[i][1] = ((si.counters_avg_min_max_var_per_step[i][1] * step) + counter_time_step) / (step + 1);
1340// min courant
1341 si.counters_avg_min_max_var_per_step[i][2] = std::min(si.counters_avg_min_max_var_per_step[i][2],counter_time_step);
1342// max courant

1344// variance courante

1346 si.counters_avg_min_max_var_per_step[i][4] += counter_time_step*counter_time_step;
1347 si.counters_avg_min_max_var_per_step[i][4] /= step+1;
1348 si.counters_avg_min_max_var_per_step[i][4] -= si.counters_avg_min_max_var_per_step[i][1] * si.counters_avg_min_max_var_per_step[i][1];
1349// avec les erreurs d'arrondis, on peut obtenir une variance negative
1350 if (si.counters_avg_min_max_var_per_step[i][4] < 0)
1351 si.counters_avg_min_max_var_per_step[i][4] = 0.0;
