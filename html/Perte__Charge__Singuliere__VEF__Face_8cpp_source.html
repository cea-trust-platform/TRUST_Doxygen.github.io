59void Perte_Charge_Singuliere_VEF_Face::remplir_num_faces(Entree& s)
60{
61 const Domaine& le_domaine = equation().probleme().domaine();
62 const Zone_VEF& zone_VEF = ref_cast(Zone_VEF,equation().zone_dis().valeur());
63 int taille_bloc = zone_VEF.nb_elem();
64 num_faces.resize(taille_bloc);

66// int nfac_tot = mp_sum(num_faces.size());

83DoubleTab& Perte_Charge_Singuliere_VEF_Face::ajouter(DoubleTab& resu) const
84{
85 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
86// const DoubleVect& volumes_entrelaces = zone_VEF.volumes_entrelaces();
87 const DoubleVect& porosite_surf = zone_VEF.porosite_face();
88 const DoubleTab& vit = la_vitesse->valeurs();
89
90 int nb_faces = num_faces.size();
91 int numfa;
92 double Ud; //!< vitesse debitante
93 double area;
94 double coefK; //!< pressure loss
95
96 for (int i=0; i<nb_faces; i++)
97 {
98 numfa = num_faces[i];
99
100 for (int j=0; j<dimension; j++)
101 {
102 Ud = vit(numfa,j)*porosite_surf[numfa];
103// area=zone_VEF.face_surfaces(numfa);
104 area=direction_perte_charge() < 0 ? zone_VEF.face_surfaces(numfa) : std::fabs(zone_VEF.face_normales(numfa,direction_perte_charge())); //!< Taking account of inclined plane
105 coefK=K();
106// if (j!=direction_perte_charge()) coefK=0.;
107 resu(numfa,j) -= 0.5*coefK*Ud*std::fabs(Ud)*area*porosite_surf[numfa];
108 }
109 }
110 return resu;
111}

119void Perte_Charge_Singuliere_VEF_Face::contribuer_a_avec(const DoubleTab& inco, Matrice_Morse& matrice) const
120{
121 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
122 const DoubleVect& porosite_surf = zone_VEF.porosite_face();
123 const DoubleTab& vit = la_vitesse->valeurs();
124
125 int nb_faces = num_faces.size();
126 int numfa;
127 double Ud; //!< vitesse debitante
128 double area;
129 double coefK; //!< pressure loss
130
131 for (int i=0; i<nb_faces; i++)
132 {
133 numfa = num_faces[i];
134
135 for (int j=0; j<dimension; j++)
136 {
137// Ud = vit(numfa,j)*porosite_surf[numfa];
138 Ud = porosite_surf[numfa];
139 double Ud2= vit(numfa,j)*porosite_surf[numfa];
140 int n0=numfa*dimension+j;
141// area=zone_VEF.face_surfaces(numfa);
142 area=direction_perte_charge() < 0 ? zone_VEF.face_surfaces(numfa) : std::fabs(zone_VEF.face_normales(numfa,direction_perte_charge())); //!< Taking account of inclined plane
143 coefK=K();
144// if (j!=direction_perte_charge()) coefK=0.;
145 matrice.coef(n0,n0)+= 0.5*coefK*Ud*std::fabs(Ud2)*area*porosite_surf[numfa];
146 }
