27Entree& Champ_Fonc_Tabule_Morceaux::readOn(Entree& is)
28{
29 Nom nom;
30 is >> nom;
31 interprete_get_domaine(nom);
32 mon_domaine->creer_tableau_elements(table_idx);
33 table_idx = -1; //!< pour planter si on a oublie une sous-zone
34
35 int nbcomp;
36 is >> nbcomp;
37
38 fixer_nb_comp(nbcomp);
39 valeurs_.resize(0, nbcomp);
40 mon_domaine->creer_tableau_elements(valeurs_);
41
42 is >> nom;
43 if(Motcle(nom) != Motcle("{") )
44 {
45 Cerr << "Error while reading a " << que_suis_je() << finl;
46 Cerr << "We expected a { instead of " << nom << finl;
47 Process::exit();
48 }
49
50 is >> nom;
51 while (nom != Nom("}"))
52 {
53 CHTAB ch_lu;
54 /* 1. lecture de la sous-zone */
55 REF(Sous_Zone) refssz = les_sous_zones.add(mon_domaine->ss_zone(nom));
56 Sous_Zone& ssz = refssz.valeur();
57
58 /* 2. lecture des champs parametres */
59 int old_table_syntax_ = 0;
60 is >> nom;
61 if (nom != "{")
62 {
63 ch_lu.noms_champs_parametre_.push_back(nom.getString());
64 old_table_syntax_ = 1;
65 Cerr << "Errror reading old syntax: " << que_suis_je() << " " << domaine().le_nom() << " " << nbcomp << " { " << ssz.le_nom() << " " << nom << " 1 { ... } ... }" << finl;
66 Cerr << "New syntax is like : " << que_suis_je() << " " << domaine().le_nom() << " " << nbcomp << " { " << ssz.le_nom() << " { problem_name " << nom << " } { ... } ... }" << finl;
67 Cerr << "Check the doc or release notes." << finl;
68 Process::exit();
69 }
70 else
71 {
72 assert (old_table_syntax_ == 0);
73 while (true)
74 {
75 is >> nom;
76 if (nom == "}") break;
77 ch_lu.noms_pbs_.push_back(nom.getString());
78 is >> nom;
79 ch_lu.noms_champs_parametre_.push_back(nom.getString());
80 }
81 }
82 const int nb_param = ch_lu.noms_champs_parametre_.size();
83
84 if (old_table_syntax_ && ch_lu.noms_pbs_.size() != 0 && nb_param != 1)
85 {
86 Cerr << "What ??? Big problem in Champ_Fonc_Tabule::readOn !!!" << finl;
87 throw;
88 }
89
90 is >> nom;
91 if (nom == "{")
92 {
93 /* 1. lecture de la grille de parametres */
94 DoubleVects params;
95 for (int n = 0; n < nb_param; n++)
96 {
97 const int nb_val = lire_dimension(is, que_suis_je());
98 DoubleVect param(nb_val);
99 for (int i = 0; i < nb_val; i++) is >> param[i];
100 params.add(param);
101 }
102
103 /* 2. lecture des valeurs des parametres */
104// taille totale du tableau de valeurs
105 int size = nbcomp;
106 for (int n = 0; n < nb_param; n++) size *= params[n].size();
107
108// lecture : tout dans un tableau 1D
109 DoubleVect tab_valeurs(size);
110 for (int i = 0; i < size; i++) is >> tab_valeurs[i];
111 ch_lu.la_table.remplir(params, tab_valeurs);
112
113 is >> nom;
114 if (nom != "}")
115 {
116 Cerr << "Error reading from an object of type Champ_Fonc_Tabule" << finl;
117 Cerr << "We expected keyword } instead of " << nom << finl;
118 Process::exit();
119 }
120 }
121 else
122 {
123 Cerr << "Error reading from an object of type Champ_Fonc_Tabule" << finl;
124 Cerr << "We expected keyword { or fonction instead of " << nom << finl;
125 Process::exit();
126 }
127
128 for (int e = 0; e < ssz.nb_elem_tot(); e++) table_idx(ssz(e)) = champs_lus.size();
129 champs_lus.push_back(ch_lu);
130 is >> nom;
131 }
132 return is;

135void Champ_Fonc_Tabule_Morceaux::mettre_a_jour(double time)
136{
137 if (!init_)
138 {
139 initialiser(time);
140 return;
141 }
142 DoubleTab& tab = valeurs();
143 Domaine& le_domaine = mon_domaine.valeur();
144 const IntTab& les_elems = le_domaine.zone(0).les_elems();
145 const int nb_som_elem = le_domaine.zone(0).nb_som_elem(), nb_elem_tot = le_domaine.zone(0).nb_elem_tot();
146
147// check whether valeur_aux_elems is needed
148 bool calc_centers = false;
149 for (auto && ch : champs_lus)
150 for (int i = 0; i < ch.champs_parametre_.size(); i++)
151 {
152 assert(ch.champs_parametre_[i]->valeurs().dimension(1) == 1 || ch.champs_parametre_[i]->valeurs().dimension(1) == tab.dimension(1));
153 if (tab.get_md_vector() != ch.champs_parametre_[i]->valeurs().get_md_vector())
154 ch.needs_projection.push_back(true), calc_centers = true;
155 else ch.needs_projection.push_back(false);
156 }
157
158// valeur aux elems : on les stocke dans un map {"pb_champ" : champ_aux_elems, ...}
159 std::map<std::string, DoubleTab> val_params_aux_elems;
160 if (calc_centers)
161 {
162 DoubleTab centres_de_gravites(0, dimension);
163 le_domaine.creer_tableau_elements(centres_de_gravites);
164
165 for (int i = 0; i < le_domaine.zone(0).nb_elem_tot(); i++)
166 {
167 int nb_som = 0, s, r;
168 for (int j = 0; j < nb_som_elem && (s = les_elems(i, j)) >= 0; j++)
169 for (r = 0, nb_som++; r < dimension; r++)
170 centres_de_gravites(i, r) += le_domaine.coord(s, r);
171
172 for (r = 0; r < dimension; r++) centres_de_gravites(i, r) /= nb_som;
173 }
174
175 IntVect les_polys(nb_elem_tot);
176 for(int elem = 0; elem < nb_elem_tot; elem++) les_polys(elem) = elem;
177
178// Estimate the field parameter on cells:
179 for (auto && ch : champs_lus)
180 for (int i = 0; i < ch.champs_parametre_.size(); i++)
181 {
182 std::string nc = ch.noms_pbs_[i] + "_" + ch.noms_champs_parametre_[i];
183 if (!val_params_aux_elems.count(nc))
184 {
185 if (ch.needs_projection[i])
186 {
187 DoubleTab vp(nb_elem_tot, ch.champs_parametre_[i]->valeurs().dimension(1));
188 ch.champs_parametre_[i]->valeur_aux_elems(centres_de_gravites, les_polys, vp);
189 val_params_aux_elems[nc] = vp;
190 }
191 else
192 val_params_aux_elems[nc] = ch.champs_parametre_[i]->valeurs();
193 }
194 }
195 }
196 else
197 {
198// Estimate the field parameter on cells:
199 for (auto && ch : champs_lus)
200 for (int i = 0; i < ch.champs_parametre_.size(); i++)
201 {
202 std::string nc = ch.noms_pbs_[i] + "_" + ch.noms_champs_parametre_[i];
203 if (!val_params_aux_elems.count(nc)) val_params_aux_elems[nc] = ch.champs_parametre_[i]->valeurs();
204 }
205 }
206
207// Compute the field according to the parameter field
208 const int nbcomp = tab.dimension(1);
209 for (int e = 0; e < le_domaine.zone(0).nb_elem(); e++)
210 if (nbcomp == 1 && champs_lus[table_idx(e)].noms_champs_parametre_.size() == 1)
211 {
212 std::string nc = champs_lus[table_idx(e)].noms_pbs_[0] + "_" + champs_lus[table_idx(e)].noms_champs_parametre_[0];
213 const double v = val_params_aux_elems[nc](e, 0);
214 tab(e, 0) = champs_lus[table_idx(e)].la_table.val_simple(v);
215 }
216 else for (int n = 0; n < nbcomp; n++)
217 {
218 std::vector<double> vals;
219 for (unsigned int k = 0; k < champs_lus[table_idx(e)].noms_champs_parametre_.size(); k++)
220 {
221 std::string nc = champs_lus[table_idx(e)].noms_pbs_[k] + "_" + champs_lus[table_idx(e)].noms_champs_parametre_[k];
222 vals.push_back(val_params_aux_elems[nc](e, champs_lus[table_idx(e)].champs_parametre_[k]->valeurs().dimension(1) == 1 ? 0 : n));
223 }
224 tab(e, n) = champs_lus[table_idx(e)].la_table.val(vals, n);
225 }
226 tab.echange_espace_virtuel();
227}

229int Champ_Fonc_Tabule_Morceaux::initialiser(const double time)
230{
231 for (auto &&ch : champs_lus)
232 for (unsigned int i = 0; i < ch.noms_champs_parametre_.size(); i++)
233 {
234 REF(Champ_base) champ;
235 const Probleme_base& pb_ch = ref_cast(Probleme_base, Interprete::objet(Nom(ch.noms_pbs_[i])));
236 champ = pb_ch.get_champ(Nom(ch.noms_champs_parametre_[i]));
237 ch.champs_parametre_.add(champ);
238 }
239 init_ = true;
240 mettre_a_jour(time);
241 return 1;
242}
