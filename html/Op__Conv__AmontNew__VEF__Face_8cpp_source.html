42static inline void convbis(const double psc,const int num1,const int num2,
43 double T1, double T2 , int comp, int ncomp,
44 DoubleTab& resu, DoubleVect& fluent)
45{
46 int amont;
47 double flux;
48
49 if (psc >= 0)
50 {
51 amont = num1;
52 if(comp==0)
53 fluent[num2] += psc;
54 }
55 else
56 {
57 amont = num2;
58 if(comp==0)
59 fluent[num1] -= psc;
60 }
61
62 if(amont==num1)
63 flux = T1*psc;
64 else
65 flux = T2*psc;
66 if(ncomp>1)
67 {
68 resu(num1,comp) -= flux;
69 resu(num2,comp) += flux;
70 }
71 else
72 {
73 resu(num1) -= flux;
74 resu(num2) += flux;
75 }
76}
