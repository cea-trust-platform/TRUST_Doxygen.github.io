62double Op_Dift_VEF_Face::calculer_dt_stab() const
63{
64// 12/04: Premiers efforts d'optimisation du code a tenir
65// compte lors de la factorisation des pas de temps de stabilite
66
67// Calcul de dt_stab
68// La diffusivite est constante par elements donc
69// il faut calculer dt_diff pour chaque element et
70// dt_stab=Min(dt_diff (K) = h(K)*h(K)/(2*dimension*diffu2_(K)))
71// ou diffu2_ est la somme des 2 diffusivite laminaire et turbulente
72
73// GF
74// alpha_dt_stab=(alpha+alpha_t)*alpha_dt_stab/alpha
75// alpha_dt_stab=(nu+diff_nu_turb)*valeurs_diffusivite_dt/nu
76
77// On remplit le tableau nu contenant la diffusivite en chaque elem
78 remplir_nu(nu_);
79
80 double dt_stab=1.e30;
81 double coef;
82 const Zone_VEF& la_zone_VEF = la_zone_vef.valeur();
83// const DoubleVect& porosite_elem = la_zone_VEF.porosite_elem();
84
85 const Zone& la_zone= la_zone_VEF.zone();
86
87 DoubleVect diffu_turb(diffusivite_turbulente()->valeurs());
88 DoubleTab diffu(nu_);
89 if (equation().que_suis_je().debute_par("Convection_Diffusion_Temp"))
90 {
91 double rhocp = mon_equation->milieu().capacite_calorifique().valeurs()(0, 0) * mon_equation->milieu().masse_volumique().valeurs()(0, 0);
92 diffu_turb /= rhocp;
93 diffu /= rhocp;
94 }
95 double alpha;
96
97 int la_zone_nb_elem=la_zone.nb_elem();
98 if (has_champ_masse_volumique())
99 {
100 const DoubleTab& rho_elem = get_champ_masse_volumique().valeurs();
101 assert(rho_elem.size_array()==la_zone_VEF.nb_elem_tot());
102 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
103 {
104 alpha = diffu[num_elem] + diffu_turb[num_elem]; //!< PQ : 06/03
105 alpha/=rho_elem[num_elem];
106// dt=1/(dimension/(pas*pas))/(2*alpha)
107
108 coef=la_zone_VEF.carre_pas_maille(num_elem)/(2.*dimension*(alpha+DMINFLOAT));
109
110// dt_stab=min(dt);
111 if (coef<dt_stab) dt_stab = coef;
112 }
113 }
114 else
115 {
116
117 const Champ_base& champ_diffusivite = diffusivite_pour_pas_de_temps();
118 const DoubleTab& valeurs_diffusivite = champ_diffusivite.valeurs();
119 double valeurs_diffusivite_dt=-1;
120 int unif_diffu_dt;
121 const Nature_du_champ nature_champ = equation().inconnue()->nature_du_champ();
122
123 if (sub_type(Champ_Uniforme,champ_diffusivite))
124 {
125 valeurs_diffusivite_dt=valeurs_diffusivite(0,0);
126 unif_diffu_dt=1;
127 }
128 else
129 unif_diffu_dt=0;
130 if (nature_champ!=multi_scalaire)
131 {
132 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
133 {
134 alpha = diffu[num_elem] + diffu_turb[num_elem]; //!< PQ : 06/03
135 if(unif_diffu_dt==0)
136 valeurs_diffusivite_dt=valeurs_diffusivite(num_elem,0);
137 alpha*=valeurs_diffusivite_dt/(diffu[num_elem]+DMINFLOAT);
138// dt=1/(dimension/(pas*pas))/(2*alpha)
139 coef=la_zone_VEF.carre_pas_maille(num_elem)/(2.*dimension*(alpha+DMINFLOAT));
140 assert(coef>=0);
141// dt_stab=min(dt);
142 if (coef<dt_stab) dt_stab = coef;
143 }
144 }
145 else
146 {
147 int nb_comp = valeurs_diffusivite.line_size();
148 for (int nc=0; nc<nb_comp; nc++)
149 {
150 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
151 {
152 alpha = diffu(num_elem,nc) + diffu_turb[num_elem];
153// if(unif_diffu_dt==0)
154 valeurs_diffusivite_dt=valeurs_diffusivite(0,nc);
155 alpha*=valeurs_diffusivite_dt/(diffu(num_elem,nc)+DMINFLOAT);
156 coef=la_zone_VEF.carre_pas_maille(num_elem)/(2.*dimension*(alpha+DMINFLOAT));
157
158 if (coef<dt_stab) dt_stab = coef;
159 }
160 }
161 }
162 }
163
164 dt_stab = Process::mp_min(dt_stab);
165 return dt_stab;
166}

169void Op_Dift_VEF_Face::calculer_pour_post(Champ& espace_stockage,const Nom& option,int comp) const
170{
171 if (Motcle(option)=="stabilite")
172 {
173 DoubleTab& es_valeurs = espace_stockage->valeurs();
174
175 if (la_zone_vef.non_nul())
176 {
177// On remplit le tableau nu contenant la diffusivite en chaque elem
178 remplir_nu(nu_);
179
180 double coef;
181 const Zone_VEF& la_zone_VEF = la_zone_vef.valeur();
182 const Zone& la_zone= la_zone_VEF.zone();
183 const DoubleVect& diffu_turb=diffusivite_turbulente()->valeurs();
184 double alpha;
185
186 int la_zone_nb_elem=la_zone.nb_elem();
187 if (has_champ_masse_volumique())
188 {
189 const DoubleTab& rho_elem = get_champ_masse_volumique().valeurs();
190 assert(rho_elem.size_array()==la_zone_VEF.nb_elem_tot());
191 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
192 {
193 alpha = nu_[num_elem] + diffu_turb[num_elem]; //!< PQ : 06/03
194 alpha/=rho_elem[num_elem];
195 coef=la_zone_VEF.carre_pas_maille(num_elem)/(2.*dimension*alpha);
196 es_valeurs(num_elem) = coef;
197 }
198 }
199 else
200 {
201 const Champ_base& champ_diffusivite = diffusivite_pour_pas_de_temps();
202 const DoubleTab& valeurs_diffusivite = champ_diffusivite.valeurs();
203 double valeurs_diffusivite_dt=-1;
204 int unif_diffu_dt;
205 if (sub_type(Champ_Uniforme,champ_diffusivite))
206 {
207 valeurs_diffusivite_dt=valeurs_diffusivite(0,0);
208 unif_diffu_dt=1;
209 }
210 else
211 unif_diffu_dt=0;
212 for (int num_elem=0; num_elem<la_zone_nb_elem; num_elem++)
213 {
214 alpha = nu_[num_elem] + diffu_turb[num_elem]; //!< PQ : 06/03
215 if(unif_diffu_dt==0)
216 valeurs_diffusivite_dt=valeurs_diffusivite(num_elem);
217 alpha*=valeurs_diffusivite_dt/nu_[num_elem];
218 coef=la_zone_VEF.carre_pas_maille(num_elem)/(2.*dimension*alpha);
219 es_valeurs(num_elem) = coef;
220 }
221 }
222
223 assert(est_egal(mp_min_vect(es_valeurs),calculer_dt_stab()));
224 }
225 }
226 else
227 Op_Dift_VEF_base::calculer_pour_post(espace_stockage,option,comp);
228}

239void Op_Dift_VEF_Face::ajouter_cas_scalaire(const DoubleVect& inconnue,
240 DoubleVect& resu, DoubleTab& tab_flux_bords,
241 const DoubleVect& nu,
242 const DoubleVect& nu_turb,
243 const Zone_Cl_VEF& zone_Cl_VEF,
244 const Zone_VEF& zone_VEF ) const
245{
246 const IntTab& elem_faces = zone_VEF.elem_faces();
247 const IntTab& face_voisins = zone_VEF.face_voisins();
248 int i,j,num_face;
249 int nb_faces = zone_VEF.nb_faces();
250 int nb_faces_elem = zone_VEF.zone().nb_faces_elem();
251 double valA, d_nu;
252 int nb_front=zone_VEF.nb_front_Cl();
253
254// modif pour imprimer les flux sur les bords
255 int size_flux_bords=zone_VEF.nb_faces_bord();
256 tab_flux_bords.resize(size_flux_bords,1);
257 tab_flux_bords=0.;
258
259 const int premiere_face_int=zone_VEF.premiere_face_int();
260
261// contient -1 si la face n'est pas periodique et numero face_assso sinon
262 ArrOfInt marq( zone_VEF.nb_faces_tot());
263 marq=-1;
264// On traite les faces bord
265 for (int n_bord=0; n_bord<nb_front; n_bord++)
266 {
267 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
268 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
269 int num1 = 0;
270 int num2 = le_bord.nb_faces_tot();
271 int nb_faces_bord_reel = le_bord.nb_faces();
272
273 if (sub_type(Periodique,la_cl.valeur()))
274 {
275 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
276 int fac_asso;
277 for (int ind_face=num1; ind_face<num2; ind_face++)
278 {
279 fac_asso = la_cl_perio.face_associee(ind_face);
280 fac_asso = le_bord.num_face(fac_asso);
281 num_face = le_bord.num_face(ind_face);
282 marq[num_face]=fac_asso;
283 for (int kk=0; kk<2; kk++)
284 {
285 int elem = face_voisins(num_face,kk);
286 d_nu = nu(elem)+nu_turb(elem);
287 for (i=0; i<nb_faces_elem; i++)
288 {
289 if ( ( (j= elem_faces(elem,i)) > num_face ) && (j != fac_asso) )
290 {
291 valA = viscA(num_face,j,elem,d_nu);
292 resu(num_face)+=valA*inconnue(j);
293 resu(num_face)-=valA*inconnue(num_face);
294 if(j<nb_faces) //!< face reelle
295 {
296 resu(j)+=0.5*valA*inconnue(num_face);
297 resu(j)-=0.5*valA*inconnue(j);
298 }
299 }
300 }
301 }
302 }
303 }
304 else //!< Il n'y a qu'une seule composante, donc on traite une equation scalaire (pas la vitesse) on a pas a utiliser le tau tangentiel (les lois de paroi thermiques ne calculent pas d'echange turbulent a la paroi pour l'instant
305//
306//
307//
308 {
309 const DoubleTab& face_normale = zone_VEF.face_normales();
310 const DoubleVect& vol = zone_VEF.volumes();
311 const Equation_base& my_eqn = zone_Cl_VEF.equation();
312 const RefObjU& modele_turbulence = my_eqn.get_modele(TURBULENCE);
313 if (sub_type(Modele_turbulence_scal_base,modele_turbulence.valeur()))
314 {
315 const Modele_turbulence_scal_base& mod_turb_scal = ref_cast(Modele_turbulence_scal_base,modele_turbulence.valeur());
316 const Turbulence_paroi_scal& loiparth = mod_turb_scal.loi_paroi();
317 if (loiparth->use_equivalent_distance())
318 {
319 const DoubleVect& d_equiv=loiparth->equivalent_distance(n_bord);
320// d_equiv contient la distance equivalente pour le bord
321// Dans d_equiv, pour les faces qui ne sont pas paroi_fixe (eg periodique, symetrie, etc...)
322// il y a la distance geometrique grace a l'initialisation du tableau dans la loi de paroi.
323
324// Les lois de parois ne s'appliquent qu'aux cas ou la CL
325// est de type temperature imposee, car dans les autres cas
326// (flux impose et adiabatique) le flux a la paroi est connu et fixe.
327 int nb_dim_pb=Objet_U::dimension;
328 const Cond_lim_base& cl_base=la_cl.valeur();
329 int ldp_appli=0;
330 if (sub_type(Scalaire_impose_paroi,cl_base))
331 ldp_appli=1;
332 else if (loiparth->get_flag_calcul_ldp_en_flux_impose() )
333 {
334 if ((sub_type(Neumann_paroi,cl_base))||(sub_type(Neumann_homogene,cl_base)))
335 ldp_appli=1;
336 }
337 if (ldp_appli)
338 {
339 DoubleVect le_mauvais_gradient(nb_dim_pb);
340 for (int ind_face=num1; ind_face<num2; ind_face++)
341 {
342// Tf est la temperature fluide moyenne dans le premier element
343// sans tenir compte de la temperature de paroi.
344 double Tf=0.;
345 double bon_gradient=0.; //!< c'est la norme du gradient de temperature normal a la paroi calculee a l'aide de la loi de paroi.
346 le_mauvais_gradient=0.;
347 num_face = le_bord.num_face(ind_face);
348 int elem1 = face_voisins(num_face,0);
349 if(elem1==-1) elem1 = face_voisins(num_face,1);
350 double surface_face=zone_VEF.face_surfaces(num_face);
351 double surface_pond;
352
353 for (i=0; i<nb_faces_elem; i++)
354 {
355 if ( (j= elem_faces(elem1,i)) != num_face )
356 {
357 surface_pond = 0.;
358 for (int kk=0; kk<nb_dim_pb; kk++)
359 surface_pond -= (face_normale(j,kk)*zone_VEF.oriente_normale(j,elem1)*face_normale(num_face,kk)*
360 zone_VEF.oriente_normale(num_face,elem1))/(surface_face*surface_face);
361 Tf+=inconnue(j)*surface_pond;
362 }
363
364 for(int kk=0; kk<nb_dim_pb; kk++)
365 le_mauvais_gradient(kk)+=inconnue(j)*face_normale(j,kk)*zone_VEF.oriente_normale(j,elem1);
366 }
367 le_mauvais_gradient/=vol(elem1);
368// mauvais_gradient = le_mauvais_gradient.n
369 double mauvais_gradient=0;
370 for(int kk=0; kk<nb_dim_pb; kk++)
371 mauvais_gradient+=le_mauvais_gradient(kk)*face_normale(num_face,kk)/surface_face;
372
373// inconnue(num_face) est la temperature de paroi : Tw.
374// On se fiche du signe de bon gradient car c'est la norme du gradient
375// de temperature dans l'element.
376// Ensuite ce sera multiplie par le vecteur normal a la face de paroi
377// qui lui a les bons signes.
378 bon_gradient=(Tf-inconnue(num_face))/d_equiv(ind_face)*(-zone_VEF.oriente_normale(num_face,elem1));
379
380 double nutotal=nu(elem1)+nu_turb(elem1);
381 for (i=0; i<nb_faces_elem; i++)
382 {
383 j= elem_faces(elem1,i);
384 double correction=0.;
385 for(int kk=0; kk<nb_dim_pb; kk++)
386 {
387 double resu2 = nutotal*(bon_gradient-mauvais_gradient)*face_normale(num_face,kk)*face_normale(j,kk)* (-zone_VEF.oriente_normale(j,elem1))/surface_face;
388
389 correction+=resu2;
390 }
391
392 resu(j)+=correction;
393
394 if (marq[j]!=-1)
395 resu[marq[j]]+=correction;
396// la face num_face n'est pas periodique mar(num_face)==-1
397// flux_bord n'est necessaire que sur les faces reelles
398 if(j==num_face && j<size_flux_bords)
399 {
400// flux_bords(j,0)+=resu1;
401 tab_flux_bords(j,0)-=correction;
402 }
403 }
404 }
405 }//!< CL Temperature impose
406 }//!< loi de paroi
407 }//Equation Convection_Diffusion_Turbulente
408
409// Fin de la correction du gradient dans la premiere maille
410// par la loi de paroi thermique.
411 for (int ind_face=num1; ind_face<num2; ind_face++)
412 {
413 num_face = le_bord.num_face(ind_face);
414 int elem1 = face_voisins(num_face,0);
415 d_nu = nu(elem1)+nu_turb(elem1);
416 for (i=0; i<nb_faces_elem; i++)
417 {
418 if (( (j= elem_faces(elem1,i)) > num_face ) || (ind_face>=nb_faces_bord_reel))
419 {
420 valA = viscA(num_face,j,elem1,d_nu);
421 if (ind_face<nb_faces_bord_reel)
422 {
423 double flux=valA*(inconnue(j)-inconnue(num_face));
424 resu(num_face)+=flux;
425 tab_flux_bords(num_face,0) -= flux;
426 }
427 if(j<nb_faces) //!< face reelle
428 {
429 double flux=valA*(inconnue(num_face)-inconnue(j));
430 if (j<premiere_face_int)
431 tab_flux_bords(j,0)-=flux;
432 resu(j)+=flux;
433 }
434 }
435 }
436 }
437 }
438 }
439
440// Faces internes :
441
442 for (num_face=premiere_face_int; num_face<nb_faces; num_face++)
443 {
444 for (int kk=0; kk<2; kk++)
445 {
446 int elem = face_voisins(num_face,kk);
447 d_nu = nu(elem)+nu_turb(elem);
448 for (i=0; i<nb_faces_elem; i++)
449 {
450 if ( (j= elem_faces(elem,i)) > num_face )
451 {
452 int el1,el2;
453 int contrib=1;
454 if(j>=nb_faces) //!< C'est une face virtuelle
455 {
456 el1 = face_voisins(j,0);
457 el2 = face_voisins(j,1);
458 if((el1==-1)||(el2==-1))
459 contrib=0;
460 }
461 if(contrib)
462 {
463 valA = viscA(num_face,j,elem,d_nu);
464 resu(num_face)+=valA*inconnue(j);
465 resu(num_face)-=valA*inconnue(num_face);
466 if(j<nb_faces) //!< On traite les faces reelles
467 {
468 resu(j)+=valA*inconnue(num_face);
469 resu(j)-=valA*inconnue(j);
470 }
471 }
472 }
473 }
474 }
475 }
476
477// Neumann :
478 for (int n_bord=0; n_bord<nb_front; n_bord++)
479 {
480 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
481
482 if (sub_type(Neumann_paroi,la_cl.valeur()))
483 {
484 const Neumann_paroi& la_cl_paroi = ref_cast(Neumann_paroi, la_cl.valeur());
485 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
486 int ndeb = le_bord.num_premiere_face();
487 int nfin = ndeb + le_bord.nb_faces();
488 for (int face=ndeb; face<nfin; face++)
489 {
490 resu[face] += la_cl_paroi.flux_impose(face-ndeb)*zone_VEF.face_surfaces(face);
491 tab_flux_bords(face,0) = la_cl_paroi.flux_impose(face-ndeb)*zone_VEF.face_surfaces(face);
492 }
493 }
494 else if (sub_type(Echange_externe_impose,la_cl.valeur()))
495 {
496 const Echange_externe_impose& la_cl_paroi = ref_cast(Echange_externe_impose, la_cl.valeur());
497 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
498 int ndeb = le_bord.num_premiere_face();
499 int nfin = ndeb + le_bord.nb_faces();
500 for (int face=ndeb; face<nfin; face++)
501 {
502 resu[face] += la_cl_paroi.h_imp(face-ndeb)*(la_cl_paroi.T_ext(face-ndeb)-inconnue(face))*zone_VEF.face_surfaces(face);
503 tab_flux_bords(face,0) = la_cl_paroi.h_imp(face-ndeb)*(la_cl_paroi.T_ext(face-ndeb)-inconnue(face))*zone_VEF.face_surfaces(face);
504 }
505 }
506 else if (sub_type(Neumann_homogene,la_cl.valeur())
507 || sub_type(Symetrie,la_cl.valeur())
508 || sub_type(Neumann_sortie_libre,la_cl.valeur()))
509 {
510 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
511 int ndeb = le_bord.num_premiere_face();
512 int nfin = ndeb + le_bord.nb_faces();
513 for (int face=ndeb; face<nfin; face++)
514 tab_flux_bords(face,0) = 0.;
515 }
516 }

519void Op_Dift_VEF_Face::ajouter_cas_vectoriel(const DoubleTab& inconnue,
520 DoubleTab& resu, DoubleTab& tab_flux_bords,
521 const DoubleTab& nu,
522 const DoubleTab& nu_turb,
523 const Zone_Cl_VEF& zone_Cl_VEF,
524 const Zone_VEF& zone_VEF,
525 int nbr_comp
526 ) const
527{
528 const IntTab& face_voisins = zone_VEF.face_voisins();
529 int nb_faces = zone_VEF.nb_faces();
530 int nb_elem = zone_VEF.nb_elem();
531 const DoubleTab& face_normale = zone_VEF.face_normales();
532
533 assert(nbr_comp>1);
534
535// On dimensionne et initialise le tableau des bilans de flux:
536 tab_flux_bords.resize(zone_VEF.nb_faces_bord(),nbr_comp);
537 tab_flux_bords=0.;
538
539// Construction du tableau grad_ si necessaire
540 if(!grad_.get_md_vector().non_nul())
541 {
542 grad_.resize(0, Objet_U::dimension, Objet_U::dimension);
543 zone_VEF.zone().creer_tableau_elements(grad_);
544 }
545 grad_=0.;
546
547 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
548 int nb_cl=les_cl.size();
549
550 Champ_P1NC::calcul_gradient(inconnue,grad_,zone_Cl_VEF);
551 if (le_modele_turbulence.valeur().utiliser_loi_paroi())
552 Champ_P1NC::calcul_duidxj_paroi(grad_,nu,nu_turb,tau_tan_,zone_Cl_VEF);
553
554 grad_.echange_espace_virtuel();
555
556 DoubleTab Re;
557 Re.resize(0, Objet_U::dimension, Objet_U::dimension);
558 zone_VEF.zone().creer_tableau_elements(Re);
559 Re = 0.;
560
561 if (le_modele_turbulence.valeur().calcul_tenseur_Re(nu_turb, grad_, Re))
562 {
563 Cerr << "On utilise une diffusion turbulente non linaire dans NS" << finl;
564 for (int elem=0; elem<nb_elem; elem++)
565 for (int i=0; i<nbr_comp; i++)
566 for (int j=0; j<nbr_comp; j++)
567 Re(elem,i,j) *= nu_turb[elem];
568 }
569 else
570 {
571 for (int elem=0; elem<nb_elem; elem++)
572 for (int i=0; i<nbr_comp; i++)
573 for (int j=0; j<nbr_comp; j++)
574 Re(elem,i,j) = nu_turb[elem]*(grad_(elem,i,j) + grad_(elem,j,i));
575 }
576 Re.echange_espace_virtuel();
577
578
579// Calcul du terme diffusif
580// boucle sur les CL
581 for (int n_bord=0; n_bord<nb_cl; n_bord++)
582 {
583 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
584 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());

587 int num2 = num1 + le_bord.nb_faces();
588
589 if (sub_type(Periodique,la_cl.valeur()))
590 {
591 for (int num_face=num1; num_face<num2; num_face++)
592 {
593 for (int kk=0; kk<2; kk++)
594 {
595 int elem = face_voisins(num_face,kk);
596 int ori = 1-2*kk;
597
598 for (int i=0; i<nbr_comp; i++)
599 for (int j=0; j<nbr_comp; j++)
600 resu(num_face,i) -= ori*face_normale(num_face,j)
601 *(nu[elem]*grad_(elem,i,j) + Re(elem,i,j));
602 } //!< Fin de la boucle sur les elements ayant la face comnune
603 } //!< Fin de la boucle sur les faces
604 } //!< Fin du cas periodique
605 else
606 {
607 for (int num_face=num1; num_face<num2; num_face++)
608 {
609 int elem=face_voisins(num_face,0);
610
611 for (int i=0; i<nbr_comp; i++)
612 for (int j=0; j<nbr_comp; j++)
613 {
614 double flux = face_normale(num_face,j)
615 *(nu[elem]*grad_(elem,i,j) + Re(elem,i,j));
616 resu(num_face,i) -= flux;
617 tab_flux_bords(num_face,i) -= flux;
618 }
619 } //!< Fin de la boucle sur les faces
620 }
621 } //!< Fin du traitement des CL
622
623//
624// Faces internes
625//
626 int n0 = zone_VEF.premiere_face_int();
627 for (int num_face=n0; num_face<nb_faces; num_face++)
628 {
629 for (int kk=0; kk<2; kk++)
630 {
631 int elem = face_voisins(num_face,kk);
632 int ori = 1-2*kk;
633
634 for (int i=0; i<nbr_comp; i++)
635 for (int j=0; j<nbr_comp; j++)
636 {
637 resu(num_face,i) -= ori*face_normale(num_face,j)
638 *(nu[elem]*grad_(elem,i,j) + Re(elem,i,j));
639 }
640 } //!< Fin de la boucle sur les 2 elements comnuns a la face
641 } //!< Fin de la boucle sur les faces internes
642
643// Symmetry condition
644 for (int n_bord=0; n_bord<nb_cl; n_bord++)
645 {
646 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
647 if (sub_type(Symetrie,la_cl.valeur()))
648 {
649 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
650 int ndeb = le_bord.num_premiere_face();
651 int nfin = ndeb + le_bord.nb_faces();
652 for (int face=ndeb; face<nfin; face++)
653 tab_flux_bords(face,0) = 0.;
654 }
655 }

658void Op_Dift_VEF_Face::ajouter_cas_multi_scalaire(const DoubleTab& inconnue,
659 DoubleTab& resu, DoubleTab& tab_flux_bords,

715 resu(j,nc)+=0.5*valA*inconnue(num_face,nc);
716 resu(j,nc)-=0.5*valA*inconnue(j,nc);

718 }
719 }
720 }
721 }
722 }
723 }
724 else //!< on traite une equation scalaire (pas la vitesse) on a pas a utiliser le tau tangentiel (les lois de paroi thermiques ne calculent pas d'echange turbulent a la paroi pour l'instant
725//
726//
727 {
728 const DoubleTab& face_normale = zone_VEF.face_normales();
729 const DoubleVect& vol = zone_VEF.volumes();
730 const Equation_base& my_eqn = zone_Cl_VEF.equation();

735 const Turbulence_paroi_scal& loiparth = mod_turb_scal.loi_paroi();
736 if (loiparth->use_equivalent_distance())
737 {
738 const DoubleVect& d_equiv=loiparth->equivalent_distance(n_bord);
739// d_equiv contient la distance equivalente pour le bord
740// Dans d_equiv, pour les faces qui ne sont pas paroi_fixe (eg periodique, symetrie, etc...)
741// il y a la distance geometrique grace a l'initialisation du tableau dans la loi de paroi.
742
743// Les lois de parois ne s'appliquent qu'aux cas ou la CL
744// est de type temperature imposee, car dans les autres cas
745// (flux impose et adiabatique) le flux a la paroi est connu et fixe.
746 int nb_dim_pb=Objet_U::dimension;
747 if(sub_type(Scalaire_impose_paroi,la_cl.valeur()))
748 {
749 DoubleVect le_mauvais_gradient(nb_dim_pb);
750 for (int ind_face=num1; ind_face<num2; ind_face++)
751 {
752 for (int nc=0; nc<nb_comp; nc++)
753 {
754// Tf est la temperature fluide moyenne dans le premier element
755// sans tenir compte de la temperature de paroi.
756 double Tf=0.;
757 double bon_gradient=0.; //!< c'est la norme du gradient de temperature normal a la paroi calculee a l'aide de la loi de paroi.
758 le_mauvais_gradient=0.;
759 num_face = le_bord.num_face(ind_face);
760 int elem1 = face_voisins(num_face,0);
761 if(elem1==-1) elem1 = face_voisins(num_face,1);

763 double surface_pond;
764
765 for (i=0; i<nb_faces_elem; i++)
766 {
767 if ( (j= elem_faces(elem1,i)) != num_face )
768 {
769 surface_pond = 0.;
770 for (int kk=0; kk<nb_dim_pb; kk++)
771 surface_pond -= (face_normale(j,kk)*zone_VEF.oriente_normale(j,elem1)*face_normale(num_face,kk)*
772 zone_VEF.oriente_normale(num_face,elem1))/(surface_face*surface_face);
773 Tf+=inconnue(j,nc)*surface_pond;
774 }
775
776 for(int kk=0; kk<nb_dim_pb; kk++)
777 le_mauvais_gradient(kk)+=inconnue(j,nc)*face_normale(j,kk)*zone_VEF.oriente_normale(j,elem1);
778 }
779 le_mauvais_gradient/=vol(elem1);
780// mauvais_gradient = le_mauvais_gradient.n
781 double mauvais_gradient=0;
782 for(int kk=0; kk<nb_dim_pb; kk++)
783 mauvais_gradient+=le_mauvais_gradient(kk)*face_normale(num_face,kk)/surface_face;
784
785// inconnue(num_face) est la temperature de paroi : Tw.
786// On se fiche du signe de bon gradient car c'est la norme du gradient
787// de temperature dans l'element.
788// Ensuite ce sera multiplie par le vecteur normal a la face de paroi
789// qui lui a les bons signes.
790 bon_gradient=(Tf-inconnue(num_face,nc))/d_equiv(ind_face)*(-zone_VEF.oriente_normale(num_face,elem1));
791
792 double nutotal=nu(elem1,nc)+nu_turb(elem1);
793 for (i=0; i<nb_faces_elem; i++)
794 {
795 j= elem_faces(elem1,i);
796 double correction=0.;
797 for(int kk=0; kk<nb_dim_pb; kk++)
798 {
799 double resu2 = nutotal*(bon_gradient-mauvais_gradient)*face_normale(num_face,kk)*face_normale(j,kk)* (-zone_VEF.oriente_normale(j,elem1))/surface_face;
800
801 correction+=resu2;
802 }
803
804 resu(j,nc)+=correction;
805
806 if (marq[j]!=-1)
807 resu(marq[j],nc)+=correction;
808// la face num_face n'est pas periodique mar(num_face)==-1
809// flux_bord n'est necessaire que sur les faces reelles
810 if(j==num_face && j<size_flux_bords)
811 {
812// flux_bords(j,0)+=resu1;
813 tab_flux_bords(j,nc)-=correction;
814 }
815 }
816 }
817 }
818 }//!< CL Temperature imposee
819 }//!< loi de paroi
820 }//Equation Convection_Diffusion_Turbulente
821
822// Fin de la correction du gradient dans la premiere maille
823// par la loi de paroi thermique.
824 for (int ind_face=num1; ind_face<num2; ind_face++)
825 {
826 num_face = le_bord.num_face(ind_face);
827 int elem1 = face_voisins(num_face,0);
828 for (i=0; i<nb_faces_elem; i++)
829 {
830 if (( (j= elem_faces(elem1,i)) > num_face ) || (ind_face>=nb_faces_bord_reel))
831 {
832 for (int nc=0; nc<nb_comp; nc++)
833 {
834 d_nu = nu(elem1,nc)+nu_turb(elem1);
835 valA = viscA(num_face,j,elem1,d_nu);
836 if (ind_face<nb_faces_bord_reel)
837 {
838 resu(num_face,nc)+=valA*inconnue(j,nc);
839 resu(num_face,nc)-=valA*inconnue(num_face,nc);
840// modif pour imprimer les flux sur les bords
841 tab_flux_bords(num_face,nc) -= valA*(inconnue(j,nc)-inconnue(num_face,nc));
842 }
843 if(j<nb_faces) //!< face reelle
844 {
845 resu(j,nc)+=valA*inconnue(num_face,nc);
846 resu(j,nc)-=valA*inconnue(j,nc);
847 }
848 }
849 }
850 }
851 }
852 }
853 }
854
855// Faces internes :
856 const int premiere_face_int=zone_VEF.premiere_face_int();
857 for (num_face=premiere_face_int; num_face<nb_faces; num_face++)
858 {
859 for (int kk=0; kk<2; kk++)
860 {
861 int elem = face_voisins(num_face,kk);
862 for (i=0; i<nb_faces_elem; i++)
863 {
864 if ( (j= elem_faces(elem,i)) > num_face )
865 {
866 int el1,el2;
867 int contrib=1;
868 if(j>=nb_faces) //!< C'est une face virtuelle
869 {
870 el1 = face_voisins(j,0);
871 el2 = face_voisins(j,1);
872 if((el1==-1)||(el2==-1))
873 contrib=0;
874 }
875 if(contrib)
876 {
877 for (int nc=0; nc<nb_comp; nc++)
878 {
879 d_nu = nu(elem,nc)+nu_turb(elem);
880 valA = viscA(num_face,j,elem,d_nu);
881 resu(num_face,nc)+=valA*inconnue(j,nc);
882 resu(num_face,nc)-=valA*inconnue(num_face,nc);
883 if(j<nb_faces) //!< On traite les faces reelles
884 {
885 resu(j,nc)+=valA*inconnue(num_face,nc);
886 resu(j,nc)-=valA*inconnue(j,nc);
887 }
888 }
889 }
890 }
891 }
892 }
893 }
894
895// Neumann :
896 for (int n_bord=0; n_bord<nb_front; n_bord++)
897 {
898 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
899

904 int ndeb = le_bord.num_premiere_face();
905 int nfin = ndeb + le_bord.nb_faces();
906 for (int face=ndeb; face<nfin; face++)

908 for (int nc=0; nc<nb_comp; nc++)

910 resu(face,nc) += la_cl_paroi.flux_impose(face-ndeb,nc)*zone_VEF.face_surfaces(face);
911 tab_flux_bords(face,nc) = la_cl_paroi.flux_impose(face-ndeb,nc)*zone_VEF.face_surfaces(face);
912 }
913 }
914 }
915 else if (sub_type(Echange_externe_impose,la_cl.valeur()))
916 {
917 const Echange_externe_impose& la_cl_paroi = ref_cast(Echange_externe_impose, la_cl.valeur());
918 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
919 int ndeb = le_bord.num_premiere_face();
920 int nfin = ndeb + le_bord.nb_faces();
921 for (int face=ndeb; face<nfin; face++)
922 {
923 for (int nc=0; nc<nb_comp; nc++)
924 {
925 resu(face,nc) += la_cl_paroi.h_imp(face-ndeb,nc)*(la_cl_paroi.T_ext(face-ndeb,nc)-inconnue(face,nc))*zone_VEF.face_surfaces(face);
926 tab_flux_bords(face,nc) = la_cl_paroi.h_imp(face-ndeb,nc)*(la_cl_paroi.T_ext(face-ndeb,nc)-inconnue(face,nc))*zone_VEF.face_surfaces(face);
927 }
928 }
929 }
930 else if (sub_type(Neumann_homogene,la_cl.valeur())
931 || sub_type(Symetrie,la_cl.valeur())
932 || sub_type(Neumann_sortie_libre,la_cl.valeur()))
933 {
934 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
935 int ndeb = le_bord.num_premiere_face();
936 int nfin = ndeb + le_bord.nb_faces();
937 for (int face=ndeb; face<nfin; face++)
938 for (int nc=0; nc<nb_comp; nc++)
939 tab_flux_bords(face,nc) = 0.;
940 }
941 }
