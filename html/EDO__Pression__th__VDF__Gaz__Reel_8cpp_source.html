85double EDO_Pression_th_VDF_Gaz_Reel::resoudre(double Pth_n)
86{
87 int n_bord ;
88 for (n_bord=0; n_bord<la_zone->nb_front_Cl(); n_bord++)
89 {
90 const Cond_lim& la_cl = la_zone_Cl->les_conditions_limites(n_bord);
91 if (sub_type(Neumann_sortie_libre, la_cl.valeur()))
92 return Pth_n;
93 }
94
95 double Pth;
96 const DoubleTab& tab_vit = ref_cast(Navier_Stokes_std,le_fluide_->vitesse()->equation()).vitesse()->valeurs();
97 const DoubleTab& tab_hnp1 = le_fluide_->inco_chaleur().valeurs(); //actuel
98 const DoubleTab& tab_hn = le_fluide_->inco_chaleur().passe(); //passe
99 const DoubleTab& tab_rho = le_fluide_->masse_volumique().valeurs(); //actuel
100 const Loi_Etat& loi_ = le_fluide_->loi_etat();
101// const DoubleVect& tab_rhon = loi_->rho_n(); //passe
102
103 int elem, nb_elem=la_zone->nb_elem(),i;
104 double V = 0; //mesure du domaine
105 double Fn = 0; //integrale 1 a l'etape n
106 double Fnp1 = 0; //integrale 1 a l'etape n+1
107 double S = 0; //second membre
108
109 double dt = le_fluide_->vitesse()->equation().schema_temps().pas_de_temps();
110 double v,al,b, bnp1, hn,hnp1, divu;
111
112 const IntTab& elem_faces = la_zone->elem_faces();
113 DoubleTrav divU(tab_vit.dimension(0), 1);
114 ref_cast(Navier_Stokes_std,le_fluide_->vitesse()->equation()).operateur_divergence().calculer(tab_vit,divU);
115 DoubleTrav gradh(tab_vit.dimension(0));
116 DoubleTrav Hstar(tab_vit.dimension(0));
117 for (elem=0 ; elem<nb_elem ; elem++)
118 {
119 Hstar(elem) = .5*(tab_hn(elem)+tab_hnp1(elem));
120 }
121 calculer_grad(Hstar,gradh);
122 DoubleTab u_gradh(nb_elem);
123 int f1,f2;
124 for (elem=0 ; elem<nb_elem ; elem++)
125 {
126 u_gradh(elem)=0;
127 for (i=0 ; i<dimension ; i++)
128 {
129 f1 = elem_faces(elem,i);
130 f2 = elem_faces(elem,i+dimension);
131 u_gradh(elem) += .25* (gradh(f1)+gradh(f2)) * (tab_vit(f1)+tab_vit(f2));
132 }
133 }
134
135 for (elem=0 ; elem<nb_elem ; elem++)
136 {
137 v = la_zone->volumes(elem);
138 V += v;
139 hn = tab_hn(elem);
140 hnp1 = tab_hnp1(elem);
141 al = loi_->Drho_DT(Pth_n,hn) / loi_->Drho_DP(Pth_n,hn);
142 b = tab_rho(elem) / loi_->Drho_DP(Pth_n,hn);
143 bnp1 = tab_rho(elem) / loi_->Drho_DP(Pth_n,hnp1);
144// S += al.dh/dt
145 S -= v*al * ((hnp1-hn)/dt);
146 divu = divU(elem);
147// S -= al.u.gradT*
148 S -= v*al * u_gradh(elem);
149// F += b.div(U)
150 Fn += v*b * divu;
151 Fnp1 += v*bnp1 * divu;
152 }
153
154 Pth = Pth_n + dt/V *(S-Fn);
155// Cerr<<"Volume="<<V<<finl;
156// Cerr<<"Fn="<<Fn<<finl;
157// Cerr<<"S="<<S<<finl;
158// Cerr<<"Pression thermo recalculee (expl) = "<<Pth<<finl;
159 Pth = Pth_n + dt/V *(S-.5*(Fn+Fnp1));
160// Cerr<<"Fnp1="<<Fnp1<<finl;
161// Cerr<<"Pression thermo recalculee (impl) = "<<Pth<<finl;
162 double tmp=0,r;
163 int k=0;
164 while (std::fabs(tmp-Pth)/Pth>1e-9 && k++<20)
165 {
166 tmp = Pth;
167 Fnp1 = 0;
168 for (elem=0 ; elem<nb_elem ; elem++)
169 {
170 v = la_zone->volumes(elem);
171 hnp1 = tab_hnp1(elem);
172 r = loi_->calculer_masse_volumique(Pth,hnp1);
173 bnp1 = r / loi_->Drho_DP(Pth,hnp1);
174 for (i=0 ; i<dimension ; i++)
175 {
176 Fnp1 += v*bnp1 * (tab_vit(elem_faces(elem,i+dimension))-tab_vit(elem_faces(elem,i)))/la_zone->dim_elem(elem,i);
177 }
178 }
179 Pth = Pth_n + dt/V *(S-.5*(Fn+Fnp1));
180// Cerr<<"Fnp1="<<Fnp1<<finl;
181 Cerr<<"Pression thermo recalculee (impl"<<k<<") = "<<Pth<<finl;
182 }
183
184 return Pth;
185}
