49Reorienter_tetraedres::Sens Reorienter_tetraedres::test_orientation_tetra(IntTab& les_elems, int num_element, const DoubleTab& coord_sommets) const
50{
51 static const int SOM_Z = 0;
52 static const int SOM_A = 1;
53 static const int SOM_B = 2;
54 static const int SOM_C = 3;
55
56 const int som_Z = les_elems(num_element,SOM_Z);
57 const int som_A = les_elems(num_element,SOM_A);
58 const int som_B = les_elems(num_element,SOM_B);
59 const int som_C = les_elems(num_element,SOM_C);
60
61 double ZA[3], ZB[3], ZC[3], pdtvect[3], pdtscal;
62 int k;
63 for (k=0 ; k<dimension ; k++)
64 {
65 ZA[k] = coord_sommets(som_A,k) - coord_sommets(som_Z,k);
66 ZB[k] = coord_sommets(som_B,k) - coord_sommets(som_Z,k);
67 ZC[k] = coord_sommets(som_C,k) - coord_sommets(som_Z,k);
68 }
69
70// calcul pdt vect ZAxZB
71 pdtvect[0] = ZA[1]*ZB[2] - ZA[2]*ZB[1];
72 pdtvect[1] = ZA[2]*ZB[0] - ZA[0]*ZB[2];
73 pdtvect[2] = ZA[0]*ZB[1] - ZA[1]*ZB[0];
74
75// calcul du pdtscal pdtvect.ZC
76 pdtscal = 0.;
77 for (k=0 ; k<dimension ; k++)
78 {
79 pdtscal += pdtvect[k] * ZC[k];
80 }
81
82 if (pdtscal<0.)
83 {
84// le pdt scalaire est negatif : il s'agit d'un tetraedre mal oriente
85#ifdef _AFFDEBUG
86 {
87 Process::Journal<<" element "<<num_element<<" indirect"<<finl;
88 }
89#endif
90 return INDIRECT;
91 }
92#ifdef _AFFDEBUG
93 {
94 Process::Journal<<" element "<<num_element<<" direct"<<finl;
95 }
96#endif
97
98 return DIRECT;
99}

101Reorienter_tetraedres::Sens Reorienter_tetraedres::reorienter_tetra(IntTab& les_elems, int num_element) const
102{
103 static const int SOM_A = 1;
104 static const int SOM_B = 2;
105
106// pour reorienter le tetraedre, on va permuter les sommets 1 et 2
107 int tmp;
108 tmp = les_elems(num_element,SOM_A);
109 les_elems(num_element,SOM_A) = les_elems(num_element,SOM_B);
110 les_elems(num_element,SOM_B) = tmp;
111 return DIRECT;
112}
