128void Zone_PolyMAC::orthocentrer()
129{
130 const IntTab& f_s = face_sommets(), &e_s = zone().les_elems(), &e_f = elem_faces();
131 const DoubleTab& xs = zone().domaine().coord_sommets(), &nf = face_normales_;
132 const DoubleVect& fs = face_surfaces();
133 int i, j, e, f, s, np;
134 DoubleTab M(0, dimension + 1), X(dimension + 1, 1), S(0, 1), vp; //pour les systemes lineaires
135 M.set_smart_resize(1), S.set_smart_resize(1), vp.set_smart_resize(1);
136 IntTrav b_f_ortho, b_e_ortho; //!< b_{f,e}_ortho(f/e) = 1 si la face / l'element est orthocentre
137 creer_tableau_faces(b_f_ortho), zone().creer_tableau_elements(b_e_ortho);
138
139 /* 1. orthocentrage des faces (en dimension 3) */
140 Cerr << zone().domaine().le_nom() << " : ";
141 if (dimension < 3) b_f_ortho = 1; //les faces (segments) sont deja orthcentrees!
142 else for (f = 0; f < nb_faces_tot(); f++)
143 {
144// la face est-elle deja orthocentree?
145 double d2min = DBL_MAX, d2max = 0, d2;
146 for (i = 0, np = 0; i < f_s.dimension(1) && (s = f_s(f, i)) >= 0; i++, np++)
147 d2min = std::min(d2min, d2 = dot(&xs(s, 0), &xs(s, 0), &xv_(f, 0), &xv_(f, 0))), d2max = std::max(d2max, d2);
148 if ((b_f_ortho(f) = (d2max / d2min - 1 < 1e-8))) continue;
149
150// peut-on l'orthocentrer?
151 M.resize(np + 1, 4), S.resize(np + 1, 1);
152 for (i = 0; i < np; i++)
153 for (j = 0, S(i, 0) = 0, M(i, 3) = 1; j < 3; j++)
154 S(i, 0) += 0.5 * std::pow(M(i, j) = xs(f_s(f, i), j) - xv_(f, j), 2);
155 for (j = 0, S(np, 0) = M(np, 3) = 0; j < 3; j++) M(np, j) = nf(f, j) / fs(f);
156 if (kersol(M, S, 1e-12, NULL, X, vp) > 1e-8) continue; //la face n'a pas d'orthocentre
157
158// contrainte : ne pas diminuer la distance entre xv et chaque arete de plus de 50%
159 double r2min = DBL_MAX;
160 for (i = 0; i < f_s.dimension(1) && (s = f_s(f, i)) >= 0; i++)
161 {
162 int s2 = f_s(f, i + 1 < f_s.dimension(1) && f_s(f, i + 1) >= 0 ? i + 1 : 0),
163 a = som_arete[s].at(s2);
164 std::array<double, 3> vec = cross(3, 3, &xv_(f, 0), &ta_(a, 0), &xs(s, 0));
165 r2min = std::min(r2min, dot(&vec[0], &vec[0]));
166 }
167 if (dot(&X(0, 0), &X(0, 0)) > 0.25 * r2min) continue; //on s'eloigne trop du CG
168
169 for (i = 0, b_f_ortho(f) = 1; i < dimension; i++)
170 if (std::fabs(X(i, 0)) > 1e-8) xv_(f, i) += X(i, 0);
171 }
172 Cerr << 100. * mp_somme_vect(b_f_ortho) / Process::mp_sum(nb_faces()) << "% de faces orthocentrees" << finl;
173
174 /* 2. orthocentrage des elements */
175 Cerr << zone().domaine().le_nom() << " : ";
176 for (e = 0; e < nb_elem_tot(); e++)
177 {
178// l'element est-il deja orthocentre?
179 double d2min = DBL_MAX, d2max = 0, d2;
180 for (i = 0, np = 0; i < e_s.dimension(1) && (s = e_s(e, i)) >= 0; i++, np++)
181 d2min = std::min(d2min, d2 = dot(&xs(s, 0), &xs(s, 0), &xp_(e, 0), &xp_(e, 0))), d2max = std::max(d2max, d2);
182 if ((b_e_ortho(e) = (d2max / d2min - 1 < 1e-8))) continue;
183
184// pour qu'on puisse l'orthocentrer, il faut que ses faces le soient
185 for (i = 0, j = 1; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++) j = j && b_f_ortho(f);
186 if (!j) continue;
187
188// existe-il un orthocentre?
189 M.resize(np, dimension + 1), S.resize(np, 1);
190 for (i = 0; i < np; i++)
191 for (j = 0, S(i, 0) = 0, M(i, dimension) = 1; j < dimension; j++)
192 S(i, 0) += 0.5 * std::pow(M(i, j) = xs(e_s(e, i), j) - xp_(e, j), 2);
193 if (kersol(M, S, 1e-12, NULL, X, vp) > 1e-8) continue; //l'element n'a pas d'orthocentre
194
195// contrainte : ne pas diminuer la distance entre xp et chaque face de plus de 50%
196 double rmin = DBL_MAX;
197 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
198 rmin = std::min(rmin, std::fabs(dot(&xp_(e, 0), &nf(f, 0), &xv_(f, 0)) / fs(f)));
199 if (dot(&X(0, 0), &X(0, 0)) > 0.25 * rmin * rmin) continue; //on s'eloigne trop du CG
200
201 for (i = 0, b_e_ortho(e) = 1; i < dimension; i++)
202 if (std::fabs(X(i, 0)) > 1e-8) xp_(e, i) += X(i, 0);
203 }
204 Cerr << 100. * mp_somme_vect(b_e_ortho) / Process::mp_sum(nb_elem()) << "% d'elements orthocentres" << finl;
205}

207/* "clamping" a 0 des coeffs petits dans M1/W1/M2/W2 */
208inline void clamp(DoubleTab& m)
209{
210 for (int i = 0; i < m.dimension(0); i++)
211 for (int j = 0; j < m.dimension(1); j++)
212 for (int n = 0; n < m.dimension(2); n++)
213 if (1e6 * std::abs(m(i, j, n)) < std::abs(m(i, i, n)) + std::abs(m(j, j, n))) m(i, j, n) = 0;
214}

216// matrices locales par elements (operateurs de Hodge) permettant de faire des interpolations :
217// normales aux faces -> tangentes aux faces duales : (nu x_ef.v) = m2 (|f|n_ef.v)
218void Zone_PolyMAC::M2(const DoubleTab *nu, int e, DoubleTab& m2) const
219{
220 int i, j, k, f, n, N = nu ? nu->dimension(1) : 1, e_nu = nu && nu->dimension_tot(0) == 1 ? 0 : e, n_f, d, D = dimension;
221 const IntTab& e_f = elem_faces(), &f_e = face_voisins();
222 const DoubleTab& xe = xp(), &xf = xv(), &nf = face_normales();
223 const DoubleVect& ve = volumes();
224 for (n_f = 0; n_f < e_f.dimension(1) && e_f(e, n_f) >= 0; ) n_f++; //nombre de faces de e
225 double prefac, fac, beta = n_f == D + 1 ? 1. / D : D == 2 ? 1. / sqrt(2) : 1. / sqrt(3); //stabilisation : DGA sur simplexes, SUSHI sinon
226 m2.resize(n_f, n_f, N), m2 = 0;
227 DoubleTrav v_e(n_f, D), v_ef(n_f, n_f, D); //interpolations du vecteur complet : non stabilisee en e, stabilisee en (e, f)
228 for (i = 0; i < n_f; i++)
229 for (f = e_f(e, i), d = 0; d < D; d++) v_e(i, d) = (xf(f, d) - xe(e, d)) / ve(e);
230 for (i = 0; i < n_f; i++)
231 for (f = e_f(e, i), prefac = D * beta / std::abs(dot(&xf(f, 0), &nf(f, 0), &xe(e, 0))), j = 0; j < n_f; j++)
232 for (fac = prefac * ((j == i) - (e == f_e(f, 0) ? 1 : -1) * dot(&nf(f, 0), &v_e(j, 0))), d = 0; d < D; d++)
233 v_ef(i, j, d) = v_e(j, d) + fac * (xf(f, d) - xe(e, d));
234// matrice!
235 for (m2 = 0, i = 0; i < n_f; i++)
236 for (j = 0; j < n_f; j++)
237 if (j < i)
238 for (n = 0; n < N; n++) m2(i, j, n) = m2(j, i, n); //sous la diagonale -> avec l'autre cote
239 else for (k = 0; k < n_f; k++)
240 for (f = e_f(e, k), fac = std::abs(dot(&xf(f, 0), &nf(f, 0), &xe(e, 0))) / D, n = 0; n < N; n++)
241 m2(i, j, n) += fac * nu_dot(nu, e_nu, n, &v_ef(k, i, 0), &v_ef(k, j, 0));
242 clamp(m2);
243}

245// tangentes aux faces duales -> normales aux faces : nu|f|n_ef.v = w2.(x_ef.v)
246void Zone_PolyMAC::W2(const DoubleTab *nu, int e, DoubleTab& w2) const
247{
248 int i, j, k, f, n, N = nu ? nu->dimension(1) : 1, e_nu = nu && nu->dimension_tot(0) == 1 ? 0 : e, n_f, d, D = dimension;
249 const IntTab& e_f = elem_faces(), &f_e = face_voisins();
250 const DoubleTab& xe = xp(), &xf = xv(), &nf = face_normales();
251 const DoubleVect& ve = volumes();
252 for (n_f = 0; n_f < e_f.dimension(1) && e_f(e, n_f) >= 0; ) n_f++; //nombre de faces de e
253 double prefac, fac, beta = n_f == D + 1 ? 1. / D : D == 2 ? 1. / sqrt(2) : 1. / sqrt(3); //stabilisation : DGA sur simplexes, SUSHI sinon
254 w2.resize(n_f, n_f, N), w2 = 0;
255 DoubleTrav v_e(n_f, D), v_ef(n_f, n_f, D); //interpolations du vecteur complet : non stabilisee en e, stabilisee en (e, f)
256 for (i = 0; i < n_f; i++)
257 for (f = e_f(e, i), d = 0; d < D; d++) v_e(i, d) = (e == f_e(f, 0) ? 1 : -1) * nf(f, d) / ve(e);
258 for (i = 0; i < n_f; i++)
259 for (f = e_f(e, i), prefac = D * beta * (e == f_e(f, 0) ? 1 : -1) / std::abs(dot(&xf(f, 0), &nf(f, 0), &xe(e, 0))), j = 0; j < n_f; j++)
260 for (fac = prefac * ((j == i) - dot(&xf(f, 0), &v_e(j, 0), &xe(e, 0))), d = 0; d < D; d++)
261 v_ef(i, j, d) = v_e(j, d) + fac * nf(f, d);
262// matrice!
263 for (i = 0; i < n_f; i++)
264 for (j = 0; j < n_f; j++)
265 if (j < i)
266 for (n = 0; n < N; n++) w2(i, j, n) = w2(j, i, n); //sous-diagonale -> on copie l'autre cote
267 else for (k = 0; k < n_f; k++)
268 for (f = e_f(e, k), fac = std::abs(dot(&xf(f, 0), &nf(f, 0), &xe(e, 0))) / D, n = 0; n < N; n++)
269 w2(i, j, n) += fac * nu_dot(nu, e_nu, n, &v_ef(k, i, 0), &v_ef(k, j, 0));
270 clamp(w2);
271}

273// normales aux aretes duales -> tangentes aux aretes : (nu|a|t_a.v) = m1 (S_ea.v)
274void Zone_PolyMAC::M1(const DoubleTab *nu, int e, DoubleTab& m1) const
275{
276 int i, j, k, f, a, s, sb, sgn, n, N = nu ? nu->dimension(1) : 1, e_nu = nu && nu->dimension_tot(0) == 1 ? 0 : e, n_a, d, D = dimension;
277 const IntTab& e_f = elem_faces(), &f_s = face_sommets(), &e_a = D < 3 ? e_f : zone().elem_aretes(), &a_s = D < 3 ? f_s : zone().aretes_som(); //en 2D, les aretes sont les faces!
278 const DoubleTab& xe = xp(), &xf = xv(), &xs = zone().domaine().coord_sommets();
279 const DoubleVect& ve = volumes();
280 for (n_a = 0; n_a < e_a.dimension(1) && e_a(e, n_a) >= 0;) n_a++; //nombre d'aretes autour de e
281 double prefac, fac, vecz[3] = { 0, 0, 1 }, beta = n_a == 3 * D - 3 ? 1. / D : D == 2 ? 1. / sqrt(2) : 1. / sqrt(3); //stabilisation : DGA sur simplexes, SUSHI sinon
282 m1.resize(n_a, n_a, N), m1 = 0;
283 DoubleTrav S_ea(n_a, D), v_e(n_a, D), v_ea(n_a, n_a, D); //vecteur "surface duale x normale" (oriente comme a), interpolations non stabilisees / stabilisees
284// construction de S_ea
285 if (D < 3)
286 for (i = 0; i < n_a; i++) //2D : arete <-> face, avec orientation du premier au second sommet de f_s
287 {
288 auto vec = cross(D, 3, &xf(f = e_f(e, i), 0), vecz, &xe(e, 0)); //rotation de (xf - xe)
289 for (sgn = dot(&xs(f_s(f, 1), 0), &vec[0], &xs(f_s(f, 0), 0)) > 0 ? 1 : -1, d = 0; d < D; d++) S_ea(i, d) = sgn * vec[d]; //avec la bonne orientation
290 }
291 else for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
292 for (j = 0; j < f_s.dimension(1) && (s = f_s(f, j)) >= 0; j++) //3D: une contribution par couple (f, a)
293 {
294 sb = f_s(f, j + 1 < f_s.dimension(1) && f_s(f, j + 1) >= 0 ? j + 1 : 0); //autre sommet
295 a = som_arete[s].at(sb), k = std::find(&e_a(e, 0), &e_a(e, 0) + n_a, a) - &e_a(e, 0); //arete, son indice dans e_a
296 auto vec = cross(D, D, &xf(f, 0), &xa_(a, 0), &xe(e, 0), &xe(e, 0)); //non oriente
297 for (sgn = dot(&vec[0], &ta_(a, 0)) >= 0 ? 1 : -1, d = 0; d < D; d++) S_ea(k, d) += sgn * vec[d] / 2;
298 }
299// v_e (interpolation non stabilisee)
300 for (i = 0; i < n_a; i++)
301 for (a = e_a(e, i), s = a_s(a, 0), sb = a_s(a, 1), d = 0; d < D; d++) v_e(i, d) = (xs(sb, d) - xs(s, d)) / ve(e);
302// v_ea (interpolation stabilisee)
303 for (i = 0; i < n_a; i++)
304 for (a = e_a(e, i), s = a_s(a, 0), sb = a_s(a, 1), prefac = D * beta / dot(&xs(sb, 0), &S_ea(i, 0), &xs(s, 0)), j = 0; j < n_a; j++)
305 for (fac = prefac * ((j == i) - dot(&S_ea(i, 0), &v_e(j, 0))), d = 0; d < D; d++)
306 v_ea(i, j, d) = v_e(j, d) + fac * (xs(sb, d) - xs(s, d));
307// matrice!
308 for (i = 0; i < n_a; i++)
309 for (j = 0; j < n_a; j++)
310 if (j < i)
311 for (n = 0; n < N; n++) m1(i, j, n) = m1(j, i, n); //sous-diagonale -> on copie l'autre cote
312 else for (k = 0; k < n_a; k++)
313 for (a = e_a(e, k), s = a_s(a, 0), sb = a_s(a, 1), fac = dot(&xs(sb, 0), &S_ea(k, 0), &xs(s, 0)) / D, n = 0; n < N; n++)
314 m1(i, j, n) += fac * nu_dot(nu, e_nu, n, &v_ea(k, i, 0), &v_ea(k, j, 0));
315 clamp(m1);
316}
