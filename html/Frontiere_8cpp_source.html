291//! Ajoute les sommets (et faces) de la frontiere<br>passee en parametre a l'objet (Frontiere).
292void Frontiere::add(const Frontiere& front)
293{
294 const Faces& a_ajouter=front.faces();
295 int nbf1=les_faces.nb_faces();
296 int nbf2=a_ajouter.nb_faces();
297// max to treat the case where my front is empty
298 int nbs=std::max(les_faces.nb_som_faces(), a_ajouter.nb_som_faces());
299 int face;
300 les_faces.les_sommets().resize(nbf1+nbf2, nbs);
301 for(face=0; face<nbf2; face++)
302 for(int som=0; som<nbs; som++)
303 les_faces.sommet(nbf1+face, som)=a_ajouter.sommet(face, som);
304
305 if(a_ajouter.voisins().nb_dim() == 1)
306 return;
307
308 int nb_voisins = a_ajouter.voisins().dimension(1);
309 les_faces.voisins().resize(nbf1+nbf2, nb_voisins);
310 for(face=0; face<nbf2; face++)
311 for(int voisin=0; voisin<nb_voisins; voisin++)
312 les_faces.voisin(nbf1+face, voisin)=a_ajouter.voisin(face, voisin);
313}

324//! Renvoie la trace sur la frontiere du tableau aux elements y
325void Frontiere::trace_elem_local(const DoubleTab& y, DoubleTab& x) const
326{
327 const int size = nb_faces(), nb_compo_ = y.line_size();
328
329// On dimensionne x si ce n'est pas fait
330 if (x.size_array() == 0 && size != 0)
331 x.resize(size, nb_compo_);
332 else if (x.dimension(0) != size || nb_compo_ != x.line_size())
333 {
334 Cerr << "Call to Frontiere::trace_elem with a DoubleTab x not located on boundary faces or don't have the same number of components" << finl;
335 Process::exit();
336 }
337 for (int i = 0; i < size; i++)
338 {
339 int elem = faces().voisin(i, 0);
340 if (elem == -1)
341 elem = faces().voisin(i, 1);
342
343 for(int j = 0; j < nb_compo_; j++)
344 x(i, j) = y(elem, j);
345 }
346}

348//
349//! Renvoie la trace sur la frontiere du tableau aux noeuds y
350void Frontiere::trace_som_local(const DoubleTab& y, DoubleTab& x) const
351{
352 const IntTab& som_face = les_sommets_des_faces();
353 const int size = nb_faces(), nb_compo_ = y.line_size();
354 const int nsomfa = som_face.dimension(1);
355
356// On dimensionne x si ce n'est pas fait
357 if (x.size_array() == 0 && size != 0)
358 x.resize(size, nb_compo_);
359 else if (x.dimension(0) != size || nb_compo_ != x.line_size())
360 {
361 Cerr << "Call to Frontiere::trace_elem with a DoubleTab x not located on boundary faces or don't have the same number of components" << finl;
362 Process::exit();
363 }
364
365 for (int i = 0; i < size; i++)
366 for(int j=0; j<nb_compo_; j++)
367 {
368 double s = 0.;
369 for (int isom = 0; isom < nsomfa; isom++)
370 s += y(som_face(i, isom), j);
371
372 s /= nsomfa;
373 x(i, j) = s;
374 }
375}
