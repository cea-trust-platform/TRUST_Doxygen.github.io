57void Sous_zone_VF::discretiser()
58{
59 Cerr << "Sous_zone_VF::discretiser : discretization of the subarea "
60 << la_sous_zone->le_nom() << "... ";
61
62 la_zone_VF=ref_cast(Zone_VF,la_zone_dis.valeur());
63
64 int nb_faces_tot=la_zone_VF->nb_faces_tot();
65
66// Stockages temporaires :
67 IntTab faces_internes;
68 faces_internes.resize(nb_faces_tot);
69 int nb_faces_internes=0;
70
71 IntTab faces_bord_0;
72 faces_bord_0.resize(nb_faces_tot);
73 int nb_faces_bord_0=0;
74
75 IntTab faces_bord_1;
76 faces_bord_1.resize(nb_faces_tot);
77 int nb_faces_bord_1=0;
78
79 IntTab faces_bord;
80 faces_bord.resize(nb_faces_tot);
81 int nb_faces_bord=0;
82
83 IntTab elem_app;
84 elem_app.resize(la_zone_VF->nb_elem_tot());
85 elem_app=0;
86
87// Remplissage de elem_app
88 for (int i=0; i<la_sous_zone->nb_elem_tot(); i++)
89 elem_app[la_sous_zone.valeur()(i)]=1;
90
91// Boucle sur les faces
92 for (int i=0; i<nb_faces_tot; i++)
93 {
94 int elem0,elem1;
95// Quels elements lui sont voisins ?
96 elem0=la_zone_VF->face_voisins(i,0);
97 elem1=la_zone_VF->face_voisins(i,1);
98// Remplissage des tableaux temporaires.
99 if (elem1==-1) //!< face au bord de la zone
100 {
101 if (elem_app[elem0]) //!< dans la sous_zone
102 {
103 faces_bord(nb_faces_bord++)=i;
104 }
105 }
106 else if (elem0==-1) //!< face au bord de la zone
107 {
108 if (elem_app[elem1]) //!< dans la sous_zone
109 {
110 faces_bord(nb_faces_bord++)=i;
111 }
112 }
113 else //!< face interne a la zone
114 {
115 if (elem_app[elem0] && elem_app[elem1]) //!< interne a la sous_zone
116 {
117 faces_internes(nb_faces_internes++)=i;
118 }
119 else if (elem_app[elem0]) //!< seul elem0 est dans la sous-zone
120 {
121 faces_bord_0(nb_faces_bord_0++)=i;
122 }
123 else if (elem_app[elem1]) //!< seul elem1 est dans la sous-zone
124 {
125 faces_bord_1(nb_faces_bord_1++)=i;
126 }
127 }
128 }
129
130// Regroupement des resultats
131 premiere_face_bord_0_ = nb_faces_internes;
132 premiere_face_bord_1_ = premiere_face_bord_0_+nb_faces_bord_0;
133 premiere_face_bord_ = premiere_face_bord_1_+nb_faces_bord_1;
134 les_faces_.resize(premiere_face_bord_+nb_faces_bord);
135 for (int i=0; i<nb_faces_internes; i++)
136 les_faces_(i)=faces_internes(i);
137 for (int i=0; i<nb_faces_bord_0; i++)
138 les_faces_(premiere_face_bord_0_+i)=faces_bord_0(i);
139 for (int i=0; i<nb_faces_bord_1; i++)
140 les_faces_(premiere_face_bord_1_+i)=faces_bord_1(i);
141 for (int i=0; i<nb_faces_bord; i++)
142 les_faces_(premiere_face_bord_+i)=faces_bord(i);
143
144// Calcul des volumes entrelaces
145 const DoubleVect& volumes=la_zone_VF->volumes();
146 volumes_entrelaces_.resize(nb_faces_bord_0+nb_faces_bord_1);
147 for (int i=0; i<nb_faces_bord_0; i++)
148 {
149 volumes_entrelaces_(i)=volumes(la_zone_VF->face_voisins(les_faces_(premiere_face_bord_0_+i),0));
150 }
151 for (int i=0; i<nb_faces_bord_1; i++)
152 {
153 volumes_entrelaces_(nb_faces_bord_0+i)=volumes(la_zone_VF->face_voisins(les_faces_(premiere_face_bord_1_+i),1));
154 }
155 Cerr << "OK" << finl;
156
157// Affichage du volume des sous zones (ca peut etre pratique)
158 double volume_sous_zone=0;
159 for (int j=0; j<la_sous_zone->nb_elem_tot(); j++)
160 {
161// On ne compte que les mailles reelles
162 if (la_sous_zone.valeur()(j)<la_zone_VF->zone().nb_elem())
163 volume_sous_zone+=volumes(la_sous_zone.valeur()(j));
164 }
165 volume_sous_zone=mp_sum(volume_sous_zone);
166 if (je_suis_maitre())
167 Cerr << "Volume of subarea " << la_sous_zone->le_nom() << ": " << volume_sous_zone << finl;
168}
