71void Traitement_particulier_NS_canal_VEF::remplir_Y(DoubleVect& tabY, DoubleVect& tabcompt, int& aNy) const
72{
73// On va initialiser les differents parametres membres de la classe
74// utiles au calcul des differentes moyennes
75// Initialisation de : Y, compt
76
77 const Zone_dis_base& zdisbase = mon_equation->inconnue().zone_dis_base();
78 const Zone_VF& zone_VF=ref_cast(Zone_VF, zdisbase);
79 const DoubleTab& xv = zone_VF.xv();
80 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF, zdisbase);
81 const ArrOfInt& faces_doubles = zone_VEF.faces_doubles();
82 int nb_faces = zone_VF.nb_faces();
83 int num_face,j,indic,trouve;
84 double y;
85
86 j=0;
87 indic = 0;
88
89 tabY.resize(200);
90 tabcompt.resize(200);
91
92 tabY = -100.;
93 tabcompt = 0;
94
95// Remplissage du tableau Y
96//
97
98// GF PQ. on ajoute 2 a compt dans le cas des faces standards, 1 pour les faces doubles
99// et on divise compt a la fin par 2.
100// pour etre coherent sur le nombre de faces vis a vis de calculer_moyenne_spatiale_vitesse_rho_mu
101
102 for (num_face=0; num_face<nb_faces; num_face++)
103 {
104 int c = (faces_doubles[num_face]==1) ? 1 : 2 ;
105 y = xv(num_face,1);
106 trouve = 0;
107
108 for (j=0; j<indic+1; j++)
109 {
110 if(est_egal(y,tabY[j]))
111 {
112 tabcompt[j] +=c;
113 j=indic+1;
114 trouve = 1;
115 break;
116 }
117 }
118 if (trouve==0)
119 {
120 tabY[indic]=y;
121 tabcompt[indic] +=c;
122 indic++;
123
124 tabY.resize(indic+1);
125 tabY(indic)=-100.;
126 tabcompt.resize(indic+1);
127 }
128 }
129
130 aNy = indic;
131
132 tabY.resize(aNy);
133 tabcompt.resize(aNy);
134 tabcompt/=2;
135}

138void Traitement_particulier_NS_canal_VEF::calculer_moyenne_spatiale_vitesse_rho_mu(DoubleTab& val_moy) const
139{
140 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
141 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF, zdisbase);
142 const DoubleTab& xv = zone_VEF.xv();
143 const IntTab& face_voisins = zone_VEF.face_voisins();
144 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
145 double y,u,v,wl;
146 int nb_faces = zone_VEF.nb_faces();
147 int num_face,i,elem0,elem1;
148 const ArrOfInt& faces_doubles = zone_VEF.faces_doubles();
149 double c;
150
151 const Fluide_base& le_fluide = ref_cast(Fluide_base,mon_equation->milieu());
152 const DoubleTab& visco_dyn = le_fluide.viscosite_dynamique()->valeurs();
153 const DoubleTab& tab_rho_face = le_fluide.masse_volumique().valeurs();
154 int taille_mu=visco_dyn.dimension(0);
155 int taille_rho=tab_rho_face.dimension(0);
156
157 if (visco_dyn.nb_dim()>2)
158 {
159 Cerr<<"Error in Traitement_particulier_NS_canal_VEF::calculer_moyenne_spatiale_vitesse_rho_mu()"<<finl;
160 Process::exit();
161 }
162
163 for (num_face=0; num_face<nb_faces; num_face++)
164 {
165 c = (faces_doubles[num_face]==1) ? 0.5 : 1. ;
166
167 y=xv(num_face,1);
168
169 u = vitesse(num_face,0);
170 v = vitesse(num_face,1);
171
172 for (i=0; i<Ny; i++)
173 if(est_egal(y,Y[i])) break;
174
175 val_moy(i,0) += c*u;
176 val_moy(i,1) += c*v;
177 val_moy(i,3) += c*u*u;
178 val_moy(i,4) += c*v*v;
179 val_moy(i,6) += c*u*v;
180
181 if(dimension==2) val_moy(i,9) += c*sqrt(u*u); //vitesse tangentielle pour calcul du frottement
182
183 if(dimension==3)
184 {
185
186 wl = vitesse(num_face,2);
187
188 val_moy(i,2) += c*wl;
189 val_moy(i,5) += c*wl*wl;
190 val_moy(i,7) += c*u*wl;
191 val_moy(i,8) += c*v*wl;
192 val_moy(i,9) += c*sqrt(u*u+wl*wl); //vitesse tangentielle pour calcul du frottement
193 }
194
195
196 if (taille_rho==1) val_moy(i,10) += c*tab_rho_face(0,0);
197 else val_moy(i,10) += c*tab_rho_face[num_face];
198
199
200
201 elem0 = face_voisins(num_face,0);
202 elem1 = face_voisins(num_face,1);
203
204 if (taille_mu==1)
205 val_moy(i,11) += c*visco_dyn(0,0);
206 else
207 {
208 if (elem1!=-1)
209 {
210 val_moy(i,11) += c*0.5*(visco_dyn(elem0,0)+visco_dyn(elem1,0));
211 }
212 else
213 {
214 val_moy(i,11) += c*visco_dyn(elem0,0);
215 }
216 }
217 }

259void Traitement_particulier_NS_canal_VEF::calculer_moyenne_spatiale_Temp(DoubleTab& val_moy) const
260{
261 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
262 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF, zdisbase);
263 const DoubleTab& xv = zone_VEF.xv();
264 const DoubleTab& temperature = Temp.valeur().valeurs();
265 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
266 double y,u,v,wl,T;
267 int nb_faces = zone_VEF.nb_faces();
268 int num_face,i;
269 const ArrOfInt& faces_doubles = zone_VEF.faces_doubles();
270 double c;
271
272
273 for (num_face=0; num_face<nb_faces; num_face++)
274 {
275 c = (faces_doubles[num_face]==1) ? 0.5 : 1. ;
276
277 y=xv(num_face,1);
278
279 T = temperature[num_face];
280 u = vitesse(num_face,0);
281 v = vitesse(num_face,1);
282
283 for (i=0; i<Ny; i++)
284 if(est_egal(y,Y[i])) break;
285
286 val_moy(i,13) += c*T;
287 val_moy(i,14) += c*T*T;
288 val_moy(i,15) += c*u*T;
289 val_moy(i,16) += c*v*T;
290
291 if(Objet_U::dimension==3)
292 {
293 wl = vitesse(num_face,2);
294
295 val_moy(i,17) += c*wl*T;
296 }
297 }
