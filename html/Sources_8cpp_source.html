64//! Lecture d'une liste de sources sur un flot d'entree.<br>Lit une liste de sources separees par des virgules<br>et le sajoute a la liste.<br>format:<br>{<br>bloc de lecture d'une source<br>[, bloc de lecture d'une source]<br>...<br>}
65/** \return Entree\&<br>
66Signification: le flot d'entree modifie<br>
67Contraintes: */
68Entree& Sources::readOn(Entree& is)
69{
70 Nom accouverte="{";
71 Nom accfermee="}";
72 Nom virgule=",";
73 Nom typ;
74 is >> typ;
75 if (typ!=accouverte)
76 {
77 Cerr << "We were waiting for { before " << typ << finl;
78 exit();
79 }
80 Source t;
81 is >> typ;
82 if(typ==accfermee)
83 return is;
84 while(1)
85 {
86 Source& so=add(t);
87 Cerr << "Reading of term " << typ << finl;
88 Cerr << "and typing: ";
89// Cout << ">>>>>>>>>>>>>>>>>>>> Type de Source = " << typ << finl;
90 so.typer(typ,mon_equation.valeur());
91 so->associer_eqn(mon_equation.valeur());
92 is >> so.valeur();
93 is >> typ;
94 if(typ==accfermee)
95 return is;
96 if(typ!=virgule)
97 {
98 Cerr << typ << " : we expected a ',' or a '}'" << finl;
99 exit();
100 }
101 assert (typ==virgule);
102 is >> typ;
103 }
104}
