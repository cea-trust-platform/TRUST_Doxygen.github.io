68double Op_Diff_PolyMAC_Face::calculer_dt_stab() const
69{
70 const Zone_PolyMAC& zone = la_zone_poly_.valeur();
71 const IntTab& e_f = zone.elem_faces();
72 const DoubleTab& nf = zone.face_normales(),
73 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL,
74 *a_r = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.champ_conserve().passe() : (has_champ_masse_volumique() ? &get_champ_masse_volumique().valeurs() : NULL); /* produit alpha * rho */
75 const DoubleVect& pe = zone.porosite_elem(), &vf = zone.volumes_entrelaces(), &ve = zone.volumes();
76 update_nu();
77
78 int i, e, f, n, N = equation().inconnue().valeurs().line_size();
79 double dt = 1e10;
80 DoubleTrav flux(N), vol(N);
81 for (e = 0; e < zone.nb_elem(); e++)
82 {
83 for (flux = 0, vol = pe(e) * ve(e), i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
84 for (n = 0; n < N; n++)
85 flux(n) += zone.nu_dot(&nu_, e, n, &nf(f, 0), &nf(f, 0)) / vf(f);
86 for (n = 0; n < N; n++)
87 if ((!alp || (*alp)(e, n) > 0.25) && flux(n)) /* sous 0.5e-6, on suppose que l'evanescence fait le job */
88 dt = std::min(dt, vol(n) * (a_r ? (*a_r)(e, n) : 1) / flux(n));
89 }
90 return Process::mp_min(dt);
91}

93void Op_Diff_PolyMAC_Face::dimensionner_blocs_ext(int aux_only, matrices_t matrices, const tabs_t& semi_impl) const
94{
95 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, le_champ_inco.non_nul() ? le_champ_inco.valeur() : equation().inconnue().valeur());
96 const Zone_PolyMAC& zone = la_zone_poly_.valeur();
97 const IntTab& e_f = zone.elem_faces(), &f_s = zone.face_sommets(), &e_a = zone.zone().elem_aretes(), &fcl = ch.fcl();
98 const std::string& nom_inco = ch.le_nom().getString();
99 if (!matrices.count(nom_inco)) return; //pas de bloc diagonal -> rien a faire
100 Matrice_Morse& mat = *matrices.at(nom_inco), mat2;
101 update_nu();
102 ConstDoubleTab_parts p_inco(ch.valeurs());
103 int i, j, k, e, f, fb, a, ab, s, n, N = ch.valeurs().line_size(), nf_tot = zone.nb_faces_tot(), d, db, D = dimension, N_nu = nu_.line_size(), semi = semi_impl.count(nom_inco);
104
105 IntTrav stencil(0, 2);
106 stencil.set_smart_resize(1);
107 Cerr << "Op_Diff_PolyMAC_Face::dimensionner() : ";
108
109 /* bloc (faces, aretes) : rot [(lambda grad)^u]*/
110 if (!semi && !aux_only)
111 for (f = 0; f < zone.nb_faces(); f++)
112 for (i = 0; i < f_s.dimension(1) && (s = f_s(f, i)) >= 0; i++)
113 for (a = D < 3 ? s : zone.som_arete[s].at(f_s(f, i + 1 < f_s.dimension(1) && f_s(f, i + 1) >= 0 ? i + 1 : 0)), n = 0; n < N; n++)
114 stencil.append_line(N * f + n, N * (nf_tot + a) + n);
115
116 /* blocs (aretes, faces) et (aretes, aretes) : avec M2 et W1 dans chaque element */
117 Matrice33 L(0, 0, 0, 0, 0, 0, 0, 0, D < 3), iL; //tenseur de diffusion dans chaque element, son inverse
118 DoubleTrav inu, m2, w1; //au format compris par zone.nu_dot
119 nu_.nb_dim() == 2 ? inu.resize(1, N) : nu_.nb_dim() == 3 ? inu.resize(1, N, D) : inu.resize(1, N, D, D);
120 m2.set_smart_resize(1), w1.set_smart_resize(1);
121 if (!semi)
122 for (e = 0; e < zone.nb_elem_tot(); e++)
123 {
124// tenseur de diffusion diagonal ou anisotrope diagonal : inverse faciles!
125 if (nu_.nb_dim() < 4)
126 for (i = 0; i < N_nu; i++) inu.addr()[i] = 1. / nu_.addr()[N_nu * e + i];
127 else for (n = 0; n < N; n++) //sinon : une matrice a inverser par composante
128 {
129 for (d = 0; d < D; d++)
130 for (db = 0; db < D; db++) L(d, db) = nu_(e, n, d, db);
131 Matrice33::inverse(L, iL);
132 for (d = 0; d < D; d++)
133 for (db = 0; db < D; db++) inu(0, n, d, db) = iL(d, db);
134 }
135 zone.M2(&inu, e, m2);
136 if (D > 2) zone.W1(&nu_, e, w1); //uniquement en 3D: en 2D, matrice diagonale
137
138// bloc (aretes, faces): en parcourant les aretes de chaque face
139 if (!aux_only)
140 for (i = 0; i < m2.dimension(0); i++)
141 for (f = e_f(e, i), j = 0; j < f_s.dimension(1) && (s = f_s(f, j)) >= 0; j++)
142 if ((a = D < 3 ? s : zone.som_arete[s].at(f_s(f, j + 1 < f_s.dimension(1) && f_s(f, j + 1) >= 0 ? j + 1 : 0))) < (D < 3 ? zone.zone().nb_som() : zone.zone().nb_aretes()))
143 for (k = 0; k < m2.dimension(1); k++)
144 for (fb = e_f(e, k), n = 0; n < N; n++)
145 if (fcl(f, 0) == 2 || m2(i, k, n)) //si f est Symetrie, alors il y a aussi une partie en ve -> dependance complete
146 stencil.append_line(N * (nf_tot + a) + n, N * fb + n);
147// bloc (aretes, aretes) : avec m1 si D = 3 (sinon, fait ensuite)
148 if (D > 2)
149 for (i = 0; i < w1.dimension(0); i++)
150 if ((a = e_a(e, i)) < zone.zone().nb_aretes())
151 for (j = 0; j < w1.dimension(1); j++)
152 for (ab = e_a(e, j), n = 0; n < N; n++)
153 if (w1(i, j, n)) stencil.append_line(N * (!aux_only * nf_tot + a) + n, N * (!aux_only * nf_tot + ab) + n);
154 }
155 if (semi || D < 3)
156 for (s = 0; s < (D < 3 ? zone.nb_som() : zone.zone().nb_aretes()); s++)
157 for (n = 0; n < N; n++) stencil.append_line(N * (!aux_only * nf_tot + s) + n, N * (!aux_only * nf_tot + s) + n);
158
159 tableau_trier_retirer_doublons(stencil);
160 Cerr << "OK" << finl;
161 Matrix_tools::allocate_morse_matrix(aux_only ? p_inco[1].size_totale() : ch.valeurs().size_totale(), aux_only ? p_inco[1].size_totale() : ch.valeurs().size_totale(), stencil, mat2);
162 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
163}

167void Op_Diff_PolyMAC_Face::ajouter_blocs_ext(int aux_only, matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
168{
169 const Champ_Face_PolyMAC& ch = ref_cast(Champ_Face_PolyMAC, le_champ_inco.non_nul() ? le_champ_inco.valeur() : equation().inconnue().valeur());
170 const Conds_lim& cls = ch.zone_Cl_dis().les_conditions_limites();
171 const Zone_PolyMAC& zone = la_zone_poly_.valeur();
172 const IntTab& e_f = zone.elem_faces(), &f_s = zone.face_sommets(), &e_a = zone.zone().elem_aretes(), &fcl = ch.fcl(), &f_e = zone.face_voisins();
173 const std::string& nom_inco = ch.le_nom().getString();
174 Matrice_Morse* mat = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
175 update_nu();
176 int i, j, k, e, f, fb, a, ab, s, n, N = ch.valeurs().line_size(), nf_tot = zone.nb_faces_tot(), d, db, D = dimension, N_nu = nu_.line_size(), semi = semi_impl.count(nom_inco);
177 double vecz[3] = { 0, 0, 1 }, v_cl[3], t = equation().schema_temps().temps_courant();
178 const DoubleTab& xp = zone.xp(), &xv = zone.xv(), &xs = zone.zone().domaine().coord_sommets(), &xa = D < 3 ? xs : zone.xa(), &ta = zone.ta(), &nf = zone.face_normales(),
179 &inco = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : ch.valeurs();
180 const DoubleVect& la = zone.longueur_aretes(), &vf = zone.volumes_entrelaces(), &fs = zone.face_surfaces(), &ve = zone.volumes();
181
182 /* que faire avec les variables auxiliaires ? */
183 if (aux_only) use_aux_ = 0; /* 1) on est en train d'assembler le systeme de resolution des variables auxiliaires lui-meme */
184 else if (mat && !semi) t_last_aux_ = t, use_aux_ = 0; /* 2) on est en implicite complet : pas besoin de mat_aux / var_aux */
185 else if (t_last_aux_ < t) update_aux(t); /* 3) premier pas a ce temps en semi-implicite : on calcule les variables auxiliaires a t et on les stocke dans var_aux */
186 ConstDoubleTab_parts p_inco(inco); /* deux parties de l'inconnue */
187 const DoubleTab& omega = use_aux_ ? var_aux : p_inco[1]; /* les variables auxiliaires peuvent etre soit dans inco/semi_impl (cas 1), soit dans var_aux (cas 2) */
188
189 /* bloc (faces, aretes) : rot [(lambda grad)^u]*/
190 if (!aux_only)
191 for (f = 0; f < zone.nb_faces(); f++)
192 for (i = 0; i < f_s.dimension(1) && (s = f_s(f, i)) >= 0; i++)
193 {
194 a = D < 3 ? s : zone.som_arete[s].at(f_s(f, i + 1 < f_s.dimension(1) && f_s(f, i + 1) >= 0 ? i + 1 : 0)); //indice d'arete
195 auto vec = zone.cross(3, D, D < 3 ? vecz : &ta(a, 0), &xv(f, 0), NULL, &xa(a, 0));
196 int sgn = zone.dot(&nf(f, 0), &vec[0]) > 0 ? 1 : -1; //orientation arete-face
197 for (n = 0; n < N; n++) secmem(f, n) -= sgn * vf(f) / fs(f) * (D < 3 ? 1 : la(a)) * omega(a, n);
198 if (mat && !semi)
199 for (n = 0; n < N; n++) (*mat)(N * f + n, N * (nf_tot + a) + n) += sgn * vf(f) / fs(f) * (D < 3 ? 1 : la(a));
200 }
201
202 /* blocs (aretes, faces) et (aretes, aretes) : avec M2 et W1 dans chaque element */
203 Matrice33 L(0, 0, 0, 0, 0, 0, 0, 0, D < 3), iL; //tenseur de diffusion dans chaque element, son inverse et le carre de celui-ci
204 DoubleTrav dL(N), inu, m2, w1; //determinant, inverse (au format compris par zone.nu_dot), matrices M2(iL) / W1(L)
205 nu_.nb_dim() == 2 ? inu.resize(1, N) : nu_.nb_dim() == 3 ? inu.resize(1, N, D) : inu.resize(1, N, D, D);
206 m2.set_smart_resize(1), w1.set_smart_resize(1);
207 if (!aux_only && mat && semi)
208 for (a = 0; a < xa.dimension(0); a++)
209 for (n = 0; n < N; n++) /* en semi-implicite : egalites w_a^+ = var_aux */
210 secmem(nf_tot + a, n) += omega(a, n) - ch.valeurs()(nf_tot + a, n), (*mat)(N * (nf_tot + a) + n, N * (nf_tot + a) + n)++;
211 else if (mat && !semi)
212 for (e = 0; e < zone.nb_elem_tot(); e++) /* en implicite : vraies equations */
213 {
214// tenseur de diffusion diagonal ou anisotrope diagonal : inverses faciles!
215 if (nu_.nb_dim() == 2)
216 for (n = 0; n < N; n++) inu(0, n) = 1. / nu_(e, n), dL(n) = std::pow(nu_(e, n), D);
217 else if (nu_.nb_dim() == 3)
218 for (n = 0; n < N; n++)
219 for (d = 0, dL(n) = 1; d < D; d++) inu(0, n, d) = 1. / nu_(e, n, d), dL(n) *= nu_(e, n, d);
220 if (nu_.nb_dim() < 4)
221 for (i = 0; i < N_nu; i++) inu.addr()[i] = 1. / nu_.addr()[N_nu * e + i];
222 else for (n = 0; n < N; n++) //sinon : une matrice a inverser par composante
223 {
224 for (d = 0; d < D; d++)
225 for (db = 0; db < D; db++) L(d, db) = nu_(e, n, d, db);
226 dL(n) = Matrice33::inverse(L, iL); //renvoie le determinant!
227 for (d = 0; d < D; d++)
228 for (db = 0; db < D; db++) inu(0, n, d, db) = iL(d, db);
229 }
230 zone.M2(&inu, e, m2);
231 if (D > 2) zone.W1(&nu_, e, w1); //uniquement en 3D: en 2D, matrice diagonale
232
233// bloc (aretes, faces): en parcourant les aretes de chaque face
234 for (i = 0; i < m2.dimension(0); i++)
235 for (f = e_f(e, i), j = 0; j < f_s.dimension(1) && (s = f_s(f, j)) >= 0; j++)
236 if ((a = D < 3 ? s : zone.som_arete[s].at(f_s(f, j + 1 < f_s.dimension(1) && f_s(f, j + 1) >= 0 ? j + 1 : 0))) < (D < 3 ? zone.zone().nb_som() : zone.zone().nb_aretes()))
237 {
238 auto vec = zone.cross(3, D, D < 3 ? vecz : &ta(a, 0), &xv(f, 0), NULL, &xa(a, 0));
239 int sgn = (e == f_e(f, 0) ? 1 : -1) * zone.dot(&nf(f, 0), &vec[0]) > 0 ? 1 : -1; //orientation arete-face (dans le sens sortant de e)
240 for (k = 0; k < m2.dimension(1); k++)
241 for (fb = e_f(e, k), n = 0; n < N; n++)//partie x_e -> x_f avec m2 + partie x_f -> x_a avec v_e si bord de Neumann / Symetrie
242 {
243 double coeff = m2(i, k, n) + (fcl(f, 0) == 2 ? zone.nu_dot(&inu, 0, n, &xa(a, 0), &xv(fb, 0), &xv(f, 0), &xp(e, 0)) / ve(e) : 0);
244 if (!coeff) continue;
245 secmem(!aux_only * nf_tot + a, n) += sgn * coeff * inco(fb, n) * fs(fb) * (e == f_e(fb, 0) ? 1 : -1);
246 if (!aux_only) (*mat)(N * (nf_tot + a) + n, N * fb + n) -= sgn * coeff * fs(fb) * (e == f_e(fb, 0) ? 1 : -1);
247 }
248 if (fcl(f, 0) == 3 && sub_type(Dirichlet, cls[fcl(f, 1)].valeur()))
249 for (n = 0; n < N; n++) //si bord de Dirichlet : partie x_f -> x_a avec la vitesse donnee par la CL
250 {
251 for (d = 0; d < D; d++) v_cl[d] = ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), N * d + n); //v impose
252 secmem(!aux_only * nf_tot + a, n) += sgn * zone.nu_dot(&inu, 0, n, &xa(a, 0), v_cl, &xv(f, 0));
253 }
254 }
255
256// bloc (aretes, aretes) : avec m1 si D = 3, diagonale * surface si D = 2
257 if (D == 2)
258 {
259 for (i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
260 for (j = 0; j < 2; j++)
261 if ((s = f_s(f, j)) < zone.nb_som())
262 {
263 auto vec = zone.cross(D, D, &xv(f, 0), &xs(s, 0), &xp(e, 0), &xp(e, 0));
264 double surf = std::abs(vec[2]) / 2; //surface du triangle (e, f, s)
265 for (n = 0; n < N; n++) secmem(!aux_only * nf_tot + s, n) -= surf * inco(nf_tot + s, n) / dL(n);
266 for (n = 0; n < N; n++) (*mat)(N * (!aux_only * nf_tot + s) + n, N * (!aux_only * nf_tot + s) + n) += surf / dL(n);
267 }
268 }
269 else for (i = 0; i < w1.dimension(0); i++)
270 if ((a = e_a(e, i)) < zone.zone().nb_aretes())
271 for (j = 0; j < w1.dimension(1); j++)
272 for (ab = e_a(e, j), n = 0; n < N; n++)
273 if (w1(i, j, n))
274 {
275 secmem(!aux_only * nf_tot + a, n) -= w1(i, j, n) * la(ab) * inco(nf_tot + ab, n) / dL(n);
276 (*mat)(N * (!aux_only * nf_tot + a) + n, N * (!aux_only * nf_tot + ab) + n) += w1(i, j, n) * la(ab) / dL(n);
277 }
278 }
279 i++;
280}
