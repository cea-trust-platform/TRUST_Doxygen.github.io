120 const int nb_faces_elem=elem_faces.dimension(1);

122 assert(nb_faces_elem==(dimension+1));
123 {
124// calcul de la CFL.
125 DoubleVect& fluent_ = fluent;
126 double psc;

136 psc+=vitesse_face_absolue(num_face,i)*face_normales(num_face,i);

144// soit on a transporte=phi*transporte_ et vitesse_face=vitesse_

146// cela d~pend si on transporte avec phi u ou avec u.
147 const DoubleTab& transporte=modif_par_porosite_si_flag(transporte_2,transporte_,!marq,porosite_face);
148 const DoubleTab& tab_vitesse=modif_par_porosite_si_flag(vitesse_face_absolue,vitesse_face_,marq,porosite_face);
149// DoubleTab ubar(transporte);

157 if (transporte_bar) ch.filtrer_L2(uT);
158 if (transportant_bar) la_vitesse.filtrer_L2(u);
159
160 const int nb_comp =resu.line_size();
161 DoubleTab grad_uT(nb_comp, dimension);
162 DoubleTab grad_phi(nb_faces_elem,dimension);
163 int elem, num_face, j, i, face;
164 double d_inv=(1./nb_faces_elem);
165
166 resu=0;
167 DoubleTab sigma(dimension, dimension);
168 DoubleTab contrib_elem(nb_faces_elem, nb_comp);
169 for (elem=0; elem<nb_elem_tot; elem++)
170 {
171// for elem
172 sigma=0, contrib_elem=0, grad_uT=0., grad_phi=0.;
173 for(num_face=0; num_face<nb_faces_elem; num_face++)
174 {
175// for num_face
176 face=elem_faces(elem, num_face);
177 double cc=1.;
178 if(face_voisins(face,0)!=elem)
179 cc*=-1;
180 for (i=0; i<dimension; i++)
181 {
182 grad_phi(num_face,i)=face_normales(face,i)*cc;
183 for (j=0; j<nb_comp; j++)
184 {
185 sigma(i,j)+=u(face,i)*uT(face,j);
186 grad_uT(j,i)+=uT(face,j)*grad_phi(num_face,i);
187 }
188 }
189 }
190
191 if(antisym==1)
192 {
193 for(num_face=0; num_face<nb_faces_elem; num_face++)
194 {
195// for num_face
196 face=elem_faces(elem, num_face);
197 for (i=0; i<dimension; i++)
198 for (j=0; j<nb_comp; j++)
199 contrib_elem(num_face,j)-=0.5*d_inv*(u(face,i)*grad_uT(j,i)-sigma(i,j)*grad_phi(num_face,i));//!< Ui Uj,i Vj - Ui Vj,i Uj
200 }
201 }
202 else
203 {
204 for(num_face=0; num_face<nb_faces_elem; num_face++)
205 {
206// for num_face
207 face=elem_faces(elem, num_face);
208 for (i=0; i<dimension; i++)
209 for (j=0; j<nb_comp; j++)
210 contrib_elem(num_face,j)-=d_inv*(u(face,i)*grad_uT(j, i));//!< Ui Uj,i Vj
211 }
212 }
213
214 for(num_face=0; num_face<nb_faces_elem; num_face++)
215 {
216 face=elem_faces(elem, num_face);
217 for (i=0; i<nb_comp; i++)
218 resu(face,i)+=contrib_elem(num_face,i);
219 }
220 }
221
222 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
223 {
224 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
225 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
226 int nb_faces_bord_tot=le_bord.nb_faces_tot();
227 int num10 = 0, num20 = le_bord.nb_faces_tot();
228
229 if (sub_type(Periodique,la_cl.valeur()))
230 {
231 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
232 int face_associee;
233 IntVect fait(nb_faces_bord_tot);
234 fait = 0;
235
236 for (int ind_face=num10; ind_face<num20; ind_face++)
237 {
238 face = le_bord.num_face(ind_face);
239 if (fait(ind_face) == 0)
240 {
241 fait(ind_face) = 1;
242 face_associee=la_cl_perio.face_associee(ind_face);
243 fait(face_associee) = 1;
244 face_associee = le_bord.num_face(face_associee);
245
246 for (int comp=0; comp<nb_comp; comp++)
247 {
248 double som;
249 som=resu(face_associee,comp)+resu(face,comp);
250 resu(face,comp)=resu(face_associee,comp)=som;
251 }
252 }//!< if fait
253 }//!< for face
254
255 }//!< sub_type Perio
256
257 else if ((antisym==1) && (!sub_type(Symetrie,la_cl.valeur())) )
258 {
259// Cerr << "Ajout des termes de bords :" << finl;
260 int num1 = le_bord.num_premiere_face();
261 int nb_faces=le_bord.nb_faces();
262 int num2 = num1 + nb_faces;
263
264 for (face=num1; face<num2; face++)
265 {
266 int comp;
267 double psc = 0;
268 for (comp=0; comp<nb_comp; comp++)
269 psc += face_normales(face,comp)*u(face, comp);
270 psc *=0.5;
271 for (comp=0; comp<nb_comp; comp++)
272 resu(face, comp)-=psc*uT(face, comp);
273
274 }
275 }
276 }//!< for nbord
277
278 if(filtrer_resu)
279 ch.filtrer_resu(resu);
280
281// On desactive le calcul et l'impression des energies u'......
282// on garde le codage au cas ou..
283 if (0)
284 {
285 Cerr << "filtrage petites echelles : " << finl;
286
287 DoubleTab ubar(transporte);
288 ch.filtrer_L2(ubar);
289 DoubleTab uprime(transporte);
290 uprime-=ubar;
291
292 const int nb_faces = zone_VEF.nb_faces();
293 const DoubleVect& volumes_entrelaces=zone_VEF.volumes_entrelaces();
294 int face2, k;
295 double psc=0., pscprim=0., ec=0., ecprime=0.;
296 int deb=zone_VEF.premiere_face_int();
297 for(face2=deb; face2<nb_faces; face2++)
298 for(k=0; k< dimension; k++)
299 {
300 double v=volumes_entrelaces(face2);
301 double up, u2, r;
302 if(nb_comp>1)
303 {
304 up=uprime(face2,k);
305 uprime(face2,k)*=v;
306 u2=ubar(face2,k);
307 r=resu(face2,k);
308 }
309 else
310 {
311 up=uprime(face2);
312 uprime(face2)*=v;
313 u2=ubar(face2);
314 r=resu(face2);

316 pscprim+=r*up;
317 psc+=r*u2;
318 ec+=u2*u2*v;
319 ecprime+=up*up*v;
320 }
321
322
323
324 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
325 {
326 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
327 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
328 int num1 = le_bord.num_premiere_face();
329 int nb_faces_b=le_bord.nb_faces();
330 int num2 = num1 + nb_faces_b;

342 fait(face2-num1) = 1;
343 face_associee=la_cl_perio.face_associee(face2-num1);
344 fait(face_associee) = 1;
345 face_associee+=num1;
346 double v=volumes_entrelaces(face2);
347 double up, u2, r;
348 for(k=0; k< dimension; k++)
349 {
350 if(nb_comp>1)

352 up=(uprime(face2,k));
353 uprime(face2,k)*=v;
354 uprime(face_associee,k)=uprime(face2,k);
355 u2=ubar(face2,k);
356 r=resu(face2,k);
357 }

359 {
360 up=uprime(face2);

362 uprime(face_associee)=uprime(face2);

370 }

372 }
373 }
374 else
375 {
376 for (face2=num1; face2<num2; face2++)
377 {
378 double v=volumes_entrelaces(face2);
379 double up, u2, r;
380 for(k=0; k< dimension; k++)
381 {
382 if(nb_comp>1)
383 {
384 up=(uprime(face2,k));
385 uprime(face2,k)*=v;
386 u2=ubar(face2,k);
387 r=resu(face2,k);
388 }
389 else
390 {
391 up=uprime(face2);
392 uprime(face2)*=v;
393 u2=ubar(face2);
394 r=resu(face2);
395 }
396 pscprim+=r*up;
397 psc+=r*u2;
398 ec+=u2*u2*v;
399 ecprime+=up*up*v;
400 }
401 }
402 }
403 }
404 Cerr << "(u,u) "<< ec <<finl;

410 pourcent=100*sqrt(ecprime/ec);
411 Cerr << "||u'||/||u|| : " << pourcent << " %" << finl;
412 }
413 resu+=sauv;
414 modifier_flux(*this);
415 return resu;
416}
