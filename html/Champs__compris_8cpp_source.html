69bool Champs_compris::has_champ(const Motcle& motcle, REF(Champ_base)& ref_champ) const
70{
71 if (motcle=="??")
72 {
73 Cerr<<"Champs_compris::get_champ()"<<finl;
74 Cerr<<"No field can be requested using the identifier \'??\'"<<finl;
75 exit();
76 }
77 CONST_LIST_CURSEUR(REF(Champ_base)) curseur = liste_champs_;
78 while (curseur)
79 {
80 ref_champ = curseur.valeur().valeur();
81 if (ref_champ.le_nom() == motcle) //!< case insensitive test
82 return true;
83 else
84 {
85 const Noms& syno = ref_champ->get_synonyms();
86 int nb_syno = syno.size();
87 for (int s = 0; s < nb_syno; s++)
88 {
89 if (syno[s] == motcle) //!< case insensitive test
90 return true;
91 }
92 int nb_composantes = ref_champ->nb_comp();
93 for (int i = 0; i < nb_composantes; i++)
94 {
95 if (motcle == ref_champ->nom_compo(i)) //!< case insensitive test
96 return true;
97 }
98 }
99 ++curseur;
100 }
101 return false;
102}
