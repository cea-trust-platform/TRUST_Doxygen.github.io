110int Assembleur_P_EF::assembler_rho_variable(Matrice& la_matrice, const Champ_Don_base& rho)
111{
112// On multiplie par la masse volumique aux sommets
113 if (!sub_type(Champ_Fonc_Q1_EF, rho))
114 {
115 Cerr << "La masse volumique n'est pas aux sommets dans Assembleur_P_EF::assembler_rho_variable." << finl;
116 Process::exit();
117 }
118 const DoubleVect& volumes_som=ref_cast(Zone_EF, la_zone_EF.valeur()).volumes_sommets_thilde();
119 const DoubleVect& masse_volumique=rho.valeurs();
120 DoubleVect quantitee_som(volumes_som);
121 int size=quantitee_som.size_array();
122 for (int i=0; i<size; i++)
123 quantitee_som(i)=(volumes_som(i)*masse_volumique(i));
124
125// On assemble la matrice
126 return assembler_mat(la_matrice,quantitee_som,1,1);
