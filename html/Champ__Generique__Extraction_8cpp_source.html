94// Precondition: L extraction est actuellement limitee aux champs inconnus du probleme
95// Elle est realisee uniquement sur des frontieres planes
96//! Extraction des valeurs d un champ (trace ou champ frontiere) sur un bord du domaine
97const Champ_base& Champ_Generique_Extraction::get_champ(Champ& espace_stockage) const
98{
99
100 Champ source_espace_stockage;
101 const Champ_Generique_base& source = get_source(0);
102 const Champ_base& source_stockage = source.get_champ(source_espace_stockage);
103
104 const DoubleTab& source_valeurs = source_stockage.valeurs();
105 const Zone_dis_base& zone_dis_source = source.get_ref_zone_dis_base();
106 const Nature_du_champ& nature_source = source_stockage.nature_du_champ();
107
108// Le test suivant n est pas dans completer() car utilise equation() : la reference doit etre initialisee
109 if (methode_=="champ_frontiere")
110 {
111 int num_bord = zone_dis_source.rang_frontiere(nom_fr_);
112 const Champ_Inc_base& source_inconnue = ref_cast(Champ_Inc_base,source_stockage);
113 const Cond_lim& cl = source_inconnue.equation().zone_Cl_dis().les_conditions_limites(num_bord);
114 if (sub_type(Dirichlet_homogene,cl.valeur()))
115 {
116 Cerr<<"A boundary condition of type Dirichlet homogeneous does not use field boundary"<<finl;
117 Cerr<<"The extraction of field "<<source_stockage.le_nom()<<" with the option champ_frontiere"<<finl;
118 Cerr<<"is therefore not possible for this type of boundary condition"<<finl;
119 Cerr<<"You can possibly remove the option champ_frontiere for postprocess the trace of the field"<<finl;
120 exit();
121 }
122 }
123
124
125// Probleme pour discretisation de la zone du domaine d extraction (dimension)
126//
127
128 int nb_comp_source = source_stockage.nb_comp();
129 double temps = source.get_time();
130
131 Nom type_source, type_espace_stockage, disc;
132 int nb_comp = 0;
133 int nb_ddl = 0;
134
135 if (zone_dis_source.que_suis_je().debute_par("Zone_VDF"))
136 disc = "VDF";
137 else if (zone_dis_source.que_suis_je().debute_par("Zone_VEF"))
138 disc = "VEF";
139 else if (zone_dis_source.que_suis_je().debute_par("Zone_EF"))
140 disc = "EF";
141 else
142 {
143 Cerr<<"Champ_Generique_Extraction::get_champ()"<<finl;
144 Cerr<<"We do not recognize the type of area discretized"<<zone_dis_source.que_suis_je()<<finl;
145 exit();
146 }
147
148 type_espace_stockage = "Champ_Fonc_P0_";
149 type_espace_stockage += disc;
150
151 if (source_stockage.que_suis_je()=="Champ_Face")
152 nb_comp = dimension;
153 else
154 nb_comp = nb_comp_source;
155
156// La trace rend systematiquement un tableau de valeurs aux faces de depart
157// et un champ frontiere discretise est dimensionne par rapport au nombre de faces
158// de la frontiere : dans les deux cas nb_ddl = nb_faces
159 const Frontiere_dis_base& fr_dis = zone_dis_source.frontiere_dis(nom_fr_);
160 const Frontiere& la_frontiere = fr_dis.frontiere();
161 nb_ddl = la_frontiere.nb_faces();
162
163// La zone discretisee a associer n est actuellement pas disponible
164// On associe pas de zone_discretisee et on ne fixe pas nb_valeurs_nodales
165 espace_stockage.typer(type_espace_stockage);
166//
167 espace_stockage->fixer_nb_comp(nb_comp);
168//
169 espace_stockage->fixer_nature_du_champ(nature_source);
170
171 DoubleTab& espace_valeurs = espace_stockage->valeurs();
172 const int N = source_valeurs.line_size();
173 espace_valeurs.resize(nb_ddl,nb_comp);
174
175 if (methode_=="trace") //On prend la trace du champ sur un bord
176 {
177 source_stockage.trace(fr_dis,espace_valeurs,temps,0);
178 }
179 else if (methode_=="champ_frontiere") //On recupere le champ frontiere
180 {
181 int num_bord = zone_dis_source.rang_frontiere(nom_fr_);
182 const Champ_Inc_base& source_inconnue = ref_cast(Champ_Inc_base,source_stockage);
183 const Champ_front& champ_fr = source_inconnue.equation().zone_Cl_dis().les_conditions_limites(num_bord)->champ_front();
184
185 if (sub_type(Champ_front_uniforme,champ_fr.valeur()))
186 {
187 for (int j = 0; j < N; j++)
188 for (int i=0; i<nb_ddl; i++)
189 espace_valeurs(i,j) = champ_fr.valeur()(0,j);
190 }
191 else
192 espace_valeurs = champ_fr->valeurs();
193 }
194
195// L espace de stockage n a pas actuellement d espace virtuel
196//
197
198 return espace_stockage.valeur();
199}

245// Description :
246// La methode completer complete le domaine d extraction declare dans le jeu de donnes
247// auquel la classe fait reference:
248// - ajout d une zone (de type Pave)
249// - on determine les coordonnees des sommets contenus dans le plan d extraction
250// Actuellement on ne procede pas a la discretisation de la zone du domaine d extraction (pb dimension)
251void Champ_Generique_Extraction::completer(const Postraitement_base& post)
252{
253 Champ_Gen_de_Champs_Gen::completer(post);
254 const Objet_U& ob = interprete().objet(dom_extrac_);
255 if (!sub_type(Domaine, ob))
256 {
257 Cerr << "Error in Champ_Generique_Extraction:get_champ"<<finl;
258 Cerr<<"We do not retrieve domain"<<finl;
259 exit();
260 }
261 domaine_ = ref_cast(Domaine,ob);
262 const Champ_Generique_base& source = get_source(0);
263
264 if (methode_=="champ_frontiere")
265 {
266 if (sub_type(Champ_Generique_refChamp,source))
267 {
268 Champ source_espace_stockage;
269 const Champ_base& source_stockage = source.get_champ(source_espace_stockage);
270 if (!sub_type(Champ_Inc_base,source_stockage))
271 {
272 Cerr<<"The method "<<methode_<<" can be applied to extract only unknown fields of the problem."<<finl;
273 Cerr<<"The field to extract "<<source_stockage.le_nom()<<" do not satisfied this requirement."<<finl;
274 exit();
275 }
276 }
277 else
278 {
279 const Noms nom_champ = get_property("nom");
280 Cerr<<"The method "<<methode_<<" can be applied to extract only unknown fields of the problem."<<finl;
281 Cerr<<"The source of the post-processing field "<<nom_champ[0]<<" do not encapsulate an unknown field."<<finl;
282 exit();
283 }
284
285 }
286
287 const Domaine& dom = source.get_ref_domain();
288 if (dom==domaine_.valeur())
289 {
290 Cerr<<"Error in Champ_Generique_Extraction:get_champ"<<finl;
291 Cerr<<"The domain is the same as the source's domain."<<finl;
292 exit();
293 }
294 const Zone_dis_base& zone_dis_source = source.get_ref_zone_dis_base();
295 const Zone_VF& zvf_source = ref_cast(Zone_VF,zone_dis_source);
296// Pour eviter un crash:
297 if (zvf_source.nb_frontiere_dis()==0)
298 {
299 Cerr << "You can't extract a field onto a boundary of the domain " << dom.le_nom() << finl;
300 Cerr << "because there is NO boundaries defined on it." << finl;
301 if (sub_type(Champ_Generique_Interpolation,source))
302 {
303 Cerr << "It is not possible to extract on a boundary a field interpolated other than the calculation domain." << finl;
304 Cerr << "So, change the domain in Interpolation definition into domain " << ref_cast(Champ_Generique_Interpolation,source).get_source(0).get_ref_domain().le_nom() << finl;
305 }
306 exit();
307 }
308 const Frontiere_dis_base& fr_dis = zone_dis_source.frontiere_dis(nom_fr_);
309 const Frontiere& la_frontiere = fr_dis.frontiere();
310 const Type_Face type_face_source = la_frontiere.faces().type_face();
311 int nb_faces = la_frontiere.nb_faces();
312
313 const DoubleTab& sommets_source = dom.les_sommets();
314 int num_premiere_face = la_frontiere.num_premiere_face();
315 Entity index1,index2;
316 index1 = FACE;
317 index2 = NODE;
318 const IntTab& face_sommets = source.get_ref_connectivity(index1,index2);
319 int nb_som_faces = zvf_source.nb_som_face();
320
321 Nom type_zone,type_elem;
322 if (type_face_source==Faces::segment_2D || type_face_source==Faces::segment_2D_axi)
323 type_elem = "Segment"; //!< Pour MC2
324 else if (type_face_source==5)
325 type_elem = "Segment"; //!< Pour MC2
326 else if (type_face_source==Faces::quadrangle_3D)
327 type_elem = "Quadrangle";
328 else if (type_face_source==Faces::triangle_3D)
329 type_elem = "Triangle";
330 else if (type_face_source==Faces::point_1D)
331 type_elem = "Point";
332// Cas suivant possible en parallele
333 else if ((type_face_source==Faces::vide_0D) && (nb_faces==0))
334 {
335 if (zvf_source.que_suis_je().debute_par("Zone_VDF"))
336 type_elem = "Rectangle";
337 else if (zvf_source.que_suis_je().debute_par("Zone_VEF"))
338 type_elem = "Triangle";
339 else
340 {
341 Cerr<<"The type of zone discretized"<<zvf_source.que_suis_je()<<" is not recognized"<<finl;
342 exit();
343 }
344 }
345 else
346 {
347 Cerr<<"The type of faces of the boundary name "<<nom_fr_<<" is neither type Faces::quadrangle_3D nor Faces::triangle_3D"<<finl;
348 Cerr<<"We do not realize the extraction"<<finl;
349 exit();
350 }
351
352// Verification que le domaine d'extraction utilise est correct:
353 if (domaine_->nb_som()!=0)
354 {
355 if (domaine_->zone(0).les_elems().dimension(0) != nb_faces ||
356 domaine_->zone(0).les_elems().dimension(1) != nb_som_faces)
357 {
358 Cerr << "Error when extracting the field " << nom_post_ << ": The " << domaine_.le_nom() << " domain can't be used cause already built with different support !" << finl;
359 Cerr << "You could create an empty domain with no operations on it before extracting this field." << finl;
360 Process::exit();
361 }
362 }
363 Zone zz;
364 domaine_->add(zz);
365 Zone& ajoutee = domaine_->zone(0);
366 ajoutee.typer(type_elem);
367 ajoutee.associer_domaine(domaine_);
368
369 IntTab& mes_elems_zone = ajoutee.les_elems();
370 mes_elems_zone.reset();
371 mes_elems_zone.resize(nb_faces, nb_som_faces);
372
373// Destruction du descripteur parallele, et on reconstruit...
374 domaine_->les_sommets().reset();
375 domaine_->les_sommets() = sommets_source;
376 for (int face = 0; face < nb_faces; face++)
377 for (int s = 0; s < nb_som_faces; s++)
378 mes_elems_zone(face, s) = face_sommets(num_premiere_face + face, s);
379 NettoieNoeuds::nettoie(domaine_);
380
381// Renumerotation
382 ajoutee.associer_domaine(domaine_);
383// Discretisation du domaine d'extraction
384 discretiser_domaine();
385}
