40int Convection_Diffusion_Chaleur_WC::lire_motcle_non_standard(const Motcle& mot, Entree& is)
41{
42 if (mot=="convection")
43 {
44 Convection_Diffusion_Chaleur_Fluide_Dilatable_base::lire_motcle_non_standard(mot,is);
45
46// l'equation de la chaleur en quasi compressible contient un terme source dP_tot/dt
47 Cerr << "Source term creation of the energy equation :"<< finl;
48 Source t;
49 Source& so=les_sources.add(t);
50 Nom type_so = "Source_WC_Chaleur_";
51 Nom disc = discretisation().que_suis_je();
52 if (disc=="VEFPreP1B") disc = "VEF";
53 type_so += disc;
54 so.typer_direct(type_so);
55 so->associer_eqn(*this);
56 Cerr<<so->que_suis_je()<<finl;
57 Cerr << "Typing of the convection operator : ";
58 return 1;
59 }
60 else
61 return Convection_Diffusion_Chaleur_Fluide_Dilatable_base::lire_motcle_non_standard(mot,is);
62 return 1;
63}
