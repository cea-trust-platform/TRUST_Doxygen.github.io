256inline int Zone_VF::numero_sommet_local(int som, int elem) const
257{
258 int nse=zone().nb_som_elem();

470//! renvoie dans el0 le numero de l'elt a l'interieur<br>renvoie dans elf le numero de l'elt fictif (-1 si il n'existe pas)
471inline void Zone_VF::face_voisins_reel_fictif(int face,int& el0,int& elf) const
472{
473 assert(face<premiere_face_int());
474 el0 = face_voisins(face,0);
475 elf = face_voisins(face,1);
476 if (el0==-1)
477 {
478 el0=elf;
479 elf=face_voisins_fictifs_(face,0);
480 }
481 if (elf==-1) elf=face_voisins_fictifs_(face,1);
482
483}

647inline int Zone_VF::orientation_si_definie(int num_face) const
648{
649 for (int dir=0; dir<dimension; dir++)
650 {
651 if (est_egal(std::fabs(face_normales(num_face, dir)), face_surfaces(num_face)))
652 return dir;
653 }
654 return -1;
655}
656
657inline int Zone_VF::orientation(int num_face) const
658{
659 for (int dir=0; dir<dimension; dir++)
660 {
661 if (est_egal(std::fabs(face_normales(num_face, dir)), face_surfaces(num_face)))
662 return dir;
663 }
664 Cerr<<"Your mesh is not axis oriented."<<finl;
665 Cerr<<"Try recoding using face_normales(). Example for g(face)=gravite[orientation[face]];" << finl;
666 Cerr<<"Compute something like: g(face)=Sum(i=0;i<gravite.size();i++) of gravite[i]*face_normales(face,i)/surface(face);" << finl;
667 exit();
668 return -1;
669}
