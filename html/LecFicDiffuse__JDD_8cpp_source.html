62//! Ouverture du fichier. Cette methode doit etre appelee<br>par tous les processeurs du groupe.<br>Valeur de retour: 1 si ok, 0 sinon
63int LecFicDiffuse_JDD::ouvrir(const char* name,
64 IOS_OPEN_MODE mode)
65{
66 int ok = 0;
67
68 if(Process::je_suis_maitre())
69 {
70
71 Cout <<"Reading data file "<<finl;
72 int fin_lu=0;
73 Motcle fin("fin|end");
74 EFichier file_;
75 ok = file_.ouvrir(name, mode);
76 if (!ok )
77 {
78 Cerr << "File " << name << " does not exist (LecFicDiffuse_JDD)" << finl;
79 Process::exit();
80 }
81 SChaine prov;
82 Nom motlu;
83 file_>>motlu;
84 int nb_accolade=0;
85 int nb_accolade_sa=-1;
86 while (1)
87 {
88 if (file_.eof())
89 {
90 break;
91 }
92 if (motlu=="{")
93 nb_accolade++;
94 if (motlu=="}")
95 nb_accolade--;
96 if (fin==motlu)
97 {
98 fin_lu=1;
99 nb_accolade_sa=nb_accolade;
100 }
101 if (motlu=="#")
102 {
103// Cerr<<" on passe les commentaires"<<finl;
104 file_>>motlu;
105 while (motlu!="#")
106 {
107 if (file_.eof())
108 {
109 Cerr<<"Problem while reading some # "<<finl;
110 Cerr<<" # not closed "<<finl;
111 if (fin_lu)
112 Cerr<<" Keyword Fin is may be already readen."<<finl;
113 else
114 Process::exit();
115 break;
116 }
117 file_ >> motlu;
118
119 }
120 }
121
122 else if (motlu=="/*")
123 {
124 // Cerr<<" on passe le bloc de commentaires"<<finl;
125
126 int ouvrante=1;
127 while (ouvrante!=0)
128 {
129 file_ >>motlu;
130
131 if (file_.eof())
132 {
133 Cerr<<"Problem while reading some /* "<<finl;
134 Cerr<<" not closed "<<finl;
135 if (fin_lu)
136 Cerr<<" Keword Fin is may be already readen."<<finl;
137 else
138 Process::exit();
139 break;
140
141 }
142 if (motlu=="/*") ouvrante++;
143 if (motlu=="*/") ouvrante--;
144
145
146 }
147
148 }
149
150 else if ((motlu.find("}")!=-1 && motlu != "}") || (motlu.find("{")!=-1 && motlu != "{") )
151 {
152 Cout << "Error while reading '" << motlu << "' from datafile. Check for missing space character" << finl;
153 Process::exit();
154 }
155
156 /* GF ne pas reactiver ce morceau de code car cela
157 est complique pour la traduction des jdd en python
158 else if (motlu=="##")
159 {
160
161 Cerr<<" The comment line is crossed."<<finl;
162 std::string ligne;
163 std::getline(file_.get_ifstream(), ligne);
164
165 }
166 */
167 else
168 {
169 prov<<motlu<<" ";
170 if( apply_verif )
171 verifie(motlu);
172 }
173 file_>>motlu;
174 }
175 if (fin_lu!=0)
176 nb_accolade=nb_accolade_sa;
177 if (nb_accolade!=0)
178 {
179 if (nb_accolade<0)
180 Cerr<<"Error perhaps extra \"}\" in data file"<<finl;
181 else
182 Cerr<<"Error check for missing \"}\" in data file"<<finl;
183 Process::exit();
184 }
185
186 chaine_.init(prov.get_str());
187 Cout <<finl;
188
189 Process::Journal()<<"JDD interpreted: "<<finl<<prov.get_str()<<finl<<finl;
190 }
191 envoyer_broadcast(ok, 0);
192 return ok;
193}
