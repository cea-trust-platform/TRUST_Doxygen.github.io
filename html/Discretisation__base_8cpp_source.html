51
52// On passe une directive qui doit etre reconnue par le type
53// de discretisation. Les directives sont des mots cles :
54// la casse est ignoree et il ne doit pas y avoir d'espace.
55// Exemples de directives :
56// "CHAMP_P0",
57// "CHAMP_FACES",
58// "VITESSE",
59// "PRESSION",
60// "TEMPERATURE",
61// Parfois, le nombre de composantes depend de la discretisation
62// (exemple pour la vitesse : 1 composante en VDF, 3 en VEF)
63// Si on met nb_comp = -1, la discretisation choisit le nombre
64// approprie, sinon elle utilise la valeur fournie.

205//! Fonction outil pour fixer les membres communs a tous les types<br>de champs (utilisee dans creer_champ)
206void Discretisation_base::champ_fixer_membres_communs(
207 Champ_base& ch,
208 const Zone_dis_base& z,
209 const Nom& type,
210 const Nom& nom,
211 const Nom& unite,
212 int nb_comp,
213 int nb_ddl,
214 double temps)
215{
216 ch.associer_zone_dis_base(z);
217 ch.nommer(nom);
218 ch.fixer_nb_comp(nb_comp);
219 if ((nb_comp>1)&& (nb_comp==dimension))
220 {
221 Noms noms(nb_comp);
222 if (!axi)
223 {
224 noms[0]=nom+"X";
225 noms[1]=nom+"Y";
226 if (nb_comp>2)
227 noms[2]=nom+"Z";
228 }
229 else
230 {
231 noms[0]=nom+"R";
232 noms[1]=nom+"teta";
233 if (nb_comp>2)
234 noms[2]=nom+"Z";
235 }
236 ch.fixer_noms_compo(noms);
237 }
238 ch.fixer_unite(unite);
239 ch.fixer_nb_valeurs_nodales(nb_ddl);
240 ch.changer_temps(temps);
241}

355void Discretisation_base::nommer_completer_champ_physique(const Zone_dis_base& zone_dis, const Nom& nom_champ, const Nom& unite, Champ_base& le_champ,
356 const Probleme_base& pb) const
357{
358// on nomme le champ et l'unite
359 le_champ.nommer(nom_champ);
360 le_champ.fixer_unite(unite);
361 if (sub_type(Champ_Fonc_Tabule,le_champ))
362 {
363 Noms& noms_variables = ref_cast(Champ_Fonc_Tabule, le_champ).noms_champs_parametre();
364 Noms& noms_pbs = ref_cast(Champ_Fonc_Tabule, le_champ).noms_problemes();
365 VECT(REF(Champ_base)) les_ch_eq;
366 for (int i = 0; i < noms_variables.size(); i++)
367 {
368 REF(Champ_base) champ;
369 const Probleme_base& pb_ch = noms_pbs.size() == 0 ? pb : ref_cast(Probleme_base, Interprete::objet(noms_pbs[i]));
370 champ = pb_ch.get_champ(Motcle(noms_variables[i]));
371 les_ch_eq.add(champ);
372 }
373 modifier_champ_tabule(zone_dis,ref_cast(Champ_Fonc_Tabule,le_champ), les_ch_eq);
374 }
375}

387Nom Discretisation_base::get_name_of_type_for(const Nom& class_operateur, const Nom& type_operateur,const Equation_base& eqn ,const REF(Champ_base)& champ_sup) const
388{
389 Nom type;
390 if (class_operateur=="Source")
391 {
392 type = type_operateur;
393 Nom disc = eqn.discretisation().que_suis_je();
394
395 int isQC=eqn.probleme().is_dilatable();
396
397 if ((isQC) && ((eqn.que_suis_je()!="Transport_K_Eps")
398 && (eqn.que_suis_je()!="Transport_K_Eps_Bas_Reynolds")
399 && (eqn.que_suis_je()!="Transport_K_Eps_Realisable")
400 && (eqn.que_suis_je()!="Transport_K_Eps_V2")))
401 type+="_QC";
402
403// sauf pour le term boussinesq
404 if( disc=="VEFPreP1B" )
405 {
406 if ( (Motcle(type_operateur) == "boussinesq_temperature") || (Motcle(type_operateur) == "boussinesq_concentration") || (Motcle(type_operateur) == "boussinesq") )
407 {
408 disc="VEFPreP1B" ;
409 }
410 else
411 {
412 disc="VEF";
413 }
414 }
415 Nom champ = (eqn.inconnue()->que_suis_je());
416 if (champ == "Champ_Q1NC")
417 champ = "Champ_P1NC";
418
419 champ.suffix("Champ_");
420 type+="_";
421 type+=disc;
422 {
423 type+="_";
424 type+=champ;
425 }
426 return type;
427 }
428 else if (class_operateur=="Solveur_Masse")
429 {
430 type="Masse_";
431
432 Nom discr=eqn.discretisation().que_suis_je();
433// le Solveur est commun aux discretisations VEF et VEFP1B
434 if(discr=="VEFPreP1B" )
435 discr="VEF";
436
437 type+=discr;
438
439 Nom type_ch=eqn.inconnue()->que_suis_je();
440 if (type_ch == "Champ_Q1NC")
441 type_ch = "Champ_P1NC";
442
443 if (type_ch.debute_par("Champ_P0_VDF"))
444 type_ch = "Champ_P0_VDF";
445
446 if (type_ch.debute_par("Champ_Face"))
447 type_ch = "Champ_Face";
448
449 type_ch.suffix("Champ");
450 type+=type_ch;
451
452 }
453 else if (class_operateur=="Operateur_Grad")
454 {
455 type="Op_Grad_";
456
457 Nom type_pb = eqn.probleme().que_suis_je();
458 if (type_pb=="Probleme_SG")
459 {
460 type+=(type_pb.suffix("Probleme_"));
461 type+="_";
462 }
463
464 Nom discr=eqn.discretisation().que_suis_je();
465
466 type+=discr;
467 type+="_";
468 Nom type_inco=eqn.inconnue()->que_suis_je();
469 if (type_inco == "Champ_Q1NC")
470 type_inco = "Champ_P1NC";
471
472 type+=(type_inco.suffix("Champ_"));
473
474
475// Test pour appliquer un gradient a un Champ_P1NC a une composante en VEF
476// Typage a revoir (revison des operateurs)
477 if ((eqn.inconnue()->le_nom()!="vitesse") && (eqn.inconnue()->que_suis_je()=="Champ_P1NC"))
478 type="Op_Grad_P1NC_to_P0";
479
480// Test pour appliquer un gradient a un Champ_P0 a une composante en VDF
481// Typage a revoir (revison des operateurs)
482 if ((eqn.inconnue()->le_nom()!="vitesse") && (eqn.inconnue()->que_suis_je()=="Champ_P0_VDF"))
483 type="Op_Grad_P0_to_Face";
484
485 return type;
486 }
487 else if (class_operateur=="Operateur_Div")
488 {
489
490 type="Op_Div_";
491
492 Nom discr=eqn.discretisation().que_suis_je();
493 Nom type_inco=eqn.inconnue()->que_suis_je();
494
495 type+=discr;
496
497
498 type+="_";
