86//! Calcule la masse molaire du melange (M)<br>M depend de la mase molaire de chaque espece et de la composition du melange (Yi)
87void Loi_Etat_Multi_GP_QC::calculer_masse_molaire(DoubleTab& tab_masse_mol_mel) const
88{
89 const int size = tab_masse_mol_mel.size();
90 ArrOfDouble inv_M(size), numer_M(size);
91
92 for (int i=0; i<liste_Y.size(); i++)
93 {
94 double M_i=liste_especes(i).valeur().masse_molaire();
95 const DoubleTab& Y_i=liste_Y(i).valeur().valeurs();
96 double min_Y_i=local_min_vect(Y_i);
97 double max_Y_i=local_max_vect(Y_i);
98 if ((min_Y_i<0.-dtol_fraction_)||(max_Y_i>1.+dtol_fraction_))
99 {
100 Cerr << "Warning : (min_Y_i<-" << dtol_fraction_ << ")||(max_Y_i>1+" << dtol_fraction_ <<") for the " << i << "th mass fraction:" << finl;
101 Cerr << " min_Y_i = " << min_Y_i << finl;
102 Cerr << " max_Y_i = " << max_Y_i << finl;
103 if (!ignore_check_fraction_) Process::exit();
104 }
105 for (int elem=0; elem<size; elem++)
106 {
107 numer_M[elem] += Y_i(elem,0);
108 inv_M[elem] += Y_i(elem,0)/M_i;
109 }
110 }
111
112 assert(liste_Y(0).valeur().valeurs().size()==size);
113
114 for (int elem=0; elem<size; elem++)
115 if (inv_M[elem]>1.e-8)
116 tab_masse_mol_mel(elem,0) = numer_M[elem] / inv_M[elem];
117}

161//! Corrections explicites pour les fractions massiques pour forcer<br>la somme a 1 et sans valeurs negatives
162void Loi_Etat_Multi_GP_QC::rabot(int futur)
163{
164 DoubleTab test( liste_Y(0).valeur().valeurs()) ;
165 test=0;
166
167 for (int i=0; i<liste_Y.size(); i++)
168 {
169 DoubleTab& Y_i=liste_Y(i).valeur().futur(futur);
170 double min_Y_i=local_min_vect(Y_i);
171 double max_Y_i=local_max_vect(Y_i);
172 Cerr << "Verification of the " << i << "th mass fraction:" << finl;
173 Cerr << " min_Y_i = " << min_Y_i << finl;
174 Cerr << " max_Y_i = " << max_Y_i << finl;
175 if ((min_Y_i<0.-dtol_fraction_)||(max_Y_i>1.+dtol_fraction_))
176 {
177 Cerr << " Warning : (min_Y_i<-" << dtol_fraction_ << ")||(max_Y_i>1+" << dtol_fraction_ <<")" << finl;
178 if (!ignore_check_fraction_) Process::exit();
179 }
180 for (int elem=0; elem<Y_i.size(); elem++)
181 {
182 if (Y_i(elem,0)<0.)
183 {
184 Y_i(elem,0)=0.;
185 Cerr << " Y_i forced to " << Y_i(elem,0) << " on node " << elem << finl;
186 }
187 if (Y_i(elem,0)>1.)
188 {
189 Y_i(elem,0)=1.;
190 Cerr << " Y_i forced to " << Y_i(elem,0) << " on node " << elem << finl;
191 }
192 }
193 test+=Y_i;
194 }
195 double min_s=mp_min_vect(test);
196 double max_s=mp_max_vect(test);
197 Cerr << "Verification of the sum of the mass fractions:" << finl;
198 Cerr << " Sum(min_Y_i) = " << min_s << finl;
199 Cerr << " Sum(max_Y_i) = " << max_s << finl;
200 for (int i=0; i<liste_Y.size(); i++)
201 {
202 DoubleTab& Y_i=liste_Y(i).valeur().futur(futur);
203 for (int elem=0; elem<Y_i.size(); elem++) Y_i(elem,0)/=(test(elem)+DMINFLOAT);
204 }
205 if ((!est_egal(min_s,1.,dtol_fraction_))||(!est_egal(max_s,1.,dtol_fraction_)))
206 {
207 Cerr << " Warning: the sum of the mass fractions is not equal to 1." <<finl;
208 if (!ignore_check_fraction_) Process::exit();
209 }
210}

275//! Calcule la viscosite dynamique de reference (depend des Yi)
276void Loi_Etat_Multi_GP_QC::calculer_mu_wilke()
277{
278// With Wilke formulation: https://aip.scitation.org/doi/pdf/10.1063/1.1747673
279 const int size = liste_Y(0).valeur().valeurs().size(), list_size = liste_Y.size();
280 DoubleTab phi(size);
281 DoubleTab mu(size);
282 mu = 0.;
283 phi = 0.;
284
285 for (int i=0; i<list_size; i++)
286 {
287 phi = 0.;
288 double& M_i=liste_especes(i).valeur().masse_molaire();
289 double mu_i=liste_especes(i).valeur().viscosite_dynamique().valeurs()(0,0);
290 for (int j=0; j<list_size; j++)
291 if (j!=i)
292 {
293 double& M_j=liste_especes(j).valeur().masse_molaire();
294 double mu_j=liste_especes(j).valeur().viscosite_dynamique().valeurs()(0,0);
295 double a=1.+sqrt(mu_i/mu_j)*pow(M_j/M_i,0.25);
296 double b=sqrt(8.*(1.+(M_i/M_j)));
297 /*
298 * E Saikali : XXX XXX XXX
299 * I think that a term is missing in this formulation
300 * (phi_ij must be multiplied by Mi/Mj since we use mass fractions and not mol fractions !)
301 */
302 double phi_ij=a*a/b;
303
304 const DoubleVect& y_j = liste_Y(j).valeur().valeurs();
305// node is elem (VDF) or face (VEF)
306 for (int node=0; node<y_j.size(); node++) phi(node) += y_j(node)*phi_ij;
307 }
308// We add the mass fraction when i = j
309 const DoubleVect& y_i = liste_Y(i).valeur().valeurs();
310 for (int node=0; node<y_i.size(); node++) mu(node) += mu_i * y_i(node)/ ( y_i(node) + phi(node) );
311 }
312 calculer_tab_mu(mu, size);
313}

328//! Calcule la viscosite dynamique sur Schmidt
329void Loi_Etat_Multi_GP_QC::calculer_mu_sur_Sc()
330{
331 const Champ_Don& mu = le_fluide->viscosite_dynamique();
332 const DoubleTab& tab_mu = mu.valeurs();
333 Champ_Don& mu_sur_Sc = le_fluide->mu_sur_Schmidt();
334 DoubleTab& tab_mu_sur_Sc = mu_sur_Sc.valeurs();
335
336 if (!sub_type(Champ_Uniforme,mu_sur_Sc.valeur()))
337 {
338 const int n = tab_mu_sur_Sc.size();
339 if (sub_type(Champ_Uniforme,mu.valeur()))
340 for (int i=0 ; i<n ; i++) tab_mu_sur_Sc(i,0) = tab_mu(0,0) / Sc_;
341 else
342 for (int i=0 ; i<n ; i++) tab_mu_sur_Sc(i,0) = tab_mu(i,0)/Sc_;
343 }
344 else //mu_sur_Sc uniforme
345 tab_mu_sur_Sc(0,0) = tab_mu(0,0) / Sc_;
346
347 double temps_champ = mu->temps();
348 mu_sur_Sc.valeur().changer_temps(temps_champ);
349 tab_mu_sur_Sc.echange_espace_virtuel();
350}
