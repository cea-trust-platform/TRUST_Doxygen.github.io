63 Nom fichier = dom.le_nom();

68 if (Process::je_suis_maitre())
69 Cerr << "In Distanceparoi::interpreter : Generate faces" << finl;
70 Faces* les_faces_ptr=new(Faces);
71 Faces& les_faces= *les_faces_ptr;
72
73// la_zone.creer_faces(les_faces);
74
75 {
76// bloc a factoriser avec Zone_VF.cpp :
77 Type_Face type_face = la_zone.type_elem().type_face(0);
78 les_faces.typer(type_face);
79 les_faces.associer_zone(la_zone);
80
81 Static_Int_Lists connectivite_som_elem;
82 const int nb_sommets_tot = la_zone.domaine().nb_som_tot();
83 const IntTab& elements = la_zone.les_elems();
84
85 construire_connectivite_som_elem(nb_sommets_tot,
86 elements,
87 connectivite_som_elem,
88 1 /* include virtual elements */);
89
90 Faces_builder faces_builder;
91 IntTab elem_faces; //!< Tableau dont on aura pas besoin
92 faces_builder.creer_faces_reeles(la_zone,
93 connectivite_som_elem,
94 les_faces,
95 elem_faces);
96 }
97 if (Process::je_suis_maitre())
98 Cerr << "In Distanceparoi::interpreter : Generate faces finished" << finl;
99
100 DoubleTab xp; //!< centres de gravite des elements
101 DoubleTab xv; //!< centres de gravite des faces
102 const DoubleTab& xs=dom.coord_sommets(); //!< coordonnees des sommets
103
104 les_faces.calculer_centres_gravite(xv);
105 la_zone.calculer_centres_gravite(xp);
106
107 int nb_elem_tot = xp.dimension(0);
108 int dim = xv.dimension(1);
109
110 DoubleVect wall_length(nb_elem_tot);
111 wall_length=10e+15;
112 DoubleVect dist_min(wall_length);
113
114 ifstream fic1(fichier);
115 if (fic1)
116 {
117 Cerr << " file " << fichier << " already exits : to generate a new one, this file has to be deleted" << finl;
118 exit();
119 }
120 else
121 {
122 if (Process::je_suis_maitre())
123 Cerr << " Generation of the external file named : " << fichier << finl;
124
125 for (int b=0; b<nb_paroi; b++)
126 {
127 int rang=la_zone.rang_frontiere(nom_paroi[b]);
128 const Frontiere& la_frontiere=la_zone.frontiere(rang);
129
130// pas de methode const pour Frontiere::les_sommets_des_faces();
131 const IntTab& sommets_face = (ref_cast(Frontiere,la_frontiere)).les_sommets_des_faces();
132
133 int ndeb = la_frontiere.num_premiere_face();
134 int nb_faces = la_frontiere.nb_faces();
135 int nfin = ndeb + nb_faces;
136
137 int pourcent=0;
138
139 for (int face=ndeb; face<nfin; face++)
140 {
141
142 if((((face-ndeb)*100)/nb_faces)>pourcent)
143 {
144 pourcent=((face-ndeb)*100)/nb_faces;
145 Cerr <<"\r"<< pourcent << " % performs of " << nom_paroi[b] ;
146 }
147
148 if (dim==2)
149 {
150 double compx_0=(xv(face,0));
151 double compy_0=(xv(face,1));
152
153 int som1=sommets_face(face-ndeb,0);
154 int som2=sommets_face(face-ndeb,1);
155
156 double nx= xs(som1,1)-xs(som2,1);
157 double ny=-(xs(som1,0)-xs(som2,0));
158
159 double norme=sqrt(nx*nx+ny*ny);
160 if (norme==0)
161 {
162 Cerr << "Error in Distanceparoi::interpreter_" << finl;
163 Cerr << "Two nodes " << som1 << " and " << som2 << " seems to have same coordinates:" << finl;
164 Cerr << "xs(" << som1 << ")=" << xs(som1,0) <<" ys(" << som1 << ")= " << xs(som1,1) << finl;
165 Cerr << "xs(" << som2 << ")=" << xs(som2,0) <<" ys(" << som2 << ")= " << xs(som2,1) << finl;
166 Cerr << "Check your mesh and suppress same nodes or contact TRUST support." << finl;
167 exit();
168 }
169
170 if ( (la_zone.chercher_elements(compx_0+nx, compy_0+ny)) < 0 ) norme*=-1.;
171
172 nx/=norme;
173 ny/=norme;
174
175
176 for (int num_elem=0; num_elem<nb_elem_tot; num_elem++)
177 {
178 double min_loc=dist_min[num_elem];
179
180 double compx=(compx_0-xp(num_elem,0));
181 if (compx<min_loc)
182 {
183 double compy=(compy_0-xp(num_elem,1));
184 if (compy<min_loc)
185 {
186
187 double w_dist=sqrt(compx*compx+compy*compy);
188
189 if (w_dist<min_loc)
190 {
191 double compx_vect=compx*nx;
192 double compy_vect=compy*ny;
193// double compz_vect=compz*nz;
194
195 double w_distproj=sqrt(compx_vect*compx_vect+compy_vect*compy_vect);
196 wall_length[num_elem]=w_distproj;
197 dist_min[num_elem]=w_dist;
198 }
199
200 }
201 }
202 }
203 }
204
205 else if (dim==3)
206 {
207 double compx_0=(xv(face,0));
208 double compy_0=(xv(face,1));
209 double compz_0=(xv(face,2));
210
211 int som1=sommets_face(face-ndeb,0);
212 int som2=sommets_face(face-ndeb,1);
213 int som3=sommets_face(face-ndeb,2);
214
215 double x1= xs(som2,0)-xs(som1,0);
216 double y1= xs(som2,1)-xs(som1,1);
217 double z1= xs(som2,2)-xs(som1,2);
218
219 double x2= xs(som3,0)-xs(som1,0);
220 double y2= xs(som3,1)-xs(som1,1);
221 double z2= xs(som3,2)-xs(som1,2);
222
223 double nx= y1*z2 - z1*y2 ;
224 double ny= z1*x2 - x1*z2 ;
225 double nz= x1*y2 - y1*x2 ;
226
227 double norme=sqrt(nx*nx+ny*ny+nz*nz);
228
229 if ( (la_zone.chercher_elements(compx_0+nx, compy_0+ny, compz_0+nz)) < 0 ) norme*=-1.;
230
231 nx/=norme;
232 ny/=norme;
233 nz/=norme;
234
235
236 for (int num_elem=0; num_elem<nb_elem_tot; num_elem++)
237 {
238 double min_loc=dist_min[num_elem];
239
240 double compx=(compx_0-xp(num_elem,0));
241 if (compx<min_loc)
242 {
243 double compy=(compy_0-xp(num_elem,1));
244 if (compy<min_loc)
245 {
246 double compz=(compz_0-xp(num_elem,2));
247 if (compz<min_loc)
248 {
249
250 double w_dist=sqrt(compx*compx+compy*compy+compz*compz);
251
252 if (w_dist<min_loc)
253 {
254 double compx_vect=compx*nx;
255 double compy_vect=compy*ny;
256 double compz_vect=compz*nz;
257
258 double w_distproj=sqrt(compx_vect*compx_vect+compy_vect*compy_vect+compz_vect*compz_vect);
259 wall_length[num_elem]=w_distproj;
260 dist_min[num_elem]=w_dist;
261 }
262 }
263 }
264 }
265 }
266 }
267 else
268 {
269 Cerr << "ERROR : Dimension 2 or 3 are required to calculate wall distance "<< finl;
270 exit();
271 }
272 }//!< fin face
273 Cerr<<finl;
274 }//!< fin nb_paroi
275 }//(!fic1)
276
277 Cerr<<" Max distance"<< local_max_vect(dist_min)<<" "<< local_max_vect(wall_length)<<finl;
278
279
280
281 SFichierBin fic_sauv(fichier);
282
283 fic_sauv<<nom_paroi<<finl;
284 int ntot=(nb_elem_tot*(1+Objet_U::dimension));
285// entete pour pouvoir passer les blocs
286// fic_sauv<<1 <<finl<<ntot << finl<< (int)1 <<ntot<<finl<<ntot<<finl;
287 fic_sauv << (int)1 <<ntot << (int)1 << ntot << finl << ntot << finl;
288// fic_sauv << (int)1 << ntot << (int)1 << ntot << ntot;
289 for (int p=0; p<nb_elem_tot; p++)
290 {
291 fic_sauv<<xp(p,0)<<" "<<xp(p,1);
292 if (Objet_U::dimension==3) fic_sauv<<" "<<xp(p,2);
293 fic_sauv<<" "<<wall_length(p)<<finl;
294 }
295
296 fic_sauv << ntot << (int)0 << (int)0 << (int)0 << (int)1 << (int)0 << finl << ntot << finl << (int)1 << (int)0 << (int)0 <<finl;
297
298 if (format=="formatte")
299 {
300 fichier = dom.le_nom();
301 fichier += "_Wall_length_formatted.xyz";
302 SFichier fic_sauv_format(fichier);
303 fic_sauv_format<<nom_paroi<<finl;
304 int ntot2=(nb_elem_tot*(1+Objet_U::dimension));
305// entete pour pouvoir passer les blocs
306 fic_sauv_format << (int)1 <<ntot2 << (int)1 << ntot2 << finl << ntot2 << finl;
307 for (int p=0; p<nb_elem_tot; p++)
308 {
309 fic_sauv_format<<xp(p,0)<<" "<<xp(p,1);
310 if (Objet_U::dimension==3) fic_sauv_format<<" "<<xp(p,2);
311 fic_sauv_format<<" "<<wall_length(p)<<finl;
312 }
313 fic_sauv_format << ntot2 << (int)0 << (int)0 << (int)0 << (int)1 << (int)0 << finl << ntot2 << finl << (int)1 << (int)0 << (int)0 <<finl;
314 }
315
316 delete les_faces_ptr;
317 return is;
318}
