26

242int Champ_Q4_implementation::imprime_Q4(Sortie& os, int ncomp) const
243{
244 const Champ_base& cha=le_champ();
245 int nb_compo_=cha.nb_comp();
246 const Zone& la_zone=zone();

250// On recalcule les centres de gravite :
251 const DoubleTab pos_som = la_zone.domaine().coord_sommets();
252 os << nb_som_tot << finl;
253 for (som=0; som<nb_som_tot; som++)
254 {
255 if (Objet_U::dimension==3)
256 os << pos_som(som,0) << " " << pos_som(som,1) << " " << pos_som(som,2) << " " ;
257 if (Objet_U::dimension==2)
258 os << pos_som(som,0) << " " << pos_som(som,1) << " " ;
259 if (nb_compo_ == 1)
260 os << val(som) << finl;
261 else
262 os << val(som,ncomp) << finl;
263 }
264 os << finl;
