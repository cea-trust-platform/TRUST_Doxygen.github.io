63int Param::read(Entree& is,int with_acco)
64{
65 assert(with_acco==1);
66 Motcle motlu;
67 Motcle accolade_ouverte("{");
68 Motcle accolade_fermee("}");
69 is >> motlu;
70 if (motlu != accolade_ouverte)
71 {
72 Cerr << "Error while reading the parameters of "<<proprietaire_<< finl;
73 Cerr << "We expected : " << accolade_ouverte << " and not "<<motlu<<finl;
74 motlu.exit();
75 }
76 is >> motlu;
77 while (motlu != accolade_fermee)
78 {
79 bool found=false;
80 Motcle mot(motlu);
81 LIST_CURSEUR(Objet_a_lire) curseur(list_parametre_a_lire_);
82 while ((curseur)&&(!(found)))
83 {
84 Objet_a_lire& obj=curseur.valeur();
85
86 if (obj.comprend_name(mot))
87 {
88 obj.read(mot,is);
89 list_parametre_lu_.add(mot);
90 found=true;
91 break;
92 }
93 ++curseur;
94 }
95 if (!found)
96 {
97 Cerr<<"!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!"<<finl;
98 Cerr<<motlu << " is not understood by "<<proprietaire_<<finl;
99 Cerr<<"The understood keywords are : ";
100 for (int i=0; i<list_parametre_a_lire_.size(); i++)
101 Cerr<<list_parametre_a_lire_(i).get_names_message()<<" ";
102 Cerr<<finl;
103// Evolution des options du mot cle Transformation
104// On previent les utilisateurs
105 if (proprietaire_=="Transformation")
106 {
107 if (motlu=="fonction")
108 {
109 Cerr << "Syntax has changed after the 1.6.3 version and fonction keyword is obsoleted." << finl;
110 Cerr << "Have a look at the TRUST documentation about Transformation keyword." << finl;
111 Cerr << "You should replace 'fonction' by 'methode formule expression nb_comp'" << finl;
112 Cerr << "where nb_comp is the number of components of your formulae following the old keyword 'fonction'" << finl;
113 Process::exit();
114 }
115 if (motlu=="vecteur")
116 {
117 Cerr << "Syntax has changed after the 1.6.3 version." << finl;
118 Cerr << "Have a look at the TRUST documentation about Transformation keyword" << finl;
119 Cerr << "You should replace 'vecteur' by 'methode vecteur expression'" << finl;
120 Process::exit();
121 }
122 if (motlu=="produit_scalaire")
123 {
124 Cerr << "Syntax has changed after the 1.6.3 version." << finl;
125 Cerr << "Have a look at the TRUST documentation about Transformation keyword" << finl;
126 Cerr << "You should replace 'produit_scalaire' by 'methode produit_scalaire'" << finl;
127 Process::exit();
128 }
129 }
130 Process::exit();
131 }
132 is >>motlu;
133 }
134 return 1;

184int Param::check()
185{
186 {
187// ok on a fini de lire est ce que tous les attributs non optionnels ont ete lus ?
188 int err=0;
189 LIST_CURSEUR(Objet_a_lire) curseur(list_parametre_a_lire_);
190 for (; curseur; ++curseur)
191 {
192 Objet_a_lire& obj=curseur.valeur();
193 if (!obj.is_optional())
194 {
195 Motcle name_var(obj.get_name());
196 if (!list_parametre_lu_.contient(name_var))
197 {
198 Cerr<<obj.get_name() <<" of " << proprietaire_ <<" has not been read while it is required."<<finl;
199 err=1;
200 }
201 }
202 }
203 if (err)
204 return 0;
205// Process::exit();
206
207 }
208// tests des autres conditions
209 int val=1;
210 LIST(Nom) list_mot;
211 int size=list_parametre_a_lire_.size();
212 for (int i=0; i<size; i++)
213 {
214 Nom mot_lu("is_read_");
215 mot_lu+= list_parametre_a_lire_(i).get_name();
216 list_mot.add(mot_lu);
217 }
218 for (int i=0; i<size; i++)
219 {
220 const Objet_a_lire& obj=list_parametre_a_lire_(i);
221 if (obj.is_type_simple())
222 {
223 Nom mot_lu("value_of_");
224 mot_lu+=list_parametre_a_lire_(i).get_name();
225 list_mot.add(mot_lu);
226 }
227 }
228 int size_agrandi=list_mot.size();
229 int nb_condition=list_conditions_.size();
230 for (int cond=0; cond<nb_condition; cond++)
231 {
232 Nom condition=list_conditions_(cond);
233 Parser_U parser;
234 parser.setNbVar(size_agrandi);
235 for (int j=0; j<size_agrandi; j++)
236 parser.addVar(list_mot(j));
237 parser.setString(condition);
238 parser.parseString();
239 for (int j=0; j<size; j++)
240 parser.setVar(j,0);
241 for (int lu=0; lu<list_parametre_lu_.size(); lu++)
242 {
243 Nom mot_lu("is_read_");
244 mot_lu+= list_parametre_lu_(lu);
245 parser.setVar(mot_lu,1);
246 }
247 for (int inc=size; inc<size_agrandi; inc++)
248 {
249 Nom mot_lu(list_mot(inc));
250 double xx=get_value(mot_lu);
251 parser.setVar(mot_lu,xx);
252 }
253 if (parser.eval()==0)
254 {
255 val=0;
256 Cerr<<" Condition "<< list_nom_conditions_(cond)<<" not verified: "<<condition<<" it's equal to "<<parser.eval()<<finl;
257 Cerr<<list_message_erreur_conditions_(cond)<<finl;
258 }
259
260
261 }
262 return val;
263}

265// retourne une liste de mot venant de nom.split('|')
266LIST(Nom) split_mot(const Nom& nom)
267{
268 LIST(Nom) res;
269 Nom n0(nom);
270// on cherche le dernier |
271 const char * marq= strchr(n0,'|');
272 while (marq)
273 {
274 Nom n1(n0);
275 n1.prefix(marq);
276 res.add(n1);
277
278 marq++;
279 Nom n1b(n0);
280 n1b.prefix(marq);
281 Nom n2(n0);
282 n2.suffix(n1b);
283// Cerr<<" IIIII "<< n0 <<" "<<n1<<" "<<n2<<finl;
284 n0=n2;
285 marq= strchr(n0,'|');
286 }
287// Cerr<<" "<<n0<<finl;
288 res.add(n0);
289// Cerr<<"YYY "<<res<<finl;
290 return res;
291}

473{
474 int size=list_parametre_a_lire_.size();

476 for (int i=0; i<size; i++)

484 {
485 return obj.get_value();

487 }

489 Cerr<<mot_lu<<" not found in Param::get_value"<<finl;
490 Cerr<<"the understood keywords are : ";
