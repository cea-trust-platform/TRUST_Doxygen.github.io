125inline void Schema_Comm_Vecteurs::add(int pe, int size, ArrOfInt& procs, ArrOfInt& buf_sizes,
126 int align_size)
127{
128 assert(status_ == BEGIN_INIT);
129 assert(size >= 0);
130 int& x = buf_sizes[pe];
131 if (x == 0 && size > 0)
132 {
133 const int n = procs.size_array();
134 if (n > 0 && procs[n-1] > pe)
135 sorted_ = 0;
136 procs.append_array(pe);
137 }
138 x = ((x+align_size-1)&(~(align_size-1))) + size; //!< Padding before block
139}

169//! renvoie un tableau contenant les "size" valeurs suivantes<br>recues du processeur pe lors de la communication en cours.<br>Attention:<br>Le tableau renvoye est une reference a un tableau interne qui n'est valide que<br>jusqu'au prochain appel a une methode get_next_xxx.
170template<>
171inline ArrOfInt& Schema_Comm_Vecteurs::get_next_area_template<int>(int pe, int size)
172{
173 ALIGN_SIZE(sdata_.buf_pointers_[pe], sizeof(int));
174 assert(check_next_area(pe, BLOCSIZE_INT(size)));
175 int *bufptr = (int *) (sdata_.buf_pointers_[pe]);
176// attention a l'arithmetique de pointeurs, ajout d'une taille en octets
177 sdata_.buf_pointers_[pe] += BLOCSIZE_INT(size);
178 tmp_area_int_.ref_data(bufptr, size);
179 return tmp_area_int_;
180}
