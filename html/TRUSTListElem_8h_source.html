25//! : Classe qui sert a representer une liste de reels int/double precision. On ne peut pas utiliser la classe container<br>List avec des objets du type int/double car int/double est un type predefini du C++ qui ne possede pas les fonctions exigees par List< >.
26template<typename _TYPE_>
27class TRUSTListElem
28{
29public :
30 friend class TRUSTList<_TYPE_>;
31
32 TRUSTListElem() { suivant_ = this; } //!< Construit une liste vide.
33 TRUSTListElem(const _TYPE_ x) : data(x), suivant_(0) { } //!< Construit le singleton (x).
34 TRUSTListElem(const TRUSTListElem&);

37 inline int est_vide() const { return suivant_ == this; } //!< renvoie 1 si la liste est vide. 0 sinon

60template<typename _TYPE_>
61TRUSTListElem<_TYPE_>::TRUSTListElem(const TRUSTListElem<_TYPE_>& list)
62{
63 if (list.est_vide()) suivant_ = this;
64 else

66 data = list.data;
67 if (list.suivant_)
68 {

73 }

76template<typename _TYPE_>
77TRUSTListElem<_TYPE_>::~TRUSTListElem()
78{
79 if (est_vide()) suivant_ = 0;
80
81 if (suivant_)
82 {
83// On ne garde pas la version delete suivant_ car sinon on est limite par le nombre d'appel recursif possible
84// delete suivant_;
85 TRUSTListElem<_TYPE_> *poignee, *pr;
86 pr = suivant_;
87 suivant_ = 0;
88 while (pr)
89 {
90 poignee = pr->suivant_;
91 pr->suivant_ = 0;
92 delete pr;
93 pr = poignee;
94 }
95 }
96}

109//! Operateur de comparaison de deux listes
110/** \return 1 si les listes sont egales, 0 sinon */
111template<typename _TYPE_>
112int operator ==(const TRUSTListElem<_TYPE_>& list1 , const TRUSTListElem<_TYPE_>& list2)
113{
114 int retour = 1;
115 if (list1.data != list2.data) retour = 0;
116 if ((!list1.est_dernier()) && (list2.est_dernier())) retour = 0;
117 if ((list1.est_dernier()) && (!list2.est_dernier())) retour = 0;
118 if ((!list1.est_dernier()) && (!list2.est_dernier())) retour = (*list1.suivant_ == *list2.suivant_);
119 return retour;
120}
