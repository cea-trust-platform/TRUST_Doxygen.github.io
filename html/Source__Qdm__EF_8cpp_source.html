66DoubleTab& Source_Qdm_EF::ajouter(DoubleTab& resu) const
67{
68 const Zone_EF& zone_EF = la_zone_EF.valeur();
69 int ncomp=dimension;
70 const IntTab& elems= zone_EF.zone().les_elems() ;
71 int nb_som_elem=zone_EF.zone().nb_som_elem();
72 int nb_elems=zone_EF.zone().nb_elem_tot();
73 const DoubleTab& IPhi_thilde=zone_EF.IPhi_thilde();
74 int is_source_unif=0;
75
76 if (sub_type(Champ_Uniforme,la_source.valeur()))
77 is_source_unif=1;
78 const DoubleTab& tab_source=la_source.valeur().valeurs();
79 for (int num_elem=0; num_elem<nb_elems; num_elem++)
80 for (int comp=0; comp<ncomp; comp++)
81 {
82 double sourcel ;
83 int cc=0;
84 if (!is_source_unif) cc=num_elem;
85 sourcel=tab_source(cc,comp);
86 for (int i=0; i<nb_som_elem; i++)
87 {
88// assert faux si on a des porosites
89// assert(est_egal(zone_EF.volumes(num_elem)/nb_som_elem,IPhi_thilde(num_elem,i)));
90 resu(elems(num_elem,i),comp)+=sourcel*IPhi_thilde(num_elem,i);
91 }
92 }
93
94 return resu;
95}
