198vect_impl& vect_impl::add(const Objet_U& data_to_add)
199{
200 DerObjU* tempo;

202 tempo = new DerObjU[new_size];
203 for(int i=0; i<sz; i++)
204 (tempo[i]).deplace(data[i]);

213/* Append a VECT to a VECT */
214vect_impl& vect_impl::add(const vect_impl& data_to_add)
215{
216 int sz_data_to_add = data_to_add.size();
217 DerObjU* tempo;
218 int new_size=sz+sz_data_to_add;
219 tempo = new DerObjU[new_size];
220 for(int i=0; i<sz; i++)
221 (tempo[i]).deplace( data[i]);
222 if (data) delete [] data;
223 data=tempo;
224 sz=new_size;
225 for(int j=0; j<sz_data_to_add; j++)
226 {
227 data[sz-sz_data_to_add+j]= data_to_add[j];
228 }
229 assert(sz>=0);
