103inline void Op_Ecart_type::associer(const Zone_dis_base& une_zdis,const Champ_Generique_base& le_champ,double t1,double t2)
104{
105 Champ espace_stockage_source;
106 const Champ_base& source = le_champ.get_champ(espace_stockage_source);
107 Nom type_le_champ = source.que_suis_je();
108
109 int renomme;
110 renomme=0;
111 if (type_le_champ.debute_par("Champ"))
112 renomme=1;
113
114 type_le_champ.suffix("Champ_");
115 type_le_champ.suffix("Fonc_");
116 Nom type("Champ_Fonc_");
117
118 if (renomme==1)
119 type+=type_le_champ;
120 else
121 type = type_le_champ;
122
123 integrale_carre_champ.typer(type);
124 integrale_carre_champ.associer_zone_dis_base(une_zdis);
125 integrale_carre_champ.associer(le_champ,2,t1,t2);
126}
