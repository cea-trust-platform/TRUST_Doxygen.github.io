23template <typename DERIVED_T> template <typename Type_Double>
24inline void Eval_Conv_VDF_Elem<DERIVED_T>::flux_face(const DoubleTab& inco, const int face, const Dirichlet_entree_fluide& la_cl, const int num1, Type_Double& flux) const
25{
26 const int i = elem_(face,0), j = elem_(face,1), ncomp = flux.size_array();
27 const double psc = dt_vitesse(face)*surface_porosite(face);
28 if (i != -1)
29 for (int k = 0; k < ncomp; k++) flux[k] = (psc > 0) ? -psc*inco(i,k) : -psc*la_cl.val_imp(face-num1,k);
30 else for (int k = 0; k < ncomp; k++) flux[k] = (psc > 0) ? -psc*la_cl.val_imp(face-num1,k) : -psc*inco(j,k); /* j != -1 */
31}

43template <typename DERIVED_T> template <typename Type_Double>
44inline void Eval_Conv_VDF_Elem<DERIVED_T>::flux_face(const DoubleTab& inco, const int face, const Periodique& la_cl, const int num1, Type_Double& flux) const
45{
46 const int i = elem_(face,0), j = elem_(face,1), ncomp = flux.size_array();
47 const double psc = dt_vitesse(face)*surface_porosite(face);
48
49 if (!DERIVED_T::IS_AMONT)
50 {
51 const int i_0 = amont_amont_(face,0),j_1 = amont_amont_(face,1);
52
53 if (DERIVED_T::IS_CENTRE || DERIVED_T::IS_CENTRE4) qcentre_<Type_Double>(psc,i,j,i_0,j_1,face,inco,flux); //!< on applique le schema centre 2 ou 4
54 else quick_fram_(psc,i,j,i_0,j_1,face,inco,flux); //!< on applique le schema Quick
55
56 for (int k = 0; k < ncomp; k++) flux[k] *= -1;
57 }
58 else for (int k = 0; k < ncomp; k++) flux[k] = (psc > 0) ? -psc*inco(i,k) : -psc*inco(j,k); /* AMONT */
59}

61template <typename DERIVED_T> template <typename Type_Double>
62inline void Eval_Conv_VDF_Elem<DERIVED_T>::flux_faces_interne(const DoubleTab& inco, const int face, Type_Double& flux) const
63{
64 const int i = elem_(face,0), j = elem_(face,1), ncomp = flux.size_array();
65 const double psc = dt_vitesse(face)*surface_porosite(face);
66 if (!DERIVED_T::IS_AMONT)
67 {
68 const int i_0 = amont_amont_(face,0), j_1 = amont_amont_(face,1);
69 if (DERIVED_T::IS_CENTRE)
70 {
71 qcentre_<Type_Double>(psc,i,j,i_0,j_1,face,inco,flux);
72 for (int k=0; k<ncomp; k++) flux[k] *= -1;
73 }
74 else if (DERIVED_T::IS_CENTRE4)
75 {
76 if ( (i_0 == -1) || (j_1 == -1) )
77 for (int k=0; k<ncomp; k++) flux[k] = -psc*(inco(i,k)+inco(j,k))/2.; //!< on applique le schema centre2
78 else //!< on applique le schema centre4
79 {
80 qcentre_<Type_Double>(psc,i,j,i_0,j_1,face,inco,flux);
81 for (int k=0; k<ncomp; k++) flux[k] *= -1;
82 }
83 }
84 else
85 {
86 /* *****************************************************************************************************************************************************
87 * Alexandre C. 19/02/03 : We do not use the first order upwind scheme as it was done before because in LES computations all turbulence is damped
88 * and we can not recover a proper behavior of physical turbulent characteristics, especially when using wall-functions.
89 * Therefore we use the 2nd order centered scheme.
90 * *****************************************************************************************************************************************************
91 * Pierre L. 14/10/04: Correction car le centre explose sur le cas VALIDA On revient au quick en essayant d'ameliorer: on prend le quick si psc est
92 * encore favorable pour avoir les 3 points necessaires au calcul du quick. Cela est deja ce qui est fait pour le quick-sharp de l'evaluateur aux faces.
93 * *****************************************************************************************************************************************************/
94 if ( (i_0 == -1 && psc >= 0 ) || (j_1 == -1 && psc <= 0 ) )
95 for (int k=0; k<ncomp; k++) flux[k] = (psc > 0) ? -psc*inco(i,k) : -psc*inco(j,k);
96 else //!< on applique le schema Quick
97 {
98 quick_fram_(psc,i,j,i_0,j_1,face,inco,flux);
99 for (int k=0; k<ncomp; k++) flux[k] *= -1;
100 }
101 }
102 }
103 else for(int k=0; k<ncomp; k++) flux[k] = (psc > 0) ? -psc*inco(i,k) : -psc*inco(j,k); /* AMONT */
104}

110template <typename DERIVED_T> template <typename Type_Double>

112{
113 assert (aii.size_array() == ajj.size_array());
114 const int i = elem_(face,0), ncomp = aii.size_array();
115 const double psc = dt_vitesse(face)*surface_porosite(face);
116 if (i != -1)
117 {
118 if (DERIVED_T::IS_CENTRE || DERIVED_T::IS_CENTRE4)
119 {
120 if (psc > 0)
121 for (int k = 0; k < ncomp; k++) aii[k] = -psc; //!< TODO : FIXME : Yannick help :/
122 }
123 else

125 {
126 aii[k] = (psc > 0) ? psc : 0.;
127 ajj[k] = 0.;
128 }
129 }
130 else //!< j != -1
131 {
132 if (DERIVED_T::IS_CENTRE || DERIVED_T::IS_CENTRE4)
133 {
134 if (psc < 0)
135 for (int k = 0; k < ncomp; k++) ajj[k] = -psc;

138 for (int k = 0; k < ncomp; k++)
139 {
140 ajj[k] = (psc < 0) ? -psc : 0.;
141 aii[k] = 0.;
142 }
143 }
