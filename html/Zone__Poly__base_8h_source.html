74//! class Zone_Poly_base<br>Classe instanciable qui derive de Zone_VF.<br>Cette classe contient les informations geometriques que demande<br>la methode des Volumes Elements Finis (element de Crouzeix-Raviart)<br>La classe porte un certain nombre d'informations concernant les faces<br>Dans cet ensemble de faces on fait figurer aussi les faces du bord et<br>des joints. Pour manipuler les faces on distingue 2 categories:<br>- les faces non standard qui sont sur un joint, un bord ou qui sont<br>internes tout en appartenant a un element du bord<br>- les faces standard qui sont les faces internes n'appartenant pas<br>a un element du bord<br>Cette distinction correspond au traitement des conditions aux limites:les<br>faces standard ne "voient pas" les conditions aux limites.<br>L'ensemble des faces est numerote comme suit:<br>- les faces qui sont sur une Zone_joint apparaissent en premier<br>(dans l'ordre du vecteur les_joints)<br>- les faces qui sont sur une Zone_bord apparaissent ensuite<br>(dans l'ordre du vecteur les_bords)<br>- les faces internes non standard apparaissent ensuite<br>- les faces internes standard en dernier<br>Finalement on trouve regroupees en premier toutes les faces non standard<br>qui vont necessiter un traitement particulier<br>On distingue deux types d'elements<br>- les elements non standard : ils ont au moins une face de bord<br>- les elements standard : ils n'ont pas de face de bord<br>Les elements standard (resp. les elements non standard) ne sont pas ranges<br>de maniere consecutive dans l'objet Zone. On utilise le tableau<br>rang_elem_non_std pour acceder de maniere selective a l'un ou<br>l'autre des types d'elements

276inline double Zone_Poly_base::dot(const double *a, const double *b, const double *ma, const double *mb) const
277{
278 double res = 0;
279 for (int i = 0; i < dimension; i++) res += (a[i] - (ma ? ma[i] : 0)) * (b[i] - (mb ? mb[i] : 0));
280 return res;
281}

283/* produit vectoriel de deux vecteurs (toujours 3D, meme en 2D) */
284inline std::array<double, 3> Zone_Poly_base::cross(int dima, int dimb, const double *a, const double *b, const double *ma, const double *mb) const
285{
286 std::array<double, 3> va = {{ 0, 0, 0 }}, vb = {{ 0, 0, 0 }}, res;
287 for (int i = 0; i < dima; i++) va[i] = a[i] - (ma ? ma[i] : 0);
288 for (int i = 0; i < dimb; i++) vb[i] = b[i] - (mb ? mb[i] : 0);
289 for (int i = 0; i < 3; i++) res[i] = va[(i + 1) % 3] * vb[(i + 2) % 3] - va[(i + 2) % 3] * vb[(i + 1) % 3];
290 return res;
291}

293/* produit matricel et transposee de DoubleTab */
294static inline DoubleTab prod(DoubleTab a, DoubleTab b)
295{
296 int i, j, k, m = a.dimension(0), n = a.dimension(1), p = b.dimension(1);
297 assert(n == b.dimension(0));
298 DoubleTab r(m, p);
299 for (i = 0; i < m; i++)
300 for (j = 0; j < n; j++)
301 for (k = 0; k < p; k++) r(i, k) += a(i, j) * b(j, k);
302 return r;
303}
304static inline DoubleTab transp(DoubleTab a)
305{
306 int i, j, m = a.dimension(0), n = a.dimension(1);
307 DoubleTab r(n, m);
308 for (i = 0; i < m; i++)
309 for (j = 0; j < n; j++) r(j, i) = a(i, j);
310 return r;
311}

313/* minimise ||M.x - b||_2, met le noyau de M dans P et retourne le residu */
314static inline double kersol(const DoubleTab& M, DoubleTab& b, double eps, DoubleTab *P, DoubleTab& x, DoubleTab& S)
315{
316 int i, nk, m = M.dimension(0), n = M.dimension(1), k = std::min(m, n), l = std::max(m, n), w = 5 * l, info, iP, jP;
317 double res2 = 0;
318 char a = 'A';
319// lapack en mode Fortran -> on decompose en fait Mt!!
320 DoubleTab A = M, U(m, m), Vt(n, n), W(w), iS(n, m);
321 S.resize(k);
322 F77NAME(dgesvd)(&a, &a, &n, &m, A.addr(), &n, S.addr(), Vt.addr(), &n, U.addr(), &m, W.addr(), &w, &info);
323 for (i = 0, nk = n; i < k && S(i) > eps * S(0); i++) nk--;
324 if (P) P->resize(n, nk);
325 for (i = 0, jP = -1; i < n; i++)
326 if (i < k && S(i) > eps * S(0)) iS(i, i) = 1 / S(i); //terme diagonal de iS
327 else if (P)
328 for (iP = 0, jP++; iP < n; iP++) (*P)(iP, jP) = Vt(i, iP); //colonne de V -> colonne de P
329 x = prod(transp(Vt), prod(iS, prod(transp(U), b)));
330 DoubleTab res = prod(M, x);
331 for (i = 0; i < m; i++) res2 += std::pow(res(i, 0) - b(i, 0), 2);
332 return sqrt(res2);
333}

369// renvoie le produit scalaire a.nu.b quelle que soient le nombre de composantes et le type de tenseur de nu
370inline double Zone_Poly_base::nu_dot(const DoubleTab* nu, int e, int n, const double *a, const double *b, const double *ma, const double *mb) const
371{
372 if (!nu) return dot(a, b, ma, mb);
373 int d, db, D = dimension;
374 double resu = 0;
375 if (nu->nb_dim() == 2) resu += (*nu)(e, n) * dot(a, b, ma, mb); //isotrope
376 else if (nu->nb_dim() == 3)
377 for (d = 0; d < D; d++) //anisotrope diagonal
378 resu += (*nu)(e, n, d) * (a[d] - (ma ? ma[d] : 0)) * (b[d] - (mb ? mb[d] : 0));
379 else for (d = 0; d < D; d++)
380 for (db = 0; db < D; db++)
381 resu += (*nu)(e, n, d, db) * (a[d] - (ma ? ma[d] : 0)) * (b[db] - (mb ? mb[db] : 0));
382 return resu;
383}
