60int Champ_Generique_refChamp::lire_motcle_non_standard(const Motcle& mot, Entree& is)
61{
62 if (mot=="Pb_champ")
63 {
64// Lecture du nom du probleme et du nom du champ
65 is >> nom_pb_ >> nom_champ_;
66// Recherche du probleme parmi les objets connus de l'interprete
67 const Objet_U& ob = interprete().objet(nom_pb_);
68 if (!sub_type(Probleme_base, ob))
69 {
70 Cerr << "Error in Champ_Generique_refChamp ::lire(keyword=\"pb_champ\"\n"
71 << " The object " << nom_pb_ << " is not a Probleme_base" << finl;
72 exit();
73 }
74 Probleme_base& pb = ref_cast_non_const(Probleme_base, ob);
75 ref_pb_ = pb;
76// Recherche du champ "nom_champ" dans le probleme:
77 REF(Champ_base) ref_champ;
78 Noms liste_noms;
79 pb.get_noms_champs_postraitables(liste_noms);
80 pb.creer_champ(nom_champ_);
81 ref_champ = pb.get_champ(nom_champ_);
82 ref_champ.valeur().corriger_unite_nom_compo();
83 set_ref_champ(ref_champ.valeur());
84 return 1;
85 }
86 else if (mot=="nom_source")
87 {
88 Nom id_source;
89 is >> id_source;
90 nommer(id_source);
91 return 1;
92 }
93 else
94 return Champ_Generique_base::lire_motcle_non_standard(mot,is);
95 return 1;
96}

126const Noms Champ_Generique_refChamp::get_property(const Motcle& query) const
127{
128
129 Motcles motcles(5);
130 motcles[0] = "nom";
131 motcles[1] = "nom_cible";
132 motcles[2] = "unites";
133 motcles[3] = "composantes";
134 motcles[4] = "synonyms";
135
136 int rang = motcles.search(query);
137 switch(rang)
138 {
139 case 0:
140 {
141 Noms mots(1);
142 mots[0] = nom_post_;
143 return mots;
144 break;
145 }
146 case 1:
147 {
148 Noms mots(1);
149 mots[0] = nom_champ_;
150 return mots;
151 break;
152 }
153 case 2 :
154 {
155 const Noms mots0= get_ref_champ_base().unites();
156 ref_cast_non_const(Champ_base,get_ref_champ_base()).corriger_unite_nom_compo();
157 const Noms mots = get_ref_champ_base().unites();
158 if (mots.size()!=mots0.size())
159 {
160 Cerr<<"iuuuuuu"<<mots<<" "<<mots0<<finl;
161 exit();
162 }
163 for (int i=0; i<mots.size(); i++)
164 {
165 if (mots0[i]!=mots[i])
166 {
167 Cerr <<" iiiiiiiiii"<<mots0[i]<< " "<<mots[i]<<finl;
168 exit();
169 }
170 }
171 return mots;
172 break;
173 }
174 case 4 :
175 {
176 const Noms mots = get_ref_champ_base().get_synonyms();
177
178 return mots;
179 break;
180 }
181 case 3 :
182 {
183
184 {
185 const Noms mots = get_ref_champ_base().noms_compo();
186 int nb_comp = mots.size();
187
188 Noms compo(nb_comp);
189 for (int i=0; i<nb_comp; i++)
190 {
191 Nom nume(i);
192 compo[i] = nom_post_+nume;
193 }
194 return compo;
195 break;
196 }
197
198 }
199 default :
200 {
201 Cerr<<"The identifiable properties are : "<<motcles<<finl;
202 exit();
203 }
204 }
205// Pour compilation
206// On n'arrive jamais ici
207 return get_property(query);
208
209}

215//! Si le champ n'est pas un champ discret : exception Champ_Generique_erreur("INVALID")<br>Sinon, renvoie la localisation du champ pour le support "index".<br>Si index == -1 (valeur par defaut), leve une exception si le champ est multisupport.<br>Sinon, leve une exception si l'index est superieur au nombre de localisations du champ.
216Entity Champ_Generique_refChamp::get_localisation(const int index) const
217{
218 Entity loc;
219// Pour initialisation
220 loc = NODE;
221
222 const Champ_base& ch = get_ref_champ_base();
223 const Zone_dis_base& z_dis_base = get_ref_zone_dis_base();
224
225// Champs discrets a une seule localisation :
226 if ((sub_type(Champ_Inc_P0_base, ch) || sub_type(Champ_Fonc_P0_base, ch)) && index <= 0)
227 {
228 loc = ELEMENT;
229 }
230 else if ((sub_type(Champ_Inc_P1_base, ch)|| sub_type(Champ_Fonc_P1_base, ch)) && index <= 0)
231 {
232 loc = NODE;
233 }
234 else if ((ch.que_suis_je().debute_par("Champ_Face_PolyMAC")
235 || ch.valeurs().dimension(0) == ref_cast(Zone_VF,z_dis_base).nb_faces()) && index <= 0)
236 {
237 loc = FACE;
238 }
239 else
240 {
241// Champs discrets a plusieurs localisations
242 Nom message="Invalid localization used for postprocessing the field ";
243 message+=ch.le_nom()+". Change your data file.";
244 throw Champ_Generique_erreur(message);
245 }
246 return loc;
247}

417//
418//! Associe le champ et determine sa localisation.
419void Champ_Generique_refChamp::set_ref_champ(const Champ_base& champ)
420{
421 assert(!ref_champ_.non_nul());
422 ref_champ_ = champ;
423
424// Determination de la localisation du champ
425// On code quelques champs de types connus...
426// Pour les autres, il faut les interpoler quelque part.
427// L'ideal serait la localisation soit une propriete du champ discret lui-meme.
428
429 Nom type = champ.que_suis_je();
430 type.majuscule();
431
432 if (type.debute_par("CHAMP_P0"))
433 {
434 localisation_ = "ELEMENTS";
435 }
436 else if (type == "CHAMP_P1")
437 {
438 localisation_ = "SOMMETS";
439 }
440 else if (type.debute_par("CHAMP_FACE"))
441 {
442 localisation_ = "FACES";
443 }
444 else if (type == "CHAMP_P1NC")
445 {
446 localisation_ = "FACES";
447 }
448 else
449 {
450 localisation_ = "LOCALISATION_INCONNUE";
451 }
452}
