72int Partitionneur_Metis::lire_motcle_non_standard(const Motcle& mot, Entree& is)
73{
74 if (mot=="pmetis")
75 {
76 Cerr << " Algorithm PMETIS" << finl;
77 algo_ = PMETIS;
78 return 1;
79 }
80 else if (mot=="kmetis")
81 {
82 Cerr << " Algorithm KMETIS" << finl;
83 algo_ = KMETIS;
84 return 1;
85 }
86 else if (mot=="match_type")
87 {
88 not_implemented(mot);
89 return 1;
90 }
91 else if (mot=="initial_partition_type")
92 {
93 not_implemented(mot);
94 return 1;
95 }
96 else if (mot=="refinement_type")
97 {
98 not_implemented(mot);
99 return 1;
100 }
101 else
102 return Partitionneur_base::lire_motcle_non_standard(mot,is);
103
104 if(Process::nproc() >1)
105 {
106 Cerr << "WARNING! You're using a sequential algorithm on " << nproc() << "processors " << finl;
107 Cerr << "Use PARMETIS for parallel domain cutting" << finl;
108 Process::exit();
109 }
110 return 1;
111}

125//! Calcule le graphe de connectivite pour Metis, appelle le partitionneur<br>et remplit elem_part (pour chaque element, numero de la partie qui lui<br>est attribuee).<br>Les parties sont equilibrees de facon a minimiser le nombre de faces de joint<br>et a equilibrer le nombre d'elements par partie.
126void Partitionneur_Metis::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
127{
128#ifdef NO_METIS
129 Cerr << "METIS is not compiled with this version. Use another partition tool like Tranche." << finl;
130 Process::exit();
131#else
132 if (!ref_domaine_.non_nul())
133 {
134 Cerr << "Error in Partitionneur_Metis::construire_partition\n";
135 Cerr << " The domain has not been associated" << finl;
136 exit();
137 }
138 if (nb_parties_ <= 0)
139 {
140 Cerr << "Error in Partitionneur_Metis::construire_partition\n";
141 Cerr << " The parts number has not been initialized" << finl;
142 exit();
143 }
144
145// Cas particulier: si nb_parts == 1, METIS ne veut rien faire...
146 if (nb_parties_ == 1)
147 {
148
149 int nb_elem = ref_domaine_.valeur().zone(0).nb_elem_tot();
150 if (use_segment_to_build_connectivite_elem_elem_==1)
151 nb_elem = ref_domaine_.valeur().nb_som();
152 elem_part.resize(nb_elem);
153 elem_part = 0;
154 return;
155 }
156
157 if (ref_domaine_.valeur().zone(0).nb_elem() == 0)
158 return;
159
160 Cerr << "Partitionneur_Metis::construire_partition" << finl;
161 Cerr << " Construction of graph connectivity..." << finl;
162 Static_Int_Lists graph_elements_perio;
163// const Domaine& dom = ref_domaine_.valeur();
164 Domain_Graph graph;
165 if (use_segment_to_build_connectivite_elem_elem_==0)
166 {
167 graph.construire_graph_elem_elem(ref_domaine_.valeur(), liste_bords_periodiques_,
168 use_weights_,
169 graph_elements_perio);
170
171 }
172 else
173 {
174 graph.construire_graph_from_segment(ref_domaine_.valeur(), use_weights_);
175
176 }
177 std::vector<int> partition(graph.nvtxs);
178 int int_parts = nb_parties_;
179 int edgecut = 0; //!< valeur renvoyee par metis (nombre total de faces de joint)
180
181 switch(algo_)
182 {
183 case PMETIS:
184 {
185 Cerr << "===============" << finl;
186 Cerr << "Call for PMETIS" << finl;
187 Cerr << "===============" << finl;
188// Voir le manual.pdf de METIS 5.0
189 int options[METIS_NOPTIONS];
190 METIS_SetDefaultOptions(options);
191// options[METIS_OPTION_PTYPE]=METIS_PTYPE_RB|METIS_PTYPE_KWAY; // Methode de partitionnement
192// options[METIS_OPTION_OBJTYPE]=METIS_OBJTYPE_CUT|METIS_OBJTYPE_VOL; // Objective type
193// options[METIS_OPTION_CTYPE]=METIS_CTYPE_SHEM|METIS_CTYPE_RM; // Matching scheme during coarsening
194// options[METIS_OPTION_IPTYPE]=METIS_IPTYPE_GROW; // Algorithm during initial partitioning
195// options[METIS_OPTION_RTYPE]=METIS_RTYPE_FM; // Algorithm for refinement
196 options[METIS_OPTION_NCUTS]=nb_essais_; //!< Nombre de partitionnements testes pour en prendre le meilleur
197 options[METIS_OPTION_NUMBERING]=0; //!< Numerotation C qui demarre a 0
198 options[METIS_OPTION_DBGLVL]=111111111; //!< Mode verbose maximal options[METIS_OPTION_NO2HOP]=1; // 5.1.0: not perform any 2-hop matchings (as 5.0.3)
199//
200 int ncon=1;
201
202// Implementation reduite (plusieurs valeurs par defaut->NULL) pour METIS 5.0
203 int status = METIS_PartGraphRecursive(&graph.nvtxs, &ncon, graph.xadj.addr(),
204 graph.adjncy.addr(), graph.vwgts.addr(), NULL, graph.ewgts.addr(),
205 &int_parts, NULL, NULL, options,
206 &edgecut, partition.data());
207 if (status != METIS_OK)
208 {
209 Cerr << "Call to METIS failed." << finl;
210 if (status == METIS_ERROR_INPUT) Cerr << "It seems there is an input error." << finl;
211 if (status == METIS_ERROR_MEMORY) Cerr << "It seems it couldn't allocate enough memory." << finl;
212 if (status == METIS_ERROR) Cerr << "It seems there is a METIS internal error." << finl;
213 Cerr << "Contact TRUST support." << finl;
214 exit();
215 }
216 Cerr << "===============" << finl;
217 break;
218 }
219 case KMETIS:
220 {
221 Cerr << " Call for KMETIS" << finl;
222 int options[METIS_NOPTIONS];
223 METIS_SetDefaultOptions(options);
224// options[METIS_OPTION_PTYPE]=METIS_PTYPE_RB|METIS_PTYPE_KWAY; // Methode de partitionnement
225// options[METIS_OPTION_OBJTYPE]=METIS_OBJTYPE_CUT|METIS_OBJTYPE_VOL; // Objective type
226// options[METIS_OPTION_CTYPE]=METIS_CTYPE_SHEM; // Matching scheme during coarsening
227// options[METIS_OPTION_IPTYPE]=METIS_IPTYPE_GROW; // Algorithm during initial partitioning
228// options[METIS_OPTION_RTYPE]=METIS_RTYPE_FM; // Algorithm for refinement
229 options[METIS_OPTION_NCUTS]=nb_essais_; //!< Nombre de partitionnements testes pour en prendre le meilleur
230 options[METIS_OPTION_NUMBERING]=0; //!< Numerotation C qui demarre a 0
231 options[METIS_OPTION_DBGLVL]=111111111; //!< Mode verbose maximal
232 int ncon=1;
233// Conseil de la doc Metis 4.0 : METIS_PartGraphKway si int_parts>8, METIS_PartGraphRecursive sinon...
234// En effet semble plus rapide, mais edgecut en sortie est moins bon...
235 int status = METIS_PartGraphKway(&graph.nvtxs, &ncon, graph.xadj.addr(),
236 graph.adjncy.addr(), graph.vwgts.addr(), NULL, graph.ewgts.addr(),
237 &int_parts, NULL, NULL, options ,
238 &edgecut, partition.data());
239 if (status != METIS_OK)
240 {
241 Cerr << "Call to METIS failed." << finl;
242 if (status == METIS_ERROR_INPUT) Cerr << "It seems there is an input error." << finl;
243 if (status == METIS_ERROR_MEMORY) Cerr << "It seems it couldn't allocate enough memory." << finl;
244 if (status == METIS_ERROR) Cerr << "It seems there is a METIS internal error." << finl;
245 Cerr << "Contact TRUST support." << finl;
246 exit();
247 }
248 break;
249 }
250 default:
251 {
252 Cerr << "Internal error Partitionneur_Metis: not coded" << finl;
253 exit();
254 }
255 }
256 Cerr << "Partitioning quality : edgecut = " << edgecut << finl;
257 Cerr << "-> It is roughly the total number of edges (faces) which will be shared by the processors." << finl;
258 Cerr << "-> The lesser this number is, the lesser the total volume of communication between processors." << finl;
259 Cerr << "-> You can increase nb_essais option (default 1) to try to reduce (but at a higher CPU cost) this number." << finl;
260 Cerr << "===============" << finl;
261
262 const int n = graph.nvtxs;
263 elem_part.resize(n);
264 for (int i = 0; i < n; i++)
265 elem_part[i] = partition[i];
266
267// Correction de la partition pour la periodicite. (***)
268 if (graph_elements_perio.get_nb_lists() > 0)
269 {
270 Cerr << "Correction of the partition for the periodicity" << finl;
271 corriger_bords_avec_liste(ref_domaine_.valeur(),
272 liste_bords_periodiques_,
273 0,
274 elem_part);
275 Cerr << " If this number is high, we can improve the splitting with the option use_weights\n"
276 << " but it takes more memory)" << finl;
277 }
278
279 if (use_segment_to_build_connectivite_elem_elem_==0)
280 {
281 Cerr << "Correction elem0 on processor 0" << finl;
282 corriger_elem0_sur_proc0(elem_part);
283 }
284
285#endif
286}
