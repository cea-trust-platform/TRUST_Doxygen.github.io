86inline void Eval_Darcy_VEF_Face::mettre_a_jour()
87{
88 int nb_faces_tot = la_zone.valeur().nb_faces_tot();
89 if (sub_type(Champ_Uniforme,diffusivite_.valeur()))
90 {
91 db_diffusivite = diffusivite_.valeur()(0,0);
92 }
93 else
94 {
95 const DoubleTab& val_diff = diffusivite_.valeur().valeurs();
96 const IntTab& face_vois = la_zone.valeur().face_voisins();
97 const DoubleVect& volumes = la_zone.valeur().volumes();
98 db_diffusivite = 0.;
99// Compute nu(fac)=(nu(elem1).vol(elem1)+nu(elem2).vol(elem2))/(vol(elem1)+vol(elem2))
100// Viscosity at face is the volume weighted average of viscosity on elements
101 for (int fac=0; fac<nb_faces_tot; fac++)
102 {
103 int elem1 = face_vois(fac,0);
104 int elem2 = face_vois(fac,1);
105 double vol = 0;
106 if (elem1!=-1)
107 {
108 db_diffusivite(fac) += val_diff(elem1)*volumes(elem1);
109 vol += volumes(elem1);
110 }
111 if (elem2!=-1)
112 {
113 db_diffusivite(fac) += val_diff(elem2)*volumes(elem2);
114 vol += volumes(elem2);
115 }
116 db_diffusivite(fac) /= vol;
117 }
118 }
119}

122inline void Eval_Darcy_VEF_Face::calculer_terme_source(int num_face, DoubleVect& source, const DoubleVect& volumes) const
123{
124 int size=source.size();
125 for (int i=0; i<size; i++)
126 source(i) = -db_diffusivite(num_face)/modK->getK(porosite)*volumes(num_face)*porosite_surf(num_face)*(vitesse->valeurs()(num_face,i));
127}
