42void Champ_Fonc_Tabule_P0_VDF::mettre_a_jour(double t)
43{
44 const Zone_VF& zone_VF = la_zone_VF.valeur();
45 const Table& table = la_table.valeur();
46 DoubleTab& mes_valeurs = valeurs();
47 const int nb_elem = zone_VF.nb_elem(), nb_elem_tot = zone_VF.nb_elem_tot(), nb_param = les_ch_param.size();
48 DoubleTabs val_params_aux_elems;
49 for (int i = 0; i < nb_param; i++)
50 {
51 assert(les_ch_param[i]->valeurs().dimension(1) == 1 || les_ch_param[i]->valeurs().dimension(1) == mes_valeurs.dimension(1));
52 DoubleTab vp(nb_elem_tot, les_ch_param[i]->valeurs().dimension(1));
53 val_params_aux_elems.add(vp);
54 }
55 const DoubleTab& centres_de_gravites = zone_VF.xp();
56 IntVect les_polys(nb_elem_tot);
57 for(int elem = 0; elem < nb_elem_tot; elem++) les_polys(elem) = elem;
58
59// Estimate the field parameter on cells:
60 for (int i = 0; i < nb_param; i++)
61 les_ch_param[i].valeur().valeur_aux_elems(centres_de_gravites, les_polys, val_params_aux_elems[i]);
62// Compute the field according to the parameter field
63 if (table.isfonction() != 2)
64 {
65 const int nbcomp = mes_valeurs.dimension(1);
66 for (int num_elem = 0; num_elem < nb_elem; num_elem++)
67 for (int ncomp = 0; ncomp < nbcomp; ncomp++)
68 {
69 std::vector<double> vals;
70 for (int n = 0; n < nb_param; n++) vals.push_back(val_params_aux_elems[n](num_elem, les_ch_param[n]->valeurs().dimension(1) == 1 ? 0 : ncomp));
71 mes_valeurs(num_elem, ncomp) = table.val(vals, ncomp);
72 }
73 }
74 else
75 {
76 table.valeurs(val_params_aux_elems[0], centres_de_gravites, t, mes_valeurs);
77 }
78
79
80 Champ_Fonc_base::mettre_a_jour(t);
81}
