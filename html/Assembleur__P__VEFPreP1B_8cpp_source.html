51static double estim_cond(Matrice& A)
52{
53 int i=A.valeur( ).nb_lignes();
54// Cout << "rang " << i << finl;
55 DoubleVect p(A.valeur( ).nb_lignes());
56 for(int j=0; j<i; j++)
57 p(j)=drand48();
58 double normep=sqrt(mp_prodscal(p,p));
59 p/=normep;
60 DoubleVect r(A.valeur( ).nb_lignes());
61 double lmax=1.e-36, lmin=1.e24;
62// double cond=-1;
63 i=10000;
64 while(i-- )
65 {
66// cond=lmax/lmin;
67 A.valeur( ).multvect(p,r);
68 double pscpr=mp_prodscal(p,r);
69// Cout << "lambda " << i << " : " << pscpr << " , " << lmax/lmin << finl;
70 double pscrr=mp_prodscal(r,r);
71 double epsilon=-pscpr/pscrr;
72 p.ajoute(epsilon,r);
73 double normepbis=sqrt(mp_prodscal(p,p));
74// Cout << "normep : " << normep << finl;
75 p/=normepbis;
76 if(pscpr>lmax)
77 lmax=pscpr;
78 else if(pscpr<lmin)
79 lmin=pscpr;
80 }
81// Cout << "lmax " << lmax << " lmin = " << lmin << finl;
82// Cout << p << finl;
83 if (A.valeur( ).nb_lignes()<30) A.valeur( ).imprimer_formatte(Cerr);
84
85 return(lmax/lmin);
86}

218int Assembleur_P_VEFPreP1B::assembler_mat(Matrice& la_matrice,const DoubleVect& quantitee_entrelacee, int incr_pression, int resoudre_en_u)
219{
220// On fixe les drapeaux de Assembleur_base
221 set_resoudre_increment_pression(incr_pression);
222 set_resoudre_en_u(resoudre_en_u);
223
224 SolveurSys& solveur_pression = ref_cast(Navier_Stokes_std, mon_equation.valeur()).solveur_pression();
225 bool read_matrix = false;
226 if (sub_type(Solv_Petsc, solveur_pression.valeur()))
227 read_matrix = ref_cast(Solv_Petsc, solveur_pression.valeur()).read_matrix();
228 if (read_matrix) //!< Lecture de la matrice dans un fichier
229 {
230 la_matrice.typer("Matrice_Petsc");
231// ref_cast(Matrice_Petsc, la_matrice.valeur()).RestoreMatrixFromFile();
232 }
233 else //!< Assemblage de la matrice
234 {
235 const Zone_VEF_PreP1b& zone_vef = zone_Vef();
236 Cerr << "Assemblage de la matrice de pression" << (zone_vef.get_alphaE() ? " P0" : "")
237 << (zone_vef.get_alphaS() ? " P1" : "") << (zone_vef.get_alphaA() ? " Pa" : "") << " en cours..." << finl;
238
239// Les decoupages doivent etre de largeur de joint de 2
240// si le support P1 ou Pa est utilise...
241 if (Process::nproc() > 1 &&
242 zone_vef.zone().nb_joints() &&
243 zone_vef.zone().joint(0).epaisseur() < 2 &&
244 (zone_vef.get_alphaS() || zone_vef.get_alphaA()))
245 {
246 Cerr << "Ghost cells width of " << zone_vef.zone().joint(0).epaisseur()
247 << " is not enough for assembling VEFPreP1B pressure matrix" << finl;
248 Cerr << "for parallel calculation. Partition your mesh with larg_joint option set to 2." << finl;
249 Process::exit();
250 }
251
252 DoubleTab inverse_quantitee_entrelacee;
253 const Zone_Cl_VEF& zone_Cl_VEF = la_zone_Cl_VEF.valeur();
254 calculer_inv_volume(inverse_quantitee_entrelacee, zone_Cl_VEF, quantitee_entrelacee);
255 int P0 = 0;
256 int P1 = P0 + zone_vef.get_alphaE();
257 int Pa = P1 + zone_vef.get_alphaS();
258 int nombre_supports = Pa + zone_vef.get_alphaA();
259 DoubleVect coef_som;
260 if (zone_vef.get_alphaS())
261 {
262 zone_vef.zone().creer_tableau_elements(coef_som);
263 for (int elem = 0; elem < coef_som.size_totale(); elem++)
264 coef_som[elem] = Op_Grad_VEF_P1B_Face::calculer_coef_som(elem, zone_Cl_VEF, zone_vef);
265 coef_som.echange_espace_virtuel();
266 }
267
268// Assemblage de la matrice complete selon les supports choisis
269 Matrice_Bloc_Sym& la_matrice_bloc_sym_de_travail = ref_cast(Matrice_Bloc_Sym, la_matrice_de_travail_.valeur());
270 if (la_matrice_bloc_sym_de_travail.nb_bloc_lignes()==0)
271 {
272 la_matrice_bloc_sym_de_travail.dimensionner(nombre_supports, nombre_supports);
273 if (zone_vef.get_alphaE())
274 assemblerP0P0(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P0, P0),
275 inverse_quantitee_entrelacee);
276
277 if (zone_vef.get_alphaS())
278 assemblerP1P1(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P1, P1),
279 inverse_quantitee_entrelacee, coef_som);
280
281 if (zone_vef.get_alphaE() && zone_vef.get_alphaS())
282 assemblerP0P1(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P0, P1),
283 inverse_quantitee_entrelacee, coef_som);
284
285 if (zone_vef.get_alphaA())
286 assemblerPaPa(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(Pa, Pa),
287 inverse_quantitee_entrelacee);
288
289 if (zone_vef.get_alphaS() && zone_vef.get_alphaA())
290 assemblerP1Pa(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P1, Pa),
291 inverse_quantitee_entrelacee, coef_som);
292
293 if (zone_vef.get_alphaE() && zone_vef.get_alphaA())
294 assemblerP0Pa(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P0, Pa),
295 inverse_quantitee_entrelacee);
296 }
297// On met a zero la matrice meme si elle a ete correctement construite et remplie dans les methodes
298// assemblerPiPi et on la remplit a nouveau. Pourquoi? Pour avoir une couverture de tests
299// suffisante des methodes updatePiPi en attendant de factoriser correctement les
300// methodes assemblerPiPi et updatePiPi
301 zero(la_matrice_bloc_sym_de_travail);
302 {
303 if (zone_vef.get_alphaE())
304 updateP0P0(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P0, P0),
305 inverse_quantitee_entrelacee);
306
307 if (zone_vef.get_alphaS())
308 {
309 updateP1P1(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P1, P1),
310 inverse_quantitee_entrelacee, coef_som);
311 if (zone_vef.get_cl_pression_sommet_faible() == 0)
312 modifieP1P1neumann(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P1, P1),
313 inverse_quantitee_entrelacee, coef_som);
314 }
315 if (zone_vef.get_alphaE() && zone_vef.get_alphaS())
316 updateP0P1(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P0, P1),
317 inverse_quantitee_entrelacee, coef_som);
318
319 if (zone_vef.get_alphaA())
320 updatePaPa(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(Pa, Pa),
321 inverse_quantitee_entrelacee);
322
323 if (zone_vef.get_alphaS() && zone_vef.get_alphaA())
324 updateP1Pa(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P1, Pa),
325 inverse_quantitee_entrelacee, coef_som);
326
327 if (zone_vef.get_alphaE() && zone_vef.get_alphaA())
328 updateP0Pa(zone_vef, zone_Cl_VEF, la_matrice_bloc_sym_de_travail.get_bloc(P0, Pa),
329 inverse_quantitee_entrelacee);
330 }
331 int ordre_matrice = mp_sum(la_matrice_bloc_sym_de_travail.nb_lignes());
332 Cerr << "Order of the matrix = " << ordre_matrice << finl;
333
334// Methode verifier
335 char *theValue = getenv("TRUST_VERIFIE_MATRICE_VEF");
336 if (theValue != NULL) verifier(*this, la_matrice_bloc_sym_de_travail, zone_vef, inverse_quantitee_entrelacee);
337
338//
339// Changement de base eventuel P0P1->P1Bulle
340//
341 if (changement_base())
342 changer_base_matrice(la_matrice_de_travail_); //!< A->A~
343
344//
345// Modification de la matrice si pas de pression de reference
346//
347 modifier_matrice(la_matrice_de_travail_);
348
349// Conversion eventuelle en Matrice_Morse_Sym
350 if (ref_cast(Navier_Stokes_std, mon_equation.valeur()).solveur_pression().supporte_matrice_morse_sym() &&
351 zone_vef.get_alphaE() != nombre_supports) //!< On n'est pas en P0
352 {
353//
354// La matrice retournee est une Matrice_Morse_Sym nettoyee
355// si le solveur utilisee supporte ce type de matrice
356// et si on n'est pas en P0 seulement (dans ce cas la, la
357// conversion en Mat_Morse_Sym n'apporte rien, et plante le SSOR
358// car la matrice morse contient alors des parties virtuelles
359// alors qu'il n'y a pas d'items communs et on tombe sur
360// l'assert assert(*tab2_ptr<=n); Deux autres solutions:
361// -Modifier le SSOR pour ne resoudre que la partie reelle
362// -Dans le cas de P0 seul, il faudrait vider VV et RV dans la Mat_Bloc_Sym
363//
364 la_matrice.typer("Matrice_Morse_Sym");
365 ref_cast(Matrice_Bloc_Sym, la_matrice_de_travail_.valeur()).BlocSymToMatMorseSym(
366 ref_cast(Matrice_Morse_Sym, la_matrice.valeur()));
367 ref_cast(Matrice_Morse_Sym, la_matrice.valeur()).compacte(
368 2);//!< Suppression des coefficients nuls et quasi non nuls
369 }
370 else
371 {
372//
373// La matrice retournee est une Matrice_Bloc_Sym nettoyee
374//
375 la_matrice = la_matrice_de_travail_;
376 for (int i = 0; i < nombre_supports; i++)
377 for (int j = i; j < nombre_supports; j++)
378 {
379 Matrice_Bloc_Sym& mat_bloc_sym = ref_cast(Matrice_Bloc_Sym, la_matrice.valeur());
380 Matrice_Bloc& bloc_ij = ref_cast(Matrice_Bloc, mat_bloc_sym.get_bloc(i, j).valeur());
381 ref_cast(Matrice_Morse, bloc_ij.get_bloc(0, 0).valeur()).compacte(
382 2); //!< Suppression des coefficients nuls et quasi non nuls
383 }
384 }
385 }
386
387// Si pas deja fait, on prends un solveur (SolveurPP1B) qui fera les changements de base pour la solution et le second membre
388 if (changement_base() && solveur_pression.valeur().que_suis_je() != "SolveurPP1B")
389 {
390 SolveurSys solveur_pression_lu = solveur_pression;
391 solveur_pression.typer("SolveurPP1B");
392 SolveurPP1B& solveur_pression_PP1B = ref_cast(SolveurPP1B, solveur_pression.valeur());
393 solveur_pression_PP1B.associer(*this, solveur_pression_lu);
394 }
395
396//
397// Affichage eventuel du conditionnement de la matrice
398//
399 char* theValue2 = getenv("TRUST_CONDITIONNEMENT_MATRICE");
400 if(theValue2 != NULL)
401 Cout << "Estimation du conditionnement de la matrice: " << estim_cond(la_matrice)<<finl;
402
403 return 1;
404}

406int Assembleur_P_VEFPreP1B::modifier_secmem(DoubleTab& b)
407{
408 const Zone_VEF_PreP1b& la_zone = zone_Vef();
409
410// Verification sur le support Pa
411 if (la_zone.get_alphaA())
412 {
413// Verification sur les aretes que:
414 const ArrOfInt& renum_arete_perio=la_zone.get_renum_arete_perio();
415 const ArrOfInt& ok_arete=la_zone.get_ok_arete();
416 int npa=la_zone.numero_premiere_arete();
417// b n'a pas forcement son espace virtuel a jour
418 int nb_aretes=la_zone.zone().nb_aretes();
419 for(int i=0; i<nb_aretes; i++)
420 if(!ok_arete[i] && b(npa+i)!=0.) //!< Les aretes superflues ont une valeur nulle
421 {
422 Cerr << "Pb div Aretes, la pression sur l'arete " << i << " (qui est superflue) n'est pas nulle." << finl;
423 Process::exit();
424 }
425 else if ( (renum_arete_perio[i]!=i) && b(npa+i)!=0.) //!< Les aretes periodiques ont une valeur nulle
426 {
427 Cerr << "Pb div Aretes Perio, la pression sur l'arete " << i << " (qui est periodique) n'est pas nulle." << finl;
428 Process::exit();
429 }
430 }
431
432// Verification sur le support P1
433 if (la_zone.get_alphaS())
434 {
435// Verification que la pression sur les sommets periodiques est nulle
436 const Domaine& dom=la_zone.zone().domaine();
437 int nps=la_zone.numero_premier_sommet();
438 int ns=la_zone.zone().nb_som();
439 for(int i=0; i<ns; i++)
440 {
441 int k=dom.get_renum_som_perio(i);
442 if((k!=i)&& b(nps+i)!=0.)
443 {
444 Cerr << "Pb div Som, la pression sur le sommet " << i << " (qui est periodique) n'est pas nulle." << finl;
445 Cerr << "En terme clair, le second membre n'est pas nul sur un sommet periodique." << finl;
446 if (b(nps+i)!=b(nps+k))
447 {
448 Cerr << "En outre, le second membre n'a pas la meme valeur sur les 2 sommets periodiques." << finl;
449 Cerr << "b(nps+i)=" << b(nps+i) << " <> b(nps+k)=" << b(nps+k) << finl;
450 }
451 Cerr << "Il y'a probabilite que le modele utilise soit mal implemente pour" << finl;
452 Cerr << "une condition de periodicite. Contacter le support TRUST." << finl;
453 Process::exit();
454 }
455 }
456 }
457
458 if (get_resoudre_en_u())
459 {
460 const Zone_VEF_PreP1b& zone_VEF = zone_Vef();
461 const Zone_Cl_VEF& zone_Cl = la_zone_Cl_VEF.valeur();
462
463 const DoubleVect& porosite_face = zone_VEF.porosite_face();
464
465 const int nb_cond_lim = zone_Cl.nb_cond_lim();
466
467 /**************************/
468 /* Recuperation de Gpoint */
469 /**************************/
470 DoubleTab Gpoint(equation().inconnue().valeurs());
471 Gpoint=0.;
472 int Gpoint_nul = 1; //!< Drapeau pour economiser potentiellement un echange_espace_virtuel
473 for (int cond_lim=0; cond_lim<nb_cond_lim; cond_lim++)
474 {
475 const Cond_lim_base& cl_base = zone_Cl.les_conditions_limites(cond_lim).valeur();
476
477 const Front_VF& front_VF = ref_cast(Front_VF,cl_base.frontiere_dis());
478 const Champ_front_base& champ_front = cl_base.champ_front().valeur();
479
480 const int ndeb = front_VF.num_premiere_face();
481 const int nfin = ndeb+front_VF.nb_faces();
