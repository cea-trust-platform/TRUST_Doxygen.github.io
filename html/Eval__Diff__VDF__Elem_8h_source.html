33 /*
34 * Notes about that MACRO's predefined in the previous version of TRUST (just to understand when,where and why (not sure of why :) )
35 *
36 * - MULTD --TRUST : predefined for Diff/Dift const_Elem / var_Elem / var_Elem_aniso (not Multi-Inco) --TrioCFD : predefined for Diff/Dift K_Eps_var_VDF_Elem & K_Eps_Bas_Re_VDF_var_Elem
37 *
38 * - D_AXI --TRUST : predefined for Diff/Dift Axi operators --TrioCFD : predefined for Diff/Dift Axi operators K_Eps
39 *
40 * - DEQUIV --TRUST : predefined for Dift (not Diff) Multi_inco_const_Elem / Multi_inco_var_Elem && Axi operators const_Elem_Axi / var_Elem_Axi / Multi_inco_var_Elem_Axi / Multi_inco_const_Elem_Axi
41 *
42 * - MODIF_DEQ --TRUST : predefined for Dift (not Diff) const_Elem / var_Elem (no Axi, no Multi_Inco)
43 *
44 * - ISQUASI --TrioCFD : predefined Diff_K_Eps_QC_VDF_Elem && Diff_K_Eps_QC_var_VDF_Elem
45 *
46 * These static constexpr bools & Methods that will generalize the implementation almost for all operator classes. The previous MACRO methodology do not exist anymore
47 * See their generic declaration later & override them in the derived class if needed to stay coherent with the previous macros
48 */
49 static constexpr bool IS_MULTD = true, IS_AXI = false, IS_DEQUIV = false, IS_MODIF_DEQ = false, IS_QUASI = false, IS_ANISO = false, IS_RANS = false;
