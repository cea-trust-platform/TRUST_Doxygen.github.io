31Entree& Force_Tchen_PolyMAC_P0::readOn(Entree& is)
32{
33// identification des phases
34 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
35
36 if (!pbm || pbm->nb_phases() == 1) Process::exit(que_suis_je() + " : not needed for single-phase flow!");
37 for (int n = 0; n < pbm->nb_phases(); n++) //recherche de n_l, n_g : phase {liquide,gaz}_continu en priorite
38 if (pbm->nom_phase(n).debute_par("liquide") && (n_l < 0 || pbm->nom_phase(n).finit_par("continu"))) n_l = n;
39
40 if (n_l < 0) Process::exit(que_suis_je() + " : liquid phase not found!");
41
42 return is;
43}

45void Force_Tchen_PolyMAC_P0::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
46{
47 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
48 if (!matrices.count(ch.le_nom().getString())) return; //rien a faire
49 Matrice_Morse& mat = *matrices.at(ch.le_nom().getString()), mat2;
50
51 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0, equation().zone_dis().valeur());
52 const DoubleTab& inco = ch.valeurs();
53 const IntTab& fcl = ch.fcl();
54
55 /* stencil : diagonal par bloc pour les vitesses aux faces, puis chaque composante des vitesses aux elems */
56 IntTrav stencil(0, 2);
57 stencil.set_smart_resize(1);
58 int N = inco.line_size(), D = dimension, nf_tot = zone.nb_faces_tot(), nf = zone.nb_faces();
59
60 /* faces */
61 for (int f = 0; f < nf; f++)
62 if (fcl(f, 0) < 2)
63 for (int k = 0 ; k<N ; k++)
64 if (k != n_l) //phase gazeuse
65 stencil.append_line( N * f + k , N * f + n_l) ;
66 /* elements */
67 for (int e = 0; e < zone.nb_elem_tot(); e++)
68 for (int d = 0 ; d <D ; d++)
69 for (int k = 0 ; k<N ; k++)
70 if (k != n_l) //phase gazeuse
71 stencil.append_line( N *(nf_tot + D * e + d) + k , N *(nf_tot + D * e + d) + n_l) ;
72
73 tableau_trier_retirer_doublons(stencil);
74 Matrix_tools::allocate_morse_matrix(inco.size_totale(), inco.size_totale(), stencil, mat2);
75 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
76
77}

79void Force_Tchen_PolyMAC_P0::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
80{
81 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
82 Matrice_Morse *mat = matrices.count(ch.le_nom().getString()) ? matrices.at(ch.le_nom().getString()) : NULL;
83
84 const Zone_PolyMAC_P0& zone = ref_cast(Zone_PolyMAC_P0, equation().zone_dis().valeur());
85 const IntTab& f_e = zone.face_voisins(), &fcl = ch.fcl();
86 const DoubleVect& pe = zone.porosite_elem(), &pf = zone.porosite_face(), &ve = zone.volumes(), &vf = zone.volumes_entrelaces();
87
88 const DoubleTab& inco = ch.valeurs(), &pvit = ch.passe(),
89 &alpha = ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe(),
90 &rho = equation().milieu().masse_volumique().passe(),
91 &vf_dir = zone.volumes_entrelaces_dir();
92
93 double pas_tps = equation().probleme().schema_temps().pas_de_temps();
94 int N = inco.line_size(), D = dimension, nf_tot = zone.nb_faces_tot(), nf = zone.nb_faces();
95
96 /* elements */
97 for (int e = 0; e < zone.nb_elem_tot(); e++)
98 for (int d = 0 ; d <D ; d++)
99 for (int k = 0 ; k<N ; k++)
100 if (k != n_l) //phase gazeuse
101 {
102 double fac = pe(e) * ve(e) * alpha(e, k) * rho(e, n_l) ;
103
104 secmem(nf_tot + D * e + d, k) += fac * (inco(nf_tot + D * e + d, n_l)-pvit(nf_tot + D * e + d, n_l))/pas_tps;
105 secmem(nf_tot + D * e + d, k) -= fac * (inco(nf_tot + D * e + d, n_l)-pvit(nf_tot + D * e + d, n_l))/pas_tps;
106 if (mat)
107 {
108 (*mat)( N *(nf_tot + D * e + d) + k , N *(nf_tot + D * e + d) + n_l) -= fac/pas_tps ;
109 (*mat)( N *(nf_tot + D * e + d) + n_l, N *(nf_tot + D * e + d) + n_l) += fac/pas_tps ;
110 }
111 }
112
113 /* faces */
114 for (int f = 0; f < nf; f++)
115 if (fcl(f, 0) < 2)
116 for (int k = 0 ; k<N ; k++)
117 if (k != n_l) //phase gazeuse
118 {
119 double alpha_loc = f_e(f, 1)>=0 ? alpha(f_e(f, 0), k) * vf_dir(f, 0)/vf(f) + alpha(f_e(f, 1), k) * vf_dir(f, 1)/vf(f) : alpha(f_e(f, 0), k) ;
120 double rho_loc = f_e(f, 1)>=0 ? rho(f_e(f, 0), n_l) * vf_dir(f, 0)/vf(f) + rho(f_e(f, 1), n_l) * vf_dir(f, 1)/vf(f) : rho(f_e(f, 0), n_l) ;
121 double fac = pf(f) * vf(f) * alpha_loc * rho_loc ;
122
123 secmem(f, k ) += fac * (inco(f, n_l)-pvit(f, n_l))/pas_tps;
124 secmem(f,n_l) -= fac * (inco(f, n_l)-pvit(f, n_l))/pas_tps;
125 if (mat)
126 {
127 (*mat)( N * f + k , N * f + n_l) -= fac/pas_tps ;
128 (*mat)( N * f + n_l, N * f + n_l) += fac/pas_tps ;
129 }
130 }
131
132}
