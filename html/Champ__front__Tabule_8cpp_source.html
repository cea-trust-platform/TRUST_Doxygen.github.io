57//! Lit les valeurs du champ tabule a partir d'un flot d'entree.<br>Format:<br>Champ_front_Tabule nb_comp { nval tps_1...tps_nval vrel_1...vrel_nval }
58/** \return Entree\&<br>
59Signification: le flot d'entree modifie<br>
60Contraintes: */
61Entree& Champ_front_Tabule::readOn(Entree& is)
62{
63 Motcle motlu;
64 int nb_val;
65 Motcle accolade_ouverte("{");
66 Motcle accolade_fermee("}");
67
68 int dim,i,j;
69 dim=lire_dimension(is,que_suis_je());
70 fixer_nb_comp(dim);
71
72 is >> motlu;
73 if (motlu == accolade_ouverte)
74 {
75 nb_val=lire_dimension(is,que_suis_je());
76 DoubleVect param(nb_val);
77 DoubleTab tab_valeurs(nb_val, dim);
78 for (i=0; i<nb_val; i++)
79 is >> param[i];
80 for (i=0; i<nb_val; i++)
81 for (j=0; j<dim; j++)
82 is >> tab_valeurs(i,j);
83 la_table.remplir(param,tab_valeurs);
84 is >> motlu;
85 if (motlu != accolade_fermee)
86 {
87 Cerr << "Error reading from an object of type Champ_front_Tabule" << finl;
88 Cerr << "We expected keyword } instead of " << motlu << finl;
89 exit();
90 }
91 }
92 else
93 {
94 Cerr << "Error reading from an object of type Champ_front_Tabule" << finl;
95 Cerr << "We expected keyword { instead of " << motlu << finl;
96 exit();
97 }
98 return is;
99}
