82void Champ_Front_Fonction::mettre_a_jour(double temps)
83{
84 const Champ_base& ch=ref_pb.valeur().get_champ(nom_champ_parametre_);
85 const Front_VF& fvf = ref_cast(Front_VF, frontiere_dis());
86 const Zone_VF& zvf = ref_cast(Zone_VF, frontiere_dis().zone_dis());
87 DoubleTab& tab=valeurs_au_temps(temps);
88 DoubleTab tab_ch(tab), pos(tab.dimension_tot(0), dimension);
89 ch.trace(frontiere_dis(),tab_ch,temps,0);
90 assert(tab.dimension(0)==tab_ch.dimension(0));
91 for (int i = 0, j; i < fvf.nb_faces_tot(); i++)
92 for (j = 0; j < dimension; j++) pos(i, j) = zvf.xv(fvf.num_face(i), j);
93 la_table.valeurs(tab_ch, pos, temps, tab);
94}
