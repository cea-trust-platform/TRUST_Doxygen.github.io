83template <class _TYPE_>
84DoubleTab& T_It_Sou_VDF_Face<_TYPE_>::ajouter(DoubleTab& resu) const
85{
86 ((_TYPE_&) (evaluateur_source_face)).mettre_a_jour();
87
88 assert(resu.nb_dim() < 3);
89 const int ncomp = resu.line_size();
90
91 DoubleVect& bilan = so_base->bilan();
92 bilan=0;
93 const int nb_faces_tot = la_zone.valeur().nb_faces_tot();
94 coef.resize(nb_faces_tot,Array_base::NOCOPY_NOINIT);
95 coef=1;
96 if (equation_divisee_par_rho())
97 {
98 const Milieu_base& milieu = la_zcl->equation().milieu();
99 const Champ_base& rho = milieu.masse_volumique().valeur();
100 if (sub_type(Champ_Uniforme,rho)) coef = rho(0,0);
101 else
102 {
103 const DoubleTab& val_rho = rho.valeurs();
104 const IntTab& face_vois = la_zone.valeur().face_voisins();
105 const DoubleVect& volumes = la_zone.valeur().volumes();
106 coef = 0.;
107 for (int fac = 0; fac < nb_faces_tot; fac++)
108 {
109 const int elem1 = face_vois(fac,0), elem2 = face_vois(fac,1);
110 double vol = 0.;
111 if (elem1 != -1)
112 {
113 coef(fac) += val_rho(elem1)*volumes(elem1);
114 vol += volumes(elem1);
115 }
116 if (elem2 != -1)
117 {
118 coef(fac) += val_rho(elem2)*volumes(elem2);
119 vol += volumes(elem2);
120 }
121 coef(fac) /= vol;
122 }
123 }
124 }
125
126 if (ncomp == 1)
127 {
128 ajouter_faces_bords<SingleDouble>(ncomp,resu);
129 ajouter_faces_internes<SingleDouble>(ncomp,resu);
130 }
131 else
132 {
133 ajouter_faces_bords<ArrOfDouble>(ncomp,resu);
134 ajouter_faces_internes<ArrOfDouble>(ncomp,resu);
135 }
136 return resu;
137}

139template <class _TYPE_> template <typename Type_Double>
140DoubleTab& T_It_Sou_VDF_Face<_TYPE_>::ajouter_faces_bords(const int ncomp, DoubleTab& resu) const
141{
142 Type_Double source(ncomp);
143 DoubleVect& bilan = so_base->bilan();
144 for (int num_cl = 0; num_cl < la_zone->nb_front_Cl(); num_cl++)
145 {
146 const Cond_lim& la_cl = la_zcl->les_conditions_limites(num_cl);
147 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
148 const int ndeb = le_bord.num_premiere_face(), nfin = ndeb + le_bord.nb_faces();
149 if ( (sub_type(Dirichlet,la_cl.valeur())) || (sub_type(Dirichlet_homogene,la_cl.valeur())) ) { /* Do nothing */ }
150 else
151 for (int num_face = ndeb; num_face < nfin; num_face++)
152 {
153 evaluateur_source_face.calculer_terme_source_bord(num_face,source);
154 for (int k = 0; k < ncomp; k++)
155 {
156 resu(num_face,k) += source[k];
157 double contribution = (faces_doubles(num_face)==1) ? 0.5 : 1 ;
158 bilan(k) += contribution * coef(num_face) * source[k];
159 }
160 }
161 }
162 return resu;
163}

173 for (int k=0; k<ncomp; k++)

175 resu(num_face,k) += source[k];
176 double contribution = (faces_doubles(num_face)==1) ? 0.5 : 1 ;
177 bilan(k) += contribution * coef(num_face) * source[k];
178 }
179 }
