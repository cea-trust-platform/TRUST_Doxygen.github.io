63//! Lecture d'une condition aux limite de type Echange_impose_base<br>a partir d'un flot d'entree.<br>On doit lire le coefficient d'echange global H_imp et la temperature<br>a la frontiere T_ext.<br>Format:<br>"h_imp" type_champ_front bloc de lecture champ<br>"T_ext" type_champ_front bloc de lecture champ
64/** \return Entree\&<br>
65Signification: le flot d'entree modifie<br>
66Contraintes: */
67Entree& Echange_impose_base::readOn(Entree& s )
68{
69 Motcle motlu;
70 Motcles les_motcles(2);
71 {
72 les_motcles[0] = "h_imp";
73 les_motcles[1] = "T_ext";
74 }
75
76 int ind = 0;
77 while (ind < 2)
78 {
79 s >> motlu;
80 int rang = les_motcles.search(motlu);
81
82 switch(rang)
83 {
84 case 0:
85 {
86 s >> h_imp_;
87 break;
88 }
89 case 1:
90 {
91 s >> T_ext();
92 break;
93 }
94 default:
95 {
96 Cerr << "Erreur a la lecture de la condition aux limites de type Echange_impose " << finl;
97 Cerr << "On attendait " << les_motcles << "a la place de " << motlu << finl;
98 exit();
99 }
100 }
101
102 ind++;
103
104 }
105
106 return s ;
107}
