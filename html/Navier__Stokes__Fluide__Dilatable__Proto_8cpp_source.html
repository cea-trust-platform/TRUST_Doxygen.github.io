34// Multiply density by velocity and return density*velocity (mass flux)
35DoubleTab& Navier_Stokes_Fluide_Dilatable_Proto::rho_vitesse_impl(const DoubleTab& tab_rho,const DoubleTab& vit,DoubleTab& rhovitesse) const
36{
37 const int n = vit.dimension(0), ncomp = vit.line_size();
38
39 for (int i=0 ; i<n ; i++)
40 for (int j=0 ; j<ncomp ; j++) rhovitesse(i,j) = tab_rho(i,0)*vit(i,j);
41
42 rhovitesse.echange_espace_virtuel();
43 Debog::verifier("Navier_Stokes_Fluide_Dilatable_Proto::rho_vitesse : ", rhovitesse);
44 return rhovitesse;
45}

47int Navier_Stokes_Fluide_Dilatable_Proto::impr_impl(const Navier_Stokes_std& eqn,Sortie& os) const
48{
49 const Fluide_Dilatable_base& fluide_dil = ref_cast(Fluide_Dilatable_base,eqn.fluide());
50 const DoubleTab& vit = eqn.vitesse().valeurs(), rho = fluide_dil.rho_face_np1();
51 DoubleTab mass_flux(vit);
52 rho_vitesse_impl(rho,vit,mass_flux);
53
54 DoubleTab array;
55 array.copy(eqn.div().valeurs(), Array_base::NOCOPY_NOINIT); //!< init structure uniquement
56 if (tab_W.get_md_vector().non_nul())
57 {
58 operator_egal(array, tab_W ); //, VECT_REAL_ITEMS); //!< initialise
59 array*=-1;
60 }
61 else
62 {
63// remarque B.M.: certaines implementations de cette methode ne font pas echange espace virtuel:
64 fluide_dil.secmembre_divU_Z(array);
65 array*=-1;
66 }
67
68 array.echange_espace_virtuel();
69 eqn.operateur_divergence().ajouter(mass_flux, array);
70 double LocalMassFlowRateError = mp_max_abs_vect(array); //!< max|sum(rho*u*ndS)|
71
72 os << "-------------------------------------------------------------------"<< finl;
73 os << "Cell balance mass flow rate control for the problem " << eqn.probleme().le_nom() << " : " << finl;
74 os << "Absolute value : " << LocalMassFlowRateError << " kg/s" << finl; ;
75
76// On divise array par vol(i)
77 eqn.operateur_divergence().volumique(array);
78
79// On divise par un rho moyen
80 double rho_moyen = mp_moyenne_vect(rho), dt = eqn.probleme().schema_temps().pas_de_temps();
81 double bilan_massique_relatif = mp_max_abs_vect(array) * dt / rho_moyen;
82 os << "Relative value : " << bilan_massique_relatif << finl; //!< =max|LocalMassFlowRateError/(rho_moyen*Vol(i)/dt)|
83
84// Calculation as OpenFOAM: http://foam.sourceforge.net/docs/cpp/a04190_source.html
85// It is relative errors (normalized by the volume/dt)
86 double TotalMass = rho_moyen * eqn.probleme().domaine().zone(0).volume_total();
87 double local = LocalMassFlowRateError / ( TotalMass / dt ), global = mp_somme_vect(array) / ( TotalMass / dt );
88 cumulative_ += global;
89
90 os << "time step continuity errors : sum local = " << local << ", global = " << global << ", cumulative = " << cumulative_ << finl;
91
92 if (local > 0.01)
93 {
94 Cerr << "The mass balance is too bad (relative value > 1%)." << finl;
95 Cerr << "Please check and lower the convergence value of the pressure solver." << finl;
96 Process::exit();
97 }
98 return 1;
99}

114//! Calcule la derivee en temps de l'inconnue vitesse,<br>i.e. l'acceleration dU/dt et la renvoie.<br>Appelle Equation_base::derivee_en_temps_inco(DoubleTab\& )<br>Calcule egalement la pression.
115/** \return DoubleTab\&<br>
116Signification: le tableau des valeurs de l'acceleration (derivee de la vitesse)<br>
117Contraintes: */
118DoubleTab& Navier_Stokes_Fluide_Dilatable_Proto::derivee_en_temps_inco_impl(Navier_Stokes_std& eqn,DoubleTab& vpoint)
119{
120 const Fluide_Dilatable_base& fluide_dil=ref_cast(Fluide_Dilatable_base,eqn.milieu());
121 DoubleTab& press = eqn.pression().valeurs(), vit = eqn.vitesse().valeurs();
122 DoubleTab secmem(press);
123 DoubleTrav inc_pre(press);
124
125 if (!tab_W.get_md_vector().non_nul())
126 {
127 tab_W.copy(secmem, Array_base::NOCOPY_NOINIT); //!< copie la structure initialisation sinon plantage assert lors du remplissage dans EDO_Pression_th_VEF::secmembre_divU_Z_VEFP1B
128//
129 tab_W = 0.;
130 }
131
132 DoubleTab rhoU;
133 rhoU.copy(vit, Array_base::NOCOPY_NOINIT); //!< copie la structure
134
135// Get champ gradP
136 REF(Champ_base) gradient_pression;
137
138 try
139 {
140 gradient_pression = eqn.get_champ("gradient_pression");
141 }
142 catch (Champs_compris_erreur)
143 {
144 Cerr<<" l'equation ne comprend pas gradient_pression "<<finl;
145 Process::exit();
146 }
147
148 DoubleTab& gradP=gradient_pression.valeur().valeurs();
149 DoubleTrav Mmoins1grad(gradP);
150
151// We use the incremental pressure-projection algorithm (Chorin)
152
153// Step 1 : prepare operators and solve for a provisional velocity u*
154 prepare_and_solve_u_star(eqn,fluide_dil,rhoU, vpoint);
155
156// Step 2 : solve the poisson equation for the pressure increment (Pi = P^n+1 - P^n)
157// Attention the matrix has a constant coefficient as the variable of NS is rhoU and not U !!!
158 solve_pressure_increment(eqn,fluide_dil,rhoU,secmem,inc_pre,vpoint);
159
160// Step 3 : compute P^n+1 & compute the correct velocity u^n+1
161 correct_and_compute_u_np1(eqn,fluide_dil,rhoU,Mmoins1grad,inc_pre,gradP,vpoint);
162
163 return vpoint;
164}

166void Navier_Stokes_Fluide_Dilatable_Proto::assembler_avec_inertie_impl(const Navier_Stokes_std& eqn, Matrice_Morse& mat_morse,
167 const DoubleTab& present, DoubleTab& secmem)
168{
169// ****** avant inertie ******
170// diffusion en div(mu grad u ) or on veut impliciter en rho * u => on divise les contributions par le rho_face associe
171// GF on ajoute apres avoir contribuer pour avoir les bons flux bords
172 DoubleTrav rhovitesse(present);
173
174// Op diff
175 eqn.operateur(0).l_op_base().contribuer_a_avec(present,mat_morse);
176 eqn.operateur(0).ajouter(secmem);
177
178 const Fluide_Dilatable_base& fluide_dil=ref_cast(Fluide_Dilatable_base,eqn.milieu());
179 const DoubleTab& tab_rho_face_np1 = fluide_dil.rho_face_np1(), tab_rho_face_n=fluide_dil.rho_face_n();
180 const int nb_compo = present.line_size();
181 const IntVect& tab1 = mat_morse.get_tab1(), tab2 = mat_morse.get_tab2();
182
183 DoubleVect& coeff = mat_morse.get_set_coeff();
184 for (int i=0; i<mat_morse.nb_lignes(); i++)
185 for (int k=tab1(i)-1; k<tab1(i+1)-1; k++)
186 {
187 int j = tab2(k)-1;
188 double rapport = tab_rho_face_np1(j/nb_compo);
189 coeff(k) /= rapport;
190 }
191
192 rho_vitesse_impl(tab_rho_face_np1,present,rhovitesse); //!< rho*U
193
194// Op conv
195 eqn.operateur(1).l_op_base().contribuer_a_avec(rhovitesse,mat_morse);
196 eqn.operateur(1).ajouter(rhovitesse,secmem);
197
198// sources
199 eqn.sources().ajouter(secmem);
200 eqn.sources().contribuer_a_avec(present,mat_morse);
201
202// on resout en rho u on stocke donc rho u dans present
203 rho_vitesse_impl(tab_rho_face_np1,present,ref_cast_non_const(DoubleTab,present));
204 mat_morse.ajouter_multvect(present,secmem);
205
206 /*
207 * contribution a la matrice de l'inertie :
208 * on divisie la diagonale par rhon+1 face
209 * on ajoute l'inertie de facon standard
210 * on remultiplie la diagonale par rhon+1
211 */
212
213// ajout de l'inertie
214 const double dt=eqn.schema_temps().pas_de_temps();
215 eqn.solv_masse().ajouter_masse(dt,mat_morse,0);
216
217 rho_vitesse_impl(tab_rho_face_n,eqn.inconnue().passe(),rhovitesse);
218 eqn.solv_masse().ajouter_masse(dt,secmem,rhovitesse,0);
219
220// blocage_cl faux si dirichlet u!=0 !!!!!! manque multiplication par rho
221 for (int op=0; op< eqn.nombre_d_operateurs(); op++) eqn.operateur(op).l_op_base().modifier_pour_Cl(mat_morse,secmem);
222
223 /*
224 * correction finale pour les dirichlets
225 * on ne doit pas imposer un+1 mais rho_un+1 => on multiplie dons le resu par rho_face_np1
226 */
227 const Conds_lim& lescl=eqn.zone_Cl_dis().les_conditions_limites();
228 int nbcondlim=lescl.size();
229
230 for (int icl=0; icl<nbcondlim; icl++)
231 {
232 const Cond_lim_base& la_cl_base = lescl[icl].valeur();
233 if (sub_type(Dirichlet,la_cl_base))
234 {
235
236 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl_base.frontiere_dis());
237 int ndeb = la_front_dis.num_premiere_face();
238 int nfin = ndeb + la_front_dis.nb_faces();
239 for (int num_face=ndeb; num_face<nfin; num_face++)
240 {
241 if (present.line_size()==1) secmem(num_face)*=tab_rho_face_np1(num_face);
242 else
243 for (int dir=0; dir<Objet_U::dimension; dir++) secmem(num_face,dir)*=tab_rho_face_np1(num_face);
244 }
245 }
246 }
247}

250void Navier_Stokes_Fluide_Dilatable_Proto::assembler_blocs_avec_inertie(const Navier_Stokes_std& eqn, matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl)
251{
252 statistiques().begin_count(assemblage_sys_counter_);
253 const std::string& nom_inco = eqn.inconnue().le_nom().getString();
254 Matrice_Morse *mat = matrices.count(nom_inco)?matrices.at(nom_inco):NULL;
255 const DoubleTab& present = eqn.inconnue()->valeurs();
256
257// ****** avant inertie ******
258// diffusion en div(mu grad u ) or on veut impliciter en rho * u => on divise les contributions par le rho_face associe
259// GF on ajoute apres avoir contribuer pour avoir les bons flux bords
260 DoubleTrav rhovitesse(present);
261
262// Op diff
263 eqn.operateur(0).l_op_base().ajouter_blocs(matrices, secmem, semi_impl);
264
265 const Fluide_Dilatable_base& fluide_dil=ref_cast(Fluide_Dilatable_base,eqn.milieu());
266 const DoubleTab& tab_rho_face_np1 = fluide_dil.rho_face_np1(), tab_rho_face_n=fluide_dil.rho_face_n();
267 const int nb_compo = present.line_size();
268 const IntVect& tab1 = mat->get_tab1(), tab2 = mat->get_tab2();
269
270 DoubleVect& coeff = mat->get_set_coeff();
271 for (int i=0; i<mat->nb_lignes(); i++)
272 for (int k=tab1(i)-1; k<tab1(i+1)-1; k++)
273 {
274 int j = tab2(k)-1;
275 double rapport = tab_rho_face_np1(j/nb_compo);
276 coeff(k) /= rapport;
277 }
278
279 rho_vitesse_impl(tab_rho_face_np1,present,rhovitesse); //!< rho*U
280
281// Op conv
282 eqn.operateur(1).l_op_base().ajouter_blocs(matrices, secmem, {{nom_inco,rhovitesse}});
283 statistiques().end_count(assemblage_sys_counter_);
284
285// sources
286 statistiques().begin_count(source_counter_);
287 for (int i = 0; i < eqn.sources().size(); i++)
288 eqn.sources()(i).valeur().ajouter_blocs(matrices, secmem, semi_impl);
289 statistiques().end_count(source_counter_);
290
291 statistiques().begin_count(assemblage_sys_counter_);
292// on resout en rho u on stocke donc rho u dans present
293 rho_vitesse_impl(tab_rho_face_np1,present,ref_cast_non_const(DoubleTab,present));
294 mat->ajouter_multvect(present,secmem);
295
296 DoubleTab tmp(secmem);
297 tmp = 0.0;
298 eqn.operateur_gradient().valeur().ajouter_blocs(matrices, tmp, semi_impl);
299 secmem-=tmp;
300
301 /*
302 * contribution a la matrice de l'inertie :
303 * on divisie la diagonale par rhon+1 face
304 * on ajoute l'inertie de facon standard
305 * on remultiplie la diagonale par rhon+1
306 */
307
308// ajout de l'inertie
309 const double dt=eqn.schema_temps().pas_de_temps();
310 eqn.solv_masse().ajouter_masse(dt,*mat,0);
311 rho_vitesse_impl(tab_rho_face_n,eqn.inconnue().passe(),rhovitesse);
312 eqn.solv_masse().ajouter_masse(dt,secmem,rhovitesse,0);
313
314// blocage_cl faux si dirichlet u!=0 !!!!!! manque multiplication par rho
315 for (int op=0; op< eqn.nombre_d_operateurs(); op++) eqn.operateur(op).l_op_base().modifier_pour_Cl(*mat,secmem);
316
317 /*
318 * correction finale pour les dirichlets
319 * on ne doit pas imposer un+1 mais rho_un+1 => on multiplie dons le resu par rho_face_np1
320 */
321 const Conds_lim& lescl=eqn.zone_Cl_dis().les_conditions_limites();
322 int nbcondlim=lescl.size();
323
324 for (int icl=0; icl<nbcondlim; icl++)
325 {
326 const Cond_lim_base& la_cl_base = lescl[icl].valeur();
327 if (sub_type(Dirichlet,la_cl_base))
328 {
329
330 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl_base.frontiere_dis());
331 int ndeb = la_front_dis.num_premiere_face();
332 int nfin = ndeb + la_front_dis.nb_faces();
333 for (int num_face=ndeb; num_face<nfin; num_face++)
334 {
335 if (present.line_size()==1) secmem(num_face)*=tab_rho_face_np1(num_face);
336 else
337 for (int dir=0; dir<Objet_U::dimension; dir++) secmem(num_face,dir)*=tab_rho_face_np1(num_face);
338 }
339 }
340 }
341 secmem.echange_espace_virtuel();
342 statistiques().end_count(assemblage_sys_counter_);
343
344}

358void Navier_Stokes_Fluide_Dilatable_Proto::prepare_and_solve_u_star(Navier_Stokes_std& eqn,
359 const Fluide_Dilatable_base& fluide_dil,
360 DoubleTab& rhoU, DoubleTab& vpoint)
361{
362 const DoubleTab& tab_rho_face_n =fluide_dil.rho_face_n(), tab_rho_face_np1=fluide_dil.rho_face_np1();
363 const DoubleTab& tab_rho = fluide_dil.rho_discvit(); //!< rho avec la meme discretisation que la vitesse
364 const DoubleTab& vit = eqn.vitesse().valeurs();
365
366 fluide_dil.secmembre_divU_Z(tab_W); //Calcule W=-dZ/dt, 2nd membre de l'equation div(rhoU) = W
367 vpoint=0;
368
369// ajout diffusion (avec la viscosite dynamique)
370 if (!eqn.schema_temps().diffusion_implicite()) eqn.operateur(0).ajouter(vpoint);
371
372 DoubleTab& rhovitesse = ref_cast_non_const(DoubleTab,eqn.rho_la_vitesse().valeurs());
373 rho_vitesse_impl(tab_rho,vit,rhovitesse);
374
375// ajout convection utilise rhovitesse
376 if (!eqn.schema_temps().diffusion_implicite()) eqn.operateur(1).ajouter(rhovitesse,vpoint);
377 else
378 {
379 DoubleTrav trav(vpoint);
380 eqn.derivee_en_temps_conv(trav,rhovitesse);
381 vpoint = trav;
382 }
383
384// ajout source
385 eqn.sources().ajouter(vpoint);
386
387// ajout de gradP
388 eqn.corriger_derivee_expl(vpoint);
389
390 const Champ_base& rho_vit=eqn.get_champ("rho_comme_v");
391 ref_cast_non_const(DoubleTab,rho_vit.valeurs())=tab_rho_face_np1;
392
393 if (eqn.schema_temps().diffusion_implicite())
394 {
395 DoubleTrav secmemV(vpoint);
396 secmemV = vpoint;
397 double dt = eqn.schema_temps().pas_de_temps();
398 /*
399 * secmemV contient M(rhonp1 unp1 -rhon un)/dt
400 * M-1 secmemv*dt+rhon un -rhonp1un= rhonp1 (unp1 -unp )
401 * dt/rhnhonp1 =(unp1-un)/dt
402 * M-1 secmemV/rhonp1 + (rhon -rhonp1)/rhonp1/dt un
403 *
404 * on modifie le solveur masse pour diviser par rhonp1
405 * (pratique aussi pour la diffusion implicite)
406 */
407
408 eqn.solv_masse()->set_name_of_coefficient_temporel("rho_comme_v");
409 eqn.solv_masse().appliquer(secmemV);
410 DoubleTrav dr(tab_rho_face_n);
411 for (int i=0; i<dr.size_totale(); i++) dr(i)=(tab_rho_face_n(i)/tab_rho_face_np1(i)-1.)/dt;
412
413// on sert de vpoint pour calculer
414 rho_vitesse_impl(dr,vit,vpoint);
415 secmemV += vpoint;
416
417 DoubleTab delta_u(eqn.inconnue().futur());
418 eqn.Gradient_conjugue_diff_impl(secmemV, delta_u ) ;
419
420 /*
421 * delta_u=unp1 -un => delta_u + un=unp1
422 * (delat_u + un)*rhonp1 = rhonp1 * unp1
423 * (delat_u + un)*rhonp1 - rhon * un= rhonp1 * unp1 - rhon * un
424 */
425
426 delta_u *= dt;
427 delta_u += vit;
428 rho_vitesse_impl(tab_rho_face_np1,delta_u,vpoint);
429 vpoint -= rhovitesse;
430 vpoint /= dt;
431 eqn.solv_masse()->set_name_of_coefficient_temporel("no_coeff");
432 }
433 else eqn.solv_masse().appliquer(vpoint);
434
435 update_vpoint_on_boundaries(eqn,fluide_dil,vpoint);
436
437} /* END prepare_and_solve_u_star */

439void Navier_Stokes_Fluide_Dilatable_Proto::update_vpoint_on_boundaries(const Navier_Stokes_std& eqn,
440 const Fluide_Dilatable_base& fluide_dil,
441 DoubleTab& vpoint)
442{
443// on ajoute durho/dt au bord Dirichlet car les solveur masse a mis a zero
444// NOTE : en incompressible le terme est rajoute par modifier_secmem
445 const double dt_ = eqn.schema_temps().pas_de_temps();
446 const DoubleTab& tab_rho_face_n =fluide_dil.rho_face_n(), tab_rho_face_np1=fluide_dil.rho_face_np1();
447 const DoubleTab& vit = eqn.vitesse().valeurs();
448 const Conds_lim& lescl=eqn.zone_Cl_dis().les_conditions_limites();
449 const IntTab& face_voisins = eqn.zone_dis().valeur().face_voisins();
450 const int nbcondlim = lescl.size(), taille = vpoint.line_size();
451
452 if (taille==1)
453 if (orientation_VDF_.size() == 0)
454 orientation_VDF_.ref(ref_cast(Zone_VF,eqn.zone_dis().valeur()).orientation());
455
456 for (int icl=0; icl<nbcondlim; icl++)
457 {
458 const Cond_lim_base& la_cl_base = lescl[icl].valeur();
459 if (sub_type(Dirichlet,la_cl_base))
460 {
461 const Front_VF& la_front_dis = ref_cast(Front_VF,la_cl_base.frontiere_dis());
462 const Dirichlet& diri=ref_cast(Dirichlet,la_cl_base);
463 const int ndeb = la_front_dis.num_premiere_face(), nfin = ndeb + la_front_dis.nb_faces();
464
465 if (taille==1) //!< VDF //
466 {
467 for (int num_face=ndeb; num_face<nfin; num_face++)
468 {
469 int n0 = face_voisins(num_face, 0);
470 if (n0 == -1) n0 = face_voisins(num_face, 1);
471
472// GF en cas de diffsion implicite vpoint!=0 on ignrore l'ancienne valeur
473 vpoint(num_face)=(diri.val_imp(num_face-ndeb,orientation_VDF_(num_face))*tab_rho_face_np1(num_face)-
474 vit(num_face)*tab_rho_face_n(num_face))/dt_;
475 }
476 }
477 else //!< VEF //
478 {
479 for (int num_face=ndeb; num_face<nfin; num_face++)
480 for (int jj=0; jj<Objet_U::dimension; jj++)
481 {
482// GF en cas de diffsion implicite vpoint!=0 on ignrore l'ancienne valeur
483 vpoint(num_face,jj)=(tab_rho_face_np1(num_face)*diri.val_imp(num_face-ndeb,jj)
484 -tab_rho_face_n(num_face)*vit(num_face,jj))/dt_;
485 }
486 }
487 }
488 }

490} /* END update_vpoint_on_boundaries */

492void Navier_Stokes_Fluide_Dilatable_Proto::solve_pressure_increment(Navier_Stokes_std& eqn,
493 const Fluide_Dilatable_base& fluide_dil,
494 DoubleTab& rhoU, DoubleTab& secmem,
495 DoubleTab& inc_pre, DoubleTab& vpoint)
496{
497 const DoubleTab& tab_rho_face_n =fluide_dil.rho_face_n(), tab_rho_face_np1=fluide_dil.rho_face_np1();
498 const DoubleTab& vit = eqn.vitesse().valeurs();

500
501// Resolution pression
502 vpoint.echange_espace_virtuel();
503
504// Compute rhoU(n) :
505 rho_vitesse_impl(tab_rho_face_n,vit,rhoU);
506
507// Add source term to vpoint if interfaces
508 Probleme_base& prob=eqn.probleme();
509 DoubleTrav vpoint0(vpoint) ;
510 vpoint0 = vpoint ;

534// Correction du second membre d'apres les conditions aux limites :
535 eqn.assembleur_pression().modifier_secmem(secmem);
536 eqn.solveur_pression().resoudre_systeme(eqn.matrice_pression().valeur(),secmem,inc_pre);
537
538} /* END solve_pressure_increment */

540void Navier_Stokes_Fluide_Dilatable_Proto::correct_and_compute_u_np1(Navier_Stokes_std& eqn,
541 const Fluide_Dilatable_base& fluide_dil,
542 DoubleTab& rhoU,DoubleTab& Mmoins1grad,
543 DoubleTab& inc_pre,DoubleTab& gradP,
544 DoubleTab& vpoint)
545{
546 const DoubleTab& tab_rho_face_np1=fluide_dil.rho_face_np1();
547 const DoubleTab& vit = eqn.vitesse().valeurs();
548 DoubleTab& press = eqn.pression().valeurs();
549 const double dt_ = eqn.schema_temps().pas_de_temps();
550
551// On a besoin de l'espace virtuel de la pression pour calculer le gradient plus bas
552// et modifier_solution ne fait pas toujours l'echange_espace_virtuel.
553// On suppose que pression et inc_pre ont leur espace virtuel a jour
554// On fait pression += inc_pre:
555 operator_add(press, inc_pre, VECT_ALL_ITEMS);
556 eqn.assembleur_pression().modifier_solution(press);
557
558// Correction de la vitesse en pression : M-1 Bt P
559 eqn.solv_masse().appliquer(gradP);
560 vpoint += gradP; //!< M-1 F
561
562 press.echange_espace_virtuel();
563 eqn.operateur_gradient().calculer(press, gradP);
564
565// On conserve Bt P pour la prochaine fois.
566 Mmoins1grad = gradP;
567 eqn.solv_masse().appliquer(Mmoins1grad);
568
569// Correction en pression
570 vpoint -= Mmoins1grad;
571
572// vpoint = (rhoU(n+1)-rhoU(n))/dt
573 vpoint *= dt_;
574 vpoint += rhoU; //!< rhoU(n+1)
575
576// Compute U(n+1):
577 const int n = vpoint.dimension(0), ncomp = vpoint.line_size();
578 for (int i=0 ; i<n ; i++)
579 for (int j=0 ; j< ncomp; j++) vpoint(i,j) /= tab_rho_face_np1(i);
580
581// Compute (U(n+1)-U(n))/dt :
582 vpoint -= vit;
583 vpoint /= dt_;
584
585 vpoint.echange_espace_virtuel();
586 Debog::verifier("Navier_Stokes_Fluide_Dilatable_base::derivee_en_temps_inco, vpoint : ", vpoint);
587
588} /* END correct_and_compute_u_np1 */
