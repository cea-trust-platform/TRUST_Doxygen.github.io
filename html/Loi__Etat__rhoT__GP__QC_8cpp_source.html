32Entree& Loi_Etat_rhoT_GP_QC::readOn( Entree& is )
33{
34 Nom expression_;
35
36 Param param(que_suis_je());
37 param.ajouter("Cp",&Cp_,Param::REQUIRED);//!< XD_ADD_P double Specific heat at constant pressure of the gas Cp.
38 param.ajouter("Prandtl",&Pr_); //!< XD_ADD_P double Prandtl number of the gas Pr=mu*Cp/lambda
39 param.ajouter("rho_xyz",&rho_xyz_); //!< XD_ADD_P field_base Defined with a Champ_Fonc_xyz to define a constant rho with time (space dependent)
40 param.ajouter("rho_t",&expression_); //!< XD_ADD_P chaine Expression of T used to calculate rho. This can lead to a variable rho, both in space and in time.
41 param.lire_avec_accolades(is);
42
43 if (expression_ == "??" && !rho_xyz_.non_nul())
44 {
45 Cerr << "Error in Loi_Etat_rhoT_GP_QC::readOn !" << finl;
46 Cerr << "The closure equation of rho is not read in your data file !" << finl;
47 Cerr << "Either use rho_t followed by an expression of T," << finl;
48 Cerr << "or use rho_xyz followed by a Champ_Fonc_xyz !" << finl;
49 Process::exit();
50 }
51
52 if (expression_ != "??")
53 {
54 is_exp_ = true;
55 Cerr << "Parsing the expression " << expression_ << finl;
56 parser_.setNbVar(1);
57 parser_.setString(expression_);
58 parser_.addVar("T");
59 parser_.parseString();
60 }
61 else
62 assert(rho_xyz_.valeur().que_suis_je() == "Champ_Fonc_xyz" );
63
64 return is;
65}

67// Method used to initialize rho from a Champ_Fonc_xyz
68// In this case the DoubleTab rho_ is initialized and remains constant with time
69void Loi_Etat_rhoT_GP_QC::initialiser_rho()
70{
71 int isVDF = 0;
72 if (le_fluide->masse_volumique().valeur().que_suis_je()=="Champ_Fonc_P0_VDF") isVDF = 1;
73// We know that mu is always stored on elems
74 int nb_elems = le_fluide->viscosite_dynamique().valeurs().size();
75// The Champ_Don rho_xyz_ is evaluated on elements
76 assert (rho_xyz_.valeur().valeurs().size() == nb_elems);
77
78 if (isVDF)
79 {
80// Disc VDF => rho_ & rho_xyz_ on elems => we do nothing
81 rho_.resize(nb_elems, 1);
82 DoubleTab& fld = rho_xyz_.valeur().valeurs();
83 for (int i=0; i<nb_elems; i++) rho_(i,0)=fld(i,0);
84 }
85 else
86 {
87// Disc VEF => rho on faces ...
88 int nb_faces = le_fluide->masse_volumique().valeurs().size(); //!< rho on faces in VEF
89 rho_.resize(nb_faces, 1);
90
91 Champ_base& ch_rho = le_fluide->masse_volumique().valeur();
92 ch_rho.affecter_(rho_xyz_);
93 DoubleTab& fld = ch_rho.valeurs();
94 for (int i = 0; i < nb_faces; i++) rho_(i,0)= fld(i,0);
95 }
96}
