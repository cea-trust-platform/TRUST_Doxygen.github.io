218void Process::exit(const Nom& message ,int i)
219{
220 if(je_suis_maitre())
221 {
222 Cerr << message << finl;
223 Cerr.flush();
224 SFichier hier("hierarchie.dump");
225 hier << "\n KEYWORDS\n";
226 Type_info::hierarchie(hier);
227 hier << "\n SYNONYMS\n";
228 Synonyme_info::hierarchie(hier);
229 if (!get_disable_stop() && Process::je_suis_maitre())
230 {
231 Nom nomfic( Objet_U::nom_du_cas() );
232 nomfic += ".stop";
233 {
234 SFichier ficstop( nomfic );
235 ficstop <<message<<finl;
236 }
237 }
238 }
239 Journal() << message << finl;
240 if( journal_shared_ && journal_file_open_)
241 end_journal(verbose_level_);
242
243
244 if (exception_sur_exit)
245 {
246// Lancement d'une exception (utilise par Execute_parallel)
247 throw TriouError("Error in trust ",Process::me());
248 }
249 else
250 {
251 int abort=0;
252#ifdef MPI_
253#ifdef INT_is_64_
254#define MPI_ENTIER MPI_LONG
255#else
256#define MPI_ENTIER MPI_INT
257#endif
258 if (Process::nproc()>1)
259 {
260 const MPI_Comm& mpi_comm=ref_cast(Comm_Group_MPI,PE_Groups::groupe_TRUST()).get_mpi_comm();
261 int tag = 666;
262 int buffer[1]= {1};
263 MPI_Request request;
264
265// Envoi non bloquant vers me()+1
266 int to_pe = (me()==nproc()-1?0:me()+1);
267 MPI_Isend(buffer, 1, MPI_ENTIER, to_pe, tag, mpi_comm, &request);
268
269// Reception non bloquante depuis me()-1
270 int from_pe = (me()==0?nproc()-1:me()-1);
271 MPI_Irecv(buffer, 1, MPI_ENTIER, from_pe, tag, mpi_comm, &request);
272
273// Attente
274 sleep(1);
275
276// Test si me() a recu de me()-1
277 True_int ok;
278 MPI_Status status;
279 MPI_Test(&request,&ok,&status);
280 if (!ok)
281 abort=1;
282 }
283#endif
284 if (abort)
285 {
286 if (!je_suis_maitre())
287 {
288 std_err_ << "!!! TRUST process number " << Process::me() << " exited unexpectedly ! See error message at the end of the file " << journal_file_name_ << " -> Aborting calculation..." << finl;
289 }
290 PE_Groups::groupe_TRUST().abort();
291 }
292 else
293 {
294#ifdef MPI_
295 MPI_Finalize();
296#endif
297 PE_Groups::finalize();
298 }
299 }
300// On force exit();
301 if (i==0) i=-1;
302 ::exit(i); //Seul ::exit utilise dans le code
303}

394//! Initialisation du journal<br>Parametre: verbose_level<br>Signification: les messages de niveau <= verbose_level seront affiches,<br>les autres seront mis a la poubelle.<br>Parametre: file_name<br>Signification: si pointeur nul, tout le monde ecrit dans cerr, sinon<br>c'est le nom du fichier (doit etre different sur chaque processeur)<br>Parametre: append<br>Signification: indique si on ouvre le fichier en mode append ou pas.
395void init_journal_file(int verbose_level, int journal_shared, const char * file_name, int append)
396{
397 journal_shared_ = journal_shared;
398 if( journal_shared_ )
399 {
400 if(journal_file_open_)
401 end_journal(verbose_level);
402 }
403 else
404 end_journal(verbose_level);
405
406 if (verbose_level > 0)
407 {
408 if (file_name)
409 {
410 if(!journal_shared_)
411 {
412 IOS_OPEN_MODE mode = ios::out;
413 if (append)
414 mode = ios::app;
415 if (!journal_file_.ouvrir(file_name, mode))
416 {
417 Cerr << "Fatal error in init_journal_file: cannot open journal file" << finl;
418 Process::exit();
419 }
420 }
421 else
422 {
423 if(append)
424 Cerr << "Process.cpp::init_journal_file : append mode is not possible with HDF5!\n"
425 << "If " << file_name << " already exists, it will be overwritten" << finl;
426 }
427 journal_file_open_ = 1;
428 journal_file_name_ = file_name;
429 }
430 }
431 verbose_level_ = verbose_level;
432}
