57//! Lit les valeurs du champ tabule en temps a partir<br>d'un flot d'entree.<br>On lit le nombre de composante du champ (nb_comp) et on<br>remplit une table a partir des nval couples (vrel_i, tps_i).<br>Format:<br>Champ_Tabule_Temps nb_comp { nval tps_1...tps_nval vrel_1...vrel_nval }
58/** \return Entree\&<br>
59Signification: le flot d'entree modifie<br>
60Contraintes: */
61Entree& Champ_Tabule_Temps::readOn(Entree& is)
62{
63 Motcle motlu;
64 int nb_val;
65 Motcle accolade_ouverte("{");
66 Motcle accolade_fermee("}");
67 nb_compo_=lire_dimension(is,que_suis_je());
68 if (nb_compo_ <= 0)
69 {
70 Cerr << "Error in Champ_Tabule_Temps::readOn : nb_compo <= 0" << finl;
71 assert(0);
72 exit();
73 }
74 dimensionner(1, nb_compo_);
75 is >> motlu;
76 if (motlu == accolade_ouverte)
77 {
78 nb_val=lire_dimension(is,que_suis_je());
79 DoubleVect param(nb_val);
80 DoubleTab les_valeurs(nb_val, nb_compo_);
81 for (int i = 0; i < nb_val; i++)
82 is >> param[i];
83 for (int i = 0; i < nb_val; i++)
84 for (int k = 0; k < nb_compo_; k++)
85 is >> les_valeurs(i, k);
86 la_table.remplir(param, les_valeurs);
87
88 is >> motlu;
89 if (motlu != accolade_fermee)
90 {
91 Cerr << "Error reading from an object of type Champ_Tabule_Temps" << finl;
92 Cerr << "We expected keyword } instead of " << motlu << finl;
93 exit();
94 }
95 }
96 else
97 {
98 Cerr << "Error reading from an object of type Champ_Tabule_Temps" << finl;
99 Cerr << "We expected keyword { instead of " << motlu << finl;
100 exit();
101 }
102 changer_temps(-1e9);
103// mettre_a_jour(0.);
104 return is;
105}
