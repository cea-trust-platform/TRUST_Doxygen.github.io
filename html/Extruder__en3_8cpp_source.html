89//! Fonction principale de l'interprete Extruder_en3<br>Triangule 1 a 1 toutes les zones du domaine<br>specifie par la directive.<br>On triangule la zone grace a la methode:<br>void Extruder_en3::extruder(Zone\& zone) const<br>Extruder_en3 signifie ici transformer en triangle des<br>elements geometrique d'une zone.
90/** \return Entree\&<br>
91Signification: le flot d'entree<br>
92Contraintes: */
93Entree& Extruder_en3::interpreter_(Entree& is)
94{
95 if (Objet_U::dimension!=2)
96 {
97 Cerr << " We only can extruder_en3 the 2D meshes ! " << finl;
98 exit();
99 }
100
101 int nb_dom=0;
102 Noms nom_dom;
103 Cerr << "The format has changed after the 1.5.3. A list of domains must now be readen as follows :" << finl;
104 Cerr << "domaine N dom1 dom2 domN" << finl;
105 Param param(que_suis_je());
106 param.ajouter("domaine",&nom_dom,Param::REQUIRED);
107 param.ajouter("nb_tranches",&NZ_,Param::REQUIRED);
108 param.ajouter_arr_size_predefinie("direction",&direction_,Param::REQUIRED);
109 param.ajouter("nom_cl_devant",&nom_dvt_);
110 param.ajouter("nom_cl_derriere",&nom_derriere_);
111 param.lire_avec_accolades_depuis(is);
112 nb_dom=nom_dom.size();
113
114// creation du tableau de correspondance des indices
115// pour assurer le decoupage conforme entre domaines
116//
117
118// PQ : 07/09/08 : le decoupage des prismes apres extrusion est base sur une numerotation
119// globale des indices qui permet d'assurer la conformite des maillages. Ces indices references
120// dans le tableau nums sont issus du regroupement des domaines en un seul suivant la methode
121// se trouvant dans Mailler.cpp.
122// On procede par concatenation des domaines dans "dom_tot" de maniere a ne pas interferer
123// avec les domaines et les zones d'origines.
124
125 Domaine dom_tot;
126 for(int i=0; i<nb_dom; i++)
127 {
128// on parcourt les domaines a l'envers pour retrouver l'ancien comportement
129 associer_domaine(nom_dom[nb_dom-1-i]);
130 Domaine& domi=domaine(i);
131 Zone& zone_tot=dom_tot.add(domi.zone(0));
132 zone_tot.associer_domaine(dom_tot);
133 int nb_som=domi.coord_sommets().dimension(0);
134 IntVect num(nb_som);
135 dom_tot.ajouter(domi.coord_sommets(), num);
136 zone_tot.renum(num);
137 zone_tot.associer_domaine(dom_tot);
138 Scatter::uninit_sequential_domain(dom_tot);
139 dom_tot.comprimer();
140 dom_tot.zone(0).fixer_premieres_faces_frontiere();
141 dom_tot.zone(0).type_elem().associer_zone(dom_tot.zone(0));
142
143//
144// extrusion des domaines
145//
146 Scatter::uninit_sequential_domain(domi);
147 extruder(domi,num);
148 Scatter::init_sequential_domain(domi);
149 }
150
151 return is;
152}

156void Extruder_en3::extruder(Domaine& dom, const IntVect& num)
157{
158 Zone& zone = dom.zone(0);
159 if(zone.type_elem()->que_suis_je() == "Triangle")
160 {
161 int oldnbsom = zone.nb_som();
162 IntTab& les_elems=zone.les_elems();
163 int oldsz=les_elems.dimension(0);
164 double dx = direction_[0]/NZ_;
165 double dy = direction_[1]/NZ_;
166 double dz = direction_[2]/NZ_;
167
168 Faces les_faces;
169 {
170// bloc a factoriser avec Zone_VF.cpp :
171 Type_Face type_face = zone.type_elem().type_face(0);
172 les_faces.typer(type_face);
173 les_faces.associer_zone(zone);
174
175 Static_Int_Lists connectivite_som_elem;
176 const int nb_sommets_tot = zone.domaine().nb_som_tot();
177 const IntTab& elements = zone.les_elems();
178
179 construire_connectivite_som_elem(nb_sommets_tot,
180 elements,
181 connectivite_som_elem,
182 1 /* include virtual elements */);
183
184 Faces_builder faces_builder;
185 IntTab elem_faces; //!< Tableau dont on aura pas besoin
186 faces_builder.creer_faces_reeles(zone,
187 connectivite_som_elem,
188 les_faces,
189 elem_faces);
190 }
191
192 int newnbsom = oldnbsom*(NZ_+1);
193 DoubleTab new_soms(newnbsom, 3);
194 DoubleTab& coord_sommets=dom.les_sommets();
195 Objet_U::dimension=3;
196
197// les sommets du maillage 2D sont translates en premier
198 for (int i=0; i<oldnbsom; i++)
199 {
200 double x = coord_sommets(i,0);
201 double y = coord_sommets(i,1);
202 double z=0.;
203 if (coord_sommets.dimension(1)>2)
204 z=coord_sommets(i,2);
205 for (int k=0; k<=NZ_; k++)
206 {
207 new_soms(k*oldnbsom+i,0)=x;
208 new_soms(k*oldnbsom+i,1)=y;
209 new_soms(k*oldnbsom+i,2)=z;
210
211 x += dx;
212 y += dy;
213 z += dz;
214 }
215 }
216
217 coord_sommets.resize(0);
218 dom.ajouter(new_soms);
219
220 int newnbelem = 3*NZ_*oldsz;
221 IntTab new_elems(newnbelem, 4); //!< les nouveaux elements
222 for (int i=0; i<oldsz; i++)
223 {
224 int i1=les_elems(i,0);
225 int i2=les_elems(i,1);
226 int i3=les_elems(i,2);
227
228 int ii1 = num[i1];
229 int ii2 = num[i2];
230 int ii3 = num[i3];
231 int i4=i1+oldnbsom;
232 int i5=i2+oldnbsom;
233 int i6=i3+oldnbsom;
234
235 for (int k=0; k<NZ_; k++)
236 {
237 int j=3*k*oldsz+3*i;
238 if (ii1>ii2 && ii1>ii3)
239 {
240 new_elems(j,0) = i1;
241 new_elems(j,1) = i2;
242 new_elems(j,2) = i3;
243 new_elems(j,3) = i4;
244 if (ii3>ii2)
245 {
246 new_elems(j+1,0) = i2;
247 new_elems(j+1,1) = i3;
248 new_elems(j+1,2) = i4;
249 new_elems(j+1,3) = i6;
250
251 new_elems(j+2,0) = i2;
252 new_elems(j+2,1) = i4;
253 new_elems(j+2,2) = i5;
254 new_elems(j+2,3) = i6;
255 }
256 else
257 {
258 new_elems(j+1,0) = i2;
259 new_elems(j+1,1) = i3;
260 new_elems(j+1,2) = i4;
261 new_elems(j+1,3) = i5;
262
263 new_elems(j+2,0) = i3;
264 new_elems(j+2,1) = i4;
265 new_elems(j+2,2) = i5;
266 new_elems(j+2,3) = i6;
267 }
268 }
269
270 if (ii2>ii1 && ii2>ii3)
271 {
272 new_elems(j,0) = i1;
273 new_elems(j,1) = i2;
274 new_elems(j,2) = i3;
275 new_elems(j,3) = i5;
276 if (ii1>ii3)
277 {
278 new_elems(j+1,0) = i1;
279 new_elems(j+1,1) = i3;
280 new_elems(j+1,2) = i4;
281 new_elems(j+1,3) = i5;
282
283 new_elems(j+2,0) = i3;
284 new_elems(j+2,1) = i4;
285 new_elems(j+2,2) = i5;
286 new_elems(j+2,3) = i6;
287 }
288 else
289 {
290 new_elems(j+1,0) = i1;
291 new_elems(j+1,1) = i3;
292 new_elems(j+1,2) = i5;
293 new_elems(j+1,3) = i6;
294
295 new_elems(j+2,0) = i1;
296 new_elems(j+2,1) = i4;
297 new_elems(j+2,2) = i5;
298 new_elems(j+2,3) = i6;
299 }
300 }
301
302 if (ii3>ii1 && ii3>ii2)
303 {
304 new_elems(j,0) = i1;
305 new_elems(j,1) = i2;
306 new_elems(j,2) = i3;
307 new_elems(j,3) = i6;
308 if (ii2>ii1)
309 {
310 new_elems(j+1,0) = i1;
311 new_elems(j+1,1) = i2;
312 new_elems(j+1,2) = i5;
313 new_elems(j+1,3) = i6;
314
315 new_elems(j+2,0) = i1;
316 new_elems(j+2,1) = i4;
317 new_elems(j+2,2) = i5;
318 new_elems(j+2,3) = i6;
319 }
320 else
321 {
322 new_elems(j+1,0) = i1;
323 new_elems(j+1,1) = i2;
324 new_elems(j+1,2) = i4;
325 new_elems(j+1,3) = i6;
326
327 new_elems(j+2,0) = i2;
328 new_elems(j+2,1) = i4;
329 new_elems(j+2,2) = i5;
330 new_elems(j+2,3) = i6;
331 }
332 }
333
334 mettre_a_jour_sous_zone(zone,i,j,3);
335 i1+=oldnbsom;
336 i2+=oldnbsom;
337 i3+=oldnbsom;
338 i4+=oldnbsom;
339 i5+=oldnbsom;
340 i6+=oldnbsom;
341 }
342 }
343// Reconstruction de l'octree
344 zone.invalide_octree();
345 zone.typer("Tetraedre");
346
347 les_elems.ref(new_elems);
348 construire_bords(dom, les_faces,oldnbsom, oldsz, num);
349 }
350 else
351 {
352 Cerr << "It is not known yet how to extrude "
353 << zone.type_elem()->que_suis_je() <<"s"<<finl;
354 }
355}

358//! Creation des bords du domaine extrude
359void Extruder_en3::construire_bords(Domaine& dom, Faces& les_faces, int oldnbsom, int oldsz, const IntVect& num)
360{
361 Zone& zone = dom.zone(0);
362 IntTab& les_elems=zone.les_elems();
363// Les bords:
364 {
365 LIST_CURSEUR(Bord) curseur(zone.faces_bord());
366 while(curseur)
367 {
368 Faces& les_faces_du_bord=curseur->faces();
369 construire_bord_lateral(les_faces_du_bord, les_faces, oldnbsom, num);
370 ++curseur;
371 }
372 }
373// Les raccords:
374 {
375 LIST_CURSEUR(Raccord) curseur(zone.faces_raccord());
376 while(curseur)
377 {
378 Faces& les_faces_du_bord=curseur.valeur()->faces();
379 construire_bord_lateral(les_faces_du_bord, les_faces, oldnbsom, num);
380 ++curseur;
381 }
382 }
383
384// Devant
385 if(nom_dvt_=="NULL")
386 {
387 Cerr << "We don't associate any boundary to the front of the domain " << dom.le_nom() << finl;
388 }
389 else
390 {
391 Bord& devant = zone.faces_bord().add(Bord());
392 devant.nommer(nom_dvt_);
393 Faces& les_faces_dvt=devant.faces();
394 les_faces_dvt.typer(Faces::triangle_3D);
395
396 IntTab som_dvt(oldsz, 3);
397 les_faces_dvt.voisins().resize(oldsz, 2);
398 les_faces_dvt.voisins()=-1;
399
400 for (int i=0; i<oldsz; i++)
401 {
402 int i0=les_elems(3*i,0);
403 int i1=les_elems(3*i,1);
404 int i2=les_elems(3*i,2);
405
406 som_dvt(i,0) = i0;
407 som_dvt(i,1) = i1;
408 som_dvt(i,2) = i2;
409 }
410 les_faces_dvt.les_sommets().ref(som_dvt);
411 }
412
413// Derriere
414 if(nom_derriere_=="NULL")
415 {
416 Cerr << "We don't associate any boundary to the back of the domain " << dom.le_nom() << finl;
417 }
418 else
419 {
420 Bord& derriere = zone.faces_bord().add(Bord());
421 derriere.nommer(nom_derriere_);
422 Faces& les_faces_der=derriere.faces();
423 les_faces_der.typer(Faces::triangle_3D);
424
425 IntTab som_der(oldsz, 3);
426 les_faces_der.voisins().resize(oldsz, 2);
427 les_faces_der.voisins()=-1;
428
429 for (int i=0; i<oldsz; i++)
430 {
431 int i0=les_elems(3*i,0);
432 int i1=les_elems(3*i,1);
433 int i2=les_elems(3*i,2);
434
435 som_der(i,0) = i0+oldnbsom*NZ_;
436 som_der(i,1) = i1+oldnbsom*NZ_;
437 som_der(i,2) = i2+oldnbsom*NZ_;
438 }
439 les_faces_der.les_sommets().ref(som_der);
440 }
441}

444//! Creation d'un bord lateral
445void Extruder_en3::construire_bord_lateral(Faces& les_faces_du_bord, Faces& les_faces, int oldnbsom, const IntVect& num)
446{
447// oldnbsom = number of nodes of the 2D mesh
448 int nb_faces = les_faces_du_bord.nb_faces();
449 IntTab les_sommets(2*nb_faces*NZ_,3);
450 for (int i=0; i<nb_faces; i++)
451 {
452 int i0=les_faces_du_bord.sommet(i,0);
453 int i1=les_faces_du_bord.sommet(i,1);
454 int ii0= num[i0];
455 int ii1= num[i1];
456 for (int k=0; k<NZ_; k++)
457 {
458 int j=2*nb_faces*k+2*i;
459
460 if (ii0>ii1)
461 {
462 les_sommets(j,0) = i0;
463 les_sommets(j,1) = i1;
464 les_sommets(j,2) = i0+oldnbsom;
465
466 les_sommets(j+1,0) = i1;
467 les_sommets(j+1,1) = i0+oldnbsom;
468 les_sommets(j+1,2) = i1+oldnbsom;
469 }
470 else
471 {
472 les_sommets(j,0) = i0;
473 les_sommets(j,1) = i1;
474 les_sommets(j,2) = i1+oldnbsom;
475
476 les_sommets(j+1,0) = i0;
477 les_sommets(j+1,1) = i0+oldnbsom;
478 les_sommets(j+1,2) = i1+oldnbsom;
479 }
480 i0+=oldnbsom;
481 i1+=oldnbsom;
482 }
483 }
484
485 les_faces_du_bord.typer(Faces::triangle_3D);
486 les_faces_du_bord.les_sommets().ref(les_sommets);
487 les_faces_du_bord.voisins().resize(2*nb_faces*NZ_, 2);
488 les_faces_du_bord.voisins()=-1;
489}
