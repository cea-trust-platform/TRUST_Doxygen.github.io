223//! Methode appelee par Zone_VF::discretiser apres la creation<br>des faces reelles.<br>On reordonne les faces de sorte a placer les faces "non standard"<br>au debut de la liste des faces. Les faces non standard sont celles<br>dont les volumes de controles sont modifies par les conditions aux<br>limites.
224void Zone_EF::reordonner(Faces& les_faces)
225{
226 if (Process::je_suis_maitre())
227 Cerr << "Zone_EF::reordonner les_faces " << finl;
228
229// Construction de rang_elem_non_std_ :
230// C'est un vecteur indexe par les elements du domaine.
231// size() = nb_elem()
232// size_tot() = nb_elem_tot()
233// Valeurs dans le tableau :
234// rang_elem_non_std_[i] = -1 si l'element i est standard,
235// sinon
236// rang_elem_non_std_[i] = j, ou j est l'indice de l'element dans
237// les tableaux indexes par les elements non standards (par exemple
238// le tableau Zone_Cl_EF::type_elem_Cl_).
239// Un element est non standard s'il est voisin d'une face frontiere.
240 {
241 const Domaine& dom = zone().domaine();
242 const int nb_elements = nb_elem();
243 const int nb_faces_front = zone().nb_faces_frontiere();
244 dom.creer_tableau_elements(rang_elem_non_std_);
245// rang_elem_non_std_.resize(nb_elements);
246// Scatter::creer_tableau_distribue(dom, Joint::ELEMENT, rang_elem_non_std_);
247 rang_elem_non_std_ = -1;
248 int nb_elems_non_std = 0;
249// D'abord on marque les elements non standards avec rang_elem_non_std_[i] = 0
250 for (int i_face = 0; i_face < nb_faces_front; i_face++)
251 {
252 const int elem = les_faces.voisin(i_face, 0);
253 if (rang_elem_non_std_[elem] < 0)
254 {
255 rang_elem_non_std_[elem] = 0;
256 nb_elems_non_std++;
257 }
258 }
259 nb_elem_std_ = nb_elements - nb_elems_non_std;
260 rang_elem_non_std_.echange_espace_virtuel();
261 int count = 0;
262 const int size_tot = rang_elem_non_std_.size_totale();
263// On remplace le marqueur "0" par l'indice j.
264 for (int elem = 0; elem < size_tot; elem++)
265 {
266 if (rang_elem_non_std_[elem] == 0)
267 {
268 rang_elem_non_std_[elem] = count;
269 count++;
270 }
271 }
272 }
273
274// Construction du tableau de renumerotation des faces. Ce tableau,
275// une fois trie dans l'ordre croissant donne l'ordre des faces dans
276// la zone_VF. La cle de tri est construite de sorte a pouvoir retrouver
277// l'indice de la face a partir de la cle par la formule :
278// indice_face = cle % nb_faces
279 const int nbfaces = les_faces.nb_faces();
280 ArrOfInt sort_key(nbfaces);
281 {
282 nb_faces_std_ =0;
283 const int nb_faces_front = zone().nb_faces_frontiere();
284// Attention : face_voisins_ n'est pas encore initialise, il
285// faut passer par les_faces.voisins() :
286 const IntTab& facevoisins = les_faces.voisins();
287// On place en premier les faces de bord:
288 int i_face;
289 for (i_face = 0; i_face < nbfaces; i_face++)
290 {
291 int key = -1;
292 if (i_face < nb_faces_front)
293 {
294// Si la face est au bord, elle doit etre placee au debut
295// (en fait elle ne doit pas etre renumerotee)
296 key = i_face;
297 }
298 else
299 {
300 const int elem0 = facevoisins(i_face, 0);
301 const int elem1 = facevoisins(i_face, 1);
302// Ces faces ont toujours deux voisins.
303 assert(elem0 >= 0 && elem1 >= 0);
304 if (rang_elem_non_std_[elem0] >= 0 || rang_elem_non_std_[elem1] >= 0)
305 {
306// Si la face est voisine d'un element non standard, elle
307// doit etre classee juste apres les faces de bord:
308 key = i_face;
309 }
310 else
311 {
312// Face standard : a la fin du tableau
313 key = i_face + nbfaces;
314 nb_faces_std_++;
315 }
316 }
317 sort_key[i_face] = key;
318 }
319 sort_key.ordonne_array();
320
321// On transforme a nouveau la cle en numero de face:
322 for (i_face = 0; i_face < nbfaces; i_face++)
323 {
324 const int key = sort_key[i_face] % nbfaces;
325 sort_key[i_face] = key;
326 }
327 }
328// On reordonne les faces:
329 {
330 IntTab& faces_sommets = les_faces.les_sommets();
331 IntTab old_tab(faces_sommets);
332 const int nb_som_faces = faces_sommets.dimension(1);
333 for (int i = 0; i < nbfaces; i++)
334 {
335 const int old_i = sort_key[i];
336 for (int j = 0; j < nb_som_faces; j++)
337 faces_sommets(i, j) = old_tab(old_i, j);
338 }
339 }
340
341 {
342 IntTab& faces_voisins = les_faces.voisins();
343 IntTab old_tab(faces_voisins);
344 for (int i = 0; i < nbfaces; i++)
345 {
346 const int old_i = sort_key[i];
347 faces_voisins(i, 0) = old_tab(old_i, 0);
348 faces_voisins(i, 1) = old_tab(old_i, 1);
349 }
350 }
351
352// Calcul de la table inversee: reverse_index[ancien_numero] = nouveau numero
353 ArrOfInt reverse_index(nbfaces);
354 {
355 for (int i = 0; i < nbfaces; i++)
356 {
357 const int j = sort_key[i];
358 reverse_index[j] = i;
359 }
360 }
361// Renumerotation de elem_faces:
362 {
363// Nombre d'indices de faces dans le tableau
364 const int nb_items = elem_faces_.size();
365 ArrOfInt& array = elem_faces_;
366 for (int i = 0; i < nb_items; i++)
367 {
368 const int old = array[i];
369 array[i] = reverse_index[old];
370 }
371 }
372// Mise a jour des indices des faces de joint:
373 {
374 Joints& joints = zone().faces_joint();
375 const int nbjoints = joints.size();
376 for (int i_joint = 0; i_joint < nbjoints; i_joint++)
377 {
378 Joint& un_joint = joints[i_joint];
379 ArrOfInt& indices_faces = un_joint.set_joint_item(Joint::FACE).set_items_communs();
380 const int nbfaces2 = indices_faces.size_array();
381 assert(nbfaces2 == un_joint.nb_faces()); //!< renum_items_communs rempli ?
382 for (int i = 0; i < nbfaces2; i++)
383 {
384 const int old = indices_faces[i]; //!< ancien indice local
385 indices_faces[i] = reverse_index[old];
386 }
387// Les faces de joint ne sont plus consecutives dans le
388// tableau: num_premiere_face n'a plus ne sens
389 un_joint.fixer_num_premiere_face(-1);
390 }
391 }
392}

423void Zone_EF::discretiser()
424{
425 verifie_compatibilite_domaine();
426
427 Zone& zone_geom=zone();
428 typer_elem(zone_geom);
429 Elem_geom_base& elem_geom = zone_geom.type_elem().valeur();
430 Zone_VF::discretiser();
431
432// Correction du tableau facevoisins:
433// A l'issue de Zone_VF::discretiser(), les elements voisins 0 et 1 d'une
434// face sont les memes sur tous les processeurs qui possedent la face.
435// Si la face est virtuelle et qu'un des deux elements voisins n'est
436// pas connu (il n'est pas dans l'epaisseur du joint), l'element voisin
437// vaut -1. Cela peut etre un voisin 0 ou un voisin 1.
438// On corrige les faces virtuelles pour que, si un element voisin n'est
439// pas connu, alors il est voisin1. Le voisin0 est donc toujours valide.
440 {
441 IntTab& face_vois = face_voisins();
442 const int debut = nb_faces();
443 const int fin = nb_faces_tot();
444 for (int i = debut; i < fin; i++)
445 {
446 if (face_voisins(i, 0) == -1)
447 {
448 face_vois(i, 0) = face_vois(i, 1);
449 face_vois(i, 1) = -1;
450 }
451 }
452 }
453
454// Verification de la coherence entre l'element geometrique et
455// l'elemnt de discretisation
456
457
458 if (sub_type(Segment_EF,type_elem_.valeur()))
459 {
460 if (!sub_type(Segment,elem_geom))
461 {
462 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
463 exit();
464 }
465 }
466 else if (sub_type(Tri_EF,type_elem_.valeur()))
467 {
468 if (!sub_type(Triangle,elem_geom))
469 {
470 Cerr << " Le type de l'element geometrique " <<
471 elem_geom.que_suis_je() << " est incorrect" << finl;
472 Cerr << " Seul le type Triangle est compatible avec la discretisation EF en dimension 2" << finl;
473 Cerr << " Il faut trianguler le domaine lorsqu'on utilise le mailleur interne" ;
474 Cerr << " en utilisant l'instruction: Trianguler nom_dom" << finl;
475 exit();
476 }
477 }
478 else if (sub_type(Tetra_EF,type_elem_.valeur()))
479 {
480 if (!sub_type(Tetraedre,elem_geom))
481 {
482 Cerr << " Le type de l'element geometrique " <<
483 elem_geom.que_suis_je() << " est incorrect" << finl;
484 Cerr << " Seul le type Tetraedre est compatible avec la discretisation EF en dimension 3" << finl;
485 Cerr << " Il faut tetraedriser le domaine lorsqu'on utilise le mailleur interne";
486 Cerr << " en utilisant l'instruction: Tetraedriser nom_dom" << finl;
487 exit();
488 }
489 }
490
491 else if (sub_type(Quadri_EF,type_elem_.valeur()))
492 {
493
494 if (!sub_type(Quadrangle_VEF,elem_geom))
495 {
496 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
497 exit();
498 }
499 }
500 else if (sub_type(Hexa_EF,type_elem_.valeur()))
501 {
502
503 if (!sub_type(Hexaedre_VEF,elem_geom))
504 {
505 Cerr << " Le type de l'element geometrique " << elem_geom.que_suis_je() << " est incorrect" << finl;
506 exit();
507 }
508 }
509
510 int num_face;
511
512
513// On remplit le tableau face_normales_;
514// Attention : le tableau face_voisins n'est pas exactement un
515// tableau distribue. Une face n'a pas ses deux voisins dans le
516// meme ordre sur tous les processeurs qui possedent la face.
517// Donc la normale a la face peut changer de direction d'un
518// processeur a l'autre, y compris pour les faces de joint.
519 {
520 const int n = nb_faces();
521 face_normales_.resize(n, dimension);
522// const Domaine & dom = zone().domaine();
523// Scatter::creer_tableau_distribue(dom, Joint::FACE, face_normales_);
524 creer_tableau_faces(face_normales_);
525 const IntTab& face_som = face_sommets();
526 const IntTab& face_vois = face_voisins();
527 const IntTab& elem_face = elem_faces();
528 const int n_tot = nb_faces_tot();
529 for (num_face = 0; num_face < n_tot; num_face++)
530 {
531 type_elem_.normale(num_face,
532 face_normales_,
533 face_som, face_vois, elem_face, zone_geom);
534 }
535 }
536
537 Zone_VF::calculer_porosites();
538// calculer_volumes_entrelaces();
539
540 zone().domaine().creer_tableau_sommets(volumes_sommets_thilde_);
541}
