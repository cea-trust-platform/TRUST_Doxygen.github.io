99//! Retourne dans le DoubleTab x la trace sur le raccord distant du DoubleTab y localise aux elements du domaine distant
100void Raccord_distant_homogene::trace_elem_distant(const DoubleTab& y, DoubleTab& x) const
101{
102 assert(est_initialise());
103 const int nb_compo_= y.line_size();
104
105 const IntTab& send_data = Tab_Envoi();
106 const ArrOfInt& recv_data = Tab_Recep();
107 const Faces& lesfaces = faces();
108 const int n1 = send_data.dimension(0);
109 const int n2 = recv_data.size_array();
110
111// On dimensionne x si ce n'est pas fait
112 if (x.size_array()==0 && n2!=0)
113 x.resize(n2, nb_compo_);
114 else if (x.dimension(0) != n2)
115 {
116 Cerr << "Call to Frontiere::trace_elem with a DoubleTab x not located on boundary faces." << finl;
117 Process::exit();
118 }
119 Schema_Comm schema;
120 schema.set_send_recv_pe_list(send_pe_list_, recv_pe_list_);
121 schema.begin_comm();
122 for (int i = 0; i < n1; i++)
123 {
124 const int pe_dest = send_data(i, 0);
125 const int face = send_data(i, 1);
126 int item = lesfaces.voisin(face,0);
127 if (item == -1)
128 item = lesfaces.voisin(face,1);
129
130 for (int j=0; j<nb_compo_; j++)
131 schema.send_buffer(pe_dest) << y(item,j);
132 }
133 schema.echange_taille_et_messages();
134
135 for (int i = 0; i < n2; i++)
136 {
137 const int pe_source = recv_data[i];
138 for(int j=0; j<nb_compo_; j++)
139 schema.recv_buffer(pe_source) >> x(i,j);
140 }
141 schema.end_comm();
142}

145//! Retourne dans le DoubleTab x la trace sur le raccord distant du DoubleTab y localise aux faces
146void Raccord_distant_homogene::trace_face_distant(const DoubleTab& y, DoubleTab& x) const
147{
148 assert(est_initialise());
149 int n, N = y.line_size();
150
151 const IntTab& send_data = Tab_Envoi();
152 const ArrOfInt& recv_data = Tab_Recep();
153 const int n1 = send_data.dimension(0);
154 const int n2 = recv_data.size_array();
155
156// On dimensionne x si ce n'est pas fait
157 if (x.size_array()==0 && n2!=0)
158 {
159 if (y.nb_dim() == 2) x.resize(n2, y.dimension(1));
160 else if (y.nb_dim() == 3) x.resize(n2, y.dimension(1), y.dimension(2));
161 else if (y.nb_dim() == 4) x.resize(n2, y.dimension(1), y.dimension(2), y.dimension(3));
162 }
163
164 Schema_Comm schema;
165 schema.set_send_recv_pe_list(send_pe_list_, recv_pe_list_);
166 schema.begin_comm();
167 for (int i = 0; i < n1; i++)
168 {
169 const int pe_dest = send_data(i, 0);
170 const int face = num_premiere_face() + send_data(i, 1);
171 for (n = 0; n < N; n++) schema.send_buffer(pe_dest) << y.addr()[N * face + n];
172 }
173 schema.echange_taille_et_messages();
174 for (int i = 0; i < n2; i++)
175 {
176 const int pe_source = recv_data[i];
177 for (n = 0; n < N; n++) schema.recv_buffer(pe_source) >> x.addr()[N * i + n];
178 }
179 schema.end_comm();
180}

216void Raccord_distant_homogene::completer()
217{
218 const IntTab& send_data = Tab_Envoi();
219 const ArrOfInt& recv_data = Tab_Recep();
220 const int n1 = send_data.dimension(0);
221 const int n2 = recv_data.size_array();
222 send_pe_list_.set_smart_resize(1);
223 recv_pe_list_.set_smart_resize(1);
224 const int nbproc = Process::nproc();
225 ArrOfBit flags(nbproc);
226 flags = 0;
227 for (int i = 0; i < n1; i++)
228 {
229 const int pe = send_data(i, 0);
230 if (!flags.testsetbit(pe))
231 send_pe_list_.append_array(pe);
232 }
233 flags = 0;
234 for (int i = 0; i < n2; i++)
235 {
236 const int pe = recv_data[i];
237 if (!flags.testsetbit(pe))
238 recv_pe_list_.append_array(pe);
239 }
240 send_pe_list_.ordonne_array();
241 recv_pe_list_.ordonne_array();
242}

245//! Initialise le raccord distant avec la frontiere et la zone discretisee opposees au raccord distant, et la zone discretisee du raccord distant
246void Raccord_distant_homogene::initialise(const Frontiere& opposed_boundary, const Zone_dis_base& opposed_zone_dis, const Zone_dis_base& zone_dis)
247{
248 Raccord_distant_homogene& raccord_distant = *this;
249 if(raccord_distant.est_initialise())
250 {
251 Cerr << "Remote connection " << raccord_distant.le_nom() << " already initialized." << finl;
252 Cerr << "Error, contact TRUST support." << finl;
253 exit();
254 }
255 int dim=Objet_U::dimension;
256 raccord_distant.nom_frontiere_voisine()=opposed_boundary.le_nom();
257 const Zone_VF& zone_dis_vf = ref_cast(Zone_VF, zone_dis);
258// On va identifier les faces par leur centres de gravite
259 int parts = Process::nproc();
260 DoubleTabs remote_xv(parts);
261 int prem_face2 = raccord_distant.num_premiere_face();
262 int nb_face2 = raccord_distant.nb_faces();
263 int moi = Process::me();
264 remote_xv[moi].resize(nb_face2,dim);
265 for (int ind_face=0 ; ind_face < nb_face2 ; ind_face++)
266 {
267 int face = prem_face2 + ind_face;
268 for (int j=0 ; j<dim ; j++)
269 remote_xv[moi](ind_face,j) = zone_dis_vf.xv(face,j);
270 }
271
272// Puis on echange les tableaux des centres de gravites
273// envoi des tableaux
274 for (int p = 0; p < parts; p++)
275 envoyer_broadcast(remote_xv[p], p);
276
277 ArrsOfInt racc_vois(parts);
278 for (int p = 0; p < parts; p++)
279 racc_vois[p].set_smart_resize(1);
280
281#ifdef MEDCOUPLING_
282// On traite les informations, chaque proc connait tous les XV
283// Si le proc porte un morceau du raccord_distant
284 int prem_face1 = opposed_boundary.num_premiere_face();
285 int nb_face1 = opposed_boundary.nb_faces();
286 if (nb_face1>0)
287 {
288 const Zone_VF& opposed_zone_dis_vf = ref_cast(Zone_VF,opposed_zone_dis);
289 const DoubleTab& local_xv = opposed_zone_dis_vf.xv();
290
291 ArrOfInt& Recep=raccord_distant.Tab_Recep();
292 Recep.resize_array(nb_face1);
293// double tolerance = 1e-8; //not very tolerant, are we?
294 double tolerance = Objet_U::precision_geom * 100 ; //default value 1e-8 not very tolerant, are we?
295
296// DataArrayDoubles des xv locaux et de tous les remote_xv (a la suite)
297 std::vector<MCAuto<DataArrayDouble> > vxv(parts);
298 std::vector<const DataArrayDouble*> cvxv(parts);
299 for (int p = 0; p < parts; p++)
300 {
301 vxv[p] = DataArrayDouble::New();
302 vxv[p]->useExternalArrayWithRWAccess(remote_xv[p].addr(), remote_xv[p].dimension(0), remote_xv[p].dimension(1));
303 cvxv[p] = vxv[p];
304 }
305 MCAuto<DataArrayDouble> remote_xvs(DataArrayDouble::Aggregate(cvxv)), local_xvs(DataArrayDouble::New());
306 local_xvs->alloc(nb_face1, dim);
307 for (int ind_face = 0; ind_face < nb_face1; ind_face++)
308 for (int j = 0; j < dim; j++)
309 local_xvs->setIJ(ind_face, j, local_xv(prem_face1 + ind_face, j));
310
311// indices des points de remote_xvs les plus proches de chaque point de local_xv
312 MCAuto<DataArrayInt> glob_idx(DataArrayInt::New());
313
314 glob_idx = remote_xvs->findClosestTupleId(local_xvs);
315
316// pour chaque face de local_xv : controle de la tolerance, remplissage de tableau
317 for (int ind_face = 0, face1 = prem_face1; ind_face<nb_face1; ind_face++, face1++)
318 {
319// retour de l'indice global (glob_idx(ind_face)) au couple (proc, ind_face2)
320 int proc = 0, ind_face2 = glob_idx->getIJ(ind_face, 0);
321 while (ind_face2 >= remote_xv[proc].dimension(0)) ind_face2 -= remote_xv[proc].dimension(0), proc++;
322 assert(ind_face2 < remote_xv[proc].dimension(0));
323
324// controle de la tolerance
325 double distance2 = 0;
326 for (int j=0; j<dim; j++)
327 {
328 double x1=local_xv(face1,j);
329 double x2=remote_xv[proc](ind_face2,j);
330 distance2 += (x1-x2)*(x1-x2);
331 }
332 if (distance2 > tolerance * tolerance)
333 {
334 Cerr << "Warning, there is no remote face found on the local boundary " << opposed_boundary.le_nom() << " for the face local number " << ind_face << " (";
335 for (int j=0; j<dim; j++) Cerr << " " << local_xv(face1,j);
336 Cerr << " )" << finl << "the nearest face on the remote boundary " << raccord_distant.le_nom() << " is the face " << ind_face2 << " (";
337 for (int j=0; j<dim; j++) Cerr << " " << remote_xv[proc](ind_face2,j);
338 Cerr << " )" << finl << "which is located at a distance of " << sqrt(distance2) << " and it is above the geometric tolerance " << tolerance << finl;
339 Cerr << "Check your mesh or contact TRUST support." << finl;
340 Process::exit();
341 }
342
343// remplissage des tableaux
344 Recep[ind_face]=proc;
345 racc_vois[proc].append_array(ind_face2);
346 }
347 }
348#else
349 Cerr<<"Raccord_distant_homogene needs TRUST compiled with MEDCoupling."<<finl;
350 exit();
351#endif
352 ArrsOfInt facteurs(Process::nproc());
353 envoyer_all_to_all(racc_vois, facteurs);
354
355 IntTab& Envoi = raccord_distant.Tab_Envoi();
356 if (nb_face2 > 0)
357 Envoi.resize(nb_face2, 2);
358 int ind = 0;
359 for (int p = 0; p < parts; p++)
360 {
361 const ArrOfInt& facteur = facteurs[p];
362 const int sz = facteur.size_array();
363 for (int d=0 ; d<sz ; d++)
364 {
365 Envoi(ind,0) = p;
366 Envoi(ind,1) = facteur[d];
367 ind++;
368 }
369 }
370 raccord_distant.e() = 0;
371 raccord_distant.est_initialise() = 1;
372 raccord_distant.completer();
373 Cerr <<"Initialize the remote connection " << zone_dis.zone().domaine().le_nom() << "/" << raccord_distant.le_nom() << "<-" << opposed_zone_dis.zone().domaine().le_nom() << "/" << opposed_boundary.le_nom() << finl;
374}
