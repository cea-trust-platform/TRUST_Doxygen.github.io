209int Couplage_U::indice_probleme(const Nom& nom_pb) const
210{
211 int i=0;
212 while ((i< nb_problemes()))
213 {
214 if(probleme(i).le_nom()==nom_pb)
215 {
216 return i;
217 }
218 i++;
219 }
220 Cerr << "Problem " << nom_pb
221 << " not found in Couplage_U " << le_nom() << finl;
222 exit();
223// Avoid compiler warning
224 return 0;
