131inline void Op_Correlation::associer(const Zone_dis_base& une_zdis, const Champ_Generique_base& le_champ_a, const
132 Champ_Generique_base& le_champ_b, double t1,double t2)
133{
134 Champ espace_stockage_source;
135 const Champ_base& source = le_champ_a.get_champ(espace_stockage_source);
136 Nom type_le_champ = source.que_suis_je();
137
138 int renomme;
139 renomme=0;
140 if (type_le_champ.debute_par("Champ"))
141 renomme=1;
142
143 type_le_champ.suffix("Champ_");
144 type_le_champ.suffix("Fonc_");
145 Nom type("Champ_Fonc_");
146 if (renomme==1)
147 type+=type_le_champ;
148 else
149 type = type_le_champ;
150
151 integrale_tps_ab_.typer(type);
152 integrale_tps_ab_.associer_zone_dis_base(une_zdis);
153 integrale_tps_ab_.associer(le_champ_a,le_champ_b,1,1,t1,t2);
154}
