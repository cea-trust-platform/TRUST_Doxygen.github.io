30Entree& Solv_GCP_NS::readOn(Entree& is )
31{
32 Motcle accolade_ouverte("{");
33 Motcle accolade_fermee("}");
34 Motcles les_parametres(6);
35 {
36 les_parametres[0] = "seuil";
37 les_parametres[1] = "solveur1";
38 les_parametres[2] = "solveur0";
39 les_parametres[3] = "impr";
40 les_parametres[4] = "precond";
41 les_parametres[5] = "quiet";
42 }
43 int rang;
44
45 Motcle motlu;
46 is >> motlu;
47 if (motlu != accolade_ouverte)
48 {
49 Cerr << "Error when reading the parameters of the conjugate gradient SSOR " << finl;
50 Cerr << "One expected : " << accolade_ouverte << finl;
51 exit();
52 }
53 is >> motlu;
54 while (motlu != accolade_fermee)
55 {
56 rang = les_parametres.search(motlu);
57 switch(rang)
58 {
59 case 0:
60 {
61 is >> seuil_;
62 break;
63 }
64 case 1:
65 {
66 is >> solveur_poisson1;
67 solveur_poisson1.nommer("poisson_solver1");
68 break;
69 }
70 case 2:
71 {
72 is >> solveur_poisson0;
73 solveur_poisson0.nommer("poisson_solver0");
74 break;
75 }
76 case 3:
77 {
78 fixer_limpr(1);
79 break;
80 }
81 case 4:
82 {
83 is >> le_precond_;
84 break;
85 }
86 case 5:
87 {
88 fixer_limpr(-1);
89 break;
90 }
91 default :
92 {
93 Cerr << "Error when reading the parameters of the conjugate gradient SSOR " << finl;
94 Cerr << "One expected " << les_parametres << " instead of " << motlu << finl;
95 exit();
96 }
97 }
98 is >> motlu;
99 }
100 return is;
101}

103static inline void corriger(DoubleVect& X,
104 double alpha,
105 double beta,
106 const DoubleVect& X1,
107 const DoubleVect& X0,
108 int prems, int n)
109{
110 int sz=prems+n;
111 int i=0;
112 for(; i<prems; i++)
113 X(i)+=beta*X0(i);
114 for(; i<sz; i++)
115 X(i)+=alpha*X1(i-prems);
116}

119static inline void injecter(const DoubleVect& X, DoubleVect& tmp,
120 int prems, int n)
121{
122 int sz=prems+n;
123 for(int i=prems; i<sz; i++)
124 tmp(i)=X(i-prems);
125}
126static inline void extraire(DoubleVect& X, const DoubleVect& tmp,
127 int prems, int n)
128{
129 int sz=prems+n;
130 for(int i=prems; i<sz; i++)
131 X(i-prems)=tmp(i);
132}

141int Solv_GCP_NS::resoudre_systeme(const Matrice_Base& matrice,
142 const DoubleVect& secmem,
143 DoubleVect& solution,
144
145 int niter_max)
146{
147// A00 X0 + A01 X1 = F0
148// A10 X0 + A11 X1 = F1
149// X1 = A11^(-1) (F1 - A10 X0)
150// X0 = (A00 -A01 A11^(-1) A10) (F0 - A01 A11^(-1) F1)
151//
152// Gradient conjugue sur ce systeme preconditionne par A00
153//
154// Cerr << "resoudre_systeme : secmem" << secmem << finl;
155
156// int n = secmem.size();
157 assert(solution.size_totale() == secmem.size_totale());
158 const int n = secmem.size_totale();
159
160 const Matrice_Bloc_Sym& matbloc=ref_cast(Matrice_Bloc_Sym, matrice);
161 if (matbloc.nb_bloc_lignes()!=2 || matbloc.nb_bloc_colonnes()!=2) erreur();
162
163 const Matrice_Base& A00=matbloc.get_bloc(0,0);
164 const Matrice_Base& A01=matbloc.get_bloc(0,1);
165 const Matrice_Base& A11=matbloc.get_bloc(1,1);
166 int n0=A00.nb_lignes();
167 int n1=A11.nb_lignes();
168 int nmax_min = 100;
169 int nmax= std::max(Process::mp_sum(n), nmax_min);
170 int niter = 0;
171 double dold,dnew,alfa;
172
173 solution=0.0;
174// Creation de tableaux distribues X1 et F1 sur les sommets
175 ConstDoubleTab_parts mo_solution(solution);
176 DoubleVect X1=mo_solution[1];
177// DoubleVect X1;
178// champ.domaine().creer_tableau_sommets(X1);
179 DoubleVect F1(X1);
180 if (X1.size_totale()!=n1) erreur();
181
182 DoubleVect resu(solution);
183
184// Creation de tableaux distribues X0 et F0 sur les elements
185// DoubleVect X0;
186// champ.domaine().creer_tableau_elements(X0);
187// ConstDoubleTab_parts mo_solution(solution);
188 DoubleVect X0=mo_solution[0];
189 DoubleVect F0(X0);
190 if (X0.size_totale()!=n0) erreur();
191
192// F0 :
193 extraire(F0, secmem, 0, n0);
194 DoubleVect residu(F0);
195// F1 :
196 extraire(F1, secmem, n0, n1);
197// X1 = A11^(-1) F1
198 solveur_poisson1.resoudre_systeme(A11, F1, X1);
199 injecter(X1, solution, n0, n1);
200// F0=A01 X1
201 A01.multvect(X1, F0);
202// residu=-F0+A01 X1
203 residu -= F0;
204 residu *=-1;
205 DoubleVect g(F0);
206 double norme = mp_norme_vect(residu);
207 if(le_precond_.non_nul())
208 le_precond_.preconditionner(A00,residu,g);
209 else
210 solveur_poisson0.resoudre_systeme(A00, residu, g);
211
212 DoubleVect p(g);
213 p *= -1.;
214
215 dold = mp_prodscal(residu,g);
216 dnew = dold;
217 if (limpr()>-1)
218 Cout << "GCP NS residue = " << norme << " " << finl;
219 double s=0;
220 while ( ( norme > seuil_ ) && (niter++ < nmax) )
221 {
222// F1=A10 p
223 p.echange_espace_virtuel();
224 A01.multvectT(p, F1);
225// X1=-A11^(-1) A10p
226 if ((limpr()==1)&& (je_suis_maitre())) Cout << "Solveur1: " ;
227 solveur_poisson1.resoudre_systeme(A11, F1, X1);
228 X1*=-1.;
229// F0=(A00 -A01 A11^(-1) A10) p
230 X1.echange_espace_virtuel();
231 A01.multvect(X1, F0);
232 A00.ajouter_multvect(p, F0);
233 s = mp_prodscal(F0,p);
234 alfa = dold/(s);
235 corriger(solution, alfa, alfa, X1, p, n0, n1);
236 residu.ajoute(alfa,F0);
237 norme = mp_norme_vect(residu);
238 if(norme>seuil_ && dnew>0)
239 {
240 if ((limpr()==1)&& (je_suis_maitre())) Cout << "Solveur0: " ;
241 if(le_precond_.non_nul())
242 le_precond_.preconditionner(A00,residu,g);
243 else
244 solveur_poisson0.resoudre_systeme(A00, residu, g);
245 dnew = mp_prodscal(residu,g);
246 p *= (dnew/dold);
247 p -= g;
248 dold = dnew;
249 if (dnew < 0)
250 {
251 Cerr << "Error : dnew<0 in GCP NS solver." << finl;
252 Cerr << "Try to reduce the threshold value of solveur0" << finl;
253 Cerr << "to a value inferior to those of the GCP NS solver." << finl;
254 exit();
255 }
256 }
257 if ((limpr()==1)&& (je_suis_maitre()))
258 {
259 Cout << "GCP NS residue = " << norme << " " << finl;
260 if ((niter % 15) == 0) Cout << finl ;
261 }
262 }
263 if(norme > seuil_)
264 {
265 Cerr << "No convergence after : " << niter << " iterations\n";
266 Cerr << " Residue : "<< norme << "\n";
267 }
268 solution.echange_espace_virtuel();
269 return(niter);
270}
