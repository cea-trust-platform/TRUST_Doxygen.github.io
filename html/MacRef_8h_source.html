15// Modification B.Mathieu 07/2004 :
16// But :
17// Avoir un pointeur direct vers l'objet pour que .valeur() soit
18// le plus rapide possible (donc inline)
19// Probleme :
20// Si on met un pointeur comme ceci dans la classe O_Uptr
21// Objet_U * ptr;
22// alors il faut que la classe _TYPE_ soit definie pour pouvoir calculer
23// la conversion de type suivante dans la methode "valeur()" :
24// _TYPE_* typed_ptr = (_TYPE_*) ptr;
25// (probleme avec l'heritage multiple).
26// Par exemple, ceci ne fonctionne pas :
27// class Type_base; // forward declaration
28// class REF(_TYPE_) : public Ref_ {
29// // Probleme : cette conversion de type de fonctionne pas :
30// // (warning sur HP, bug avec gcc)
31// inline valeur() { return * ((_TYPE_*) ptr); };
32// }
33// Donc, soit il faut laisser tomber la declaration forward,
34// soit il ne faut pas mettre "valeur()" en inline.
35// Pour avoir les deux il faut que le pointeur soit du type _TYPE_*.
36
