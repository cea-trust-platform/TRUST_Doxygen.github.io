19template <Ordre_RK _ORDRE_ > template<Ordre_RK _O_>
20enable_if_t<_O_ == Ordre_RK::DEUX_WILLIAMSON || _O_ == Ordre_RK::TROIS_WILLIAMSON || _O_ == Ordre_RK::QUATRE_WILLIAMSON, int>
21TRUSTSchema_RK<_ORDRE_>::faire_un_pas_de_temps_eqn_base_generique(Equation_base& eqn)
22{
23 static constexpr int NB_PTS = (_ORDRE_ == Ordre_RK::DEUX_WILLIAMSON) ? 2 : 3;
24
25// Warning sur les 100 premiers pas de temps si facsec est egal a 1 pour faire reflechir l'utilisateur
26 if (nb_pas_dt() >= 0 && nb_pas_dt() <= NW && facsec_ == 1) print_warning(NW);
27
28 DoubleTab& xi = eqn.inconnue().valeurs(), &xip1 = eqn.inconnue().futur();
29 DoubleTab present(xi), qi(xi);
30
31 for (int i = 0; i < NB_PTS; i++)
32 {
33// on fait ca : q_i = a_{i-1} * q_{i-1} + dt * f(x_{i-1})
34 qi *= get_a<_ORDRE_,NB_PTS>()[i];
35 qi.ajoute(dt_, eqn.derivee_en_temps_inco(xip1));
36
37// on fait ca : x_i = x_{i-1} + b_i * q_i
38 xi.ajoute(get_b<_ORDRE_,NB_PTS>()[i], qi);
39 }
40
41 xip1 = xi;
42 xi -= present;
43 xi /= dt_;
44 update_critere_statio(xi, eqn);
45
46// Update boundary condition on futur:
47 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(), temps_courant() + pas_de_temps());
48 xip1.echange_espace_virtuel();
49
50// xi = x0;
51 xi = present;
52
53 return 1;
54}

58TRUSTSchema_RK<_ORDRE_>::faire_un_pas_de_temps_eqn_base_generique(Equation_base& eqn)
59{
60// Warning sur les 100 premiers pas de temps si facsec est egal a 1 pour faire reflechir l'utilisateur
61 if (nb_pas_dt() >= 0 && nb_pas_dt() <= NW && facsec_ == 1) print_warning(NW);
62
63 static constexpr bool IS_DEUX = (_O_ == Ordre_RK::DEUX_CLASSIQUE) , IS_TROIS = (_O_ == Ordre_RK::TROIS_CLASSIQUE), IS_QUATRE = (_O_ == Ordre_RK::QUATRE_CLASSIQUE);
64 static constexpr int NB_PTS = IS_DEUX ? 2 : ( IS_TROIS ? 3 : 4);
65 static constexpr int NB_BUTCHER = IS_DEUX ? 0 : ( IS_TROIS ? 1 : ( IS_QUATRE ? 2 : 3 ));
66
67 DoubleTab& present = eqn.inconnue().valeurs(), &futur = eqn.inconnue().futur();
68 DoubleTabs ki(NB_PTS); //!< just for initializing the array structure ...
69
70 for (int i = 0; i < NB_PTS; i++) ki[i] = present;
71
72 DoubleTrav sauv(present);
73 sauv = present; //!< sauv = y0
74
75// Step 1
76 eqn.derivee_en_temps_inco(ki[0]); //!< ki[0] = f(y0)
77 ki[0] *= dt_; //!< ki[0] = h * f(y0)
78
79 for (int step = 1; step < NB_PTS; step++ ) //!< ATTENTION : ne touche pas !
80 {
81 present = sauv; //!< back to y0
82 for (int i = 0; i < step; i++) present.ajoute(get_butcher_coeff<_ORDRE_,NB_PTS-1>().at(step-1).at(i), ki[i]); //!< Et ouiiiiiiii :-)
83
84 eqn.derivee_en_temps_inco(ki[step]);
85 ki[step] *= dt_;
86 }
87
88 futur = sauv; //!< futur = y1 = y0
89 for (int i = 0; i < NB_PTS; i++) futur.ajoute(BUTCHER_TAB.at(NB_BUTCHER).at(i), ki[i]);
90
91 update_critere_statio(futur, eqn);
92
93// Update boundary condition on futur:
94 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(), temps_courant() + pas_de_temps());
95 futur.echange_espace_virtuel();
96
97 present = sauv; //!< back to y0
98
99 return 1;
100}
