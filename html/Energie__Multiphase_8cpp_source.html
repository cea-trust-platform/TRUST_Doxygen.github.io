394void Energie_Multiphase::calculer_alpha_rho_e(const Objet_U& obj, DoubleTab& val, DoubleTab& bval, tabs_t& deriv)
395{
396 const Equation_base& eqn = ref_cast(Equation_base, obj);
397 const Fluide_base& fl = ref_cast(Fluide_base, eqn.milieu());
398 const Champ_base& ch_rho = fl.masse_volumique().valeur();
399 const Champ_Inc_base& ch_alpha = ref_cast(Pb_Multiphase, eqn.probleme()).eq_masse.inconnue(),
400 &ch_en = ref_cast(Champ_Inc_base, fl.energie_interne()), //toujours un Champ_Inc
401 *pch_rho = sub_type(Champ_Inc_base, ch_rho) ? &ref_cast(Champ_Inc_base, ch_rho) : NULL; //pas toujours un Champ_Inc
402 const DoubleTab& alpha = ch_alpha.valeurs(), &rho = ch_rho.valeurs(), &en = ch_en.valeurs();
403
404 /* valeurs du champ */
405 int i, n, N = val.line_size(), Nl = val.dimension_tot(0), cR = sub_type(Champ_Uniforme, ch_rho);
406 for (i = 0; i < Nl; i++)
407 for (n = 0; n < N; n++) val(i, n) = alpha(i, n) * rho(!cR * i, n) * en(i, n);
408
409 /* on ne peut utiliser valeur_aux_bords que si ch_rho a une zone_dis_base */
410 DoubleTab b_al = ch_alpha.valeur_aux_bords(), b_rho, b_en = ch_en.valeur_aux_bords();
411 int Nb = b_al.dimension_tot(0);
412 if (ch_rho.a_une_zone_dis_base()) b_rho = ch_rho.valeur_aux_bords();
413 else b_rho.resize(Nb, N), ch_rho.valeur_aux(ref_cast(Zone_VF, eqn.zone_dis().valeur()).xv_bord(), b_rho);
414 for (i = 0; i < Nb; i++)
415 for (n = 0; n < N; n++) bval(i, n) = b_al(i, n) * b_rho(i, n) * b_en(i, n);
416
417 DoubleTab& d_a = deriv["alpha"];//derivee en alpha : rho * en
418 for (d_a.resize(Nl, N), i = 0; i < Nl; i++)
419 for (n = 0; n < N; n++) d_a(i, n) = rho(!cR * i, n) * en(i, n);
420
421 /* derivees a travers rho et en */
422 const tabs_t d_vide = {}, &d_rho = pch_rho ? pch_rho->derivees() : d_vide, &d_en = ch_en.derivees();
423 std::set<std::string> vars; //liste de toutes les derivees possibles
424 for (auto && d_c : d_rho) vars.insert(d_c.first);
425 for (auto && d_c : d_en) vars.insert(d_c.first);
426
427 for (auto && var : vars)
428 {
429 const DoubleTab *dr = d_rho.count(var) ? &d_rho.at(var) : NULL, *de = d_en.count(var) ? &d_en.at(var) : NULL;
430 DoubleTab& d_v = deriv[var];
431 for (d_v.resize(Nl, N), i = 0; i < Nl; i++)
432 for (n = 0; n < N; n++)
433 d_v(i, n) = alpha(i, n) * ((dr ? (*dr)(i, n) * en(i, n) : 0) + (de ? rho(!cR * i, n) * (*de)(i, n) : 0));
434 }
435}
