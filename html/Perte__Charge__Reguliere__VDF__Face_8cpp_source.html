58void Perte_Charge_Reguliere_VDF_Face::remplir_num_faces(Nom& nom_sous_zone)
59{
60 const Domaine& le_domaine = equation().probleme().domaine();
61 const Zone_VDF& zone_VDF = ref_cast(Zone_VDF,equation().zone_dis().valeur());
62 const IntTab& elem_faces = zone_VDF.elem_faces();
63 const IntTab& face_voisins = zone_VDF.face_voisins();
64 const DoubleVect& volumes = zone_VDF.volumes();
65 const DoubleTab& xv = zone_VDF.xv() ;
66 const Sous_Zone& la_sous_zone = le_domaine.ss_zone(nom_sous_zone);
67 int nb_poly_ss_zone = la_sous_zone.nb_elem_tot();
68 int k = direction_perte_charge();
69 DoubleVect xv_(dimension) ;
70 int dir_a_faire ;
71 int face_associee,nfac;
72 int num_poly;
73 num_faces.resize(nb_poly_ss_zone*6);
74 corr_front_ss.resize(nb_poly_ss_zone*6);
75
76 corr_front_ss = 1. ;
77
78 nfac = 0 ;
79
80
81 IntVect num_loc(zone_VDF.nb_elem_tot());
82 num_loc = -1;
83 int num_elem,num_face;
84 for (num_elem=0; num_elem<nb_poly_ss_zone; num_elem++)
85 {
86 num_loc[la_sous_zone(num_elem)] = num_elem;
87
88 }
89 for (int direction = 0; direction<dimension; direction++)
90 {
91 Cerr << " Perte_Charge_Reguliere_VDF_Face::remplir num_face en direction " << direction << finl ;
92
93 if ((direction == k) || (dir[direction] == 1 ))
94 {
95 dir_a_faire = k ;
96 if (dir[direction] == 1 ) dir_a_faire = direction ;
97 for (num_elem=0; num_elem<nb_poly_ss_zone; num_elem++)
98 {
99 num_poly = la_sous_zone(num_elem);
100 num_face = elem_faces(num_poly,dir_a_faire);
101 num_faces[nfac++] = num_face;
102 for (int ifa=0; ifa<dimension; ifa++) xv_(ifa)= xv(num_face,ifa) ;
103 corr_front_ss[nfac-1] *= correction_direction( xv_ , dir_a_faire );
104
105 int num_poly_vois0 = face_voisins(num_face,0);
106 if (num_poly_vois0 != -1)
107 if (num_loc[num_poly_vois0] == -1) //!< le poly voisin n'est pas dans la sous_zone
108 {
109 corr_front_ss[nfac-1]*= volumes(num_poly)/(volumes(num_poly)+volumes(num_poly_vois0)) ;
110 }
111 face_associee = elem_faces(num_poly,dir_a_faire+dimension);
112 int num_poly_vois1 = face_voisins(face_associee,1);
113 if (num_poly_vois1 != -1)
114 {
115 if (num_loc[num_poly_vois1] == -1) //!< le poly voisin n'est pas dans la sous_zone
116 {
117 num_faces[nfac++] = face_associee;
118 for (int ifa=0; ifa<dimension; ifa++) xv_(ifa) = xv(face_associee,ifa) ;
119 corr_front_ss[nfac-1] *= correction_direction( xv_, dir_a_faire);
120 corr_front_ss[nfac-1] *= volumes(num_poly)/(volumes(num_poly)+volumes(num_poly_vois1)) ;
121 }
122 }
123 else
124 {
125 num_faces[nfac++]=face_associee;
126 }
127 }
128 }
129 }
130 num_faces.resize(nfac);
131 corr_front_ss.resize(nfac);
132
133}

135DoubleTab& Perte_Charge_Reguliere_VDF_Face::ajouter_(const DoubleTab& inco,DoubleTab& resu) const
136{
137
138 const Zone_VDF& zone_VDF = la_zone_VDF.valeur();
139 const IntTab& face_voisins = zone_VDF.face_voisins();
140 const DoubleVect& volumes_entrelaces = zone_VDF.volumes_entrelaces();
141 const IntTab& elem_faces = zone_VDF.elem_faces();
142 const DoubleVect& porosite_surf = zone_VDF.porosite_face();
143 const DoubleTab& vit = la_vitesse->valeurs();
144 int ndeb_faces_int = zone_VDF.premiere_face_int();
145// on prend nu et non mu
146 const Champ_Don& nu = le_fluide->viscosite_cinematique();
147
148
149 double d_visco=-1;
150 int l_visco_unif=0;
151
152 if (sub_type(Champ_Uniforme,nu.valeur()))
153 {
154 const Champ_Uniforme& ch_nu = ref_cast(Champ_Uniforme,nu.valeur());
155 d_visco = ch_nu(0,0);
156 l_visco_unif = 1;
157 }
158
159 {
160 int nb_faces,numfa;
161
162 double Cf,CK,U,U_abs,Reynolds;
163 const DoubleTab& visco = nu.valeurs();
164 {
165 nb_faces = num_faces.size();
166 for (int i=0; i<nb_faces; i++)
167 {
168 numfa = num_faces[i];
169 U = inco[numfa];
170 U_abs = std::fabs(vit[numfa]);
171
172 if (!l_visco_unif)
173 {
174 if (numfa < ndeb_faces_int)
175 {
176 int n0 = face_voisins(numfa,0);
177 if (n0 != -1)
178 d_visco = visco[n0];
179 else
180 d_visco = visco[face_voisins(numfa,1)];
181 }
182 else
183 d_visco = 0.5*(visco[face_voisins(numfa,0)]+visco[face_voisins(numfa,1)]);
184
185 }
186 if (couronne_tube == 1)
187 {
188 if (vit!=inco)
189 {
190 Cerr<<"not implemented " <<finl;
191 exit();
192 }
193 double U_res=0.,vit1,vit2,div,corr_sign=1.0;
194 int ori,el,fa1,fa2,fa3,fa4 ;
195// modifications ajoutees par fabio :
196 ori = zone_VDF.orientation(numfa);
197 vit1 = 0;
198 vit2 = 0;
199 div = 4.;
200 for (int ind=0; ind <2; ind++)
201 {
202 el = face_voisins(numfa,ind);
203 if (el!=-1)
204 {
205 fa1 = elem_faces(el,(ori+1));
206 fa2 = elem_faces(el,(ori+1+dimension)%(2*dimension));
207 vit1 += vit[fa1] + vit[fa2] ;
208 if (dimension == 3)
209 {
210 fa3 = elem_faces(el,(ori+2));
211 fa4 = elem_faces(el,(ori+2+dimension)%(2*dimension));
212 vit2 += vit[fa3] + vit[fa4];
213 }
214 }
215 else div = 2.;
216 }
217 vit1 = vit1/div;
218 U_res = U*U+vit1*vit1;
219 if (dimension == 3)
220 {
221 vit2 = vit2/div;
222 U_res += vit2*vit2;
223 }
224// U_res = sqrt(U_res)/rasu();
225 U_res = sqrt(U_res);
226 Reynolds = U_res*d()/d_visco;
227 if (!Cf_utilisateur)
228 Cf = calculer_Cf_blasius(Reynolds);
229 else
230 Cf = CF();
231 CK = -0.5*Cf/D();
232 if (U != 0)
233 {
234 corr_sign = U/std::fabs(U);
235 }
236 resu[numfa] += CK*U_res*U_res*volumes_entrelaces[numfa]*porosite_surf[numfa]*corr_front_ss[i]*corr_sign;
237 }
238 else
239 {
240 Reynolds = U_abs*d()/d_visco;
241 if (!Cf_utilisateur)
242 Cf = calculer_Cf_blasius(Reynolds);
243 else
244 Cf = CF();
245 CK = -0.5*Cf/D();
246 resu[numfa] += CK*U*U_abs*volumes_entrelaces[numfa]*porosite_surf[numfa]*corr_front_ss[i];
247 }
248 }
249 }
250
251 }
252 return resu;
253}
