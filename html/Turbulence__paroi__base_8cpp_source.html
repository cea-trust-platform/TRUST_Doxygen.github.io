107void Turbulence_paroi_base::creer_champ(const Motcle& motlu)
108{
109 if (motlu=="u_star" && !champ_u_star_.non_nul())
110 {
111 int nb_comp = 1;
112 Noms noms(1);
113 noms[0]="u_star";
114 Noms unites(1);
115 unites[0] = "m2/s2";
116 double temps=0.;
117 const Equation_base& equation=mon_modele_turb_hyd->equation();
118 const Discretisation_base& discr = equation.probleme().discretisation();
119 discr.discretiser_champ("champ_face",equation.zone_dis(),scalaire,noms,unites,nb_comp,temps,champ_u_star_);
120 champs_compris_.ajoute_champ(champ_u_star_);
121 }

124const Champ_base& Turbulence_paroi_base::get_champ(const Motcle& nom) const
125{
126 if (nom==champ_u_star_.le_nom())
127 {
128// Initialisation a 0 du champ volumique u_star
129 DoubleTab& valeurs = champ_u_star_->valeurs();
130 valeurs=0;
131 const Equation_base& my_eqn=mon_modele_turb_hyd->equation();
132 if (tab_u_star_.size_array()>0)
133 {
134// Boucle sur les frontieres pour recuperer u_star si tab_u_star dimensionne
135 const Zone_Cl_dis& zcl=my_eqn.zone_Cl_dis();
136 int nb_front=my_eqn.zone_dis().valeur().nb_front_Cl();
137 for (int n_bord=0; n_bord<nb_front; n_bord++)
138 {
139 const Cond_lim& la_cl = zcl.les_conditions_limites(n_bord);
140 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
141 int ndeb = le_bord.num_premiere_face();
142 int nfin = ndeb + le_bord.nb_faces();
143 for (int num_face=ndeb; num_face<nfin; num_face++)
144 valeurs(num_face)=tab_u_star_(num_face);
145 }
146 }
147 valeurs.echange_espace_virtuel();
148// Met a jour le temps du champ:
149 champ_u_star_.mettre_a_jour(my_eqn.schema_temps().temps_courant());
150 return champs_compris_.get_champ(nom);
151 }
152 else
153 return champs_compris_.get_champ(nom);
154}
