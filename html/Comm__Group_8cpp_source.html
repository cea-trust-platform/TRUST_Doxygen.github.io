81//! Cette fonction doit etre appelee simultanement<br>par tous les PEs du groupe current_group avec les memes parametres.<br>Les processeurs de la pe_list sont les rangs dans current_group() des<br>processeurs du nouveau groupe. Le maitre du groupe est le premier de la liste,<br>le rang du processeur courant, s'il est dans le groupe est determine par<br>son rang dans la liste. Il ne doit pas y avoir de doublon.<br>Cette fonction est appelee par les methodes init_group des classes derivees.
82void Comm_Group::init_group(const ArrOfInt& pe_list)
83{
84// Tous les processeurs du groupe courant doivent arriver ici
85 const Comm_Group& current = PE_Groups::current_group();
86 current.barrier(0);
87
88 nproc_ = pe_list.size_array();
89 if (nproc_ == 0)
90 {
91 Cerr << "Comm_Group::set_group_properties : empty process list" << finl;
92 exit();
93 }
94 rank_ = -1;
95 const Comm_Group& groupe_trio = PE_Groups::groupe_TRUST();
96 local_ranks_.resize_array(groupe_trio.nproc());
97 local_ranks_ = -1;
98 world_ranks_.resize_array(nproc_);
99
100 for (int i = 0; i < nproc_; i++)
101 {
102// rank du pe dans current_group()
103 const int pe = pe_list[i];
104 if (check_enabled())
105 {
106 int pe_check = pe;
107 envoyer_broadcast(pe_check, 0);
108 if (pe_check != pe)
109 {
110 Cerr << "Comm_Group::set_group_properties : processes have different pe_lists" << finl;
111 exit();
112 }
113 }
114 if (pe < 0 || pe >= current.nproc_)
115 {
116 Cerr << "Comm_Group::set_group_properties : process "
117 << pe << " is not in current_group()" << finl;
118 exit();
119 }
120// rank du pe dans le groupe_TRUST
121 const int world_rank = current.world_ranks_[pe];
122
123 if (local_ranks_[world_rank] >= 0)
124 {
125 Cerr << "Comm_Group::set_group_properties : duplicate pe in pe_list" << finl;
126 exit();
127 }
128
129 world_ranks_[i] = world_rank;
130 local_ranks_[world_rank] = i;
131 if (pe == current.rank_)
132 rank_ = i;
133 }
134}
