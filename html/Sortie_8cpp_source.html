170Sortie& Sortie::operator<<(const Separateur& ob)
171{
172 if (bin_)
173 {
174// En binaire on n'ecrit pas les separateurs
175 }
176 else
177 {
178 switch (ob.get_type())
179 {
180 case Separateur::ENDL:
181// On ecrit "\n" et pas endl...
182// C'est peut-etre une mauvaise idee
183// (*ostream_) << '\n';
184// GF pb sous windows avec ancienne ligne
185 (*ostream_)<<endl;
186// Flush (important pour les fichiers de log)
187 ostream_->flush();
188 break;
189 case Separateur::SPACE:
190 (*ostream_) << ' ';
191 break;
192 }
193 }
194 return *this;
195}
