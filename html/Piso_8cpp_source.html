74Entree& Piso::lire(const Motcle& motlu,Entree& is)
75{
76 Motcles les_mots(1);
77 {
78 les_mots[0] = "nb_corrections_max";
79 }
80
81 int rang = les_mots.search(motlu);
82 switch(rang)
83 {
84 case 0:
85 {
86 is >> nb_corrections_max_;
87 if (nb_corrections_max_ < 2)
88 {
89 Cerr<<"There must be at least two corrections steps for the PISO algorithm."<<finl;
90 exit();
91 }
92 break;
93 }
94
95 default :
96 {
97 Cerr << "Keyword : " << motlu << " is not understood in " << que_suis_je() << finl;
98 exit();
99 }
100 }
101 return is;
102}

127void Piso::iterer_NS(Equation_base& eqn,DoubleTab& current,DoubleTab& pression,
128 double dt,Matrice_Morse& matrice,double seuil_resol,DoubleTrav& secmem,int nb_ite,int& converge, int& ok)
129{
130 converge = 1;
131 if (nb_ite>1) return;
132 Navier_Stokes_std& eqnNS = ref_cast(Navier_Stokes_std,eqn);
133 Parametre_implicite& param_eqn = get_and_set_parametre_implicite(eqn);
134 SolveurSys& le_solveur_ = param_eqn.solveur();
135
136 DoubleTrav gradP(current);
137 DoubleTrav correction_en_pression(pression);
138 DoubleTrav resu(current);
139 int is_dilat = eqn.probleme().is_dilatable();
140
141 double vitesse_norme,vitesse_norme_old ;
142 double pression_norme,pression_norme_old ;
143 vitesse_norme_old = 1.e10 ;
144 pression_norme_old = 1.e10 ;
145
146// int deux_entrees = 0;
147// if (current.nb_dim()==2) deux_entrees = 1;
148 Operateur_Grad& gradient = eqnNS.operateur_gradient();
149 Operateur_Div& divergence = eqnNS.operateur_divergence();
150
151// int nb_comp = 1;
152// int nb_dim = current.nb_dim();
153
154// if(nb_dim==2) nb_comp = current.dimension(1);
155
156// Construction de matrice et resu
157// matrice = A[Un] = M/delta_t + CONV +DIFF
158// resu = A[Un]Un -(A[Un]Un-Ss) + Sv -BtPn
159
160// <IBM> Taking into account penality term for Immersed Boundary Method
161 const int i_source_PDF = eqnNS.get_i_source_pdf();
162 if (i_source_PDF != -1)
163 {
164 Source_PDF_base& src = dynamic_cast<Source_PDF_base&>((eqnNS.sources())[i_source_PDF].valeur());
165 Cerr<<"Immersed Interface: Dirichlet velocity in momentum equation for PDF (if any)."<<finl;
166// Terme PDF IB value : -rho/delta_t ksi_gamma/epsilon U_gamma
167 DoubleTab secmem_pdf(resu);
168 src.calculer_pdf(secmem_pdf);
169 resu -= secmem_pdf;
170 resu.echange_espace_virtuel();
171
172// Terme en temps : -rho/delta_t ksi_gamma Un
173 int i_traitement_special = 101;
174 if (eqnNS.nombre_d_operateurs() > 1)
175 {
176 if (eqnNS.vitesse_pour_transport().le_nom()=="rho_u") i_traitement_special = 1;
177 }
178 DoubleTrav secmem_pdf_time(resu);
179 src.calculer(secmem_pdf_time, i_traitement_special);
180 secmem_pdf += secmem_pdf_time;
181
182// Sauvegarde de secmem_pdf
183 secmem_pdf.echange_espace_virtuel();
184 src.set_sec_mem_pdf(secmem_pdf);
185
186 DoubleTab coeff;
187 coeff = eqnNS.get_champ_coeff_pdf_som();
188 if (eqnNS.get_gradient_pression_qdm_modifie()==1)
189 {
190 Cerr<<"(IBM) Immersed Interface: modified pressure gradient in momentum equation."<<finl;
191 gradP/=coeff;
192 }
193 gradP.echange_espace_virtuel();
194 }
195// </IBM>
196
197 gradient.valeur().calculer(pression,gradP);
198 if (eqnNS.has_interface_blocs()) //si l'interface blocs est disponible, on l'utilise
199 {
200 eqnNS.assembler_blocs_avec_inertie({{ "vitesse", &matrice }}, resu);
201 if (eqnNS.discretisation().que_suis_je().debute_par("PolyMAC"))
202 matrice.ajouter_multvect(current, resu); //pour ne pas etre en increment
203 }
204 else //sinon, on passe par ajouter/contribuer
205 {
206 resu -= gradP;
207 first_special_treatment( eqn, eqnNS, current, dt, resu );
208 eqnNS.assembler_avec_inertie(matrice,current,resu);
209 }
210
211 le_solveur_.valeur().reinit();
212
213// sometimes we need a second special treatement like for ALE for example
214 second_special_treatment( eqn, current, resu, matrice );
215
216// Construction de matrice_en_pression_2 = BD-1Bt[Un]
217// Assemblage reeffectue seulement pour algorithme Piso (avancement_crank_==0)
218 Matrice& matrice_en_pression_2 = eqnNS.matrice_pression();
219 SolveurSys& solveur_pression_ = eqnNS.solveur_pression();
220
221 if (avancement_crank_==0)
222 {
223 assembler_matrice_pression_implicite(eqnNS,matrice,matrice_en_pression_2);
224 solveur_pression_.valeur().reinit();
225 }
226
227// Etape predicteur
228// Resolution du systeme A[Un]U* = -BtPn + Sv + Ss
229// current = U*
230 le_solveur_.resoudre_systeme(matrice,resu,current);
231
232 test_imposer_cond_lim(eqn,current,"apres resolution ",0);
233 current.echange_espace_virtuel();
234 Debog::verifier("Piso::iterer_NS current apres solveur",current);
235
236// Calcul de secmem = BU* (en incompressible) BU* -drho/dt (en quasi-compressible)
237 if (is_dilat)
238 {
239 if (with_d_rho_dt_)
240 {
241 Fluide_Dilatable_base& fluide_dil = ref_cast(Fluide_Dilatable_base,eqn.milieu());
242 fluide_dil.secmembre_divU_Z(secmem);
243 secmem *= -1;
244 }
245 else secmem = 0;
246 divergence.ajouter(current,secmem);
247 }
248 else
249 divergence.calculer(current,secmem);
250 secmem *= -1;
251 secmem.echange_espace_virtuel();
252 Debog::verifier("Piso::iterer_NS secmem",secmem);
253
254// GF il ne faut pas modifier le scd membre le terme en du/dt au bord a deja ete pris en compte dans la resolution precedente
255// eqnNS.assembleur_pression().valeur().modifier_secmem(secmem);
256
257// Etape de correction 1
258 Cout << "Solving mass equation :" << finl;
259// Description du cas implicite
260// Resolution du systeme (BD-1Bt)P' = Bu* (D-1 = M-1 pour le cas implicite)
261// correction_en_pression = P' pour Piso et correction_en_pression = delta_t*P' pour implicite
262 eqnNS.assembleur_pression()->modifier_secmem_pour_incr_p(pression, 1. / dt, secmem);
263 solveur_pression_.resoudre_systeme(matrice_en_pression_2.valeur(),
264 secmem,correction_en_pression);
265 correction_en_pression.echange_espace_virtuel();
266 Debog::verifier("Piso::iterer_NS apres correction_pression",correction_en_pression);
267
268 if (avancement_crank_==1)
269 {
270// calcul de Un+1
271// Calcul de Bt(delta_t*delta_P)
272 gradient.valeur().multvect(correction_en_pression,gradP);
273 eqn.solv_masse().appliquer(gradP);
274 if ((i_source_PDF != -1) && (eqnNS.get_correction_vitesse_modifie()==1))
275 {
276 Cerr<<"(IBM) Immersed Interface: modified velocity correction."<<finl;
277 DoubleTab coeff;
278 coeff = eqnNS.get_champ_coeff_pdf_som();
279 gradP/=coeff;
280 gradP.echange_espace_virtuel();
281 }
282
283// Calcul de Un+1 = U* -delta_t*delta_P
284 current -= gradP;
285 eqn.solv_masse().corriger_solution(current, current); //pour PolyMAC_P0 : sert a corriger ve
286 current.echange_espace_virtuel();
287 divergence.calculer(current,secmem);
288
289// Calcul de Pn+1 = Pn + (delta_t*delta_P)/delat_t
290 Debog::verifier("Piso::iterer_NS correction avant dt",correction_en_pression);
291 correction_en_pression /= dt;
292
293// <IBM> Immersed Interface: modified pressure correction.
294 if ((i_source_PDF != -1) && (eqnNS.get_correction_pression_modifie()==1))
295 {
296 Cerr<<"Immersed Interface: modified pressure correction."<<finl;
297 DoubleTab coeff;
298 coeff = eqnNS.get_champ_coeff_pdf_som();
299 const Source_PDF_base& src = dynamic_cast<Source_PDF_base&>((eqnNS.sources())[i_source_PDF].valeur());
300 src.correct_pressure(coeff,pression,correction_en_pression);
301 }
302 else
303 {
304 pression += correction_en_pression;
305 }
306// </IBM>
307 eqnNS.assembleur_pression().valeur().modifier_solution(pression);
308 pression.echange_espace_virtuel();
309 Debog::verifier("Piso::iterer_NS pression finale",pression);
310 Debog::verifier("Piso::iterer_NS current final",current);
311 if (is_dilat)
312 {
313// on redivise par rho_np_1 avant de sortir
314 diviser_par_rho_np1_face(eqn,current);
315 }
316 return;
317 }
318
319// calcul de la correction en vitesse premiere etape (DU' =-Bt P)
320
321// Calcul de P* = Pn + P'
322 pression += correction_en_pression;
323 eqnNS.assembleur_pression().valeur().modifier_solution(pression);
324// Resolution du systeme D[Un]U' = -BtP'
325 DoubleTrav correction_en_vitesse(current);
326 calculer_correction_en_vitesse(correction_en_pression,gradP,correction_en_vitesse,matrice,gradient);
327
328// Calcul de U** = U* + U'
329 current += correction_en_vitesse;
330 test_imposer_cond_lim(eqn,current,"apres premiere correction ",0);
331 Debog::verifier("Piso::iterer_NS arpes cor pression",pression);
332 Debog::verifier("Piso::iterer_NS arpes cor vitesse",current);
333
334// Etape correcteur 2
335 for (int compt=0; compt<nb_corrections_max_-1; compt++)
336 {
337 correction_en_vitesse.echange_espace_virtuel();
338// Resolution du systeme D resu = EU' + (resu2=0) pour stocker resu = D-1EU'
339 DoubleTrav resu2(resu);
340 int status = inverser_par_diagonale(matrice,resu2,correction_en_vitesse,resu);
341
342 if (status!=0) exit();
343// calcul de P'' BD-1Bt P''= -div(D-1EU')
344
345 resu.echange_espace_virtuel();
346// Calcul de B(D-1EU')
347 divergence.calculer(resu,secmem);
348 secmem *= -1;
349 secmem.echange_espace_virtuel();
350
351// Resolution du systeme (BD-1Bt)P'' = (BD-1E)U'
352// correction_en_pression = P''
353 correction_en_pression = 0;
354 solveur_pression_.resoudre_systeme(matrice_en_pression_2.valeur(),
355 secmem,correction_en_pression);
356
357#ifdef DEBUG
358// Pour verifier que le calcul du gradient ne modifie pas la pression
359 DoubleTrav correction_en_pression_mod(pression);
360 correction_en_pression_mod = correction_en_pression;
361#endif
362// Resolution du systeme D[Un]U'' = -BtP''
363// correction_en_vitesse = U''
364 calculer_correction_en_vitesse(correction_en_pression,gradP,correction_en_vitesse,matrice,gradient);
365
366#ifdef DEBUG
367 correction_en_pression_mod -= correction_en_pression;
368 assert(max_abs(correction_en_pression_mod)==0);
369#endif
370
371// Calcul de U'' = U'' + D-1EU'
372 correction_en_vitesse += resu;
373// ajout des increments
374
375 vitesse_norme = mp_norme_vect(correction_en_vitesse);
376 pression_norme = mp_norme_vect(correction_en_pression);
377
378 if ( (vitesse_norme>vitesse_norme_old) || (pression_norme>pression_norme_old) )
379 {
380 Cout <<"PISO : "<< compt+1 <<" corrections to perform the projection."<< finl;
381 if (is_dilat)
382 {
383// on redivise par rho_np_1 avant de sortir
384 diviser_par_rho_np1_face(eqn,current);
385 }
386 return ;
387 }
388
389 vitesse_norme_old = vitesse_norme;
390 pression_norme_old = pression_norme;
391// Calcul de P** = P* + P''
392 pression += correction_en_pression;
393 eqnNS.assembleur_pression().valeur().modifier_solution(pression);
394
395// Calcul de U*** = U** + U''
396 current += correction_en_vitesse;
397 test_imposer_cond_lim(eqn,current,"apres correction (int)__LINE__",0);
398
399 Debog::verifier("Piso::iterer_NS apres correction pression",pression);
400 Debog::verifier("Piso::iterer_NS apres correction vitesse",current);
401 }
402 if (is_dilat)
403 {
404 diviser_par_rho_np1_face(eqn,current);
405// ref_cast(Navier_Stokes_QC,eqn).impr_impl(eqnNS, Cout);
406 }
407 current.echange_espace_virtuel();
408// divergence.calculer(current, secmem); Cerr<<" ici DIVU "<<mp_max_abs_vect(secmem)<<finl;;
409 Cout <<"PISO : "<<nb_corrections_max_<<" corrections to perform the projection."<< finl;
410}
