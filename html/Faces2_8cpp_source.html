39//! Fonction hors classe<br>Renvoie le numero du plus petit (au sens de la<br>numerotation des sommets) sommet d'une face
40/** \return int<br>
41Signification: le numero du plus petit sommet la face specifiee<br>
42Contraintes: */
43int ppsf(const Faces& faces, int face, int nb_som)
44{
45 int sommet=faces.sommet(face,0);
46 for(int imin=1; imin < nb_som; imin++)
47 sommet=std::min(sommet,faces.sommet(face,imin));

91int same(const Faces& faces1, int f1,
92 const Faces& faces2, int f2,
93 int nb_som)
94{
95 int ok=1;
96 for(int i=0; ((i<nb_som) && ok ); i++)
97 {
98 int nok=1;
99 for(int j=0; ((j<nb_som) && nok ); j++)
100 {
101 if( faces1.sommet(f1, i) == faces2.sommet(f2, j))

103 }
104 if(nok) ok=0; //!< sommet pas trouve ==> pas meme face
