136void Traitement_particulier_NS_canal_VDF::remplir_Tab_recap(IntTab& Tab_rec) const
137{
138 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
139 const Zone_VDF& zone_VDF=ref_cast(Zone_VDF, zdisbase);
140 const DoubleTab& xp = zone_VDF.xp();
141 const IntTab& elem_faces = zone_VDF.elem_faces();
142
143 int face; //recepteur des faces
144 int elem_test,elem_test2; //!< element de test pour les ficitfs
145 int nb_elem_tot = zone_VDF.zone().nb_elem_tot(); //!< nombre total d'elements (reel + fict)
146 int nb_elems = zone_VDF.zone().nb_elem();
147
148 IntTab trouve(1);//!< tableau des elements deja effectue
149 double y=0;
150 int i,num_elem; //!< compteurs
151 int q=1; //Curseur pour les tableau haut
152 trouve[0]=0;
153
154
155 Tab_rec.resize(nb_elems,3); //!< On dimenssione le tableau.
156
157 for (num_elem=nb_elems; num_elem<nb_elem_tot; num_elem++) //!< boucle sur les elements fictifs
158 {
159 face = elem_faces(num_elem,1+dimension);
160 elem_test=zone_VDF.elem_voisin(num_elem,face,0);
161 face = elem_faces(num_elem,1);
162 elem_test2=zone_VDF.elem_voisin(num_elem,face,1);
163
164 if ((elem_test>0) && (elem_test<nb_elems)) //!< si l'element en dessus est un element reel alors
165 {
166 trouve[q-1]=elem_test;
167 q =q +1;
168 trouve.resize(q);
169
170 Tab_rec(elem_test,0)=num_elem; //!< on affecte la meme valeur aux deux case haut et bas
171 Tab_rec(elem_test,1)=num_elem; //ainsi la fonction qui calcul les valeurs voie un element normal.
172
173 y=xp(elem_test,1);
174 for (i=0; i<Ny; i++)
175 {
176 if(est_egal(y,Y[i]))
177 break;
178 }
179
180 Tab_rec(elem_test,2)=i; //!< on garde la valeur de i pour ne pas reefectuer la boucle a chaque pas de temps.
181 }
182 else if ((elem_test2<nb_elems)&&(elem_test2>0)) //sinon si l'element en dessous est un element reel alors
183 {
184 trouve[q-1]=elem_test2;
185 q =q +1;
186 trouve.resize(q);
187
188 Tab_rec(elem_test2,0)=num_elem; //!< on affecte la meme valeur aux deux case haut et bas
189 Tab_rec(elem_test2,1)=num_elem; //ainsi la fonction qui calcul les valeurs voie un element normal.
190
191 y=xp(elem_test2,1);
192 for (i=0; i<Ny; i++)
193 {
194 if(est_egal(y,Y[i]))
195 break;
196 }
197 Tab_rec(elem_test2,2)=i; //!< on garde la valeur de i pour ne pas reefectuer la boucle a chaque pas de temps.
198
199 }
200// sinon rien
201 }
202
203 Cerr << "Traitement particulier canal : Il y a une amelioration a apporter aux face de bord !! " << finl;
204 for (num_elem=0; num_elem<nb_elems; num_elem++)
205 {
206 q=0;//!< on utilise le compteur q qui ne nous sert plus pour verifier si on a trouver un equivalent.
207 for(i=0; i<(trouve.size()-1); i++) //!< trouve est une case trop grand, mais plutot que de le redimentionner on utilise le critere taille -1
208 if(num_elem==trouve[i])
209 {
210 q = 0; //!< on met fixe q qui ne peu repondre au prochain test. // correction on fixe q =0 car c'etais un faux probleme.
211 break;
212 }
213// en realite lambda explose a l'interface.
214 if(q==0) //
215 {
216 face=elem_faces(num_elem,1); //face inferieure
217 elem_test=zone_VDF.elem_voisin(num_elem,face,1);
218
219
220 if (elem_test+1)
221 {
222 Tab_rec(num_elem,1)=elem_test; //!< faux si elem_test=-1 sinon remplit avec l'element en dessous
223 }
224 else
225 {
226 Tab_rec(num_elem,1)=zone_VDF.elem_voisin(num_elem,elem_faces(num_elem,1+dimension),0); //!< on le traite alors comme un virtuel
227 }
228
229 face= elem_faces(num_elem,1+dimension); //face superieure
230 elem_test=zone_VDF.elem_voisin(num_elem,face,0);
231
232 if (elem_test+1)
233 {
234 Tab_rec(num_elem,0)=elem_test; //!< faux si elem_test=-1 sinon remplit avec l'element au dessus
235 }
236 else
237 {
238 Tab_rec(num_elem,0)=zone_VDF.elem_voisin(num_elem,elem_faces(num_elem,1),1); //!< on le traite alors comme un virtuel
239 }
240
241 y = xp(num_elem,1);
242 for (i=0; i<Ny; i++)
243 if(est_egal(y,Y[i])) break;
244
245 Tab_rec(num_elem,2)=i;
246
247 }
248
249 }

252void Traitement_particulier_NS_canal_VDF::calculer_moyenne_spatiale_vitesse_rho_mu(DoubleTab& val_moy) const
253{
254 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
255 const Zone_VDF& zone_VDF=ref_cast(Zone_VDF, zdisbase);
256// const DoubleTab& xp = zone_VDF.xp();
257 const IntTab& elem_faces = zone_VDF.elem_faces();
258 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
259 double u,v,wl;
260 int nb_elems = zone_VDF.zone().nb_elem();
261 int num_elem,i;
262 int face_x_0,face_y_0,face_y_1,face_z_0;
263
264 const Fluide_base& le_fluide = ref_cast(Fluide_base,mon_equation->milieu());
265 const DoubleTab& visco_dyn = le_fluide.viscosite_dynamique()->valeurs();
266 const DoubleTab& tab_rho_elem = le_fluide.masse_volumique().valeurs();
267 int taille_mu=visco_dyn.dimension(0);
268 int taille_rho=tab_rho_elem.dimension(0);
269
270 for (num_elem=0; num_elem<nb_elems; num_elem++)
271 {
272// y=xp(num_elem,1);
273
274 face_x_0 = elem_faces(num_elem,0);
275// face_x_1 = elem_faces(num_elem,dimension);
276 face_y_0 = elem_faces(num_elem,1);
277 face_y_1 = elem_faces(num_elem,1+dimension);
278
279// PQ : 12/10 : pour eviter de moyenner localement la vitesse u et w
280// on fait le choix de "deplacer" celles-ci
281// du centre de la face au centre de l'element
282// en se basant sur le principe que l'ecoulement est homogene
283// suivant les plans xz
284// Pour v, la moyenne s'impose si l'on veut revenir au centre des elements
285
286// u = .5*(vitesse[face_x_0]+vitesse[face_x_1]);
287 u = vitesse[face_x_0];
288 v = .5*(vitesse[face_y_0]+vitesse[face_y_1]);
289
290 i= Tab_recap(num_elem,2);
291
292 val_moy(i,0) += u;
293 val_moy(i,1) += v;
294 val_moy(i,3) += u*u;
295 val_moy(i,4) += v*v;
296 val_moy(i,6) += u*v;
297
298 if(dimension==2) val_moy(i,9) += sqrt(u*u); //vitesse tangentielle pour calcul du frottement
299
300 if(dimension==3)
301 {
302 face_z_0 = elem_faces(num_elem,2);
303// face_z_1 = elem_faces(num_elem,2+dimension);
304
305// w = .5*(vitesse[face_z_0]+vitesse[face_z_1]);
306 wl = vitesse[face_z_0];
307
308 val_moy(i,2) += wl;
309 val_moy(i,5) += wl*wl;
310 val_moy(i,7) += u*wl;
311 val_moy(i,8) += v*wl;
312 val_moy(i,9) += sqrt(u*u+wl*wl); //vitesse tangentielle pour calcul du frottement
313 }
314
315 if (taille_rho==1) val_moy(i,10) += tab_rho_elem(0,0);
316 else val_moy(i,10) += tab_rho_elem[num_elem];
317
318
319 if (taille_mu==1) val_moy(i,11) += visco_dyn(0,0);
320 else val_moy(i,11) += visco_dyn[num_elem];
321 }

347void Traitement_particulier_NS_canal_VDF::calculer_moyenne_spatiale_Temp(DoubleTab& val_moy) const
348{
349 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();
350 const Zone_VDF& zone_VDF=ref_cast(Zone_VDF, zdisbase);
351// const DoubleTab& xp = zone_VDF.xp();
352 const IntTab& elem_faces = zone_VDF.elem_faces();
353 const DoubleTab& temperature = Temp.valeur().valeurs();
354 const DoubleTab& vitesse = mon_equation->inconnue().valeurs();
355 double u,v,wl,T;
356 int nb_elems = zone_VDF.zone().nb_elem();
357 int num_elem,i;
358 int face_x_0,face_x_1,face_y_0,face_y_1,face_z_0,face_z_1;
359
360 for (num_elem=0; num_elem<nb_elems; num_elem++)
361 {
362// y=xp(num_elem,1);
363
364 T = temperature[num_elem];
365
366 face_x_0 = elem_faces(num_elem,0);
367 face_x_1 = elem_faces(num_elem,dimension);
368 face_y_0 = elem_faces(num_elem,1);
369 face_y_1 = elem_faces(num_elem,1+dimension);
370
371 u = .5*(vitesse[face_x_0]+vitesse[face_x_1]);
372 v = .5*(vitesse[face_y_0]+vitesse[face_y_1]);
373
374
375 i=Tab_recap(num_elem,2);
376// for (i=0; i<Ny; i++) if(est_egal(y,Y[i])) break;
377
378 val_moy(i,13) += T;
379 val_moy(i,14) += T*T;
380 val_moy(i,15) += u*T;
381 val_moy(i,16) += v*T;
382
383 if(dimension==3)
384 {
385 face_z_0 = elem_faces(num_elem,2);
386 face_z_1 = elem_faces(num_elem,2+dimension);
387
388 wl = .5*(vitesse[face_z_0]+vitesse[face_z_1]);
389
390 val_moy(i,17) += wl*T;
391 }
392 }
