109//! Cree l'objet Champ_front_calc representant la trace d'un champ<br>inconnue sur une frontiere a partir des noms:<br>- du probleme portant l'inconnue<br>- du bord concerne (la frontiere)<br>- de l'inconnue
110void Champ_front_calc::creer(const Nom& nom_pb,
111 const Nom& nom_bord,
112 const Motcle& nom_inco)
113{
114 nom_autre_pb_ = nom_pb;
115 nom_autre_bord_ = nom_bord;
116 REF(Probleme_base) autre_pb;
117 Objet_U& ob=Interprete::objet(nom_autre_pb_);
118 if(sub_type(Probleme_base, ob))
119 {
120 autre_pb = ref_cast(Probleme_base, ob);
121 }
122 else
123 {
124 Cerr << "We did not find problem with name " << nom_pb << finl;
125 exit();
126 }
127 REF(Champ_base) rch;
128 rch = autre_pb->get_champ(nom_inco);
129 if (sub_type(Champ_Inc_base,rch.valeur()))
130 {
131 l_inconnue=ref_cast(Champ_Inc_base, rch.valeur()) ;
132 fixer_nb_comp(rch.valeur().nb_comp());
133 }
134 else
135 {
136 Cerr << autre_pb->le_nom() << " did not have unknown field with name " << nom_inco << finl;
137 exit();
138 }
139}

141void Champ_front_calc::completer()
142{
143// Check/initialize Raccord boundaries in parallel:
144 if (nproc()>1)
145 {
146 const Zone_dis_base& zone_dis_opposee = front_dis().zone_dis();
147 const Zone_dis_base& zone_dis_locale = frontiere_dis().zone_dis();
148 const Frontiere& frontiere_opposee = front_dis().frontiere();
149 const Frontiere& frontiere_locale = frontiere_dis().frontiere();
150 if (distant_ && !sub_type(Raccord_distant_homogene, frontiere_opposee))
151 {
152 const Nom& nom_domaine_oppose = zone_dis_opposee.zone().domaine().le_nom();
153 Cerr << "Error, the boundary " << frontiere_opposee.le_nom() << " should be a Raccord." << finl;
154 Cerr << "Add in your data file between the definition and the partition of the domain " << nom_domaine_oppose << " :" << finl;
155 Cerr << "Modif_bord_to_raccord " << nom_domaine_oppose << " " << frontiere_opposee.le_nom() << finl;
156 exit();
157 }
158 if (distant_==1)
159 {
160 Raccord_distant_homogene& raccord_distant = ref_cast_non_const(Raccord_distant_homogene, frontiere_opposee);
161 if (!raccord_distant.est_initialise())
162 raccord_distant.initialise(frontiere_locale, zone_dis_locale, zone_dis_opposee);
163 }
164 }
165}
