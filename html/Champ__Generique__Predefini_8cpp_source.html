82const Noms Champ_Generique_Predefini::get_property(const Motcle& query) const
83{
84 Motcles motcles(1);
85 motcles[0] = "unites";
86 int rang = motcles.search(query);
87 switch(rang)
88 {
89 case 0:
90 {
91 Noms mots(1);
92 if (Motcle(type_champ_)=="ENERGIE_CINETIQUE_TOTALE")
93 mots[0] = "kg.m2/s2";
94 else if (Motcle(type_champ_)=="ENERGIE_CINETIQUE_ELEM")
95 mots[0] = "kg/(m.s2)";
96 else if (Motcle(type_champ_)=="VISCOSITE_TURBULENTE")
97 mots[0] = "m2/s";
98 else if (Motcle(type_champ_)=="VISCOUS_FORCE_X")
99 mots[0] = "kg.m2/s";
100 else if (Motcle(type_champ_)=="VISCOUS_FORCE_Y")
101 mots[0] = "kg.m2/s";
102 else if (Motcle(type_champ_)=="VISCOUS_FORCE_Z")
103 mots[0] = "kg.m2/s";
104 else if (Motcle(type_champ_)=="PRESSURE_FORCES_X")
105 mots[0] = "kg.m2/s";
106 else if (Motcle(type_champ_)=="PRESSURE_FORCES_Y")
107 mots[0] = "kg.m2/s";
108 else if (Motcle(type_champ_)=="PRESSURE_FORCES_Z")
109 mots[0] = "kg.m2/s";
110 else if (Motcle(type_champ_)=="TOTAL_FORCE_X")
111 mots[0] = "kg.m2/s";
112 else if (Motcle(type_champ_)=="TOTAL_FORCE_Y")
113 mots[0] = "kg.m2/s";
114 else if (Motcle(type_champ_)=="TOTAL_FORCE_Z")
115 mots[0] = "kg.m2/s";
116 else if (Motcle(type_champ_)=="VISCOUS_FORCE")
117 mots[0] = "kg.m2/s";
118 else if (Motcle(type_champ_)=="PRESSURE_FORCE")
119 mots[0] = "kg.m2/s";
120 else if (Motcle(type_champ_)=="TOTAL_FORCE")
121 mots[0] = "kg.m2/s";
122 return mots;
123 break;
124 }
125 }
126 return champ_->get_property(query);
