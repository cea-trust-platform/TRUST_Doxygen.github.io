64inline double Eval_Puiss_Th_QC_VEF_Face::calculer_terme_source_standard(int num_face) const
65{
66 double source;
67 if (sub_type(Champ_Uniforme,la_puissance.valeur().valeur()))
68 source = puissance(0,0)*volumes_entrelaces(num_face) ;
69 else
70 {
71 double P0,P1;
72 int elem0 = face_voisins(num_face,0);
73 int elem1 = face_voisins(num_face,1);
74 double V0 = volumes(elem0);
75 double V1 = volumes(elem1);
76 if (puissance.nb_dim()==1)
77 {
78 P0 = puissance(elem0);
79 P1 = puissance(elem1);
80 }
81 else
82 {
83 P0 = puissance(elem0,0);
84 P1 = puissance(elem1,0);
85 }
86 source = (P0*V0 + P1*V1) / nb_faces_elem;
87 }
88 return (source*porosite_surf(num_face));
89}

101 double V0 = volumes(elem0);
102 double V1 = (elem1 != -1?volumes(elem1):0);
103 if (puissance.nb_dim()==1)

105 P0 = puissance(elem0);

112 }
113 source = (P0*V0 + P1*V1) / nb_faces_elem;
114 }
115 return (source*porosite_surf(num_face));
