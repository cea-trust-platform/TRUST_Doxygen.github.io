53inline double Eval_Puiss_Th_VEF_Face::calculer_terme_source_standard(int num_face) const
54{
55 double source;
56 if (sub_type(Champ_Uniforme,la_puissance.valeur().valeur()))
57 source = puissance(0,0)*volumes_entrelaces(num_face);
58 else
59 source = (puissance(face_voisins(num_face,0),0)*volumes(face_voisins(num_face,0)) + puissance(face_voisins(num_face,1),0)*volumes(face_voisins(num_face,1)))/(nb_faces_elem);
60
61 return (source*porosite_surf(num_face));
62}

64inline double Eval_Puiss_Th_VEF_Face::calculer_terme_source_non_standard(int num_face) const
65{
66 double source;
67 if (sub_type(Champ_Uniforme,la_puissance.valeur().valeur()))
68 source = puissance(0,0)*volumes_entrelaces_Cl(num_face);
69 else
70 {
71 if (face_voisins(num_face,1) != -1)
72 source = (puissance(face_voisins(num_face,0),0)*volumes(face_voisins(num_face,0)) + puissance(face_voisins(num_face,1),0)*volumes(face_voisins(num_face,1)))/(nb_faces_elem);
73 else
74 source = (puissance(face_voisins(num_face,0),0)*volumes(face_voisins(num_face,0)))/(nb_faces_elem);
75 }
76 return (source*porosite_surf(num_face));
77}
