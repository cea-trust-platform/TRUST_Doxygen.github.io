35Entree& Porosites::interpreter(Entree& is)
36{
37
38 int i;
39 Nom nom_pb, nom_ss_zone;
40 is >> nom_pb;
41 if(! sub_type(Probleme_base, objet(nom_pb)))
42 {
43 Cerr << nom_pb << " is of type " << objet(nom_pb).que_suis_je() << finl;
44 Cerr << "We waited an object of type Probleme_base" << finl;
45 exit();
46 }
47 Probleme_base& pb=ref_cast(Probleme_base, objet(nom_pb));
48
49 is >> nom_ss_zone;
50 if(! sub_type(Sous_Zone, objet(nom_ss_zone)))
51 {
52 Cerr << nom_ss_zone << " is of type " << objet(nom_ss_zone).que_suis_je() << finl;
53 Cerr << "We waited an object of type Sous_Zone" << finl;
54 exit();
55 }
56
57 Sous_Zone& ssz=ref_cast(Sous_Zone, objet(nom_ss_zone));
58
59 double porosite_volu=1;
60 DoubleVect porosites_surf;
61
62 Param param(que_suis_je());
63 param.ajouter("Volumique",&porosite_volu);
64 param.ajouter("Surfacique",&porosites_surf);
65 param.ajouter_condition("value_of_Volumique_gt_0","Volumic porosity must be greater than 0");
66 param.lire_avec_accolades_depuis(is);
67
68 double porsurfmin=porosites_surf.local_min_vect();
69 if(porsurfmin<=0)
70 {
71 Cerr << "You entered a surfacic porosity <=0 " << finl;
72 exit();
73 }
74
75 const Zone_dis& la_zone_dis = pb.equation(0).zone_dis();
76 if (!la_zone_dis.non_nul())
77 {
78 Cerr << "You must have discretized the problem before defining the porosities" << finl;
79 exit();
80 }
81
82// On passe par la zone_VF
83
84 Zone_VF& zvf=ref_cast(Zone_VF, pb.equation(0).zone_dis().valeur());
85 if (zvf.que_suis_je().debute_par("Zone_VEF"))
86 {
87 Cerr << "Porosites should no longer be used in VEF." << finl;
88 Cerr << "Porosites_champ should be used instead." << finl;
89 exit();
90 }
91 DoubleVect& porosite_elem = zvf.porosite_elem();
92 DoubleVect& porosite_face = zvf.porosite_face();
93 const IntTab& elem_faces = zvf.elem_faces();
94 int nb_faces_elem = zvf.zone().nb_faces_elem();
95 int elem;
96
97 if( (porosite_volu!=1) && (porsurfmin!=1) )
98 {
99 for (i=0; i<ssz.nb_elem_tot(); i++)
100 {
101 elem = ssz(i);
102 porosite_elem(elem)=porosite_volu;
103 for (int j=0; j<nb_faces_elem; j++)
104 {
105 double norme_normale =0;
106 int iii;
107 for (iii=0; iii<dimension; iii++)
108 norme_normale+=zvf.face_normales(elem_faces(elem, j),iii)*zvf.face_normales(elem_faces(elem, j),iii);
109
110 porosite_face(elem_faces(elem, j))=0.;
111 for (iii=0; iii<dimension; iii++)
112 {
113 porosite_face(elem_faces(elem, j))+= porosites_surf(iii)*std::fabs(zvf.face_normales(elem_faces(elem, j),iii))
114 *std::fabs(zvf.face_normales(elem_faces(elem, j),iii))/norme_normale;
115 }
116 }
117 }
118 }
119 else if(porosite_volu!=1)
120 {
121 Cerr << " The volumic porosities are used to calculate the surfacic porosities " << finl ;
122 for (i=0; i<ssz.nb_elem_tot(); i++)
123 {
124 elem = ssz(i);
125 porosite_elem(elem)=porosite_volu;
126 }
127 int nb_elem_tot = zvf.zone().nb_elem_tot();
128 for (i=0; i<nb_elem_tot; i++)
129 {
130 for (int j=0; j<nb_faces_elem; j++)
131 {
132 int face_glob = elem_faces(i, j) ;
133 int num_poly_vois = zvf.face_voisins(face_glob,1);
134 int num_poly_vois1 = zvf.face_voisins(face_glob,0);
135
136 if ((num_poly_vois != -1) && (num_poly_vois1 != -1))
137 porosite_face(face_glob) = (porosite_elem(num_poly_vois1)+porosite_elem(num_poly_vois))/2. ;
138 else porosite_face(face_glob) = porosite_elem(i) ;
139
140 }
141 }
142 }
143 else if(porsurfmin!=1)
144 {
145 Cerr << " The surfacic porosities are used to calculate the volumic porosities " << finl ;
146 for (i=0; i<ssz.nb_elem_tot(); i++)
147 {
148 elem = ssz(i);
149 for (int j=0; j<nb_faces_elem; j++)
150 {
151 double norme_normale =0;
152 int iii;
153 for (iii=0; iii<dimension; iii++)
154 norme_normale+=zvf.face_normales(elem_faces(elem, j),iii)*zvf.face_normales(elem_faces(elem, j),iii);
155
156 porosite_face(elem_faces(elem, j))=0.;
157 for (iii=0; iii<dimension; iii++)
158 {
159 porosite_face(elem_faces(elem, j))+= porosites_surf(iii)*zvf.face_normales(elem_faces(elem, j),iii)
160 *zvf.face_normales(elem_faces(elem, j),iii)/norme_normale;
161 }
162 }
163 }
164 }
165 return is;
166}
