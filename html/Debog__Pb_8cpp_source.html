53
54 Param param(que_suis_je());

66 param.ajouter("exit_on_error", &exit_on_error_);
67 param.lire_avec_accolades(is);
68
69 Objet_U& obj = Interprete::objet(nom_pb_);
70 if (!sub_type(Probleme_base, obj))
71 {
72 Cerr << "Error in Debog::interpreter: " << obj << " is not a Problem_base object" << finl;
73 Process::exit();
74 }
75
76 Probleme_base& pb = ref_cast(Probleme_base, obj);
77 if (pb.domaine_dis().nombre_de_zones() == 0)
78 {
79 Cerr<<finl;

81 Cerr<<"It's maybe because you have put"<<finl;
82 Cerr<<"the command line Debog before to have associated"<<finl;
83 Cerr<<"the discretization to the problem."<<finl;
84 }
85 ref_pb_ = pb;
86
87 if (mode_db_ == 0)
88 {
89// Ecriture:
90 write_geometry_data();
91 write_debog_data_file_.ouvrir(fichier_debog_);
92 write_debog_data_file_.setf(ios::scientific);
93 write_debog_data_file_.precision(20);

97 read_geometry_data();

99 debog_data_file_.set_error_action(Entree::ERROR_EXIT);

101

103

107 return is;

116void Debog_Pb::goto_msg(const char *const message)
117{
118 assert(mode_db_ == 1);
119 Nom dummy;
120 Nom n;
121 int num;
122 Nom msg(message); //!< copie sans les espaces avant/apres
123 while (msg.debute_par(" "))
124 msg.suffix(" ");
125 while (msg.finit_par(" "))
126 msg.prefix(" ");
127 while (1)
128 {
129// read until we find a "msg" word:
130 do
131 {
132 debog_data_file_ >> dummy;
133 }
134 while (dummy != "msg");
135
136 debog_data_file_ >> dummy;
137 if (dummy != ":")
138 {
139 Cerr << "Error in Debog_Pb::goto_msg(" << msg << ")\n invalid data in DEBOG file" << finl;
140 exit();
141 }
142 num = -1;
143 debog_data_file_ >> num;
144 debog_data_file_ >> dummy;
145 if (dummy != ":")
146 {
147 Cerr << "Error in Debog_Pb::goto_msg(" << msg << ")\n invalid data in DEBOG file" << finl;
148 exit();
149 }
150 n = "";
151 while (1)
152 {
153 debog_data_file_ >> dummy;
154 if (dummy == "FinMsg")
155 break;
156 n += dummy;
157 n += " ";
158 }
159 if (n.debute_par(msg))
160 break;
161 if (Process::je_suis_maitre())
162 {
163 log_file_ << "Skipping message " << n << " (looking for " << msg << ")" << finl;
164 }
165 }
166 if (Process::je_suis_maitre())
167 log_file_ << "Reading message " << num << "(file) " << debog_msg_count_ << "(current) " << msg << finl;
168
169 detailed_log_file_ << "Reading message " << num << "(file) " << debog_msg_count_ << "(current) " << msg << finl;
170}

209void Debog_Pb::add_renum_item(const DoubleTab& coord_ref, const DoubleTab& coord_par, const MD_Vector& md, const Nom& id)
210{
211 register_item(md, id);
212
213 const double epsilon = Objet_U::precision_geom;
214
215// On cree un octree contenant les points du domaine de reference
216// et on cherche chaque point du domaine local dans cet octree.
217 Octree_Double octree;
218 octree.build_nodes(coord_ref, 1 /* include_virtual_items */);
219
220// Add a new renum array:
221 renum_array_.add(IntVect());
222 IntVect& renum = renum_array_[renum_array_.size()-1];
223 MD_Vector_tools::creer_tableau_distribue(md, renum, Array_base::NOCOPY_NOINIT);
224
225 if (renum.size_totale() != coord_par.dimension_tot(0))
226 {
227 Cerr << "Internal error in Debog::add_renum_item: coordinate array does not match md descriptor" << finl;
228 Process::exit();
229 }
230 const int dim = coord_par.dimension(1);
231 ArrOfDouble center(dim);
232 ArrOfInt elements;
233 elements.set_smart_resize(1);
234 const int n = renum.size_totale();
235 for (int i = 0; i < n; i++)
236 {
237 for (int j = 0; j < dim; j++)
238 center[j] = coord_par(i,j);
239 octree.search_elements_box(center, epsilon, elements);
240// la premiere recherche renvoie tous les sommets potentiellement proches.
241// il faut faire ensuite un test sur chaque sommet:
242 octree.search_nodes_close_to(center, coord_ref, elements, epsilon);
243 const int k = elements.size_array();
244 if (k != 1)
245 {
246 Cerr << "Debog::add_renum_item: Error. Id=" << id
247 << "\n Item with following coordinates was found " << k << " times within epsilon=" << epsilon
248 << "\n in the reference geometry: " << center << finl;
249 if (noeuds_doubles_ignores_==0)
250 {
251 Cerr << "If you wan't to discard nodes comparison in the mesh, add 'noeuds_doubles_ignores 1' option in debog_pb keyword." << finl;
252 exit();
253 }
254 else
255 {
256 Cerr << "Warning, this item will be ignored during the comparison..." << finl;
257 renum[i] = -1;
258 }
259 }
260// renum[i] is the index of the i-th local item in the reference geometry
261 renum[i] = elements[0];
262 }

265void Debog_Pb::read_geometry_data()
266{
267 const Domaine& dom = ref_pb_.valeur().domaine();
268 const Zone_dis_base& zd = ref_pb_.valeur().domaine_dis().zone_dis(0).valeur();
269 const Zone_VF& zvf = ref_cast(Zone_VF, zd);
270 {
271 DoubleTab coord_som_seq; //!< sommets
272 DoubleTab xp_seq; //!< centres des elements Il faut passer dans un groupe monoprocesseur pour Domaine::readOn:
273//
274 {
275 DERIV(Comm_Group) group;
276 ArrOfInt liste_procs(1); //!< Liste de 1 processeur contenant le proc 0
277 PE_Groups::create_group(liste_procs, group, 1);
278 if (PE_Groups::enter_group(group.valeur()))
279 {
280 EFichier f(fichier_domaine_);
281 Domaine dom_seq;
282 f >> dom_seq;
283 coord_som_seq = dom_seq.coord_sommets();
284 const Elem_geom_base& elem = dom_seq.zone(0).type_elem().valeur();
285 elem.calculer_centres_gravite(xp_seq);
286 PE_Groups::exit_group();
287 }
288 }
289// Tous les processeurs recoivent les coordonnees des sommets de reference
290// (detruire le descripteur sinon printOn refuse d'envoyer le vecteur)
291 coord_som_seq.set_md_vector(MD_Vector());
292 envoyer_broadcast(coord_som_seq, 0);
293// Idem avec les centres des elements
294 xp_seq.set_md_vector(MD_Vector());
295 envoyer_broadcast(xp_seq, 0);
296
297// Calculer les renumerotations:
298 const DoubleTab& coord_som = dom.coord_sommets();
299 const MD_Vector& md_som = dom.coord_sommets().get_md_vector();
300 add_renum_item(coord_som_seq, coord_som, md_som, "SOM");
301 const DoubleTab& xp = zvf.xp();
302// Le tableau xp n'a pas de structure parallele, on la prend dans les elements...
303 const MD_Vector& md_elem = dom.zone(0).les_elems().get_md_vector();
304 add_renum_item(xp_seq, xp, md_elem, "ELEM");
305 }
306 {
307 DoubleTab xv_seq;
308 DoubleTab xa_seq;
309 int avec_aretes = 1;
310 if (Process::je_suis_maitre())
311 {
312 EFichier f(fichier_faces_);
313 f.precision(20);
314 {
315 IntTab elem_faces;
316 f >> elem_faces; //!< non utilise
317 }
318 f >> xv_seq;
319 {
320 IntTab face_voisins;
321 f >> face_voisins; //!< non utilise
322 }
323 f.set_error_action(Entree::ERROR_EXCEPTION);
324 try
325 {
326 f >> xa_seq;
327 }
328 catch (Entree_Sortie_Error)
329 {
330 avec_aretes = 0;
331 }
332 }
333 envoyer_broadcast(xv_seq, 0);
334 const DoubleTab& xv = zvf.xv();
335 const MD_Vector& md_face = zvf.md_vector_faces();
336 add_renum_item(xv_seq, xv, md_face, "FACE");
337
338 envoyer_broadcast(avec_aretes, 0);
339 if (avec_aretes)
340 {
341 envoyer_broadcast(xa_seq, 0);
342 const DoubleTab& xa = zvf.xa();
343 const MD_Vector& md_arete = zvf.md_vector_aretes();
344 add_renum_item(xa_seq, xa, md_arete, "ARETE");
345 }
346 }

373//! md_lignes: descripteur des indices de lignes (cad descripteur du vecteur b dans A*x=b)<br>mb_colonnes: idem, indices colonnes (cad descripteur du vecteur x dans A*x=b)
374void Debog_Pb::verifier_matrice(const char *const msg, const Matrice_Base& matrice, const MD_Vector& md_lignes, const MD_Vector& md_colonnes)
375{
376// Attention: cette implementation ne marche que pour les types md_colonnes enregistres dans renum_array_
377// (cad types simples sommets, elements, faces, aretes, et pas les MD_Vector_composite)
378// et uniquement pour linesize==1 (une ligne et une colonne de la matrice par item)
379
380// Pour chaque colonne i de la matrice, ecriture du produit matrice * vecteur_i
381// avec vecteur_i[j] = (i==j);
382 DoubleVect vecteur_i;
383 DoubleVect tmp;
384 MD_Vector_tools::creer_tableau_distribue(md_colonnes, vecteur_i);
385 MD_Vector_tools::creer_tableau_distribue(md_lignes, tmp);
386 Nom id;
387
388// Boucle sur les items sequentiels du vecteur x de A*x=b
389 const int nb_colonnes = md_colonnes.valeur().nb_items_seq_tot();
390
391 for (int i = 0; i < nb_colonnes; i++)
392 {
393 int renum_i;
394 if (mode_db_ == 0)
395 {
396 renum_i = i;
397 }
398 else
399 {
400// Recherche l'indice local correspondant a l'indice sequentiel i
401// Si i est un item virtuel on le prend quand meme: vecteur_i aura son espace virtuel a jour
402// Si i n'existe pas sur ce processeur, on met -1
403 renum_i = -1;
404// Renum_colonnes[j] est l'indice dans le vecteur de reference sequentiel de l'item j sur ce processeur
405 const IntVect& renum_colonnes = find_renum_vector(md_colonnes, id);
406 const int sz = renum_colonnes.size_totale();
407 for (int j = 0; j < sz; j++)
408 {
409 if (renum_colonnes[j] == i)
410 {
411 renum_i = j;
412 break;
413 }
414 }
415 }
416 if (renum_i >= 0)
417 vecteur_i[renum_i] = 1.;
418 matrice.multvect(vecteur_i, tmp);
419 Nom msg2(msg);
420 msg2 += " produit avec base ";
421 msg2 += Nom(i);
422 verifier(msg2, tmp);
423 if (renum_i >= 0)
424 vecteur_i[renum_i] = 0.;
425 }
426}
