84// Description
85// Provides afield with a name, a time interval, components, and no field ownership.
86// The geometry includes all nodes of the domain and all elements of the sous_zone or domain.
87// WEC : optimization possible here (recalculated each time)
88void Champ_input_P0::getTemplate(TrioField& afield) const
89{
90
91 afield.clear();
92 afield.setName(le_nom().getChar());
93
94 afield._time1=mon_pb->presentTime();
95 afield._time2=mon_pb->futureTime();
96
97 afield._nb_field_components=nb_comp();
98
99 /* coordonnees */
100 const DoubleTab& sommets=mon_pb->domaine().les_sommets();
101 afield._nbnodes=sommets.dimension(0);
102 afield._mesh_dim = afield._space_dim = sommets.dimension(1);
103 affecte_double_avec_doubletab(&afield._coords,sommets);
104
105 /* connectivites */
106 const Zone_VF& zvf = ref_cast(Zone_VF, zone_dis_base());
107 afield._nb_elems = ma_sous_zone.non_nul() ? nb_elems_reels_sous_zone_ : zvf.nb_elem();
108 Motcle type_elem = zvf.zone().type_elem()->que_suis_je();
109 if (type_elem != "POLYEDRE") //cas simple -> il suffit de copier les_elems
110 {
111 const IntTab& conn = zvf.zone().les_elems();
112// le seul moyen qu'on a d'eviter que des polygones soient pris pour des quadrilateres est d'avoir un tableau de connectivite de largeur > 4...
113 afield._nodes_per_elem = std::max(conn.dimension(1), type_elem == "POLYGONE" ? (int) 5 : 0);
114 afield._connectivity = new int[afield._nb_elems * afield._nodes_per_elem];
115 for (int i = 0; i < afield._nb_elems; i++)
116 for (int j = 0; j < afield._nodes_per_elem; j++)
117 {
118 const int e = ma_sous_zone.non_nul() ? ma_sous_zone.valeur()[i] : i; //numero de l'element
119 afield._connectivity[afield._nodes_per_elem * i + j] = j < conn.dimension(1) ? conn(e, j) : -1;
120 }
121 }
122 else //polyedres -> il faut reconstruire une connectivite de type MEDCoupling a la main
123 {
124 afield._nodes_per_elem = zvf.elem_faces().dimension(1) * (zvf.face_sommets().dimension(1) + 1); //un -1 apres chaque face
125 int *p = afield._connectivity = new int[afield._nb_elems * afield._nodes_per_elem];
126 for (int i = 0, j, k, e, f, s; i < (ma_sous_zone.non_nul() ? ma_sous_zone.valeur().nb_elem_tot() : zvf.nb_elem()); i++)
127 {
128 if (ma_sous_zone.non_nul() && ma_sous_zone.valeur()[i] >= zvf.nb_elem()) continue; //element non reel de la sous-zone -> on saute
129 e = ma_sous_zone.non_nul() ? ma_sous_zone.valeur()[i] : i; //numero de l'element
130 int *pf = p + afield._nodes_per_elem; //fin de la ligne
131 /* insertion de la connectivite de chaque face, suivie d'un -1 */
132 for (j = 0; j < zvf.elem_faces().dimension(1) && (f = zvf.elem_faces(e, j)) >= 0; j++, *p = -1, p++)
133 for (k = 0; k < zvf.face_sommets().dimension(1) && (s = zvf.face_sommets(f, k)) >= 0; k++) *p = s, p++;
134 /* des -1 jusqu'a la ligne suivante */
135 for ( ; p < pf; p++) *p = -1;
136 }
137 }
138}

141// Fills valeurs_ on either the whole domain or a subzone
142void Champ_input_P0::setValue(const TrioField& afield)
143{
144 if (ma_sous_zone.non_nul())
145 {
146 const Sous_Zone& ssz=ma_sous_zone.valeur();
147 if (afield._nb_elems != nb_elems_reels_sous_zone_)
148 throw WrongArgument(mon_pb.le_nom().getChar(),"setInputField","afield","should have the same _nb_elems as returned by getInputFieldTemplate");
149 assert(valeurs_.dimension(1)==nb_compo_);
150 if (afield._nb_field_components!=valeurs_.dimension(1))
151 throw WrongArgument(mon_pb.le_nom().getChar(),"setInputField","afield","should have the same _nb_field_components as returned by getInputFieldTemplate");
152 for (int i=0, j, k = -1; i<ssz.nb_elem_tot(); i++)
153 if (ssz[i] < nb_elems_reels_loc_)
154 for (j = 0, k++ ; j<nb_compo_; j++)
155 valeurs_(ssz[i],j)=afield._field[k*nb_compo_+j];
156 }
157 else
158 Champ_Input_Proto::setValueOnTab(afield,valeurs_);
159
160}
