52void PolyMAC_P0_discretisation::grad_u(const Zone_dis& z,const Zone_Cl_dis& zcl,const Champ_Inc& ch_vitesse,Champ_Fonc& ch) const
53{
54 const Champ_Face_PolyMAC_P0& vit = ref_cast(Champ_Face_PolyMAC_P0,ch_vitesse.valeur());
55 const Zone_PolyMAC_P0& zone_poly = ref_cast(Zone_PolyMAC_P0, z.valeur());
56 const Zone_Cl_PolyMAC& zone_cl_poly = ref_cast(Zone_Cl_PolyMAC, zcl.valeur());
57
58 ch.typer("grad_Champ_Face_PolyMAC_P0");
59
60 grad_Champ_Face_PolyMAC_P0& ch_grad_u = ref_cast(grad_Champ_Face_PolyMAC_P0,ch.valeur()); //
61
62 ch_grad_u.associer_zone_dis_base(zone_poly);
63 ch_grad_u.associer_zone_Cl_dis_base(zone_cl_poly);
64 ch_grad_u.associer_champ(vit);
65 ch_grad_u.nommer("gradient_vitesse");
66 ch_grad_u.fixer_nb_comp(dimension * dimension * vit.valeurs().line_size());
67
68 for (int n = 0; n<ch_grad_u.valeurs().line_size(); n++)
69 {
70 Nom phase = Nom(n);
71 if (dimension == 2)
72 {
73 ch_grad_u.fixer_nom_compo(dimension*n+0,Nom("dU_")+phase); //!< dU
74 ch_grad_u.fixer_nom_compo(dimension*n+1,Nom("dV_")+phase); //!< dV
75 }
76 else
77 {
78 ch_grad_u.fixer_nom_compo(dimension*n+0,Nom("dU_")+phase); //!< dU
79 ch_grad_u.fixer_nom_compo(dimension*n+1,Nom("dV_")+phase); //!< dV
80 ch_grad_u.fixer_nom_compo(dimension*n+2,Nom("dW_")+phase); //!< dW
81 }
82 }
83 ch_grad_u.fixer_nature_du_champ(multi_scalaire); //!< tensoriel pour etre precis
84 ch_grad_u.fixer_nb_valeurs_nodales(-1);
85 ch_grad_u.fixer_unite("s-1");
86 ch_grad_u.changer_temps(-1); //!< so it is calculated at time 0
87}
