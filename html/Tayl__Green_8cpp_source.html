29DoubleTab& Tayl_Green::valeur_aux(const DoubleTab& positions, DoubleTab& tab_valeurs) const
30{
31 assert(tab_valeurs.size_totale() == positions.size());
32 for (int i = 0; i < tab_valeurs.dimension(0); i++)
33 {
34 tab_valeurs(i, 0) = -sin(kwave * positions(i, 0)) * cos(kwave * positions(i, 1));
35 tab_valeurs(i, 1) = cos(kwave * positions(i, 0)) * sin(kwave * positions(i, 1));
36 if (dimension == 3) tab_valeurs(i, 2) = 0.;
37 }
38 return tab_valeurs;
39}

41DoubleVect& Tayl_Green::valeur_aux_compo(const DoubleTab& positions, DoubleVect& tab_valeurs, int ncomp) const
42{
43 assert(tab_valeurs.size_totale() == positions.dimension(0));
44 if (ncomp == 0)
45 for (int i = 0; i < tab_valeurs.size(); i++)
46 tab_valeurs(i) = -sin(kwave * positions(i, 0)) * cos(kwave * positions(i, 1));
47 else if (ncomp == 1)
48 for (int i = 0; i < tab_valeurs.size(); i++)
49 tab_valeurs(i) = cos(kwave * positions(i, 0)) * sin(kwave * positions(i, 1));
50 else if (ncomp == 2)
51 for (int i = 0; i < tab_valeurs.size(); i++) tab_valeurs(i) = 0;
52
53 return tab_valeurs;
