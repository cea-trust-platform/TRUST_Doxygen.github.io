43int Champ_front_xyz_debit::initialiser(double tps, const Champ_Inc_base& inco)
44{
45// Cerr << "Champ_front_xyz_debit::initialiser" << finl;
46 if (!Champ_front_normal::initialiser(tps,inco))
47 return 0;
48 const Front_VF& le_bord= ref_cast(Front_VF,frontiere_dis());
49 flow_rate_.valeur().initialiser(tps,inco);
50 if ( !sub_type(Champ_front_uniforme,flow_rate_.valeur()) && !sub_type(Ch_front_input_uniforme,flow_rate_.valeur()) && !sub_type(Champ_front_t,flow_rate_.valeur()) && !sub_type(Champ_front_Tabule,flow_rate_.valeur()))
51 {
52 Cerr << "\nError in Champ_front_xyz_debit::initialiser() \nflow_rate must be of type champ_front_uniforme, ch_front_input_uniforme, champ_front_tabule or champ_front_fonc_t!" << finl;
53 exit();
54 }
55 if (!flow_rate_alone_)
56 {
57 velocity_profil_.valeur().initialiser(tps,inco);
58 if (velocity_profil_.valeur().nb_comp()!=dimension)
59 {
60 Cerr << "\nError in Champ_front_xyz_debit::initialiser() \nvelocity_profil must have " << dimension << " components!" << finl;
61 exit();
62 }
63 }
64 initialiser_coefficient(inco, tps);
65
66// For tabular sizing only
67 normal_vectors_=les_valeurs[0].valeurs();
68 integrale_.resize(flow_rate_.valeur().valeurs().line_size());
69 DoubleTab velocity_user;
70 if (!flow_rate_alone_)
71 velocity_user=velocity_profil_.valeur().valeurs();
72 calculer_normales_et_integrale(le_bord,velocity_user);
73
74// the flow rate Q_user and the velocity profil U_user are imposed on the boundary
75// so the velocity field u is computed as following :
76// u = ( Q_user U_user n)/ Integrale( U_user . n dS)
77// where n is the normal vector associated to the boundary
78
79// fill at every time
80 int nb_cases=les_valeurs->nb_cases(); //!< cases number of the wheel
81 for (int t=0; t<nb_cases; t++)
82 {
83 DoubleTab& velocity_field=les_valeurs[t].valeurs(); //!< values of the field at time t, type Roue_ptr
84 calculer_champ_vitesse(le_bord, velocity_field, velocity_user, tps);
85 }
86 return 1;
87}

89void Champ_front_xyz_debit::calculer_normales_et_integrale(const Front_VF& le_bord, DoubleTab& velocity_user)
90{
91 const int N = flow_rate_.nb_comp();
92 const Zone_VF& zone_VF = ref_cast(Zone_VF,zone_dis());
93 integrale_ = 0.;
94
95 for(int i = 0; i < le_bord.nb_faces_tot(); i++)
96 {
97 int f = le_bord.num_face(i);
98 const double dS = zone_VF.face_surfaces(f);
99 for (int n = 0; n < N; ++n)
100 {
101 double u_scal_n = 0;
102 for(int j = 0; j < dimension; j++)
103 {
104 normal_vectors_(i, j) = -zone_VF.normalized_boundaries_outward_vector(f, 1.)(j); //!< inward normal
105 u_scal_n += normal_vectors_(i, j) * (velocity_user.size() ? velocity_user(velocity_user.size() > dimension ? i : 0, j) : normal_vectors_(i, j));
106 }
107 if (i < le_bord.nb_faces()) integrale_(n) += dS * u_scal_n / coeff_(i, n) * zone_VF.porosite_face(f); //real faces only
108 }
109 }
110 for (int n = 0; n < N; ++n)
111 integrale_(n) = mp_sum(integrale_(n));
112}

121void Champ_front_xyz_debit::calculer_champ_vitesse(const Front_VF& le_bord, DoubleTab& velocity_field, DoubleTab& velocity_user, double temps)
122{
123 const int N = flow_rate_.nb_comp();
124
125 for(int i = 0; i < le_bord.nb_faces_tot(); i++)
126 for (int n = 0; n < N; ++n)
127 for(int j = 0; j < dimension; j++)
128 {
129 double v_mult = flow_rate_.valeur().valeurs_au_temps(temps)(0, n) / integrale_(n) ; //the profile/normals must be multiplied by this to get the correct inward flow
130 double n_mult = flow_rate_alone_ ? normal_vectors_(i,j) : (velocity_user(velocity_user.size() > dimension ? i : 0, j));
131 velocity_field(i, N * j + n) = v_mult * n_mult;
132 }
133}

201void Champ_front_xyz_debit::mettre_a_jour(double temps)
202{
203// Cerr << "Champ_front_xyz_debit::mettre_a_jour temps = " << temps << finl;
204 if (update_coeff_) update_coeff(temps);
205 DoubleTab velocity_user;
206
207 flow_rate_.valeur().mettre_a_jour(temps);
208 if (!flow_rate_alone_)
209 {
210 velocity_profil_.valeur().mettre_a_jour(temps);
211 velocity_user=velocity_profil_.valeur().valeurs_au_temps(temps);
212 }
213 else
214 {
215 velocity_user.resize(1,dimension);
216 velocity_user=1;
217 }
218 if (update_coeff_) calculer_normales_et_integrale(ref_cast(Front_VF,frontiere_dis()),velocity_user);
219 DoubleTab& velocity_field=valeurs_au_temps(temps); //!< values of the field at time t, type Roue_ptr
220 const Front_VF& le_bord= ref_cast(Front_VF,frontiere_dis());
221 if (velocity_field.size_array())
222 calculer_champ_vitesse(le_bord, velocity_field, velocity_user, temps);
223}
