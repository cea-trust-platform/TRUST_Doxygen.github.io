52int Decouper_multi::lire_motcle_non_standard(const Motcle& mot, Entree& is)
53{
54 if (mot=="domaine") //parametres de decoupage d'un domaine
55 {
56 Decouper decoup;
57 decoup.lire(is);
58 if (decoupeurs.count(decoup.nom_domaine.getString())) //decoupeur deja lu
59 Process::exit(Nom("Decouper_multi: domain ") + decoup.nom_domaine + "already read!");
60 else decoupeurs[decoup.nom_domaine.getString()] = decoup; //sinon, on l'ajoute
61 }
62 else return -1;
63 return 0;
64}

67Entree& Decouper_multi::interpreter(Entree& is)
68{
69 int i, j, k, l, d, s, ns, count = 0;
70// lecture des domaines et des raccords
71 Param param(que_suis_je());
72 param.ajouter_non_std("domaine",(this),Param::REQUIRED);
73 param.ajouter("tolerance", &tolerance);
74 param.lire_avec_accolades_depuis(is);
75
76 /* partition des domaines */
77 std::vector<IntVect> v_proc; //partition de chaque domaine
78 std::vector<Decouper*> v_dec; //decoupeurs
79 std::vector<const Domaine*> v_dom; //domaines
80 std::vector<Static_Int_Lists> v_se; //connectivites
81 std::vector<MCAuto<DataArrayDouble>> v_da; //DataArrayDouble de coordonnes
82 std::vector<const DataArrayDouble*> v_pda; //et des pointeurs (ppur Aggregate)
83 std::vector<int> off = { 0 }; //offset des sommets de chaque domaine dans le tableau aggrege
84 for (auto &&n_d : decoupeurs)
85 {
86 const Domaine& dom = n_d.second.ref_domaine.valeur();
87 const DoubleTab& coord = dom.coord_sommets();
88 v_proc.push_back(IntVect()), v_dec.push_back(&n_d.second), v_dom.push_back(&dom);
89 v_se.push_back(Static_Int_Lists()), v_da.push_back(DataArrayDouble::New());
90 v_da.back()->useExternalArrayWithRWAccess((double *) coord.addr(), coord.dimension(0), coord.dimension(1));
91 v_pda.push_back(v_da.back()), off.push_back(off.back() + coord.dimension(0));
92
93 Partitionneur_base& partitionneur = n_d.second.deriv_partitionneur.valeur();
94 partitionneur.declarer_bords_periodiques(n_d.second.liste_bords_periodiques);
95 partitionneur.construire_partition(v_proc.back(), n_d.second.nb_parts_tot);
96 construire_connectivite_som_elem(dom.nb_som(), dom.zone(0).les_elems(), v_se.back(), 1);
97 }
98
99#ifdef MEDCOUPLING_
100 /* concatenation des coordonnees et recherche de sommets coincidents */
101 Cerr << "Decouper_multi: searching for coinciding vertices ... ";
102 MCAuto<DataArrayDouble> da(DataArrayDouble::Aggregate(v_pda)); //tous les sommets!
103 DataArrayInt* S_i = NULL, *S = NULL; //groupes de sommets coincidants : S([S_i(i), S_i(i + 1)[)
104 da->findCommonTuples(tolerance, -1, S, S_i); //!< * heavy lifting *
105
106 /* pour chaque groupe de sommets, recherche de tous les procs les touchant et ajout a chaque sommet des procs qui ne le possedent pas encore */
107 std::vector<std::map<int, std::set<int>>> v_sp(v_dec.size()); //v_sp[d][s] = { processeurs supplementaires ayant besoin du sommet s du domaine d }
108 std::vector<std::set<int>> procs; //processeurs possedant chaque sommet du groupe...
109 std::set<int> u_procs; //et leur union
110 std::vector<std::array<int, 2>> v_ds; //liste (domaine, sommet)
111 for (i = 0; i + 1 < S_i->getNumberOfTuples(); i++)
112 if ((ns = S_i->getIJ(i + 1, 0) - S_i->getIJ(i, 0)) > 1) //pas besoin de traiter les sommets seuls
113 {
114 for (count++, procs.resize(ns), v_ds.resize(ns), j = S_i->getIJ(i, 0), k = 0; j < S_i->getIJ(i + 1, 0); j++, k++)
115 {
116 for (s = S->getIJ(j, 0), d = 0; s >= off[d + 1]; ) d++; //d : indice du domaine contenant s
117 s -= off[d], v_ds[k] = {{ d, s }}; //stockage du couple (d, s)
118 for (procs[k].clear(), l = 0; l < v_se[d].get_list_size(s); l++) procs[k].insert(v_proc[d](v_se[d](s, l))); //processeurs connectes
119 }
120 for (u_procs.clear(), k = 0; k < ns; k++)
121 for (auto &&pr : procs[k]) u_procs.insert(pr); //union
122 for (k = 0; k < ns; k++) /* on ajoute a chaque sommet les procs auxquels il n'est pas deja connecte */
123 if (procs[k].size() < u_procs.size())
124 {
125 std::set<int>& dest = v_sp[v_ds[k][0]][v_ds[k][1]]; //ou on doit inserer
126 std::set_difference(u_procs.begin(), u_procs.end(), procs[k].begin(), procs[k].end(), std::inserter(dest, dest.end()));
127 }
128 }
129 S_i->decrRef(), S->decrRef();
130 Cerr << count << " sommets trouves!" << finl;
131
132#else
133 Process::exit("Decouper_multi requires MEDCoupling!");
134#endif
135
136 /* ecriture des domaines avec les sommets raccord qu'on a trouves */
137 for (d = 0; d < (int) v_dec.size(); d++)
138 {
139 Static_Int_Lists som_raccord;
140 ArrOfInt sizes(v_dom[d]->nb_som()); //tailles des listes par sommet
141 sizes = 0;
142 for (auto && s_p : v_sp[d]) sizes[s_p.first] = s_p.second.size();
143 som_raccord.set_list_sizes(sizes);
144 for (auto && s_p : v_sp[d])
145 {
146 i = 0;
147 for (auto &&p : s_p.second) som_raccord.set_value(s_p.first, i, p), i++;
148 }
149 v_dec[d]->ecrire(v_proc[d], &som_raccord); //!< * heavier lifting *
150 }
151 return is;
152}
