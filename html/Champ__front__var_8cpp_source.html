77//! Initialisation en debut de calcul.<br>Dimensionne le tableau de valeurs et cree son espace virtuel.<br>Les classes derivees doivent imperativement appeler cette methode.
78/** \return 0 en cas d'erreur, 1 sinon.<br>

81int Champ_front_var::initialiser(double temps, const Champ_Inc_base& inco)
82{
83 const Frontiere& frontiere = frontiere_dis().frontiere();
84 const int n = les_valeurs->nb_cases();
85 const int nbc = nb_comp();
86 for(int i = 0; i < n; i++)
87 {
88 DoubleTab& tab = les_valeurs[i].valeurs();
89// B.M. la methode completer dimensionne parfois le tableau.
90// On a un peut tout et n'importe quoi en entree: parfois tableau vide,
91// parfois nb_dim==1, etc...
92// Attention, les valeurs existantes doivent etre conservees s'il y en a !
93 tab.resize(tab.dimension(0), nbc);
94 frontiere.creer_tableau_faces(tab);
95 tab.echange_espace_virtuel();
96 }
97 return 1;
98}
