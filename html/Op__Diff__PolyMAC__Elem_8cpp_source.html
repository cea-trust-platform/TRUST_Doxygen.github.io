87double Op_Diff_PolyMAC_Elem::calculer_dt_stab() const
88{
89 const Zone_PolyMAC& zone = la_zone_poly_.valeur();
90 const IntTab& e_f = zone.elem_faces();
91 const DoubleTab& nf = zone.face_normales(),
92 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL,
93 &diffu = diffusivite_pour_pas_de_temps().valeurs(), &lambda = diffusivite().valeurs();
94 const DoubleVect& pe = zone.porosite_elem(), &vf = zone.volumes_entrelaces(), &ve = zone.volumes();
95 update_nu();
96
97 int i, e, f, n, N = equation().inconnue().valeurs().dimension(1), cD = diffu.dimension(0) == 1, cL = lambda.dimension(0) == 1;
98 double dt = 1e10;
99 DoubleTrav flux(N);
100 for (e = 0; e < zone.nb_elem(); e++)
101 {
102 for (flux = 0, i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
103 for (n = 0; n < N; n++)
104 flux(n) += zone.nu_dot(&nu_, e, n, &nf(f, 0), &nf(f, 0)) / vf(f);
105 for (n = 0; n < N; n++)
106 if ((!alp || (*alp)(e, n) > 1e-3) && flux(n)) /* sous 0.5e-6, on suppose que l'evanescence fait le job */
107 dt = std::min(dt, pe(e) * ve(e) * (alp ? (*alp)(e, n) : 1) * (lambda(!cL * e, n) / diffu(!cD * e, n)) / flux(n));
108 if (dt < 0) abort();
109 }
110 return Process::mp_min(dt);
111}

138 stencil[i].resize(0, 2), stencil[i].set_smart_resize(1);

148 {
149 zone[0].get().W2(&diffu[0].get(), e, w2); //interpolation : [n_ef.nu grad T]_f = w2_{ff'} (T_f' - T_e)
150// element <-> toutes ses faces (non Dirichlet)
151 if (!aux_only && !semi)
152 for (i = 0; i < w2.dimension(0); i++)
153 if (!semi && fcl[0](f = e_f[0](e, i), 0) < 6)
154 {
155 if (e < zone[0].get().nb_elem())
156 for (n = 0; n < N[0]; n++) stencil[0].append_line(N[0] * e + n, N[0] * (ne_tot[0] + f) + n);
157 if (f < zone[0].get().nb_faces())
158 for (n = 0; n < N[0]; n++) stencil[0].append_line(N[0] * (ne_tot[0] + f) + n, N[0] * e + n);
159 }
160// face <-> face (si les deux sont non Dirichlet)
161 for (i = 0; i < w2.dimension(0); i++)
162 if ((f = e_f[0](e, i)) >= zone[0].get().nb_faces()) continue; //face virtuelle -> rien
163 else if (semi || fcl[0](f = e_f[0](e, i), 0) > 5)
164 for (n = 0; n < N[0]; n++) //Dirichlet ou semi-implicite -> diagonale
165 stencil[0].append_line(N[0] * (!aux_only * ne_tot[0] + f) + n, N[0] * (!aux_only * ne_tot[0] + f) + n);
166 else for (j = 0; j < w2.dimension(1); j++)
167 for (fb = e_f[0](e, j), n = 0; n < N[0]; n++) //cas reel
168 if (fcl[0](fb, 0) < 6 && w2(i, j, n)) stencil[0].append_line(N[0] * (!aux_only * ne_tot[0] + f) + n, N[0] * (!aux_only * ne_tot[0] + fb) + n), tpfa(f, n) &= (i == j);
169 }
170 /* problemes distants : pour les Echange_contact */
171 const Echange_contact_PolyMAC *pcl;
172 if (!semi)
173 for (i = 0; i < cls[0].get().size(); i++)
174 if ((pcl = sub_type(Echange_contact_PolyMAC, cls[0].get()[i].valeur()) ? &ref_cast(Echange_contact_PolyMAC, cls[0].get()[i].valeur()) : NULL))
175 for (pcl->init_f_dist(), j = 0, p = std::find(op_ext.begin(), op_ext.end(), &pcl->o_diff.valeur()) - op_ext.begin(); j < pcl->fvf->nb_faces(); j++)
176 {
177 f = pcl->fvf->num_face(j), o_f = pcl->f_dist(j), o_e = f_e[p](o_f, 0); //faces cote local/distant, elem cote distant
178 zone[p].get().W2(&diffu[p].get(), o_e, w2); //matrice w2 de l'autre cote
179 k = std::find(&e_f[p](o_e, 0), &e_f[p](o_e, 0) + w2.dimension(0), o_f) - &e_f[p](o_e, 0); //indice de o_f dans o_e
180 if (!aux_only)
181 for (n = 0; n < N[0]; n++)
182 for (m = (N[0] == N[p]) * n; m < (N[0] == N[p] ? n + 1 : N[p]); m++)
183 stencil[p].append_line(N[0] * (ne_tot[0] + f) + n, N[p] * o_e + m); //face <-> elem : compo par compo si N[0] == N[p], tout sinon
184 for (l = 0; l < w2.dimension(0); l++)
185 if (fcl[p](fb = e_f[p](o_e, l), 0) < 6)
186 for (n = 0; n < N[0]; n++)
187 for (m = (N[0] == N[p]) * n; m < (N[0] == N[p] ? n + 1 : N[p]); m++)
188 if (w2(k, l, m)) stencil[p].append_line(N[0] * (!aux_only * ne_tot[0] + f) + n, N[p] * (!aux_only * ne_tot[p] + fb) + m);
189 }
190
191 for (i = 0; i < n_ext; i++)
192 if (mat[i])
193 {
194 tableau_trier_retirer_doublons(stencil[i]);
195 Matrice_Morse mat2;
196 Matrix_tools::allocate_morse_matrix(aux_only ? v_part[0][1].size_totale() : inco[0].get().size_totale(), aux_only ? v_part[i][1].size_totale() : inco[i].get().size_totale(), stencil[i], mat2);
197 mat[i]->nb_colonnes() ? *mat[i] += mat2 : *mat[i] = mat2;
198 }
199
200 for (auto &&st : stencil) n_sten += st.dimension(0); //n_sten : nombre total de points du stencil de l'operateur
201 if (!aux_only) Cerr << "width " << Process::mp_sum(n_sten) * 1. / (N[0] * zone[0].get().mdv_elems_faces.valeur().nb_items_seq_tot())
202 << " " << mp_somme_vect(tpfa) * 100. / (N[0] * zone[0].get().md_vector_faces().valeur().nb_items_seq_tot()) << "% TPFA " << finl;
203}

206{

208 const std::string& nom_inco = (le_champ_inco.non_nul() ? le_champ_inco.valeur() : equation().inconnue()).le_nom().getString();
209 int i, j, k1, k2, e, f, fb, n, M, n_ext = op_ext.size(), semi = semi_impl.count(nom_inco), d, D = dimension;

225 fs.push_back(std::ref(zone[i].get().face_surfaces()));
226 xp.push_back(std::ref(zone[i].get().xp())), xv.push_back(std::ref(zone[i].get().xv()));

228 cls.push_back(std::ref(op_ext[i]->equation().zone_Cl_dis().les_conditions_limites()));
229 diffu.push_back(ref_cast(Op_Diff_PolyMAC_Elem, *op_ext[i]).nu());

235 }
236
237 /* que faire avec les variables auxiliaires ? */
238 double t = equation().schema_temps().temps_courant(), dt = equation().schema_temps().pas_de_temps(), fac, prefac;
239 if (aux_only) use_aux_ = 0; /* 1) on est en train d'assembler le systeme de resolution des variables auxiliaires lui-meme */
240 else if (mat[0] && !semi) t_last_aux_ = t + dt, use_aux_ = 0; /* 2) on est en implicite complet : pas besoin de mat_aux / var_aux, on aura les variables a t + dt */
241 else if (t_last_aux_ < t) update_aux(t); /* 3) premier pas a ce temps en semi-implicite : on calcule les variables auxiliaires a t et on les stocke dans var_aux */
242 for (i = 0; i < n_ext; i++) v_aux.push_back(use_aux_ ? ref_cast(Op_Diff_PolyMAC_Elem, *op_ext[i]).var_aux : v_part[i][1]); /* les variables auxiliaires peuvent etre soit dans inco/semi_impl (cas 1), soit dans var_aux (cas 2) */
243
244 DoubleTrav w2, flux(N[0]), acc(N[0]);
245 for (w2.set_smart_resize(1), e = 0; e < ne_tot[0]; e++)
246 {
247 zone[0].get().W2(&diffu[0].get(), e, w2); //interpolation : [n_ef.nu grad T]_f = w2_{ff'} (T_f' - T_e)
248
249 for (i = 0; i < w2.dimension(0); i++) //seconds membres
250 {
251 for (f = e_f[0](e, i), flux = 0, j = 0; j < w2.dimension(1); j++)
252 for (fb = e_f[0](e, j), n = 0; n < N[0]; n++)
253 flux(n) += w2(i, j, n) * (v_aux[0](fb, n) - inco[0](e, n));

261 }
262
263 if (semi || !mat[0]) continue;
264// matrice: elem-elem
265 for (acc = 0, i = 0; i < w2.dimension(0); i++)
266 for (j = 0; j < w2.dimension(1); j++)
267 for (n = 0; n < N[0]; n++) acc(n) += w2(i, j, n);
268 if (!aux_only && e < zone[0].get().nb_elem())
269 for (n = 0; n < N[0]; n++) (*mat[0])(N[0] * e + n, N[0] * e + n) += acc(n);
270// matrice: elem-face

272 for (i = 0; i < w2.dimension(0); i++)
273 if (fcl[0](f = e_f[0](e, i), 0) < 6)
274 {
275 for (acc = 0, j = 0; j < w2.dimension(1); j++)
276 for (n = 0; n < N[0]; n++) acc(n) += w2(i, j, n);
277 if (f < zone[0].get().nb_faces())
278 for (n = 0; n < N[0]; n++) (*mat[0])(N[0] * (ne_tot[0] + f) + n, N[0] * e + n) -= acc(n);
279 if (e < zone[0].get().nb_elem())
280 for (n = 0; n < N[0]; n++) (*mat[0])(N[0] * e + n, N[0] * (ne_tot[0] + f) + n) -= acc(n);
281 }
282// matrice : face-face
283 for (i = 0; i < w2.dimension(0); i++)
284 if (fcl[0](f = e_f[0](e, i), 0) < 6 && f < zone[0].get().nb_faces())
285 for (j = 0; j < w2.dimension(1); j++)
286 if (fcl[0](fb = e_f[0](e, j), 0) < 6)
287 for (n = 0; n < N[0]; n++)
288 if (w2(i, j, n))
289 (*mat[0])(N[0] * (!aux_only * ne_tot[0] + f) + n, N[0] * (!aux_only * ne_tot[0] + fb) + n) += w2(i, j, n);
290 }
291
292// contributions restantes aux equations aux faces
293 for (f = 0; f < zone[0].get().nb_faces(); f++)
294 if (!aux_only && semi && mat[0])
295 for (n = 0; n < N[0]; n++) //semi-implicite : T_f^+ = var_aux
296 secmem(ne_tot[0] + f, n) += v_aux[0](f, n) - (le_champ_inco.non_nul() ? le_champ_inco->valeurs() : equation().inconnue().valeurs())(ne_tot[0] + f, n), (*mat[0])(N[0] * (ne_tot[0] + f) + n, N[0] * (ne_tot[0] + f) + n)++;
297 else if (fcl[0](f, 0) == 0 || fcl[0](f, 0) == 5) continue; //face interne ou Neumann_val_ext -> rien
298 else if (corr[0]) //Pb_Multiphase avec flux parietal -> on ne traite (pour le moment) que Dirichlet et Echange_contact
299 {
300 if (fcl[0](f, 0) == 2) abort(); //Echange_global_impose -> on ne sait pas faire (que vaut T_paroi dans l'element?)
301 const Echange_contact_PolyMAC *ech = fcl[0](f, 0) == 3 ? &ref_cast(Echange_contact_PolyMAC, cls[0].get()[fcl[0](f, 1)].valeur()) : NULL;
302 int o_p = ech ? ech->o_idx : -1, o_f = ech ? ech->f_dist(fcl[0](f, 2)) : -1;
303 const Pb_Multiphase& pbm = ref_cast(Pb_Multiphase, equation().probleme());
304 const DoubleTab& alpha = pbm.eq_masse.inconnue().passe(), &dh = zone[0].get().diametre_hydraulique_elem(), &press = pbm.eq_qdm.pression().passe(),
305 &vit = pbm.eq_qdm.inconnue().passe(), &lambda = pbm.milieu().conductivite().passe(), &mu = ref_cast(Fluide_base, pbm.milieu()).viscosite_dynamique().passe(),
306 &rho = pbm.milieu().masse_volumique().passe(), &Cp = pbm.milieu().capacite_calorifique().passe();
307 DoubleTrav qpk(N[0]), dTf_qpk(N[0], N[0]), dTp_qpk(N[0]), qpi(N[0], N[0]), dTf_qpi(N[0], N[0], N[0]), dTp_qpi(N[0], N[0]), v(N[0], D), nv(N[0]);
308 for (e = f_e[0](f, 0), i = 0; i < e_f[0].get().dimension(1) && (fb = e_f[0](e, i)) >= 0; i++)
