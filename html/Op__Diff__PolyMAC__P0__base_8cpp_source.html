74 int N = eq.inconnue().valeurs().line_size(), N_mil = eq.milieu().masse_volumique().non_nul() ? eq.milieu().masse_volumique().valeurs().line_size() : N;
75 int N_diff = diffusivite().valeurs().line_size(), D = dimension, N_nu = std::max(N * dimension_min_nu(), N_diff);

90int Op_Diff_PolyMAC_P0_base::impr(Sortie& os) const
91{
92 const Zone& ma_zone=la_zone_poly_->zone();
93 const int impr_mom=ma_zone.Moments_a_imprimer();
94 const int impr_sum=(ma_zone.Bords_a_imprimer_sum().est_vide() ? 0:1);
95 const int impr_bord=(ma_zone.Bords_a_imprimer().est_vide() ? 0:1);
96 const Schema_Temps_base& sch = la_zcl_poly_->equation().probleme().schema_temps();
97 DoubleTab& tab_flux_bords= flux_bords();
98 int nb_comp = tab_flux_bords.nb_dim() > 1 ? tab_flux_bords.dimension(1) : 0;
99 DoubleVect bilan(nb_comp);
100 const int nb_faces = la_zone_poly_->nb_faces_tot();
101 DoubleTab xgr(nb_faces,dimension);
102 xgr=0.;
103 if (impr_mom)
104 {
105 const DoubleTab& xgrav = la_zone_poly_->xv();
106 const ArrOfDouble& c_grav=ma_zone.cg_moments();
107 for (int num_face=0; num_face <nb_faces; num_face++)
108 for (int i=0; i<dimension; i++)
109 xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
110 }
111 int k,face;
112 int nb_front_Cl=la_zone_poly_->nb_front_Cl();
113 DoubleTrav flux_bords2( 5, nb_front_Cl , nb_comp) ;
114 flux_bords2=0;
115 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
116 {
117 const Cond_lim& la_cl = la_zcl_poly_->les_conditions_limites(num_cl);
118 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
119 int ndeb = frontiere_dis.num_premiere_face();
120 int nfin = ndeb + frontiere_dis.nb_faces();
121 for (face=ndeb; face<nfin; face++)
122 {
123 for(k=0; k<nb_comp; k++)
124 {
125 flux_bords2(0,num_cl,k)+=tab_flux_bords(face, k);
126 if (ma_zone.Bords_a_imprimer_sum().contient(frontiere_dis.le_nom()))
127 flux_bords2(3,num_cl,k)+=tab_flux_bords(face, k);
128 } /* fin for k */
129 if (impr_mom)
130 {
131 if (dimension==2)
132 {
133 flux_bords2(4,num_cl,0)+=tab_flux_bords(face,1)*xgr(face,0)-tab_flux_bords(face,0)*xgr(face,1);
134 }
135 else
136 {
137 flux_bords2(4,num_cl,0)+=tab_flux_bords(face,2)*xgr(face,1)-tab_flux_bords(face,1)*xgr(face,2);
138 flux_bords2(4,num_cl,1)+=tab_flux_bords(face,0)*xgr(face,2)-tab_flux_bords(face,2)*xgr(face,0);
139 flux_bords2(4,num_cl,2)+=tab_flux_bords(face,1)*xgr(face,0)-tab_flux_bords(face,0)*xgr(face,1);
140 }
141 }
142 } /* fin for face */
143 }
144 mp_sum_for_each_item(flux_bords2);
145
146 if (je_suis_maitre() && nb_comp > 0)
147 {
148// SFichier Flux;
149 if (!Flux.is_open()) ouvrir_fichier(Flux,"",1);
150// SFichier Flux_moment;
151 if (!Flux_moment.is_open()) ouvrir_fichier(Flux_moment,"moment",impr_mom);
152// SFichier Flux_sum;
153 if (!Flux_sum.is_open()) ouvrir_fichier(Flux_sum,"sum",impr_sum);
154 Flux.add_col(sch.temps_courant());
155 if (impr_mom) Flux_moment.add_col(sch.temps_courant());
156 if (impr_sum) Flux_sum.add_col(sch.temps_courant());
157 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
158 {
159 for(k=0; k<nb_comp; k++)
160 {
161 Flux.add_col(flux_bords2(0,num_cl,k));
162 if (impr_sum) Flux_sum.add_col(flux_bords2(3,num_cl,k));
163 bilan(k)+=flux_bords2(0,num_cl,k);
164 }
165 if (dimension==3)
166 {
167 for (k=0; k<nb_comp; k++)
168 if (impr_mom) Flux_moment.add_col(flux_bords2(4,num_cl,k));
169 }
170 else
171 {
172 if (impr_mom) Flux_moment.add_col(flux_bords2(4,num_cl,0));
173 }
174 } /* fin for num_cl */
175 for(k=0; k<nb_comp; k++)
176 Flux.add_col(bilan(k));
177 Flux << finl;
178 if (impr_sum) Flux_sum << finl;
179 if (impr_mom) Flux_moment << finl;
180 }
181 const LIST(Nom)& Liste_Bords_a_imprimer = la_zone_poly_->zone().Bords_a_imprimer();
182 if (!Liste_Bords_a_imprimer.est_vide() && nb_comp > 0)
183 {
184 EcrFicPartage Flux_face;
185 ouvrir_fichier_partage(Flux_face,"",impr_bord);
186 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
187 {
188 const Frontiere_dis_base& la_fr = la_zcl_poly_->les_conditions_limites(num_cl).frontiere_dis();
189 const Cond_lim& la_cl = la_zcl_poly_->les_conditions_limites(num_cl);
190 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
191 int ndeb = frontiere_dis.num_premiere_face();
192 int nfin = ndeb + frontiere_dis.nb_faces();
193 if (ma_zone.Bords_a_imprimer().contient(la_fr.le_nom()))
194 {
195 if(je_suis_maitre())
196 {
197 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps ";
198 sch.imprimer_temps_courant(Flux_face);
199 Flux_face << " : " << finl;
200 }
201 for (face=ndeb; face<nfin; face++)
202 {
203 if (dimension == 2)
204 Flux_face << "# Face a x= " << la_zone_poly_->xv(face,0) << " y= " << la_zone_poly_->xv(face,1) << " : ";
205 else if (dimension == 3)
206 Flux_face << "# Face a x= " << la_zone_poly_->xv(face,0) << " y= " << la_zone_poly_->xv(face,1) << " z= " << la_zone_poly_->xv(face,2) << " : ";
207 for(k=0; k<nb_comp; k++)
208 Flux_face << tab_flux_bords(face, k) << " ";
209 Flux_face << finl;
210 }
211 Flux_face.syncfile();
212 }
213 }
214 }
215 return 1;
216}

249void Op_Diff_PolyMAC_P0_base::update_nu() const
250{
251 const Zone_PolyMAC_P0& zone = la_zone_poly_.valeur();
252 const DoubleTab& nu_src = diffusivite().valeurs();
253 int e, i, m, n, c_nu = nu_src.dimension_tot(0) == 1, d, db, D = dimension;
254 int N = equation().inconnue().valeurs().line_size(), N_mil = equation().milieu().masse_volumique().non_nul() ? equation().milieu().masse_volumique().valeurs().line_size() : N,
255 N_nu = nu_.line_size(), N_nu_src = nu_src.line_size(), mult = N_nu / N;
256 assert(N_nu % N == 0);
257
258 /* nu_ : si necessaire, on doit etendre la champ source */
259 if (N_nu == N && N_nu_src == N_mil)
260 for (e = 0; e < zone.nb_elem_tot(); e++)
261 for (n = 0; n < N; n++) nu_.addr()[N_nu * e + n] = nu_src(!c_nu * e, n); //facile
262 else if (N_nu == N * D && N_nu_src == N_mil)
263 for (e = 0; e < zone.nb_elem_tot(); e++)
264 for (n = 0; n < N; n++)
265 for (d = 0; d < D; d++) //diagonal
266 nu_(e, n, d) = nu_src(!c_nu * e, n);
267 else if (N_nu == N * D * D && (N_nu_src == N_mil || N_nu_src == N_mil * D))
268 for (e = 0; e < zone.nb_elem_tot(); e++)
269 for (n = 0; n < N; n++) //complet
270 for (d = 0; d < D; d++)
271 for (db = 0; db < D; db++) nu_(e, n, d, db) = (d == db) * nu_src(!c_nu * e, N_nu_src == N_mil ? n : D * n + d);
272 else abort();
273
274 /* ponderation de nu par la porosite et par alpha (si pb_Multiphase) */
275 const DoubleTab *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
276 for (e = 0; e < zone.nb_elem_tot(); e++)
277 for (n = 0, i = 0; n < N; n++)
278 for (m = 0; m < mult; m++, i++)
279 nu_.addr()[N_nu * e + i] *= zone.porosite_elem()(e) * (alp ? std::max((*alp)(e, n), 1e-8) : 1);
280
281 /* modification par une classe fille */
282 modifier_nu(nu_);
283
284 nu_a_jour_ = 1;
285}
