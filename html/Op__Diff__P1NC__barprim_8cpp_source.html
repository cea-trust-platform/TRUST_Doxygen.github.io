91double Op_Diff_P1NC_barprim::calculer_dt_stab() const
92{
93 double dt_stab;
94 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
95 double alpha = 1.e-38;
96// pas mp_max ?
97 if (diffusivite_pour_pas_de_temps().valeurs().size() > 0)
98 alpha = local_max_vect(diffusivite_pour_pas_de_temps().valeurs());
99 dt_stab = zone_VEF.carre_pas_du_maillage()/(2.*dimension*(alpha+DMINFLOAT));
100 return dt_stab;
101}

104void Op_Diff_P1NC_barprim::calculer_divergence(const DoubleTab& grad,
105 const DoubleVect& nu,
106 DoubleTab& resu) const
107{
108 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
109 const Zone& zone = zone_VEF.zone();
110 const DoubleTab& face_normales = zone_VEF.face_normales();
111 const IntTab& elem_faces=zone_VEF.elem_faces();
112 const IntTab& face_voisins=zone_VEF.face_voisins();
113 int nfe=zone.nb_faces_elem();
114 int nb_elem_tot=zone.nb_elem_tot();
115 int elem,indice,face,comp,comp2;
116 ArrOfDouble sigma(dimension);
117 double normale;
118 for(elem=0; elem<nb_elem_tot; elem++)
119 {
120 for(indice=0; indice<nfe; indice++)
121 {
122 face = elem_faces(elem,indice);
123 double signe=1;
124 if(elem!=face_voisins(face,0))
125 signe=-1;
126 for(comp2=0; comp2<dimension; comp2++)
127 {
128 normale=signe*face_normales(face,comp2);
129 for(comp=0; comp<dimension; comp++)
130 resu(face, comp)-=nu(elem)*grad(elem,comp,comp2)*normale;
131 }
132 }
133 }
134 {
135 int nb_bords=la_zone_vef->nb_front_Cl();
136 for (int n_bord=0; n_bord<nb_bords; n_bord++)
137 {
138 const Cond_lim& la_cl = la_zcl_vef->les_conditions_limites(n_bord);
139 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
140// Cerr << "la_cl = " << la_cl.valeur() << finl;
141 int num1 = le_bord.num_premiere_face();
142 int num2 = num1 + le_bord.nb_faces();
143 if ((la_cl->que_suis_je() == "Paroi_fixe"))
144 {
145// Cerr << "Paroi_fixe" << finl;
146 for (face=num1; face<num2; face++)
147 {
148 for(comp2=0; comp2<dimension; comp2++)
149 {
150 normale=face_normales(face,comp2);
151 for(comp=0; comp<dimension; comp++)
152 resu(face, comp)=0.;
153 }
154 }
155 }
156 else if (sub_type(Periodique,la_cl.valeur()))
157 {
158 const Periodique& la_cl_perio = ref_cast(Periodique,
159 la_cl.valeur());
160 ArrOfInt fait(le_bord.nb_faces());
161 for (face=num1; face<num2; face++)
162 {
163 if(!fait[face-num1])
164 {
165 int face_associee=num1+la_cl_perio.face_associee(face-num1);
166 fait[face-num1]=(fait[face_associee-num1]=1);
167 for (comp=0; comp<dimension; comp++)
168 resu(face_associee,comp) =
169 (resu(face,comp)+= resu(face_associee,comp));
170 }
171 }
172 }
173 }
174 }
175}

192DoubleTab& Op_Diff_P1NC_barprim::calculer(const DoubleTab& inconnue, DoubleTab& resu) const
193{
194// Cerr << "Op_Diff_P1NC_barprim::calculer" << finl;
195 int nb_elem = la_zone_vef->zone().nb_elem_tot();
196// const DoubleTab& xp=la_zone_vef->xp();
197 DoubleVect nu;
198 {
199 nu.resize(nb_elem);
200 nu=(diffusivite().valeurs())(0,0);
201 }
202 resu=0;
203
204 const Champ_P1NC& ch=ref_cast(Champ_P1NC,(equation().inconnue().valeur()));
205 DoubleTab gradient_bar;
206 DoubleTab ubar(ch.valeurs());
207 ch.filtrer_L2(ubar);
208 calculer_gradientP1NC(ubar, la_zone_vef.valeur(),la_zcl_vef.valeur() ,
209 gradient_bar);
210 DoubleTab resu_bar(resu);
211 calculer_divergence(gradient_bar, nu, resu_bar);
212 ch.filtrer_resu(resu_bar);
213 resu+=resu_bar;
214 resu.echange_espace_virtuel();
215 modifier_flux(*this);
216 return resu;
217}
