375void Echange_contact_Correlation_VEF::calculer_CL()
376{
377 Schema_Comm schema_comm;
378 const Joints& joints = zone_Cl_dis().zone().faces_joint();
379 const int nb_voisins = joints.size();
380 ArrOfInt send_pe_list(nb_voisins);
381 int i;
382 for (i = 0; i < nb_voisins; i++)
383 send_pe_list[i] = joints[i].PEvoisin();
384 schema_comm.set_send_recv_pe_list(send_pe_list, send_pe_list);
385
386 const double ymin = coord(1);
387 const double ymax = coord(N-2);
388 const double Tmin = T(1);
389 const double Tmax = T(N-2);
390
391 T_CL0 = Tinf;
392 T_CL1 = Tsup;
393
394 schema_comm.begin_comm();
395 for (i = 0; i < nb_voisins; i++)
396 {
397 const int pe = send_pe_list[i];
398 schema_comm.send_buffer(pe) << ymin << ymax << Tmin << Tmax;
399 }
400 schema_comm.echange_taille_et_messages();
401 for (i = 0; i < nb_voisins; i++)
402 {
403 const int pe = send_pe_list[i];
404 double ymin_voisin, ymax_voisin, Tmin_voisin, Tmax_voisin;
405 schema_comm.recv_buffer(pe) >> ymin_voisin >> ymax_voisin >> Tmin_voisin >> Tmax_voisin;
406 if (ymax_voisin < ymin)
407 T(0) = T_CL0 = 0.5 * (Tmax_voisin + Tmin);
408 else if (ymin_voisin > ymax)
409 T(N-1) = T_CL1 = 0.5 * (Tmin_voisin + Tmax);
410 else
411 {
412 Cerr << "Erreur dans Echange_contact_Correlation_VEF::calculer_CL() (decoupage incompatible ?)";
413 exit();
414 }
415 }
416 schema_comm.end_comm();

437 */
438double Echange_contact_Correlation_VEF::calculer_coefficient_echange(int i)
439{
440 double Re,Pr;
441 Re = std::fabs(getQh()*getDh(i)/getMu(i));
442 Pr = getMu(i)*getCp()/getLambda(i);
443 fct_Nu.setVar("Re",Re);

445 return fct_Nu.eval()*getLambda(i)/getDh(i);

487
488 const int ndeb = ma_front_vf.num_premiere_face();
489 const int nb_faces_bord = ma_front_vf.nb_faces();
490
491// N entre par l'utilisateur = Nombre de mailles 1D
492 N=N/Process::nproc()+2;
493// N modifie pour que cela soit le nombre de noeuds 1D par processeur
494
495// Ici, on suppose que l'utilisateur a decoupe en tranche selon la direction 1D
496 double delta=(xsup-xinf)/Process::nproc();

498 xsup=xinf+delta;
499// on dimensionne
500 coord.resize(N);
501 U.resize(N);
502 T.resize(N);
503 Qvol.resize(N);
504 rho.resize(N);
505 mu.resize(N);
506 lambda.resize(N);
507 vol.resize(N);
508 diam.resize(N);
509 h_correlation.resize(N);
510
511// on remplit les coordonnees des noeuds du maillages 1D dans le fluide
512 coord(0)=xinf;
513 coord(N-1)=xsup;
514 double dx = (xsup-xinf)/(N-2);
515 for(int i=1; i<N-1; i++)
516 {
517 coord(i) = xinf+dx*(i-0.5);
518 }
519
520// on remplit les correspondances elements solides -> elements fluides
521 correspondance_solide_fluide.resize(nb_faces_bord);
522 for(int i=0; i<nb_faces_bord; i++)
523 {
524 double min=std::fabs(xv(i+ndeb,dir)-coord(1));
525 double tmp=0.;

528 {
529 if ((tmp=std::fabs(xv(i+ndeb,dir)-coord(j))) < min)

531 min = tmp;
532 jmin = j;
533 }
534 }
535// Cerr << "min=" << min << " jmin=" << jmin << finl;
536 correspondance_solide_fluide(i) = jmin;
537 }
538// Cerr << Process::me() << finl;
539// Cerr << correspondance_solide_fluide << finl;
540
541 if (avec_rayo == 1)
542 {
543 const double precision = 1.e-6;
544 const int MAX_FACES_PATCH = 500; //!!! a revoir
545 int nb_patches = 2*(N-2);
546 patches_rayo.resize(nb_patches,MAX_FACES_PATCH);
547 nb_faces_patch.resize(nb_patches);
548 IntVect traite(nb_faces_bord);

580 for (int idim=0; idim<dimension; idim++)
581 {

583 v02[idim] = coord_som(s2,idim)-coord_som(s0,idim);
584 }
585 ArrOfDouble normale(dimension);
586 normale[0] = v01[1]*v02[2]-v01[2]*v02[1];
587 normale[1] = -v01[0]*v02[2]+v01[2]*v02[0];
588 normale[2] = v01[0]*v02[1]-v01[1]*v02[0];
589
590 for(int j=0; j<nb_faces_bord; j++)
591 {
592 if ((traite(j) == 0) && (correspondance_solide_fluide(j) == face_corresp))
593 {
594 int sj_0 = face_sommets(ndeb+j,0);
595 double pscal = 0.;
596 for (int idim=0; idim<dimension; idim++)
597 {
598 pscal += normale[idim]*(coord_som(sj_0,idim)-coord_som(s0,idim));

612 }
613 }
614 }
615 nb_faces_patch(compteur_patches) = compteur_face;
616 compteur_patches++;
617 }
618 }
619
620
621 }
622// calcul du diametre hydraulique
623 for (int i=0; i<N; i++)
624 {

627 }

666 ma(0) = 1.;
667 ma(N-1) = 1.;
668 sm(0) = T_CL0;
669 sm(N-1) = T_CL1;
670 mc(0) = 0.;
671 mb(N-2) = 0.;
672 int i;
673 for (i=1; i<N-1; i++)
674 {
675 const double dtrhoCp = dt/rho(i)/Cp;
676 const double dz1 = coord(i+1)-coord(i);
677 const double dz2 = coord(i)-coord(i-1);
678 const double l1 = 0.5*(lambda(i+1)+lambda(i));
679 const double l2 = 0.5*(lambda(i)+lambda(i-1));
680
681 ma(i)=1+dtrhoCp*(sgn*Cp*debit/dz1+2*(l1/dz1+l2/dz2)/(dz1+dz2));
682
683 sm(i) = T(i)+dtrhoCp*Qvol(i);
684
685 mc(i) = dtrhoCp*(0.5*(1-sgn)*Cp*debit/dz1-2*l1/dz1/(dz1+dz2));
686
687 mb(i-1) = dtrhoCp*(-2*l2/dz2/(dz1+dz2));
688 mb(i-1) += dtrhoCp*(-0.5*(1+sgn)*Cp*debit/dz1);
689 }
690
691
692
693
694 DoubleVect xx(N);
695 xx=0.;
696 Solv_TDMA::resoudre(ma,mb,mc,sm,xx,N);
697 for (i=0; i<N; i++)
698 T(i) = xx(i);
699}
