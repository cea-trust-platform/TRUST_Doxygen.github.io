30double Op_Diff_VDF_Elem_base::calculer_dt_stab() const
31{
32// Calcul du pas de temps de stabilite :
33//
34// - La diffusivite est uniforme donc :
35//
36// dt_stab = 1/(2*diffusivite*Max(coef(elem)))
37//
38// avec:
39// coef = 1/(dx*dx) + 1/(dy*dy) + 1/(dz*dz)
40//
41// i decrivant l'ensemble des elements du maillage
42//
43// Rq : On ne balaie pas l'ensemble des elements puisque
44// le max de coeff est atteint sur l'element qui realise
45// a la fois le min de dx le min de dy et le min de dz
46 double dt_stab = DMAXFLOAT;
47 const Zone_VDF& zone_VDF = iter.zone();
48 const DoubleTab& diffu = has_champ_masse_volumique() ? diffusivite().valeurs() : diffusivite_pour_pas_de_temps().valeurs();
49
50
51 if (sub_type(Champ_Uniforme,diffusivite_pour_pas_de_temps()) && !has_champ_masse_volumique())
52 {
53// GF le max permet de traiter le multi_inco
54 double alpha=max_array(diffu);
55
56 double coef = 1/(zone_VDF.h_x()*zone_VDF.h_x()) + 1/(zone_VDF.h_y()*zone_VDF.h_y());
57
58 if (dimension == 3) coef += 1/(zone_VDF.h_z()*zone_VDF.h_z());
59
60 if (alpha==0) dt_stab = DMAXFLOAT;
61 else dt_stab = 0.5/(alpha*coef);
62 }
63 else
64 {
65 double alpha_loc;
66 const int nb_elem = diffu.dimension(0);
67 for (int elem = 0 ; elem < nb_elem; elem++)
68 {
69 double h = 0;
70 for (int i=0 ; i<dimension; i++)
71 {
72 double l = zone_VDF.dim_elem(elem,i);
73 h += 1./(l*l);
74 }
75 alpha_loc = diffu(elem,0);
76 for (int ncomp=1; ncomp<diffu.dimension(1); ncomp++) alpha_loc = std::max(alpha_loc,diffu(elem,ncomp));
77 if (has_champ_masse_volumique())
78 {
79 const DoubleTab& rho = get_champ_masse_volumique().valeurs();
80 alpha_loc/= rho(elem);
81 }
82 double dt_loc = 0.5/((alpha_loc+DMINFLOAT)*h);
83 if (dt_loc<dt_stab) dt_stab = dt_loc;
84 }
85 }
86
87 return Process::mp_min(dt_stab);
88}

90void Op_Diff_VDF_Elem_base::contribuer_termes_croises(const DoubleTab& inco, const Probleme_base& autre_pb, const DoubleTab& autre_inco, Matrice_Morse& matrice) const
91{
92 const Zone_VDF& zone = iter.zone();
93 const IntTab& f_e = zone.face_voisins();
94 const Zone_Cl_VDF& zcl = iter.zone_Cl();
95 int l;
96
97// boucle sur les cl pour trouver un paroi_contact
98 for (int i = 0; i < zone.nb_front_Cl(); i++)
99 {
100 const Cond_lim& la_cl = zcl.les_conditions_limites(i);
101 if (!la_cl.valeur().que_suis_je().debute_par("Paroi_Echange_contact")) continue; //pas un Echange_contact
102 const Echange_contact_VDF& cl = ref_cast(Echange_contact_VDF, la_cl.valeur());
103 if (cl.nom_autre_pb() != autre_pb.le_nom()) continue; //not our problem
104
105 std::map<int, std::pair<int, int>> f2e;
106 const Front_VF& fvf = ref_cast(Front_VF, cl.frontiere_dis());
107 for (int j = 0; j < cl.item.dimension(0); j++)
108 if ((l = cl.item(j)) >= 0)
109 {
110 int f = fvf.num_face(j);
111 int e = f_e(f, 0) == -1 ? f_e(f, 1) : f_e(f, 0);
112 f2e[f] = std::make_pair(e, l);
113 }
114 iter.ajouter_contribution_autre_pb(inco, matrice, la_cl, f2e);
115 }

118void Op_Diff_VDF_Elem_base::dimensionner_termes_croises(Matrice_Morse& matrice, const Probleme_base& autre_pb, int nl, int nc) const
119{
120 const Champ_P0_VDF& ch = ref_cast(Champ_P0_VDF, equation().inconnue().valeur());
121 const Zone_VDF& zone = iter.zone();
122 const IntTab& f_e = zone.face_voisins();
123 const Conds_lim& cls = iter.zone_Cl().les_conditions_limites();
124 int i, j, l, f, n, N = ch.valeurs().line_size();
125
126 IntTab stencil(0, 2);
127 stencil.set_smart_resize(1);
128 for (i = 0; i < cls.size(); i++)
129 if (sub_type(Echange_contact_VDF, cls[i].valeur()))
130 {
131 const Echange_contact_VDF& cl = ref_cast(Echange_contact_VDF, cls[i].valeur());
132 if (cl.nom_autre_pb() != autre_pb.le_nom()) continue; //not our problem
133
134 /* stencil */
135 const Front_VF& fvf = ref_cast(Front_VF, cl.frontiere_dis());
136 for (j = 0; j < cl.item.dimension(0); j++)
137 if ((l = cl.item(j)) >= 0)
138 {
139 f = fvf.num_face(j);
140 int e = f_e(f, 0) == -1 ? f_e(f, 1) : f_e(f, 0);
141 for (n = 0; n < N; n++) stencil.append_line(N * e + n, N * l + n);
142 }
143 }
144
145 tableau_trier_retirer_doublons(stencil);
146 Matrix_tools::allocate_morse_matrix(nl, nc, stencil, matrice);
147}

172 mat[i]->nb_colonnes() ? *mat[i] += mat2 : *mat[i] = mat2;

176void Op_Diff_VDF_Elem_base::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
177{
178 statistiques().begin_count(diffusion_counter_);
179 if (!op_ext_init_) init_op_ext();
180
181 const std::string& nom_inco = equation().inconnue().le_nom().getString();
182 int n_ext = op_ext.size(); //pour la thermique monolithique
183 std::vector<Matrice_Morse *> mat(n_ext);
184 std::vector<const DoubleTab *> inco(n_ext); //inconnues
185
186 mat[0] = matrices.count(nom_inco) ? matrices.at(nom_inco) : NULL;
187 inco[0] = semi_impl.count(nom_inco) ? &semi_impl.at(nom_inco) : &op_ext[0]->equation().inconnue().valeur().valeurs();
188
189 if(mat[0]) iter.ajouter_contribution(*inco[0], *mat[0]);
190 iter.ajouter(*inco[0],secmem);
191
192 if (equation().domaine_application() == Motcle("Hydraulique"))
193// On est dans le cas des equations de Navier_Stokes
194 {
195// Ajout du terme supplementaire en V/(R*R) dans le cas des coordonnees axisymetriques
196 if(Objet_U::bidim_axi == 1)
197 {
198 const Zone_VDF& zvdf=iter.zone();
199 const DoubleTab& xv=zvdf.xv();
200 const IntVect& ori=zvdf.orientation();
201 const IntTab& face_voisins=zvdf.face_voisins();
202 const DoubleVect& volumes_entrelaces=zvdf.volumes_entrelaces();
203 int face, nb_faces=zvdf.nb_faces();//, cst;
204 DoubleTrav diffu_tot(zvdf.nb_elem_tot());
205 double db_diffusivite;
206 Nom nom_eq=equation().que_suis_je();
207 if ((nom_eq == "Navier_Stokes_standard")||(nom_eq == "Navier_Stokes_QC")||(nom_eq == "Navier_Stokes_FT_Disc"))
208 {
209 const Eval_Diff_VDF& eval=dynamic_cast<const Eval_Diff_VDF&> (iter.evaluateur());
210 const Champ_base& ch_diff=eval.get_diffusivite();
211 const DoubleTab& tab_diffusivite=ch_diff.valeurs();
212 if (tab_diffusivite.size() == 1)
213 diffu_tot = tab_diffusivite(0,0);
214 else
215 diffu_tot = tab_diffusivite;
216
217 for(face=0; face<nb_faces; face++)
218 if(ori(face)==0)
219 {
220 int elem1=face_voisins(face,0);
221 int elem2=face_voisins(face,1);
222 if(elem1==-1)
223 db_diffusivite=diffu_tot(elem2);
224 else if (elem2==-1)
225 db_diffusivite=diffu_tot(elem1);
226 else
227 db_diffusivite=0.5*(diffu_tot(elem2)+diffu_tot(elem1));
228 double r= xv(face,0);
229 if(r>=1.e-24)
230 {
231 if(mat[0]) (*mat[0])(face,face)+=db_diffusivite*volumes_entrelaces(face)/(r*r);
232 secmem(face) -=(*inco[0])(face)*db_diffusivite*volumes_entrelaces(face)/(r*r);
233
234 }
235 }
236 }
237 else if (equation().que_suis_je() == "Navier_Stokes_Interface_avec_trans_masse" ||
238 equation().que_suis_je() == "Navier_Stokes_Interface_sans_trans_masse" ||
239 equation().que_suis_je() == "Navier_Stokes_Front_Tracking" ||
240 equation().que_suis_je() == "Navier_Stokes_Front_Tracking_BMOL")
241 {
242// Voir le terme source axi dans Interfaces/VDF
243 }
244 else
245 {
246 Cerr << "Probleme dans Op_Diff_VDF_base::contribuer_a_avec avec le type de l'equation" << finl;
247 Cerr << "on n'a pas prevu d'autre cas que Navier_Stokes_std" << finl;
248 exit();
249 }
250 }
251 }
252
253 for (int i = 1; i < n_ext; i++)
254 {
255 std::string nom_mat = nom_inco + "/" + op_ext[i]->equation().probleme().le_nom().getString();
256 mat[i] = matrices.count(nom_mat) ? matrices.at(nom_mat) : NULL;
257 if(mat[i])
258 {
259 inco[i] = semi_impl.count(nom_mat) ? &semi_impl.at(nom_mat) : &op_ext[i]->equation().inconnue().valeur().valeurs();
260 contribuer_termes_croises(*inco[i], op_ext[i]->equation().probleme(), op_ext[i]->equation().inconnue().valeurs(), *mat[i]);
261 }
262 }
263 statistiques().end_count(diffusion_counter_);
264
265}
