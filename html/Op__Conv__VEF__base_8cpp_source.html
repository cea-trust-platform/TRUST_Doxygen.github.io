67double Op_Conv_VEF_base::calculer_dt_stab() const
68{
69 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
70 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
71 const DoubleVect& volumes_entrelaces = zone_VEF.volumes_entrelaces();
72 const DoubleVect& volumes_entrelaces_Cl = zone_Cl_VEF.volumes_entrelaces_Cl();
73 remplir_fluent(fluent);
74 if (vitesse().le_nom()=="rho_u" && equation().probleme().is_dilatable())
75 diviser_par_rho_si_dilatable(fluent,equation().milieu());
76
77 double dt_face,dt_stab =1.e30;
78
79// On traite les conditions aux limites
80// Si une face porte une condition de Dirichlet on n'en tient pas compte
81// dans le calcul de dt_stab
82 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
83 {
84 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
85 if ( (sub_type(Dirichlet,la_cl.valeur()))
86 ||
87 (sub_type(Dirichlet_homogene,la_cl.valeur()))
88 )
89 ;
90 else
91 {
92 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
93 int ndeb = le_bord.num_premiere_face();
94 int nfin = ndeb + le_bord.nb_faces();
95 for (int num_face=ndeb; num_face<nfin; num_face++)
96 {
97 dt_face = volumes_entrelaces_Cl(num_face)/(fluent[num_face]+1.e-30);
98 dt_stab = (dt_face < dt_stab) ? dt_face : dt_stab;
99 }
100 }
101 }
102
103// On traite les faces internes non standard
104 int ndeb = zone_VEF.premiere_face_int();
105 int nfin = zone_VEF.premiere_face_std();
106
107 for (int num_face=ndeb; num_face<nfin; num_face++)
108 {
109 dt_face = volumes_entrelaces_Cl(num_face)/(fluent[num_face]+DMINFLOAT);
110 dt_stab =(dt_face < dt_stab) ? dt_face : dt_stab;
111 }
112
113// On traite les faces internes standard
114 ndeb = nfin;
115 nfin = zone_VEF.nb_faces();
116 for (int num_face=ndeb; num_face<nfin; num_face++)
117 {
118 dt_face = volumes_entrelaces(num_face)/(fluent[num_face]+DMINFLOAT);
119 dt_stab =(dt_face < dt_stab) ? dt_face : dt_stab;
120 }
121
122 dt_stab = Process::mp_min(dt_stab);
123// astuce pour contourner le type const de la methode
124 Op_Conv_VEF_base& op = ref_cast_non_const(Op_Conv_VEF_base,*this);
125 op.fixer_dt_stab_conv(dt_stab);
126 if (vitesse().le_nom()=="rho_u" && equation().probleme().is_dilatable())
127 multiplier_par_rho_si_dilatable(fluent,equation().milieu());
128
129 return dt_stab;
130}

153// dans le calcul de dt_stab

170 es_valeurs(num_face) = dt_face;
171 }
172 }
173 }

176 int ndeb = zone_VEF.premiere_face_int();
177 int nfin = zone_VEF.premiere_face_std();
178
179 for (int num_face=ndeb; num_face<nfin; num_face++)
180 {

265//Description
266// Calculation of local time: Vect of size number of faces of the domain
267// This is the equivalent of "Op_Conv_VEF_base :: calculer_dt_stab ()"
268void Op_Conv_VEF_base::calculer_dt_local(DoubleTab& dt_face) const
269{
270 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
271 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
272 const DoubleVect& volumes_entrelaces = zone_VEF.volumes_entrelaces();
273 const DoubleVect& volumes_entrelaces_Cl = zone_Cl_VEF.volumes_entrelaces_Cl();
274
275 int nb_faces= zone_VEF.nb_faces();
276 dt_face=(volumes_entrelaces);
277 remplir_fluent(fluent);
278
279 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
280 {
281 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
282 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
283 int ndeb = le_bord.num_premiere_face();
284 int nfin = ndeb + le_bord.nb_faces();
285 for (int num_face=ndeb; num_face<nfin; num_face++)
286 {
287 if( sup_strict(fluent[num_face], 1.e-30) )
288 dt_face(num_face)= volumes_entrelaces_Cl(num_face)/fluent[num_face];
289 else
290 dt_face(num_face) = -1.;
291 }
292 }
293
294 //Non-standard internal faces
295 int ndeb = zone_VEF.premiere_face_int();
296 int nfin = zone_VEF.premiere_face_std();
297
298 for (int num_face=ndeb; num_face<nfin; num_face++)
299 {
300 if( sup_strict(fluent[num_face], 1.e-30) )
301 dt_face(num_face)= volumes_entrelaces(num_face)/fluent[num_face];
302 else
303 dt_face(num_face) = -1.;
304 }
305
306 //The standard internal faces
307 ndeb = nfin;
308 nfin = zone_VEF.nb_faces();
309 for (int num_face=ndeb; num_face<nfin; num_face++)
310 {
311 if( sup_strict(fluent[num_face], 1.e-30) )
312 dt_face(num_face)= volumes_entrelaces(num_face)/fluent[num_face];
313 else
314 dt_face(num_face) = -1.;
315 }
316
317 double max_dt_local= dt_face.mp_max_abs_vect();
318 for(int i=0; i<nb_faces; i++)
319 {
320 if(! sup_strict(dt_face(i), 1.e-16))
321 dt_face(i) = max_dt_local;
322 }
323 dt_face.echange_espace_virtuel();
324
325 for (int n_bord=0; n_bord<zone_VEF.nb_front_Cl(); n_bord++)
326 {
327 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
328 if (sub_type(Periodique,la_cl.valeur()))
329 {
330 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
331 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
332 int nb_faces_bord=le_bord.nb_faces();
333 for (int ind_face=0; ind_face<nb_faces_bord; ind_face++)
334 {
335 int ind_face_associee = la_cl_perio.face_associee(ind_face);
336 int face = le_bord.num_face(ind_face);
337 int face_associee = le_bord.num_face(ind_face_associee);
338 if (!est_egal(dt_face(face),dt_face(face_associee),1.e-8))
339 {
340 dt_face(face) = std::min(dt_face(face),dt_face(face_associee));
341 }
342 }
343 }
344 }
345 dt_face.echange_espace_virtuel();
346
347// dt_conv_locaux=dt_face;
348}
