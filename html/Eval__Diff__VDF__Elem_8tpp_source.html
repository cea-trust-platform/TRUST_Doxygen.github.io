51{
52 const int i = elem_(face,0), j = elem_(face,1), ncomp = flux.size_array();
53 const double d0 = la_zone->dist_face_elem0_period(face,i,la_cl.distance()), d1 = la_zone->dist_face_elem1_period(face,j,la_cl.distance());
54
55 for (int k = 0; k < ncomp; k++)
56 {
57 const int ori = DERIVED_T::IS_ANISO ? orientation(face) : k;
58 const double heq = compute_heq(d0,i,d1,j,ori);
59
60 flux[k] = DERIVED_T::IS_QUASI ? heq*(inco(j,k)/dv_mvol(j) - inco(i,k)/dv_mvol(i))*surface(face)*porosite(face) : heq*(inco(j,k) - inco(i,k))*surface(face)*porosite(face);

91template <typename DERIVED_T> template <typename Type_Double>
92inline void Eval_Diff_VDF_Elem<DERIVED_T>::flux_face(const DoubleTab& inco, const int boundary_index, const int face, const int local_face, const Echange_externe_impose& la_cl, const int num1, Type_Double& flux) const
93{
94 if (DERIVED_T::IS_QUASI) not_implemented_k_eps(__func__);
95
96// C.L de type Echange_externe_impose : 1/h_total = (1/h_imp) + (e/diffusivite) : La C.L fournit h_imp ; il faut calculer e/diffusivite
97 double heq, h_total_inv, e;
98 const int i = elem_(face,0), j = elem_(face,1), ncomp = flux.size_array();
99
100 if (DERIVED_T::IS_MODIF_DEQ) e = ind_Fluctu_Term()==1 ? Dist_norm_bord_externe_(face) : equivalent_distance(boundary_index,local_face);
101 else e = DERIVED_T::IS_DEQUIV ? equivalent_distance(boundary_index,local_face) : Dist_norm_bord_externe_(face);
102
103 int elem_opp = -1;
104 if (sub_type(Echange_interne_impose, la_cl))
105 {
106// In this case, T_ext might change within the timestep (e.g. Newton's algorithm used for the
107// variable gap model). Make sure we get the most up-to-date value.
108 const Echange_interne_impose& cl = ref_cast(Echange_interne_impose, la_cl);
109 const Champ_front_calc_interne& Text = ref_cast(Champ_front_calc_interne, cl.T_ext().valeur());
110 const IntTab& fmap = Text.face_map();
111 int opp_face = fmap[face-num1]+num1; //!< num1 is the index of the first face
112 int elem1 = la_zone->face_voisins(opp_face, 0);
113 elem_opp = (elem1 != -1) ? elem1 : la_zone->face_voisins(opp_face, 1);
114 }
115
116// XXX : E. Saikali 08/03/2021 : The test of a zero diffusion was not done before. I think it should be like that
117// Attention for DIFT OPERATORS : nu_2 and not nu_1 (only laminar part)
118 if ( i != -1 )
119 {
120 for (int k = 0; k < ncomp; k++)
121 {
122 const int ori = DERIVED_T::IS_ANISO ? orientation(face) : k;
123 const double h_imp = la_cl.h_imp(face-num1,k), T_ext = (elem_opp == -1) ? la_cl.T_ext(face-num1,k) : inco(elem_opp,k);
124 if ( nu_2(i,ori) == 0.0 ) heq = 0.0;
125 else
126 {
127 h_total_inv = 1.0/h_imp + e/nu_2(i,ori);
128 heq = 1.0 / h_total_inv;
129 }
130 flux[k] = heq * (T_ext-inco(i,k))*surface(face);
131 }
132 }
133 else //!< j != -1
134 {
135 for (int k = 0; k < ncomp; k++)
136 {
137 const int ori = DERIVED_T::IS_ANISO ? orientation(face) : k;
138 const double h_imp = la_cl.h_imp(face-num1,k), T_ext = (elem_opp == -1) ? la_cl.T_ext(face-num1,k) : inco(elem_opp,k);
139 if ( nu_2(j,ori) == 0.0 ) heq = 0.0;
140 else
141 {
142 h_total_inv = 1.0/h_imp + e/nu_2(j,ori);
143 heq = 1.0 / h_total_inv;
144 }
145 flux[k] = heq*(inco(j,k)-T_ext)*surface(face);
146 }
147 }

150template <typename DERIVED_T> template <typename Type_Double>
151inline void Eval_Diff_VDF_Elem<DERIVED_T>::flux_faces_interne(const DoubleTab& inco, const int face, Type_Double& flux) const
152{
153 const int i = elem_(face,0), j = elem_(face,1), ncomp = flux.size_array();
154 double heq, d0 = Dist_face_elem0(face,i), d1 = Dist_face_elem1(face,j);
155 for (int k = 0; k < ncomp; k++)
156 {
157 const int ori = DERIVED_T::IS_ANISO ? orientation(face) : k;
158 if (DERIVED_T::IS_RANS)
159 {
160 heq = compute_heq(d0,i, d1,j,ori); //!< pas d'assert pour k-eps !
161 flux[k] = DERIVED_T::IS_QUASI ? heq*(inco(j,k)/dv_mvol(j) - inco(i,k)/dv_mvol(i))*surface(face)*porosite(face) : heq*(inco(j,k)-inco(i,k))*surface(face)*porosite(face);
162 }
163 else
164 {
165 if (nu_1(i,ori) == 0.0 && nu_1(j,ori) == 0.0) heq = 0.;
166 else
167 {
168 assert(nu_1(i,ori) != 0.0 && nu_1(j,ori) != 0.0);
169 heq = compute_heq(d0,i, d1,j,ori);
170 }
171 flux[k] = heq*(inco(j,k)-inco(i,k))*surface(face)*porosite(face);
172 }
173 }
