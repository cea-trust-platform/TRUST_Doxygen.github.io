93DoubleTab& Convection_Diffusion_Espece_Multi_QC::derivee_en_temps_inco(DoubleTab& derivee)
94{
95 if (schema_temps().diffusion_implicite())
96 {
97// on ne peut pas pour l'instanrt aller plus vite que la CFL
98 double dt=le_schema_en_temps->pas_de_temps();
99 double dt_op = operateur(1).calculer_pas_de_temps();
100 if(inf_strict(dt_op,dt))
101 {
102 Cerr<<"diffusion_implicite cannot be achieved for the case dt > dt_op_conv"<<finl;
103 Process::exit();
104 }
105 }
106
107 DoubleTrav secmem(derivee);
108
109// on commence par retirer phi*div(1 U)
110 const DoubleTab& frac_mass = inconnue().valeurs();
111
112 int n = frac_mass.dimension_tot(0);
113 DoubleTrav unite(frac_mass);
114 unite=1;
115
116// on change temporairement la zone_cl
117 operateur(1).l_op_base().associer_zone_cl_dis(zcl_modif_.valeur());
118 operateur(1).ajouter(unite,secmem);
119 operateur(1).l_op_base().associer_zone_cl_dis(zone_Cl_dis().valeur());
120
121 for (int i=0; i<n; i++)
122 secmem(i)=-secmem(i)*frac_mass(i);
123
124// suite + standard
125 operateur(1).ajouter(secmem);
126 les_sources.ajouter(secmem);
127 solveur_masse.appliquer(secmem);
128
129 solveur_masse->set_name_of_coefficient_temporel("masse_volumique");
130 if (schema_temps().diffusion_implicite())
131 {
132 derivee=inconnue().valeurs();
133 Equation_base::Gradient_conjugue_diff_impl(secmem, derivee);
134 }
135 else
136 {
137 DoubleTrav secmem_bis(secmem);
138 operateur(0).ajouter(secmem_bis);
139 solveur_masse.appliquer(secmem_bis);
140 secmem += secmem_bis;
141 derivee = secmem;
142 }
143 solveur_masse->set_name_of_coefficient_temporel("no_coeff");
144 return derivee;
145}

237void Convection_Diffusion_Espece_Multi_QC::assembler_blocs_avec_inertie(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl)
238{
239 statistiques().begin_count(assemblage_sys_counter_);
240 const std::string& nom_inco = inconnue().le_nom().getString();
241 const DoubleTab& inco = inconnue().valeurs();
242 Matrice_Morse *mat = matrices.count(nom_inco)?matrices.at(nom_inco):NULL;
243
244 secmem=0;
245 const IntVect& tab1= mat->get_tab1();
246
247 DoubleVect& coeff=mat->get_set_coeff();
248
249 const DoubleTab& rho=get_champ("masse_volumique").valeurs();
250 operateur(0).l_op_base().ajouter_blocs(matrices, secmem, semi_impl);
251
252 int ndl=rho.dimension(0);
253// on divise par rho chaque ligne
254 for (int som=0; som<ndl; som++)
255 {
256 double inv_rho=1/rho(som);
257 for (int k=tab1(som)-1; k<tab1(som+1)-1; k++)
258 coeff(k)*=inv_rho;
259 secmem(som)*=inv_rho;
260 }
261
262// ajout de la convection
263 operateur(1).l_op_base().ajouter_blocs(matrices, secmem, semi_impl);
264
265// on retire Divu1 *inco
266 DoubleTrav unite(inco),divu1(inco);
267 unite=1;
268
269 {
270// on change temporairement la zone_cl
271 operateur(1).l_op_base().associer_zone_cl_dis(zcl_modif_.valeur());
272 operateur(1).ajouter(unite,divu1);
273 operateur(1).l_op_base().associer_zone_cl_dis(zone_Cl_dis().valeur());
274 }
275
276 for (int i=0; i<ndl; i++)
277 {
278 secmem(i)-=divu1(i)*inco(i);
279 if(mat) (*mat)(i,i)+=divu1(i);
280 }
281 statistiques().end_count(assemblage_sys_counter_);
282
283 statistiques().begin_count(source_counter_);
284 for (int i = 0; i < sources().size(); i++)
285 sources()(i).valeur().ajouter_blocs(matrices, secmem, semi_impl);
286 statistiques().end_count(source_counter_);
287
288 statistiques().begin_count(assemblage_sys_counter_);
289 if(mat) mat->ajouter_multvect(inco,secmem);
290
291 schema_temps().ajouter_blocs(matrices, secmem, *this);
292
293 if (!discretisation().que_suis_je().debute_par("PolyMAC"))
294 modifier_pour_Cl(*mat,secmem);
295
296 statistiques().end_count(assemblage_sys_counter_);
297
298}
