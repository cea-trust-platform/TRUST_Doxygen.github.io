215void Op_Dift_Stab_VEF_Face::modifie_pour_Cl(const DoubleTab& inconnue, DoubleTab& resu) const
216{
217 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
218 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
219 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
220 const DoubleVect& inconnueVect = inconnue;
221 DoubleVect& resuVect = resu;
222 const int nb_bords=les_cl.size();
223 const int nb_comp=resu.line_size();
224
225 DoubleTab& tab_flux_bords = flux_bords_;
226 tab_flux_bords.resize(zone_VEF.nb_faces_bord(),nb_comp);
227 tab_flux_bords=0.;
228
229 int num1=0, num2=0, n_bord=0;
230 int face=0, face_associee=0;
231 int ind_face=0, dim=0;
232 double surface=0., flux=0.;
233
234 for (n_bord=0; n_bord<nb_bords; n_bord++)
235 {
236 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
237 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
238
239 num1 = 0;
240 num2 = le_bord.nb_faces();
241
242 if (sub_type(Periodique,la_cl.valeur()))
243 {
244 const Periodique& la_cl_perio = ref_cast(Periodique,la_cl.valeur());
245
246 for (ind_face=num1; ind_face<num2; ind_face++)
247 {
248 face=le_bord.num_face(ind_face);
249 face_associee=le_bord.num_face(la_cl_perio.face_associee(ind_face));
250
251 if (face<face_associee)
252 for (dim=0; dim<nb_comp; dim++)
253 {
254 resuVect[face*nb_comp+dim]+=resuVect[face_associee*nb_comp+dim];
255 resuVect[face_associee*nb_comp+dim]=resuVect[face*nb_comp+dim];
256 }
257 }
258 }
259 if (sub_type(Neumann_paroi,la_cl.valeur()))
260 {
261 const Neumann_paroi& la_cl_paroi = ref_cast(Neumann_paroi,la_cl.valeur());
262
263 for (ind_face=num1; ind_face<num2; ind_face++)
264 {
265 face=le_bord.num_face(ind_face);
266 surface=zone_VEF.surface(face);
267
268 for (dim=0; dim<nb_comp; dim++)
269 {
270 flux=la_cl_paroi.flux_impose(ind_face,dim)*surface;
271 resuVect[face*nb_comp+dim]-=flux;
272 tab_flux_bords(face,dim)=flux;
273 }
274 }
275 }
276 if (sub_type(Echange_externe_impose,la_cl.valeur()))
277 {
278 const Echange_externe_impose& la_cl_paroi = ref_cast(Echange_externe_impose,la_cl.valeur());
279
280 for (ind_face=num1; ind_face<num2; ind_face++)
281 {
282 face=le_bord.num_face(ind_face);
283 surface=zone_VEF.surface(face);
284
285 for (dim=0; dim<nb_comp; dim++)
286 {
287 flux=la_cl_paroi.h_imp(ind_face,dim)*(la_cl_paroi.T_ext(ind_face,dim)-inconnueVect[face*nb_comp+dim])*surface;
288 resuVect[face*nb_comp+dim]-=flux;
289 tab_flux_bords(face,dim)=flux;
290 }
291 }
292 }
293 if (sub_type(Neumann_homogene,la_cl.valeur())
294 || sub_type(Symetrie,la_cl.valeur())
295 || sub_type(Neumann_sortie_libre,la_cl.valeur()))
296 {
297 for (ind_face=num1; ind_face<num2; ind_face++)
298 {
299 face=le_bord.num_face(ind_face);
300
301 for (dim=0; dim<nb_comp; dim++)
302 tab_flux_bords(face,dim)=0.;
303 }
304 }
305
306 }

309void Op_Dift_Stab_VEF_Face::ajouter_operateur_centre(const DoubleTab& Aij, const DoubleTab& inconnueTab, DoubleTab& resuTab) const
310{
311 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
312
313 const int nb_elem_tot=zone_VEF.nb_elem_tot();
314 const int nb_faces_elem=zone_VEF.zone().nb_faces_elem();
315 const int nb_comp=resuTab.line_size();
316 const DoubleVect& inconnue = inconnueTab;
317 DoubleVect& resu = resuTab;
318 int elem=0, facei_loc=0,facei=0;
319 int facej_loc=0,facej=0, dim=0;
320 double inc_i=0., inc_j=0., delta_ij=0.;
321
322 const IntTab& elem_faces=zone_VEF.elem_faces();
323
324 for (elem=0; elem<nb_elem_tot; elem++)
325 for (facei_loc=0; facei_loc<nb_faces_elem; facei_loc++)
326 {
327 facei=elem_faces(elem,facei_loc);
328
329 for (facej_loc=facei_loc+1; facej_loc<nb_faces_elem; facej_loc++)
330 {
331 facej=elem_faces(elem,facej_loc);
332
333 const double aij=Aij(elem,facei_loc,facej_loc);
334
335 for (dim=0; dim<nb_comp; dim++)
336 {
337 inc_i=inconnue[facei*nb_comp+dim];
338 inc_j=inconnue[facej*nb_comp+dim];
339 delta_ij=aij*(inc_j-inc_i);
340
341 resu[facei*nb_comp+dim]+=delta_ij;
342 resu[facej*nb_comp+dim]-=delta_ij;
343 }
344 }
