230double Op_Diff_VEFP1NCP1B_Face::calculer_dt_stab() const
231{
232 const Zone_VEF_PreP1b& zone_VEF=zone_VEFPreP1B();
233 const Zone& zone=zone_VEF.zone();
234 const Zone_Cl_VEF& zone_Cl_VEF=la_zcl_vef.valeur();
235 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
236
237 const int nb_faces=zone_VEF.nb_faces();
238 const int nb_faces_tot=zone_VEF.nb_faces_tot();
239
240 const DoubleVect& volumes_entrelaces=zone_VEF.volumes_entrelaces();
241 const DoubleVect& porosite_elem = zone_VEF.porosite_elem();
242
243 DoubleTab coeffOperateur(nb_faces_tot);
244 coeffOperateur=0.;
245
246 DoubleTab nu;
247
248 const int nb_bords=les_cl.size();
249 const int marq = phi_psi_diffuse(equation());
250
251 int face=0;
252 int ind_face=0;
253 int num1=0,num2=0;
254 int n_bord=0;
255
256 double dt_stab=DMAXFLOAT;
257
258// Calcul de la porosite
259 remplir_nu(nu_);
260 modif_par_porosite_si_flag(nu_,nu,!marq,porosite_elem);
261
262// Calcul : contribution des parties P0, P1 et Pa au dt_stab
263 if (alphaE) calculer_dt_stab_elem(nu,coeffOperateur);
264 if (alphaS)
265 {
266 DoubleTab nu_p1;
267 zone.domaine().creer_tableau_sommets(nu_p1);
268 remplir_nu_p1(nu,nu_p1);
269 calculer_dt_stab_som(nu_p1,coeffOperateur);
270 }
271 if (alphaA)
272 {
273 DoubleTab nu_pA;
274 zone_VEF.creer_tableau_aretes(nu_pA);
275 remplir_nu_pA(nu,nu_pA);
276 calculer_dt_stab_aretes(nu_pA,coeffOperateur);
277 }
278// Calcul : modification pour tenir compte de la matrice de masse
279 for (face=0; face<nb_faces; face++)
280 {
281 coeffOperateur(face)/=volumes_entrelaces(face);
282 assert(coeffOperateur(face)>=0.);
283 coeffOperateur(face)=1./(coeffOperateur(face)+DMINFLOAT);
284 }
285
286// Calcul : modification pour les faces de Dirichlet
287 for (n_bord=0; n_bord<nb_bords; n_bord++)
288 {
289 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
290 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
291
292 num1=0;
293 num2=le_bord.nb_faces();
294
295 if (sub_type(Dirichlet_homogene,la_cl.valeur()) ||
296 sub_type(Dirichlet,la_cl.valeur())
297 )
298 for (ind_face=num1; ind_face<num2; ind_face++)
299 {
300 face=le_bord.num_face(ind_face);
301 coeffOperateur(face)=1.e20;
302 }
303 }
304
305// Calcul du pas de temps de stabilite
306// : on en a besoin que sur les faces reelles
307 for (face=0; face<nb_faces; face++)
308 if (coeffOperateur(face)<dt_stab)
309 dt_stab=coeffOperateur(face);
310
311 dt_stab=Process::mp_min(dt_stab);
312 return dt_stab;
313}

315void Op_Diff_VEFP1NCP1B_Face::
316calculer_dt_stab_elem(const DoubleTab& nu, DoubleTab& coeffOperateur) const
317{
318 const Zone_VEF_PreP1b& zone_VEF=zone_VEFPreP1B();
319 const Zone& zone=zone_VEF.zone();
320 const Zone_Cl_VEF& zone_Cl_VEF=la_zcl_vef.valeur();
321 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
322
323 const DoubleTab& face_normales=zone_VEF.face_normales();
324 const DoubleVect& volumes=zone_VEF.volumes();
325
326 const IntTab& elem_faces=zone_VEF.elem_faces();
327
328 const int nb_elem_tot=zone.nb_elem_tot();
329 const int nb_faces_elem=zone.nb_faces_elem();
330 const int nb_bords=les_cl.size();
331
332 int elem=0;
333 int face=0,face_loc=0;
334 int faceAss=0,faceAss_loc=0;
335 int dim=0;
336 int ind_face=0;
337 int num1=0,num2=0;
338 int n_bord=0;
339
340 double psc=0.;
341 double volume=0.;
342 double nu_elem=0.;
343 double coeff=0.;
344
345 for (elem=0; elem<nb_elem_tot; elem++)
346 {
347 volume=volumes(elem);
348 nu_elem=nu_(elem);
349
350 for (face_loc=0; face_loc<nb_faces_elem; face_loc++)
351 {
352 face=elem_faces(elem,face_loc);
353
354 psc=0.;
355 for (dim=0; dim<dimension; dim++)
356 psc+=face_normales(face,dim)*face_normales(face,dim);
357
358 coeff=nu_elem;
359 coeff/=volume;
360 coeff*=psc;
361 coeff*=convexite_;
362 coeffOperateur(face)+=coeff;
363 }
364 }
365
366 for (n_bord=0; n_bord<nb_bords; n_bord++)
367 {
368 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
369 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
370
371 num1=0;
372 num2=le_bord.nb_faces();
373
374 if (sub_type(Periodique,la_cl.valeur()))
375 {
376 const Periodique& la_cl_perio = ref_cast(Periodique, la_cl.valeur());
377
378 for (ind_face=num1; ind_face<num2; ind_face++)
379 {
380 face=le_bord.num_face(ind_face);
381 faceAss_loc=la_cl_perio.face_associee(ind_face);
382 faceAss=le_bord.num_face(faceAss_loc);
383
384 if (face<faceAss)
385 {
386 coeffOperateur(faceAss)+=coeffOperateur(face);
387 coeffOperateur(face)=coeffOperateur(faceAss);
388 }
389 }
390 }
391 }

425DoubleVect& Op_Diff_VEFP1NCP1B_Face::
426calculer_gradient_elem(const DoubleVect& inconnue) const
427{
428 const Zone_VEF_PreP1b& zone_VEF = zone_VEFPreP1B();
429
430 const Zone& zone = zone_VEF.zone();
431 const DoubleTab& face_normales = zone_VEF.face_normales();
432 const DoubleVect& volumes = zone_VEF.volumes();
433 const IntTab& elem_faces=zone_VEF.elem_faces();
434 const IntTab& face_voisins=zone_VEF.face_voisins();
435
436 const int nb_faces_elem=zone.nb_faces_elem();
437 const int nb_elem_tot=zone.nb_elem_tot();
438 int elem=0,face_loc=0,face=0,compi=0,compj=0;
439
440 double signe=0.;
441 double volume=0.;
442
443// Valeurs INTEGRALES du gradient_p0_
444 for(elem=0; elem<nb_elem_tot; elem++)
445 for(face_loc=0; face_loc<nb_faces_elem; face_loc++)
446 {
447 face=elem_faces(elem,face_loc);
448
449 signe=1;
450 if(elem!=face_voisins(face,0)) signe=-1;
451
452 for(compi=0; compi<dim_ch_; compi++)
453 for(compj=0; compj<dimension; compj++)
454 gradient_p0_(elem,compi,compj)+=signe*
455 inconnue[face*dim_ch_+compi]*
456 face_normales(face,compj);
457 }
458
459// Valeurs NODALES du gradient_p0_
460 for (elem=0; elem<nb_elem_tot; elem++)
461 {
462 volume = volumes(elem);
463
464 for (compi=0; compi<dim_ch_; compi++)
465 for (compj=0; compj<dimension; compj++)
466 gradient_p0_(elem,compi,compj)/=(coeff_*volume);
467 }
468
469// REMARQUE : cet echange_espace_virtuel() est NECESSAIRE
470// dans le cas ou alphaS==1 et/ou alphaA=1
471 if (alphaS) gradient_p0_*=convexite_;
472 gradient_p0_.echange_espace_virtuel();
473 Debog::verifier("OpDifP1NCP1B Gradient P0 : ",gradient_p0_);
474
475 return gradient_p0_;
476}

478DoubleVect& Op_Diff_VEFP1NCP1B_Face::
479calculer_gradient_som(const DoubleVect& inconnue) const
480{
481 const Zone_VEF_PreP1b& zone_VEF = zone_VEFPreP1B();
482 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
483 const Zone& zone = zone_VEF.zone();
484 const Domaine& dom=zone.domaine();
485
486 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
487
488 const int nb_faces_elem=zone.nb_faces_elem();
489 const int nb_som_face=zone_VEF.nb_som_face();
490 const int nb_elem_tot=zone_VEF.nb_elem_tot();
491 const int nb_som=zone_VEF.nb_som();
492 const int nb_som_tot=zone_VEF.nb_som_tot();
493 const int nb_bords =les_cl.size();
494 int elem=0,face_loc=0,som_loc=0,face=0;
495 int compi=0,compj=0,som=0,num1=0,num2=0;
496 int i=0,ind_face=0;
497 int n_bord=0;
498
499 const double coeff_som=1./(dimension)/(dimension+1);
500 double signe=0.;
501
502 const DoubleTab& face_normales = zone_VEF.face_normales();
503 const DoubleVect& volume_aux_sommets=zone_VEF.volume_aux_sommets();
504
505 DoubleTab secmem(gradient_p1_);
506 ArrOfDouble sigma(dimension);
507
508 DoubleVect secmemij;
509 DoubleVect gradij;
510 dom.creer_tableau_sommets(secmemij);
511 dom.creer_tableau_sommets(gradij);
512
513 const IntTab& som_elem=zone.les_elems();
514 const IntTab& elem_faces=zone_VEF.elem_faces();
515 const IntTab& face_voisins=zone_VEF.face_voisins();
516 const IntTab& face_sommets=zone_VEF.face_sommets();
517
518// Le second membre du systeme a inverser
519 secmem=0.;
520 for(elem=0; elem<nb_elem_tot; elem++)
521 {
522 sigma = 0;
523 for(face_loc=0; face_loc<nb_faces_elem; face_loc++)
524 {
525 face = elem_faces(elem,face_loc);
526
527 for(compi=0; compi<dim_ch_; compi++)
528 sigma[compi]+=inconnue[face*dim_ch_+compi];
529 }
530
531 for(face_loc=0; face_loc<nb_faces_elem; face_loc++)
532 {
533 som = dom.get_renum_som_perio(som_elem(elem,face_loc));
534 face = elem_faces(elem,face_loc);
535
536 signe=1;
537 if(elem!=face_voisins(face,0)) signe=-1;
538
539 for(compi=0; compi<dim_ch_; compi++)
540 for(compj=0; compj<dimension; compj++)
541 secmem(som,compi,compj)+=coeff_som*signe*
542 sigma[compi]*face_normales(face,compj);
543 }
544 }
545
546 secmem.echange_espace_virtuel();
547 Debog::verifier("OpDifP1NCP1B secmem, avant CL : ", secmem);
548
549// Les conditions aux limites pour le second membre
550 for (n_bord=0; n_bord<nb_bords; n_bord++)
551 {
552 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
553 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
554
555 num1=0;
556 num2=le_bord.nb_faces_tot();
557
558 if (sub_type(Dirichlet_homogene,la_cl.valeur()))
559 {
560// On ne fait rien et c'est normal
561 }
562 else if (sub_type(Dirichlet,la_cl.valeur()))
563 {
564 const Dirichlet& dirichlet =
565 ref_cast(Dirichlet,la_cl.valeur());
566
567 double x=0.,y=0.,z=0.;
568 double inconnue_pt=0.;
569 double temps = equation().schema_temps().temps_courant();
570
571 const DoubleTab& coord_sommets = dom.coord_sommets();
572
573 if (sub_type(Champ_front_txyz,dirichlet.champ_front().valeur()))
574 {
575 const Champ_front_txyz& champ_front =
576 ref_cast(Champ_front_txyz,dirichlet.champ_front().valeur());
577
578 for (ind_face=num1; ind_face<num2; ind_face++)
579 {
580 face=le_bord.num_face(ind_face);
581
582 for(som_loc=0; som_loc<nb_som_face; som_loc++)
583 {
584 som=dom.get_renum_som_perio(face_sommets(face,som_loc));
585
586// Formule d'integration numerique exacte pour les polynomes de degre 2
587 if (dimension==2) //formule de Simpson
588 {
589// Coordonnees du sommet "som"
590 x=coord_sommets(som,0);
591 y=coord_sommets(som,1);
592
593// Valeur de l'inconnue au point d'integration
594 for (compi=0; compi<dim_ch_; compi++)
595 {
596 inconnue_pt=
597 champ_front.valeur_au_temps_et_au_point(temps,som,x,y,z,compi);
598
599 for (compj=0; compj<dimension; compj++)
600 secmem(som,compi,compj) +=
601 1./6*(2*inconnue[face*dim_ch_+compi]+inconnue_pt)
602 *face_normales(face,compj) ;
603 }
604 }//fin du if sur dimension==2
605
606 else //formule exacte pour les polynomes de degre 2
607 {
608// On suppose que l'element considere est un TETRAEDRE
609 for (i=1; i<3; i++)
610 {
611 int som2=face_sommets(face,(som_loc+i)%nb_som_face);
612 som2=dom.get_renum_som_perio(som2);
613
614// Coordonnees des points d'integration
615 x=(coord_sommets(som,0)+coord_sommets(som2,0))/2.;
616 y=(coord_sommets(som,1)+coord_sommets(som2,1))/2.;
617 z=(coord_sommets(som,2)+coord_sommets(som2,2))/2.;
618
619// Vitesse au point d'integration
620 for (compi=0; compi<dim_ch_; compi++)
621 {
622 inconnue_pt=
623 champ_front.valeur_au_temps_et_au_point(temps,som,x,y,z,compi);
624
625 for (compj=0; compj<dimension; compj++)
626 secmem(som, compi, compj) += 1./dimension*
627 1/2.*inconnue_pt*face_normales(face,compj) ;
628 }
629 }
630 }//fin du else sur la dimension

855calculer_divergence_som(DoubleVect& div) const
856{
857 const Zone_VEF_PreP1b& zone_VEF = zone_VEFPreP1B();
858 const Zone& zone = zone_VEF.zone();
859 const Domaine& dom=zone.domaine();
860
861 const DoubleTab& face_normales = zone_VEF.face_normales();
862 ArrOfDouble sigma(dimension);
863
864 const IntTab& som_elem=zone.les_elems();
865 const IntTab& elem_faces=zone_VEF.elem_faces();
866 const IntTab& face_voisins=zone_VEF.face_voisins();
867
868 const int nb_faces_elem=zone.nb_faces_elem();
869 const int nb_som_face=zone_VEF.nb_som_face();
870 const int nb_elem_tot=zone_VEF.nb_elem_tot();
871 int elem=0,face_loc=0,face_loc2=0,face=0;
872 int compi=0,compj=0,som=0,ind_face=0;
873 int num1=0, num2=0,som_loc=0;
874
875 static double coeff_som=1./(dimension)/(dimension+1);
876 double signe=0.;
877
878// Algorithme sans tenir compte des CL
879 for(elem=0; elem<nb_elem_tot; elem++)
880 for(face_loc=0; face_loc<nb_faces_elem; face_loc++)
881 {
882 som=dom.get_renum_som_perio(som_elem(elem,face_loc));
883 face=elem_faces(elem,face_loc);
884
885 signe=1;
886 if(elem!=face_voisins(face,0)) signe=-1;
887
888 for(compj=0; compj<dimension; compj++)
889 sigma[compj]=signe*face_normales(face,compj);
890
891 for(face_loc2=0; face_loc2<nb_faces_elem; face_loc2++)
892 for(compi=0; compi<dim_ch_; compi++)
893 for(compj=0; compj<dimension; compj++)
894 div[elem_faces(elem,face_loc2)*dim_ch_+compi]-=
895 coeff_som*gradient_p1_(som,compi,compj)*sigma[compj];
896 }
897
898// Les conditions aux limites
899 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();
900 const Conds_lim& les_cl = zone_Cl_VEF.les_conditions_limites();
901 const IntTab& face_sommets = zone_VEF.face_sommets();
902 const int nb_bords =les_cl.size();
903
904 DoubleTab gradient_bord(dim_ch_,dimension);
905
906 for (int n_bord=0; n_bord<nb_bords; n_bord++)
907 {
908 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
909 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
910 num1 = 0;
911 num2 = le_bord.nb_faces();
912
913 if (sub_type(Neumann,la_cl.valeur()) ||
914 sub_type(Neumann_val_ext,la_cl.valeur()) ||
915 sub_type(Neumann_homogene,la_cl.valeur()) ||
916 sub_type(Symetrie,la_cl.valeur())
917 )
918 {
919 for (ind_face=num1; ind_face<num2; ind_face++)
920 {
921// Le numero de la face (qui peut etre virtuelle)
922 face=le_bord.num_face(ind_face);
923
924// Le numero de l'element voisin
925 elem=face_voisins(face,0);
926 assert(elem!=-1);
927
928// Calcul du gradient au milieu de la face de bord
929// On prend une integration numerique approchee
930 gradient_bord=0.;
931 for (som_loc=0; som_loc<nb_som_face; som_loc++)
932 {
933 som=dom.get_renum_som_perio(face_sommets(face,som_loc));
934
935 for(compi=0; compi<dim_ch_; compi++)
936 for(compj=0; compj<dimension; compj++)
937 gradient_bord(compi,compj)+=
938 gradient_p1_(som,compi,compj);
939 }
940 gradient_bord/=dimension;
941
942// Calcul divergence au bord : formule d'integration numerique
943// exacte pour polynome d'ordre 1
944 for(compi=0; compi<dim_ch_; compi++)
945 for(compj=0; compj<dimension; compj++)
946 div[face*dim_ch_+compi]-=
947 gradient_bord(compi,compj)
948 *face_normales(face,compj);
949
950 }//fin du for sur "ind_face"
951
952 }//fin du if sur "Neumann_paroi", "Neumann", "Symetrie"
953
954 }//fin du for sur "n_bords"
955
956 div.echange_espace_virtuel();
957 Debog::verifier("OpDifP1NCP1B divergence P1 : ", div);
958 return div;
