33void Zone_EF_axi::calculer_Bij_gen(DoubleTab& bij)
34{
35 const IntTab& les_elems=zone().les_elems() ;
36 int nbsom_face=nb_som_face();
37 int nbelem=nb_elem_tot();
38 int nbsom_elem=zone().nb_som_elem();
39 const IntTab& elemfaces=elem_faces_;
40 int nbface_elem=zone().nb_faces_elem();
41
42 const DoubleTab& les_coords = domaine_axi().coord_sommets();
43
44 const DoubleVect& surf = face_surfaces();
45
46 for (int elem=0; elem<nbelem; elem++)
47 {
48 double x0 = origine_repere_(elem,0);
49 double y0 = origine_repere_(elem,1);
50
51 for (int i=0; i<nbsom_elem; i++)
52 {
53// on cherche les faces contribuantes ,ce n'est pas optimal
54 for (int f=0; f<nbface_elem; f++)
55 {
56 int face=elemfaces(elem,f);
57 for (int s=0; s<nbsom_face; s++)
58 {
59// on cherche les faces contribuantes ,ce n'est pas optimal
60 if (face_sommets_(face,s)==les_elems(elem,i))
61 {
62 double x1 = les_coords(les_elems(elem,0),0);
63 double x2 = les_coords(les_elems(elem,1),0);
64 double y1 = les_coords(les_elems(elem,0),1);
65 double y2 = les_coords(les_elems(elem,1),1);
66
67 double r1 = sqrt((x1-x0)*(x1-x0)+(y1-y0)*(y1-y0));
68 double r2 = sqrt((x2-x0)*(x2-x0)+(y2-y0)*(y2-y0));
69
70 for (int j=0; j<dimension; j++)
71 {
72 if (surf(face)>0)
73 bij(elem,i,j)+=face_normales(face,j)*oriente_normale(face,elem)*(r1+r2)*M_PI/surf(face);
74 else
75 bij(elem,i,j)+=face_normales(face,j)*oriente_normale(face,elem)*(r1+r2)*M_PI;
76 }
77 }
78 }
79 }
80 }
81 }
82}

84void Zone_EF_axi::calculer_IPhi()
85{
86 int nbelem=zone().nb_elem();
87 int nb_som_elem=zone().nb_som_elem();
88 assert(nb_som_elem==2);
89
90 IPhi_.resize(nbelem,nb_som_elem);
91 domaine_axi().creer_tableau_elements(IPhi_);
92// Scatter::creer_tableau_distribue(zone().domaine(), Joint::ELEMENT, IPhi_);
93 IPhi_thilde_=IPhi_;
94
95 const IntTab& les_elems=zone().les_elems() ;
96
97 const DoubleTab& les_coords = domaine_axi().coord_sommets();
98
99 for (int elem=0; elem<nbelem; elem++)
100 {
101 double x1 = les_coords(les_elems(elem,0),0);
102 double x2 = les_coords(les_elems(elem,1),0);
103 double y1 = les_coords(les_elems(elem,0),1);
104 double y2 = les_coords(les_elems(elem,1),1);
105
106 double x0 = origine_repere_(elem,0);
107 double y0 = origine_repere_(elem,1);
108
109 double r1 = sqrt((x1-x0)*(x1-x0)+(y1-y0)*(y1-y0));
110 double r2 = sqrt((x2-x0)*(x2-x0)+(y2-y0)*(y2-y0));
111 IPhi_(elem,0) = (M_PI/3.)*(r2*r2+r1*r2-2.*r1*r1);
112 IPhi_(elem,1) = (M_PI/3.)*(2.*r2*r2-r1*r2-r1*r1);
113
114 for (int s=0; s<nb_som_elem; s++)
115 IPhi_thilde_(elem,s)=IPhi_(elem,s)*porosite_elem_(elem);
116 }
117
118 IPhi_.echange_espace_virtuel();
119 IPhi_thilde_.echange_espace_virtuel();
120}

122void Zone_EF_axi::remplir_tableau_origine()
123{
124 DomaineAxi1d& domax = domaine_axi();
125 const Champ& orig = domax.champ_origine();
126
127 origine_repere_.resize(0,Objet_U::dimension-1);
128 domax.creer_tableau_elements(origine_repere_);
129
130 if (orig.non_nul())
131 {
132 const DoubleTab& positions = xp();
133
134 IntVect les_polys(nb_elem());
135 for(int e = 0; e < nb_elem(); e++)
136 les_polys(e) = e;
137
138 orig->valeur_aux_elems(positions, les_polys, origine_repere_);
139 }
140 else
141 origine_repere_=0.;
142
143 domax.associer_origine_repere(origine_repere_);
144}
