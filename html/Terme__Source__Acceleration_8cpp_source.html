70{

74// Valeurs par defaut:

95 int fini = 0;
96 while (!fini)
97 {
98 s >> motlu;
99 const int i_mot = les_mots.search(motlu);
100 if (je_suis_maitre() && i_mot < 5)

256//! Calcul de la valeur du champ la_source aux faces en fonction de<br>- calculer_vitesse_faces()<br>- champ_acceleration_<br>- omega_<br>- domegadt_<br>- centre_rotation_<br>Parametre: champ_source<br>Signification: un champ discretise aux elements de la zone_VF dans lequel<br>on stocke le resultat du calcul. Espace virtuel a jour.
257const DoubleTab&
258Terme_Source_Acceleration::calculer_la_source(DoubleTab& acceleration_aux_faces) const
259{
260 const Zone_VF& zone_VF = ref_cast(Zone_VF, equation().zone_dis().valeur());;
261
262 const DoubleTab& centre_faces = zone_VF.xv();
263
264 double a[3] = {0., 0., 0.}; //!< champ acceleration
265 double w[3] = {0., 0., 0.}; //!< omega
266 double dw[3] = {0., 0., 0.}; //!< domegadt
267 double c[3] = {0., 0., 0.}; //!< centre_gravite
268 const int dim = Objet_U::dimension;
269
270// ****************************************************************
271// INITIALISATION DE a, w, dw, c
272 int rotation_solide = 0;
273 {
274 int j;
275
276 if (champ_acceleration_.non_nul())
277 {
278 const DoubleTab& a_ = champ_acceleration_.valeur().valeurs();
279 for (j = 0; j < dim; j++)
280 a[j] = a_(0, j);
281 }
282
283 if (omega_.non_nul())
284 {
285// L'utilisateur a specifie un mouvement de rotation solide
286 const DoubleTab& champ_w = omega_ .valeur().valeurs();
287 const DoubleTab& champ_dw = domegadt_ .valeur().valeurs();
288 const DoubleTab& champ_c = centre_rotation_.valeur().valeurs();
289// Attention: en 2D, le vecteur rotation est typiquement oriente
290// dans la direction Z : donc boucle jusqu'a 3 dans tous les cas:
291 for (j = 0; j < 3; j++) w[j] = champ_w(0,j);
292 for (j = 0; j < 3; j++) dw[j] = champ_dw(0,j);
293 for (j = 0; j < dim; j++) c[j] = champ_c(0,j);
294 rotation_solide = 1;
295 }
296 }
297
298// Si on a un terme source de rotation solide, on a besoin des trois
299// composantes de la vitesse du fluide aux faces:
300
301// Un espace de stockage pour la vitesse si elle doit etre
302// calculee (en VDF, on calcule les trois composantes a chaque face)
303 DoubleTab vitesse_faces_stockage;
304// Calcul du champ de vitesse aux faces (a partir de l'inconnue
305// de l'equation de N.S.)
306 const DoubleTab& vitesse_faces =
307 (rotation_solide)
308 ? calculer_vitesse_faces(vitesse_faces_stockage)
309 : vitesse_faces_stockage;
310
311// **************************************************
312// BOUCLE SUR LES FACES
313 int i_face;
314 const int nb_faces = acceleration_aux_faces.dimension(0);
315
316// Vecteur vitesse du fluide dans l'element courant
317// (attention: initialisation de la composante Z pour le 2D)
318 double v[3] = {0., 0., 0.};
319// Vecteur (centre_face - centre_rotation)
320// (idem)
321 double xgr[3] = {0., 0., 0.};
322
323 for (i_face = 0; i_face < nb_faces; i_face++)
324 {
325
326 int j;
327 double src[3];
328 src[0] = -a[0]; //!< Terme d'acceleration
329 src[1] = -a[1];
330 src[2] = -a[2]; //!< =0. en 2D
331
332// Formules identiques en 2D et en 3D :
333// en 2D, les composantes z de v et xgr sont nulles.
334 if (rotation_solide)
335 {
336 for (j = 0; j < dim; j++) //!< en 2D, v[3]=0.
337 v[j] = vitesse_faces(i_face, j);
338
339 for (j = 0; j < dim; j++) //!< en 2D, xgr[3]=0.
340 xgr[j] = centre_faces(i_face, j) - c[j];
341
342 if (option_ != ENTRAINEMENT_SEUL)
343 {
344 src[0] += -2. * (w[1] * v[2] - w[2] * v[1]);
345 src[1] += -2. * (w[2] * v[0] - w[0] * v[2]);
346 src[2] += -2. * (w[0] * v[1] - w[1] * v[0]);
347 src[0] += dw[2] * xgr[1] - dw[1] * xgr[2];
348 src[1] += dw[0] * xgr[2] - dw[2] * xgr[0];
349 src[2] += dw[1] * xgr[0] - dw[0] * xgr[1];
350 }
351 if (option_ != CORIOLIS_SEUL)
352 {
353 src[0] += w[2]*w[2]*xgr[0] - w[0]*w[2]*xgr[2] - w[0]*w[1]*xgr[1] + w[1]*w[1]*xgr[0];
354 src[1] += w[0]*w[0]*xgr[1] - w[0]*w[1]*xgr[0] - w[1]*w[2]*xgr[2] + w[2]*w[2]*xgr[1];
355 src[2] += w[1]*w[1]*xgr[2] - w[1]*w[2]*xgr[1] - w[2]*w[0]*xgr[0] + w[0]*w[0]*xgr[2];
356 }
357 }
358 for (j = 0; j < dim; j++)
359 acceleration_aux_faces(i_face, j) = src[j];
360 }
361 return acceleration_aux_faces;
362}
