35Entree& Redresser_hexaedres_vdf::interpreter_(Entree& is)
36{
37 if (Process::nproc()>1)
38 {
39 Cerr << "Redresser_hexaedres_vdf can't be used during a parallel calculation." << finl;
40 exit();
41 }
42 associer_domaine(is);
43 DoubleTab& les_sommets = domaine().les_sommets();
44 int nb_sommets = les_sommets.dimension(0);
45 double epsilon = Objet_U::precision_geom;
46 int ok=0;
47// On type en rectangle ou hexaedre
48 const Elem_geom_base& elem=domaine().zone(0).type_elem().valeur();
49 Cerr << "Attempt on the mesh ";
50 if (sub_type(Rectangle,elem)) Cerr << "to correct rectangles..." << finl;
51 else if (sub_type(Quadrangle_VEF,elem)) Cerr << "to correct VEF quadrangles into rectangles..." << finl;
52 else if (sub_type(Hexaedre,elem)) Cerr << "to correct hexahedra..." << finl;
53 else if (sub_type(Hexaedre_VEF,elem)) Cerr << "to correct VEF hexahedra into regular hexahedra..." << finl;
54 else
55 {
56 Cerr << "This type of element ("<<elem.que_suis_je()<<") is not supported by Redresser_hexaedres_vdf" << finl;
57 exit();
58 }
59 if (dimension==2) domaine().zone(0).typer("Rectangle");
60 if (dimension==3) domaine().zone(0).typer("Hexaedre");
61 Elem_geom_base& new_elem=domaine().zone(0).type_elem().valeur();
62 do
63 {
64 double correction_max = 0;
65 Octree_Double octree;
66 octree.build_nodes(les_sommets, 0 /* do not include virtual nodes */);
67// direction indique la coordonnee qu'on va corriger
68 for (int direction = 0; direction < dimension; direction++)
69 {
70 ArrOfInt marqueurs(nb_sommets); //!< initialise a zero
71 ArrOfInt liste_sommets;
72 liste_sommets.set_smart_resize(1);
73 int prochain_sommet = 0;
74 while (prochain_sommet < nb_sommets)
75 {
76// Cherche le prochain sommet non marque:
77 for (; prochain_sommet < nb_sommets; prochain_sommet++)
78 if (! marqueurs[prochain_sommet])
79 break;
80 if (prochain_sommet == nb_sommets)
81 break;
82// Trouver tous les sommets qui sont dans une boite tres grande
83// en dehors de la direction dir:
84 double x = les_sommets(prochain_sommet, 0);
85 double y = les_sommets(prochain_sommet, 1);
86 double z = (dimension == 3) ? les_sommets(prochain_sommet, 2) : 0.;
87 double xmin = (direction==0) ? (x-epsilon) : (-DMAXFLOAT);
88 double xmax = (direction==0) ? (x+epsilon) : (+DMAXFLOAT);
89 double ymin = (direction==1) ? (y-epsilon) : (-DMAXFLOAT);
90 double ymax = (direction==1) ? (y+epsilon) : (+DMAXFLOAT);
91 double zmin = (direction==2) ? (z-epsilon) : (-DMAXFLOAT);
92 double zmax = (direction==2) ? (z+epsilon) : (+DMAXFLOAT);
93 octree.search_elements_box(xmin, ymin, zmin, xmax, ymax, zmax, liste_sommets);
94// Pour chaque sommet, corriger deux des coordonnees
95// (on met les coordonnees du point trouve)
96 const int n = liste_sommets.size_array();
97 for (int i = 0; i < n; i++)
98 {
99 const int j = liste_sommets[i];
100// le premier sommet sert de reference, ne pas le corriger:
101 if (j == prochain_sommet)
102 continue;
103// coordonnee de reference (on aligne tous les autres points du
104// plan de maillage sur celui-la:
105 const double coord_ref = les_sommets(prochain_sommet, direction);
106
107// verifier que le sommet est bien dans la boite (l'octree
108// renvoie les sommets "potentiellement" a l'interieur
109 const double coord = les_sommets(j, direction);
110 if (std::fabs(coord-coord_ref) > epsilon)
111 continue;
112
113 correction_max=std::max(correction_max,std::fabs(coord_ref-les_sommets(j, direction)));
114// Corriger:
115 les_sommets(j, direction) = coord_ref;
116 marqueurs[j] = 1; //!< sommet traite !
117 }
118 marqueurs[prochain_sommet] = 1;
119 }
120 }
121 epsilon*=10;
122 Cerr << "Redresser_hexaedres_vdf, epsilon=" << epsilon << " maximum_correction=" << correction_max << finl;
123 if (epsilon>1)
124 {
125 Cerr << "Redresser_hexaedres_vdf failed" << finl;
126 Cerr << "it has failed to make regular hexahedra on the mesh." << finl;
127 exit();
128 }
129 if (dimension==2)
130 ok=ref_cast(Rectangle,new_elem).reordonner_elem();
131 else if (dimension==3)
132 ok=ref_cast(Hexaedre,new_elem).reordonner_elem();
133 }
134 while (ok==-1);
135 Cerr << "End and success of Redresser_hexaedres_vdf" << finl;
136 return is;
137}
