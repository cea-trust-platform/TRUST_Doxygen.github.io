141// Loi analytique avec raccordement des comportements
142// asymptotiques de la temperature adimensionnee T+
143// avec Kader:
144// sous-couche conductrice : T+=Pr y+
145// zone logarithmique : T+=2.12*ln(y+)+Beta
146inline double Turbulence_paroi_scal_base::T_plus(double y_plus, double Pr)
147{
148 double Gamma = (0.01*pow(Pr*y_plus,4.))/(1.+5.*pow(Pr,3.)*y_plus);
149 double Beta = pow(3.85*pow(Pr,1./3.)-1.3,2.)+Prdt_sur_kappa_*log(Pr);
150 return Pr*y_plus*exp(-Gamma) + (Prdt_sur_kappa_*log(1.+y_plus) + Beta)*exp(-1./(Gamma+1e-20));
151}
