310void paroi_contact::complement(Nom& ajout)
311{
312 int rayo = is_pb_rayo();
313 if (!rayo)
314 {
315 if (ref_cast(Probleme_base,Interprete::objet(nom_autre_pb)).milieu().is_rayo_transp())
316 rayo = 2;
317 else if (ref_cast(Probleme_base,Interprete::objet(nom_autre_pb)).milieu().is_rayo_semi_transp())
318 rayo = 1;
319 }
320
321 const Nom& nom_mon_pb=mon_equation->probleme().le_nom();
322 if (nom_mon_pb==nom_autre_pb)
323 {
324 Cerr<<"Error in "<<que_suis_je()<<" the name of the other problem is the same as my problem :"<<nom_mon_pb<<" on the boundary :"<< nom_bord_<<finl;
325 exit();
326 }
327
328 if (is_pb_VDF() || is_pb_PolyMAC() || is_pb_PolyMAC_P0())
329 {
330 if (is_pb_VDF())
331 {
332 ajout = "paroi_echange_contact_VDF ";
333 if (rayo == 2)
334 ajout = "Echange_contact_Rayo_transp_VDF ";
335 if (rayo == 1)
336 ajout = "Paroi_Echange_contact_rayo_semi_transp_VDF ";
337 }
338 else
339 ajout = is_pb_PolyMAC() ? "paroi_echange_contact_PolyMAC " : "paroi_echange_contact_PolyMAC_P0 ";
340
341 ajout += nom_autre_pb;
342 ajout += " ";
343 ajout += nom_autre_bord + " ";
344 ajout += mon_equation->inconnue().le_nom() + " 1.e10";
345 }
346 else
347 {
348
349 if (rayo==1)
350 {
351 ajout="paroi_temperature_imposee_rayo_semi_transp ";
352 ajout+="Champ_Front_contact_rayo_semi_transp_VEF ";
353 }
354 else if (rayo==2)
355 {
356 ajout="paroi_temperature_imposee_rayo_transp ";
357 ajout+="Champ_Front_contact_rayo_transp_VEF ";
358 }
359 else
360 {
361 ajout="scalaire_impose_paroi ";
362 ajout+="Champ_Front_contact_VEF ";
363 }
364 ajout+=nom_mon_pb;
365 ajout+=" ";
366 ajout+=nom_bord_;
367 ajout+=" ";
368 ajout+=nom_autre_pb;
369 ajout+=" ";
370 ajout+=nom_autre_bord;
371 }
