93template<AJOUTE_SCAL _T_>
94DoubleTab& Op_Dift_EF_Q1::ajouter_scalaire_template(const DoubleTab& tab_inconnue, DoubleTab& resu) const
95{
96 static constexpr bool IS_GEN = (_T_ == AJOUTE_SCAL::GEN), IS_D3_8 = (_T_ == AJOUTE_SCAL::D3_8), IS_D2_4 = (_T_ == AJOUTE_SCAL::D2_4);
97
98 const Zone_EF& zone_ef = ref_cast(Zone_EF, equation().zone_dis().valeur());
99
100 const int N = IS_GEN ? resu.line_size() : 1;
101 const int nb_som_elem = IS_D3_8 ? 8 : ( IS_D2_4 ? 4 : zone_ef.zone().nb_som_elem() /* IS_GEN */);
102 const int const_dimension = IS_D3_8 ? 3 : ( IS_D2_4 ? 2 : Objet_U::dimension /* IS_GEN */);
103 const int dim_fois_nbn = nb_som_elem * const_dimension;
104
105 const DoubleVect& volumes_thilde = zone_ef.volumes_thilde(), &volumes = zone_ef.volumes();
106
107 const DoubleTab& bij = zone_ef.Bij();
108 int nb_elem_tot = zone_ef.zone().nb_elem_tot();
109 const IntTab& elems = zone_ef.zone().les_elems();
110
111 remplir_nu(nu_);
112 DoubleVect diffu_turb(diffusivite_turbulente()->valeurs());
113 DoubleTab diffu(nu_);
114
115 const double *bij_ptr = bij.addr();
116 const double *inco_ptr = tab_inconnue.addr();
117
118#define bij_(elem,i,j) bij_ptr[elem*dim_fois_nbn+i*const_dimension+j]
119#define inconnue_(som,a) inco_ptr[som * N + a]
120#define resu_(som,a) resu_ptr[som * N + a]
121
122 ArrOfDouble pr(N);
123 for (int elem = 0; elem < nb_elem_tot; elem++)
124 if (elem_contribue(elem))
125 {
126 double pond = volumes_thilde(elem) / volumes(elem) / volumes(elem) * (diffu[elem] + diffu_turb[elem]);
127
128 for (int i1 = 0; i1 < nb_som_elem; i1++)
129 {
130 int glob = elems(elem, i1);
131 for (int yy = 0; yy < N; yy++) pr[yy] = 0;
132
133 for (int i2 = 0; i2 < nb_som_elem; i2++)
134 {
135 int glob2 = elems(elem, i2);
136
137 {
138 double prod = 0;
139 for (int b = 0; b < const_dimension; b++)
140 {
141 prod += bij_(elem,i1,b)*bij_(elem,i2,b);
142 }
143 for (int n = 0; n < N; n++) pr[n] += prod * inconnue_(glob2, n);
144 }
145 }
146 resu(glob) -= pr[0] * pond;
147 }
148 }
149
150// on ajoute la contribution des bords
151 ajouter_bords(tab_inconnue, resu);
152 return resu;
153
154#undef bij_
155#undef inconnue_
156#undef resu_
157}
