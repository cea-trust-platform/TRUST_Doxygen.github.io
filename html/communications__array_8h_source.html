23// Ces methodes sont volontairement separees de communications.h car plus dangereuses a l'usage.
24// Les tableaux passes en parametres doivent absolument avoir au moins n cases et n doit etre
25// identique sur tous les processeurs qui communiquent entre eux.
26
27// Attention: le template ne marche que pour les types simples (pas Objet_U !)
28// On ne passe pas par un buffer Entree / Sortie mais on envoie directement le tableau sous sa forme binaire.
29// Un seul message envoye, sauf en mode check() ou on envoie aussi la taille pour verifier.
30template<typename _TYPE_>
31typename std::enable_if<std::is_arithmetic<_TYPE_>::value,int >::type
32inline envoyer_array(const _TYPE_ *objet, int n, int source, int cible, int canal)
33{
34 const Comm_Group& grp = PE_Groups::current_group();
35 const int moi = grp.rank();
36 if (moi == cible)
37 return 0;
38
39 if (source != moi && source != -1)
40 return 0;
41
42 const char *data = (const char*) objet;
43 const int sz = sizeof(_TYPE_) * n;
44 if (cible < 0)
45 {
46 const int nbproc = grp.nproc();
47 for (int i = 0; i < nbproc; i++)
48 {
49 if (i != moi)
50 {
51// En mode check, on verifie que n est bien le meme sur l'expediteur et le recepteur
52 if (grp.check_enabled())
53 grp.send(i, &sz, sizeof(int), canal);
54 if (sz > 0)
55 grp.send(i, data, sz, canal);
56 }
57 }
58 }
59 else
60 {
61 if (grp.check_enabled())
62 grp.send(cible, &sz, sizeof(int), canal);
63 if (sz > 0)
64 grp.send(cible, data, sz, canal);
65 }
66 return 1;
67}

76template<typename _TYPE_>
77typename std::enable_if<std::is_arithmetic<_TYPE_>::value,int >::type
78inline recevoir_array(const _TYPE_ *objet, int n, int source, int cible, int canal)
79{
80 const Comm_Group& grp = PE_Groups::current_group();
81 const int moi = grp.rank();
82 if (moi == source)
83 return 0;
84
85 if (cible != moi && cible != -1)
86 return 0;
87
88 const int sz = sizeof(_TYPE_) * n;
89 if (grp.check_enabled())
90 {
91 int sz_check;
92 grp.recv(source, &sz_check, sizeof(int), canal);
93 if (sz_check != sz)
94 {
95 Cerr << "Fatal error in template<typename _TYPE_> int recevoir_array : incorrect size\n" << " sent=" << sz_check << " expected=" << sz << finl;
96 Process::exit();
97 }
98 }
99 char *data = (char*) objet;
100 if (sz > 0)
101 grp.recv(source, data, sz, canal);
102 return 1;
103}

114inline envoyer_broadcast_array(_TYPE_ *objet, int n, int source)
115{

118 {
119 int copie_n = n;
120 grp.broadcast(&copie_n, sizeof(int), source);
121 if (copie_n != n)
122 {
123 Cerr << "Error in template<typename _TYPE_> envoyer_broadcast_array !" << finl;
124 Process::exit();
125 }
126 }
127 grp.broadcast(objet, sizeof(_TYPE_) * n, source);
128 return 1;
129}
