31Entree& Saturation_constant::readOn(Entree& is)
32{
33 Param param(que_suis_je());
34 param.ajouter("Tsat", &tsat_,Param::REQUIRED);
35 param.ajouter("Psat", &psat_,Param::REQUIRED);
36 param.ajouter("Lvap", &lvap_);
37 param.ajouter("Hlsat", &hls_);
38 param.ajouter("Hvsat", &hvs_);
39 param.ajouter("tension_superficielle", &sigma__);
40 param.lire_avec_accolades_depuis(is);
41// verifications hlsat/hvsat/lvap
42 const int i = (lvap_ > 0) + (hls_ > 0) + (hvs_ > 0);
43 if (i != 2) Process::exit(que_suis_je() + " Please give 2 properties among {Lvap, Hlsat, Hvsat}");
44 if (lvap_ > 0 && hls_ > 0) hvs_ = hls_ + lvap_;
45 else if (lvap_ > 0 && hvs_ > 0) hls_ = hvs_ - lvap_;
46 else if (hls_ > 0 && hvs_ > 0) lvap_ = hvs_ - hls_;
47 else Process::exit(que_suis_je() + "bad parameters");
48 return is;
49}
