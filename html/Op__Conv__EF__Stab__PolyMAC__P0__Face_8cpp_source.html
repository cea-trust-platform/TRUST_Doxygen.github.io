68double Op_Conv_EF_Stab_PolyMAC_P0_Face::calculer_dt_stab() const
69{
70 double dt = 1e10;
71 const Zone_Poly_base& zone = la_zone_poly_.valeur();
72 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
73 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face(), &ve = zone.volumes(), &pe = zone.porosite_elem(), &vf = zone.volumes_entrelaces();
74 const DoubleTab& vit = vitesse_->valeurs(), &vfd = zone.volumes_entrelaces_dir(),
75 *alp = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
76 const IntTab& e_f = zone.elem_faces(), &f_e = zone.face_voisins(), &fcl = ch.fcl();
77 int i, e, f, n, N = vit.line_size();
78 DoubleTrav flux(N), vol(N); //somme des flux pf * |f| * vf, volume minimal des mailles d'elements/faces affectes par ce flux
79
80 for (e = 0; e < zone.nb_elem(); e++)
81 {
82 for (vol = pe(e) * ve(e), flux = 0, i = 0; i < e_f.dimension(1) && (f = e_f(e, i)) >= 0; i++)
83 for (n = 0; n < N; n++)
84 {
85 flux(n) += pf(f) * fs(f) * std::max((e == f_e(f, 1) ? 1 : -1) * vit(f, n), 0.); //seul le flux entrant dans e compte
86 if (0 && fcl(f, 0) < 2) vol(n) = std::min(vol(n), pf(f) * vf(f) / vfd(f, e != f_e(f, 0)) * vf(f)); //prise en compte de la contribution aux faces
87 }
88 for (n = 0; n < N; n++)
89 if ((!alp || (*alp)(e, n) > 1e-3) && flux(n)) dt = std::min(dt, vol(n) / flux(n));
90 }
91
92 return Process::mp_min(dt);
93}

95void Op_Conv_EF_Stab_PolyMAC_P0_Face::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
96{
97 const Zone_Poly_base& zone = la_zone_poly_.valeur();
98 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
99 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces(), &fcl = ch.fcl();
100 const DoubleTab& nf = zone.face_normales();
101 const DoubleVect& fs = zone.face_surfaces();
102
103 const std::string& nom_inco = ch.le_nom().getString();
104 if (!matrices.count(nom_inco) || semi_impl.count(nom_inco)) return; //pas de bloc diagonal ou semi-implicite -> rien a faire
105 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
106 const Masse_ajoutee_base *corr = pbm && pbm->has_correlation("masse_ajoutee") ? &ref_cast(Masse_ajoutee_base, pbm->get_correlation("masse_ajoutee").valeur()) : NULL;
107 Matrice_Morse& mat = *matrices.at(nom_inco), mat2;
108
109 int i, j, k, e, eb, f, fb, fc, ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), m, n, N = equation().inconnue().valeurs().line_size(), d, D = dimension;
110
111 IntTab stencil(0, 2);
112 stencil.set_smart_resize(1);
113
114 /* agit uniquement aux elements; diagonale omise */
115 for (f = 0; f < zone.nb_faces_tot(); f++)
116 if (f_e(f, 0) >= 0 && (f_e(f, 1) >= 0 || fcl(f, 0) == 3))
117 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
118 for (j = 0; j < 2 && (eb = f_e(f, j)) >= 0; j++)
119 {
120 for (k = 0; k < e_f.dimension(1) && (fb = e_f(e, k)) >= 0; k++)
121 if (fb < zone.nb_faces() && fcl(fb, 0) < 2)
122 {
123 if ((fc = zone.equiv(f, i, k)) >= 0) //equivalence : face -> face
124 for (n = 0; fcl(fc, 0) < 2 && n < N; n++)
125 for (m = (corr ? 0 : n); m < (corr ? N : n + 1); m++) stencil.append_line(N * fb + n, N * fc + m);
126 else if (f_e(f, 1) >= 0)
127 for (d = 0; d < D; d++) //pas d'equivalence : elem -> face
128 if(std::fabs(nf(fb, d)) > 1e-6 * fs(fb))
129 for (n = 0; n < N; n++)
130 for (m = (corr ? 0 : n); m < (corr ? N : n + 1); m++)
131 stencil.append_line(N * fb + n, N * (nf_tot + D * eb + d) + m); //elem -> face
132 }
133 for (d = 0; d < D; d++)
134 for (n = 0; n < N; n++)
135 for (m = (corr ? 0 : n); m < (corr ? N : n + 1); m++) //elem->elem
136 stencil.append_line(N * (nf_tot + D * e + d) + n, N * (nf_tot + D * eb + d) + m);
137 }
138
139 tableau_trier_retirer_doublons(stencil);
140 Matrix_tools::allocate_morse_matrix(N * (nf_tot + D * ne_tot), N * (nf_tot + D * ne_tot), stencil, mat2);
141 mat.nb_colonnes() ? mat += mat2 : mat = mat2;
142}

146void Op_Conv_EF_Stab_PolyMAC_P0_Face::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, const tabs_t& semi_impl) const
147{
148 const Zone_Poly_base& zone = la_zone_poly_.valeur();
149 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
150 const Conds_lim& cls = la_zcl_poly_.valeur().les_conditions_limites();
151 const IntTab& f_e = zone.face_voisins(), &e_f = zone.elem_faces(), &fcl = ch.fcl();
152 const DoubleTab& vit = ch.passe(), &nf = zone.face_normales(), &vfd = zone.volumes_entrelaces_dir();
153 const DoubleVect& fs = zone.face_surfaces(), &pe = porosite_e, &pf = porosite_f, &ve = zone.volumes();
154
155 /* a_r : produit alpha_rho si Pb_Multiphase -> par semi_implicite, ou en recuperant le champ_conserve de l'equation de masse */
156 const std::string& nom_inco = ch.le_nom().getString();
157 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()) : NULL;
158 const Masse_ajoutee_base *corr = pbm && pbm->has_correlation("masse_ajoutee") ? &ref_cast(Masse_ajoutee_base, pbm->get_correlation("masse_ajoutee").valeur()) : NULL;
159 const DoubleTab& inco = semi_impl.count(nom_inco) ? semi_impl.at(nom_inco) : ch.valeurs(),
160 *a_r = !pbm ? NULL : semi_impl.count("alpha_rho") ? &semi_impl.at("alpha_rho") : &pbm->eq_masse.champ_conserve().valeurs(),
161 *alp = pbm ? &pbm->eq_masse.inconnue().passe() : NULL, &rho = equation().milieu().masse_volumique().passe();
162 Matrice_Morse *mat = matrices.count(nom_inco) && !semi_impl.count(nom_inco) ? matrices.at(nom_inco) : NULL;
163
164 int i, j, k, e, eb, f, fb, fc, fd, nf_tot = zone.nb_faces_tot(), m, n, N = inco.line_size(), d, D = dimension, comp = !incompressible_;
165 double mult;
166
167 DoubleTrav dfac(2, N, N), masse(N, N);
168 for (f = 0; f < zone.nb_faces_tot(); f++)
169 if (f_e(f, 0) >= 0 && (f_e(f, 1) >= 0 || fcl(f, 0) == 1 || fcl(f, 0) == 3))
170 {
171 for (i = 0, dfac = 0; i < 2; i++)
172 {
173// masse : diagonale + masse ajoutee si correlation
174 for (masse = 0, e = f_e(f, f_e(f, i) >= 0 ? i : 0), n = 0; n < N; n++) masse(n, n) = a_r ? (*a_r)(e, n) : 1;
175 if (corr) corr->ajouter(&(*alp)(e, 0), &rho(e, 0), masse);
176// contribution a dfac
177 for (e = f_e(f, i), eb = f_e(f, i), n = 0; n < N; n++)
178 for (m = 0; m < N; m++)
179 dfac(fcl(f, 0) == 1 ? 0 : i, n, m) += fs(f) * vit(f, m) * pe(eb >= 0 ? eb : f_e(f, 0)) * masse(n, m)
180 * (1. + (vit(f, m) * (i ? -1 : 1) >= 0 ? 1. : vit(f, m) ? -1. : 0.) * alpha) / 2;
181 }
182 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
183 {
184 for (k = 0; k < e_f.dimension(1) && (fb = e_f(e, k)) >= 0; k++)
185 if (fb < zone.nb_faces() && fcl(fb, 0) < 2) //partie "faces"
186 {
187 if ((fc = zone.equiv(f, i, k)) >= 0 || f_e(f, 1) < 0)
188 for (j = 0; j < 2; j++) //equivalence : face fd -> face fb
189 {
190 eb = f_e(f, j), fd = (j == i ? fb : fc); //element/face sources
191 mult = (fd < 0 || zone.dot(&nf(fb, 0), &nf(fd, 0)) > 0 ? 1 : -1) * (fd >= 0 ? pf(fd) / pe(eb) : 1); //multiplicateur pour passer de vf a ve
192 for (n = 0; n < N; n++)
193 for (m = 0; m < N; m++)
194 if (dfac(j, n, m))
195 {
196 double fac = (i ? -1 : 1) * vfd(fb, e != f_e(fb, 0)) * dfac(j, n, m) / ve(e);
197 if (fd >= 0) secmem(fb, n) -= fac * mult * inco(fd, m); //autre face calculee
198 else for (d = 0; d < D; d++) //CL de Dirichlet
199 secmem(fb, n) -= fac * nf(fb, d) / fs(fb) * ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), N * d + m);
200 if (comp) secmem(fb, n) += fac * inco(fb, m); //partie v div(alpha rho v)
201 if (!mat) continue;
202 if (fd >= 0 && fcl(fd, 0) < 2) (*mat)(N * fb + n, N * fd + m) += fac * mult;
203 if (comp) (*mat)(N * fb + n, N * fb + m) -= fac;
204 }
205 }
206 else for (j = 0; j < 2; j++)
207 for (eb = f_e(f, j), d = 0; d < D; d++)
208 if (std::fabs(nf(fb, d)) > 1e-6 * fs(fb))
209 for (n = 0; n < N; n++)
210 for (m = 0; m < N; m++)
211 if (dfac(j, n, m))
212 {
213// pas d'equivalence : mu_f * n_f * operateur aux elements
214 double fac = (i ? -1 : 1) * vfd(fb, e != f_e(fb, 0)) * dfac(j, n, m) / ve(e) * nf(fb, d) / fs(fb);
215 secmem(fb, n) -= fac * inco(nf_tot + D * eb + d, m);
216 if (comp) secmem(fb, n) += fac * inco(nf_tot + D * e + d, m);
217 if (!mat || !fac) continue;
218 (*mat)(N * fb + n, N * (nf_tot + D * eb + d) + m) += fac;
219 if (comp) (*mat)(N * fb + n, N * (nf_tot + D * e + d) + m) -= fac;
220 }
221 }
222 for (j = 0; j < 2; j++)
223 for (eb = f_e(f, j), d = 0; d < D; d++)
224 for (n = 0; n < N; n++)
225 for (m = 0; m < N; m++)
226 if (dfac(j, n, m)) //partie "elem"
227 {
228 double fac = (i ? -1 : 1) * dfac(j, n, m);
229 secmem(nf_tot + D * e + d, n) -= fac * (eb >= 0 ? inco(nf_tot + D * eb + d, m)
230 : ref_cast(Dirichlet, cls[fcl(f, 1)].valeur()).val_imp(fcl(f, 2), N * d + m));
231 if (comp) secmem(nf_tot + D * e + d, n) += fac * inco(nf_tot + D * e + d, m); //partie v div(alpha rho v)
232 if (!mat) continue;
233 if (eb >= 0) (*mat)(N * (nf_tot + D * e + d) + n, N * (nf_tot + D * eb + d) + m) += fac;
234 if (comp) (*mat)(N * (nf_tot + D * e + d) + n, N * (nf_tot + D * e + d) + m) -= fac;
235 }
236 }
237 }
