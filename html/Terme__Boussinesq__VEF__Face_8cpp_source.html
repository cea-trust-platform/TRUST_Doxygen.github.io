46DoubleTab& Terme_Boussinesq_VEF_Face::ajouter(DoubleTab& resu) const
47{
48 const Zone_VEF& zone_VEF = la_zone_VEF.valeur();
49 const IntTab& face_voisins = zone_VEF.face_voisins();
50 const DoubleTab& face_normales = zone_VEF.face_normales();
51 const DoubleTab& xp = zone_VEF.xp();
52 const DoubleTab& xv = zone_VEF.xv();
53 const DoubleVect& porosite_surf = zone_VEF.porosite_face();
54 const DoubleTab& param = equation_scalaire().inconnue().valeurs();
55 const DoubleTab& beta_valeurs = beta().valeur().valeurs();
56 const DoubleVect& g = gravite().valeurs();
57
58 int nb_dim = param.line_size(), nbcomp = resu.line_size();
59// Verifie la validite de T0:
60 check();
61
62// Boucle sur toutes les faces
63 int nb_faces = zone_VEF.nb_faces();
64 for (int face=0; face<nb_faces; face++)
65 {
66 int elem1 = face_voisins(face,0), elem2 = face_voisins(face,1);
67 double delta_param = 0;
68 for (int dim=0; dim<nb_dim; dim++)
69 delta_param += valeur(beta_valeurs,elem1,elem2,dim)*(Scalaire0(dim)-param(face,dim));
70
71 for (int comp=0; comp<dimension; comp++)
72 {
73 double delta_coord;
74 if (elem2==-1) //!< Face de bord
75 delta_coord = xv(face,comp) - xp(elem1,comp);
76 else
77 delta_coord = xp(elem2,comp) - xp(elem1,comp);
78 for (int compo=0; compo<nbcomp; compo++)
79 resu(face,compo) += delta_param*delta_coord*face_normales(face,compo)*g(comp)*porosite_surf(face);
80 }
81 }
82 return resu;
83}
