116//! Effectue un pas de temps d'Euler explicite<br>sur l'equation passee en parametre.
117/** \return int<br>
118Signification: renvoie toujours 1<br>
119Contraintes: */
120int Schema_Explicite_Multi_TimeStep_base::faire_un_pas_de_temps_eqn_base(Equation_base& eqn)
121{
122 authorized_equation(eqn);
123
124 double time_step = pas_de_temps(); //tn+1-tn
125 double inv_time_step = 1/time_step;
126
127// Un
128 const DoubleTab& present = eqn.inconnue().valeurs();
129 Debog::verifier("Schema_Explicite_Multi_TimeStep_base::faire_un_pas_de_temps_eqn_base, present",present);
130
131// Un+1
132 DoubleTab& futur = eqn.inconnue().futur();
133
134// sert pour la pression et les couplages
135 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
136
137 DoubleTab dudt(futur);
138 /*
139 if (nb_pas_dt() > nb_pas_dt_seuil())
140 {
141 Cout <<"Use of "<<que_suis_je()<<" scheme"<<finl;
142 }
143 else
144 {
145 Cout <<"Use of explicit Euler scheme for time step "<<nb_pas_dt()<<" in "<<que_suis_je()<<" scheme"<<finl;
146 } */
147
148// Compute du/dt
149 eqn.derivee_en_temps_inco(dudt);
150
151// Contribution de l'inconnue au temps n
152 futur = dudt;
153 futur *= time_step;
154 futur += present;
155
156 dudt=futur;
157 dudt-=present;
158 dudt*=inv_time_step;
159
160 Debog::verifier("Schema_Explicite_Multi_TimeStep_base::faire_un_pas_de_temps_eqn_base, dudt",dudt);
161
162 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(),temps_courant()+pas_de_temps());
163 update_critere_statio(dudt, eqn);
164 futur.echange_espace_virtuel();
165
166// Debog::verifier("Schema_Explicite_Multi_TimeStep_base::faire_un_pas_de_temps_eqn_base, futur fin",futur);
167 return 1;
168}

170void Schema_Explicite_Multi_TimeStep_base::authorized_equation(const Equation_base& eqn)
171{
172 Nom equation_name(eqn.que_suis_je());
173 bool authorized = true;
174
175 authorized &= (bool) equation_name.find("FT");
176 authorized &= (bool) equation_name.find("Front_Tracking");
177 authorized &= (bool) equation_name.find("QC");
178 authorized &= (bool) equation_name.find("Quasi_Compressible");
179
180 if (!authorized)
181 {
182 Cerr<<"Error in "<<que_suis_je()<<"::authorized_equation()"<<finl;
183 Cerr<<"Equation "<<equation_name<<" is not allowed with "<<que_suis_je()<<" time scheme"<<finl;
184 Process::exit();
185 }
