60int Entree_Fichier_base::ouvrir(const char* name, IOS_OPEN_MODE mode)
61{
62 if(ifstream_)
63 delete ifstream_;
64 IOS_OPEN_MODE ios_mod=mode;
65 if (bin_)
66 {
67 ios_mod=ios_mod|ios::binary;
68 }
69 ifstream_ = new ifstream(name,ios_mod);
70 int ok = ifstream_->good();
71 set_istream(ifstream_);
72#ifdef INT_is_64_
73 if (bin_)
74 {
75 Nom test;
76 (*this) >> test;
77 if (test!="INT64")
78 {
79 Cerr<<"Opening " <<name<< " which is an int32 binary file..."<<finl;
80 delete ifstream_;
81// on recharge le fichier
82 ifstream_ = new ifstream(name,ios_mod);
83 is_different_int_size_=true;
84 ok = ifstream_->good();
85 set_istream(ifstream_);
86 }
87 }
88#else
89 if (bin_)
90 {
91 Nom test;
92 (*this) >> test;
93 if (test=="INT64")
94 {
95 Cerr<<"Opening " << name<< " which is an int64 binary file..."<<finl;
96 is_different_int_size_=true;
97 }
98 else
99 {
100 delete ifstream_;
101// on recharge le fichier
102 ifstream_ = new ifstream(name,ios_mod);
103 ok = ifstream_->good();
104 set_istream(ifstream_);
105 }
106 }
107#endif
108 return ok;
109}
