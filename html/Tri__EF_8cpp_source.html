41//! remplit le tableau face_normales dans la Zone_EF
42void Tri_EF::normale(int num_Face,DoubleTab& Face_normales,
43 const IntTab& Face_sommets,
44 const IntTab& Face_voisins,
45 const IntTab& elem_faces,
46 const Zone& zone_geom) const
47{
48 const DoubleTab& les_coords = zone_geom.domaine().coord_sommets();
49
50 double x1,y1;
51 double nx,ny;
52 int no3;
53 int f0;
54 int n0 = Face_sommets(num_Face,0);
55 int n1 = Face_sommets(num_Face,1);
56 x1 = les_coords(n0,0)-les_coords(n1,0);
57 y1 = les_coords(n0,1)-les_coords(n1,1);
58 nx = -y1;
59 ny = x1;
60
61// Orientation de la normale de elem1 vers elem2
62// pour cela recherche du sommet de elem1 qui n'est pas sur la Face
63 int elem1 = Face_voisins(num_Face,0);
64 if ( (f0 = elem_faces(elem1,0)) == num_Face )
65 f0 = elem_faces(elem1,1);
66 if ( (no3 = Face_sommets(f0,0)) != n0 && no3 != n1 )
67 ;
68 else
69 no3 = Face_sommets(f0,1);
70
71 x1 = les_coords(no3,0) - les_coords(n0,0);
72 y1 = les_coords(no3,1) - les_coords(n0,1);
73
74 if ( (nx*x1+ny*y1) > 0 )
75 {
76 Face_normales(num_Face,0) = - nx;
77 Face_normales(num_Face,1) = - ny;
78 }
79 else
80 {
81 Face_normales(num_Face,0) = nx;
82 Face_normales(num_Face,1) = ny;
83 }
84}

88void Tri_EF::calcul_vc(const ArrOfInt& Face,ArrOfDouble& vc,
89 const ArrOfDouble& vs,const DoubleTab& vsom,
90 const Champ_Inc_base& vitesse,int type_cl) const
91{
92 const Zone_EF& zone_EF = ref_cast(Zone_EF,vitesse.zone_dis_base());
93 const DoubleVect& porosite_face = zone_EF.porosite_face();
94// Cerr << " type_cl " << type_cl << finl;
95 switch(type_cl)
96 {
97 case 0: //!< le triangle n'a pas de Face de Dirichlet
98 {
99 vc[0] = vs[0]/3;
100 vc[1] = vs[1]/3;
101 break;
102 }
103
104 case 1: //!< le triangle a une Face de Dirichlet :la Face 2
105 {
106 vc[0]= vitesse(Face[2],0)*porosite_face[Face[2]];
107 vc[1]= vitesse(Face[2],1)*porosite_face[Face[2]];
108// vc[0]= vitesse(Face[2],0);
109// vc[1]= vitesse(Face[2],1);
110 break;
111 }
112
113 case 2: //!< le triangle a une Face de Dirichlet :la Face 1
114 {
115 vc[0]= vitesse(Face[1],0)*porosite_face[Face[1]];
116 vc[1]= vitesse(Face[1],1)*porosite_face[Face[1]];
117// vc[0]= vitesse(Face[1],0);
118// vc[1]= vitesse(Face[1],1);
119 break;
120 }
121
122 case 4: //!< le triangle a une Face de Dirichlet :la Face 0
123 {
124 vc[0]= vitesse(Face[0],0)*porosite_face[Face[0]];
125 vc[1]= vitesse(Face[0],1)*porosite_face[Face[0]];
126// vc[0]= vitesse(Face[0],0);
127// vc[1]= vitesse(Face[0],1);
128 break;
129 }
130
131 case 3: //!< le triangle a deux faces de Dirichlet :les faces 1 et 2
132 {
133 vc[0]= vsom(0,0);
134 vc[1]= vsom(0,1);
135 break;
136 }
137
138 case 5: //!< le triangle a deux faces de Dirichlet :les faces 0 et 2
139 {
140 vc[0]= vsom(1,0);
141 vc[1]= vsom(1,1);
142 break;
143 }
144
145 case 6: //!< le triangle a deux faces de Dirichlet :les faces 0 et 1
146 {
147 vc[0]= vsom(2,0);
148 vc[1]= vsom(2,1);
149 break;
150 }
151
152 } //!< fin du switch
153
154}

159//! calcule les coord xg du centre d'un element non standard<br>calcule aussi idirichlet=nb de faces de Dirichlet de l'element<br>si idirichlet=2, n1 est le numero du sommet confondu avec G
160void Tri_EF::calcul_xg(DoubleVect& xg, const DoubleTab& x,
161 const int type_elem_Cl,int& idirichlet,int& n1,int& ,int& ) const
162{
163 int j,dim=xg.size();
164 switch(type_elem_Cl)
165 {
166
167 case 0: //!< le triangle n'a pas de Face de dirichlet: il a 3 Facettes le point G est le barycentre des sommets du triangle
168//
169 {
170 for (j=0; j<dim; j++)
171 xg[j]=(x(0,j)+x(1,j)+x(2,j))/3;
172
173 idirichlet=0;
174 break;
175 }
176
177 case 1: //!< le triangle a une Face de dirichlet: la Face 2 le point G est le barycentre des sommets de la Face 2
178//
179
180 {
181 for (j=0; j<dim; j++)
182 xg[j]=(x(0,j)+x(1,j))/2;
183
184 idirichlet=1;
185 break;
186 }
187
188 case 2: //!< le triangle a une Face de dirichlet: la Face 1 le point G est le barycentre des sommets de la Face 1
189//
190
191 {
192 for (j=0; j<dim; j++)
193 xg[j]=(x(0,j)+x(2,j))/2;
194
195 idirichlet=1;
196 break;
197 }
198
199 case 4: //!< le triangle a une Face de dirichlet: la Face 0 le point G est le barycentre des sommets de la Face 0
200//
201
202 {
203 for (j=0; j<dim; j++)
204 xg[j]=(x(1,j)+x(2,j))/2;
205
206 idirichlet=1;
207 break;
208 }
209
210 case 6 : //!< le triangle a deux faces de Dirichlet : les faces 0,1 le point G est le sommet 2 du triangle
211//
212
213 {
214 for (j=0; j<dim; j++)
215 xg[j]=x(2,j);
216
217 idirichlet=2;
218 n1 = 2;
219 break;
220
221 }
222
223 case 5 : //!< le triangle a deux faces de Dirichlet : les faces 0,2 le point G est le sommet 1 du triangle
224//
225
226 {
227 for (j=0; j<dim; j++)
228 xg[j]=x(1,j);
229
230 idirichlet=2;
231 n1 = 1;
232 break;
233
234 }
235
236 case 3 : //!< le triangle a deux faces de Dirichlet : les faces 1,2 le point G est le sommet 0 du triangle
237//
238

240 for (j=0; j<dim; j++)
241 xg[j]=x(0,j);
