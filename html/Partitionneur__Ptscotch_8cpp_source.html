77//! Calcule le graphe de connectivite, appelle le partitionneur<br>et remplit elem_part (pour chaque element, numero de la partie qui lui<br>est attribuee).
78void Partitionneur_Ptscotch::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
79{
80#ifndef PTSCOTCH_
81 Cerr << "Ptscotch is not compiled with this version. Use another partition tool like Tranche." << finl;
82 Process::exit();
83#else
84 if (!ref_domaine_.non_nul())
85 {
86 Cerr << "Error in Partitionneur_Ptscotch::construire_partition\n";
87 Cerr << " The domain has not been associated" << finl;
88 exit();
89 }
90 if (nb_parties_ <= 0)
91 {
92 Cerr << "Error in Partitionneur_Ptscotch::construire_partition\n";
93 Cerr << " The parts number has not been initialized" << finl;
94 exit();
95 }
96
97// Cas particulier: si nb_parts == 1, METIS ne veut rien faire...
98// ToDo: I don't know is that's the case with Ptscotch
99 if (nb_parties_ == 1)
100 {
101
102 int nb_elem = ref_domaine_.valeur().zone(0).nb_elem();
103 elem_part.resize_array(nb_elem);
104 elem_part = 0;
105 return;
106 }
107
108 if (ref_domaine_.valeur().zone(0).nb_elem() == 0)
109 return;
110
111 Cerr << "Partitionneur_Ptscotch::construire_partition" << finl;
112 Cerr << " Construction of graph connectivity..." << finl;
113 Static_Int_Lists graph_elements_perio;
114// const Domaine& dom = ref_domaine_.valeur();
115 Domain_Graph graph;
116 graph.construire_graph_elem_elem(ref_domaine_.valeur(), liste_bords_periodiques_,
117 use_weights_,
118 graph_elements_perio);
119
120
121 const int n = ref_domaine_.valeur().zone(0).nb_elem();
122 int* partition = new int[n];
123
124 SCOTCH_randomReset();
125 SCOTCH_Dgraph scotch_graph;
126 SCOTCH_dgraphInit(&scotch_graph, Comm_Group_MPI::get_trio_u_world());
127 SCOTCH_dgraphBuild(&scotch_graph,
128 0, //!< baseval , base first indice 0
129 n, //!< vertlocnbr , nb of local graph nodes
130 n, //!< vertlocmax , should be set to vertlocnbr for graphs without holes
131 graph.xadj.addr(), //!< vertloctab[vertnbr+1] , index vertex table
132 0, //!< vendloctab , index end vertex table if disjoint, set to zero
133 graph.ewgts.addr(), //graph.ewgts, //!< veloloctab , graph vertices loads, set to zero
134 0, //!< vlblocltab , vertex label array : global vertex index
135 graph.xadj[n], //!< edgelocnbr , number of edges
136 graph.xadj[n], //!< edgelocsiz , same as edgelocnbr if edgeloctab is compact
137 graph.adjncy.addr(), //!< edgeloctab[edgelocnbr], global indexes of edges
138 graph.edgegsttab.addr(), //!< edgegsttab , optional, should be computed internally, set to zero
139 0); //!< edloloctab , graph edges loads, set to zero
140
141 SCOTCH_Strat scotch_strategy;
142 SCOTCH_stratInit(&scotch_strategy);
143
144 SCOTCH_dgraphPart(&scotch_graph,nb_parties_,&scotch_strategy,partition);
145
146
147 SCOTCH_stratExit(&scotch_strategy);
148 SCOTCH_dgraphExit(&scotch_graph);
149
150 MD_Vector_tools::creer_tableau_distribue(ref_domaine_.valeur().zone(0).md_vector_elements(), elem_part);
151 for (int i = 0; i < n; i++)
152 elem_part[i] = partition[i];
153
154 delete [] partition;
155
156// Correction de la partition pour la periodicite. (***)
157 if (graph_elements_perio.get_nb_lists() > 0)
158 {
159 Cerr << "Correction of the partition for the periodicity" << finl;
160 corriger_bords_avec_liste(ref_domaine_.valeur(),
161 liste_bords_periodiques_,
162 0,
163 elem_part);
164 Cerr << " If this number is high, we can improve the splitting with the option use_weights\n"
165 << " but it takes more memory)" << finl;
166 }
167
168 Cerr << "Correction elem0 on processor 0" << finl;
169 corriger_elem0_sur_proc0(elem_part);
170 elem_part.echange_espace_virtuel();
171
172#endif
173}
