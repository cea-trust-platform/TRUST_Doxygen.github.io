38void PlaqThVDF::mettre_a_jour(double )
39{
40 const Equation_base& eqn = ma_zone_cl_dis->equation();
41 const Zone_VDF& la_zone_VDF=ref_cast(Zone_VDF, eqn.zone_dis().valeur());
42 const Front_VF& front= ref_cast(Front_VF,frontiere_dis());
43
44 const Milieu_base& le_milieu=eqn.probleme().milieu();
45 h/=(le_milieu.masse_volumique()(0,0)*le_milieu.capacite_calorifique()(0,0));
46
47// Calcul de himp :
48 const RefObjU& modele_turbulence = eqn.get_modele(TURBULENCE);
49 if (modele_turbulence.non_nul() && sub_type(Modele_turbulence_scal_base,modele_turbulence.valeur()))
50 {
51 const Modele_turbulence_scal_base& modele = ref_cast(Modele_turbulence_scal_base,modele_turbulence.valeur());
52 const Turbulence_paroi_scal& loipar = ref_cast(Turbulence_paroi_scal,modele.loi_paroi());
53 Champ_front_calc& ch=ref_cast(Champ_front_calc, T_ext().valeur());
54 ch.creer(eqn.probleme().le_nom(),frontiere_dis().le_nom(),eqn.inconnue().le_nom());
55// const Milieu_base& le_milieu=eqn.probleme().milieu();
56 h_imp_.typer("Champ_front_uniforme");
57 DoubleTab& tab = h_imp_->valeurs();
58 h_imp_->fixer_nb_comp(1);
59 int nbfs2 = front.nb_faces()/2;
60 tab.resize(front.nb_faces(),1);
61 int boundary_index=-1;
62 int nb_boundaries=la_zone_VDF.zone().nb_front_Cl();
63 for (int n_bord=0; n_bord<nb_boundaries; n_bord++)
64 {
65 if (la_zone_VDF.front_VF(n_bord).le_nom() == front.le_nom())
66 boundary_index=n_bord;
67 }
68 for(int face=0; face < nbfs2; face++)
69 {
70// double e1 = loipar.d_equiv(face);
71// double e2 = loipar.d_equiv(nbfs2+face);
72 int local_face=la_zone_VDF.front_VF(boundary_index).num_local_face(face);
73 int local_face2=la_zone_VDF.front_VF(boundary_index).num_local_face(nbfs2+face);
74 double e1 = loipar.valeur().equivalent_distance(boundary_index,local_face);
75 double e2 = loipar.valeur().equivalent_distance(boundary_index,local_face2);
76 tab(face,0) = tab(nbfs2+face,0) =
77 2./(1./h+e1/le_milieu.diffusivite()(0,0)
78 +e2/le_milieu.diffusivite()(0,0));
79 }
80 }
81 else if (sub_type(Convection_Diffusion_Temperature,eqn))
82 {
83 Champ_front_calc& ch = ref_cast(Champ_front_calc,T_ext().valeur());
84 ch.creer(eqn.probleme().le_nom(),frontiere_dis().le_nom(),
85 eqn.inconnue().le_nom());
86// const Milieu_base& le_milieu=eqn.probleme().milieu();
87 h_imp_.typer("Champ_front_uniforme");
88 DoubleTab& tab= h_imp_->valeurs();
89 tab.resize(1,1);
90 h_imp_->fixer_nb_comp(1);
91 double e1 = la_zone_VDF.dist_norm_bord(front.num_premiere_face());
92 double e2 = la_zone_VDF.dist_norm_bord(front.nb_faces()/2);
93 tab(0,0) = 2./(1./h+e1/le_milieu.diffusivite()(0,0)
94 +e2/le_milieu.diffusivite()(0,0));
95 }
96
97// Calcul de T_ext :
98 const DoubleTab& Temp= eqn.inconnue()->valeurs();
99 DoubleTab& tab= T_ext()->valeurs();
100 tab.resize(front.nb_faces(),1);
101 int face, el1, el2;
102 int premiere = front.num_premiere_face();
103 int nbfs2=front.nb_faces()/2;
104 int derniere = premiere + nbfs2;
105 for(face=premiere; face < derniere; face++)
106 {
107 int num=face-premiere;
108 el1=la_zone_VDF.face_voisins(face,0);
109 el2=la_zone_VDF.face_voisins(face+nbfs2,1);
110 tab(num+nbfs2,0)=tab(num,0)=(Temp(el1)+Temp(el2))*0.5;
111 };
112}
