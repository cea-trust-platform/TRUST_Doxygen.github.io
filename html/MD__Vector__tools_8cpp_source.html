48template <class VECT, class TAB>
49static void creer_tableau_distribue_(const MD_Vector& md, VECT& v, Array_base::Resize_Options opt)
50{
51 if (v.get_md_vector().non_nul())
52 {
53// Si ce message apparait et qu'on est sur de ce qu'on fait,
54// faire ceci avant d'appeler creer_tableau_distribue:
55// vect.set_md_vector(MD_Vector()); // Annule le descripteur et garde le vecteur.
56// ou
57// vect.reset(); // Detruit completement le contenu du vecteur
58 Cerr << "Internal error in MD_Vector_tools::creer_tableau_distribue:\n"
59 << " Vector already has a parallel vector structure" << finl;
60 Process::exit();
61 }
62 int sz = md.valeur().get_nb_items_tot();
63// Attention, sz_r peut valoir -1 dans certains cas. Alors le test n==sz_r sera toujours faux,
64// mais c'est bien ce qu'on veut...
65 int sz_r = md.valeur().get_nb_items_reels();
66 int err = 0;
67 TAB* vv = dynamic_cast<TAB*>(&v);
68 if (vv)
69 {
70 TAB& t = *vv;
71 const int n = t.dimension_tot(0);
72 if (n == sz_r || n == 0) t.resize_dim0(sz, opt);
73 else if (n == sz) { /* ok no resize */ }
74 else { err = 1; }
75 }
76 else
77 {
78 const int n = v.size_totale();
79 if (n == sz_r || n == 0) { v.resize(sz, opt); }
80 else if (n == sz) { /* ok no resize */ }
81 else { err = 1; }
82 }
83 if (err)
84 {
85 Cerr << "Internal error in MD_Vector_tools::creer_tableau_distribue:\n"
86 << " Input vector has wrong size or dimension(0): expected 0 or size_reele=" << sz_r << finl;
87 Process::exit();
88 }
89 v.set_md_vector(md);
90}

101//! transforme v en un tableau parallele ayant la structure md.<br>md doit est non nul !<br>Les dimension(i>=1) du tableau v (si c'est un IntTab ou DoubleTab) sont conservees,<br>les dimension(0) et dimension_tot(0) sont modifiees en fonction du nombre d'items<br>specifies dans le MD_Vector.<br>La dimension initiale du vecteur doit etre soit 0, soit md.get_nb_items_reels(),<br>soit md.get_nb_items_tot(). Si besoin, la taille du tableau est modifiee et on<br>initialise le tableau selon opt.<br>ATTENTION, echange_espace_virtuel() n'est PAS appele.<br>Les cases virtuelles ne sont pas initialisees...
102void MD_Vector_tools::creer_tableau_distribue(const MD_Vector& md, Array_base& v, Array_base::Resize_Options opt)
103{
104 if (!md.non_nul())
105 {
106 Cerr << "Error in MD_Vector_tools::creer_tableau_distribue(): MD_Vector is null" << finl;
107 Process::exit();
108 }
109
110 IntVect* intV = dynamic_cast<IntVect*>(&v);
111 DoubleVect* doubleV = dynamic_cast<DoubleVect*>(&v);
112
113 if (intV) creer_tableau_distribue_<IntVect, IntTab>(md, *intV, opt);
114 else if (doubleV) creer_tableau_distribue_<DoubleVect, DoubleTab>(md, *doubleV, opt);
115 else
116 {
117 Cerr << "Internal error in MD_Vector_tools::creer_tableau_distribue(const MD_Vector & md, Array_base & v):\n"
118 << " Invalid array type " << v.que_suis_je()
119 << "\n Array must be a subtype of IntVect or DoubleVect" << finl;
120 Process::exit();
121 }
122}

124template <typename _TYPE_>
125void echange_espace_virtuel_(const MD_Vector& md, TRUSTVect<_TYPE_>& v, const Echange_EV_Options& opt = echange_ev_opt_default)
126{
127 const MD_Vector_base& mdv = md.valeur();
128 const int is_double = (std::is_same<_TYPE_,double>::value) ? 1 : 0;
129
130 if (md == last_md && v.line_size() == last_linesize && last_isdouble == is_double && last_opt == opt) { /* Do nothing si double et pas 1er passage */ }
131 else
132 {
133 last_md = md;
134 last_linesize = v.line_size();
135 last_isdouble = is_double;
136 last_opt = opt;
137 comm.begin_init();
138 mdv.initialize_comm(opt, comm, v);
139 comm.end_init();
140 }
141 comm.begin_comm();
142 mdv.prepare_send_data(opt, comm, v);
143 comm.exchange();
144 mdv.process_recv_data(opt, comm, v);
145 comm.end_comm();
146}

148template<typename _TYPE_>
149void echange_espace_virtuel1_(const MD_Vector& md, TRUSTVect<_TYPE_>& v, MD_Vector_tools::Operations_echange opt)
150{
151 switch(opt)
152 {
153 case MD_Vector_tools::ECHANGE_EV:
154 echange_espace_virtuel_(md, v, echange_ev_opt_default);
155 break;
156 case MD_Vector_tools::EV_SOMME:
157 echange_espace_virtuel_(md, v, Echange_EV_Options(Echange_EV_Options::SUM));
158 break;
159 case MD_Vector_tools::EV_SOMME_ECHANGE:
160 echange_espace_virtuel_(md, v, Echange_EV_Options(Echange_EV_Options::SUM));
161 echange_espace_virtuel_(md, v, echange_ev_opt_default);
162 break;
163 case MD_Vector_tools::EV_MAX:
164 echange_espace_virtuel_(md, v, Echange_EV_Options(Echange_EV_Options::MAX));
165 break;
166 case MD_Vector_tools::EV_MINCOL1:
167 echange_espace_virtuel_(md, v, Echange_EV_Options(Echange_EV_Options::MINCOL1));
168 break;
169 default:
170 Cerr << "echange_espace_virtuel1_ operation not implemented" << finl;
171 Process::exit();
172 }
173}

200template <class FlagsType>
201static int get_seq_flags(const MD_Vector_base& mdv, FlagsType& flags, int startidx, int line_size)
202{
203 int count = 0;
204 if (sub_type(MD_Vector_base2, mdv))
205 {
206 const MD_Vector_base2& ms = ref_cast(MD_Vector_base2, mdv);
207 const int nblocs = ms.blocs_items_to_sum_.size_array() >> 1;
208 const int *ptr = ms.blocs_items_to_sum_.addr();
209 int j = startidx;
210 for (int i = nblocs; i; i--)
211 {
212// On pourrait optimiser pour les ArrOfBit en ajoutant a cette classe
213// une methode setflag/clearflag(range_begin, range_end);
214 int jmax = startidx + (*(ptr++)) * line_size; //!< debut du prochain bloc d'items sequentiels
215 for (; j < jmax; j++)
216 clearflag(flags, j);
217 jmax = startidx + (*(ptr++)) * line_size; //!< fin du bloc d'items sequentiels
218 assert(jmax > j);
219 count += jmax - j;
220 for (; j < jmax; j++)
221 setflag(flags, j);
222 }
223// Fin du remplissage
224 int sz = mdv.get_nb_items_tot() * line_size;
225 for (; j < sz; j++)
226 clearflag(flags, j);
227 }
228 else
229 {
230 Cerr << "Error in MD_Vector_tools::get_sequential_items_flags: unsupported MD_Vector type " << mdv.que_suis_je() << finl;
231 Process::exit();
232 }
233 return count;
234}

236int MD_Vector_tools::get_sequential_items_flags(const MD_Vector& md, ArrOfBit& flags, int line_size)
237{
238 const MD_Vector_base& mdv = md.valeur();
239 const int sz = mdv.get_nb_items_tot() * line_size;
240// setbit et clearbit font |= et &=, donc valgrind ne voit pas qu'on initialise tout
241// le tableau. On initialise tout explicitement:
242 flags.resize_array(sz);
243 flags = 1; //!< valeur par defaut la plus courante (0 est le cas particulier), voir setflag(ArrOfBit)
244 return get_seq_flags(mdv, flags, 0, line_size);
245}

262//! Renvoie le nombre d'items extraits (pas le nombre de blocs)
263static int extract_blocs(const ArrOfInt src, const ArrOfInt& renum, ArrOfInt& dest)
264{
265 const int nblocs_src = src.size_array() / 2;
266 dest.set_smart_resize(1);
267 dest.resize_array(0);
268 int end_last_bloc = -1;
269 int count = 0;
270 for (int i = 0; i < nblocs_src; i++)
271 {
272 const int deb = src[i*2];
273 const int fin = src[i*2+1];
274 for (int j = deb; j < fin; j++)
275 {
276 const int rj = renum[j];
277 if (rj >= 0)
278 {
279 count++;
280// Blocs construction assumes that renum is ordered
281 assert(rj >= end_last_bloc);
282 if (rj == end_last_bloc)
283 {
284// item is contiguous to last bloc in destination array, increase bloc size:
285 end_last_bloc++;
286 dest[dest.size_array()-1] = end_last_bloc;
287 }
288 else
289 {
290// non contiguous item, create new bloc
291 end_last_bloc = rj;
292 dest.append_array(end_last_bloc);
293 end_last_bloc++;
294 dest.append_array(end_last_bloc);
295 }
296 }
297 }
298 }
299 return count;
300}

302static void creer_md_vect_renum(const IntVect& renum, const MD_Vector_std& src, MD_Vector_std& dest)
303{
304// Calcul de nb_items_tot_ et nb_items_reels_
305// nb_items_tot_ est le nombre d'items pour lesquels renum[i] >= 0
306// Les items reels sont ceux qui etaient reels dans le tableau d'origine et qui sont a conserver
307 {
308 const int src_size = src.get_nb_items_tot();
309 const int src_size_r = src.get_nb_items_reels();
310 dest.nb_items_tot_ = 0;
311 dest.nb_items_reels_ = 0;
312 for (int i = 0; i < src_size; i++)
313 {
314 if (renum[i] >= 0)
315 {
316 dest.nb_items_tot_++;
317 if (i < src_size_r)
318 dest.nb_items_reels_++;
319 }
320 }
321// Si l'attribut nb_items_reels_ est invalide dans le vecteur source, il est invalide dans le
322// vecteur resultat:
323 if (src_size_r < 0)
324 dest.nb_items_reels_ = -1;
325 }
326// Remplissage de blocs_items_to_sum_ et blocs_items_to_compute_
327// Ce sont les items pour lesquels renum[i] >= 0 et qui sont dans les blocs_items_to_sum_
328// et blocs_items_to_compute_ du descripteur source.
329 extract_blocs(src.blocs_items_to_sum_, renum, dest.blocs_items_to_sum_);
330 dest.nb_items_seq_local_ = extract_blocs(src.blocs_items_to_compute_, renum, dest.blocs_items_to_compute_);
331 dest.nb_items_seq_tot_ = Process::mp_sum(dest.nb_items_seq_local_);
332
333// ********************************************************
334// Calcul des items a recevoir: ce sont les items pour lesquels renum[i] >= 0 et qui etaient a
335// recevoir dans le descripteur source.
336// Chaque fois qu'on trouve un item a recevoir, il faut informer le processeur voisin qui possede l'item
337// que c'est item doit etre envoye (utilisation du schema_comm). Pour identifier ces items, on utilise
338// sont rang dans le tableau des items a envoyer dans le descripteur source (variable item_rank)
339//
340 const int nb_pe_voisins = src.pe_voisins_.size_array();
341 Schema_Comm schema_comm;
342 schema_comm.set_send_recv_pe_list(src.pe_voisins_, src.pe_voisins_);
343 schema_comm.begin_comm();
344// Calcul des listes items_to_recv
345// index+data forment les deux tableau d'un StaticIntLists.
346// On construit pour commencer autant de listes que de processeurs voisins dans le descripteur source.
347// On retirera a la fin les processeurs qui n'echangent pas de donnees
348 ArrOfInt dest_items_recv_index;
349 ArrOfInt dest_items_recv_data;
350 ArrOfInt dest_blocs_recv_index;
351 ArrOfInt dest_blocs_recv_data;
352
353 ArrOfInt tmp;
354 tmp.set_smart_resize(1);
355 {
356 dest_items_recv_index.resize_array(nb_pe_voisins + 1, Array_base::NOCOPY_NOINIT);
357 dest_items_recv_index[0] = 0;
358 dest_blocs_recv_index.resize_array(nb_pe_voisins + 1, Array_base::NOCOPY_NOINIT);
359 dest_blocs_recv_index[0] = 0;
360 dest.blocs_items_count_.resize_array(nb_pe_voisins, Array_base::NOCOPY_NOINIT);
361 dest.nb_items_to_items_.resize_array(nb_pe_voisins, Array_base::NOCOPY_NOINIT);
362// Preallocation de la taille maxi
363 dest_items_recv_data.resize_array(src.items_to_recv_.get_data().size_array(), Array_base::NOCOPY_NOINIT);
364 dest_items_recv_data.set_smart_resize(1);
365 dest_items_recv_data.resize_array(0);
366// On ne peut pas prevoir le nombre de blocs, il peut y en avoir plus que dans la source
367 dest_blocs_recv_data.set_smart_resize(1);
368 for (int i_pe = 0; i_pe < nb_pe_voisins; i_pe++)
369 {
370 tmp.resize_array(0);
371 int item_rank = 0; //!< Compteur d'items recus dans le descripteur source 1) Extraction des items individuels (remplissage de dest_items_recv_index, dest_items_recv_data et tmp)
372//
373//
374 const int n = src.items_to_recv_.get_list_size(i_pe);
375 for (int i = 0; i < n; i++)
376 {
377 const int j = src.items_to_recv_(i_pe, i);
378 const int rj = renum[j];
379 if (rj >= 0)
380 {
381// Ajout de cet item
382 dest_items_recv_data.append_array(rj);
383// Demander au processeur voisin d'envoyer cet item
384 tmp.append_array(item_rank);
385 }
386 item_rank++;
387 }
388// Initialisation de la taille de la liste d'items pour ce processeur:
389 dest_items_recv_index[i_pe+1] = dest_items_recv_data.size_array();
390
391// 2) Extraction des blocs d'items recus de ce processeur
392// (remplissage de dest_blocs_recv_index dest_blocs_recv_data dest.blocs_items_count_ et tmp)
393 const int nblocs = src.blocs_to_recv_.get_list_size(i_pe) / 2;
394 int received_count = 0; //!< nombre d'items recus de ce processeur
395 for (int ibloc = 0; ibloc < nblocs; ibloc++)
396 {
397 const int jdeb = src.blocs_to_recv_(i_pe, ibloc * 2);
398 const int jfin = src.blocs_to_recv_(i_pe, ibloc * 2 + 1);
399 int last_added = -2; //!< ne peut jamais etre egal a rj-1 Pour chaque item du bloc, s'il doit etre conserve, le reporter dans les blocs a recevoir:
400//
401 for (int j = jdeb; j < jfin; j++)
402 {
403 const int rj = renum[j];
404 if (rj >= 0)
405 {
406 if (rj <= last_added)
407 {
408// Les items dans les blocs a recevoir doivent etre tries dans l'ordre croissant
409// (sinon cet algorithme ne marche pas, il faut en utiliser un autre pour
410// construire les blocs, et en plus le comptage des items reels ne veut peut-etre plus rien dire)
411 Cerr << "Error in creer_md_vect_renum: renum array is not sorted: cannot extract blocs to recv" << finl;
412 Process::exit();
413 }
414// Cet item est-il contigu avec le precedent ?
415 if (last_added == rj-1)
416 {
417// Oui, augmenter la taille du dernier bloc
418 const int k = dest_blocs_recv_data.size_array();
419// fin de bloc = indice du dernier plus 1
420 dest_blocs_recv_data[k-1] = rj+1;
421 }
422 else
423 {
424// Non, creer un nouveau bloc
425 dest_blocs_recv_data.append_array(rj);
426 dest_blocs_recv_data.append_array(rj+1);
427 }
428 last_added = rj;
429// Demander au processeur voisin d'envoyer cet item
430 tmp.append_array(item_rank);
431 received_count++;
432 }
433 item_rank++;
434 }
435 }
436// Initialisation de la taille de la liste de blocs pour ce processeur:
437 dest_blocs_recv_index[i_pe+1] = dest_blocs_recv_data.size_array();
438// Nombre d'items recus de ce processeur
439 dest.blocs_items_count_[i_pe] = received_count;
440// Envoi du tableau tmp au processeur voisin:
441 schema_comm.send_buffer(src.pe_voisins_[i_pe]) << tmp;
442 }
443 }
444
445 schema_comm.echange_taille_et_messages();
446
447// ********************************************************
448// Construction de dest.items_to_send avec les infos recues
449//
450 ArrOfInt dest_items_send_index;
451 dest_items_send_index.resize_array(nb_pe_voisins + 1, Array_base::NOCOPY_NOINIT);
452 ArrOfInt dest_items_send_data;
453 ArrOfInt nb_items_to_items(nb_pe_voisins); //!< Initialise a zero
454
455// Allocation d'une taille par borne superieure:
456 dest_items_send_data.resize_array(src.items_to_send_.get_data().size_array(), Array_base::NOCOPY_NOINIT);
457 dest_items_send_index[0] = 0;
458 {
459 int count = 0;
460 int error = 0;
461 for (int i_pe = 0; i_pe < nb_pe_voisins; i_pe++)
462 {
463// nb_item_received_single est le nombre d'items dans items_to_send_ qui seront receptionnes
464// par items_to_recv_. Les suivants sont recus dans blocs_to_recv_.
465 const int j_limite_bloc = src.nb_items_to_items_[i_pe];
466 int new_j_limite_bloc = 0;
467 schema_comm.recv_buffer(src.pe_voisins_[i_pe]) >> tmp;
468 const int n = tmp.size_array();
469 for (int i = 0; i < n; i++)
470 {
471// tmp contient des indices qui pointent dans src.items_to_send_
472 int j = tmp[i];
473 if (j >= j_limite_bloc)
474 new_j_limite_bloc++; //!< Cet item est dans les blocs_to_recv_ de l'autre cote k est l'indice dans l'ancien tableau de l'item a envoyer
475//
476 int k = src.items_to_send_(i_pe, j);
477 int renum_k = renum[k];
478// Une erreur possible ici: un processeur a besoin d'un item virtuel et l'item
479// reel a ete supprime sur le processeur source. Il faut alors utiliser un
480// autre algo pour changer le proprietaire de l'item...
481 if (renum_k < 0)
482 error = 1;
483 dest_items_send_data[count++] = renum_k;
484 }
485 dest_items_send_index[i_pe+1] = count;
486 nb_items_to_items[i_pe] = new_j_limite_bloc;
487 }
488 if (error)
489 {
490 Cerr << "Internal error in MD_Vector_tools::creer_md_vect_renum:\n"
491 << " processor " << Process::me()
492 << " received a request for distant items that are not in the new vector" << finl;
493 Process::exit();
494 }
495// On ajuste a la taille definitive
496 dest_items_send_data.set_smart_resize(1);
497 dest_items_send_data.resize_array(count);
498 }
499 schema_comm.end_comm();
500
501// Construction de la liste de processeurs voisins de dest
502// (processeurs avec qui on echange des donnees)
503// Et compression des indexes des StaticIntLists pour retirer les
504// processeurs supprimes.
505 {
506 int pe_count = 0;
507 dest.pe_voisins_.resize_array(nb_pe_voisins, Array_base::NOCOPY_NOINIT);
508 for (int i_pe = 0; i_pe < nb_pe_voisins; i_pe++)
509 {
510 int flag = 0;
511 if (dest_items_recv_index[i_pe+1] - dest_items_recv_index[i_pe] > 0)
512 flag = 1;
513 if (dest_blocs_recv_index[i_pe+1] - dest_blocs_recv_index[i_pe] > 0)
514 flag = 1;
515 if (dest_items_send_index[i_pe+1] - dest_items_send_index[i_pe] > 0)
516 flag = 1;
517
518 if (flag)
519 {
520 dest_items_recv_index[pe_count+1] = dest_items_recv_index[i_pe+1];
521 dest_blocs_recv_index[pe_count+1] = dest_blocs_recv_index[i_pe+1];
522 dest_items_send_index[pe_count+1] = dest_items_send_index[i_pe+1];

663MD_Vector MD_Vector_tools::extend(const MD_Vector& src, extra_item_t& items)
664{
665 if (Process::mp_max(items.size()) == 0) return src; //rien a faire
666 /* remplissage de : recep[p] -> liste des items qu'on veut recevoir du processeur p
667 items[{p, i}] -> ou on va placer chaque item dans le MD_Vector elargi */
668 int i, j, p, nb_items_tot = src.valeur().get_nb_items_tot(), idx = nb_items_tot;
669 const MD_Vector_std *mds = NULL;
670 if (sub_type(MD_Vector_std, src.valeur())) mds = &ref_cast(MD_Vector_std, src.valeur());
671 else if (sub_type(MD_Vector_composite, src.valeur())) mds = &ref_cast(MD_Vector_composite, src.valeur()).global_md_;
672
673 /* recv[p] : items qu'on veut recevoir du processeur p : taille nrecv[p]. On en profite pour les numeroter dans items */
674 ArrsOfInt recv(Process::nproc());
675 for (p = 0; p < Process::nproc(); p++) recv[p].set_smart_resize(1);
676 std::vector<int> nrecv(Process::nproc());
677 for (auto &kv : items) recv[kv.first[0]].append_array(kv.first[1]), kv.second = idx, nrecv[kv.first[0]]++, idx++;
678
679 /* send[p] : items qu'on doit envoyer au processeur p */
680 ArrsOfInt send(Process::nproc());
681 envoyer_all_to_all(recv, send);
682
683 std::map<int, std::array<std::vector<int>, 3>> map; //map[p] = { items a envoyer a p, items a recevoir de p, blocs a recevoir de p }
684
685 for (i = 0; i < mds->pe_voisins_.size_array(); i++)
686 {
687 p = mds->pe_voisins_[i];
688 for (j = 0; j < mds->items_to_send_.get_list_size(i); j++) map[p][0].push_back(mds->items_to_send_(i, j));
689 for (j = 0; j < mds->items_to_recv_.get_list_size(i); j++) map[p][1].push_back(mds->items_to_recv_(i, j));
690 for (j = 0; j < mds->blocs_to_recv_.get_list_size(i); j++) map[p][2].push_back(mds->blocs_to_recv_(i, j));
691 }
692
693 /* ajout de ce qu'on doit envoyer/recevoir en plus */
694 for (p = 0; p < Process::nproc(); p++)
695 for (i = 0; i < send[p].size_array(); i++) map[p][0].push_back(send[p][i]);
696 for (p = 0, idx = nb_items_tot; p < Process::nproc(); idx += nrecv[p], p++)
697 if (nrecv[p]) map[p][2].insert(map[p][2].end(), { idx, idx + nrecv[p] });
698
699 /* reconstruction de tableaux pour MD_Vector_std */
700 ArrOfInt pe_voisins(map.size());
701 ArrsOfInt items_to_send(map.size()), items_to_recv(map.size()), blocs_to_recv(map.size());
702 for (p = 0, i = 0; p < Process::nproc(); nb_items_tot += nrecv[p], p++)
703 if (map.count(p))
704 {
705 pe_voisins[i] = p;
706 items_to_send[i].resize(map[p][0].size()), std::copy(map[p][0].begin(), map[p][0].end(), items_to_send[i].addr());
707 items_to_recv[i].resize(map[p][1].size()), std::copy(map[p][1].begin(), map[p][1].end(), items_to_recv[i].addr());
708 blocs_to_recv[i].resize(map[p][2].size()), std::copy(map[p][2].begin(), map[p][2].end(), blocs_to_recv[i].addr());
709 i++;
710 }
711
712 /* c'est parti */
713 MD_Vector_std mdd(nb_items_tot, mds->get_nb_items_reels(), pe_voisins, items_to_send, items_to_recv, blocs_to_recv);
714 MD_Vector res;
715 res.copy(mdd);
716 return res;
717}
