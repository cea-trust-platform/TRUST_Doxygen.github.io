54double Frontiere_ouverte_rho_u_impose::val_imp_au_temps(double temps, int i,int j) const
55{
56 double rho_u, rho;
57 int ndeb=le_champ_front->frontiere_dis().frontiere().num_premiere_face();
58 const DoubleTab& tab_rho_u=le_champ_front->valeurs_au_temps(temps);
59 assert(tab_rho_u.nb_dim()==2);
60
61 if (tab_rho_u.dimension(0)==1)
62 rho_u=tab_rho_u(0,j);
63 else
64 rho_u=tab_rho_u(i,j);
65
66 rho=le_fluide->rho_face_np1()(i+ndeb);
67 return rho_u/rho;
68}
