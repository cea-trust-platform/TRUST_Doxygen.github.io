64//! Discretisation d'un champ pour le VEF en fonction d'une directive<br>de discretisation. La directive est un Motcle comme "vitesse", "pression",<br>"temperature", "champ_elem" (cree un champ de type P0 ...), ...<br>Cette methode determine le type du champ a creer en fonction du type d'element<br>et de la directive de discretisation. Elle determine ensuite le nombre de ddl<br>et fixe l'ensemble des parametres du champ (type, nb_compo, nb_ddl, nb_pas_dt,<br>nom(s), unite(s), nature du champ et attribue un temps) et associe la Zone_dis au champ.<br>Voir le code pour avoir la correspondance entre les directives et<br>le type de champ cree.
65void VEF_discretisation::discretiser_champ(
66 const Motcle& directive, const Zone_dis_base& z,
67 Nature_du_champ nature,
68 const Noms& noms, const Noms& unites,
69 int nb_comp, int nb_pas_dt, double temps,
70 Champ_Inc& champ, const Nom& sous_type) const
71{
72 const Zone_VEF& zone_vef = ref_cast(Zone_VEF, z);
73
74 Motcles motcles(7);
75 motcles[0] = "vitesse"; //!< Choix standard pour la vitesse
76 motcles[1] = "pression"; //!< Choix standard pour la pression
77 motcles[2] = "temperature"; //!< Choix standard pour la temperature
78 motcles[3] = "divergence_vitesse"; //!< Le type de champ obtenu en calculant div v
79 motcles[4] = "gradient_pression"; //!< Le type de champ obtenu en calculant grad P
80 motcles[5] = "champ_elem"; //!< Creer un champ aux elements (de type P0)
81 motcles[6] = "champ_sommets"; //!< Creer un champ aux sommets (type P1)
82
83// Le type de champ de vitesse depend du type d'element :
84 Nom type_champ_vitesse;
85 if (sub_type(Tri_VEF, zone_vef.type_elem().valeur())
86 || sub_type(Tetra_VEF,zone_vef.type_elem().valeur()))
87 type_champ_vitesse = "Champ_P1NC";
88 else if (sub_type(Quadri_VEF,zone_vef.type_elem().valeur())
89 || sub_type(Hexa_VEF,zone_vef.type_elem().valeur()))
90 type_champ_vitesse = "Champ_Q1NC";
91 else
92 {
93 Cerr << "VEF_discretisation::discretiser_champ :\n L'element geometrique ";
94 Cerr << zone_vef.type_elem().valeur().que_suis_je();
95 Cerr << " n'est pas supporte." << finl;
96 exit();
97 }
98
99 Nom type;
100 int default_nb_comp = 0; //!< Valeur par defaut du nombre de composantes
101 int rang = motcles.search(directive);
102 switch(rang)
103 {
104 case 0:
105 type = type_champ_vitesse;
106 default_nb_comp = dimension;
107 break;
108 case 1:
109 type = "Champ_P0_VEF";
110 default_nb_comp = 1;
111 break;
112 case 2:
113 type = type_champ_vitesse;
114 default_nb_comp = 1;
115 break;
116 case 3:
117 type = "Champ_P0_VEF";
118 default_nb_comp = 1;
119 break;
120 case 4:
121 type = type_champ_vitesse;
122 default_nb_comp = dimension;
123 break;
124 case 5:
125 type = "Champ_P0_VEF";
126 default_nb_comp = 1;
127 break;
128 case 6:
129 type = "Champ_P1_VEF";
130 default_nb_comp = 1;
131 break;
132 default:
133 assert(rang < 0);
134 break;
135 }
136
137 if (directive == demande_description)
138 Cerr << "VEF_discretisation : " << motcles;
139
140// Si on n'a pas compris la directive (ou si c'est une demande_description)
141// alors on appelle l'ancetre :
142 if (rang < 0)
143 {
144 Discret_Thyd::discretiser_champ(directive, z,
145 nature, noms, unites,
146 nb_comp, nb_pas_dt, temps, champ);
147 return;
148 }
149
150// Calcul du nombre de ddl
151 int nb_ddl = 0;
152 if (type == "Champ_P0_VEF")
153 nb_ddl = z.nb_elem();
154 else if (type == type_champ_vitesse)
155 nb_ddl = zone_vef.nb_faces();
156 else if (type == "Champ_P1_VEF")
157 nb_ddl = zone_vef.nb_som();
158 else
159 assert(0);
160
161 if (nb_comp < 0)
162 nb_comp = default_nb_comp;
163 assert(nb_comp > 0);
164 creer_champ(champ, z, type, noms[0], unites[0], nb_comp, nb_ddl, nb_pas_dt, temps,
165 directive, que_suis_je());
166 if (nature == multi_scalaire)
167 {
168 champ.valeur().fixer_nature_du_champ(nature);
169 champ.valeur().fixer_unites(unites);
170 champ.valeur().fixer_noms_compo(noms);
171 }
172
173}

206//! Idem que VEF_discretisation::discretiser_champ(... , Champ_Inc)<br>Traitement commun aux champ_fonc et champ_don.<br>Cette methode est privee (passage d'un Objet_U pas propre vu<br>de l'exterieur ...)
207void VEF_discretisation::discretiser_champ_fonc_don(
208 const Motcle& directive, const Zone_dis_base& z,
209 Nature_du_champ nature,
210 const Noms& noms, const Noms& unites,
211 int nb_comp, double temps,
212 Objet_U& champ) const
213{
214// Deux pointeurs pour acceder facilement au champ_don ou au champ_fonc,
215// suivant le type de l'objet champ.
216 Champ_Fonc * champ_fonc = 0;
217 Champ_Don * champ_don = 0;
218 if (sub_type(Champ_Fonc, champ))
219 champ_fonc = & ref_cast(Champ_Fonc, champ);
220 else
221 champ_don = & ref_cast(Champ_Don, champ);
222
223 const Zone_VEF& zone_vef = ref_cast(Zone_VEF, z);
224
225 Motcles motcles(8);
226 motcles[0] = "pression"; //!< Choix standard pour la pression
227 motcles[1] = "temperature"; //!< Choix standard pour la temperature
228 motcles[2] = "divergence_vitesse"; //!< Le type de champ obtenu en calculant div v
229 motcles[3] = "champ_elem"; //!< Creer un champ aux elements (de type P0)
230 motcles[4] = "vitesse"; //!< Choix standard pour la vitesse
231 motcles[5] = "gradient_pression"; //!< Le type de champ obtenu en calculant grad P
232 motcles[6] = "champ_sommets"; //!< Creer un champ aux sommets
233 motcles[7] = "champ_face"; //!< Creer un champ aux faces
234
235
236// Le type de champ de vitesse depend du type d'element :
237 Nom type_champ_vitesse, type_champ_sommets;
238 Nom type_elem_zone = zone_vef.zone().type_elem().valeur().que_suis_je();
239 if (Motcle(type_elem_zone)!="Segment")
240 {
241 const Elem_VEF_base& elem_vef = zone_vef.type_elem().valeur();
242 if (sub_type(Tri_VEF, elem_vef) || sub_type(Tetra_VEF, elem_vef))
243 {
244 type_champ_vitesse = "Champ_Fonc_P1NC";
245 type_champ_sommets = "Champ_Fonc_P1_VEF";
246 }
247 else if (sub_type(Quadri_VEF, elem_vef) || sub_type(Hexa_VEF, elem_vef))
248 {
249 type_champ_vitesse = "Champ_Fonc_Q1NC";
250 type_champ_sommets = "Champ_Fonc_Q1_VEF";
251 }
252 else
253 {
254 Cerr << "VEF_discretisation::discretiser_champ :\n L'element geometrique ";
255 Cerr << elem_vef.que_suis_je();
256 Cerr << " n'est pas supporte." << finl;
257 exit();
258 }
259 }
260
261 Nom type;
262 int default_nb_comp = 0; //!< Valeur par defaut du nombre de composantes
263 int rang = motcles.search(directive);
264 switch(rang)
265 {
266 case 0:
267 type = "Champ_Fonc_P0_VEF";
268 default_nb_comp = 1;
269 break;
270 case 1:
271 type = type_champ_vitesse;
272 default_nb_comp = 1;
273 break;
274 case 2:
275 type = "Champ_Fonc_P0_VEF";
276 default_nb_comp = 1;
277 break;
278 case 3:
279 type = "Champ_Fonc_P0_VEF";
280 default_nb_comp = 1;
281 break;
282 case 4:
283 type = type_champ_vitesse;
284 default_nb_comp = dimension;
285 break;
286 case 5:
287 type = type_champ_vitesse;
288 default_nb_comp = dimension;
289 break;
290 case 6:
291 type = type_champ_sommets;
292 default_nb_comp = 1;
293 break;
294 case 7:
295 type = type_champ_vitesse;
296 default_nb_comp = 1;
297 break;
298
299 default:
300 assert(rang < 0);
301 break;
302 }
303
304 if (directive == demande_description)
305 Cerr << "VEF_discretisation : " << motcles;
306
307// Si on n'a pas compris la directive (ou si c'est une demande_description)
308// alors on appelle l'ancetre :
309 if (rang < 0)
310 {
311 if (champ_fonc)
312 Discret_Thyd::discretiser_champ(directive, z, nature, noms, unites,
313 nb_comp, temps, *champ_fonc);
314 else
315 Discret_Thyd::discretiser_champ(directive, z, nature, noms, unites,
316 nb_comp, temps, *champ_don);
317 return;
318 }
319
320// Calcul du nombre de ddl
321 int nb_ddl = 0;
322 if (type == "Champ_Fonc_P0_VEF") nb_ddl = z.nb_elem();
323 else if (type == type_champ_vitesse) nb_ddl = zone_vef.nb_faces();
324 else if ((type == "Champ_Fonc_P1_VEF") || (type == "Champ_Fonc_Q1_VEF")) nb_ddl = zone_vef.nb_som();
325 else assert(0);
326
327 if (nb_comp < 0)
328 nb_comp = default_nb_comp;
329 assert(nb_comp > 0);
330 if (champ_fonc)
331 creer_champ(*champ_fonc, z, type, noms[0], unites[0], nb_comp, nb_ddl, temps,
332 directive, que_suis_je());
333 else
334 creer_champ(*champ_don, z, type, noms[0], unites[0], nb_comp, nb_ddl, temps,
335 directive, que_suis_je());
336
337 if ((nature == multi_scalaire) && (champ_fonc))
338 {
339 champ_fonc->valeur().fixer_nature_du_champ(nature);
340 champ_fonc->valeur().fixer_unites(unites);
341 champ_fonc->valeur().fixer_noms_compo(noms);
342 }
343 else if ((nature == multi_scalaire) && (champ_don))
344 {
345 Cerr<<"There is no field of type Champ_Don with a multi_scalaire nature."<<finl;
346 exit();
347 }
348}

352void VEF_discretisation::distance_paroi(const Schema_Temps_base& sch,
353 Zone_dis& z, Champ_Fonc& ch) const
354{
355 Cerr << "Discretisation de la distance paroi" << finl;

358 Champ_Fonc_P0_VEF& ch_dist_paroi=ref_cast(Champ_Fonc_P0_VEF,ch.valeur());

371{
372 Cerr << "Discretisation de la vorticite " << finl;
373 const Zone_VEF& zone_VEF=ref_cast(Zone_VEF, z.valeur());
374
375 if (sub_type(Tri_VEF,zone_VEF.type_elem().valeur()) || sub_type(Tetra_VEF,zone_VEF.type_elem().valeur()))

378 const Champ_P1NC& vit = ref_cast(Champ_P1NC,ch_vitesse.valeur());

388 ch_W.fixer_nom_compo(0,"vorticitex");

509//! discretise en VEF le fluide incompressible, donc K e N
510void VEF_discretisation::proprietes_physiques_fluide_Ostwald(const Zone_dis& z, Fluide_Ostwald& le_fluide,
511 const Navier_Stokes_std& eqn_hydr, const Champ_Inc& ch_temper ) const
512{
513 Cerr << "Discretisation VEF du fluide_Ostwald" << finl;
514 const Zone_VEF& zone_vef=ref_cast(Zone_VEF, z.valeur());
515 const Champ_Inc& ch_vitesse = eqn_hydr.inconnue();
516 const Champ_P1NC& vit = ref_cast(Champ_P1NC,ch_vitesse.valeur());
517
518
519
520
521 Champ_Don& mu = le_fluide.viscosite_dynamique();
522// mu est toujours un champ_Ostwald_VEF , il faut toujours faire ce qui suit
523 mu.typer("Champ_Ostwald_VEF");
524 Champ_Ostwald_VEF& ch_mu = ref_cast(Champ_Ostwald_VEF,mu.valeur());
525 Cerr<<"associe zonedisbase VEF"<<finl;
526 ch_mu.associer_zone_dis_base(zone_vef);
527 ch_mu.associer_fluide(le_fluide);
528 ch_mu.associer_champ(vit);
529 ch_mu.associer_eqn(eqn_hydr);
530 Cerr<<"associations finies zone dis base, fluide, champ VEF"<<finl;
531 ch_mu.fixer_nb_comp(1);
532
533 Cerr<<"fait fixer_nb_valeurs_nodales"<<finl;
534 Cerr<<"nb_valeurs_nodales VEF = "<<zone_vef.nb_elem()<<finl;
535 ch_mu.fixer_nb_valeurs_nodales(zone_vef.nb_elem());
536
537 Cerr<<"fait changer_temps"<<finl;
538 ch_mu.changer_temps(vit.temps());
539 Cerr<<"mu VEF est discretise "<<finl;
540}
