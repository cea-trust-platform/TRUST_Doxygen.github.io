61Entree& Traitement_particulier_NS_THI_new::lire(Entree& is)
62{
63// valeurs par defaut
64 init = 0;
65 fac_init = 0;
66 oui_transf = 0;
67 oui_calc_spectre = 0;
68// FIN valeurs par defaut
69 Motcle accouverte = "{" , accfermee = "}" ;
70 Motcle valec="val_Ec";
71 Motcle facon="facon_init";
72 Motcle motbidon, motlu;
73 is >> motbidon ;
74 if (motbidon == accouverte)
75 {
76 Motcles les_mots(2);
77 {
78 les_mots[0] = "init_Ec";
79 les_mots[1] = "calc_spectre";
80 }
81 {
82 is >> motlu;
83 while(motlu != accfermee)
84 {
85 int rang=les_mots.search(motlu);
86 switch(rang)
87 {
88 case 0 :
89 {
90 is >> init; //!< init_Ec = 1
91 if (init!=0)
92 {
93 is >> motlu;
94 if (motlu==valec)
95 {
96 is >> Ec_init; //!< on lit la valeur de Ec_init
97 Cerr << "Avec Initialisation de l Energie Cinetique Ec_init= " << Ec_init << finl;
98 is >> motlu;
99 if (motlu==facon)
100 {
101 is >> fac_init; //!< Sur quelle valeur se base l initialisation??
102 Cerr << "Avec Initialisation de l Energie Cinetique sur Ecspat (init_fac==0) ou Ecspec (init_fac==1) : fac_init=" << fac_init << finl;
103 }
104 }
105 else
106 {
107 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI_new_VDF" << finl;
108 Cerr << "Le seul mot cle possible ici est : val_Ec" << finl;
109 Cerr << "Vous avez lu :" << motlu << finl;
110 exit();
111 }
112 }
113 break;
114 }
115 case 1 :
116 {
117 is >> oui_calc_spectre;
118 if (oui_calc_spectre!=0) Cerr << "Calcul des spectres." << finl;
119 break;
120 }
121
122 default :
123 {
124 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI_new";
125 Cerr << "Les mots cles possibles sont : init_Ec, calc_spectre, { et }" << finl;
126 Cerr << "Vous avez lu :" << motlu << finl;
127 exit();
128 break;
129 }
130 }
131 is >> motlu;
132 }
133 is >> motlu;
134 if (motlu != accfermee)
135 {
136 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI_new";
137 Cerr << "On attendait une }" << finl;
138 exit();
139 }
140 }
141 }
142 else
143 {
144 Cerr << "Erreur dans la lecture de Traitement_particulier_NS_THI_new";
145 Cerr << "On attendait une {" << finl;

147 }
148 return is;

184{
185 const char* methode_actuelle="Traitement_particulier_NS_THI_new::calcul_nb_som_dir";
186
187// Sert a calculer le nombre de sommet commun en parallele
188 double nb_som = zone.domaine().md_vector_sommets().valeur().nb_items_seq_tot();

190
