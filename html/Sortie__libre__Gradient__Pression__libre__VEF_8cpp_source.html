108 elem=face_voisins(face,1);
109 coeff[face-ndeb] = -distance_2D(face,elem,ma_zone_VEF);
110 }
111 }

114 {
115 int elem=face_voisins(face,0);
116 if (elem != -1)
117 coeff[face-ndeb] = distance_3D(face,elem,ma_zone_VEF);
118 else
119 {
120 elem=face_voisins(face,1);
121 coeff[face-ndeb] = -distance_3D(face,elem,ma_zone_VEF);
122 }
123 }

145double Sortie_libre_Gradient_Pression_libre_VEF::flux_impose(int face) const
146{
147 const Zone_Cl_dis_base& la_zone_Cl = zone_Cl_dis();
148 const Equation_base& eqn = la_zone_Cl.equation();
149 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,eqn);
150 const Zone_VEF& ma_zone_VEF = ref_cast(Zone_VEF,eqn.zone_dis().valeur());
151 const IntTab& face_voisins = ma_zone_VEF.face_voisins();
152// const Zone_Cl_VEF& zone_Cl_VEF = ref_cast(Zone_Cl_VEF,ma_zone_cl_dis.valeur());
153// const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
154 const IntTab& elem_faces = ma_zone_VEF.elem_faces();
155 const DoubleTab& face_normales = ma_zone_VEF.face_normales();
156 const Champ_P0_VEF& pre = ref_cast(Champ_P0_VEF,eqn_hydr.pression().valeur());
157
158 double Pimp, diff, grad ;
159
160 int elem1, elem2, face_adj, face_face_adj ;
161
162 double a1 = trace_pression_int[face];
163 double a2 = coeff[face];
164
165 double a3 = 0. ;
166
167 elem1 = face_voisins(face,0);
168
169 for (face_adj=0; face_adj<dimension; face_adj++)
170 {
171 face_face_adj= elem_faces(elem1,face_adj);
172
173 elem2 = face_voisins(face_face_adj,0);
174 if (elem2 != elem1)
175 {
176 diff = pre(elem2) - pre(elem1);
177
178 for (int comp=0; comp < dimension; comp++)
179 {
180 grad = diff*face_normales(face_face_adj,comp);
181 a3 += grad ;
182 }
183 }
184 }
185
186 a3 /= double(dimension) ;
187 Pimp = a1 + a2 * a3 ;
188 return Pimp ;
189
190}

200{

202 const Champ_Uniforme& rho=ref_cast(Champ_Uniforme,mil.masse_volumique().valeur());
203 double d_rho = rho(0,0);
204 if (le_champ_front.valeurs().size()==1)
205 return le_champ_front(0,0)/d_rho;
206 else if (le_champ_front.valeurs().line_size()==1)

208 else

210 exit();
