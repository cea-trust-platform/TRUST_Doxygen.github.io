55int Sortie_libre_Gradient_Pression_impose_VEFPreP1B::initialiser(double temps)
56{
57
58 Cond_lim_base::initialiser(temps);
59 const Zone_Cl_dis_base& la_zone_Cl = zone_Cl_dis();
60 const Equation_base& eqn = la_zone_Cl.equation();
61 const Navier_Stokes_std& eqn_hydr = ref_cast(Navier_Stokes_std,eqn);
62 const Champ_P1_isoP1Bulle& pression = ref_cast(Champ_P1_isoP1Bulle,eqn_hydr.pression().valeur());
63 const Zone_VEF& ma_zone_VEF = ref_cast(Zone_VEF,eqn.zone_dis().valeur());
64 const IntTab& face_voisins = ma_zone_VEF.face_voisins();
65 const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
66 pression_interne = pression;
67 int nb_faces_loc = le_bord.nb_faces_tot();
68 trace_pression_int.resize(nb_faces_loc);
69 coeff.resize(nb_faces_loc);
70 for (int ind_face=0; ind_face<nb_faces_loc ; ind_face++)
71 {
72 int face = le_bord.num_face(ind_face);
73 int elem=face_voisins(face,0);
74 if (dimension==2)
75 coeff[ind_face] = distance_2D(face,elem,ma_zone_VEF) * 3. ;
76 else
77 coeff[ind_face] = distance_3D(face,elem,ma_zone_VEF) * 4. ;
78 }
79// On n'appelle pas mettre_a_jour car la mise a jour du champ_front pourrait dependre de l'exterieur !
80 return calculer_trace_pression();
81}

93// Retourne 1 si OK, 0 sinon.
94int Sortie_libre_Gradient_Pression_impose_VEFPreP1B::calculer_trace_pression()
95{
96 if (!pression_interne.non_nul())
97 {
98 Cerr << "Sortie_libre_Gradient_Pression_impose_VEFPreP1B::calculer_trace_pression() : la pression n'est pas definie!" << finl;
99 return 0;
100 }
101 Debog::verifier("pression_interne:",pression_interne);
102 const Front_VF& le_bord = ref_cast(Front_VF,frontiere_dis());
103 int nb_faces_loc = le_bord.nb_faces_tot();
104 for (int ind_face=0; ind_face<nb_faces_loc; ind_face++)
105 {
106 int face = le_bord.num_face(ind_face);
107 trace_pression_int[ind_face] = pression_interne->valeur_au_bord(face);
108 }
109 return 1;
110}
