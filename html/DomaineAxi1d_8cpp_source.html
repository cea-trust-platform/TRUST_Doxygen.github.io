31Entree& DomaineAxi1d::readOn( Entree& is )
32{
33 const Motcle accolade_ouverte("{");
34 const Motcle accolade_fermee("}");
35
36 Motcle motlu;
37
38 is >> motlu;
39
40 if (motlu != accolade_ouverte)
41 {
42 Cerr << "Error while reading 1D axi domain " << finl;
43 Cerr << "We expected a " << accolade_ouverte << " instead of \n"
44 << motlu << finl;
45 Process::exit();
46 }
47
48 while (1)
49 {
50 is >> motlu;
51
52 if (motlu == accolade_fermee)
53 break;
54
55 if (motlu=="ORIGINE")
56 {
57 is >> champ_orig;
58 }
59 else
60 {
61 Cerr << "Error while reading 1D axi domain " << finl;
62 Cerr << "We expected a \"origine\" instead of \n"
63 << motlu << finl;
64 Process::exit();
65 }
66 }
67
68 return is;
69}
