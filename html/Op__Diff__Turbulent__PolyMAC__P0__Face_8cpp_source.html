72void Op_Diff_Turbulent_PolyMAC_P0_Face::mettre_a_jour(double temps)
73{
74 Op_Diff_PolyMAC_P0_Face::mettre_a_jour(temps);
75
76 int N = ref_cast(Pb_Multiphase, equation().probleme()).nb_phases();
77 for (int n = 0; n < N; n++)
78 if (nu_t_post_[n].non_nul()) //viscosite turbulente : toujours scalaire
79 {
80 const DoubleTab& rho = equation().milieu().masse_volumique().passe() ;
81 DoubleTab& val = nu_t_post_[n]->valeurs();
82 int nl = val.dimension(0);
83 int cR = (rho.dimension(0) == 1);
84 DoubleTrav nu_t(nl, N);
85 ref_cast(Viscosite_turbulente_base, corr.valeur()).eddy_viscosity(nu_t); //remplissage par la correlation
86 for (int i = 0; i < nl; i++) val(i, 0) = rho(!cR * i, n) * nu_t(i, n);
87 }

90void Op_Diff_Turbulent_PolyMAC_P0_Face::modifier_nu(DoubleTab& mu) const
91{
92 if (!corr.non_nul()) return; //rien a faire
93 const DoubleTab& rho = equation().milieu().masse_volumique().passe(),
94 *alpha = sub_type(Pb_Multiphase, equation().probleme()) ? &ref_cast(Pb_Multiphase, equation().probleme()).eq_masse.inconnue().passe() : NULL;
95 int i, nl = mu.dimension(0), n, N = equation().inconnue().valeurs().line_size(), cR = rho.dimension(0) == 1, d, D = dimension;
96 DoubleTrav nu_t(nl, N); //viscosite turbulente : toujours scalaire
97 ref_cast(Viscosite_turbulente_base, corr.valeur()).eddy_viscosity(nu_t); //remplissage par la correlation
98 if (mu.nb_dim() == 2) //nu scalaire
99 for (i = 0; i < nl; i++)
100 for (n = 0; n < N; n++) mu(i, n) += (alpha ? (*alpha)(i, n) : 1) * rho(!cR * i, n) * nu_t(i, n);
101 else if (mu.nb_dim() == 3) //nu anisotrope diagonal
102 for (i = 0; i < nl; i++)
103 for (n = 0; n < N; n++)
104 for (d = 0; d < D; d++) mu(i, n, d) += (alpha ? (*alpha)(i, n) : 1) * rho(!cR * i, n) * nu_t(i, n);
105 else if (mu.nb_dim() == 4) //nu anisotrope complet
106 for (i = 0; i < nl; i++)
107 for (n = 0; n < N; n++)
108 for (d = 0; d < D; d++) mu(i, n, d, d) += (alpha ? (*alpha)(i, n) : 1) * rho(!cR * i, n) * nu_t(i, n);
109 else abort();
110 mu.echange_espace_virtuel();
111}
