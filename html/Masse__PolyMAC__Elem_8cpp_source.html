61DoubleTab& Masse_PolyMAC_Elem::appliquer_impl(DoubleTab& sm) const
62{
63 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
64 const DoubleVect& ve = zone.volumes(), &pe = zone.porosite_elem();
65 const DoubleTab& der = equation().champ_conserve().derivees().at(equation().inconnue().le_nom().getString());
66
67 int e, ne_tot = zone.nb_elem_tot(), n, N = sm.line_size();
68 assert(sm.dimension_tot(0) >= ne_tot && N == der.line_size());
69
70 /* partie elem */
71 for (e = 0; e < ne_tot; e++)
72 for (n = 0; n < N; n++)
73 if (der(e, n) > 1e-10) sm(e, n) /= pe(e) * ve(e) * der(e, n);
74 else sm(e, n) = 0; //cas d'une evanescence
75
76 return sm;
77}

79void Masse_PolyMAC_Elem::dimensionner_blocs(matrices_t matrices, const tabs_t& semi_impl) const
80{
81 equation().init_champ_conserve();
82 /* une diagonale par derivee de champ_conserve_ presente dans matrices */
83 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
84 const Champ_Inc_base& cc = equation().champ_conserve();
85 int e, ne = zone.nb_elem(), n, N = cc.valeurs().line_size();
86
87 for (auto &&i_m : matrices)
88 if (cc.derivees().count(i_m.first))
89 {
90 /* nombre de composantes de la variable : autant que le champ par defaut, mais peut etre different pour la pression */
91 const DoubleTab& col = equation().probleme().get_champ(i_m.first.c_str()).valeurs(); //tableau de l'inconnue par rapport a laquelle on derive
92 int m, M = col.line_size();
93
94 IntTrav stencil(0, 2);
95 stencil.set_smart_resize(1);
96
97 for (e = 0; e < ne; e++)
98 for (n = 0, m = 0; n < N; n++, m += (M > 1)) stencil.append_line(N * e + n, M * e + m);
99 Matrice_Morse mat;
100 Matrix_tools::allocate_morse_matrix(N * equation().inconnue().valeurs().dimension_tot(0), M * col.dimension_tot(0), stencil, mat);
101 i_m.second->nb_colonnes() ? *i_m.second += mat : *i_m.second = mat;
102 }
103}

105void Masse_PolyMAC_Elem::ajouter_blocs(matrices_t matrices, DoubleTab& secmem, double dt, const tabs_t& semi_impl, int resoudre_en_increments) const
106{
107 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
108 const Champ_Inc_base& cc = equation().champ_conserve();
109 const Conds_lim& cls = equation().zone_Cl_dis()->les_conditions_limites();
110 const IntTab& fcl = ref_cast(Champ_Elem_PolyMAC, equation().inconnue().valeur()).fcl(), &f_e = zone.face_voisins();
111 const DoubleTab& present = cc.valeurs(), &passe = cc.passe();
112 const DoubleVect& ve = zone.volumes(), &pe = zone.porosite_elem(), &fs = zone.face_surfaces();
113 int e, f, n, N = cc.valeurs().line_size(), ne = zone.nb_elem();
114
115 /* second membre : avec ou sans resolution en increments*/
116 for (e = 0; e < ne; e++)
117 for (n = 0; n < N; n++)
118 secmem(e, n) += pe(e) * ve(e) * (passe(e, n) - resoudre_en_increments * present(e, n)) / dt;
119
120 /* si on n'a pas d'operateur de diffusion (operateur(0) negligeable ou operateur(0) convectif pour Masse_Multiphase), alors ajout des flux aux faces de Neumann */
121 if ( (sub_type(Op_Diff_negligeable, equation().operateur(0).l_op_base())) || (!sub_type(Operateur_Diff_base, equation().operateur(0).l_op_base())) )
122 for (f = 0; f < zone.premiere_face_int(); f++)
123 if (fcl(f, 0) == 4)
124 for (e = f_e(f, 0), n = 0; n < N; n++)
125 secmem(e, n) += fs(f) * ref_cast(Neumann_paroi, cls[fcl(f, 1)].valeur()).flux_impose(fcl(f, 2), n);
126
127 /* matrices */
128 for (auto &&i_m : matrices)
129 if (cc.derivees().count(i_m.first))
130 {
131 int m, M = equation().probleme().get_champ(i_m.first.c_str()).valeurs().line_size();
132 const DoubleTab& der = cc.derivees().at(i_m.first);
133 for (e = 0; e < ne; e++)
134 for (n = 0, m = 0; n < N; n++, m += (M > 1))
135 (*i_m.second)(N * e + n, M * e + m) += pe(e) * ve(e) * der(e, n) / dt;
136 }
137}
