161//! Calcule les volumes des elements de la zone associee.
162void Rectangle_2D_axi::calculer_volumes(DoubleVect& volumes) const
163{
164 const Zone& zone=ma_zone.valeur();//!< zone.valeur() permet d'acceder a // la zone qui est en cours de traitement
165 const Domaine& dom=zone.domaine();
166 double r,r1,r2,dr,dz;
167 int S1,S2,S3;
168 int size_tot = zone.nb_elem_tot();
169 assert(volumes.size_totale()==size_tot);
170 for (int num_poly=0; num_poly<size_tot; num_poly++)
171 {
172 S1 = zone.sommet_elem(num_poly,0);
173 S2 = zone.sommet_elem(num_poly,1);
174 S3 = zone.sommet_elem(num_poly,2);
175 r1= dom.coord(S1,0);
176 r2= dom.coord(S2,0);
177 if (r1< r2) r =r1;
178 else r =r2;
179 dr =std::fabs( dom.coord(S2,0) - dom.coord(S1,0));
180 dz =std::fabs( dom.coord(S3,1) - dom.coord(S1,1));
181 volumes[num_poly]= 2*M_PI*(r*dr+0.5*dr*dr)*dz;
182 }
183 /*
184 Rectangle::calculer_volumes(volumes);
185 */
186}
