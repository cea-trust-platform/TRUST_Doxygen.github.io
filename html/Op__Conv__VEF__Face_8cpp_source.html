65 if (type_lim=="vanleer") LIMITEUR=&vanleer;
66 if (type_lim=="vanalbada") LIMITEUR=&vanalbada;
67 if (type_lim=="chakravarthy") LIMITEUR=&chakravarthy;
68 if (type_lim=="superbee") LIMITEUR=&superbee;

71
72 if (ordre!=1 && ordre!=2 && ordre!=3)

499 {

553// Determination des faces amont pour les points C,S,S2
554 int face_amont_c=face_amont_m;
555 int face_amont_s=face_amont_m;
556 int face_amont_s2=face_amont_m;
557 if (type_op_boucle==muscl && ordre==3)
558 {
559 face_amont_c = (psc_c >= 0) ? num10 : num20;
560 face_amont_s = (psc_s >= 0) ? num10 : num20;
561 face_amont_s2 = (psc_s2 >= 0) ? num10 : num20;
562 }
563// gradient aux items element (schema centre) ou aux items face (schemas muscl)
564 int item_m=poly;
565 int item_c=poly;
566 int item_s=poly;
567 int item_s2=poly;
568 if (type_op_boucle==muscl)
569 {
570 item_m = face_amont_m;
571 item_c = face_amont_c;
572 item_s = face_amont_s;
573 item_s2 = face_amont_s2;
574 }
575
576 for (comp0=0; comp0<ncomp_ch_transporte; comp0++)
577 {
578 double flux;
579 double inco_m = (ncomp_ch_transporte==1?transporte_face(face_amont_m):transporte_face(face_amont_m,comp0));
580 if (type_op_boucle==amont || appliquer_cl_dirichlet)
581 {
582 flux = inco_m*psc_m;
583 }
584 else //!< muscl ou centre
585 {
586// Calcul de l'inconnue au centre M de la fa7
587 if (rang==-1)
588 for (j=0; j<dimension; j++)
589 inco_m+= gradient(item_m,comp0,j)*vecteur_face_facette(poly,fa7,j,dir);
590 else
591 for (j=0; j<dimension; j++)
592 inco_m+= gradient(item_m,comp0,j)*vecteur_face_facette_Cl(rang,fa7,j,dir);
593
594// Calcul de l'inconnue au sommet S, une premiere extremite de la fa7
595 double inco_s = (ncomp_ch_transporte==1?transporte_face(face_amont_s):transporte_face(face_amont_s,comp0));
596 int sommet_s = sommet_elem(poly,KEL(2,fa7));
597 for (j=0; j<dimension; j++)
598 inco_s+= gradient(item_s,comp0,j)*(-xv(face_amont_s,j)+coord_sommets(sommet_s,j));
599
600// Calcul de l'inconnue au sommet S2, la derniere extremite de la fa7 en 3D
601 double inco_s2=0;
602 if (dimension==3)
603 {
604 inco_s2 = (ncomp_ch_transporte==1?transporte_face(face_amont_s2):transporte_face(face_amont_s2,comp0));
605 int sommet_s2 = sommet_elem(poly,KEL(3,fa7));
606 for (j=0; j<dimension; j++)
607 inco_s2+= gradient(item_s2,comp0,j)*(-xv(face_amont_s2,j)+coord_sommets(sommet_s2,j));
608 }
609
610// Calcul de l'inconnue a C, une autre extremite de la fa7, intersection avec les autres fa7
611// du polyedre. C=G centre du polyedre si volume non etendu
612// xc donne par elemvef.calcul_xg()
613 double inco_c;
614 if (ordre==3)
615 {
616 inco_c = (ncomp_ch_transporte==1?transporte_face(face_amont_c):transporte_face(face_amont_c,comp0));
617 for (j=0; j<dimension; j++)
618 inco_c+= gradient(item_c,comp0,j)*(-xv(face_amont_c,j)+xc(j));
619 }
620 else
621 {
622 inco_c = dimension*inco_m-inco_s-inco_s2;
623 }
624
625// Calcul du flux sur 1 point
626 if (calcul_flux_en_un_point || option_calcul_flux_en_un_point)
627 {
628 flux = inco_m*psc_m;
629 }
630 else
631 {
632// Calcul du flux sur 3 points
633 flux = (dimension==2) ? (inco_c*psc_c + inco_s*psc_s + 4*inco_m*psc_m)/6
634 : (inco_c*psc_c + inco_s*psc_s + inco_s2*psc_s2 + 9*inco_m*psc_m)/12;
635 }
636 }
637
638// Ponderation par coefficient alpha
639 flux*=alpha;
640
641 if (ncomp_ch_transporte == 1)
642 {
643 resu(num10) -= flux;
644 resu(num20) += flux;
645 if (num10<nb_faces_bord) flux_b(num10,0) += flux;
646 if (num20<nb_faces_bord) flux_b(num20,0) -= flux;
647 }
648 else
649 {
650 resu(num10,comp0) -= flux;
651 resu(num20,comp0) += flux;
652 if (num10<nb_faces_bord) flux_b(num10,comp0) += flux;
653 if (num20<nb_faces_bord) flux_b(num20,comp0) -= flux;
654 }
655
656 }//!< boucle sur comp
657 } //!< fin de la boucle sur les facettes
658 }
659 } //!< fin de la boucle
660 alpha = 1 - alpha;
661 } //!< fin de la boucle statistiques().end_count(m2); statistiques().begin_count(m3);
662//
663//
664 int voisine;
665 nb_faces_perio = 0;
666 double diff1,diff2;
667
668// Boucle sur les bords pour traiter les conditions aux limites
669// il y a prise en compte d'un terme de convection pour les
670// conditions aux limites de Neumann_sortie_libre seulement
671 for (n_bord=0; n_bord<nb_bord; n_bord++)
672 {
673 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
674
675 if (sub_type(Neumann_sortie_libre,la_cl.valeur()))
676 {
677 const Neumann_sortie_libre& la_sortie_libre = ref_cast(Neumann_sortie_libre, la_cl.valeur());
678 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
679 int num1 = le_bord.num_premiere_face();
680 int num2 = num1 + le_bord.nb_faces();
681 for (num_face=num1; num_face<num2; num_face++)
682 {
683 psc =0;
684 for (i=0; i<dimension; i++)
685 psc += vitesse_face(num_face,i)*facenormales(num_face,i);
686 if (psc>0)
687 if (ncomp_ch_transporte == 1)
688 {
689 resu(num_face) -= psc*transporte_face(num_face);
690 flux_b(num_face,0) = -psc*transporte_face(num_face);
691 }
692 else
693 for (i=0; i<ncomp_ch_transporte; i++)
694 {
695 resu(num_face,i) -= psc*transporte_face(num_face,i);
696 flux_b(num_face,i) = -psc*transporte_face(num_face,i);
697 }
698 else
699 {
700 if (ncomp_ch_transporte == 1)
701 {
702 resu(num_face) -= psc*la_sortie_libre.val_ext(num_face-num1);
703 flux_b(num_face,0) = -psc*la_sortie_libre.val_ext(num_face-num1);
704 }
705 else
706 for (i=0; i<ncomp_ch_transporte; i++)
707 {
708 resu(num_face,i) -= psc*la_sortie_libre.val_ext(num_face-num1,i);
709 flux_b(num_face,i) = -psc*la_sortie_libre.val_ext(num_face-num1,i);
710 }
711 }
712 }
713 }
714 else if (sub_type(Periodique,la_cl.valeur()))
715 {
716 const Periodique& la_cl_perio = ref_cast(Periodique, la_cl.valeur());
717 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
718 int num1 = le_bord.num_premiere_face();
719 int num2 = num1 + le_bord.nb_faces();
720 ArrOfInt fait(le_bord.nb_faces());
721 fait = 0;
722 for (num_face=num1; num_face<num2; num_face++)
723 {
724 if (fait[num_face-num1] == 0)
725 {
726 voisine = la_cl_perio.face_associee(num_face-num1) + num1;
727
728 if (ncomp_ch_transporte == 1)
729 {
730 diff1 = resu(num_face)-tab(nb_faces_perio);
731 diff2 = resu(voisine)-tab(nb_faces_perio+voisine-num_face);
732 resu(voisine) += diff1;
733 resu(num_face) += diff2;
734 /* On ne doit pas ajouter a flux_b, c'est deja calcule au dessus
735 flux_b(voisine,0) += diff1;
736 flux_b(num_face,0) += diff2;*/
737 }
738 else
739 for (int comp=0; comp<ncomp_ch_transporte; comp++)
740 {
741 diff1 = resu(num_face,comp)-tab(nb_faces_perio,comp);
742 diff2 = resu(voisine,comp)-tab(nb_faces_perio+voisine-num_face,comp);
743 resu(voisine,comp) += diff1;
744 resu(num_face,comp) += diff2;
745 /* On ne doit pas ajouter a flux_b, c'est deja calcule au dessus
746 flux_b(voisine,comp) += diff1;
747 flux_b(num_face,comp) += diff2; */
748 }
749
750 fait[num_face-num1]= 1;
751 fait[voisine-num1] = 1;
752 }
753 nb_faces_perio++;
754 }
755 }
756 }
757
758 modifier_flux(*this);
759// statistiques().end_count(m3);
760 return resu;
761}

797{
798 modifier_matrice_pour_periodique_avant_contribuer(matrice,equation());
799 const Zone_Cl_VEF& zone_Cl_VEF = la_zcl_vef.valeur();

803 const IntTab& elem_faces = zone_VEF.elem_faces();
804 const DoubleTab& face_normales = zone_VEF.face_normales();
805 const DoubleTab& facette_normales = zone_VEF.facette_normales();

812
813 const DoubleVect& porosite_face = zone_VEF.porosite_face();

815 const DoubleTab& normales_facettes_Cl = zone_Cl_VEF.normales_facettes_Cl();
816
817 int nfac = zone.nb_faces_elem();
818 int nsom = zone.nb_som_elem();
819 const IntTab& sommet_elem = zone.les_elems();
820
821// Pour le traitement de la convection on distingue les polyedres
822// standard qui ne "voient" pas les conditions aux limites et les
823// polyedres non standard qui ont au moins une face sur le bord.
824// Un polyedre standard a n facettes sur lesquelles on applique le
825// schema de convection.
826// Pour un polyedre non standard qui porte des conditions aux limites
827// de Dirichlet, une partie des facettes sont portees par les faces.
828// En bref pour un polyedre le traitement de la convection depend
829// du type (triangle, tetraedre ...) et du nombre de faces de Dirichlet.
830
831 double psc;
832 DoubleTab pscl=0;
833 int poly,face_adj,fa7,i,j,n_bord;
834 int num_face, rang ,itypcl;
835 int num10,num20,num_som;
836 int ncomp_ch_transporte;
837 if (transporte.nb_dim() == 1)
838 ncomp_ch_transporte=1;
839 else
840 ncomp_ch_transporte= transporte.dimension(1);
841
842
843 int marq=phi_u_transportant(equation());
844
845 DoubleTab vitesse_face_;
846// soit on a transporte=phi*transporte_ et vitesse_face=vitesse_
847// soit transporte=transporte_ et vitesse_face=phi*vitesse_
848// cela depend si on transporte avec phi u ou avec u.
849 const DoubleTab& vitesse_face=modif_par_porosite_si_flag(vitesse_face_absolue,vitesse_face_,marq,porosite_face);
850 ArrOfInt face(nfac);
851 ArrOfDouble vs(dimension);
852 ArrOfDouble vc(dimension);
853 DoubleTab vsom(nsom,dimension);
854 ArrOfDouble cc(dimension);
855 DoubleVect& coeff = matrice.get_set_coeff();
856 const Elem_VEF_base& type_elemvef= zone_VEF.type_elem().valeur();
857 int istetra=0;
858 Nom nom_elem=type_elemvef.que_suis_je();
859 if ((nom_elem=="Tetra_VEF")||(nom_elem=="Tri_VEF"))
860 istetra=1;
861
862
863// Les polyedres non standard sont ranges en 2 groupes dans la Zone_VEF:
864// - polyedres bords et joints
865// - polyedres bords et non joints
866// On traite les polyedres en suivant l'ordre dans lequel ils figurent
867// dans la zone
868
869// boucle sur les polys
870 const IntTab& KEL=zone_VEF.type_elem().valeur().KEL();
871 int phi_u_transportant_yes=phi_u_transportant(equation());
872 for (poly=0; poly<nb_elem_tot; poly++)
873 {
874
875 rang = rang_elem_non_std(poly);
876 if (rang==-1)
877 itypcl=0;
878 else
879 itypcl=zone_Cl_VEF.type_elem_Cl(rang);
880
881// calcul des numeros des faces du polyedre
882 for (face_adj=0; face_adj<nfac; face_adj++)

889 vs[j]+= vitesse_face_absolue(face[i],j)*porosite_face[face[i]];
890 }
891// calcul de la vitesse aux sommets des polyedres
892// On va utliser les fonctions de forme implementees dans la classe Champs_P1_impl ou Champs_Q1_impl
893 if (istetra==1)
894 {
895 for (i=0; i<nsom; i++)
896 for (j=0; j<dimension; j++)
897 vsom(i,j) = (vs[j] - dimension*vitesse_face_absolue(face[i],j)*porosite_face[face[i]]);
898 }
899 else
900 {
901// pour que cela soit valide avec les hexa (c'est + lent a calculer...)
902 int ncomp;
903 for (j=0; j<nsom; j++)
904 {
905 num_som = sommet_elem(poly,j);
906 for (ncomp=0; ncomp<dimension; ncomp++)
907 vsom(j,ncomp) = la_vitesse.valeur_a_sommet_compo(num_som,poly,ncomp);
908 }
909 }
910
911
912// calcul de vc (a l'intersection des 3 facettes) vc vs vsom proportionnelles a la prosite
913 type_elemvef.calcul_vc(face,vc,vs,vsom,vitesse(),itypcl,porosite_face);
914 if (marq==0)
915 {
916 double porosite_poly=porosite_elem(poly);
917 for (i=0; i<nsom; i++)
918 for (j=0; j<dimension; j++)
919 vsom(i,j)/=porosite_poly;
920 for (j=0; j<dimension; j++)
921 {
922 vs[j]/= porosite_elem(poly);
923 vc[j]/=porosite_elem(poly);
924 }
925 }
926// Boucle sur les facettes du polyedre non standard:
927 for (fa7=0; fa7<nfa7; fa7++)
928 {
929 num10 = face[KEL(0,fa7)];
930 num20 = face[KEL(1,fa7)];
931// normales aux facettes
932 if (rang==-1)
933 {
934 for (i=0; i<dimension; i++)
935 cc[i] = facette_normales(poly, fa7, i);
936 }
937 else
938 {
939 for (i=0; i<dimension; i++)
940 cc[i] = normales_facettes_Cl(rang,fa7,i);
941 }
942// On applique le schema de convection a chaque sommet de la facette
943
944
945 double psc_c=0,psc_s=0,psc_m,psc_s2=0;
946 if (dimension==2)
947 {
948 for (i=0; i<dimension; i++)
949 {
950 psc_c+=vc[i]*cc[i];
951 psc_s+=vsom(KEL(2,fa7),i)*cc[i];
952 }
953 psc_m=(psc_c+psc_s)/2.;
954 }
955 else
956 {
957 for (i=0; i<dimension; i++)
958 {
959 psc_c+=vc[i]*cc[i];
960 psc_s+=vsom(KEL(2,fa7),i)*cc[i];
961 psc_s2+=vsom(KEL(3,fa7),i)*cc[i];
962 }
963 psc_m=(psc_c+psc_s+psc_s2)/3.;
964 }
965 convbisimplicite_dec(psc_m,num10,num20,transporte,ncomp_ch_transporte,coeff,matrice,porosite_face,phi_u_transportant_yes);
966 } //!< fin de boucle sur les facettes.
967
968 } //!< fin de boucle sur les polyedres.
969
970// Boucle sur les bords pour traiter les conditions aux limites
971// il y a prise en compte d'un terme de convection pour les
972// conditions aux limites de Neumann_sortie_libre seulement
973 int nb_bord = zone_VEF.nb_front_Cl();
974 for (n_bord=0; n_bord<nb_bord; n_bord++)
975 {
976
977 const Cond_lim& la_cl = zone_Cl_VEF.les_conditions_limites(n_bord);
978
979 if (sub_type(Neumann_sortie_libre,la_cl.valeur()))
980 {
981 const Front_VF& le_bord = ref_cast(Front_VF,la_cl.frontiere_dis());
982 int num1 = le_bord.num_premiere_face();
983 int num2 = num1 + le_bord.nb_faces();
984 for (num_face=num1; num_face<num2; num_face++)
985 {
986 psc =0;
987 for (i=0; i<dimension; i++)
988 psc += vitesse_face(num_face,i)*face_normales(num_face,i);
989 if (!phi_u_transportant_yes)
990 psc*=porosite_face(num_face);
991 if (psc>0)
992 {
993 for (j=0; j<ncomp_ch_transporte; j++)
994 {
995 int n0=num_face*ncomp_ch_transporte+j;
996 matrice(n0,n0)+=psc;
997 }
998 }
999 }
1000 }
1001 }
1002 modifier_matrice_pour_periodique_apres_contribuer(matrice,equation());
