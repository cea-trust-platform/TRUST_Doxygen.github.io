38//! lecture de la fonction de filtrage.<br>{<br>type BOITE|CHAPEAU|QUADRA|GAUSSIENNE|PARSER<br>demie-largeur L<br>[ omega W ]<br>[ expression FORMULE ]<br>}
39Entree& Moyenne_volumique::readOn(Entree& is )
40{
41 l_ = -1;
42 expression_parser_ = "??";
43 int type;
44 Param param(que_suis_je());
45 param.ajouter("type", & type, Param::REQUIRED /* obligatoire */);
46 param.dictionnaire("BOITE", BOITE);
47 param.dictionnaire("CHAPEAU", CHAPEAU);
48 param.dictionnaire("GAUSSIENNE", GAUSSIENNE);
49 param.dictionnaire("QUADRA", QUADRA);
50 param.dictionnaire("PARSER", PARSER);
51 param.ajouter("demie-largeur", & box_size_, Param::REQUIRED /* obligatoire */);
52 param.ajouter("omega", & l_);
53 param.ajouter("expression", & expression_parser_);
54 param.lire_avec_accolades_depuis(is);
55 switch(type)
56 {
57 case BOITE:
58 type_ = BOITE;
59 l_ = box_size_;
60 break;
61 case CHAPEAU:
62 type_ = CHAPEAU;
63 l_ = box_size_;
64 break;
65 case GAUSSIENNE:
66 type_ = GAUSSIENNE;
67 if (l_ < 0.)
68 {
69 Cerr << "Error : OMEGA must be set to >= 0" << finl;
70 barrier();
71 exit();
72 }
73 break;
74 case PARSER:
75 type_ = PARSER;
76 if (expression_parser_ == "??")
77 {
78 Cerr << "Error : EXPRESSION must be specified for the parser." << finl;
79 barrier();
80 exit();
81 }
82 {
83 std::string s(expression_parser_);
84 for (auto & c: s) c = toupper(c);
85 parser_.setString(s);
86 parser_.setNbVar(Objet_U::dimension);
87 parser_.addVar("x");
88 parser_.addVar("y");
89 if (Objet_U::dimension == 3)
90 parser_.addVar("z");
91 parser_.parseString();
92 }
93 break;
94 case QUADRA:
95 type_ = QUADRA;
96 l_ = box_size_;
97 Cerr << "l_ = " << l_ << "box_size_ = " << box_size_ << finl;
98 break;
99 default:
100 exit(); //!< Erreur interne !
101 }
102// Pour que l'octree trouve bien les elements qui sont pile-poil a la limite.
103 box_size_ += precision_geom;
104 return is;
105}
106inline double fonction_quadra(double x, double l_)
107{
108 assert(std::fabs(x) <= l_);
109 double ax = 1. - std::fabs(x) / l_;
110 ax = ax * ax;
111 if (std::fabs(x) < (l_/3.))
112 {
113 double bx = -3. / l_ * std::fabs(x) + 1.;
114 ax -= bx * bx / 3.; //!< signifie ax = ax - bx*bx/3
115 }
116 return ax * (27. / (16. * l_));
117}

119//! Evalue la fonction filtre en chaque coordonnee coord<br>Methode appelee dans la classe Calcul_integrale_locale
120void Moyenne_volumique::eval_filtre(const DoubleTab& coords, ArrOfDouble& result) const
121{
122 const int dim = Objet_U::dimension;
123 assert(dim == coords.dimension(1));
124 const int n = coords.dimension(0);
125 assert(result.size_array() == n);
126 switch(type_)
127 {
128 case PARSER:
129 {
130 for (int i = 0; i < n; i++)
131 {
132 for (int j = 0; j < dim; j++)
133 parser_.setVar(j, coords(i,j));
134 result[i] = parser_.eval();
135 }
136 break;
137 }
138 case BOITE:
139 {
140 double facteur = 0.;
141 if (dim == 2)
142 facteur = 1. / (l_ * l_ * 4.);
143 else
144 facteur = 1. / (l_ * l_ * l_ * 8.);
145
146 for (int i = 0; i < n; i++)
147 {
148 const double x = coords(i,0);
149 const double y = coords(i,1);
150 const double z = (dim==3) ? coords(i,2) : 0.;
151 double r = facteur;
152 if (x > l_ || x < -l_
153 || y > l_ || y < -l_
154 || z > l_ || z < -l_)
155 r = 0.;
156 result[i] = r;
157 }
158 break;
159 }
160 case CHAPEAU:
161 {
162 const double L2D = l_*l_*l_*l_;
163 const double L3D = L2D*l_*l_;
164 double facteur;
165 if (dim==3)
166 facteur = 1. / L3D;
167 else
168 facteur = 1. / (L2D * l_);
169 const int nbis = coords.dimension(0);
170 for (int i = 0; i < nbis; i++)
171 {
172 const double x = coords(i, 0);
173 const double y = coords(i, 1);
174 const double z = (dim == 3) ? coords(i, 2) : 0.;
175 double resu = 0.;
176 const double ax = std::fabs(x);
177 const double ay = std::fabs(y);
178 const double az = std::fabs(z);
179 if (ax <= l_ && ay <= l_ && az <= l_)
180 resu = (l_-ax) * (l_-ay) * (l_-az) * facteur;
181 result[i] = resu;
182 }
183 break;
184 }
185 case GAUSSIENNE:
186 {
187 double facteur1 = - 0.5 / (l_ * l_);
188 double facteur2 = 1. / (l_ * sqrt(2 * M_PI));
189 if (dim == 2)
190 facteur2 = facteur2 * facteur2;
191 else
192 facteur2 = facteur2 * facteur2 * facteur2;
193 for (int i = 0; i < n; i++)
194 {
195 const double x = coords(i, 0);
196 const double y = coords(i, 1);
197 const double z = (dim == 3) ? coords(i, 2) : 0.;
198 const double k = (x*x + y*y + z*z) * facteur1;
199 result[i] = exp(k) * facteur2;
200 }
201 break;
202 }
203 case QUADRA:
204 {
205 for (int i = 0; i < n; i++)
206 {
207 const double x = coords(i, 0);
208 const double y = coords(i, 1);
209 const double z = (dim == 3) ? coords(i, 2) : 0.;
210 double resu = 0;
211 if (std::fabs(x) < l_ && std::fabs(y) < l_ && std::fabs(z) < l_)
212 {
213 resu = fonction_quadra(x,l_) * fonction_quadra(y,l_);
214 if (dim == 3)
215 resu *= fonction_quadra(z,l_);
216 }
217 result[i] = resu;
218 }
219 break;
220 }
221 default:
222 {
223 Cerr << "Error in Moyenne_volumique::eval() : filter function is not initialized." << finl;
224 exit();
225 }
226 }
227}

232//! Cherche le champ de nom "nom_champ" dans le probleme de nom "nom_pb"<br>dans les objers de l'interprete.<br>Methode appelee par traiter_champs()
233int Moyenne_volumique::get_champ(const Nom& nom_pb,
234 const Nom& nom_champ,
235 REF(Champ_base) & ref_champ)
236{
237 Probleme_base& pb = ref_cast(Probleme_base, objet(nom_pb));
238// Le champ est-il defini dans les postraitements (statistiques) ?
239 const int nb_post = pb.postraitements().size();
240 Motcle mc_nom_champ(nom_champ);
241 for (int i_post = 0; i_post < nb_post; i_post++)
242 {
243 if (sub_type(Postraitement_base, pb.postraitements()[i_post].valeur()))
244 {
245 Postraitement& post = ref_cast(Postraitement, pb.postraitements()[i_post].valeur());
246 Operateurs_Statistique_tps& stats = post.les_statistiques();
247 const int nstat = stats.size();
248 for (int i_stat = 0; i_stat < nstat; i_stat++)
249 {
250 Motcle tmp(stats[i_stat].valeur().le_nom() );
251
252 if (tmp == mc_nom_champ)
253 {
254 Operateur_Statistique_tps_base& stat = stats[i_stat].valeur();
255 ref_cast_non_const(DoubleTab, stat.integrale().valeurs()) = stat.calculer_valeurs();
256 ref_champ = stat.integrale();
257 return 1;
258 }
259 }
260 }
261 }
262
263 ref_champ = pb.get_champ(nom_champ);
264 return 1;
265}

271//! fonction outil permettant de faire les calculs et d'ecrire le resultat<br>dans un fichier lata pour tous les champs d'un type donne de la liste noms_champs.<br>Methode appelee par interpreter()<br>type_champ=0 => traiter les champs aux elements<br>type_champ=1 => traiter les champs aux faces
272void Moyenne_volumique::traiter_champs(const Motcles& noms_champs,
273 const Nom& nom_pb, const Nom& nom_dom,
274 const DoubleTab& coords,
275 Format_Post_base& post,
276 double temps,
277 const Motcle& localisation)
278{
279 const Domaine& dom_post = ref_cast(Domaine, objet(nom_dom));
280 const int nb_champs = noms_champs.size();
281 if (nb_champs == 0)
282 return;
283
284 REF(Champ_base) ref_champ;
285 REF(Zone_VF) ref_zone_vf;
286 int i_champ;
287// ************************************
288// Calcul du nombre total de composantes et de ref_zone_vf
289 int nb_compo_tot = 0;
290 for (i_champ = 0; i_champ < nb_champs; i_champ++)
291 {
292 get_champ(nom_pb, noms_champs[i_champ], ref_champ);
293 const Champ_base& champ = ref_champ.valeur();
294 const Zone_VF& zvf = ref_cast(Zone_VF, champ.zone_dis_base());
295 if (i_champ == 0)
296 {
297 ref_zone_vf = zvf;
298 }
299 else
300 {
301 if (& (ref_zone_vf.valeur()) != & zvf)
302 {
303 Cerr << "Error in Moyenne_volumique::traiter_champs all the fields must be discretized on the same Zone." << finl;
304 barrier();
305 exit();
306 }
307 }
308 const int nb_compo = champ.nb_comp();
309 nb_compo_tot += nb_compo;
310 }
311
312 const Zone_VF& zone_source = ref_zone_vf.valeur();
313
314// ************************************
315// Construction d'un gros tableau contenant toutes les valeurs a traiter plus la porosite
316 DoubleTab valeurs_src;
317 const int nb_lignes = zone_source.nb_elem();
318 valeurs_src.resize(nb_lignes, nb_compo_tot + 1);
319 int count = 0;
320 {
321 DoubleTab tmp_val;
322 IntVect liste_elems(nb_lignes);
323 for (int i = 0; i < nb_lignes; i++)
324 liste_elems[i] = i;
325 const DoubleTab& xp = zone_source.xp();
326 for (i_champ = 0; i_champ < nb_champs; i_champ++)
327 {
328 get_champ(nom_pb, noms_champs[i_champ], ref_champ);
329 const Champ_base& champ = ref_champ.valeur();
330 const int nb_compo = champ.nb_comp();
331 tmp_val.reset();
332 tmp_val.resize(nb_lignes, nb_compo);
333 champ.valeur_aux_elems(xp, liste_elems, tmp_val);
334
335 for (int i = 0; i < nb_lignes; i++)
336 for (int j = 0; j < nb_compo; j++)
337 valeurs_src(i, count+j) = tmp_val(i, j);
338 count += nb_compo;
339
340 Cout << "Field name = " << ref_champ->le_nom()
341 << " Field type = " << ref_champ->que_suis_je() << finl;
342 }
343 }
344// et une colonne de 1:
345 for (int i = 0; i < nb_lignes; i++)
346 valeurs_src(i, count) = 1.;
347
348// Tableau de resultats:
349 const int nb_coords = coords.dimension(0);
350 DoubleTab resu(nb_coords, nb_compo_tot + 1);
351
352// ************************************
353// Calcul de tous les produits de convolution
354
355 calculer_convolution_champ_elem(zone_source,
356 valeurs_src,
357 coords,
358 resu);
359 Noms nom_dir;
360 nom_dir.add("_X");
361 nom_dir.add("_Y");
362 nom_dir.add("_Z");
363 count = 0;
364 for (i_champ = 0; i_champ < nb_champs; i_champ++)
365 {
366 get_champ(nom_pb, noms_champs[i_champ], ref_champ);
367 const Champ_base& champ = ref_champ.valeur();
368 const int nb_compo = champ.nb_comp();
369 DoubleTab extrait(nb_coords, nb_compo);
370 for (int i = 0; i < nb_coords; i++)
371 for (int j = 0; j < nb_compo; j++)
372 extrait(i, j) = resu(i, count + j);
373
374 Cout << "Post writting " << champ.le_nom() << finl;
375 Nom nature("scalar");
376 if (champ.nature_du_champ()==vectoriel) nature="vector";
377 post.ecrire_champ(dom_post,
378 champ.unites(),
379 champ.noms_compo(),
380 -1 /* ecrire toutes les composantes */, temps, temps,
381 noms_champs[i_champ], nom_dom, localisation,nature, extrait);
382
383 count += nb_compo;
384 }
385// Derniere colonne (porosite)
386 const int nb_compo = 1;
387 DoubleTab extrait(nb_coords, nb_compo);
388 for (int i = 0; i < nb_coords; i++)
389 for (int j = 0; j < nb_compo; j++)
390 extrait(i, j) = resu(i, count + j);
391
392 Noms noms_compo;
393 Noms unites;
394 Nom nom_moyenne;
395 unites.add("m3");
396 noms_compo.add("porosite");
397 nom_moyenne = "porosite";
398 Cout << "Porosity post writing" << finl;
399
400 post.ecrire_champ(dom_post, unites, noms_compo, -1 /* ecrire toutes les composantes */,
401 temps, temps,
402 nom_moyenne, nom_dom, localisation, "scalar",extrait);
403}

416//! Lecture des parametres dans le jeu de donnees. Format attendu:<br>Moyenne_volumique {<br>nom_pb NOM_DU_PROBLEME (ou chercher les champs sources)<br>nom_domaine DOMAINE_CIBLE (on evalue la convolution aux elements de ce domaine)<br>noms_champs N CHAMP1 CHAMP2 ... (noms des champs a filtrer dans le probleme)<br>[ nom_fichier_post NOM_SANS_EXTENSION ] (soit on donne nom_fichier et format_post,<br>soit on donne fichier_post)<br>[ format_post lata|meshtv|lml|med|... ] (par defaut lata)<br>[ fichier_post Format_Post_XXX { ... } ] (lecture par readOn du Format_Post_XXX)<br>fonction_filtre ... (format : voir Moyenne_volumique::readOn() )<br>[ localisation ELEM|SOM ]<br>}
417Entree& Moyenne_volumique::interpreter(Entree& is)
418{
419 Cerr << "Starting of Moyenne_volumique::interpreter" << finl;
420 Nom nom_pb, nom_dom;
421 Motcles noms_champs;
422 Param param(que_suis_je() + Nom("::interpreter()"));
423 const int id_elem = 0;
424 const int id_som = 1;
425 int localisation = id_elem; //!< par defaut
426 Motcle format_post("lata_v1");
427 Nom nom_fichier_post;
428 DERIV(Format_Post_base) fichier_post;
429 param.ajouter("nom_pb", & nom_pb, Param::REQUIRED);
430 param.ajouter("nom_domaine", & nom_dom, Param::REQUIRED);
431 param.ajouter("noms_champs", & noms_champs, Param::REQUIRED);
432 param.ajouter("fichier_post", & fichier_post);
433 param.ajouter("format_post", & format_post);
434 param.ajouter("nom_fichier_post", & nom_fichier_post);
435// L'objet Moyenne_volumique est un interprete, mais c'est aussi un objet
436// dont la seule propriete est la fonction filtre a utiliser... astuce:
437// on appelle le readOn de la classe pour lire la fonction filtre.
438 param.ajouter("fonction_filtre", this, Param::REQUIRED);
439 param.ajouter("localisation", & localisation);
440 param.dictionnaire("ELEM", id_elem);
441 param.dictionnaire("SOM", id_som);
442 param.lire_avec_accolades_depuis(is);
443
444// on recupere le domaine
445 const Domaine& dom = ref_cast(Domaine, objet(nom_dom));
446 if (noms_champs.size() == 0)
447 {
448 Cerr << "Moyenne_volumique : no field to treat" << finl;
449 return is;
450 }
451 Cerr << "Writing of the post-processing domain : " << nom_dom << finl;
452 REF(Champ_base) ref_champ;
453 get_champ(nom_pb, noms_champs[0], ref_champ);
454 const double temps = ref_champ.valeur().temps();
455 if (!fichier_post.non_nul())
456 {
457 if (nom_fichier_post == "??")
458 {
459 Cerr << "Error in Moyenne_volumique::interpreter:\n"
460 << " missing NOM_FICHIER_POST or FICHIER_POST keyword" << finl;
461 barrier();
462 exit();
463 }
464 if (format_post == "lata")
465 format_post = "lata_v1";
466// Astuce pour permettre le cas test de non regression en sequentiel et en parallele:
467// (il faut que le nom du fichier de sortie soit le nom du cas)
468 if (nom_fichier_post == "NOM_DU_CAS")
469 {
470 Cerr << "Post filename = NOM_DU_CAS => using " << nom_du_cas() << " instead" << finl;
471 nom_fichier_post = nom_du_cas();
472 }
473 fichier_post.typer(Motcle("FORMAT_POST_") + format_post);
474 fichier_post.valeur().initialize(nom_fichier_post, 1 /* binaire */, "SIMPLE");
475 }
476 else
477 {
478 if (nom_fichier_post != "??")
479 {
480 Cerr << "Error in Moyenne_volumique::interpreter:\n"
481 << " you cannot give NOM_FICHIER_POST and FICHIER_POST. Choose one" << finl;
482 barrier();
483 exit();
484 }
485 }
486 Format_Post_base& post = fichier_post.valeur();
487 post.ecrire_entete(temps, 0 /*reprise*/, 1 /* premier post */);
488 post.ecrire_domaine(dom, 1 /* premier_post */);
489 post.ecrire_temps(temps);
490
491// Coordonnees des centres des elements du domaine destination
492 DoubleTab coords;
493 if (localisation == id_elem)
494 {
495 dom.zone(0).calculer_centres_gravite(coords);
496// Le tableau contient aussi les elements virtuels et pas d'espace virtuel. bouh.
497 coords.resize(dom.zone(0).nb_elem(), coords.dimension(1));
498 }
499 else
500 {
501 coords = dom.les_sommets();
502 }
503
504 Motcle loc("ELEM");
505 if (localisation == id_som)
506 loc = "SOM";
507 traiter_champs(noms_champs,
508 nom_pb,
509 nom_dom,
510 coords,
511 post,
512 temps,
513 loc);
514 int fin=1;
515 post.finir(fin);
516 return is;
517}

568//
569//
570//
571//
572//! evalue le produit de convolution "filter_ * champ_source_" au point x,y,z<br>et stocke le resultat dans resu. On determine les elements a utiliser en fonction<br>de la taille du filtre en utilisant une octree.<br>On suppose que le champ source est aux elements.<br>Methode appelee par Moyenne_volumique::calculer_convolution()
573void Calcul_integrale_locale::calculer(const double x, const double y, const double z,
574 ArrOfDouble& resu)
575{
576 const double box_size = filter_.box_size();
577 octree_.search_elements_box(x - box_size, y - box_size, z - box_size,
578 x + box_size, y + box_size, z + box_size,
579 liste_elems_);
580
581 const DoubleTab& coord_items = zone_source_.xp();
582
583 const int nb_items = liste_elems_.size_array();
584 const int dim = Objet_U::dimension;
585 const int dim3 = (dim == 3);
586 filter_coords_.resize(nb_items, dim);
587 for (int i = 0; i < nb_items; i++)
588 {
589 const int item = liste_elems_[i];
590 assert(item < nb_items_reels_);
591 filter_coords_(i, 0) = coord_items(item, 0) - x;
592 filter_coords_(i, 1) = coord_items(item, 1) - y;
593 if (dim3)
594 filter_coords_(i, 2) = coord_items(item, 2) - z;
595 }
596 filter_results_.resize_array(nb_items);
597 filter_.eval_filtre(filter_coords_, filter_results_);
598
599 const DoubleVect& volumes = zone_source_.volumes();
600 const int nb_comp = champ_source_.dimension(1);
601 resu = 0.;

603 {
604 const int item = liste_elems_[i];
605 const double valeur_filtre = filter_results_[i];
606 const double volume = volumes(item);
607 const double facteur = valeur_filtre * volume;
608 for (int j = 0; j < nb_comp; j++)

610// L'integrale est discretisee grossierement comme le produit des
611// valeurs au centre de l'element fois le volume de l'element:

620//! methode generale pour calculer une convolution a partir d'un<br>champ aux elements ou aux faces. Methode appelee par calculer_convolution_champ_elem()<br>et calculer_convolution_champ_face()
621void Moyenne_volumique::calculer_convolution(const Zone_VF& zone_source,
622 const DoubleTab& champ_source,
623 const DoubleTab& coords_to_compute,
624 DoubleTab& resu) const
625{
626 assert(resu.dimension(0) == coords_to_compute.dimension(0));
627 const int dim = Objet_U::dimension;
628 const int nbproc = Process::nproc();
629 const int nb_coords_to_compute = coords_to_compute.dimension(0);
630 const int nb_coords_max = ::mp_max(nb_coords_to_compute);
631
632 int nb_comp;
633 nb_comp = champ_source.line_size();
634 assert(resu.line_size() == nb_comp);
635
636 DoubleTab coords(nbproc, 3);
637 ArrOfInt flag(nbproc);
638 ArrOfDouble resu_partiel(nb_comp);
639 DoubleTab resu_partiels(nbproc, nb_comp);
640
641 Calcul_integrale_locale integrale_locale(zone_source,
642 *this,
643 champ_source);
644
645// Boucle sur les coordonnees x locales pour lesquelles on veut calculer l'integrale I(x).
646// Si x est pres du bord, le support de la fonction filtre couvre des processeurs voisins
647// qu'il faut ajouter. Pour chaque coordonnees, on demande a tous les autres processeurs
648// de calculer la contribution. Donc on boucle sur le max du nombre de coordonnees pour
649// synchroniser les procs:
650 int i, j;
651 for (int i_coord = 0; i_coord < nb_coords_max; i_coord++)
652 {
653// Chaque processeur envoie la coordonnee a calculer a tous les processeurs:
654 if (i_coord < nb_coords_to_compute)
655 {
656 for (j = 0; j < dim; j++)
657 {
658 double x = coords_to_compute(i_coord, j);
659 for (i = 0; i < nbproc; i++)
660 coords(i, j) = x;
661 }
662 flag = 1;
663 }
664 else
665 {
666 flag = 0;
667 }
668 envoyer_all_to_all(coords, coords);
669 envoyer_all_to_all(flag, flag);
670// On a dans coord(pe, i) les coordonnees demandees par chaque processeur et
671// flag[pe] indique si le processeur a demande une coordonnee ou s'il a fini
672// sa liste coords_to_compute.
673// On calcule maintenant la contribution du processeur local aux integrales I(x)
674// pour ces coordonnees. En general, le processeur qui a cette coordonnee chez lui
675// a beaucoup de travail pour celle-ci et presque rien a faire pour les autres coordonnees
676// (si la coordonnee est loin du domaine, l'octree trouve tres rapidement la liste vide).
677// Donc les calculs sont a peu pres equilibres sur les processeurs.
678 for (i = 0; i < nbproc; i++)
679 {
680 if (flag[i])
681 {
682 integrale_locale.calculer(coords(i, 0), coords(i, 1), coords(i, 2), resu_partiel);
683 for (j = 0; j < nb_comp; j++)
684 resu_partiels(i, j) = resu_partiel[j];
685 }
686 }
687// On renvoie a chaque processeur la contribution du processeur local pour la coordonnee
688// qu'il a demande:
689 envoyer_all_to_all(resu_partiels, resu_partiels);
690// Le resultat est la somme des contributions de tous les processeurs:
691 if (i_coord < nb_coords_to_compute)
692 {
693 for (j = 0; j < nb_comp; j++)
694 {
695 double x = 0.;
696 for (i = 0; i < nbproc; i++)
697 x += resu_partiels(i, j);
698 resu(i_coord, j) = x;
699 }
700 }
701 }
702}
