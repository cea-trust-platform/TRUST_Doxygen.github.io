42//! remplit le tableau face_normales dans la Zone_poly
43void Quadri_poly::normale(int num_Face,DoubleTab& Face_normales,
44 const IntTab& Face_sommets,
45 const IntTab& Face_voisins,
46 const IntTab& elem_faces,
47 const Zone& zone_geom) const
48{
49 const DoubleTab& les_coords = zone_geom.domaine().coord_sommets();
50 double x1,y1;
51 double nx,ny;
52 double x1g=0,y1g=0;
53 double x2g=0,y2g=0;
54 double grx,gry,psc;
55 int sign=1,i;
56 int n0 = Face_sommets(num_Face,0);
57 int n1 = Face_sommets(num_Face,1);
58 x1 = les_coords(n0,0)-les_coords(n1,0);
59 y1 = les_coords(n0,1)-les_coords(n1,1);
60 nx = -y1;
61 ny = x1;
62 int elem1=Face_voisins(num_Face,0);
63 int elem2=Face_voisins(num_Face,1);
64
65// Orientation de la normale vers le plus grand numero d'elem
66// pour cela on teste d'abord si l'on est sur le bord
67 if (elem2!=-1)
68 {
69// on oriente a partir du centre de gravite
70// calcul du centre de gravite de chaque element
71 for(i=0; i<4; i++)
72 {
73 x1g+=les_coords(Face_sommets(elem_faces(elem1,i),0),0);
74 x1g+=les_coords(Face_sommets(elem_faces(elem1,i),1),0);
75 y1g+=les_coords(Face_sommets(elem_faces(elem1,i),0),1);
76 y1g+=les_coords(Face_sommets(elem_faces(elem1,i),1),1);
77 x2g+=les_coords(Face_sommets(elem_faces(elem2,i),0),0);
78 x2g+=les_coords(Face_sommets(elem_faces(elem2,i),1),0);
79 y2g+=les_coords(Face_sommets(elem_faces(elem2,i),0),1);
80 y2g+=les_coords(Face_sommets(elem_faces(elem2,i),1),1);
81 }
82
83 grx=(x2g-x1g)*0.125;
84 gry=(y2g-y1g)*0.125;
85
86// on regarde le signe du produit scalaire
87 psc=grx*nx+gry*ny;
88 if(psc<0)
89 {
90 if(elem1<elem2)
91 sign=-1;
92 }
93 else if(elem2<elem1)
94 sign=-1;
95 }
96 else
97 {
98// on oriente a partir du centre de gravite et du milieu de la
99// face courante
100
101 for(i=0; i<4; i++)
102 {
103 x1g+=les_coords(Face_sommets(elem_faces(elem1,i),0),0);
104 x1g+=les_coords(Face_sommets(elem_faces(elem1,i),1),0);
105 y1g+=les_coords(Face_sommets(elem_faces(elem1,i),0),1);
106 y1g+=les_coords(Face_sommets(elem_faces(elem1,i),1),1);
107 }
108// Cerr << "xg et yg de Face_normales: " << x1g << " " << y1g << finl;
109
110 x2g = les_coords(n0,0)+les_coords(n1,0);
111 y2g = les_coords(n0,1)+les_coords(n1,1);
112 grx=x2g*0.5-x1g*0.125;
113 gry=y2g*0.5-y1g*0.125;
114
115// Cerr << "grx et gry : " << grx << " " << gry << finl;
116// on regarde le signe du produit scalaire
117 psc=grx*nx+gry*ny;
118 if(psc<0)
119 sign=-1;
120 }
121 Face_normales(num_Face,0)=sign*nx;
122 Face_normales(num_Face,1)=sign*ny;
123}
