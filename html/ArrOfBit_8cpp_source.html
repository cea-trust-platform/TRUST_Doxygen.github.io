50//! Taille en "int" du tableau requis pour stocker un tableau de bits<br>de taille donnees.
51int ArrOfBit::calculer_int_size(int la_taille) const
52{
53 assert(la_taille >= 0);
54 int siz = la_taille >> SIZE_OF_INT_BITS;
55 if (la_taille & DRAPEAUX_INT)
56 siz++;
57 return siz;
58}

63//! Change la taille du tableau et copie les donnees<br>existantes. Si la taille est plus petite, les donnees sont<br>tronquees, et si la taille est plus grande, les nouveaux elements<br>ne sont pas initialises.
64ArrOfBit& ArrOfBit::resize_array(int n)
65{
66 if (taille == n)
67 return *this;
68 assert(n >= 0);
69 if (n > 0)
70 {
71 int oldsize = calculer_int_size(taille);
72 int newsize = calculer_int_size(n);
73 unsigned int * newdata = new unsigned int[newsize];
74 int size_copy = (newsize > oldsize) ? oldsize : newsize;
75 if (size_copy)
76 {
77 memcpy(newdata, data, size_copy);
78 delete[] data;
79 }
80 data = newdata;
81 taille = n;
82 }
83 else
84 {
85 delete[] data; //!< data!=0 sinon taille==n et on ne serait pas ici
86 data = 0;
87 taille = 0;
88 }
89 return *this;
90}

92//! Operateur copie (deep copy).
93ArrOfBit& ArrOfBit::operator=(const ArrOfBit& array)
94{
95 int newsize = calculer_int_size(array.taille);
96 if (taille != array.taille)
97 {
98 if (data)
99 {
100 delete[] data;
101 data = 0;
102 }
103 if (newsize > 0)
104 data = new unsigned int[newsize];
105 }
106 taille = array.taille;
107 if (taille)
108 memcpy(data, array.data, newsize * sizeof(unsigned int));
109 return *this;
110}

113//! Si la valeur est non nulle, met la valeur 1 dans<br>tous les elements du tableau, sinon met la valeur 0.
114
115ArrOfBit& ArrOfBit::operator=(int val)
116{
117 unsigned int valeur = val ? (~((unsigned int) 0)) : 0;
118 int size = calculer_int_size(taille);
119 int i;
120 for (i = 0; i < size; i++)

127//! Ecriture du tableau. Format:<br>n<br>0 1 0 0 1 0 ... (n valeurs)
128Sortie& ArrOfBit::printOn(Sortie& os) const
129{
130 os << taille << finl;
131 int i;
132// Un retour a la ligne tous les 32 bits,
133// Une espace tous les 8 bits
134 for (i = 0; i < taille; i++)
135 {
136 os << operator[](i);
137 if ((i & 7) == 7)
138 os << space;
139 if ((i & 31) == 31)
140 os << finl;
141 }
142// Un retour a la ligne si la derniere ligne n'etait pas terminee
143 if (i & 31)
144 os << finl;
145 return os;
146}

151Entree& ArrOfBit::readOn(Entree& is)
152{
153 int newsize;

156 operator=(0);
