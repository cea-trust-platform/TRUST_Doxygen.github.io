58//! Lit un champ uniforme par morceaux tabule dans<br>le temps a partir d'un flot d'entree.
59/** \return Entree\&<br>
60Signification: le flot d'entre modifie<br>
61Contraintes: */
62Entree& Champ_Uniforme_Morceaux_Tabule_Temps::readOn(Entree& is)
63{
64 int dim;
65 Motcle motlu;
66 Nom nom;
67 int k, poly,nb_val;
68 is >> nom;
69 mon_domaine = ref_cast(Domaine, Interprete::objet(nom));
70 Domaine& le_domaine=mon_domaine.valeur();
71 dim=lire_dimension(is,que_suis_je());
72 valeurs_.resize(0, dim);
73 le_domaine.creer_tableau_elements(valeurs_);
74 is >> nom;
75 motlu=nom;
76 if(motlu != Motcle("{") )
77 {
78 Cerr << "Error while reading a " << que_suis_je() << finl;
79 Cerr << "We expected a { instead of " << nom << finl;
80 exit();
81 }
82 is >> nom;
83 motlu=nom;
84 if(motlu != Motcle("defaut") )
85 {
86 Cerr << "Error while reading a " << que_suis_je() << finl;
87 Cerr << "We expected defaut instead of " << nom << finl;
88 exit();
89 }
90 DoubleVect val(dim);
91 for( k=0; k< dim; k++) is >> val(k);
92 for( poly=0; poly<le_domaine.zone(0).nb_elem(); poly++)
93 for( k=0; k< dim; k++)
94 valeurs_(poly,k)=val(k);
95 is >> nom;
96 motlu=nom;
97 while (motlu != Motcle("}") )
98 {
99 REF(Sous_Zone) refssz=les_sous_zones.add(le_domaine.ss_zone(nom));
100 is >> motlu;
101 if (motlu == Motcle("{") )
102 {
103 Table la_table;
104 is >> nb_val;
105 DoubleVect param(nb_val);
106 DoubleTab tab_valeurs(nb_val,dim);
107 for (int i = 0; i < nb_val; i++)
108 is >> param[i];
109 for (int i = 0; i < nb_val; i++)
110 for (k = 0; k < dim; k++)
111 is >> tab_valeurs(i, k);
112 la_table.remplir(param,tab_valeurs);
113 les_tables.add(la_table);
114 is >> motlu;
115 if (motlu != Motcle("}") )
116 {
117 Cerr << "Error while reading an object of type Champ_Uniforme_Morceaux_Tabule_Temps" << finl;
118 Cerr << "We expected keyword } instead of " << motlu << finl;
119 exit();
120 }
121 }
122 else
123 {
124 Cerr << "Error while reading an object of type Champ_Uniforme_Morceaux_Tabule_Temps" << finl;
125 Cerr << "We expected keyword { instead of " << motlu << finl;
126 exit();
127 }
128 is >> nom;
129 motlu=nom;
130 }
131 return is;
132}

150//! Mise a jour en temps du champ (tabule en temps)
151void Champ_Uniforme_Morceaux_Tabule_Temps::me_calculer(double tps)
152{
153 if (!est_egal(temps(),tps))
154 {
155 LIST_CURSEUR(REF(Sous_Zone)) curseur1(les_sous_zones);
156 LIST_CURSEUR(Table) curseur2(les_tables);
157
158 if (nb_comp() == 1)
159 {
160 DoubleVect& mes_valeurs = valeurs();
161 double val_ch;
162 while(curseur1)
163 {
164 Sous_Zone& ssz = (curseur1.valeur()).valeur();
165 Table& la_table = curseur2.valeur();
166
167 val_ch = la_table.val(tps);
168
169 for( int poly=0; poly<ssz.nb_elem_tot(); poly++)
170 mes_valeurs(ssz(poly)) = val_ch;
171
172 ++curseur1;
173 ++curseur2;
174 }
175 }
176 else
177 {
178 DoubleTab& mes_valeurs = valeurs();
179 int k,dim = nb_comp();
180 DoubleVect val_loc(dim);
181
182 while(curseur1)
183 {
184 Sous_Zone& ssz = (curseur1.valeur()).valeur();
185 Table& la_table = curseur2.valeur();
186
187 la_table.valeurs(val_loc,tps);
188
189 for(int poly=0; poly<ssz.nb_elem_tot(); poly++)
190 for( k=0; k< dim; k++)
191 mes_valeurs(ssz(poly),k) = val_loc(k);
192
193 ++curseur1;
194 ++curseur2;
195 }
196 }
197 }
198 valeurs().echange_espace_virtuel();
199}
