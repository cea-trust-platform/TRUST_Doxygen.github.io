35Entree& Solv_Cholesky::readOn(Entree& is )
36{
37 if ( Process::nproc()>1 )
38 {
39 Cerr << finl;
40 Cerr << "Cholesky solver not parallelized in TRUST!!!" << finl;
41 Cerr << "Change your solver." << finl;
42 Cerr << "Try using Petsc Cholesky solver." << finl;
43 exit();
44 }
45
46 Motcle accolade_ouverte("{");
47 Motcle accolade_fermee("}");
48 Motcles les_parametres(3);
49 {
50 les_parametres[0] = "impr";
51 les_parametres[1] = "save_matrice|save_matrix";
52 les_parametres[2] = "quiet";
53 }
54 int rang;
55
56 Motcle motlu;
57 is >> motlu;
58 if (motlu != accolade_ouverte)
59 {
60 Cerr << "Error when reading parameters for Cholesky " << finl;
61 Cerr << "One expected : " << accolade_ouverte << finl;
62 exit();
63 }
64 is >> motlu;
65 while (motlu != accolade_fermee)
66 {
67 rang = les_parametres.search(motlu);
68 switch(rang)
69 {
70 case 0:
71 {
72 fixer_limpr(1);
73 break;
74 }
75 case 1:
76 {
77 save_matrice_ = 1;
78 break;
79 }
80 case 2:
81 {
82 fixer_limpr(-1);
83 break;
84 }
85 default :
86 {
87 Cerr << "Error when reading parameters for Cholesky " << finl;
88 Cerr << "On expected " << les_parametres << " instead of " << motlu << finl;
89 exit();
90 }
91 }
92 is >> motlu;
93 }
94 return is;
95}

97int Solv_Cholesky::resoudre_systeme(const Matrice_Base& la_matrice,
98 const DoubleVect& secmem,
99 DoubleVect& solution)
100{
101 if(sub_type(Matrice_Morse_Sym,la_matrice))
102 {
103 const Matrice_Morse_Sym& matrice = ref_cast(Matrice_Morse_Sym,la_matrice);
104 return Cholesky(matrice,secmem,solution);
105 }
106 else
107 {
108 if(sub_type(Matrice_Bloc_Sym,la_matrice))
109 {
110// Conversion d'une matrice au format Matrice_Bloc_Sym au format Matrice Morse
111 if (nouvelle_matrice())
112 ref_cast(Matrice_Bloc_Sym,la_matrice).BlocSymToMatMorseSym(matrice_de_travail);
113 return Cholesky(matrice_de_travail,secmem,solution);
114 }
115 else if(sub_type(Matrice_Bloc,la_matrice))
116 {
117 const Matrice_Bloc& matrice = ref_cast(Matrice_Bloc,la_matrice);
118 if((matrice.nb_bloc_lignes()>1 && ref_cast(Matrice_Morse,matrice.get_bloc(1,0).valeur()).nb_lignes()>0))
119 {
120 Cerr<<"Solv_Cholesky : WARNING : Cholesky by blocks cannot be done"<<finl;
121 exit();
122 return(-1);
123 }
124 const Matrice_Morse_Sym& MB00 = ref_cast(Matrice_Morse_Sym,matrice.get_bloc(0,0).valeur());
125 return Cholesky(MB00,secmem,solution);
126 }
127 else
128 {
129 Cerr<<"Solv_Cholesky : Warning : one is not able to carry out preconditionning for linear systems based on Matrice_Morse_Sym or Matrice_Bloc type matrixes"<<finl;
130 exit();
131 return(-1);
132 }
133 }

136int Solv_Cholesky::Cholesky(const Matrice_Morse_Sym& matrice,
137 const DoubleVect& secmem,
138 DoubleVect& solution)
139{
140 int avec_renumerotation=1;
141 if (nouvelle_matrice())
142 {
143 fixer_nouvelle_matrice(0);
144// On rend definie la matrice dans le cas sequentiel ou Cholesky est le solveur en pression
145// En parallele, Cholesky est utilise en preconditionnement local et il ne faut
146// pas modifier la matrice locale.
147 Cerr << "Order of the matrix : " << matrice.ordre() << finl;
148// Les champs P1bulle n'ont pas de size_reelle et en parallele ca ne marche pas...
149 const int sz = secmem.size_totale();
150 if(matrice.get_est_definie()==0 && nproc()==1)
151 {
152 Cerr << "Matrix not defined : " << finl;
153 Cerr << "zero value is imposed at point 0" << finl;
154 Matrice_Morse_Sym& mat = ref_cast_non_const(Matrice_Morse_Sym,matrice);
155 mat(0,0)*=1E+6;
156 }
157 if (avec_renumerotation)
158 {
159 matrice.renumerote();
160 Fact_Cholesky(matrice.matrice_renumerotee(),sz);
161 }
162 else
163 Fact_Cholesky(matrice, sz);
164 }
165
166
167// int ordre = la_matrice.ordre();
168 DoubleVect vecteur(secmem);
169 double* vect_data = vecteur.addr();
170// solution = 0.0;
171 const int n = secmem.size_totale();//ordre;
172 int N=n;
173
174// permutation du second membre
175// subroutine dvperm (n, x, perm)
176// SPARSKIT2/FORMATS/unary.f
177 if (avec_renumerotation) F77NAME(DVPERM)(&n, vect_data, matrice.permutation().addr());
178
179
180 char UPLO = 'U'; //!< A est triangulaire superieure
181 int KD = largeur_de_bande_-1; //!< largeur de bande sup int N = n; //m_ // ordre de A (=nb col de abd_)
182//
183 int LDAB = largeur_de_bande_; //n_;//!< nb de lg de abd_
184 int LDB = n; //m_;//!< nb de lg de b
185 int NRHS = 1; //!< nb de col de b (=vecteur)
186 int INFO = 0;
187
188
189// resolution du systeme
190#ifdef CRAY
191 Nom UPLOCpp(UPLO);
192 _fcd UPLOF90;
193 nomCtonomF90(UPLOCpp,UPLOF90);
194 F77NAME(SPBTRS)(UPLOF90, &N, &KD, &NRHS,
195 matrice_bande_factorisee_fortran_.addr() ,
196 &LDAB, vecteur.addr(), &LDB, &INFO);
197#else
198 F77NAME(DPBTRS)(&UPLO, &N, &KD, &NRHS, matrice_bande_factorisee_fortran_.addr() , &LDAB, vecteur.addr(), &LDB, &INFO);
199#endif
200
201 if(INFO)
202 {
203 Cerr << "Solv_Cholesky::resoudre_systeme error : ";
204 if(INFO<0) Cerr << "F77NAME(DPBTRS) param. " << -INFO << " invalid" << finl;
205 if(INFO>0) Cerr << "singular matrix ! resolution impossible" << finl;
206 exit();
207 };
208
209
210// permutation inverse du resultat
211// subroutine dvperm (n, x, perm)
212// SPARSKIT2/FORMATS/unary.f
213 if (avec_renumerotation) F77NAME(DVPERM)(&n, vect_data, matrice.permutation_inverse().addr());
214
215 solution = vecteur;
216 solution.echange_espace_virtuel();
217// test
218// DoubleVect res;
219// Cout<<"second membre "<<secmem.norme()<<finl;
220// Cout<<"solution "<<solution.norme()<<finl;
221// res = matrice*solution - secmem;
222// Cout<<"residu "<<res.norme()<<finl;
223
224 return 1;
225}

229int Solv_Cholesky::Fact_Cholesky(const Matrice_Morse_Sym& mat1, const int size_reelle)
230{
231 largeur_de_bande_ = mat1.largeur_de_bande();
232// stockage de la matrice en format bande FORTRAN
233 matrice_bande_factorisee_fortran_.resize_array(size_reelle*largeur_de_bande_);
234
235 for (int j=0; j<size_reelle; j++)
236 {
237 int i;
238 i = j+1 - largeur_de_bande_;
239 int k = 0;
240 while (k < largeur_de_bande_)
241 {
242 if (i >= 0)
243 {
244// Cout<<"i j "<<i<<" "<<j<<" "<<mat1(i,j)<<" "<<me()<<finl;
245 const double tmpa = mat1(i,j);
246 matrice_bande_factorisee_fortran_[j * largeur_de_bande_ + k] = tmpa;
247 }
248
249 i++;
250 k++;
251 }
252 }
253
254// Factorisation de Cholesky
255
256// int DBG=1;
257// int PRC=0;
258 char UPLO = 'U';
259 int N = size_reelle;
260 int KD = largeur_de_bande_ - 1;//!< mu nbre de diagonales superieures
261 int LDAB = largeur_de_bande_; //n_
262 int INFO = 0;
263
264
265 Cerr << "Starting factorisation ..." << finl;
266#ifdef CRAY
267 Nom UPLOCpp(UPLO);
268 _fcd UPLOF90;
269 nomCtonomF90(UPLOCpp,UPLOF90);
270 SPBTRF(UPLOF90, &N, &KD,
271 matrice_bande_factorisee_fortran_.addr() , &LDAB, &INFO);
272#else
273 F77NAME(DPBTRF)(&UPLO, &N, &KD,
274 matrice_bande_factorisee_fortran_.addr() , &LDAB, &INFO);
275#endif
276 Cerr << "End factorization." << finl;
277
278// {
279// double ANORM=1.0;
280// double RCOND;
281// ArrOfDouble WORK(3*N);
282// ArrOfInt IWORK(N);
283// F77NAME(DPBCON)(&UPLO, &N, &KD,
284// matrice_bande_factorisee_fortran_.addr() , &LDAB,
285// &ANORM, &RCOND, WORK.addr(), IWORK.addr(), &INFO);
286// Cerr<<"//////////// RCOND : "<<RCOND<<finl;
287// }
288 if(INFO)
289 {
290 Cerr << "Solv_Cholesky::Fact_Cholesky() error : ";
291 if(INFO<0) Cerr << "F77NAME(DPBTRF) param. " << -INFO
292 << " invalid" << finl;
293 if(INFO>0) Cerr << "submatrix of order " << INFO
294 << " not positive definite!" << finl;
295 if (INFO<100)
296 {
297 for (int i=0; i<INFO; i++)
298 Cerr << "A(" <<i<<","<<i<<")="<<mat1(i,i)<<finl;
299 Cerr << "non-positive definite matrix ="<<finl;
300 mat1.imprimer_formatte(Cerr);
301 }
302 exit();
303 };
304
305// rovisoire : on pourrait vider matrice et matrice_renumerotee
306 return 1;
307
308}
