58Entree& Paroi_Knudsen_non_negligeable::readOn(Entree& s )
59{
60 Cerr << "Paroi_Knudsen_non_negligeable::readOn" << finl;
61 Motcle mot;

63 {
64 les_motcles[0] = "vitesse_paroi";
65 les_motcles[1] = "k";
66 }
67 int nb_champs_lus=0;
68 while (nb_champs_lus<2)
69 {
70 s >> mot ;
71 int rang = les_motcles.search(mot);
72 if (rang == 0)
73 s >> vitesse_paroi_ ;
74 if (rang == 1)
75 s >> k_ ;
76 nb_champs_lus++;
77 }
78
79 le_champ_front.typer("Champ_front_fonc_gradient");
80 le_champ_front->fixer_nb_comp(dimension);
81 return s;
82}
