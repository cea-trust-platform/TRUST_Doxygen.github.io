32//! Convertit une coordonnees reele en coordonnee entiere pour l'octree_int<br>Valeur de retour: 1 si ok, 0 si coordonnee hors de l'octree
33inline int Octree_Double::integer_position(double x, int direction, int& ix) const
34{
35 const double coord_max = (double) Octree_Int::coord_max_;
36 double rnd_x = (x - origin_[direction]) * factor_[direction];
37// 0.49 permet d'accepter une coordonnee x egale a xmin ou xmax de l'octree,
38// sinon pour un octree cree a partir de sommets, il y a un risque
39// de ne pas trouver les coordonnees des points qu'on avait mis au bord de l'octree.
40 if (rnd_x >= -0.49 && rnd_x <= coord_max + 0.49)
41 {
42 ix = (int) floor(rnd_x + 0.5);
43 return 1;
44 }
45 return 0;
46}

48// Valeur de retour: 1 s'il y a une intersection non vide avec l'octree, 0 sinon
49inline int Octree_Double::integer_position_clip(double xmin, double xmax,
50 int& x0, int& x1,
51 int direction) const
52{
53 const double coord_max = (double) Octree_Int::coord_max_;
54 xmin = (xmin - origin_[direction]) * factor_[direction];
55 xmax = (xmax - origin_[direction]) * factor_[direction];
56// pas de marge ici comme on cherche avec une boite, l'epsilon est deja
57// dans la dimension de la boite.
58 if (xmin > coord_max || xmax < 0.)
59 return 0;
60 if (xmin < .0)
61 x0 = 0;
62 else
63 x0 = (int) (floor(xmin+0.5));
64 if (xmax > coord_max)
65 x1 = Octree_Int::coord_max_;
66 else
67 x1 = (int) (floor(xmax+0.5));
68 return 1;
69}

94//! methode outil pour build_nodes et build_elements<br>(calcul des facteurs de conversion entre reels et entiers pour Octree_Int<br>epsilon est ajoute aux coordonnees min et max pour la bounding box de l'octree.
95void Octree_Double::compute_origin_factors(const DoubleTab& coords,
96 const double epsilon,
97 const int include_virtual)
98{
99// Recherche des coordonnees min et max du domaine
100 const int nb_som = include_virtual ? coords.dimension_tot(0) : coords.dimension(0);
101 if (nb_som == 0)
102 return; //!< octree vide
103
104 const int dim = coords.dimension(1);
105 dim_ = dim;
106 origin_.resize_array(3);
107 factor_.resize_array(3);
108 ArrOfDouble xmin(dim);
109 xmin= 1.e37;
110 ArrOfDouble xmax(dim);
111 xmax= -1.e-37;
112 assert(dim >= 1 && dim <= 3);
113 int i, j;
114 for (i = 0; i < nb_som; i++)
115 {
116 for (j = 0; j < dim; j++)
117 {
118 const double x = coords(i, j);
119 if (x < xmin[j])
120 xmin[j] = x;
121 if (x > xmax[j])
122 xmax[j] = x;
123 }
124 }
125 const double coord_max = (double) Octree_Int::coord_max_;
126 for (j = 0; j < dim; j++)
127 {
128 xmin[j] -= epsilon;
129 xmax[j] += epsilon;
130 origin_[j] = xmin[j];
131 if (xmax[j] - xmin[j] > 0.)
132 {
133 factor_[j] = coord_max / (xmax[j] - xmin[j]);
134 }
135 else
136 factor_[j] = 0.;
137 }
138}

146//! construit un octree contenant les points de coordonnees coords.<br>Si include_virtual=1, on stocke coords.dimension_tot(0) elements, sinon on en<br>stocke coords.dimension(0)<br>Si epsilon = 0, on construit un octree de points de taille nulle (chaque point<br>se trouve dans un seul octree_floor)<br>Sinon, on construit un octree d'elements cubiques centres sur les coords, de demie-largeur epsilon.<br>Un point peut alors se trouver dans plusieurs octree_floor.
147void Octree_Double::build_nodes(const DoubleTab& coords, const int include_virtual, const double epsilon)
148{
149 octree_int_.reset();
150 compute_origin_factors(coords, epsilon, include_virtual);
151 const int nb_som = include_virtual ? coords.dimension_tot(0) : coords.dimension(0);
152 if (nb_som == 0)
153 return; //!< octree vide
154 const int dim = coords.dimension(1);
155 if (epsilon < 0.)
156 {
157 Cerr << "Internal error in Octree_Double::build_nodes: negative epsilon" << finl;
158 Process::exit();
159 }
160 int have_epsilon = (epsilon == 0.) ? 0 : 1;
161 IntTab elements_boxes;
162 elements_boxes.resize(nb_som, have_epsilon ? (dim*2) : dim, ArrOfInt::NOCOPY_NOINIT);
163
164 for (int i = 0; i < nb_som; i++)
165 {
166 for (int j = 0; j < dim; j++)
167 {
168 int pos1 = 0;
169 const double x0 = coords(i, j);
170 double x = x0 - epsilon;
171 if (!integer_position(x, j, pos1))
172 {
173 Cerr << "Fatal error in octree : integer position outside octree" << finl;
174 Process::exit();
175 }
176 elements_boxes(i, j) = pos1;
177 if (have_epsilon)
178 {
179 pos1 = 0;
180 double xbis = x0 + epsilon;
181 if (!integer_position(xbis, j, pos1))
182 {
183 Cerr << "Fatal error in octree : integer position outside octree" << finl;
184 Process::exit();
185 }
186 elements_boxes(i, dim+j) = pos1;
187 }
188 }
189 }
190 octree_int_.build(dim, elements_boxes);
191}

197//! Construit un octree a partir d'elements volumiques decrits par des<br>ensembles de sommets. On stocke dans l'octree les parallelipipdes englobant chaque<br>element (contenant tous les sommets de l'element) plus une marge de epsilon.<br>Si include_virtual=1, on stocke elements.dimension_tot(0) elements, sinon on en<br>stocke elements.dimension(0)
198void Octree_Double::build_elements(const DoubleTab& coords, const IntTab& elements,
199 const double epsilon, const int include_virtual)
200{
201 octree_int_.reset();
202 compute_origin_factors(coords, epsilon, include_virtual);
203
204 const int nb_elems = include_virtual ? elements.dimension_tot(0) : elements.dimension(0);
205 const int nb_som_elem = elements.dimension(1);
206 const int dim = coords.dimension(1);
207 IntTab elements_boxes;
208 elements_boxes.resize(nb_elems, dim * 2, ArrOfInt::NOCOPY_NOINIT);
209 for (int i = 0; i < nb_elems; i++)
210 {
211 for (int j = 0; j < dim; j++)
212 {
213 double xmin = 1.e37;
214 double xmax = -1.e37;
215 for (int k = 0; k < nb_som_elem; k++)
216 {
217 const int som = elements(i, k);
218 const double x = coords(som, j);
219 xmin = (x<xmin) ? x : xmin;
220 xmax = (x>xmax) ? x : xmax;
221 }
222 int pos1 = 0, pos2 = 0;
223 if (!integer_position(xmin, j, pos1) || !integer_position(xmax, j, pos2))
224 {
225 Cerr << "Fatal error in octree : integer position outside octree" << finl;
226 Process::exit();
227 }
228 elements_boxes(i, j) = pos1;
229 elements_boxes(i, j+dim) = pos2;
230 }
231 }
232 octree_int_.build(dim, elements_boxes);
233}

236//! cherche tous les elements ou points ayant potentiellement une intersection<br>non vide avec la boite donnee.
237int Octree_Double::search_elements_box(double xmin, double ymin, double zmin,
238 double xmax, double ymax, double zmax,
239 ArrOfInt& elements) const
240{
241 const int dim = dim_;
242 if (dim == 0)
243 {
244 elements.resize_array(0);
245 return 0;
246 }
247 int x0 = 0, x1 = 0, y0 = 0, y1 = 0, z0 = 0, z1 = 0;
248 int ok = integer_position_clip(xmin, xmax, x0, x1, 0);
249 if ((ok) && (dim >= 1))
250 {
251 ok = integer_position_clip(ymin, ymax, y0, y1, 1);
252 if ((ok) && (dim >= 2))
253 ok = integer_position_clip(zmin, zmax, z0, z1, 2);
254 }
255 if (ok)
256 octree_int_.search_elements_box(x0, y0, z0, x1, y1, z1, elements);
257 else
258 elements.resize_array(0);
259 return elements.size_array();
260}

281//! Methode hors classe<br>Cherche parmi les sommets de la liste node_list ceux qui sont a une<br>distance inferieure a epsilon du point (x,y,z). node_list contient des indices de<br>sommets dans le tableau coords. La liste des noeuds verifiant le critere est mise<br>dans node_list. On renvoie l'indice dans le tableau coords du sommet le plus proche.
282int Octree_Double::search_nodes_close_to(double x, double y, double z,
283 const DoubleTab& coords, ArrOfInt& node_list,
284 double epsilon)
285{
286 const int n = node_list.size_array();
287 double eps2 = epsilon * epsilon;
288 int count = 0;
289 const int dim = coords.dimension(1);
290 double dmin = eps2;
291 int nearest = -1;
292 for (int i = 0; i < n; i++)
293 {
294 const int som = node_list[i];
295 double dx = x - coords(som, 0);
296 double dy = (dim >= 2) ? y - coords(som, 1) : 0.;
297 double dz = (dim >= 3) ? z - coords(som, 2) : 0.;
298 double d2 = dx * dx + dy * dy + dz * dz;
299 if (d2 < eps2)
300 {
301 node_list[count] = som;
302 if (d2 < dmin)
303 {
304 dmin = d2;
305 nearest = som;
306 }
307 count++;
308 }
309 }
310 node_list.resize_array(count);
311 return nearest;
312}
