45
46void Op_Conv_kschemas_centre_VEF::convkschemas_centre(const double dK, const int ncomp, int dim, const int poly ,
47 const int poly1, const int poly2,const int jel0,
48 const int jel1,const double psc ,const DoubleTab& tab1 ,
49 DoubleVect& tab_fluent, DoubleVect& flux,
50 const DoubleVect& rx0, const DoubleTab& gradient_elem ,DoubleVect& coord_som) const
51{
52 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
53 const DoubleTab& xv = zone_VEF.xv();
54 int comp,amont,i,j;
55 double deltat0,deltat1;
56 DoubleVect rx(dim);
57 DoubleVect dist(dim);
58 DoubleVect dist_amont(dim);
59 DoubleVect dist_aval(dim);
60 deltat0 = 0.;
61 deltat1 = 0.;
62
63 if ((poly1==-1) || (poly2==-1))
64 {
65 //**** Cas ou on applique le schema amont
66 if (psc >= 0)
67 {
68 amont = jel0;
69 tab_fluent[jel1] += psc;
70 }
71 else
72 {
73 amont = jel1;
74 tab_fluent[jel0] -= psc;
75 }
76
77 for (j=0; j<dim; j++)
78 dist(j) = coord_som(j)/3.0 +xv(amont,j)*2.0/3.0-xv(amont,j);
79
80 if (ncomp == 1)
81 {
82 flux(0) = tab1(amont);
83
84 // on centrepole le flux de la face amont sur la droite reliant les milieux des deux faces.
85 for (j=0; j<dim; j++)
86 flux(0) = flux(0)+gradient_elem(poly,0,j)*dist(j);
87
88 }
89 else
90 for (comp=0; comp<ncomp; comp++)
91 {
92 flux(comp) = tab1(amont,comp);
93 for (j=0; j<dim; j++)
94 flux(comp) = flux(comp)+gradient_elem(poly,comp,j)*dist(j);
95 }
96 }
97
98 else
99 {
100 // **** Cas ou on applique le schema centre
101
102 if (psc >= 0)
103 tab_fluent(jel1) += psc;
104 else
105 tab_fluent(jel0) -= psc;
106
107 rx = rx0;
108 rx *= 2./3.;
109
110 for (j=0; j<dim; j++)
111 {
112 dist_amont(j) = coord_som(j)/3.0 +xv(jel0,j)*2.0/3.0-xv(jel0,j) ;
113 dist_aval(j) = coord_som(j)/3.0 +xv(jel1,j)*2.0/3.0-xv(jel1,j) ;
114 }
115
116 if (ncomp == 1)
117 {
118 flux(0) = 0.5*(tab1(jel0)+tab1(jel1));
119 for (j=0; j<dim; j++)
120 flux(0) = flux(0)+0.5*gradient_elem(poly,0,j)*(dist_amont(j)+dist_aval(j));
121 for (i=0; i<dim; i++)
122 {
123 // Determination des deltat phi0 et phi1
124 deltat0 += gradient_elem(poly1,0,i)*rx(i);
125 deltat1 -= gradient_elem(poly2,0,i)*rx(i);
126 }
127
128
129 // Calcul du flux
130 flux(0) += deltat0/16. + deltat1/16. ;
131 // if (K == 0.5)
132 // flux(0) += deltat0/16. + deltat1/16. ;
133 // else {
134 // Cerr << "****ATTENTION!" << finl;
135 // flux(0) += 0.25*((1.+K)*deltat0 +(1.-K)*deltat1);
136 // }
137 }
138
139 else
140 {
141 for (comp=0; comp<ncomp; comp++)
142 {
143
144 deltat0 = deltat1 = 0.0;
145 flux(comp) = 0.5*(tab1(jel0,comp)+tab1(jel1,comp));
146 for (j=0; j<dim; j++)
147 flux(comp) = flux(comp)+0.5*gradient_elem(poly,comp,j)*(dist_amont(j)+dist_aval(j));
148 for (i=0; i<dim; i++)
149 {
150 deltat0 += gradient_elem(poly1,comp,i)*rx(i);
151 deltat1 += gradient_elem(poly2,comp,i)*rx(i);
152 }
153 // Calcul du flux
154 flux(comp) += deltat0/16. + deltat1/16. ;
155 // if (K == 0.5)
156 // flux(comp) += deltat0/16. + deltat1/16. ;
157 // else
158 // flux(comp) += 0.25*((1.+K)*deltat0 + (1.-K)*deltat1) ;
159
160 }
161 }
162 }
163}

345 } // fin du for faces

348 for (comp0=0; comp0<ncomp_ch_transporte; comp0++)
349 for (i=0; i<dimension; i++)

578
579 }
580
581 }

583 // Cerr << "fluxsom(" << comp << ") = " << fluxsom(comp) << finl;
584 }
585 }
586
587
588 ////////////////////////////////////////////////////////////////////////////
589 // on traite le centre de gravite
590 ////////////////////////////////////////////////////////////////////////////
591
592 if (ncomp_ch_transporte == 1)
593 {
594 for (j=0; j<dimension; j++)
595 {
596 xm = 0.5 *(coord(scom,j)+xv(num3,j));
597 if (psc >= 0)
598 {
599 if (poly1==-1)
600 {
601 fluxg(0) += gradient_elem(poly,0,j)*(xg(poly,j)-xm);

610 {
611 if (poly2==-1)
612 {

614 }
615 else
616 {
617 fluxg(0) += (teta*gradient_elem(poly,0,j) + (1.- teta)*gradient_elem(poly2,0,j))*(xg(poly,j)-xm);
618 }
619 }

621 }
622
623 fluxg(0) += flux(0);
624 fluxg(0) *= psc;
625 }
626 else
627 {
628 for (comp0=0; comp0<ncomp_ch_transporte; comp0++)
629 {
630 fluxg(comp0) = flux(comp0);
631 for (j=0; j<dimension; j++)
632 {
633 xm = 0.5 *(coord(scom,j)+xv(num3,j));
634 if (psc >= 0)
635 {
636 if (poly1==-1)
637 {
638 fluxg(comp0) += gradient_elem(poly,comp0,j)*(xg(poly,j)-xm);
