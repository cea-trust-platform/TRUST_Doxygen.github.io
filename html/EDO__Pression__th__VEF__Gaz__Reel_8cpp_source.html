51Signification: La nouvelle valeur de la pression<br>

53double EDO_Pression_th_VEF_Gaz_Reel::resoudre(double Pth_n)
54{
55 int n_bord ;
56 for (n_bord=0; n_bord<la_zone->nb_front_Cl(); n_bord++)
57 {
58 const Cond_lim& la_cl = la_zone_Cl->les_conditions_limites(n_bord);
59 if (sub_type(Neumann_sortie_libre, la_cl.valeur()))
60 return Pth_n;
61 }
62 double Pth;
63 const DoubleTab& tab_vit = ref_cast(Navier_Stokes_std,le_fluide_->vitesse()->equation()).vitesse()->valeurs();
64 const DoubleTab& tab_hnp1 = le_fluide_->inco_chaleur().valeurs(); //actuel
65 const DoubleTab& tab_hn = le_fluide_->inco_chaleur().passe(); //passe
66 const DoubleTab& tab_rho = le_fluide_->masse_volumique().valeurs(); //actuel
67 const Loi_Etat& loi_ = le_fluide_->loi_etat();
68// const DoubleVect& tab_rhon = loi_->rho_n(); //passe
69
70 int elem, nb_elem=la_zone->nb_elem();
71 double V = 0; //mesure du domaine
72 double Fn = 0; //integrale 1 a l'etape n
73 double Fnp1 = 0; //integrale 1 a l'etape n+1
74 double S = 0; //second membre
75
76 double dt = le_fluide_->vitesse()->equation().schema_temps().pas_de_temps();
77 double v,al,bn, bnp1, hn,hnp1, divu;
78
79 int i, face,nb_faces=la_zone->nb_faces();
80// calcul T* aux elements
81 DoubleTab HstarP0(nb_elem);
82 DoubleTab gradh(nb_faces,dimension);
83 DoubleTab u_gradh(nb_faces);
84 int nfe = la_zone->zone().nb_faces_elem();
85 const IntTab& elem_faces = la_zone->elem_faces();
86 for (elem=0 ; elem<nb_elem ; elem++)
87 {
88 HstarP0(elem) = 0;
89 for (face=0 ; face<nfe ; face++)
90 {
91 hn = tab_hn(elem_faces(elem,face));
92 hnp1 = tab_hnp1(elem_faces(elem,face));
93 HstarP0(elem) += .5*(hn+hnp1);
94 }
95 HstarP0(elem) /= nfe;
96 }
97// calcul gradT*
98 calculer_grad(HstarP0,gradh);
99// calcul u.gradT*
100 for (face=0 ; face<nb_faces ; face++)
101 {
102 u_gradh(face)=0;
103 for (i=0 ; i<dimension ; i++)
104 {
105 u_gradh(face) += gradh(face,i) * tab_vit(face,i);
106 }
107 }
108// calcul divU en P0 et P1
109 DoubleTrav divUP0(nb_elem);
110 ref_cast(Navier_Stokes_std,le_fluide_->vitesse()->equation()).operateur_divergence().calculer(tab_vit,divUP0);
111 DoubleTrav divU(nb_faces);
112 int e0,e1;
113 for (face=0 ; face<nb_faces ; face++)
114 {
115 e0 = la_zone->face_voisins(face,0);
116 e1 = la_zone->face_voisins(face,1);
117 if (e0!=-1 && e1!=-1)
118 {
119 divU(face) = .5*(divUP0(e0)+divUP0(e1));
120 }
121 else if (e0!=-1)
122 {
123 divU(face) = divUP0(e0);
124 }
125 else
126 {
127 divU(face) = divUP0(e1);
128 }
129 }
130
131 for (face=0 ; face<nb_faces ; face++)
132 {
133 v = la_zone->volumes_entrelaces(face);
134 V += v;
135 hn = tab_hn(face);
136 hnp1 = tab_hnp1(face);
137 al = loi_->Drho_DT(Pth_n,hn) / loi_->Drho_DP(Pth_n,hn);
138 bn = tab_rho(face)/ loi_->Drho_DP(Pth_n,hn);
139 bnp1 = loi_->calculer_masse_volumique(Pth_n,hnp1) / loi_->Drho_DP(Pth_n,hnp1);
140 divu = divU(face);
141
142 S -= v *al *((hnp1-hn)/dt + u_gradh(face));
143 Fn += v * bn * divu;
144 Fnp1 += v * bnp1 * divu;
145 }
146
147
148 Pth = Pth_n + dt/V *(S-Fn);
149// Cerr<<"Volume="<<V<<finl;
150// Cerr<<"Fn="<<Fn<<finl;
151// Cerr<<"S="<<S<<finl;
152// Cerr<<"Pression thermo recalculee (expl) = "<<Pth<<finl;
153 Pth = Pth_n + dt/V *(S-.5*(Fn+Fnp1));
154// Cerr<<"Fnp1="<<Fnp1<<finl;
155// Cerr<<"Pression thermo recalculee (impl) = "<<Pth<<finl;
156 double tmp=0,r;
157 int k=0;
158 while (std::fabs(tmp-Pth)/Pth>1e-9 && k++<20)
159 {
160 tmp = Pth;
161 Fnp1 = 0;
162 for (face=0 ; face<nb_faces ; face++)
163 {
164 v = la_zone->volumes_entrelaces(face);
165 hnp1 = tab_hnp1(face);
166 r = loi_->calculer_masse_volumique(Pth,hnp1);
167 bnp1 = r / loi_->Drho_DP(Pth,hnp1);
168 Fnp1 += v * bnp1 * divU(face);
169 }
170 Pth = Pth_n + dt/V *(S-.5*(Fn+Fnp1));
171// Cerr<<"Fnp1="<<Fnp1<<finl;
172 Cerr<<"Pression thermo recalculee (impl"<<k<<") = "<<Pth<<finl;
173 }
174 return Pth;
175}
