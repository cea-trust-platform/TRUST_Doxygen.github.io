21// XD champ_front_bruite front_field_base champ_front_bruite 0 Field which is variable in time and space in a random manner.
22// XD attr nb_comp entier nb_comp 0 Number of field components.
23// XD attr bloc bloc_lecture bloc 0 { [N val L val ] Moyenne m_1.....[m_i ] Amplitude A_1.....[A_ i ]}: Random nois: If N and L are not defined, the ith component of the field varies randomly around an average value m_i with a maximum amplitude A_i. NL2 White noise: If N and L are defined, these two additional parameters correspond to L, the domain length and N, the number of nodes in the domain. Noise frequency will be between 2*Pi/L and 2*Pi*N/(4*L). NL2 For example, formula for velocity: u=U0(t) v=U1(t)Uj(t)=Mj+2*Aj*bruit_blanc where bruit_blanc (white_noise) is the formula given in the mettre_a_jour (update) method of the Champ_front_bruite (noise_boundary_field) (Refer to the Ch_fr_bruite.cpp file).

65//! Lecture a partir d'un flot d'entree au format:<br>nombre_de_composantes<br>moyenne moyenne(0) ... moyenne(nombre_de_composantes-1)<br>moyenne amplitude(0) ... amplitude(nombre_de_composantes-1)
66/** \return Entree\&<br>
67Signification: le flot d'entree modifie<br>
68Contraintes: */
69Entree& Champ_front_bruite::readOn(Entree& is)
70{
71 nx=0;
72 lx=0.;
73 int dim;
74 dim=lire_dimension(is,que_suis_je());
75 Motcle motlu;
76 Motcles les_mots(4);
77 les_mots[0]="moyenne";
78 les_mots[1]="amplitude";
79 les_mots[2]="L";
80 les_mots[3]="N";
81 is >> motlu;
82 if (motlu != "{")
83 {
84 Cerr << "Error while reading a Champ_front_bruite" << finl;
85 Cerr << "We expected a { instead of " << motlu << finl;
86 exit();
87 }
88 is >> motlu;
89 while (motlu!="}")
90 {
91 int rang=les_mots.search(motlu);
92 switch(rang)
93 {
94 case 0 :
95 {
96 moyenne.resize(dim);
97 fixer_nb_comp(dim);
98 for(int i=0; i<dim; i++)
99 is >> moyenne(i);
100 break;
101 }
102 case 1:
103 {
104 amplitude.resize(dim);
105 for(int i=0; i<dim; i++)
106 is >> amplitude(i);
107 break;
108 }
109 case 2:
110 {
111 is >> lx;
112 break;
113 }
114 case 3:
115 {
116 is >> nx;
117 break;
118 }
119 default :
120 {
121 Cerr << "Error while reading a Champ_front_bruite" << finl;
122 Cerr << motlu << "is not understand here "<< finl;
123 Cerr << "We were expecting a word from " << les_mots << finl;
124 exit();
125 }
126 }
127 is >> motlu;
128 }
129 return is;
130}

166//! Mise a jour du temps<br>et retirage aleatoire des valeurs du bruit.
167void Champ_front_bruite::mettre_a_jour(double temps)
168{
169 const Frontiere& front=la_frontiere_dis->frontiere();
170 int nb_faces=front.nb_faces();
171 DoubleTab& tab=valeurs_au_temps(temps);
172
173 if (nx!=0)
174 {
175//
176// N!=0 modelisation du bruit blanc (R. Howard)
177//
178 double onde, bruit_blanc, pi, invx, d_x, tot_vel;
179 pi=2.*acos(0.);
180 invx=1./nx;
181 d_x=lx/nx;
182 for(int i=0; i<nb_faces; i++)
183 for(int j=0; j<nb_comp(); j++)
184 {
185 tot_vel=moyenne(j)+amplitude(j);
186 onde=0.;
187 for (int in_x=0; in_x<nx; in_x++)
188 onde=onde+sin((2.*pi*temps/(4*d_x+in_x*d_x)+(10.2*i+5.3*j))*tot_vel+5.6*pi);
189
190 bruit_blanc=onde*invx;
191 tab(i,j)=moyenne(j)+2.*amplitude(j)*bruit_blanc;
192 }
193 }
194 else
195//
196// N=0 donc modelisation bruit aleatoire
197//
198 for(int i=0; i<nb_faces; i++)
199 for(int j=0; j<nb_comp(); j++)
200 {
201 tab(i,j)=moyenne(j)+amplitude(j)*(-1.+2.*drand48());
202 }
203 tab.echange_espace_virtuel();
204}
