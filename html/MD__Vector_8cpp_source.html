25//! construction d'un objet MD_Vector par copie d'un<br>objet existant. C'est la methode recommandee pour creer un objet MD_Vector<br>(autrement que par copie d'un autre MD_Vector)
26void MD_Vector::copy(const MD_Vector_base& src)

28 detach();
29 int num_obj = src.duplique();

35// Precondition: suppose que ptr_ est non nul ! (sinon appeler detach())
36//! methode non inline pour detacher l'objet pointe.
37void MD_Vector::detach_()
38{
39 assert(ptr_);
40 int& count = ptr_->ref_count_;
41 assert(count > 0);
42 count--;
43 if (count == 0)
44 {
45// Detruit l'objet pointe:
46 delete ptr_;
47 }
48 ptr_ = 0;
49}

51// Precondition: suppose ptr_ == 0 et src.non_nul() (sinon appeler attach())
52//! methode non inline pour attacher l'objet en parametre.
53void MD_Vector::attach_(const MD_Vector& src)
54{
55 assert(ptr_ == 0);
56 assert(src.non_nul());
57 ptr_ = src.ptr_;
58 (ptr_->ref_count_)++;
59}

61//! renvoie 1 si les structures sont identiques, 0 sinon
62int MD_Vector::operator==(const MD_Vector& md) const
63{
64// Pour l'instant, test tres restrictif: les deux structures sont
65// identiques si et seulement si les pointeurs sont identiques,
66// (autrement dit la deuxieme a ete creee par copie de la premiere)
67// Si on veut debrancher le systeme de reference multiple et dupliquer
68// les structures a la copie, il faut faire le test complet d'egalite
69// sur toute la structure !
