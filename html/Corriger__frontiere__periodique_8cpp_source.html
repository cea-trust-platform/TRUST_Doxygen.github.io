52//! Cet interprete permet de corriger les frontieres periodiques pour etre conformes aux<br>besoins de TRUST:<br>- reordonner les faces du bord periodique pour que la face i+n/2 soit en face de la face i,<br>et toutes les faces [0 .. n/2-1] du meme cote et [n/2 .. n-1] de l'autre cote<br>- deplacer les sommets des faces periodiques si besoin (si la CAO est fausse)<br>Syntaxe:<br>Corriger_frontiere_periodique {<br>domaine NOMDOMAINE<br>bord NOMBORDPERIO<br>[ direction DIMENSION dx dy [ dz ] ]<br>[ fichier_post BASENAME ]<br>}<br>XD corriger_frontiere_periodique interprete corriger_frontiere_periodique 1 The Corriger_frontiere_periodique keyword is mandatory to first define the periodic boundaries, to reorder the faces and eventually fix unaligned nodes of these boundaries. Faces on one side of the periodic domain are put first, then the faces on the opposite side, in the same order. It must be run in sequential before mesh splitting.<br>XD attr domaine chaine domaine 0 Name of domain.<br>XD attr bord chaine bord 0 the name of the boundary (which must contain two opposite sides of the domain)<br>XD attr direction list direction 1 defines the periodicity direction vector (a vector that points from one node on one side to the opposite node on the other side). This vector must be given if the automatic algorithm fails, that is:NL2 - when the node coordinates are not perfectly periodic NL2 - when the periodic direction is not aligned with the normal vector of the boundary faces<br>XD attr fichier_post chaine fichier_post 1 .
53Entree& Corriger_frontiere_periodique::interpreter_(Entree& is)
54{
55 if (nproc() > 1)
56 {
57 Cerr << "Error in Corriger_frontiere_periodique::interpreter():\n"
58 << " this function must be run in sequential before mesh splitting." << finl;
59 barrier();
60 exit();
61 }
62 Nom nom_bord;
63 ArrOfDouble direction_perio;
64 Nom nom_dom, nom_fichier_post;
65 Param param(que_suis_je());
66 param.ajouter("domaine", &nom_dom, Param::REQUIRED);
67 param.ajouter("bord", &nom_bord, Param::REQUIRED);
68 param.ajouter("direction", &direction_perio);
69 param.ajouter("fichier_post", &nom_fichier_post);
70 param.lire_avec_accolades_depuis(is);
71 associer_domaine(nom_dom);
72 Domaine& dom = domaine();
73 const int dim = dom.coord_sommets().dimension(1);
74 int direction_perio_set;
75 if (direction_perio.size_array() == 0)
76 {
77 direction_perio_set=0;
78 Cerr << "No direction given, searching periodicity direction automatically:" << finl;
79 Reordonner_faces_periodiques::chercher_direction_perio(direction_perio, dom, nom_bord);
80 }
81 else
82 {
83 direction_perio_set=1;
84 if (direction_perio.size_array() != dim)
85 {
86 Cerr << "Error in Corriger_frontiere_periodique::interpreter: direction should be of size "
87 << dim << finl;
88 exit();
89 }
90 }
91 Cerr << "Searching and moving periodicity nodes for domain " << nom_dom << " boundary "
92 << nom_bord << finl;
93 corriger_coordonnees_sommets_perio(dom, nom_bord, direction_perio, nom_fichier_post);
94 Bord& bord = dom.zone(0).bord(nom_bord);
95 IntTab& faces = bord.faces().les_sommets();
96 const double epsilon = Objet_U::precision_geom;
97 const int ok = Reordonner_faces_periodiques::reordonner_faces_periodiques(dom, faces, direction_perio, epsilon);
98 if (!ok)
99 {
100 if (direction_perio_set)
101 Cerr << "May be you give a wrong vector for the DIRECTION option" << finl;
102 else
103 Cerr << "May be you could use the DIRECTION option to specify the vector of periodicity" << finl;
104 Cerr << "for the keyword:" << finl;
105 Cerr << "Corriger_frontiere_periodique { Domaine " << nom_dom << " Bord " << nom_bord << " } " << finl;
106 exit();
107 }
108 return is;
109}

127//! Pour chaque sommet du bord periodique, on cherche son sommet oppose dans la direction<br>+ ou - vecteur_perio et dans un rayon search_radius.<br>Parmi le couple de sommets forme, on deplace celui qui se trouve en +vecteur_perio<br>pour le mettre en face de l'autre sommet.<br>Exit en cas d'erreur (si les sommets sont trop eloignes de leur sommet associe)<br>Parametre: dom<br>Signification: le domaine a modifier. On deplace les coordonnees des sommets periodiques.<br>Parametre: nom_bord<br>Signification: le nom du bord periodique du domaine a traiter<br>Parametre: vecteur_perio<br>Signification: un vecteur de taille "dimension" qui contient le delta entre deux sommets<br>periodiques associes (voir methode chercher_direction_perio())<br>Parametre: nom_fichier_post<br>Signification: si different de "??", on cree un fichier de postraitement contenant les<br>faces du bord initial et un vecteur qui indique le deplacement applique aux sommets.
128void Corriger_frontiere_periodique::corriger_coordonnees_sommets_perio(Domaine& dom,
129 const Nom& nom_bord,
130 const ArrOfDouble& vecteur_perio,
131 const Nom& nom_fichier_post)
132{
133 if (nproc() > 1)
134 {
135 Cerr << "Error in Reordonner_faces_periodiques::corriger_coordonnees_sommets_perio\n"
136 << " this algorithm is sequential (use it before Decouper)" << finl;
137 barrier();
138 exit();
139 }
140
141 Domaine_bord domaine_bord;
142 domaine_bord.construire_domaine_bord(dom, nom_bord);
143 const ArrOfInt& renum_som = domaine_bord.get_renum_som();
144 const DoubleTab& som_bord = domaine_bord.les_sommets();
145 Octree_Double octree;
146 octree.build_nodes(som_bord, 0 /* do not include virtual nodes */);
147
148 const double epsilon_initial = dom.epsilon();
149 if (epsilon_initial == 0.)
150 {
151 Cerr << "Error in Corriger_frontiere_periodique::corriger_coordonnees_sommets_perio\n"
152 << " dom.epsilon = 0." << finl;
153 exit();
154 }
155
156 int error_flag = 0;
157
158 const int nb_som_bord = som_bord.dimension(0);
159 const int dim = som_bord.dimension(1);
160// Un tableau contenant le deplacement applique aux sommets (pour postraitement)
161 DoubleTab delta(nb_som_bord, dim);
162// Un tableau pour detecter les erreurs de periodicite: sommets associes plusieurs fois
163 ArrOfBit marker(nb_som_bord);
164 marker = 0;
165// Tableau temporaire pour l'octree:
166 ArrOfInt nodes_list;
167 nodes_list.set_smart_resize(1);
168 DoubleTab& sommets_src = dom.les_sommets();
169 ArrOfDouble coord(dim);
170 for (int som = 0; som < nb_som_bord; som++)
171 {
172 if (marker.testsetbit(som))
173 continue; //!< Sommet deja traite
174
175// Cherche le sommet oppose dans les deux directions (-1. et +1.)
176 double facteur;
177 int som2 = -1;
178// Recherche du sommet dans un rayon de plus en plus grand en commencant par epsilon:
179 double epsilon = epsilon_initial;
180 do
181 {
182 for (facteur = -1.; facteur < 1.5; facteur += 2.)
183 {
184 for (int i = 0; i < dim; i++)
185 coord[i] = som_bord(som, i) + facteur * vecteur_perio[i];
186 octree.search_elements_box(coord, epsilon, nodes_list);
187 som2 = octree.search_nodes_close_to(coord, som_bord, nodes_list, epsilon);
188 if (som2 >= 0)
189 break;
190 }
191 if (som2 >= 0)
192 break;
193// Sommet non trouve, on recommence avec un rayon de recherche plus grand
194 epsilon *= 4.;
195 }
196 while (1);
197 if (marker.testsetbit(som2))
198 {
199 Cerr << "Error in corriger_coordonnees_sommets_perio\n"
200 << " Coordinate " << coord
201 << " Closest point [ " << som_bord(som2, 0) << " " << som_bord(som2, 0) << " "
202 << ((dim==3)?som_bord(som2, 0):0.) << " ] already used for another point" << finl;
203 error_flag = 1;
204 }
205 else
206 {
207// On deplace le sommet vers lequel pointe vecteur_perio:
208 const int som_ref = som;
209 const int som_deplace = som2;
210// Indice du sommet a deplacer dans le domaine source:
211 const int s = renum_som[som_deplace];
212// Deplacement
213 for (int i = 0; i < dim; i++)
214 {
215 double old_x = som_bord(som_deplace, i);
216 double new_x = som_bord(som_ref, i) + facteur * vecteur_perio[i];
217 sommets_src(s, i) = new_x;
218 delta(som_deplace, i) = new_x - old_x;
219 }
220 }
221 }
222
223 if (nom_fichier_post != "??")
224 {
225 Domaine dom2;
226 Cerr << "Writing node displacement into file: " << nom_fichier_post << finl;
227 DERIV(Format_Post_base) fichier_post;
228 fichier_post.typer("FORMAT_POST_LATA_V1");
229 fichier_post.valeur().initialize(nom_fichier_post, 1 /* binaire */, "SIMPLE");
230 Format_Post_base& post = fichier_post.valeur();
231 post.ecrire_entete(0., 0 /*reprise*/, 1 /* premier post */);
232 post.ecrire_domaine(domaine_bord, 1 /* premier_post */);
233 post.ecrire_temps(0.);
234 Noms unites;
235 unites.add("m");
236 unites.add("m");
237 if (dim==3) unites.add("m");
238 Noms compos;
239 compos.add("dx");
240 compos.add("dy");
241 if (dim==3) compos.add("dz");
242 post.ecrire_champ(domaine_bord,
243 unites, compos, -1 /* ecrire toutes les composantes */,
244 0., 0., /* temps */
245 "vitesse", domaine_bord.le_nom(), "SOM","vector", delta);
246 int fin=1;
247 post.finir(fin);
248 }
249 Cerr << "Max of node displacement (m): " << max_abs_array(delta) << finl;
250 if (error_flag)
251 {
252 Cerr << "Could not correct node coordinates. Aborting." << finl;
253 exit();
254 }
255}
