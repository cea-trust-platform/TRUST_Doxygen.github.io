39void Source_WC_Chaleur_VEF::compute_interpolate_gradP(DoubleTab& UgradP_face, const DoubleTab& Ptot) const
40{
41 /*
42 * NOTA BENE :
43 * On a discretisé P_EOS avec une directive = temperature qui donne en VEF :
44 * Ptot => aux faces
45 * la_vitesse => faces car vef
46 * grad_Ptot => elem
47 *
48 * L'equation est une equation de scalaire => aux faces car vef
49 * On a d P_tot / d t = del P / del t + u.grad(P_tot)
50 *
51 * del P / del t => aux faces
52 * u.grad(P_tot) !!! faut interpoler grad(P_tot) aux faces
53 */
54
55 const Navier_Stokes_WC& eqHyd = ref_cast(Navier_Stokes_WC,mon_equation->probleme().equation(0));
56 const DoubleTab& la_vitesse = eqHyd.vitesse().valeurs();
57
58// On sait que grad_Ptot est un champ elem => on pense à la conductivite (elem en vef !! )
59 DoubleTab grad_Ptot;
60 const DoubleTab& lambda = eqHyd.milieu().conductivite()->valeurs();
61 const int nb = lambda.size_totale(), nbcomp = la_vitesse.line_size();
62 assert (nb == lambda.dimension_tot(0) && lambda.line_size() == 1);
63 grad_Ptot.resize(nb,nbcomp);
64
65 const Convection_Diffusion_Chaleur_WC& eq_chal = ref_cast(Convection_Diffusion_Chaleur_WC,mon_equation.valeur());
66 const Operateur_Grad& Op_Grad = eq_chal.operateur_gradient_WC();
67 Op_Grad.calculer(Ptot,grad_Ptot); //!< compute grad(P_tot)
68
69 const Zone_dis_base& zone_dis = mon_equation->inconnue().zone_dis_base();
70 const Zone_VF& zone = ref_cast(Zone_VF, zone_dis);
71 assert (zone_dis.que_suis_je() == "Zone_VEFPreP1b" || zone_dis.que_suis_je() == "Zone_VEF");
72
73// grad should be zero at boundary
74 correct_grad_boundary(zone,grad_Ptot);
75
76// We compute u*grad(P_tot) on each face
77 DoubleTab grad_Ptot_face(la_vitesse); //!< champs sur les faces
78
79// get grad_Ptot on faces
80 elem_to_face(zone,grad_Ptot,grad_Ptot_face);
81
82 const int n = la_vitesse.dimension_tot(0);
83 assert (UgradP_face.dimension_tot(0) == n && grad_Ptot_face.dimension_tot(0) == n);
84 assert (UgradP_face.line_size() == 1 && n == zone.nb_faces_tot());
85 for (int i=0 ; i <n ; i++)
86 {
87 UgradP_face(i,0) = 0.;
88 for (int j=0 ; j <nbcomp ; j++) UgradP_face(i,0) += la_vitesse(i,j) * grad_Ptot_face(i,j);
89 }
90}

93void Source_WC_Chaleur_VEF::elem_to_face(const Zone_VF& zone, const DoubleTab& grad_Ptot,DoubleTab& grad_Ptot_face) const
94{
95 const DoubleVect& vol = zone.volumes();
96 const IntTab& elem_faces = zone.elem_faces();
97 const int nb_face_elem = elem_faces.line_size(), nb_elem_tot = zone.nb_elem_tot(), nb_comp = grad_Ptot_face.line_size();
98 assert (grad_Ptot.dimension_tot(0) == nb_elem_tot && grad_Ptot_face.dimension_tot(0) == zone.nb_faces_tot());
99 assert (grad_Ptot.line_size() == nb_comp);
100
101 grad_Ptot_face = 0.;
102 for (int ele = 0; ele < nb_elem_tot; ele++)
103 for (int s = 0; s < nb_face_elem; s++)
104 {
105 const int face = elem_faces(ele,s);
106 for (int comp = 0; comp < nb_comp; comp++) grad_Ptot_face(face,comp) += grad_Ptot(ele,comp)*vol(ele);
107 }
108
109 for (int f=0; f<zone.nb_faces_tot(); f++)
110 for (int comp=0; comp<nb_comp; comp++) grad_Ptot_face(f,comp) /= volumes(f)*nb_face_elem;
111}
