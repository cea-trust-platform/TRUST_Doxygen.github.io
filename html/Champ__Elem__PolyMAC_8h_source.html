80inline DoubleTab& Champ_Elem_PolyMAC::trace(const Frontiere_dis_base& fr, DoubleTab& x, double t, int distant) const
81{
82 /* dimensionnement du tableau de destination x si necessaire */
83 const DoubleTab& src = valeurs();
84 const Front_VF& fvf = ref_cast(Front_VF, fr);
85 const Zone_PolyMAC& zone = ref_cast(Zone_PolyMAC, zone_dis_base());
86 const IntTab& f_e = zone.face_voisins();
87
88 DoubleTrav dst; //reconstruction du champ aux faces (on ne le remplit que sur le bord concerne)
89 int i, n, e, f, N = src.nb_dim() > 1 ? src.dimension(1): 1, has_faces = src.dimension_tot(0) > zone.nb_elem_tot();
90 if (!x.dimension(0) && !x.get_md_vector().non_nul()) x.resize(fvf.nb_faces(), N);
91 dst.resize(zone.nb_faces(), N);
92 for (i = 0; i < fvf.nb_faces(); i++)
93 for (n = 0, f = fvf.num_face(i), e = f_e(f, 0); n < N; n++)
94 dst(f, n) = src(has_faces ? zone.nb_elem_tot() + f : e, n);
95
96 if (distant) fr.frontiere().trace_face_distant(dst, x);
97 else fr.frontiere().trace_face_local(dst, x);
98
99 return x;
100}
