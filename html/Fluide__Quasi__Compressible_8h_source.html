67inline void Fluide_Quasi_Compressible::Resoudre_EDO_PT()
68{

72 {

85// Calcule W=-dZ/dt, 2nd membre de l'equation div(rhoU) = W
86inline void Fluide_Quasi_Compressible::secmembre_divU_Z(DoubleTab& tab_W) const
87{
88 double temps=vitesse_->temps();
89 if (temps>temps_debut_prise_en_compte_drho_dt_)
90 {
91 eos_tools_->secmembre_divU_Z(tab_W);
92 if (tab_W_old_.size_totale() > 0) //!< Relaxation eventuelle:
93 {
94 tab_W *= omega_drho_dt_;
95 tab_W.ajoute_sans_ech_esp_virt(1-omega_drho_dt_, tab_W_old_);
96 }
97 tab_W_old_ = tab_W;
98 }
99 else
100 {
101 if (je_suis_maitre())
102 Cerr<<" The drho_dt term is not taken into account for time smaller than " <<temps_debut_prise_en_compte_drho_dt_<<finl;
103 tab_W=0;
104 }
105}
