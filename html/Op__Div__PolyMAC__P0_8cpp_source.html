63void Op_Div_PolyMAC_P0::dimensionner(Matrice_Morse& matrice) const
64{
65 const Zone_PolyMAC_P0& zone = la_zone_PolyMAC_P0.valeur();
66 const IntTab& f_e = zone.face_voisins();
67 int i, e, f, n, ne_tot = zone.nb_elem_tot(), nf_tot = zone.nb_faces_tot(), N = equation().inconnue().valeurs().line_size(), D = dimension;
68
69 IntTab stencil(0,2);
70 stencil.set_smart_resize(1);
71 for (f = 0; f < zone.nb_faces(); f++)
72 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
73 for (n = 0; n < N; n++) stencil.append_line(N * e + n, N * f + n);
74
75 tableau_trier_retirer_doublons(stencil);
76 Matrix_tools::allocate_morse_matrix(N * ne_tot, N * (nf_tot + D * ne_tot), stencil, matrice);
77}

79DoubleTab& Op_Div_PolyMAC_P0::ajouter(const DoubleTab& vit, DoubleTab& div) const
80{
81 const Zone_PolyMAC_P0& zone = la_zone_PolyMAC_P0.valeur();
82 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face();
83 const IntTab& f_e = zone.face_voisins();
84 int i, e, f, n, N = vit.line_size();
85 assert(div.line_size() == N);
86
87 DoubleTab& tab_flux_bords = flux_bords_;
88 tab_flux_bords.resize(zone.nb_faces_bord(),1);
89 tab_flux_bords=0;
90
91 for (f = 0; f < zone.nb_faces(); f++)
92 {
93 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
94 for (n = 0; n < N; n++)
95 div(e, n) += (i ? -1 : 1) * fs(f) * pf(f) * vit(f, n);
96 if (f < zone.premiere_face_int())
97 for (n = 0; n < N; n++) tab_flux_bords(f, n) = fs(f) * pf(f) * vit(f, n);
98 }
99
100 div.echange_espace_virtuel();
101
102 return div;
103}
104void Op_Div_PolyMAC_P0::contribuer_a_avec(const DoubleTab&,Matrice_Morse& mat) const
105{
106 const Zone_PolyMAC_P0& zone = la_zone_PolyMAC_P0.valeur();
107 const Champ_Face_PolyMAC_P0& ch = ref_cast(Champ_Face_PolyMAC_P0, equation().inconnue().valeur());
108 const DoubleVect& fs = zone.face_surfaces(), &pf = zone.porosite_face();
109 const IntTab& f_e = zone.face_voisins();
110 int i, e, f, n, N = ch.valeurs().line_size();
111
112 for (f = 0; f < zone.nb_faces(); f++)
113 for (i = 0; i < 2 && (e = f_e(f, i)) >= 0; i++)
114 for (n = 0; n < N; n++) mat(N * e + n, N * f + n) += (i ? 1 : -1) * fs(f) * pf(f);
115}

124int Op_Div_PolyMAC_P0::impr(Sortie& os) const
125{
126
127 const int impr_bord=(la_zone_PolyMAC_P0->zone().Bords_a_imprimer().est_vide() ? 0:1);
128 SFichier Flux_div;
129 ouvrir_fichier(Flux_div,"",je_suis_maitre());
130 EcrFicPartage Flux_face;
131 ouvrir_fichier_partage(Flux_face,"",impr_bord);
132 const Schema_Temps_base& sch = equation().probleme().schema_temps();
133 double temps = sch.temps_courant();
134 if (je_suis_maitre()) Flux_div.add_col(temps);
135
136 int nb_compo=flux_bords_.dimension(1);
137// On parcours les frontieres pour sommer les flux par frontiere dans le tableau flux_bord
138 DoubleVect flux_bord(nb_compo);
139 DoubleVect bilan(nb_compo);
140 bilan = 0;
141 for (int num_cl=0; num_cl<la_zone_PolyMAC_P0->nb_front_Cl(); num_cl++)
142 {
143 flux_bord=0;
144 const Cond_lim& la_cl = la_zcl_PolyMAC_P0->les_conditions_limites(num_cl);
145 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
146 int ndeb = frontiere_dis.num_premiere_face();
147 int nfin = ndeb + frontiere_dis.nb_faces();
148 for (int face=ndeb; face<nfin; face++)
149 for(int k=0; k<nb_compo; k++)
150 flux_bord(k)+=flux_bords_(face, k);
151 for(int k=0; k<nb_compo; k++)
152 flux_bord(k)=Process::mp_sum(flux_bord(k));
153
154 if(je_suis_maitre())
155 {
156 for(int k=0; k<nb_compo; k++)
157 {
158// Ajout pour impression sur fichiers separes
159 Flux_div.add_col(flux_bord(k));
160 bilan(k)+=flux_bord(k);
161 }
162 }
163 }
164
165 if(je_suis_maitre())
166 {
167 for(int k=0; k<nb_compo; k++)
168 {
169 Flux_div.add_col(bilan(k));
170 }
171 Flux_div << finl;
172 }
173
174 for (int num_cl=0; num_cl<la_zone_PolyMAC_P0->nb_front_Cl(); num_cl++)
175 {
176 const Frontiere_dis_base& la_fr = la_zcl_PolyMAC_P0->les_conditions_limites(num_cl).frontiere_dis();
177 const Cond_lim& la_cl = la_zcl_PolyMAC_P0->les_conditions_limites(num_cl);
178 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
179 int ndeb = frontiere_dis.num_premiere_face();
180 int nfin = ndeb + frontiere_dis.nb_faces();
181 if (la_zone_PolyMAC_P0->zone().Bords_a_imprimer().contient(la_fr.le_nom()))
182 {
183 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps " << temps << " : " << finl;
184 for (int face=ndeb; face<nfin; face++)
185 {
186 if (dimension==2)
187 Flux_face << "# Face a x= " << la_zone_PolyMAC_P0->xv(face,0) << " y= " << la_zone_PolyMAC_P0->xv(face,1) << " flux=" ;
188 else if (dimension==3)
189 Flux_face << "# Face a x= " << la_zone_PolyMAC_P0->xv(face,0) << " y= " << la_zone_PolyMAC_P0->xv(face,1) << " z= " << la_zone_PolyMAC_P0->xv(face,2) << " flux=" ;
190 for(int k=0; k<nb_compo; k++)
191 Flux_face << " " << flux_bords_(face, k);
192 Flux_face << finl;
193 }
194 Flux_face.syncfile();
195 }
196 }
197
198 return 1;
199}
