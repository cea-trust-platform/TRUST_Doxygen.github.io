40static inline char char_uppercase(char c)
41{
42 if (c >= 'a' && c <= 'z')
43 c += 'A' - 'a';
44 return c;
45}
46
47static inline int strcmp_uppercase(const char *n1, const char *n2)
48{
49 int i = 0;
50 unsigned char c1, c2;
51 int delta;
52 do
53 {
54 c1 = (unsigned char) char_uppercase(n1[i]);
55 c2 = (unsigned char) char_uppercase(n2[i]);
56 delta = c1 - c2;
57 i++;
58 }
59 while ((delta == 0) && (c1 != 0) && (c2 != 0));
60 return delta;
61}

80//! Recherche le synonyme de nom "nom" dans la liste de synonymes enregistres<br>par une recherche binaire.<br>On compare les chaines converties en majuscules.<br>On range dans "index" l'indice du synonyme, s'il a ete trouve<br>et sinon l'indice du synonyme juste apres (dans ce cas, on a<br>les_synonymes[index-1]->n < nom < les_synonymes[index]->n )<br>Si le synonyme a ete trouve on renvoie 1, sinon 0.
81int Synonyme_info::search_synonyme_info_name(const char *nom, int& index)
82{
83 assert(nom != 0);
84// [imin..imax] est l'intervalle ou se trouve l'index recherche
85 int imin = 0;
86 int imax = nb_classes;
87 while (imax > imin)
88 {
89// On a toujours milieu < imax
90 int milieu = (imin + imax) / 2;
91 int comparaison = strcmp_uppercase(nom, les_synonymes[milieu]->n);
92 if (comparaison == 0)
93 {
94 index = milieu;
95 return 1;
96 }
97 if (comparaison < 0)
98 {
99// nom < les_synonymes[milieu]
100// l'index recherche est donc inferieur ou egal a "milieu"
101 imax = milieu;
102 }
103 else
104 {
105// nom > les_synonymes[milieu]
106// l'index recherche est donc strictement superieur a "milieu"
107 imin = milieu + 1;
108 }
109 }
110 index = imax;
111 return 0;
112}

154//! Methode statique appelee par les constructeurs de Synonyme_info pour<br>ajouter un nouveau synonyme a la liste des synonymes enregistres.<br>Verifie que le nom du synonyme n'existe pas encore.
155void Synonyme_info::retirer_synonyme(const char* nom)
156{
157// On cherche ou mettre le synonyme dans le tableau :
158 int index;
159 int existe_deja = search_synonyme_info_name(nom, index);
160 if (!existe_deja)
161 {
162 Cerr<<"A synonym is suppressed whereas it doesn't exist !!!!!"<<finl;
163 Process::exit();
164 }
165 else
166 {
167// Ajout du synonyme dans le tableau a l'indice "index":
168 for (int i = index; i < nb_classes-1; i++)
169 les_synonymes[i] = les_synonymes[i+1];
170 nb_classes--;
171 }
172}

177//! Methode statique appelee par les constructeurs de Synonyme_info pour<br>ajouter un nouveau synonyme a la liste des synonymes enregistres.<br>Verifie que le nom du synonyme n'existe pas encore.
178void Synonyme_info::ajouter_synonyme(const Synonyme_info& synonyme_info)
179{
180// Verifie qu'il y a assez de place dans le tableau :
181 if (les_synonymes_memsize <= nb_classes + 1)
182 {
183 static const int INCREMENT = 512;
184// Plus assez de place dans le tableau: on redimensionne.
185 les_synonymes_memsize += INCREMENT;
186 const Synonyme_info** nouveau = new const Synonyme_info*[les_synonymes_memsize];
187 for (int i = 0; i < nb_classes; i++)
188 nouveau[i] = les_synonymes[i];
189 delete[] les_synonymes;
190 les_synonymes = nouveau;
191 }
192
193// On cherche ou mettre le synonyme dans le tableau :
194 int index;
195 int existe_deja=Type_info::est_un_type(synonyme_info.n);
196 if (existe_deja)
197 {
198 Cerr<<" The synonym "<<synonyme_info.n<<" exists as a class which is forbidden !!!!"<<finl;
199 Process::exit();
200 }
201 existe_deja = search_synonyme_info_name(synonyme_info.n, index);
202 if (existe_deja)
203 {
204 Cerr<<" Synonym "<<synonyme_info.n<<" already exists, which is forbidden!!!!"<<finl;
205 Process::exit();
206 }
207 else
208 {
209// Ajout du synonyme dans le tableau a l'indice "index":
210 for (int i = nb_classes; i > index; i--)
211 les_synonymes[i] = les_synonymes[i-1];
212 les_synonymes[index] = &synonyme_info;
213
214 nb_classes++;
215 }
216}
