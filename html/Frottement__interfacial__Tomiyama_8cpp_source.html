28Entree& Frottement_interfacial_Tomiyama::readOn(Entree& is)
29{
30 Param param(que_suis_je());
31 param.ajouter("constante_gravitation", &g_);
32 param.ajouter("contamination", &contamination_);
33 param.lire_avec_accolades_depuis(is);
34 if (! ((contamination_==0)|(contamination_==1)|(contamination_==2) )) Process::exit("Frottement_interfacial_Tomiyama : only 3 contamination levels exist : 0,1,2 !");
35
36 const Pb_Multiphase *pbm = sub_type(Pb_Multiphase, pb_.valeur()) ? &ref_cast(Pb_Multiphase, pb_.valeur()) : NULL;
37
38 if (!pbm || pbm->nb_phases() == 1) Process::exit(que_suis_je() + " : not needed for single-phase flow!");
39 for (int n = 0; n < pbm->nb_phases(); n++) //recherche de n_l, n_g : phase {liquide,gaz}_continu en priorite
40 if (pbm->nom_phase(n).debute_par("liquide") && (n_l < 0 || pbm->nom_phase(n).finit_par("continu"))) n_l = n;
41 if (n_l < 0) Process::exit(que_suis_je() + " : liquid phase not found!");
42
43 for (int k = 0; k < pbm->nb_phases(); k++)
44 if (k != n_l)
45 if (!(ref_cast(Milieu_composite, pbm->milieu()).has_interface(n_l, k))) Process::exit("Frottement_interfacial_Tomiyama : one must define an interface and have a surface tension !");
46
47 return is;
48}

56void Frottement_interfacial_Tomiyama::coefficient(const DoubleTab& alpha, const DoubleTab& p, const DoubleTab& T,
57 const DoubleTab& rho, const DoubleTab& mu, const DoubleTab& sigma, double Dh,
58 const DoubleTab& ndv, const DoubleTab& d_bulles, DoubleTab& coeff) const
59{
60 int N = ndv.dimension(0);
61
62 coeff = 0;
63
64 for (int k = 0; k < N; k++)
65 if (k!=n_l)
66 {
67
68 double Re = rho(n_l) * ndv(n_l,k) * d_bulles(k)/mu(n_l) + 1.e-6;
69 double Eo = g_ * std::abs(rho(n_l)-rho(k)) * d_bulles(k)*d_bulles(k)/sigma(n_l,k);
70 double Cd = -1;
71 if (contamination_==0) Cd = std::max( std::min( 16./Re*(1+0.15*std::pow(Re, 0.687)) , 48./Re ) , 8.*Eo/(3.*(Eo+4.)));
72 if (contamination_==1) Cd = std::max( std::min( 24./Re*(1+0.15*std::pow(Re, 0.687)) , 72./Re ) , 8.*Eo/(3.*(Eo+4.)));
73 if (contamination_==2) Cd = std::max( 24./Re*(1+0.15*std::pow(Re, 0.687)) , 8.*Eo/(3.*(Eo+4.)));
74
75 coeff(k, n_l, 1) = 3./4.*Cd/d_bulles(k) * alpha(k) * rho(n_l);
76 coeff(k, n_l, 0) = coeff(k, n_l, 1) * ndv(n_l,k);
77 coeff(n_l, k, 0) = coeff(k, n_l, 0);
78 coeff(n_l, k, 1) = coeff(k, n_l, 1);
79 }
80}
