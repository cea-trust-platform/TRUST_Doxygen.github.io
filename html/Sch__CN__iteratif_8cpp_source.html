47void Sch_CN_iteratif::ajuster_facsec(type_convergence cv)
48{
49
50 if (facsec_!=last_facsec) //!< On n'ajuste qu'une fois pour un initTimeStep.
51 return;
52
53 switch (cv)
54 {
55 case DIVERGENCE:
56// le critere de divergence a ete atteint
57 facsec_=last_facsec*0.8;
58 break;
59 case NON_CONVERGENCE:
60// ni le critere de divergence ni le critere de convergence
61// n'ont ete atteints en niter_max iterations
62 facsec_=last_facsec*0.9;
63 break;
64 case CONVERGENCE_LENTE:
65// le critere de convergence a ete atteint en plus que niter_avg
66// iterations
67 facsec_=last_facsec*0.99;
68 break;
69 case CONVERGENCE_RAPIDE:
70// le critere de convergence a ete atteint en moins que niter_avg
71// iterations
72 facsec_=last_facsec*1.01;
73 break;
74 case CONVERGENCE_OK:
75// le critere de convergence a ete atteint en niter_avg iterations
76 break;
77 default:
78 break;
79 }
80 facsec_=std::min(facsec_,facsec_max);
81}

158bool Sch_CN_iteratif::iterateTimeStep(bool& converged)
159{
160
161 Probleme_base& pb = pb_base();
162 const int nb_eqn=pb.nombre_d_equations();
163
164 SFichier fic;
165 fic.ouvrir("dt_CN",ios::app);
166
167 if (je_suis_maitre())
168 {
169 fic << "Sch_CN_iteratif : pb= " << pb.le_nom()
170 << " present= " << temps_courant()
171 << " dt= " << dt_
172 << " iteration= " << iteration
173 << " last_facsec= " << last_facsec;
174 }
175 converged=true;
176 bool diverged=false;
177
178 for(int i=0; i<nb_eqn && !diverged; i++)
179 {
180 bool cv;
181 diverged = diverged || !iterateTimeStepOnEquation(i,cv);
182 converged = converged && cv;
183 }
184
185 iteration ++;
186 if (iteration<niter_min)
187 converged=false; //!< Continuer a iterer de toutes facons
188
189// Un peu de bon sens...
190 assert(!(converged&&diverged));
191
192// Si convergence
193 if (converged)
194 {
195 if (iteration<niter_avg) //!< Convergence trop rapide, augmenter facsec
196 ajuster_facsec(CONVERGENCE_RAPIDE);
197 else if (iteration>niter_avg) //!< Convergence trop lente, reduire facsec
198 ajuster_facsec(CONVERGENCE_LENTE);
199 else
200 ajuster_facsec(CONVERGENCE_OK);
201 if (je_suis_maitre())
202 {
203 fic << " facsec= " << facsec_
204 << " result= CONVERGENCE" << finl;
205 }
206 fic.close();
207 return true;
208 }
209
210// Si divergence
211 else if (diverged)
212 {
213 ajuster_facsec(DIVERGENCE);
214 if (je_suis_maitre())
215 {
216 fic << " facsec= " << facsec_
217 << " result= DIVERGENCE" << finl;
218 }
219 return false;
220 }
221
222// Si pas converge assez vite
223 else if (iteration==niter_max-1)
224 {
225 ajuster_facsec(NON_CONVERGENCE);
226 if (je_suis_maitre())
227 {
228 fic << " facsec= " << facsec_
229 << " result= NON_CONVERGENCE" << finl;
230 }
231 return false;
232 }
233
234// Sinon, en cours de convergence
235 if (je_suis_maitre())
236 {
237 fic << " facsec= " << facsec_
238 << " result= CONTINUE" << finl;
239 }
240 return true;
241}

250//! Calcule une iteration de la resolution sur l'equation i.<br>Calcule u(n+1/2,p+1)=u(n)+f(u(n+1/2,p))*dt/2<br>et u(n+1,p+1)=u(n)+f(u(n+1/2,p))*dt<br>ou f donne du/dt en fonction de u<br>Retourne true dans converged si ca ne bouge plus d'une iteration a l'autre, false sinon<br>Renvoie true si OK pour continuer a iterer, false sinon (diverge ou trop d'iterations)
251bool Sch_CN_iteratif::iterateTimeStepOnEquation(int i,bool& converged)
252{
253
254 Probleme_base& pb = pb_base();
255 Equation_base& eqn = pb.equation(i);
256 if (eqn.equation_non_resolue())
257 {
258 Cout<< "====================================================" << finl;
259 Cout<< eqn.que_suis_je()<<" equation is not solved."<<finl;
260 Cout<< "====================================================" << finl;
261// On calcule une fois la derivee pour avoir les flux bord
262 if (eqn.schema_temps().nb_pas_dt()==0)
263 {
264 DoubleTab inconnue_valeurs(eqn.inconnue().valeurs());
265 eqn.derivee_en_temps_inco(inconnue_valeurs);
266 }
267 converged=true;
268 return true;
269 }
270 double temps_intermediaire=temps_futur(1);
271 double temps_final=temps_futur(2);
272 double dt_intermediaire=temps_intermediaire-temps_courant();
273 double dt_final=temps_final-temps_courant();
274
275 DoubleTab& present = eqn.inconnue().valeurs();
276 DoubleTab& intermediaire = eqn.inconnue()->valeurs(temps_intermediaire);
277 DoubleTab& final = eqn.inconnue()->valeurs(temps_final);
278
279 DoubleTab dudt(present);
280
281 DoubleTab delta(intermediaire);
282 delta*=-1;
283
284// On impose les CLs Dirichlet au temps intermediaire.
285// En effet, les operateurs de diffusion n'utilisent que
286// l'inconnue et ne vont pas lire les CLs.
287// Cela permet en particulier d'avoir l'egalite des flux en pb
288// couple thermique VEF avec Chap_front_contact_VEF, meme avant convergence.
289// WEC : /!\ la vitesse au temps intermediaire
290// n'est pas forcement a divergence nulle.
291 eqn.zone_Cl_dis()->imposer_cond_lim(eqn.inconnue(),temps_intermediaire);
292
293// Calcul de la derivee dudt pour la valeur intermediaire de l'inconnue.
294// Bidouille : Comme les operateurs prennent par defaut le present,
295// on avance temporairement l'inconnue.
296
297 eqn.inconnue().avancer();
298 eqn.derivee_en_temps_inco(dudt);
299 eqn.inconnue().reculer();
300
301// Mise a jour des valeurs de l'inconnue aux temps intermediaire et final
302// intermediaire = present + dt_intermediaire * dudt;
303// final = present + dt_final * dudt;
304 intermediaire = dudt;
305 intermediaire*= dt_intermediaire;
306 intermediaire+= present;
307 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(),temps_intermediaire);
308 intermediaire.echange_espace_virtuel();
309 final = dudt;
310 final*= dt_final;
311 final+= present;
312 eqn.zone_Cl_dis().imposer_cond_lim(eqn.inconnue(),temps_final);
313 final.echange_espace_virtuel();
314
315 delta+=intermediaire; //!< delta = Contient u(n+1/2,p+1) - u(n+1/2,p)
316
317// Si l'equation a diverge
318 if (divergence(present,intermediaire,delta,iteration))
319 {
320 converged=false;
321 return false;
322 }
323
324// Si l'equation a converge
325 if (convergence(present,intermediaire,delta,iteration))
326 {
327 converged=true;
328 delta=final;
329 delta-=present;
330 delta/=dt_final;
331 update_critere_statio(delta, eqn);
332 }
333 else
334 {
335 set_stationnaire_atteint()=0;
336 converged=false;
337 }
338 return true;
339}

380//! Indique si le calcul iteratif a converge.<br>Critere de convergence utilise :<br>|| u(n+1,p+1) - u(n+1,p) || < seuil * || u(n+1/2,p+1) ||<br>C'est equivalent a<br>|| u(n+1,p) - u(n+1) || < seuil * || (Id-(dt/2).(df/du))^-1 || * || u(n+1/2,p+1) ||<br>ou u(n+1) est la solution exacte en n+1,<br>df/du est le lagrangien de f(u), pris en u(n+1/2)<br>et les normes sont compatibles entre matrices et vecteurs (ici norme infinie).
381/** \return bool<br>
382Signification: true=converge, false=non converge<br>
383Contraintes: */
384bool Sch_CN_iteratif::convergence(const DoubleTab& u0, const DoubleTab& up1, const DoubleTab& delta, int p) const
385{
386 double a = mp_max_abs_vect(delta);
387 double b = mp_max_abs_vect(up1);
388 int resu = (2. * a) < (seuil * b);
389 envoyer_broadcast(resu, 0); //!< pour etre certain que tout le monde fait la meme chose
390 return resu;
391}
