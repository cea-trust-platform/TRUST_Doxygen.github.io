100double Op_Diff_VEF_base::calculer_dt_stab() const
101{
102 remplir_nu(nu_);
103// La diffusivite est constante dans le domaine donc
104//
105// dt_diff = h*h/diffusivite
106
107 double dt_stab = DMAXFLOAT;
108 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
109 if (! has_champ_masse_volumique())
110 {
111
112// Methode "standard" de calcul du pas de temps
113// Ce calcul est tres conservatif: si le max de la diffusivite
114// n'est pas atteint a l'endroit ou le min de delta_h_carre est atteint,
115// le pas de temps est sous-estime.
116 const Champ_base& champ_diffusivite = diffusivite_pour_pas_de_temps();
117 const DoubleVect& valeurs_diffusivite = champ_diffusivite.valeurs();
118 double alpha_max = local_max_vect(valeurs_diffusivite);
119
120// Detect if a heat flux is imposed on a boundary through Paroi_echange_externe_impose keyword
121 double h_imp_max = -1, h_imp_temp=-2, max_conductivity = 0;
122 const Zone_Cl_VEF& la_zone_cl_vef = la_zcl_vef.valeur();
123 const Equation_base& mon_eqn = la_zone_cl_vef.equation();
124
125 for(int i=0; i<la_zone_cl_vef.nb_cond_lim(); i++)
126 {
127// loop on boundaries
128 const Cond_lim_base& la_cl = la_zone_cl_vef.les_conditions_limites(i).valeur();
129
130 if (sub_type(Echange_externe_impose,la_cl))
131 {
132 const Echange_externe_impose& la_cl_int = ref_cast(Echange_externe_impose,la_cl);
133 const Champ_front_base& le_ch_front = ref_cast( Champ_front_base, la_cl_int.h_imp().valeur());
134 const DoubleVect& tab = le_ch_front.valeurs();
135 if(tab.size() > 0)
136 {
137 h_imp_temp = local_max_vect(tab); //!< get h_imp from datafile
138 h_imp_temp = std::fabs(h_imp_temp); //!< we should take the absolute value since it can be negative!
139 h_imp_max = (h_imp_temp>h_imp_max) ? h_imp_temp : h_imp_max ; //!< Should we take the max if more than one bc has h_imp ?
140 }
141 }
142 } //!< End loop on boundaries
143
144 h_imp_max = Process::mp_max(h_imp_max);
145
146 if ((alpha_max == 0.)||(zone_VEF.nb_elem()==0))
147 dt_stab = DMAXFLOAT;
148 else
149 {
150 const double min_delta_h_carre = zone_VEF.carre_pas_du_maillage();
151 if (h_imp_max>0.0) //we have a BC with Paroi_echange_externe_impose
152 {
153// get the thermal conductivity
154 const Milieu_base& mon_milieu = mon_eqn.milieu();
155 const DoubleVect& tab_lambda = mon_milieu.conductivite().valeurs();
156 max_conductivity = local_max_vect(tab_lambda);
157// compute Biot number given by Bi = L*h/lambda.
158 double Bi = h_imp_max*sqrt(min_delta_h_carre)/max_conductivity;
159// if Bi>1, replace conductivity by h_imp*h in diffusivity. We are very conservative since h_imp_max is not necessarily located where max_conductivity is.
160 if (Bi>1.0)
161 alpha_max *= h_imp_max*sqrt(min_delta_h_carre)/max_conductivity;
162 }
163 dt_stab = min_delta_h_carre / (2. * dimension * alpha_max);
164 }
165
166 }
167 else
168 {
169 const double deux_dim = 2. * Objet_U::dimension;
170 const Champ_base& champ_diffu = diffusivite();
171 const DoubleTab& valeurs_diffu = champ_diffu.valeurs();
172 const Champ_base& champ_rho = get_champ_masse_volumique();
173 const DoubleTab& valeurs_rho = champ_rho.valeurs();
174 assert(sub_type(Champ_Fonc_P0_base, champ_rho));
175 assert(sub_type(Champ_Fonc_P0_base, champ_diffu));
176 const int nb_elem = zone_VEF.nb_elem();
177 assert(valeurs_rho.size_array()==zone_VEF.nb_elem_tot());
178// Champ de masse volumique variable.
179 for (int elem = 0; elem < nb_elem; elem++)
180 {
181 const double h_carre = zone_VEF.carre_pas_maille(elem);
182 const double diffu = valeurs_diffu(elem);
183 const double rho = valeurs_rho(elem);
184 const double dt = h_carre * rho / (deux_dim * (diffu+DMINFLOAT));
185 if (dt_stab > dt)
186 dt_stab = dt;
187 }
188 }
189 dt_stab = Process::mp_min(dt_stab);
190 return dt_stab;
191}

194void Op_Diff_VEF_base::calculer_pour_post(Champ& espace_stockage,const Nom& option,int comp) const
195{
196 if (Motcle(option)=="stabilite")
197 {
198 DoubleTab& es_valeurs = espace_stockage->valeurs();
199
200 if (la_zone_vef.non_nul())
201 {
202 remplir_nu(nu_);
203 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
204 if (! has_champ_masse_volumique())
205 {
206
207 const Champ_base& champ_diffusivite = diffusivite_pour_pas_de_temps();
208 const DoubleVect& valeurs_diffusivite = champ_diffusivite.valeurs();
209 double alpha_max = local_max_vect(valeurs_diffusivite);
210
211 if ((alpha_max == 0.)||(zone_VEF.nb_elem()==0))
212 es_valeurs = DMAXFLOAT;
213 else
214 {
215 const int nb_elem = zone_VEF.nb_elem();
216 for (int elem = 0; elem < nb_elem; elem++)
217 {
218 es_valeurs(elem) = zone_VEF.carre_pas_maille(elem) / (2. * dimension * alpha_max);
219 }
220 }
221 }
222 else
223 {
224 const double deux_dim = 2. * Objet_U::dimension;
225 const Champ_base& champ_diffu = diffusivite();
226 const DoubleTab& valeurs_diffu = champ_diffu.valeurs();
227 const Champ_base& champ_rho = get_champ_masse_volumique();
228 const DoubleTab& valeurs_rho = champ_rho.valeurs();
229 assert(sub_type(Champ_Fonc_P0_base, champ_rho));
230 assert(sub_type(Champ_Fonc_P0_base, champ_diffu));
231 const int nb_elem = zone_VEF.nb_elem();
232 assert(valeurs_rho.size_array()==zone_VEF.nb_elem_tot());
233// Champ de masse volumique variable.
234 for (int elem = 0; elem < nb_elem; elem++)
235 {
236 const double h_carre = zone_VEF.carre_pas_maille(elem);
237 const double diffu = valeurs_diffu(elem);
238 const double rho = valeurs_rho(elem);
239 const double dt = h_carre * rho / (deux_dim * diffu);
240 es_valeurs(elem) = dt;
241 }
242 }
243
244 assert(mp_min_vect(es_valeurs)==calculer_dt_stab());
245 }
246 }
247 else
248 Operateur_Diff_base::calculer_pour_post(espace_stockage,option,comp);
249}

261void Op_Diff_VEF_base::remplir_nu(DoubleTab& nu) const
262{
263 const Zone_VEF& zone_VEF = la_zone_vef.valeur();
264// On dimensionne nu
265 const DoubleTab& diffu=diffusivite().valeurs();
266 if (!nu.get_md_vector().non_nul())
267 {
268 nu.resize(0, diffu.line_size());
269 zone_VEF.zone().creer_tableau_elements(nu, Array_base::NOCOPY_NOINIT);
270 }
271 if (! diffu.get_md_vector().non_nul())
272 {
273// diffusvite uniforme
274 const int n = nu.dimension_tot(0);
275 const int nb_comp = nu.line_size();
276// Tableaux vus comme uni-dimenionnels:
277 const DoubleVect& arr_diffu = diffu;
278 DoubleVect& arr_nu = nu;
279 for (int i = 0; i < n; i++)
280 for (int j = 0; j < nb_comp; j++)
281 arr_nu[i*nb_comp + j] = arr_diffu[j];
282 }
283 else
284 {
285 assert(nu.get_md_vector() == diffu.get_md_vector());
286 assert_espace_virtuel_vect(diffu);
287 nu.inject_array(diffu);
288 }
