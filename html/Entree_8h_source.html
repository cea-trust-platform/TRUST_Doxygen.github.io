132template<typename _TYPE_>
133int Entree::get_template(_TYPE_ *ob, int n)
134{
135 static constexpr bool IS_INT = std::is_same<_TYPE_,int>::value;
136
137 assert(istream_!=0);
138 assert(n >= 0);
139 if (bin_)
140 {
141 if (is_different_int_size_ && IS_INT)
142 {
143 for (int i = 0; i < n; i++) (*this) >> ob[i];
144 }
145 else
146 {
147 char *ptr = (char*) ob;
148// En binaire: lecture optimisee en bloc:
149 std::streamsize sz = sizeof(_TYPE_);
150 sz *= n;
151// Overflow checking :
152 assert(sz / (std::streamsize )sizeof(_TYPE_) == (std::streamsize )n);
153 istream_->read(ptr, sz);
154 error_handle(istream_->fail());
155 }
156 }
157 else
158 {
159// En ascii : on passe par operator>> pour verifier les conversions
160// Attention : on appelle celui de cette classe, pas de la classe derivee
161 for (int i = 0; i < n; i++) Entree::operator>>(ob[i]);
162 }
163 return (!istream_->fail());
164}

166template <typename _TYPE_>
167Entree& Entree::operator_template(_TYPE_& ob)
168{
169 static constexpr bool IS_INT = std::is_same<_TYPE_,int>::value;
170
171 assert(istream_!=0);
172 if (bin_)
173 {
174 if (is_different_int_size_ && IS_INT)
175 {
176#ifdef INT_is_64_
177 True_int pr;
178 char * ptr = (char*) &pr;
179 istream_->read(ptr, sizeof(True_int));
180 ob=pr;
181#else
182 long pr;
183 char *ptr = (char*) &pr;
184 istream_->read(ptr, sizeof(long));
185 if (pr > INT32_MAX)
186 {
187 std::cerr << "Can't read this int64 binary file with an int32 binary: overflow." << std::endl;
188 Process::exit();
189 }
190 ob = pr;
191#endif
192 }
193 else
194 {
195 char *ptr = (char*) &ob;
196 istream_->read(ptr, sizeof(_TYPE_));
197 error_handle(istream_->fail());
198 }
199 }
200 else
201 {
202 if (check_types_)
203 {
204 char buffer[100];
205 int ok = Entree::get(buffer, 100); //!< Bien appeler get de cette classe
206 if (ok)
207 convert_to(buffer, ob);
208 }
209 else
210 {
211 (*istream_) >> ob;
212 error_handle(istream_->fail());
213 }
214 }
215 return *this;
216}
