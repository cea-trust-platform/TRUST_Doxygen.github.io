37// Methode de calcul de la valeur sur une face encadree par elem1 et elem2 d'un champ uniforme ou non a plusieurs composantes
38inline double valeur(const DoubleTab& valeurs_champ, int elem1, int elem2, const int& compo)
39{
40 if (valeurs_champ.dimension(0)==1)
41 return valeurs_champ(0,compo); //!< Champ uniforme
42 else
43 {
44 if (elem2<0) elem2 = elem1; //!< face frontiere
45 if (valeurs_champ.nb_dim()==1)
46 return 0.5*(valeurs_champ(elem1)+valeurs_champ(elem2));
47 else
48 return 0.5*(valeurs_champ(elem1,compo)+valeurs_champ(elem2,compo));
49 }
50}

52DoubleTab& Source_Generique_Face_PolyMAC::ajouter(DoubleTab& resu) const
53{
54 Champ espace_stockage;
55 const Champ_base& la_source = ch_source_->get_champ(espace_stockage); //!< Aux faces
56 const Zone_PolyMAC& zone = la_zone_PolyMAC.valeur();
57 const DoubleVect& pf = zone.porosite_face(), &vf = zone.volumes_entrelaces();
58 const IntTab& fcl = ref_cast(Champ_Face_PolyMAC, equation().inconnue().valeur()).fcl();
59 for (int f = 0, calc_cl = !sub_type(Zone_PolyMAC_P0, zone); f < zone.nb_faces(); f++)
60 if (calc_cl || fcl(f, 0) < 2)
61 resu(f) += pf(f) * vf(f) * la_source.valeurs()(f);
62 return resu;
63}
