71// Fram4 pour centre 4 (inutile je pense mais bon)
72inline double Fram4(const double s1,const double s2, const double s3,const double s4)
73{
74 double smin0 = std::min(s4,s2), smax0 = std::max(s4,s2), smin1 = std::min(s3,s1), smax1 = std::max(s3,s1);
75 double sr0 = (s3-smin0)/(smax0-smin0+DMINFLOAT), sr1 = (s2-smin1)/(smax1-smin1+DMINFLOAT);
76 double fr = 2.*std::max(std::fabs(sr0-0.5),std::fabs(sr1-0.5));
77 fr *= fr;
78 fr *= fr;
79 return std::min(fr,1.0);
80}

82// Fram pour QUICK
83inline double Fram(const double s1,const double s2, const double s3,const double s4)
84{
85 double smin0 = std::min(s4,s2), smax0 = std::max(s4,s2), smin1 = std::min(s3,s1), smax1 = std::max(s3,s1);
86 double sr0 = (std::fabs(smax0-smin0)<DMINFLOAT ? 0. : (s3-smin0)/(smax0-smin0));
87 double sr1 = (std::fabs(smax1-smin1)<DMINFLOAT ? 0. : (s2-smin1)/(smax1-smin1));
88 double fr = 2.*std::max(std::fabs(sr0-0.5),std::fabs(sr1-0.5));
89 fr *= fr;
90 fr *= fr;
91 return std::min(fr,1.0);
92}

94// Fonction de calcul de cf(limiteur de pente) dans le schema Quick-sharp
95inline double sharp2(const double utc)
96{
97 double cf;
98 if ( (utc <= -1) || (utc >= 1.5) ) cf = 0.125;
99 else if ((utc > -1) && (utc <= 0) ) cf = 0.5 + 0.375*utc;
100 else if ((utc <= 0.25) && (utc > 0) ) cf = 0.5 - 0.625*sqrt(utc);
101 else if ((utc > 0.25) && (utc <= 1.) ) cf = 0.25*(1.-utc);
102 else cf = 0.25*(utc-1.);
103 return cf;
104}

120template <typename Type_Double>
121void Eval_Conv_VDF_tools::qcentre4_impl(const int ori,const double dx, const double dxam, const double dxav, const double psc, const int num0, const int num1,
122 const int num0_0, const int num1_1, const int face, const DoubleTab& transporte,Type_Double& flux) const
123{
124 int k, ncomp = flux.size_array();
125 Type_Double T0(ncomp), T0_0(ncomp), T1(ncomp), T1_1(ncomp);
126 for (k=0; k<ncomp; k++)
127 {
128 T0[k] = transporte(num0,k);
129 T0_0[k] = transporte(num0_0,k);
130 T1[k] = transporte(num1,k);
131 T1_1[k] = transporte(num1_1,k);
132 }
133 const double g1 = -dx*dx*(dx/2+dxav)/(4*(dx+dxam+dxav)*(dx+dxam)*dxam), g2 = (dx+2*dxam)*(dx+2*dxav)/(8*dxam*(dx+dxav));
134 const double g3 = (dx+2*dxam)*(dx+2*dxav)/(8*dxav*(dx+dxam)), g4 = -dx*dx*(dx/2+dxam)/(4*(dx+dxam+dxav)*dxav*(dx+dxav));
135 for (k=0; k<ncomp; k++) flux[k] =( g1*T0_0[k] + g2*T0[k] + g3*T1[k] + g4*T1_1[k] ) * psc ;
136}

138template <typename Type_Double>
139void Eval_Conv_VDF_tools::quick_fram_impl(const int ori,const double dx, const double dm0, const double dxam0, const double dm1, const double dxam1, const double psc,
140 const int num0, const int num1, const int num0_0, const int num1_1, const int face, const DoubleTab& transporte,Type_Double& flux) const
141{
142 const int ncomp = flux.size_array();
143 double fr, T0,T0_0=0,T1,T1_1=0,trans_amont,curv;
144
145 for (int k=0; k<ncomp; k++)
146 {
147 T0 = transporte(num0,k);
148 T0_0 = (num0_0!=-1?transporte(num0_0,k):0);
149 T1 = transporte(num1,k);
150 T1_1 = (num1_1!=-1?transporte(num1_1,k):0);
151
152 if (psc > 0)
153 {
154 assert(num0_0!=-1);
155 trans_amont = T0;
156 curv = ( (T1 - T0)/dx - (T0 - T0_0)/dxam0 )/dm0 ;
157 }
158 else
159 {
160 assert(num1_1!=-1);
161 trans_amont = T1;
162 curv = ( (T1_1 - T1)/dxam1 - (T1 - T0)/dx )/dm1;
163 }
164 flux[k] = 0.5*(T0+T1) - 0.125*(dx*dx)*curv;
165// On applique le filtre Fram:
166 fr = ( num0_0 != -1 && num1_1 != -1 ) ? Fram(T0_0,T0,T1,T1_1) : 1.;
167 flux[k] = ((1.-fr)*flux[k] + fr*trans_amont)*psc;
168 }
169}
