126void Extraire_surface::extraire_surface(Domaine& domaine_surfacique,const Domaine& domaine_volumique, const Nom& nom_domaine_surfacique, const Zone_VF& zone_vf, const Nom& expr_elements,const Nom& expr_faces, int avec_les_bords, const Noms& noms_des_bords)
127{
128 domaine_surfacique.nommer(nom_domaine_surfacique);
129 Parser_U condition_elements,condition_faces;
130 condition_elements.setNbVar(dimension);
131 condition_elements.addVar("x");
132 condition_elements.addVar("y");
133 if (dimension==3)
134 condition_elements.addVar("z");
135 condition_faces.setNbVar(dimension);
136 condition_faces.addVar("x");
137 condition_faces.addVar("y");
138 if (dimension==3)
139 condition_faces.addVar("z");
140
141 condition_faces.setString(expr_faces);
142 condition_faces.parseString();
143 condition_elements.setString(expr_elements);
144 condition_elements.parseString();
145
146// Copie des sommets
147 domaine_surfacique.les_sommets()=domaine_volumique.les_sommets();
148 const DoubleTab& coord=domaine_surfacique.les_sommets();
149 Zone zone__;
150 domaine_surfacique.add(zone__);
151 Zone& zone=domaine_surfacique.zone(0);
152 zone.nommer("NO_FACE");
153 const Nom& type_elem=zone_vf.zone().type_elem().valeur().que_suis_je();
154
155 if (dimension==3)
156 {
157 if (type_elem==Motcle("Tetraedre"))
158 {
159 zone.typer("Triangle");
160 }
161 else
162 {
163 if (type_elem==Motcle("Segment"))
164 {
165 zone.typer("Point");
166 }
167 else
168 {
169 if ((type_elem==Motcle("Hexaedre"))|| (type_elem==Motcle("Hexaedre_VEF")))
170 {
171 zone.typer("Quadrangle");
172 }
173 else
174 {
175 if (type_elem==Motcle("Polyedre"))
176 {
177 zone.typer("Polygone");
178 }
179 else
180 {
181 Cerr<<"WARNING "<<type_elem<< " not coded, use Quadrangle" <<finl;
182 zone.typer("Quadrangle");
183// exit();
184 }
185 }
186 }
187 }
188 }
189 else
190 {
191 zone.typer("segment");
192 }
193 zone.associer_domaine(domaine_surfacique);
194
195 const DoubleTab& xp =zone_vf.xp();
196
197 int nb_elem=zone_vf.nb_elem();
198 IntTab marq_elem;
199
200 zone_vf.zone().creer_tableau_elements(marq_elem);
201 std::string X("X"), Y("Y"), Z("Z");
202 for (int elem=0; elem<nb_elem; elem++)
203 {
204 condition_elements.setVar(X,xp(elem,0));
205 condition_elements.setVar(Y,xp(elem,1));
206 if (dimension==3)
207 condition_elements.setVar(Z,xp(elem,2));
208 double res=condition_elements.eval();
209 if (std::fabs(res)>1e-5)
210 marq_elem(elem)=1;
211 else
212 marq_elem(elem)=0;
213
214 }
215 marq_elem.echange_espace_virtuel();
216 const DoubleTab& xv =zone_vf.xv();
217
218 int nb_faces=zone_vf.nb_faces();
219 const IntTab& face_voisin=zone_vf.face_voisins();
220
221 ArrOfInt marq(nb_faces);
222// on marque les joints
223 int nbjoints=zone_vf.nb_joints();
224
225 for(int njoint=0; njoint<nbjoints; njoint++)
226 {
227 const Joint& joint_temp = zone_vf.joint(njoint);
228 int pe_voisin=joint_temp.PEvoisin();
229 if (pe_voisin<me())
230 {
231 const IntTab& indices_faces_joint = joint_temp.joint_item(Joint::FACE).renum_items_communs();
232 const int nbfaces = indices_faces_joint.dimension(0);
233 for (int j = 0; j < nbfaces; j++)
234 {
235 int face_de_joint = indices_faces_joint(j, 1);
236 marq[face_de_joint] = -1;
237 }
238 }
239 }
240 int nb_t=0;
241
242// on flage les face sde bords qui nous interesse
243 ArrOfInt face_bord_int(nb_faces);
244 if (avec_les_bords==1)
245 face_bord_int=1;
246 if (noms_des_bords.size()!=0)
247 {
248 if (avec_les_bords!=0)
249 {
250 Cerr<<" the option avec_les_bords is incompatible with the option avec_certains_bords"<<finl;
251 exit();
252 }
253 for (int b=0; b<noms_des_bords.size(); b++)
254 {
255 const Frontiere& fr=zone_vf.frontiere_dis(zone_vf.rang_frontiere(noms_des_bords[b])).frontiere();
256 int deb=fr.num_premiere_face();
257 int fin=deb+fr.nb_faces();
258 for (int f=deb; f<fin; f++)
259 face_bord_int[f]=1;
260 }
261
262 }
263
264// on marque toutes les faces que l'on veut mettre dans le domaine
265 for (int fac=0; fac<nb_faces; fac++)
266 {
267 int elem0=face_voisin(fac,0);
268 int elem1=face_voisin(fac,1);
269 int val0=-1;
270 if (elem0!=-1) val0=marq_elem(elem0);
271 int val1=-1;
272 if (elem1!=-1) val1=marq_elem(elem1);
273
274 if (val0!=val1)
275 {
276 if (((val0*val1==0)&&(val0+val1==1))||((val0==1)&&(face_bord_int[fac]==1))||((val1==1)&&(face_bord_int[fac]==1)))
277// if (zone_test.chercher_elements(xv(fac,0),xv(fac,1),xv(fac,2))==0)
278 {
279 condition_faces.setVar(X,xv(fac,0));
280 condition_faces.setVar(Y,xv(fac,1));
281 if (dimension==3)
282 condition_faces.setVar(Z,xv(fac,2));
283 double res=condition_faces.eval();
284 if (std::fabs(res)>1e-5)
285 if (marq[fac]!=-1) //!< pas un joint, ou on est le proprietaire
286 {
287 marq[fac]=1;
288 nb_t++;
289 }
290 }
291 }
292 }
293
294 ArrOfDouble point0b(3),point1b(3),point2b(3);
295 Cerr<<"Number of elements of the new domain : "<<nb_t<<finl;
296 IntTab& les_elems=zone.les_elems();
297
298 const IntTab& face_sommets=zone_vf.face_sommets();
299 int nb_sommet_face=face_sommets.dimension(1);
300 les_elems.resize(nb_t,nb_sommet_face);
301 int nb=0;
302 ArrOfDouble normal(dimension);
303 ArrOfDouble normal_b(3);
304 for (int fac=0; fac<nb_faces; fac++)
305 if (marq[fac]==1)
306 {
307 int el1=face_voisin(fac,0);
308 if (el1==-1) el1=face_voisin(fac,1);
309 if (marq_elem(el1)!=1)
310 el1=face_voisin(fac,1);
311 for (int i=0; i<dimension; i++)
312 normal[i]=xv(fac,i)-xp(el1,i);
313 for (int s=0; s<nb_sommet_face; s++)
314 les_elems(nb,s)=face_sommets(fac,s);
315// on calcule la normale
316 if (nb_sommet_face>1)
317 {
318 if (dimension==3)
319 {
320 for (int i=0; i<3; i++)
321 {
322 point0b[i]=coord(les_elems(nb,0),i);
323 point1b[i]=coord(les_elems(nb,1),i);
324 point2b[i]=coord(les_elems(nb,2),i);
325 }
326 calcul_normal(point0b,point1b,point2b,normal_b);
327 if (dotproduct_array(normal,normal_b)<0)
328 {
329// si normal a l'envers on inverse les deux sommets
330 les_elems(nb,1)=face_sommets(fac,2);
331 les_elems(nb,2)=face_sommets(fac,1);
332 }
333 }
334 else
335 {
336 for (int i=0; i<nb_sommet_face; i++)
337 {
338
339 point0b[i]=coord(les_elems(nb,1),i)-coord(les_elems(nb,0),i);
340 }
341 double produit=normal[0]*point0b[1]-normal[1]*point0b[0];
342 if (produit<0)
343 {
344// si normal a l'envers on inverse les deux sommets
345 les_elems(nb,0)=face_sommets(fac,1);
346 les_elems(nb,1)=face_sommets(fac,0);
347 }
348 }
349 }
350 nb++;
351 }
352
353 assert(nb==nb_t);
354 NettoieNoeuds::nettoie(domaine_surfacique);
355}
