72 const Zone_dis_base& zdisbase=mon_equation->inconnue().zone_dis_base();

92 flag=1;

105

111
112

114 {

127 for (int iface=num1; iface<num2; iface++)

130 else tmp = rho(iface)*vitesse(iface,dir)*zone_VEF.face_surfaces(iface);
131 Tmoyen += tmp*tab_temperature(iface);
132 rhoUS += tmp;
133 if(tab_temperature(iface)>Tmax) Tmax=tab_temperature(iface);
134 if(tab_temperature(iface)<Tmin) Tmin=tab_temperature(iface);
135 }
136

143
144 if (Process::je_suis_maitre())
145 {
146 if (rhoUS==0)
147 {
148 Cerr << "rhoUS=0..." << finl;
149 Cerr << "May be the direction of Traitement_particulier { temperature is wrong..." << finl;
150 exit();
151 }
152 Tmoyen /= rhoUS;
153 Nom nom_fic1("Tmoyen_");
154 nom_fic1 += le_bord.le_nom();
155 SFichier fic1(nom_fic1, ios::app);
156 fic1 << temps << " " << Tmoyen << " " << Tmin << " " << Tmax << finl;
157
158 Nom nom_fic2("RhoU_");

163 }
