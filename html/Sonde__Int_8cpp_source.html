240 for (i=0; i<dimension; i++)
241 is >> extremite1(i);
242 for (i=0; i<dimension; i++)

244 for (i=0; i<dimension; i++)

253 for (j=0; j<nbre_points2; j++)
254 for (int m=0; m<nbre_points3; m++)

275 const Zone& la_zone =pb.domaine().zone(0);
276 initialiser(la_zone);
277 return is;
