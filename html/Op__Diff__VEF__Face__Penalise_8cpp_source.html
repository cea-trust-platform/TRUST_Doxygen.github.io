76//
77//
78//! methode qui calcule la vitesse au temps n+1 lorsque le<br>schema explicite est utilise.
79DoubleTab& Op_Diff_VEF_Face_Penalise::
80ajouter(const DoubleTab& inconnue, DoubleTab& resu) const
81{
82// Cerr << "J'entre dans ajouter() de penalisation" << finl;
83 int nb_composante,numero_global_face,local;
84 int face_penalisation,face;
85 double coeff;
86 IntList voisinage_ordre1,voisinage_ordre1_strict;
87 IntList voisinage_ordre2,voisinage_ordre2_strict;
88 IntList ensemble_faces;
89
90 /* On calcule le resultat du a la matrice de penalisation
91 * i.e. somme_j U_j P_i,j ou P_i,j est le terme de la matrice
92 * de penalisation.
93 */
94 for (nb_composante = 0; nb_composante < dimension; nb_composante++)
95 {
96 for (numero_global_face = 0; numero_global_face < zone_vef().nb_faces();
97 numero_global_face++)
98 {
99 /* Calcul des voisinages de numero_global_face */
100 voisinage(numero_global_face,voisinage_ordre1);
101 voisinage(voisinage_ordre1,voisinage_ordre2);
102 reduction(voisinage_ordre1,voisinage_ordre2,voisinage_ordre2_strict);
103// Cerr << "Nombre iterations " << numero_global_face << finl;
104// Cerr << "Nombre de faces " << zone_vef().nb_faces() << finl;
105
106 /* Calcul de resu(numero_face_global,nb_comp)
107 * pour le voisinage_ordre2_strict
108 */
109 for (local = 0 ; local < voisinage_ordre2_strict.size(); local++)
110 {
111 face = voisinage_ordre2_strict[local];
112 faces_communes(numero_global_face,face,ensemble_faces);
113// Cerr << "Taille voisinage_ordre_1 " << voisinage_ordre1.size()
114// << finl;
115// Cerr << "Taille voisinage_ordre2 " << voisinage_ordre2.size()
116// << finl;
117// Cerr << "Taille voisinage_ordre2_strict " << voisinage_ordre2_strict.size() << finl;
118// Cerr << "Taille ensemble_faces " << ensemble_faces.size() << finl;
119
120// if (ensemble_faces.size() != 1)
121// {
122// Cerr << "Erreur dans ajouter() de la penalisation." << finl;
123// Process::exit();
124// }
125
126 for (int mm = 0; mm < ensemble_faces.size(); mm++)
127 {
128 face_penalisation = ensemble_faces[mm];
129
130 resu(numero_global_face,nb_composante) +=
131 inconnue(face,nb_composante)*
132 signe(numero_global_face,face)*
133 coefficient_penalisation(face_penalisation)*
134 longueur(face_penalisation)*
135 1./3.;
136 }
137
138 }
139
140 /* Calcul de resu(numero_global_face,nb_comp)
141 * pour le voisinage_ordre1_strict
142 */
143 for (local = 0 ; local < voisinage_ordre1_strict.size(); local++)
144 {
145 assert(voisinage_ordre1_strict.size() == 2
146 || voisinage_ordre1_strict.size() == 4);
147
148 face = voisinage_ordre1_strict[local];
149 face_penalisation = autre_face(numero_global_face,face);
150
151 resu(numero_global_face,nb_composante) +=
152 inconnue(face,nb_composante)*
153 coefficient_penalisation(face_penalisation)*
154 longueur(face_penalisation)*
155 (-1./3.);
156 }
157
158 /* Cas 1: la face consideree est au bord du domaine.
159 * et numero_global_face = face
160 */
161 if (numero_global_face < zone_vef().nb_faces_bord())
162 for (local = 0; local < voisinage_ordre1.size(); local++)
163 {
164 assert(voisinage_ordre1.size() == 3);
165
166 face_penalisation = voisinage_ordre1[local];
167
168 if (numero_global_face == face_penalisation)
169 coeff = 1.;
170 else
171 coeff = 1./3.;
172
173 resu(numero_global_face,nb_composante) +=
174 inconnue(numero_global_face,nb_composante) *
175 coeff *
176 coefficient_penalisation(face_penalisation)*
177 longueur(face_penalisation);
178 }
179
180 /* Cas 2: la face est interne et numero_global_face = face */
181 else
182 for (local = 0; local < voisinage_ordre1_strict.size(); local++)
183 {
184 assert( voisinage_ordre1_strict.size() == 4);
185
186 face_penalisation = voisinage_ordre1_strict[local];
187
188 resu(numero_global_face,nb_composante) +=
189 inconnue(numero_global_face,nb_composante) *
190 coefficient_penalisation(face_penalisation)*
191 longueur(face_penalisation)*
192 1./3.;
193 }
194 }
195 }
196
197
198 /* Ajout de la matrice de diffusion classique */
199 Op_Diff_VEF_Face::ajouter(inconnue,resu);
200
201// Cerr << "Je sors de calcul_matrice_de_penalisation_" << finl;
202 return resu;
203}

213//
214//! methode qui calcule le voisinage d'une face.
215void Op_Diff_VEF_Face_Penalise::
216voisinage(const int Numero_face, IntList& Voisinage) const
217{
218// Cerr << "J'entre dans voisinage pour un numero face" << finl;
219
220 /* On vide la liste Voisinage pour eviter les mauvaises surprises */
221 if (! Voisinage.est_vide() ) Voisinage.vide();
222
223 /* Declaration des principaux parametres locaux. */
224 int numero_local;
225
226 /* Le nombre de faces des elements constituant la zone
227 * de discretisation.
228 * REM: on suppose que l'on ne travaille pas sur des prismes.
229 */
230 const int nb_faces_element = zone().nb_faces_elem();
231
232 /* Les elements voisins de Numero_face */
233 const int voisin1 = zone_vef().face_voisins(Numero_face,1);
234 const int voisin2 = zone_vef().face_voisins(Numero_face,0);
235
236 /* Ici, on recupere les faces de voisin*
237 * que l'on injecte dans la liste Voisinage.
238 * Il faut donc que l'element existe d'ou le premier test realise.
239 */
240 if (voisin1 != -1)
241 {
242 for (numero_local = 0; numero_local < nb_faces_element; numero_local++)
243 {
244 /* On recupere le numero global de chacune des
245 * faces de voisin*
246 */
247 const int numero_global_face =
248 zone_vef().elem_faces(voisin1,numero_local);
249
250 /* Puis on place ce numero dans la liste Voisinage. */
251 Voisinage.add_if_not(numero_global_face);
252
253 }//fin du for
254
255 }//fin du if
256
257 if (voisin2 != -1)
258 {
259 for (numero_local = 0; numero_local < nb_faces_element; numero_local++)
260 {
261 /* On recupere le numero global de chacune des
262 * faces de numero_element_*
263 */
264 const int numero_global_face =
265 zone_vef().elem_faces(voisin2,numero_local);
266
267 /* Puis on place ce numero dans la liste Voisinage. */
268 Voisinage.add_if_not(numero_global_face);
269
270 }//!< fin du for
271
272 }//fin du if
273
274// Cerr << "Je sors de voisinage pour un numero de face" << finl;
275}

303 /* Parametre interne a la boucle */

310 /* Dans la liste temporaire, on place le voisinage

324 liste_temporaire[nb_elements_liste_temporaire];
325
326 /* Enfin, on stoke dans Voisinage */
327 Voisinage.add_if_not(numero_face_dans_liste_temporaire);
328
329 }//fin du deuxieme for

331 }//fin du premier for

333// Cerr << "Je sors de voisinage pour un ensemble face" << finl;

345double Op_Diff_VEF_Face_Penalise::
346signe(const int Face1, const int Face2) const
347{
348// Cerr << "J'entre dans signe" << finl;
349
350 /* Parametres locaux a la procedure */
351 int numero_local;
352
353 /* On recupere le nombre de sommets des faces
354 * des elements constituant la zone discretisee.
355 * REM: on exclut le prisme par convention.
356 */
357 const int nb_sommets_par_face = zone_vef().nb_som_face();
358
359 /* On cree une liste qui contient les sommets de la Face2 */
360 IntList sommets_Face2;
361
362 for (numero_local = 0 ; numero_local < nb_sommets_par_face ; numero_local++)
363 sommets_Face2.add(zone_vef().face_sommets(Face2,numero_local));
364
365 /* Enfin, on regarde si des sommets de la Face2 appartiennent
366 * a la Face1.
367 * Si oui, alors on retourne 1 sinon on retourne -1
368 */
369 for (numero_local = 0; numero_local < nb_sommets_par_face ; numero_local++)
370 if ( sommets_Face2.contient(zone_vef().face_sommets(Face1,numero_local)))
371 return 1.;
372
373 return -1.;
374
375// Cerr << "Je sors de signe" << finl;
376}

387//
388//! fonction membre qui retourne le coefficient de<br>penalisation associee a chaque face du maillage primaire.
389double Op_Diff_VEF_Face_Penalise::
390coefficient_penalisation(const int Numero_face) const
391{
392// Cerr << "J'entre dans coefficient_penalisation" << finl;
393 /* Initialisation de parametres locaux */
394 double eta=0.;
395 double coefficientpenalisation = 0.;
396
397 const int voisin1 =
398 zone_vef().face_voisins(Numero_face,1);
399
400 const int voisin2 =
401 zone_vef().face_voisins(Numero_face,0);
402
403 if (voisin1 == -1 && voisin2 == -1)
404 {
405 Cerr << "Erreur dans Op_Dift_standard_Face_VEF_penalise::"
406 << "coefficient_penalisation()" << finl;
407 Process::exit();
408 }
409
410 if (voisin1 != -1 && voisin2 != -1)
411 {
412 /* Calcul des coefficients de penalisation proprement dit */
413 coefficientpenalisation = 1./diametre(voisin1);
414 eta = 1./diametre(voisin2);
415
416 coefficientpenalisation = std::min(coefficientpenalisation,eta);
417 }
418
419 if (voisin1 == -1)
420 coefficientpenalisation = 1./diametre(voisin2);
421
422 if (voisin2 == -1)
423 coefficientpenalisation = 1./diametre(voisin1);
424
425 return coefficientpenalisation;
426// Cerr << "Je sors de coefficient_penalisation" << finl;
427}

488
489 /* Declaration des parametres locaux a la procedure */
490 const IntList *liste_de_plus_petite_taille,*liste_de_plus_grande_taille;
491 int nb_element_dans_liste;
492
493 /* On teste la taille des listes passees en parametres puis
494 * on effectue l'allocation en fonction du resultat.
495 */
496 if (Liste1.size() >= Liste2.size())
497 {
498 liste_de_plus_petite_taille = &Liste2;
499 liste_de_plus_grande_taille = &Liste1;
500 }
501 else
502 {
503 liste_de_plus_petite_taille = &Liste1;
504 liste_de_plus_grande_taille = &Liste2;
505 }
506
507 /* Enfin, on supprime de la liste_de_plus_grande_taille,
508 * les elements de la liste_de_plus_petite_taille si ceux-ci
509 * sont bien dans la liste_de_plus_grande_taille.
510 */
511 for (int ll = 0 ; ll < (*liste_de_plus_grande_taille).size() ; ll++)
512 Liste_reduite.add( (*liste_de_plus_grande_taille)[ll] );
513
514 for (nb_element_dans_liste = 0;
515 nb_element_dans_liste < (*liste_de_plus_petite_taille).size();
516 nb_element_dans_liste++)
517 if
518 (Liste_reduite.contient( (*liste_de_plus_petite_taille)
519 [nb_element_dans_liste] ) )
520 Liste_reduite.suppr( (*liste_de_plus_petite_taille)
521 [nb_element_dans_liste] );
522
523// Cerr << "Je sors de reduction" << finl;
524}
