31//! VTRUSTdata alloue une zone de memoire de la taille specifiee au constructeur, et libere la zone de memoire a la destruction.<br>La memoire peut etre allouee sur le tas (avec new) ou par le mecanisme Memoire::add_trav_double.<br>"ref_count" compte le nombre de pointeurs qui font reference a "this". (permet au dernier utilisateur de l'objet de le detruire)
32/** \sa TRUSTArray */
33template<typename _TYPE_>
34class VTRUSTdata
35{
36public:
37 typedef _TYPE_ value_type;
38
39 VTRUSTdata(int size, Array_base::Storage storage);
40
41// Detruit la zone de memoire allouee.
42// Precondition : ref_count == 0 (la zone de memoire ne doit etre referencee nulle part)
43 ~VTRUSTdata()
44 {
45 assert(ref_count_ == 0);
46 if (d_ptr_trav_ == 0) delete[] data_; //!< Stockage STANDARD
47 else //!< Stockage TEMP_STORAGE
48 {
49 d_ptr_trav_->unlock();
50 d_ptr_trav_ = 0;
51 }
52 data_ = 0; //!< paranoia: si size_==-1 c'est qu'on pointe sur un zombie
53 size_ = -1; //!< (pointeur vers un objet qui a ete detruit)
54 }
55
56 inline _TYPE_ * get_data() { return data_; }
57 inline const _TYPE_ * get_data() const { return data_; }
58 inline int add_one_ref() { return ++ref_count_; } //!< Un nouveau tableau utilise cette zone memoire : incremente ref_count
59 inline int suppr_one_ref() { assert(ref_count_ > 0); return (--ref_count_); } //!< Un tableau de moins utilise cette zone memoire : decremente ref_count
60 inline int ref_count() const { return ref_count_; }
61 inline int get_size() const { return size_; }
62
63private:
64// Le constructeur par copie et l'operateur= sont interdits.
65 VTRUSTdata(const VTRUSTdata& v) = delete;
66 VTRUSTdata& operator=(const VTRUSTdata& v) = delete;
67
68// "data" est un pointeur sur une zone de memoire de taille sz * sizeof(_TYPE_), allouee par le constructeur et liberee par le destructeur.
69// Ce pointeur n'est jamais nul meme si size_==0
70 _TYPE_ * data_;
71
72// Compteur incremente par add_one_ref et decremente par suppr_one_ref. Contient le nombre d'objets TRUSTArray dont le membre "p" pointe vers "this". On a ref_count_ >= 0.
73 int ref_count_;
74
75// "sz" est la taille du tableau "data_" alloue. On a sz >= 0.
76 int size_;
77
78// Si storage est de type TEMP_STORAGE, d_ptr_trav porte la reference a la zone allouee, sinon le pointeur est nul.
79 TRUST_ptr_trav<_TYPE_> * d_ptr_trav_;
80};

92//! Construit un VTRUSTdata de taille size >= 0<br>Parametre: int s<br>Signification: taille du VTRUSTdata, il faut size >= 0<br>Parametre: Storage storage<br>Signification: indique si la memoire doit etre allouee avec "new" ou avec "memoire.add_trav_double()"<br>Valeurs par defaut: STANDARD (allocation avec "new")
93template<typename _TYPE_>
94VTRUSTdata<_TYPE_>::VTRUSTdata(int size, Array_base::Storage storage)
95{
96 const int size_warning = 100000000;
97 assert(size >= 0);
98 if (size == 0) storage = Array_base::STANDARD;
99
100 switch (storage)
101 {
102 case Array_base::STANDARD:
103 {
104 d_ptr_trav_ = 0;
105 if (size>size_warning) Cerr << "Warning: Allocating an array of " << size << " " << typeid(_TYPE_).name() << finl;
106#ifdef _EXCEPTION_
107// Allocation de la memoire sur le tas
108 try
109 {
110 data_ = new _TYPE_[size];
111 }
112 catch(...)
113 {
114 Cerr << "unable to allocate " << size << " _TYPE_ " << finl;
115 Process::exit();
116 }
117#else
118 data_ = new _TYPE_[size];
119 if(!data_)
120 {
121 Cerr << "unable to allocate " << size << "_TYPE_ " << finl;
122 Process::exit();
123 }
124#endif
125 if (size>size_warning) Cerr << " OK" << finl;
126 break;
127 }
128 case Array_base::TEMP_STORAGE:
129 {
130// Allocation de la memoire sur un tas special. La memoire ne sera pas rendue au systeme mais conservee pour une reutilisation ulterieure.
131 Memoire& memoire = Memoire::Instance();
132 if (size>size_warning) Cerr << "Warning: Allocating or reusing a " << typeid(_TYPE_).name() << " Trav of " << size << " elements ...";
133 d_ptr_trav_ = memoire.template add_trav<_TYPE_>(size);
134 assert(d_ptr_trav_ != 0);
135 data_ = d_ptr_trav_->ptr_();
136 if (size>size_warning) Cerr << " OK" << finl;
137 break;
138
139 }
140 default:
141 Process::exit();
142 }
143 ref_count_ = 1;
144 size_ = size;
145 assert(data_ != 0);
146}
