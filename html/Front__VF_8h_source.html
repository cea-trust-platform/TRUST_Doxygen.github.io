77inline int Front_VF::num_local_face(const int ind_global_face) const
78{
79 if (ind_global_face<num_premiere_face()+nb_faces())
80 {
81 return ind_global_face-num_premiere_face();
82 }
83 else
84 {
85 for (int i=0; i<frontiere().get_faces_virt().size_array(); i++)
86 {
87 if (frontiere().get_faces_virt()[i]==ind_global_face)
88 return i+nb_faces();
89 }
90 }
91 return -1;
92}
