64const Champ_base& Champ_Generique_modifier_pour_QC::get_champ(Champ& espace_stockage) const
65{
66 Champ source_espace_stockage;
67 const Champ_base& source = get_source(0).get_champ(source_espace_stockage);
68 Nature_du_champ nature_source = source.nature_du_champ();
69 int nb_comp = source.nb_comp();
70 Champ_Fonc es_tmp;
71 espace_stockage = creer_espace_stockage(nature_source,nb_comp,es_tmp);
72
73 DoubleTab& val = espace_stockage.valeurs();
74 val = source.valeurs();
75 const Fluide_Quasi_Compressible& fluide = ref_cast(Fluide_Quasi_Compressible,mon_milieu_.valeur());
76 multiplier_diviser_rho(val,fluide,diviser_);
77 val.echange_espace_virtuel();
78 return espace_stockage.valeur();
79}
