34Entree& Perte_Charge_Anisotrope_PolyMAC_Face::readOn(Entree& s )
35{
36 Cerr << "Perte_Charge_Anisotrope_PolyMAC_Face::readOn " << finl;
37 sous_zone=false;
38 int lambda_ortho_ok=0;
39 int lambda_ok=0;
40
41// Definition des mots-cles
42 Motcles les_mots(6);
43 les_mots[0] = "lambda";
44 les_mots[1] = "diam_hydr";
45 les_mots[2] = "sous_zone";
46 les_mots[3] = "direction";
47 les_mots[4] = "lambda_ortho";
48 les_mots[5] = "implicite";
49
50// Lecture et interpretation
51 Motcle motlu, accolade_fermee="}", accolade_ouverte="{";
52 s >> motlu;
53 while (motlu != accolade_ouverte)
54 {
55 Cerr << "On attendait une { a la lecture d'un " << que_suis_je() << finl;
56 Cerr << "et non : " << motlu << finl;
57 exit();
58 }
59 s >> motlu;
60 while (motlu != accolade_fermee)
61 {
62 int rang=les_mots.search(motlu);
63 switch(rang)
64 {
65 case 0 : //!< lambda
66 {
67 lambda_ok=1;
68 Nom tmp;
69 s >> tmp;
70 Cerr << "Lecture et interpretation de la fonction " << tmp << " ... ";
71 lambda.setNbVar(2+dimension);
72 lambda.setString(tmp);
73 lambda.addVar("Re");
74 lambda.addVar("t");
75 lambda.addVar("x");
76 if (dimension>1)
77 lambda.addVar("y");
78 if (dimension>2)
79 lambda.addVar("z");
80 lambda.parseString();
81 Cerr << " Ok" << finl;
82 break;
83 }
84 case 1: //!< diam_hydr
85 s >> diam_hydr;
86 break;
87 case 2: //!< sous_zone
88 s >> nom_sous_zone;
89 sous_zone=true;
90 break;
91 case 3: //!< direction
92 s >> v;
93 break;
94 case 4: //!< lambda_ortho
95 {
96 lambda_ortho_ok=1;
97 Nom tmp;
98 s >> tmp;
99 Cerr << "Lecture et interpretation de la fonction " << tmp << " ... ";
100 lambda_ortho.setNbVar(2+dimension);
101 lambda_ortho.setString(tmp);
102 lambda_ortho.addVar("Re");
103 lambda_ortho.addVar("t");
104 lambda_ortho.addVar("x");
105 if (dimension>1)
106 lambda_ortho.addVar("y");
107 if (dimension>2)
108 lambda_ortho.addVar("z");
109 lambda_ortho.parseString();
110 Cerr << " Ok" << finl;
111 break;
112 }
113 case 5:
114 {
115 s>>implicite_;
116 break;
117 }
118
119 default : //!< non compris
120 Cerr << "Mot cle \"" << motlu << "\" non compris lors de la lecture d'un "
121 << que_suis_je() << finl;
122 exit();
123 }
124 s >> motlu;
125 }
126
127// Verification de la coherence
128 if (lambda_ok==0)
129 {
130 Cerr << "Il faut definir l'expression lamba(Re)" << finl;
131 exit();
132 }
133
134 if (lambda_ortho_ok==0)
135 {
136 Cerr << "Il faut definir l'expression lamba_ortho(Re)" << finl;
137 exit();
138 }
139
140 if (diam_hydr->nb_comp()!=1)
141 {
142 Cerr << "Il faut definir le champ diam_hydr a une composante" << finl;
143 exit();
144 }
145
146 if (v->nb_comp()!=dimension)
147 {
148 Cerr << "Il faut definir le champ direction a " << dimension << " composantes" << finl;
149 exit();
150 }
151
152
153 Cerr << "Fin de Perte_Charge_Anisotrope_PolyMAC_Face::readOn" << finl;
154 return s;
155}
