191// Estimation du temps de latence MPI_allreduce
192static double estimate_allreduce(double max_bench_time)
193{
194 Process::barrier();
195 double t1 = Statistiques::get_time_now();
196 double t2 = t1;
197 int howmany = 100;
198 int i = 0;
199 for (i = 0; i < howmany; i++)
200 Process::mp_sum((int)1);
201 t2 = Statistiques::get_time_now();
202 t2 = Process::mp_min(t2-t1);
203 return t2 / (double) howmany;
204}

206inline void write_stat_file(const Nom& msg, const Stat_Results& stat, const Stat_Results& pas_de_temps, SFichier& stat_file)
207{
208 if (stat.max_time!=0)
209 {
210 stat_file << "Dont " << msg << " ";
211 if (pas_de_temps.max_count>0)
212 stat_file << stat.max_time/pas_de_temps.max_count;
213 else
214 stat_file << stat.max_time;
215 stat_file << " " << (int)(stat.max_time/pas_de_temps.max_time*100) << "%";
216 if (pas_de_temps.max_count>0)
217 {
218 double n = stat.max_count/pas_de_temps.max_count;
219 stat_file << " (" << n << " " << (n==1?"appel":"appels") << "/pas de temps)";
220 }
221 stat_file << "\n";
222 }
223}
