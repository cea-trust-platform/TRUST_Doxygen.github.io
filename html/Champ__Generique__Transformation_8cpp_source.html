51// -methode : type de transformation a realiser
52// methodes possible :
53// formule : fonction de champs generiques sources, x, y, z et t
54// vecteur : construction d un vecteur a partir des donnees fournies
55// produit_scalaire : produit scalaire de deux vecteurs specifies
56// norme : norme d un vecteur specifie
57// composante : creation d un champ scalaire a partir d une composante d un champ vectoriel
58// -expression : expression de la transformation (cas des methodes formule et vecteur)
59// -numero : numero de la composante a extraire (uniquement pour le cas de la methode composante)
60// -localisation : localisation du champ de stockage

70int Champ_Generique_Transformation::lire_motcle_non_standard(const Motcle& mot, Entree& is)
71{
72 if (mot=="expression")
73 {
74 int dim;
75 is >> dim;
76 les_fct.dimensionner(dim);
77 if ((Motcle(methode_) == "formule") && (dim!=1) )
78 {
79 Cerr << "The integer N of the expression with FORMULE method must be equal to 1." <<finl;
80 Cerr << "Use the VECTEUR expression method for N="<<dim<<finl;
81 exit();
82 }
83 for (int i=0; i<dim; i++)
84 is>>les_fct[i];
85 return 1;
86 }
87 else if (mot=="numero")
88 {
89 int num_compo;
90 is >> num_compo;
91 nb_comp_ = 1;
92 nature_ch = scalaire;
93 /* test numero composante : on ne peut pas se baser sur la dimension de l'espace dans le cas d'un champ multi_scalaire
94 Probleme : c'est nb_comp_ qui portera par la suite la dimension des composantes et il est ici initialise a 1 pour eviter bug
95 Le test est deplace dans Champ_Generique_Transformation::get_champ
96
97 if ((num_compo<0) || (num_compo>dimension-1))
98 {
99 // Cerr<<"Component number must be >=0 and < "<<dimension<<" for composante option."<<finl;
100 Cerr<<"Component number must be >=0 for composante option."<<finl;
101 exit();
102 } */
103 les_fct.dimensionner(num_compo+1);
104 return 1;
105 }
106 else
107 return Champ_Gen_de_Champs_Gen::lire_motcle_non_standard(mot,is);
108 return 1;

163void Champ_Generique_Transformation::completer(const Postraitement_base& post)
164{
165
166 bool sources_location_ok = true;
167 bool fictive_source = false;
168
169 int nb_source_fictive=0;
170 int nb_sources = get_nb_sources();
171 if (nb_sources==0)
172 {
173 nb_source_fictive = 1;
174 Champ_Generique& new_src = get_set_sources().add(Champ_Generique());
175 Nom nom_probleme = ref_cast(Postraitement,post).probleme().le_nom();
176 const Objet_U& ob = interprete().objet(nom_probleme);
177 const Probleme_base& pb = ref_cast(Probleme_base,ob);
178 const Nom& nom_champ = pb.equation(0).inconnue().le_nom();
179 Entree s_bidon;
180 Nom ajout("");
181 ajout += " Champ_Post_refChamp { Pb_champ ";
182 ajout += pb.le_nom();
183 ajout += " ";
184 ajout += nom_champ;
185 ajout += " }";
186 Entree_complete s_complete(ajout,s_bidon);
187 s_complete>>new_src;
188 nb_sources = get_nb_sources();
189 }
190
191 Champ_Gen_de_Champs_Gen::completer(post);
192
193 if (nb_source_fictive==1)
194 {
195 if (get_nb_sources()>1)
196 {
197 assert( get_set_sources().size( ) == 1 );
198// on a rajoute une source pour rien ...
199 get_set_sources().vide();
200 nb_source_fictive = 0;
201 }
202 }
203
204 nb_sources = get_nb_sources();
205
206// On determine le nombre de composantes qui sera attribue
207// a l espace de stockage rendu par ce champ de postraitement en
208// recherchant la source qui possede le plus grand nombre de composantes
209
210 nb_comp_ = 1;
211 nature_ch = scalaire;
212
213 if (nb_source_fictive==0)
214 {
215 for (int i=0; i<nb_sources; i++)
216 {
217 Champ source_espace_stockage;
218 const Champ_base& source = get_source(i).get_champ_without_evaluation(source_espace_stockage);
219
220 if ((Motcle(methode_) == "vecteur"))
221 {
222
223 if (source.nature_du_champ()!=scalaire)
224 {
225 Cerr<<que_suis_je()<<" The source fields must be of scalar nature for option vecteur."<<finl;
226 exit();
227 }
228 }
229 int nb_comp = source.nb_comp();
230 nb_comp_ = (nb_comp_<nb_comp)? nb_comp:nb_comp_;
231
232 if (source.nature_du_champ()==vectoriel)
233 nature_ch = vectoriel;
234 else if (source.nature_du_champ()==multi_scalaire)
235 nature_ch = multi_scalaire;
236 }
237 }
238// Si aucun champ source n a ete specifie on en ajoute une
239// pour donner acces a la zone discretisee ...
240
241 Noms sources_location;
242
243 if( nb_source_fictive > 0 )
244 {
245// special treatment when we have fictives sources
246// the user needs to provide a location
247 if( localisation_ == "??" )
248 {
249// On prend la localisation de l'inconnue
250
251
252 Cerr << "Error in Champ_Generique_Transformation::completer "<<finl;
253 Cerr << "No sources were specified. So you need to specify at least the location."<<finl;
254 Cerr << "Aborting..."<<finl;
255 Cerr << "ToDo for developer: guess location by taking the pb.equation(0).inconnue() localization." << finl;
256 Process::abort( );
257 }
258 else
259 {
260 fictive_source = true;
261 }
262 }
263 else
264 {
265
266// real sources
267 for( int i=0; i<nb_sources; i++)
268 {
269// loop over sources to check that all sources have the same location
270 bool use_directive = false;
271 Champ my_field;
272
273 const Champ_base& source_i = get_source( i ).get_champ_without_evaluation( my_field );
274 if( source_i.a_une_zone_dis_base( ) )
275 {
276 const Zone_dis_base& zone_source_i = source_i.zone_dis_base( );
277 if( ! sub_type( Zone_VF , zone_source_i ) )
278 {
279 Cerr << "Warning in Champ_Generique_Transformation::completer "<<finl;
280 Cerr << "The source number "<<i<<" has a zone_dis_base but not of type Zone_VF so use directive to determine the location"<<finl;
281 use_directive=true;
282 }
283 if( ! use_directive )
284 {
285 const DoubleTab& values_source_i = source_i.valeurs( );
286 const Zone_VF& zvf_source_i = ref_cast( Zone_VF, zone_source_i );
287
288
289 MD_Vector md;
290 md = values_source_i.get_md_vector( );
291
292// composite case, in particular Champ_{P0,Face}_PolyMAC...
293 if (zvf_source_i.que_suis_je().debute_par("Zone_PolyMAC") && sub_type( MD_Vector_composite, md.valeur( )))
294 {
295 const MD_Vector& md0 = ref_cast(MD_Vector_composite, md.valeur()).get_desc_part(0);
296 if (md0 == zvf_source_i.zone( ).les_elems().get_md_vector( ))
297 sources_location.add( "elem" );
298 else if (md0 == zvf_source_i.face_sommets( ).get_md_vector( ))
299 sources_location.add("faces");
300 }
301 else if( ( sub_type( MD_Vector_composite, md.valeur( ) ) ) && (localisation_=="??"))
302 {
303 Cerr << "Error in Champ_Generique_Transformation::completer "<<finl;
304 Cerr << "The source number "<<i<<" is composite and the location was not provided. It is forbidden to apply a 'transformation' on a composite source without providing a location. "<<finl;
305 Cerr << "You must perform an interpolation before or specify a location."<<finl;
306 Cerr << "Aborting..."<<finl;
307 Process::abort( );
308// if we want to deal with composite case we can do something like...
309// MD_Vector_composite md_comp = ref_cast( MD_Vector_composite, values_source_i.get_md_vector( ).valeur( ) );
310// nb_parts = md_comp.nb_parts( ); //to perform loop over parts and then get the considered part :
311// md = md_comp.get_desc_part( part );
312// when dealing with pression_pa it leads to have elem and som locations
313 }
314 else if ( md == zvf_source_i.face_sommets( ).get_md_vector( ) )
315 {
316 sources_location.add( "faces" );
317 }
318 else if( md == zvf_source_i.zone( ).les_elems( ).get_md_vector( ) )
319 {
320 sources_location.add( "elem" );
321 }
322 else if( md == zvf_source_i.zone( ).domaine( ).les_sommets( ).get_md_vector( ) )
323 {
324 sources_location.add( "som" );
325 }
326 else if( md == zvf_source_i.xa( ).get_md_vector( ) )
327 {
328 sources_location.add( "xa" );
329 }
330 else
331 {
332 Cerr << "Warning in Champ_Generique_Transformation::completer "<<finl;
333 Cerr << "Impossible to determine the localisation of the source number "<<i<<" using md_vectors so try using directive"<<finl;
334 use_directive = true;
335 }
336// Cout << "sources location "<<i<<" : "<<sources_location( i )<<finl;
337 }//end of not using directive case
338 }//end of has zone dis
339 else
340 {
341// in the case where there is no zone dis base, we look at the directive
342 use_directive=true;
343 }
344
345 if( use_directive )
346 {
347 Motcle directive = get_source( i ).get_directive_pour_discr( );
348 if (directive=="champ_elem")
349 sources_location.add( "elem" );
350 else if (directive=="champ_sommets")
351 sources_location.add( "som" );
352 else if (directive=="champ_face")
353 sources_location.add( "faces" );
354 else if (directive=="pression")
355 {
356 if (localisation_=="??")
357 {
358 Cerr << "Error in Champ_Generique_Transformation::completer"<<finl;
359 Cerr << "The directive associated to the source number "<<i<<" is pressure and the location was not provided.";
360 Cerr <<" It is forbidden to apply a 'transformation' on a composite source without providing a location. "<<finl;
361 Cerr << "You must perform an interpolation before or specify a location." <<finl;
362 Cerr << "Aborting..."<<finl;
363 Process::abort( );
364 }
365 sources_location.add( "unknown" );
366 }
367 else if ( directive == "champ_uniforme" )
368 sources_location.add( "elem" );
369 else
370 {
371 Cerr<<"Error in Champ_Generique_Transformation::completer"<<finl;
372 Cerr<<"The discretization of the first source of the generic field "<<nom_post_<<finl;
373 Cerr<<"does not correspond to any TRUST field discretization."<<finl;
374 Cerr<<directive<<finl;
375 Cerr<<"Aborting..."<<finl;
376 Process::abort( );
377 }
378 }
379 }//loop over sources
380 const int nb_locations = sources_location.size( );
381 if( nb_sources != nb_locations )
382 {
383 Cerr << "Error in Champ_Generique_Transformation::completer "<<finl;
384 Cerr << nb_sources<<" were detected but "<<nb_locations<<" were found"<<finl;
385 Cerr << "Aborting..."<<finl;
386 Process::abort( );
387 }
388 Nom reference_location = sources_location( 0 );
389 for(int i=1; i<nb_locations; i++)
390 {
391 if( sources_location( i ) != reference_location )
392 {
393 sources_location_ok = false;
394 Cerr << "Warning in Champ_Generique_Transformation::completer "<<finl;
395 Cerr << "All sources have location : "<<finl;
396 Cerr <<sources_location;
397 Cerr <<"But the specified location is "<<localisation_<<finl;
398// Cerr << "Aborting..."<<finl;
399// Process::abort( );
400 }
401 }
402 } //end dealing with nb_source_fictive <= 0
403
404 if( localisation_ == "??" )
405 {
406 if( sources_location_ok )
407 {
408 Cerr <<"sources have same location : "<< sources_location( 0 )<<finl;
409 }
410 else
411 {
412 Cerr << "Warning sources have not the same location, taking "<<sources_location( 0 )<<" as location"<<finl;
413 }
414 localisation_ = sources_location( 0 ) ;
415 }
416 else
417 {
418 if( ! fictive_source && localisation_ != sources_location( 0 ) )
419 {
420 Cerr << "Warning first source is located to "<<sources_location( 0 )<<" but the user has specified "<< localisation_<<finl;
421 }
422 }
423
424 verifier_localisation();
425 preparer_macro();
426
427 if (Motcle(methode_)=="formule")
428 {
429// on redemande le nb_sources car on a ajoute des sources par completer
430 nb_sources = get_nb_sources();
431 fxyz.dimensionner(1);
432 fxyz[0].setNbVar(4+nb_sources);
433 fxyz[0].setString(les_fct[0]);
434 fxyz[0].addVar("x");
435 fxyz[0].addVar("y");
436 fxyz[0].addVar("z");
437 fxyz[0].addVar("t");
438 for (int i=0; i<nb_sources; i++)
439 {
440 const Noms nom = get_source(i).get_property("nom");
441 fxyz[0].addVar(nom[0]);
442 Journal()<<" Name of source "<<nom[0]<<finl;
443 }
444 fxyz[0].parseString();
445 }
446}

449void projette(DoubleTab& valeurs_espace,const DoubleTab& val_source,const Zone_VF& zvf,const Motcle& loc,bool champ_normal_faces)
450{
451 const Nom& type_elem=zvf.zone().type_elem().valeur().que_suis_je();
452 if ((champ_normal_faces)||(loc!="elem")|| (type_elem!="Quadrangle"))
453 {
454 Cerr<<"option composante_normale not coded in this case"<<finl;
455 Process::exit();
456 }
457 const DoubleTab& coord=zvf.zone().domaine().coord_sommets();
458 const IntTab& elems=zvf.zone().les_elems();
459
460 int nb_v= valeurs_espace.dimension(0);
461 assert(nb_v==zvf.nb_elem());
462 assert(val_source.dimension(1)==Objet_U::dimension);
463 for (int i=0; i<nb_v; i++)
464 {
465 Vecteur3 v1,v2,normal,val;
466 int s0=elems(i,0);
467 int s1=elems(i,1);
468 int s2=elems(i,2);
469 if (Objet_U::dimension!=3)
470 {
471 Cerr<<"not implemented "<<finl;
472 assert(0);
473 Process::exit();
474 }
475 v1.set(coord(s1,0)-coord(s0,0),coord(s1,1)-coord(s0,1),coord(s1,2)-coord(s0,2));
476 v2.set(coord(s2,0)-coord(s0,0),coord(s2,1)-coord(s0,1),coord(s2,2)-coord(s0,2));
477 val.set(val_source(i,0),val_source(i,1),val_source(i,2));
478
479 Vecteur3::produit_vectoriel(v1,v2,normal);
480 double norm=Vecteur3::produit_scalaire(normal,normal);
481 double inv=1./sqrt(norm);
482 normal*=inv;
483
484
485 double prod_scal= Vecteur3::produit_scalaire(normal,val);
486 valeurs_espace(i)=prod_scal;
487 }
488
489}

503const Champ_base& Champ_Generique_Transformation::get_champ(Champ& espace_stockage) const
504{
505 const Zone_dis_base& zone_dis = get_ref_zone_dis_base();
506 Champ_Fonc es_tmp;
507
508 espace_stockage = creer_espace_stockage(nature_ch,nb_comp_,es_tmp);
509 DoubleTab& valeurs_espace = espace_stockage->valeurs();
510// Construction du tableau positions qui contient les coordonnees
511// des points (support) ou vont etre evaluees les valeurs de l espace
512// de stockage
513 const Zone_VF& zvf = ref_cast(Zone_VF,zone_dis);
514 DoubleTab positions;
515 int nb_pos;
516 int nb_elem_tot = zvf.nb_elem_tot();
517 int nb_som_tot = get_ref_domain().nb_som_tot();
518 const Motcle directive = get_directive_pour_discr();
519 bool champ_normal_faces = 0;
520 if (zvf.que_suis_je().debute_par("Zone_VDF") || zvf.que_suis_je().debute_par("Zone_PolyMAC"))
521 champ_normal_faces = 1;
522 if (localisation_ == "elem")
523 {
524 if (zvf.xp().nb_dim() != 2) /* xp() non initialise */
525 {
526 zvf.zone().calculer_centres_gravite(positions);
527 }
528 else
529 {
530 positions.resize(zvf.nb_elem(), zvf.xp().dimension(1));
531 memcpy(positions.addr(), zvf.xp().addr(), positions.size() * sizeof(double));
532 }
533 }
534 else if (localisation_ == "som")
535 positions = get_ref_domain().coord_sommets();
536 else if (localisation_ == "faces")
537 positions = zvf.xv();
538 else if (localisation_ == "elem_som")
539 {
540 const DoubleTab& som = get_ref_domain().coord_sommets();
541 int dim = som.dimension(1);
542 positions = zvf.xp();
543 positions.resize(nb_elem_tot+nb_som_tot,dim);
544 for (int i1=nb_elem_tot; i1<nb_elem_tot+nb_som_tot; i1++)
545 for (int i2=0; i2<dim; i2++)
546 positions(i1,i2) = som(i1-nb_elem_tot,i2);
547 }
548 else
549 {
550 Cerr<<"Problem in " <<que_suis_je()<<finl;
551 Cerr<<"Item "<<localisation_<<" is not recognized."<<finl;
552 exit();
553 }
554
555// Interpolation des valeurs des sources sur le support retenu et stockage dans sources_val
556
557 int nb_sources = get_nb_sources();
558 nb_pos = positions.dimension(0);
559 assert(nb_pos>0||(methode_=="formule")||(methode_=="composante_normale")||(methode_=="vecteur"));
560 double x=0;
561 double y=0;
562 double z=0;
563 DoubleTabs sources_val(nb_sources);
564 IntVect nb_comps(nb_sources);
565 Noms nom_source(nb_sources);
566 int dim_compo = 2*dimension;
567 Noms compo(dim_compo);
568 double temps;
569 temps = get_time();
570 int nb_compso = 0; //!< deplacement du parametre pour qu'il soit reconnu par l'ensemble des methodes en vue test (voir methode composante ci-dessous)
571 for (int so=0; so<nb_sources; so++)
572 {
573 const Champ_Generique_base& source = get_source(so);
574 Champ stockage_so;
575 const Champ_base& source_so = source.get_champ(stockage_so);
576 const DoubleTab& source_so_val = source_so.valeurs();
577 const Motcle directive_so = source.get_directive_pour_discr();
578
579// int nb_compso = source_so.nb_comp();
580 nb_compso = source_so.nb_comp();
581 nb_comps(so) = nb_compso;
582 const Noms nom_champ = source.get_property("nom");
583 nom_source[so] = nom_champ[0];
584 if ((Motcle(methode_)=="produit_scalaire") || (Motcle(methode_)=="norme"))
585 {
586 const Noms compo_ch = source.get_property("composantes");
587 for (int comp=0; comp<dimension; comp++)
588 compo[so*dimension+comp] = compo_ch[comp];
589 }
590
591// Si VDF et traitement d un produit scalaire ou d une norme alors on interpole
592// pour recuperer un tableau avec autant de composantes que la dimension du probleme
593
594 if (directive!=directive_so)
595 {
596 sources_val[so].resize(nb_pos,nb_compso);
597 if (directive == "CHAMP_FACE") source_so.valeur_aux_faces(sources_val[so]);
598 else source_so.valeur_aux(positions,sources_val[so]);
599 }
600 else
601 {
602 if (source_so_val.line_size()==1)
603 {
604 int nn=source_so_val.dimension_tot(0);
605 sources_val[so].resize(nn,1);
606 for (int i_loc=0; i_loc<nn; i_loc++)
607 sources_val[so](i_loc,0) = source_so_val(i_loc);
608 }
609 else
610 sources_val[so] = source_so_val;
611 }
612
613 if ((nb_pos==0)&&(valeurs_espace.dimension_tot(0)!=0))
614 {
615 int nb_pos_tmp=valeurs_espace.dimension(0);
616 if (nb_pos_tmp > sources_val[so].size())
617 {
618 Cerr << "\nError in Champ_Generique_Transformation::get_champ" << finl;
619 Cerr << nom_source(so) << " : Wrong number of elements." << finl;
620 exit();
621 }
622 }
623 }
624// Fin du remplissage de sources_val
625
626// Calcul des valeurs de l espace de stockage en fonction de methode_ selectionne
627 if ((Motcle(methode_)=="produit_scalaire") || (Motcle(methode_)=="norme"))
628 {
629 Parser_U& f = fxyz[0];
630 for (int i=0; i<nb_pos; i++)
631 {
632 for (int so=0; so<nb_sources; so++)
633 {
634 const DoubleTab& source_so_val = sources_val[so];
635 for (int j=0; j<dimension; j++)
636 f.setVar(so*dimension+j,source_so_val(i,j));
637// Optimisation plus rapide que:
638// f.setVar(compo[so*dimension+j],source_so_val(i,j));
639 }
640 valeurs_espace(i) = f.eval();
641 }
642 }
643 else if (Motcle(methode_)=="vecteur")
644 {
645 for (int j=0; j<nb_comp_; j++)
646 fxyz[j].setVar("t",temps);
647
648 if ((champ_normal_faces) && (localisation_=="faces"))
649 {
650 for (int i=0; i<nb_pos; i++)
651 {
652 x = positions(i,0);
653 y = positions(i,1);
654 z = 0;
655 if (dimension>2)
656 z = positions(i,2);
657 valeurs_espace(i) = 0;
658 for (int j=0; j<nb_comp_; j++)
659 {
660 fxyz[j].setVar(0,x);
661 fxyz[j].setVar(1,y);
662 fxyz[j].setVar(2,z);
663 for (int so=0; so<nb_sources; so++)
664 {
665 const DoubleTab& source_so_val = sources_val[so];
666 fxyz[j].setVar(so+4,source_so_val(i,0));
667 }
668 valeurs_espace(i) += fxyz[j].eval() * zvf.face_normales(i, j) / zvf.surface(i);
669 }
670 }
671 }
672 else
673 {
674 for (int i=0; i<nb_pos; i++)
675 {
676 x = positions(i,0);
677 y = positions(i,1);
678 z = 0;
679 if (dimension>2)
680 z = positions(i,2);
681
682 for (int j=0; j<nb_comp_; j++)
683 {
684 fxyz[j].setVar(0,x);
685 fxyz[j].setVar(1,y);
686 fxyz[j].setVar(2,z);
687 for (int so=0; so<nb_sources; so++)
688 {
689 const DoubleTab& source_so_val = sources_val[so];
690 fxyz[j].setVar(so+4,source_so_val(i,0));
691 }
692 valeurs_espace(i,j) = fxyz[j].eval();
693 }
694 }
695 }
696 }
697 else if (Motcle(methode_)=="composante_normale")
698 {
699 const DoubleTab& source_so_val = sources_val[0];
700 projette(valeurs_espace,source_so_val,zvf,localisation_,champ_normal_faces);
701 }
702 else if (Motcle(methode_)=="composante")
703 {
704 int num_compo = les_fct.size()-1;
705 if ((num_compo<0)||(num_compo>=nb_compso))
706 {
707 Cerr<<"Component number must be >=0 and < "<<nb_compso<<" for composante option."<<finl;
708 exit();
709 }
710
711 const DoubleTab& source_so_val = sources_val[0];
712 if ((champ_normal_faces) && (localisation_=="faces"))
713 {
714 for (int i=0; i<nb_pos; i++)
715 valeurs_espace(i) = source_so_val(i,0) * zvf.face_normales(i, num_compo) / zvf.surface(i);
716 }
717 else
718 {
719 for (int i=0; i<nb_pos; i++)
720 valeurs_espace(i) = source_so_val(i,num_compo);
721 }
722 }
723 else if (Motcle(methode_)=="formule")
724 {
725// Evaluation des valeurs de l espace de stockage en fonction de la fonction (les_fct)
726 Parser_U& f = fxyz[0];
727 f.setVar("t",temps);
728 int special=0;
729 if ((nb_pos==0)&&(valeurs_espace.dimension_tot(0)!=0))
730 {
731 special=1;
732 nb_pos=valeurs_espace.dimension(0);
733 x=1e38;
734 y=1e38;
735 z=1e38;
736 f.setVar(0,x);
737 f.setVar(1,y);
738 f.setVar(2,z);
739 }
740
741 for (int i=0; i<nb_pos; i++)
742 {
743 if (!special)
744 {
745 x = positions(i,0);
746 y = positions(i,1);
747 z = 0;
748 if (dimension>2)
749 z = positions(i,2);
750 f.setVar(0,x);
751 f.setVar(1,y);
752 f.setVar(2,z);
753 }
754 /* Plus rapide que de passer par:
755 f.setVar("x",x);
756 f.setVar("y",y);
757 f.setVar("z",z);
758 */
759 if (valeurs_espace.line_size() == 1)
760 {
761 for (int so=0; so<nb_sources; so++)
762 f.setVar(so+4,sources_val[so](i,0));
763 valeurs_espace(i) = f.eval();
764 }
765 else //!< line_size > 1
766 {
767 for (int j=0; j<nb_comp_; j++)
768 {
769 for (int so=0; so<nb_sources; so++)
770 {
771 int nbcomp_loc = nb_comps(so);
772 if (nbcomp_loc == nb_comp_)
773 f.setVar(so+4,sources_val[so](i,j));
774 else if (nbcomp_loc == 1)
775 f.setVar(so+4,sources_val[so](i,0));
776 else
777 {
778 Cerr << "The arrays of values don't have compatibles dimensions in "<<que_suis_je()<< finl;
779 exit();
780 }
781 }
782 valeurs_espace(i,j) = f.eval();
783 }
784 }
785 }
786 }

788// Voir Champ_Generique_Interpolation (localisation = som) pour le report de l'echange_espace_virtuel
789// valeurs_espace.echange_espace_virtuel();
