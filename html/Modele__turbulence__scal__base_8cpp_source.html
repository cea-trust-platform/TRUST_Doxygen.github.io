378// Contraintes:
379// Acces: NON ACCEDE

381// Effets de bord:
382// Postcondition:
383//
384//! Indique s'il faut imprimer ou non le Nusselt local
385/** \return int<br>
386Signification: renvoie 1 si on imprime, 0 sinon<br>
387Contraintes: */
388int Modele_turbulence_scal_base::limpr_nusselt(double temps_courant, double temps_prec, double dt) const
389{
390 const Schema_Temps_base& sch = mon_equation->schema_temps();
391 if (sch.nb_pas_dt()==0)
392 return 0;
393 if (dt_impr_nusselt_<=dt || ((sch.temps_final_atteint() || sch.nb_pas_dt_max_atteint() || sch.nb_pas_dt()==1 || sch.stationnaire_atteint()) && dt_impr_nusselt_!=DMAXFLOAT))
394 return 1;
395 else
396 {
397// Voir Schema_Temps_base::limpr pour information sur epsilon et modf
398 double i, j, epsilon = 1.e-8;
399 modf(temps_courant/dt_impr_nusselt_ + epsilon, &i);
400 modf(temps_prec/dt_impr_nusselt_ + epsilon, &j);
401 return ( i>j );
402 }
403}
