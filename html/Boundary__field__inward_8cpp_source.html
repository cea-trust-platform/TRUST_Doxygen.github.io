37//
38//! Initialize normal inward field given on boundaries
39
40// Remark :
41// This code is not in "Champ_front_normal::initialiser" function because in the case of using
42// Champ_front_debit, the readOn function of Champ_front_normal is not called
43// so vit_norm is not initialized and it can leads to critical values detected
44// in debug mode.
45// Moreover, when using Champ_front_debit we don't need to do all these calculations
46// That is why Boundary_field_inward was created
47int Boundary_field_inward::initialiser(double tps, const Champ_Inc_base& inco)
48{
49 if( !Champ_front_normal::initialiser(tps,inco) )
50 return 0;
51 mettre_a_jour(tps);
52 return 1;
53}

55void Boundary_field_inward::mettre_a_jour(double tps)
56{
57 const Zone_VF& zone_VF = ref_cast(Zone_VF,zone_dis());
58 const Front_VF& le_bord= ref_cast(Front_VF,frontiere_dis());
59 int first=le_bord.num_premiere_face();
60
61 int nb_cases=les_valeurs->nb_cases();
62
63 DoubleTab tab_normal_vector(le_bord.nb_faces(),dimension);
64
65// Evaluation de vit_norm:
66 vit_norm.setVar("t",tps);
67 double value = vit_norm.eval();
68
69 for(int i=0; i<le_bord.nb_faces(); i++)
70 {
71
72 int signe = -1;//we want the opposite of the normal vector
73 int global_boundary_face_number=i+first;
74// outward vector associated to the fac global_boundary_face_number
75 DoubleTab normal_vector = zone_VF.normalized_boundaries_outward_vector(global_boundary_face_number,signe);
76
77 for (int k=0; k<dimension; k++)
78 {
79 tab_normal_vector(i,k) = normal_vector(k);
80 }
81
82// Fill all moments
83 for (int t=0; t<nb_cases; t++)
84 {
85 DoubleTab& tab=les_valeurs[t].valeurs();
86 for(int j=0; j<dimension; j++)
87 {
88 tab(i,j)=tab_normal_vector(i,j);
89 tab(i,j)*=value;
90 }
91 }
92 }
93}

103Entree& Champ_front_normal_VEF::readOn(Entree& is)
104{
105 Motcle motlu;
106 Motcles les_mots(1);
107 les_mots[0]="valeur_normale";
108 is>>motlu;
109 if (motlu != les_mots[0])
110 {
111 Cerr << "Error while reading a Champ_front_normal" << finl;
112 Cerr << "We expected " << les_mots[0] << " instead of " << motlu << finl;
113 exit();
114 }
115 Nom debit;
116 is >> debit;
117 Nom exp("{ normal_value ");
118 exp+=debit;
119 exp+=" } ";
120 EChaine E(exp);
121 return Boundary_field_inward::readOn(E);
122 return is;
123}
