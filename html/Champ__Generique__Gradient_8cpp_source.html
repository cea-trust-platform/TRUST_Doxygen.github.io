36void Champ_Generique_Gradient::completer(const Postraitement_base& post)
37{
38 Champ_Gen_de_Champs_Gen::completer(post);
39 const Probleme_base& Pb = get_ref_pb_base();
40 if (sub_type(Champ_Generique_refChamp,get_source(0)))
41 {
42
43 Champ espace_stockage;
44 const Champ_base& mon_champ = get_source(0).get_champ(espace_stockage);
45 if (sub_type(Champ_Inc_base,mon_champ))
46 {
47 const Champ_Inc_base& mon_champ_inc = ref_cast(Champ_Inc_base,mon_champ);
48
49 if ( (mon_champ_inc.le_nom()=="pression") || ((mon_champ_inc.le_nom()!="vitesse") && (mon_champ_inc.que_suis_je()=="Champ_P1NC"))
50 || ((mon_champ_inc.le_nom()!="vitesse") && (mon_champ_inc.que_suis_je()=="Champ_P0_VDF")) )
51 {
52
53// On recupere l equation alors qu elle n est pas encore associee au Champ_Inc
54// On parcours les equatiosn du probleme et on identifie celle qui correspond au champ inconnu
55
56 int nb_eq = Pb.nombre_d_equations();
57 int i=0;
58
59 while (i<nb_eq)
60 {
61 const Equation_base& eq_test = Pb.equation(i);
62 if ((eq_test.inconnue().le_nom() == mon_champ_inc.le_nom()) || ((i==0) && (mon_champ_inc.le_nom()=="pression") &&
63 (eq_test.le_nom()!="Conduction")))
64 break;
65 i++;
66 }
67 const Equation_base& eqn = Pb.equation(i);
68
69 Op_Grad_.associer_eqn(eqn);
70 Op_Grad_.typer();
71 Op_Grad_.l_op_base().associer_eqn(eqn);
72
73 const Zone_dis& zdis = eqn.zone_dis();
74 const Zone_Cl_dis& zcl = eqn.zone_Cl_dis();
75 const Champ_Inc& inco = eqn.inconnue();
76 Op_Grad_->associer(zdis, zcl, inco);
77 }
78 else
79 {
80 Cerr<<"We can apply a Champ_Generique_Gradient only to the pressure field"<<finl;
81 Cerr<<"or to a field of type Champ_P1NC or Champ_P0_VDF at one component"<<finl;
82 exit();
83 }
84 }
85 else
86 {
87 Cerr<<"We can apply a Champ_Generique_Gradient only to an unknown field of the problem"<<finl;
88 exit();
89 }
90 }

111const Champ_base& Champ_Generique_Gradient::get_champ(Champ& espace_stockage) const
112{
113 Champ source_espace_stockage;
114 const Champ_base& source = get_source(0).get_champ(source_espace_stockage);
115
116 if (Op_Grad_.non_nul())
117 {
118 Champ_Fonc es_tmp;
119 espace_stockage = creer_espace_stockage(vectoriel,dimension,es_tmp);
120
121 Op_Grad_.calculer(source.valeurs(),espace_stockage.valeurs());
122
123 Nom type_op = Op_Grad_.valeur().que_suis_je();
124 if ((type_op!="Op_Grad_P1NC_to_P0") && (type_op!="Op_Grad_P0_to_Face"))
125 Op_Grad_.equation().solv_masse().appliquer(espace_stockage.valeurs());
126 }
127 else
128 {
129 const Noms nom = get_source(0).get_property("nom");
130 Cerr<<"We can apply a Champ_Generique_Gradient only to the pressure field"<<finl;
131 Cerr<<"or to a field of type Champ_P1NC or Champ_P0_VDF at one component"<<finl;
132 Cerr<<"The space of storage of field treated here "<<nom[0]<<" is of type"<<source.que_suis_je()<<finl;
133 exit();
134 }
135
136 DoubleTab& espace_valeurs = espace_stockage->valeurs();
137 espace_valeurs.echange_espace_virtuel();
138 return espace_stockage.valeur();
139}

213Entity Champ_Generique_Gradient::get_localisation(const int index) const
214{
215 Entity loc;
216 Nom type_op = Op_Grad_.valeur().que_suis_je();
217 if ((type_op=="Op_Grad_P1NC_to_P0") && (index <= 0))
218 loc = ELEMENT;
219 else if ((type_op=="Op_Grad_P0_to_Face") && (index <= 0))
220 loc=FACE;
221 else
222 {
223 Cerr << "Error of type : Gradient operator " << type_op << " is not possible for the defined field " << nom_post_ << finl;
224 throw Champ_Generique_erreur("INVALID");
225 }
226 return loc;
227}
