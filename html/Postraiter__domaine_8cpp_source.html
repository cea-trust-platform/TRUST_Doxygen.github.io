61void traite_bord(const Zone& zone,IntVect& ch_som,IntVect& ch_elem3,int num2, Faces& faces,Nom& nom_bord,Nom& fichier,const IntTab& les_elems, Format_Post_base& post, int& compteur,int& compteur0,int& compteur_reel,int& moi,int isjoint=0)
62{
63 const Domaine& dom=zone.domaine();
64 Nom nom_dom=dom.le_nom();
65 int num=-num2;
66 int nb_elem=les_elems.dimension(0);
67 int nb_elem_tot=zone.nb_elem_tot();
68
69
70 int nb_faces=faces.nb_faces();
71 int nb_som_faces=faces.nb_som_faces();
72 int j,k,som;
73
74 DoubleTab ch_elem2(nb_elem,1);
75 ch_elem2=moi;
76
77 if (isjoint!=2)
78 {
79// on a pas encore calcule ch_elem3
80 ch_elem3=moi;
81 for (j=0; j<nb_faces; j++)
82 for (k=0; k<nb_som_faces; k++)
83 {
84 som=faces.sommet(j,k);
85// cas des polyedres
86 if (som==-1) break;
87 if (ch_som(som)==0)
88 ch_som(som)=num;
89 else
90 ch_som(som)=(ch_som(som)+num)/2;
91 }
92
93// num=-Process::nproc();
94 num=-1;
95 if (nb_faces!=0)
96 {
97 if (0)
98 {
99 int taille1=les_elems.dimension(1);
100 for (int e=0 ; e < nb_elem ; e++)
101 {
102 for (j=0; j<nb_faces; j++)
103 {
104 int m=-1;
105 for (k=0; k<nb_som_faces; k++)
106 {
107 som=faces.sommet(j,k);
108 for (int kk=0; kk<taille1; kk++)
109 {
110 if (les_elems(e,kk)==som)
111 {
112 m++;;
113 break;
114 }
115 }
116 if (m!=k) break;
117 }
118 if (m==nb_som_faces-1)
119 ch_elem3[e]=num;
120 }
121 }
122 }
123 else
124 {
125 const IntTab& faces_voisin=faces.voisins();
126 if (faces_voisin.dimension(0) > 0 && local_max_vect(faces_voisin)!=-1)
127 {
128// Cerr<<" we use faces_voisins"<<finl;
129 for (j=0; j<nb_faces; j++)
130 {
131 int el1=faces_voisin(j,0);
132 if (el1==-1) el1=faces_voisin(j,1);
133 if (el1==-1)
134 {
135 Cerr << "Error, the face " << j << " of the domain " << dom.le_nom() << " has no neighbours !" << finl;
136 Cerr << "Check that the domain " << dom.le_nom() << " has been discretized." << finl;
137 Process::exit();
138 }
139 ch_elem3[el1]=num;
140 }
141 }
142 else
143 {
144 DoubleTab position(1,Objet_U::dimension);
145 IntVect res;
146 DoubleTab xv;
147 faces.associer_zone(zone);
148 faces.calculer_centres_gravite(xv);
149 for (j=0; j<nb_faces; j++)
150 {
151 for (int dir=0; dir<Objet_U::dimension; dir++)
152 position(0,dir)=xv(j,dir);
153 zone.chercher_elements(position,res);
154 if (res[0]<0)
155 {
156 position+=1e-5;
157 zone.chercher_elements(position,res);
158 if (res[0]<0)
159 {
160 position-=2e-5;
161 zone.chercher_elements(position,res);
162 }
163 }
164
165 if (res[0]<nb_elem)
166 {
167 if (res[0]<0)
168 {
169 if (isjoint==1)
170 Cerr<<"To properly view the joints, the splitting area must be complete "<<finl;
171 else
172 {
173 Cerr<<"Problem in Postraiter_domaine.cpp with the boundary "<<nom_bord<<finl;
174// abort();
175 }
176 }
177 else
178 {
179 ch_elem3[res[0]]=num;
180 }
181 }
182 else
183 {
184 if (nb_elem>=nb_elem_tot) abort();
185 }
186 }
187 }
188 }
189 }
190 }
191
192 for (int s=0; s<nb_elem; s++) ch_elem2(s,0)=(ch_elem3[s]);
193// on ajoute ce bord
194 Noms noms_post(1);
195 noms_post[0]=nom_bord;
196 Noms unites(1);
197 unites[0]="1";
198// Cerr<<Process::me() <<" ch_elem2 size"<<fichier<< " "<<nom_dom<<" " <<nom_bord<<" "<<ch_elem2.dimension(0)<<finl;
199 post.ecrire_champ(dom,unites,noms_post,-1,0.,0.,nom_bord, nom_dom, "ELEM","scalar",ch_elem2);
200}

215Entree& Postraiter_domaine::interpreter_(Entree& is)
216{
217 Nom nom_pdb;
218 Nom un_dom;
219 Param param(que_suis_je());
220 param.ajouter("format",&format_post_,Param::REQUIRED);
221 format_binaire_=1;
222 param.ajouter("binaire",&format_binaire_);
223 ecrire_frontiere_=1;
224 param.ajouter("ecrire_frontiere",&ecrire_frontiere_);
225 nom_pdb="NOM_DU_CAS";
226 param.ajouter("fichier|file",&nom_pdb);
227 joint_non_ecrit_=1;
228 param.ajouter("joints_non_postraites",&joint_non_ecrit_); //!< desactive l'ecriture des joints pratique pour comparer parallele et sequentielle
229 param.ajouter("domaine",&un_dom);
230 param.ajouter_non_std("domaines",(this));
231 param.ajouter_condition("is_read_domaine_or_is_read_domaines","Vous devez preciser domaine ou domaines dans Postraiter_domaine");
232 param.ajouter_condition("is_read_domaine_or_is_read_domaines","Interpreter Postraiter_domaine : one of the keywords domaine or domaines must be specified.");
233 param.lire_avec_accolades_depuis(is);
234 if (un_dom!=Nom())
235 {
236 associer_domaine(un_dom);
237 }
238 nb_domaine_=domaines().size();
239 if(nb_domaine_==0)
240 {
241 Cerr << "You must specify one or several domains" << finl;
242 exit();
243 }
244 if (nom_pdb=="NOM_DU_CAS")
245 {
246 nom_pdb=nom_du_cas();
247 nom_pdb+=".";
248 nom_pdb+=format_post_;
249 }
250 if (format_post_=="lata")
251 format_post_+="_V2";
252
253// On deplace la boucle sur les domaines lus
254 ecrire(nom_pdb);
255 return is;

284void Postraiter_domaine::ecrire(Nom& nom_pdb)
285{
286 Nom fichier=nom_pdb.nom_me(Process::me());
287 int compteur=-123;
288 int compteur0=-123;
289 int compteur_reel=0;
290 Nom type("Format_Post_");
291 type+=format_post_;
292 Format_Post post_typer;
293 post_typer.typer_direct(type);
294 Format_Post_base& post=ref_cast(Format_Post_base,post_typer.valeur());
295
296 Nom nom_2(nom_pdb);
297 Nom format_post_bis(format_post_);
298 if (format_post_bis.debute_par("lata"))
299 format_post_bis="lata";
300 Nom suffix(".");
301 suffix+=format_post_bis;
302 nom_2.prefix(suffix);
303 int est_le_premie_post=0;
304 if (format_post_bis!="lata")
305 post.initialize_by_default(nom_2);
306 else
307 post.initialize(nom_2,format_binaire_,"SIMPLE");
308 for (int numero_domaine=0; numero_domaine<nb_domaine_; numero_domaine++)
309 {
310 if (numero_domaine==0)
311 est_le_premie_post=1;
312 else
313 est_le_premie_post=0;
314 post.ecrire_entete(0.,0,est_le_premie_post);
315
316 const Domaine& dom=domaine(numero_domaine);
317 int reprise = 0;
318 double t_init = 0.;
319 post.preparer_post(dom.le_nom(),est_le_premie_post,reprise,t_init);
320 post.ecrire_domaine(dom, est_le_premie_post);
321 }
322
323 post.ecrire_temps(0.);
324
325 for (int numero_domaine=0; numero_domaine<nb_domaine_; numero_domaine++)
326 {
327 if (numero_domaine==0)
328 est_le_premie_post=1;
329 else
330 est_le_premie_post=0;
331 const Domaine& dom=domaine(numero_domaine);
332
333 compteur=-1;
334 post.init_ecriture(0.,-1.,est_le_premie_post,dom);
335
336 }
337 int moi=Process::me();
338 if (joint_non_ecrit_)
339 moi=0;
340
341 for (int numero_domaine=0; numero_domaine<nb_domaine_; numero_domaine++)
342 {
343 Domaine& dom=domaine(numero_domaine);
344 Nom nom_dom(dom.le_nom());
345 Zone& zone=dom.zone(0);
346 const IntTab& les_elems = zone.les_elems();
347 IntVect ch_som(zone.nb_som());
348 int nb_elem=zone.nb_elem();
349 IntVect ch_elem2(nb_elem);
350 int num=0;
351// En format LATA V2, les frontieres et joints
352// sont ecrits par defaut depuis la 1.6.6 lors
353// de l'ecriture des domaines si le domaine a ete discretise <=> indice_domaines_frontieres>0
354 if (format_post_!="lata_V2" || dom.domaines_frontieres().size()==0)
355 {
356 if (dom.domaines_frontieres().size()==0) Cerr << "Warning: it could be faster and nicer to discretize your domain before using Postraiter_domaine." << finl;
357//
358// Ecriture des frontieres
359//
360 if (ecrire_frontiere_)
361 {
362 int nglob=1;
363 int nb_bords = zone.nb_bords();
364 for (int i=0; i<nb_bords; i++)
365 {
366 num=i+nglob;
367 Nom nom_fr=zone.bord(i).le_nom();
368// le maitre envoit le nom de la frontiere
369// afin que les processeurs traitent les bords dans le meme ordre
370 if (je_suis_maitre())
371 envoyer(nom_fr,0,-1,11);
372 else
373 recevoir(nom_fr,0,me(),11);
374 Nom nom_bord;
375 if (nb_domaine_==1)
376 nom_bord = nom_fr;
377 else
378 nom_bord = zone.domaine().le_nom()+"_"+nom_fr;
379 Faces& faces=zone.frontiere(zone.rang_frontiere(nom_fr)).faces();
380 traite_bord(zone, ch_som,ch_elem2, num, faces, nom_bord, fichier, les_elems, post,compteur,compteur0,compteur_reel,moi);
381 }
382
383 nglob+=zone.nb_bords();
384 int nb_raccords = zone.nb_raccords();
385 for (int i=0; i<nb_raccords; i++)
386 {
387 num=i+nglob;
388 Nom nom_fr=zone.raccord(i).valeur().le_nom();
389 if (je_suis_maitre())
390 envoyer(nom_fr,0,-1,11);
391 else
392 recevoir(nom_fr,0,me(),11);
393 Nom nom_bord;
394 if (nb_domaine_==1)
395 nom_bord = Nom("Raccord_")+nom_fr;
396 else
397 nom_bord = zone.domaine().le_nom()+Nom("_Raccord_")+nom_fr;
398
399 Faces& faces=zone.frontiere(zone.rang_frontiere(nom_fr)).faces();
400 traite_bord(zone, ch_som,ch_elem2, num, faces, nom_bord, fichier, les_elems, post,compteur,compteur0,compteur_reel,moi);
401 }
402
403 nglob+=zone.nb_raccords();
404
405// On ecrit un champ global
406 Noms noms_post(1);
407 if (nb_domaine_==1)
408 noms_post[0] = "Bord";
409 else
410 noms_post[0] = zone.domaine().le_nom()+"_Bord";
411 Noms unites(1);
412 unites[0]="1";
413 int nb_som=zone.nb_som();
414 DoubleTab ch_som2(nb_som,1);
415 for (int j=0; j<nb_som; j++)
416 ch_som2(j,0)=(ch_som(j));
417 post.ecrire_champ(dom,unites,noms_post,-1,0.,0.,noms_post[0], nom_dom, "SOM","scalar",ch_som2);
418 }
419
420//
421// Ecriture des joints
422//
423 if (!joint_non_ecrit_)
424 for (int p=0; p<nproc(); p++)
425 {
426 int nb_joint=zone.nb_joints();
427 envoyer_broadcast(nb_joint, p);
428 for (int i=0; i<nb_joint; i++)
429 {
430 Nom nom_bord;
431 if (me()==p)
432 {
433 if (nb_domaine_==1)
434 nom_bord="Joint_Proc_";
435 else
436 nom_bord=zone.domaine().le_nom()+"Joint_Proc_";
437 Nom namep(p);
438 nom_bord+=namep;
439 nom_bord+="_avec_Proc_";
440 nom_bord+=Nom(zone.joint(i).PEvoisin());
441 }
442// Le processeur p envoie nom_bord a tout le monde
443 envoyer_broadcast(nom_bord, p);
444 Faces toto;
445 Cerr<< me()<<" INFO " << p<<" "<<nom_bord<<finl;
446 Faces& faces=(me()==p?zone.joint(i).faces():toto);
447 barrier();
448 traite_bord(zone, ch_som,ch_elem2, num, faces, nom_bord, fichier, les_elems,post,compteur,compteur0,compteur_reel,moi,1);
449 barrier();
450 }
451
452 }
453 }
454//
455// Ecriture des sous_zones
456//
457 int nb_ss_zones=dom.nb_ss_zones();
458 int nb_elem_tot=zone.nb_elem_tot();
459 for (int i=0; i<nb_ss_zones; i++)
460 {
461 const Sous_Zone& la_ss_zone=dom.ss_zone(i);
462 int nbpoly=la_ss_zone.nb_elem_tot();
463 IntVect ch_elemb2(nb_elem);
464 int cell=0;
465 for (int pol=0; pol<nbpoly; pol++)
466 {
467 int elem=la_ss_zone[pol];
468 if (elem<nb_elem)
469 {
470 ch_elemb2[elem]=(i+1);
471 cell++;
472 }
473 else if (elem>=nb_elem_tot)
474 {
475 Cerr << "This subzone is poorly defined." << finl;
476 Cerr << "It is composed of elements which do not belong" << finl;
477 Cerr << "to the domain!" << finl;
478 Process::exit();
479 }
480 }
481 int subzone_cells=mp_sum(cell);
482 Cerr<<"We handle of the subzone " << i <<" who have " << subzone_cells << " cells." << finl;
483 Faces bidon;
484 Nom nom_post;
485 nom_post="Sous_Zone_";
486 nom_post+=la_ss_zone.le_nom();
487 traite_bord(zone,ch_som,ch_elemb2,1,bidon,nom_post,fichier,les_elems,post,compteur,compteur0,compteur_reel,moi,2);
488 }
489 }
490 int fin=1;
491 post.finir(fin);
492}
