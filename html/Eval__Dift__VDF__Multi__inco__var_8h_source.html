39// Overloaded methods used by the flux computation in template class:
40 inline double nu_1_impl(int i, int compo) const
41 {

55 const double heq_turb= 0.5*(dv_diffusivite_turbulente(i)+dv_diffusivite_turbulente(j))/(d1+d0);
