46double Champ_P1_isoP1Bulle::norme_L2(const Domaine& dom) const
47{
48 const Zone_VEF_PreP1b& zvef = zone_vef();
49 const IntTab& som_elem=zvef.zone().les_elems();
50 const DoubleVect& volumes=zvef.volumes();
51
52 int prs=zvef.numero_premier_sommet();
53 DoubleTab q(valeurs());
54 double norme=0, s=0;
55 double alpha =1./(dimension*(dimension+1));
56 int i, k;
57 double somme=0;
58 int sz=q.size();
59 for(i=prs; i<sz; i++)
60 somme+=q(i);
61 somme/=(sz-prs);
62 for(i=0; i<prs; i++)
63 q(i)+=alpha*somme;
64 for(i=prs; i<sz; i++)
65 q(i)-=somme;
66 for(i=0; i<prs; i++)
67 {
68 double m=0;
69 for(k=0; k<(dimension+1); k++)
70 m+=q(prs+som_elem(i,k));
71 m*=alpha;
72 m+=q(i);
73 s+= m*volumes(i);
74 norme+= volumes(i);
75 }
76 Cerr << "norme L1 : " << s << finl;
77
78 s/=norme;
79 for(i=0; i<prs; i++)
80 q(i)-=s;
81 s=norme=0;
82 for(i=0; i<prs; i++)
83 {
84 double m=0;
85 for(k=0; k<(dimension+1); k++)
86 m+=q(prs+som_elem(i,k));
87 m*=alpha;
88 m+=q(i);
89 m*=m;
90 norme+= volumes(i)*m;
91
92 for(k=0; k<(dimension+1); k++)
93 {
94 double v=0;
95 for(int l=0; l<(dimension+1); l++)
96 {
97 if(k!=l)
98 v+=q(prs+som_elem(i,l));
99 }
100 v*=alpha;
101 v*=v;
102 v/=(dimension+1);
103 norme+= volumes(i)*v;
104 }
105 }
106 return sqrt(norme);
107}

252double Champ_P1_isoP1Bulle::valeur_au_bord(int face) const
253{
254 const DoubleTab& val = valeurs();
255 const Zone_VEF_PreP1b& zone_VEF=la_Zone_VEF_PreP1b.valeur();
256 const IntTab& face_voisins = zone_VEF.face_voisins();
257 const IntTab& som_elem=zone_VEF.zone().les_elems();
258 int nps=zone_VEF.numero_premier_sommet();
259
260 int elem = face_voisins(face,0);
261 if (face_voisins(face,1)!=-1)
262 {
263 Cerr << "Erreur dans Champ_P1_isoP1Bulle::valeur_au_bord." << finl;
264 Cerr << "La face n'est pas une face de bord." << finl;
265 exit();
266 }
267
268 int som_opp=zone_VEF.get_num_fac_loc(face, 0);
269 double val_bord = 0;
270 for(int i=0; i<(dimension+1); i++)
271 if(i!=som_opp)
272 {
273 val_bord+=val[nps+som_elem(elem, i)];
274 }
275 val_bord/=dimension;
276 return val_bord;
277}
