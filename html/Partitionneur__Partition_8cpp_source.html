71void Partitionneur_Partition::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
72{
73 assert(ref_domaine_interpol_.non_nul());
74 assert(ref_domaine_calcul_.non_nul());
75
76// Domaine dom_interpol
77 const Domaine& dom_interpol = ref_domaine_interpol_.valeur();
78 if(dom_interpol.nb_som()==0)
79 {
80 Cerr << "The domain \"" << dom_interpol.le_nom() << "\" to be split is empty !" << finl;
81 exit();
82 }
83 const Zone& zone_interpol = dom_interpol.zone(0);
84 const int nb_elem_interpol = zone_interpol.nb_elem_tot();
85 const int dim_interpol = dom_interpol.dimension;
86
87// Domaine dom_calcul
88 const Domaine dom_calcul = ref_domaine_calcul_.valeur();
89 if(dom_calcul.nb_som()==0)
90 {
91 Cerr << "The reference domain \"" << dom_calcul.le_nom() << "\" is empty !" << finl;
92 exit();
93 }
94 const Zone& zone_calcul = dom_calcul.zone(0);
95 const int nb_elem_calcul = zone_calcul.nb_elem();
96 const int dim_calcul = dom_calcul.dimension;
97
98 IntVect elem_part_calcul;
99 int nb_parts_tot_calcul;
100 EFichier file;
101 Nom nom_fichier;
102 nom_fichier=nom_du_cas()+"_"+dom_calcul.le_nom()+".dec";
103 file.ouvrir(nom_fichier);
104 if (!file.good())
105 {
106 Cerr << "Error in Partitionneur_Partition::construire_partition\n";
107 Cerr << " Failed to open file " << nom_fichier << finl;
108 Cerr << "To use the partitioner Partition, you must check" << finl;
109 Cerr << "that you use the option:" << finl;
110 Cerr << "Ecrire_decoupage " << nom_fichier << finl;
111 Cerr << "In the splitting block of the computational domain " << dom_calcul.le_nom() << finl;
112 exit();
113 }
114 file >> elem_part_calcul;
115 file >> nb_parts_tot_calcul;
116 file.close();
117
118 const int sz = elem_part_calcul.size_array();
119 if (nb_elem_calcul != sz)
120 {
121 Cerr << "Error in Partitionneur_Partition::construire_partition\n";
122 Cerr << " The file " << nom_fichier << " contains an array of " << sz << " values.\n";
123 Cerr << " The area contains " << nb_elem_calcul << " elements" << finl;
124 exit();
125 }
126
127// Centre de gravite des elements de dom_interpol
128 DoubleTab coord_g_interpol;
129// Calcul des centres de gravite des elements de dom_interpol
130 zone_interpol.calculer_centres_gravite(coord_g_interpol);
131 assert(coord_g_interpol.dimension(0) == nb_elem_interpol);
132 assert(coord_g_interpol.dimension(1) == dim_interpol);
133
134// Centre de gravite des elements de dom_calcul
135 DoubleTab coord_g_calcul;
136// Calcul des centres de gravite des elements de dom_calcul
137 zone_calcul.calculer_centres_gravite(coord_g_calcul);
138 assert(coord_g_calcul.dimension(0) == nb_elem_calcul);
139 assert(coord_g_calcul.dimension(1) == dim_calcul);
140
141 elem_part.resize(nb_elem_interpol);
142 elem_part = 0;
143// Recherche des centres de gravite les plus proche les uns des autres
144 {
145 int i,j;
146 double distance_min,distance;
147 for (i = 0; i < nb_elem_interpol; i++)
148 {
149 distance_min=1e+9;
150 double x_interpol = coord_g_interpol(i,0);
151 double y_interpol = coord_g_interpol(i,1);
152 double z_interpol = (dim_interpol == 3) ? coord_g_interpol(i,2) : 0.;
153 for (j = 0; j < nb_elem_calcul; j++)
154 {
155 double x_calcul = coord_g_calcul(j,0);
156 double y_calcul = coord_g_calcul(j,1);
157 double z_calcul = (dim_calcul == 3) ? coord_g_calcul(j,2) : 0.;
158 distance=sqrt((x_calcul-x_interpol)*(x_calcul-x_interpol)+(y_calcul-y_interpol)*(y_calcul-y_interpol)+(z_calcul-z_interpol)*(z_calcul-z_interpol));
159 if (distance < distance_min)
160 {
161 distance_min=distance;
162 elem_part[i]=elem_part_calcul[j];
163 }
164 }
165 }
166 }
167
168 int nb_parties_interpol, nb_parties_calcul;
169 nb_parties_interpol=0;
170 nb_parties_calcul=0;
171// Nombre de proc sur elem_part
172 if (elem_part.size_array() > 0)
173 nb_parties_interpol = max_array(elem_part) + 1;
174// Nombre de proc sur elem_part_calcul
175 if (elem_part_calcul.size_array() > 0)
176 nb_parties_calcul = max_array(elem_part_calcul) + 1;
177// Si partie vide, on les prends en compte
178 if (nb_parties_calcul < nb_parts_tot_calcul)
179 nb_parties_calcul = nb_parts_tot_calcul;
180 else
181 nb_parts_tot_calcul = nb_parties_calcul;
182
183// Nombre de partie total a generer
184 if ((nb_parties_interpol < nb_parties_calcul) && nb_parts_tot==-1)
185 nb_parts_tot=nb_parties_calcul;
186 if (nb_parts_tot_calcul < nb_parts_tot)
187 nb_parts_tot=nb_parts_tot_calcul;
188}
