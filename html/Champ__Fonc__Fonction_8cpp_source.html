76Entree& Sutherland::readOn(Entree& is)
77{
78 Cerr<< "Sutherland::readOn usage { prob A a C c TREF Tref }"<<finl;
79 Cerr<<"== Champ_Fonc_Fonction problem_name Temperature 1 a*(c+Tref)/(c+val)*(val/Tref)^1.5"<<finl;
80 Motcle accferme="}", accouverte="{", motlu;
81 is >> motlu;
82 Cerr<<"Reading Sutherland"<<finl;
83 if (motlu != accouverte)
84 {
85 Cerr<<" We expected "<<accouverte<<" instead of "<<motlu<<finl;
86 abort();
87 }
88
89 Motcles les_mots(3);
90 {
91 les_mots[0] = "A";
92 les_mots[1] = "C";
93 les_mots[2] = "Tref";
94 }
95
96 is >> motlu;
97 while(motlu != accferme )
98 {
99 is >> prob_;
100 int rang=les_mots.search(motlu);
101 switch(rang)
102 {
103 case 0 :
104 is >> A_;
105 break;
106 case 1 :
107 is>> C_;
108 break;
109 case 2 :
110 is>>Tref_;
111 break;
112 default :
113 {
114 Cerr<<"A "<<que_suis_je()<<" doesn't have property "<<motlu<<finl;
115 Cerr<<"We expected a word in:"<<finl<<les_mots<<finl;
116 abort();
117 }
118 }
119 is >> motlu;
120 }
121 if ((C_==-1)||(A_==-1)||(Tref_==-1))
122 {
123 Cerr<<"Sutherland object "<<le_nom()<< " is incorrect"<<finl;
124 if (C_==-1)
125 Cerr<<"You must define C in "<<le_nom()<<finl;
126 if (Tref_==-1)
127 Cerr<<"You must define Tref in "<<le_nom()<<finl;
128 if (A_==-1)
129 Cerr<<"You must define A in "<<le_nom()<<finl;
130 exit();
131 }
132
133 lire_expression();
134 return is;
135}

137void Sutherland::lire_expression()
138{
139// On cree une chaine correspondant a Sutherland
140// c.a.d Champ_Fonc_fonction 1 prob temperature A * (Tref+C)/(val+C) * pow(val/Tref,1.5);
141 Nom chaine = get_prob();
142 chaine += " temperature 1 ";
143 Nom nA(A_);
144 Nom nT(Tref_);
145 Nom nC(C_);
146 chaine+=nA;
147 chaine+="*(";
148 chaine+=nT;
149 chaine+="+";
150 chaine+=nC;
151 chaine+=")/(val+";
152 chaine+=nC;
153 chaine+=")*(val/";
154 chaine+=nT;
155 chaine+=")^1.5";
156 Cerr<<"Sutherland Ok ? Champ_Fonc_Fonction "<<chaine<<finl;
157 EChaine echaine(chaine);
158 Champ_Fonc_Fonction::readOn(echaine);
159 Cerr<<table().val(2)<<finl;
160}
