58template <class _TYPE_>
59int T_It_VDF_Face<_TYPE_>::impr(Sortie& os) const
60{
61 const Zone& ma_zone=la_zone->zone();
62 const int impr_mom=ma_zone.Moments_a_imprimer();
63 const int impr_sum=(ma_zone.Bords_a_imprimer_sum().est_vide() ? 0:1), impr_bord=(ma_zone.Bords_a_imprimer().est_vide() ? 0:1);
64 const Schema_Temps_base& sch = la_zcl->equation().probleme().schema_temps();
65 DoubleTab& tab_flux_bords=op_base->flux_bords();
66 DoubleVect bilan(tab_flux_bords.dimension(1));
67 const int nb_faces = la_zone->nb_faces_tot();
68 DoubleTab xgr(nb_faces,dimension);
69 xgr=0.;
70 if (impr_mom)
71 {
72 const DoubleTab& xgrav = la_zone->xv();
73 const ArrOfDouble& c_grav=ma_zone.cg_moments();
74 for (int num_face=0; num_face <nb_faces; num_face++)
75 for (int i=0; i<dimension; i++)
76 xgr(num_face,i)=xgrav(num_face,i)-c_grav[i];
77 }
78 int k,face, nb_front_Cl=la_zone->nb_front_Cl();
79 DoubleTrav flux_bords2( 5, nb_front_Cl , tab_flux_bords.dimension(1)) ;
80 flux_bords2=0;
81 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
82 {
83 const Cond_lim& la_cl = la_zcl->les_conditions_limites(num_cl);
84 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
85 int ndeb = frontiere_dis.num_premiere_face(), nfin = ndeb + frontiere_dis.nb_faces(), periodicite = (type_cl(la_cl)==periodique?1:0);
86 for (face=ndeb; face<nfin; face++)
87 {
88 for(k=0; k<tab_flux_bords.dimension(1); k++)
89 {
90 flux_bords2(0,num_cl,k)+=tab_flux_bords(face, k);
91 if(periodicite)
92 {
93 if( face < (ndeb+frontiere_dis.nb_faces()/2) ) flux_bords2(1,num_cl,k)+=tab_flux_bords(face, k);
94 else flux_bords2(2,num_cl,k)+=tab_flux_bords(face, k);
95 }
96 if (ma_zone.Bords_a_imprimer_sum().contient(frontiere_dis.le_nom())) flux_bords2(3,num_cl,k)+=tab_flux_bords(face, k);
97 } /* fin for k */
98 if (impr_mom)
99 {
100 if (dimension==2) flux_bords2(4,num_cl,0)+=tab_flux_bords(face,1)*xgr(face,0)-tab_flux_bords(face,0)*xgr(face,1);
101 else
102 {
103 flux_bords2(4,num_cl,0)+=tab_flux_bords(face,2)*xgr(face,1)-tab_flux_bords(face,1)*xgr(face,2);
104 flux_bords2(4,num_cl,1)+=tab_flux_bords(face,0)*xgr(face,2)-tab_flux_bords(face,2)*xgr(face,0);
105 flux_bords2(4,num_cl,2)+=tab_flux_bords(face,1)*xgr(face,0)-tab_flux_bords(face,0)*xgr(face,1);
106 }
107 }
108 } /* fin for face */
109 }
110 mp_sum_for_each_item(flux_bords2);
111 if (je_suis_maitre())
112 {
113// SFichier Flux;
114 if (!Flux.is_open()) op_base->ouvrir_fichier(Flux,"",1);
115// SFichier Flux_moment;
116 if (!Flux_moment.is_open()) op_base->ouvrir_fichier(Flux_moment,"moment",impr_mom);
117// SFichier Flux_sum;
118 if (!Flux_sum.is_open()) op_base->ouvrir_fichier(Flux_sum,"sum",impr_sum);
119 Flux.add_col(sch.temps_courant());
120 if (impr_mom) Flux_moment.add_col(sch.temps_courant());
121 if (impr_sum) Flux_sum.add_col(sch.temps_courant());
122 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
123 {
124 const Cond_lim& la_cl = la_zcl->les_conditions_limites(num_cl);
125 int periodicite = (type_cl(la_cl)==periodique?1:0);
126 for(k=0; k<tab_flux_bords.dimension(1); k++)
127 {
128 if(periodicite)
129 {
130 Flux.add_col(flux_bords2(1,num_cl,k));
131 Flux.add_col(flux_bords2(2,num_cl,k));
132 }
133 else Flux.add_col(flux_bords2(0,num_cl,k));
134
135 if (impr_sum) Flux_sum.add_col(flux_bords2(3,num_cl,k));
136 bilan(k)+=flux_bords2(0,num_cl,k);
137 }
138 if (dimension==3)
139 {
140 for (k=0; k<tab_flux_bords.dimension(1); k++)
141 if (impr_mom) Flux_moment.add_col(flux_bords2(4,num_cl,k));
142 }
143 else if (impr_mom) Flux_moment.add_col(flux_bords2(4,num_cl,0));
144 } /* fin for num_cl */
145 for(k=0; k<tab_flux_bords.dimension(1); k++) Flux.add_col(bilan(k));
146 Flux << finl;
147 if (impr_sum) Flux_sum << finl;
148 if (impr_mom) Flux_moment << finl;
149 }
150 const LIST(Nom)& Liste_Bords_a_imprimer = la_zone->zone().Bords_a_imprimer();
151 if (!Liste_Bords_a_imprimer.est_vide())
152 {
153 EcrFicPartage Flux_face;
154 op_base->ouvrir_fichier_partage(Flux_face,"",impr_bord);
155 for (int num_cl=0; num_cl<nb_front_Cl; num_cl++)
156 {
157 const Frontiere_dis_base& la_fr = la_zcl->les_conditions_limites(num_cl).frontiere_dis();
158 const Cond_lim& la_cl = la_zcl->les_conditions_limites(num_cl);
159 const Front_VF& frontiere_dis = ref_cast(Front_VF,la_cl.frontiere_dis());
160 int ndeb = frontiere_dis.num_premiere_face(), nfin = ndeb + frontiere_dis.nb_faces();
161 if (ma_zone.Bords_a_imprimer().contient(la_fr.le_nom()))
162 {
163 if(je_suis_maitre())
164 {
165 Flux_face << "# Flux par face sur " << la_fr.le_nom() << " au temps ";
166 sch.imprimer_temps_courant(Flux_face);
167 Flux_face << " : " << finl;
168 }
169 for (face=ndeb; face<nfin; face++)
170 {
171 if (dimension == 2) Flux_face << "# Face a x= " << la_zone->xv(face,0) << " y= " << la_zone->xv(face,1) << " : ";
172 else if (dimension == 3) Flux_face << "# Face a x= " << la_zone->xv(face,0) << " y= " << la_zone->xv(face,1) << " z= " << la_zone->xv(face,2) << " : ";
173 for(k=0; k<tab_flux_bords.dimension(1); k++) Flux_face << tab_flux_bords(face, k) << " ";
174 Flux_face << finl;
175 }
176 Flux_face.syncfile();
177 }
178 }
179 }
180 return 1;
181}

183template <class _TYPE_>
184DoubleTab& T_It_VDF_Face<_TYPE_>::ajouter(const DoubleTab& inco, DoubleTab& resu) const
185{
186 ((_TYPE_&) flux_evaluateur).mettre_a_jour();
187 assert(inco.nb_dim() < 3);
188 const int ncomp = inco.line_size();
189 DoubleTab& tab_flux_bords=op_base->flux_bords();
190 tab_flux_bords.resize(la_zone->nb_faces_bord(),dimension);
191 tab_flux_bords=0;
192
193 if (ncomp == 1)
194 {
195 ajouter_aretes_bords<SingleDouble>(ncomp,inco,resu);
196 ajouter_aretes_coins<SingleDouble>(ncomp,inco,resu);
197 ajouter_aretes_internes<SingleDouble>(ncomp,inco,resu);
198 ajouter_aretes_mixtes<SingleDouble>(ncomp,inco,resu);
199 ajouter_fa7_sortie_libre<SingleDouble>(ncomp,inco,resu);
200 ajouter_fa7_elem<SingleDouble>(ncomp,inco,resu);
201 }
202 else
203 {
204 ajouter_aretes_bords<ArrOfDouble>(ncomp,inco,resu);
205 ajouter_aretes_coins<ArrOfDouble>(ncomp,inco,resu);
206 ajouter_aretes_internes<ArrOfDouble>(ncomp,inco,resu);
207 ajouter_aretes_mixtes<ArrOfDouble>(ncomp,inco,resu);
208 ajouter_fa7_sortie_libre<ArrOfDouble>(ncomp,inco,resu);
209 ajouter_fa7_elem<ArrOfDouble>(ncomp,inco,resu);
210 }
211
212// On multiplie les flux au bord par rho en hydraulique (sert uniquement a la sortie)
213 multiply_by_rho_if_hydraulique(tab_flux_bords);
214 return resu;
215}

217template <class _TYPE_>
218inline void T_It_VDF_Face<_TYPE_>::multiply_by_rho_if_hydraulique(DoubleTab& tab_flux_bords) const
219{
220 Nom nom_eqn=la_zcl->equation().que_suis_je();
221 /* Modif B.Mathieu pour front-tracking: masse_volumique() invalide en f.t.*/
222 if (nom_eqn.debute_par("Navier_Stokes") && nom_eqn!="Navier_Stokes_Melange" && nom_eqn!="Navier_Stokes_FT_Disc")
223 {
224 const Champ_base& rho = la_zcl->equation().milieu().masse_volumique().valeur();
225 if (sub_type(Champ_Uniforme,rho))
226 {
227 const double coef = rho(0,0);
228 const int nb_faces_bord = la_zone->nb_faces_bord();
229 for (int face = 0; face < nb_faces_bord; face++)
230 for(int k = 0; k < tab_flux_bords.line_size(); k++) tab_flux_bords(face,k) *= coef;
231 }
232 }
