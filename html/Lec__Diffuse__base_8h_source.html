82template <typename _TYPE_>
83int Lec_Diffuse_base::get_template(_TYPE_ *ob, int n)
84{
85 int ok = 0;
86 if (Process::je_suis_maitre())
87 {
88 Entree& is = get_entree_master();
89 assert(is.get_error_action() == ERROR_CONTINUE);
90 ok = is.get(ob, n);
91 }
92 else if (!diffuse_)
93 {
94 Cerr << "Lec_Diffuse_base::get(...) can't be used with diffuse_=0 on non master process." << finl;
95 Process::exit();
96 }
97 if (diffuse_)
98 {
99 envoyer_broadcast(ok, 0);
100 if (ok)
101 envoyer_broadcast_array(ob, n, 0);
102 }
103 return error_handle(!ok);
104}
