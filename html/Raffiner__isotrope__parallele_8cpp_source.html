46void mon_construire_correspondance_items_par_coordonnees(Joints& joints, const Joint::Type_Item type_item, const DoubleTab& coord_items)
47{
48 switch(type_item)
49 {
50 case Joint::SOMMET:
51 ;
52 break;
53 case Joint::ARETE:
54 ;
55 break;
56 default:
57 Cerr << "Scatter::construire_correspondance_items_par_coordonnees unusable for item "
58 << (int)type_item
59 << finl;
60 Process::exit();
61 }
62 const int dim = Objet_U::dimension;
63 const int nb_joints = joints.size();
64
65// Indices des items de joints dans le domaine sur mon processeur
66 ArrsOfInt indices_items_locaux(nb_joints);
67// Indices des items de joints dans le domaine sur le processeur voisin
68 ArrsOfInt indices_items_distants(nb_joints);
69// Coordonnees des items correspondants (dans le meme ordre que indices_items_xxx)
70 DoubleTabs coord_items_locaux(nb_joints);
71 DoubleTabs coord_items_distants(nb_joints);
72
73// Remplissage des tableaux indices_items_locaux
74// et coord_items_locaux
75 {
76 int i_joint;
77 for (i_joint = 0; i_joint < nb_joints; i_joint++)
78 {
79 const Joint& joint = joints[i_joint];
80 ArrOfInt& items = indices_items_locaux[i_joint];
81// Remarque: **LISTE_TRI** les indices_items_locaux sont
82// tries dans l'ordre croissant:
83// Scatter::calculer_liste_complete_items_joint(joint, type_item, items);
84 items = joint.joint_item(Joint::SOMMET).items_communs();
85 const int n = items.size_array();
86 DoubleTab& coord = coord_items_locaux[i_joint];
87 int i, j;
88 coord.resize(n, dim);
89 for (i = 0; i < n; i++)
90 for (j = 0; j < dim; j++)
91 coord(i,j) = coord_items(items[i], j);
92 }
93 }
94
95// Envoi des indices et coordonnees locaux au processeur voisin
96 {
97 Schema_Comm schema_comm;
98 ArrOfInt liste_pe_voisins(nb_joints);
99 int i;
100 for (i = 0; i < nb_joints; i++)
101 liste_pe_voisins[i] = joints[i].PEvoisin();
102 schema_comm.set_send_recv_pe_list(liste_pe_voisins, liste_pe_voisins);
103 schema_comm.begin_comm();
104 for (i = 0; i < nb_joints; i++)
105 {
106 const int pe = liste_pe_voisins[i];
107 Sortie& buffer = schema_comm.send_buffer(pe);
108 buffer << indices_items_locaux[i];
109 buffer << coord_items_locaux[i];
110 }
111 schema_comm.echange_taille_et_messages();
112 for (i = 0; i < nb_joints; i++)
113 {
114 const int pe = liste_pe_voisins[i];
115 Entree& buffer = schema_comm.recv_buffer(pe);
116 buffer >> indices_items_distants[i];
117 buffer >> coord_items_distants[i];
118 }
119 schema_comm.end_comm();
120 }
121
122// Boucle sur les joints
123 {
124// Cette fois, on modifie les joints (remplissage de renum_virt_loc)
125// const int moi = Process::me();
126 for (int i_joint = 0; i_joint < nb_joints; i_joint++)
127 {
128 Joint& joint = joints[i_joint];
129 const int PEvoisin = joint.PEvoisin();
130 const ArrOfInt& indices_locaux = indices_items_locaux[i_joint];
131// ArrOfInt & indices_distants= indices_items_distants[i_joint];
132 const DoubleTab& coord_locaux = coord_items_locaux[i_joint];
133 const DoubleTab& coord_distants = coord_items_distants[i_joint];
134 const int n = indices_locaux.size_array();
135 /*
136 if (n != indices_distants.size_array()) {
137 Cerr << "Error in Scatter::remplir_renum_virt_loc on PE "
138 << Process::me();
139 Cerr << "\n Joint with PE " << joint.PEvoisin()
140 << "\n Number of items " << type_item << " on my joint : " << n;
141 Cerr << "\n Number of items " << type_item << " on the remote joint : "
142 << indices_distants.size_array()
143 << finl;
144 exit();
145 } */
146// Recherche des correspondances entre items
147 ArrOfInt corresp(n);
148 const double epsilon = Objet_U::precision_geom;
149 int nbsomr=Scatter::Chercher_Correspondance(coord_distants, coord_locaux, corresp, epsilon);
150 if (nbsomr)
151 {
152 Cerr<<"iiiiiiiiiiii"<<finl;
153// abort();
154 }
155 ArrOfInt& items_communs = joint.set_joint_item(type_item).set_items_communs();
156 int nb_items_communs_trouves=0;
157 for (int k = 0; k < n; k++)
158 if (corresp[k]>=0)
159 nb_items_communs_trouves++;
160 else
161 Cerr<<items_communs[k] << " not found in joint voisin "<<PEvoisin<<finl;
162
163
164 ArrOfInt items_communs_sa(items_communs);
165 items_communs.resize_array(nb_items_communs_trouves);
166
167 int l=0;
168
169 for (int k = 0; k < n; k++ )
170 {
171 if (corresp[k]>=0)
172 items_communs[l++]=items_communs_sa[k];
173 }
174 }
175 }
176}

189Entree& Raffiner_isotrope_parallele::interpreter(Entree& is)
190{
191 int form=0, format_hdf = 0;
192 Nom org,newd;
193 Param param(que_suis_je());
194// XD Raffiner_isotrope_parallele interprete Raffiner_isotrope_parallele 1 Refine parallel mesh in parallel
195 param.ajouter("name_of_initial_zones",&org,Param::REQUIRED); //!< XD_ADD_P chaine name of initial Zones
196 param.ajouter("name_of_new_zones",&newd,Param::REQUIRED); //!< XD_ADD_P chaine name of new Zones
197 param.ajouter("ascii",&form); //!< XD_ADD_P flag writing Zones in ascii format
198 param.ajouter_flag("single_hdf",&format_hdf); //!< XD_ADD_P flag writing Zones in hdf format
199 param.lire_avec_accolades(is);
200// Force un fichier unique au dela d'un certain nombre de rangs MPI:
201 if (Process::force_single_file(Process::nproc(), org+".Zones"))
202 format_hdf = 1;
203 int binaire=!form;
204 if (form && format_hdf)
205 {
206 Cerr << "Raffiner_isotrope_parallele::interpreter(): options 'ascii' and 'single_hdf' are mutually exclusive!" << finl;
207 Process::exit(1);
208 }
209 Domaine dom_org;
210 Noms liste_bords_periodiques;
211 org+=".Zones";
212
213 Nom copy(org);
214 copy = copy.nom_me(Process::nproc(), "p", 1);
215// bool is_hdf = FichierHDF::is_hdf5(copy);
216 LecFicDiffuse test;
217 bool is_hdf = test.ouvrir(copy) && FichierHDF::is_hdf5(copy);
218
219 if (!is_hdf)
220 {
221 LecFicDistribue fichier;
222 fichier.set_bin(binaire);
223 fichier.ouvrir(org);
224 fichier >> dom_org;
225 dom_org.set_fichier_lu(org);
226 fichier >> liste_bords_periodiques;
227 }
228 else
229 {
230 FichierHDFPar fic_hdf;
231// FichierHDF fic_hdf;
232 org = copy;
233 fic_hdf.open(org, true);
234 Entree_Brute data;
235
236 fic_hdf.read_dataset("/zone", Process::me(), data);
237
238// Feed TRUST objects:
239 data >> dom_org;
240 dom_org.set_fichier_lu(org);
241 data >> liste_bords_periodiques;
242
243 fic_hdf.close();
244 }
245
246 Scatter::uninit_sequential_domain(dom_org);
247 Domaine dom_new(dom_org);
248
249 refine_domain(dom_org,dom_new);
250
251
252 static Stat_Counter_Id stats = statistiques().new_counter(0 /* Level */, "Mailler_Parallel", 0 /* Group */);
253
254 if (nproc() > 1)
255 {
256 Scatter::uninit_sequential_domain(dom_new);
257 int nb_sommet_avant_completion=dom_new.nb_som();
258 Scatter::trier_les_joints(dom_new.zone(0).faces_joint());
259 statistiques().begin_count(stats);
260
261
262 mon_construire_correspondance_sommets_par_coordonnees(dom_new);
263
264 if (0)
265 {
266 Nom debug(newd);
267 debug+="_debug.Zones";
268 EcrFicCollecte os;
269 os.set_bin(binaire);
270
271 os.ouvrir(debug);
272 if (!binaire)
273 {
274 os.setf(ios::scientific);
275 os.precision(Objet_U::format_precision_geom);
276 }
277
278 os << dom_new;
279
280 os << liste_bords_periodiques;
281
282 }
283// else
284 {
285
286 Scatter::construire_correspondance_sommets_par_coordonnees(dom_new);
287 Scatter::calculer_renum_items_communs(dom_new.zone(0).faces_joint(), Joint::SOMMET);
288
289 statistiques().end_count(stats);
290 double maxtime = mp_max(statistiques().last_time(stats));
291 Cerr << "Scatter::construire_correspondance_sommets_par_coordonnees fin, time:"
292 << maxtime
293 << finl;
294
295 statistiques().begin_count(stats);
296
297 Scatter::construire_structures_paralleles(dom_new, liste_bords_periodiques);
298
299 statistiques().end_count(stats);
300 maxtime = mp_max(statistiques().last_time(stats));
301 Cerr << "Scatter::construire_structures_paralleles, time:" << maxtime << finl;
302
303
304 int ecrit=1;
305 if (ecrit)
306 {
307 int nb_elem_reel=dom_new.zone(0).nb_elem();
308 Scatter::uninit_sequential_domain(dom_new);
309 dom_new.zone(0).les_elems().resize( nb_elem_reel,dom_new.zone(0).les_elems().dimension(1));
310 dom_new.les_sommets().resize(nb_sommet_avant_completion,dimension);
311
312 Scatter::uninit_sequential_domain(dom_new);
313 newd+=".Zones";
314
315 if( !format_hdf )
316 {
317 EcrFicCollecte os;
318 os.set_bin(binaire);
319 os.ouvrir(newd);
320 if (!binaire)
321 {
322 os.setf(ios::scientific);
323 os.precision(Objet_U::format_precision_geom);
324 }
325 os << dom_new;
326 os << liste_bords_periodiques;
327 }
328 else
329 {
330 Sortie_Brute os_hdf;
331 os_hdf << dom_new;
332 os_hdf << liste_bords_periodiques;
333 FichierHDFPar fic_hdf;
334 newd = newd.nom_me(Process::nproc(), "p", 1);
335 fic_hdf.create(newd);
336 fic_hdf.create_and_fill_dataset_MW("/zone", os_hdf);
337 fic_hdf.close();
338 }
339 }
340 }
341 }
342 else
343 {
344 Scatter::init_sequential_domain(dom_new);
345 }
346
347 return is;
348}
