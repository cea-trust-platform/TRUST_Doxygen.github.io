50//! Constructeur.<br>Si nb_items_reels >= 0, items_to_compute contiendra un bloc avec les items reels,<br>sinon, items_to_compute prendra tous les items (jusqu'a nb_items_tot).<br>Parametre: nb_items_tot<br>Signification: valeur que prendra nb_items_tot_ (doit etre >= 0)<br>Parametre: nb_items_reels<br>Signification: valeur que prendra nb_items_reels_ (doit etre >= -1,<br>-1 signifie qu'il n'y a pas d'items "reels" groupes au debut du tableau<br>Parametre: pe_voisins<br>Signification: liste des processeurs voisins, classes dans l'ordre croissant<br>(les tableaux ArrsOfInt doivent avoir la meme taille).<br>Cette liste n'est pas reprise integralement: on retire les processeurs<br>pour lesquels toutes les trois listes d'items sont vides.<br>Parametre: items_to_send<br>Signification: pour chaque pe_voisin, liste des items a envoyer (communs et virtuels).<br>count_items_to_send_to_items_ est calcule par une communication a partir de items_to_recv.<br>Parametre: items_to_recv<br>Signification: liste d'items individuels a recevoir des differents procs<br>(souvent des items communs mais ce n'est pas obligatoire, ex front-tracking)<br>Parametre: blocs_to_recv<br>Signification: liste de blocs d'items a recevoir (voir MD_Vector_std::blocs_to_recv_)<br>(souvent les items virtuels...)
51MD_Vector_std::MD_Vector_std(int nb_items_tot, int nb_items_reels, const ArrOfInt& pe_voisins,
52 const ArrsOfInt& items_to_send, const ArrsOfInt& items_to_recv, const ArrsOfInt& blocs_to_recv)
53{
54 assert(nb_items_tot >= 0);
55 assert(nb_items_reels >= -1 && nb_items_reels <= nb_items_tot);
56 nb_items_tot_ = nb_items_tot;
57 nb_items_reels_ = nb_items_reels;
58 const int nb_voisins = pe_voisins.size_array();
59 assert(items_to_send.size() == nb_voisins);
60 assert(items_to_recv.size() == nb_voisins);
61 assert(blocs_to_recv.size() == nb_voisins);
62
63// selection: liste des indices des pe a conserver dans pe_voisins
64// (procs avec qui on echange effectivement des donnees)
65 ArrOfInt tmp, selection;
66 tmp.set_smart_resize(1);
67 selection.set_smart_resize(1);
68 int i;
69 for (i = 0; i < nb_voisins; i++)
70 if (items_to_send[i].size_array() > 0 || items_to_recv[i].size_array() > 0 || blocs_to_recv[i].size_array() > 0)
71 {
72 tmp.append_array(pe_voisins[i]);
73 selection.append_array(i);
74 }
75 const int nb_voisins2 = tmp.size_array();
76// ** pe_voisins_ **
77 pe_voisins_ = tmp;
78
79 {
80 ArrsOfInt tmpbis(nb_voisins2);
81
82 for (i = 0; i < nb_voisins2; i++)
83 tmpbis[i] = items_to_send[selection[i]];
84 items_to_send_.set(tmpbis);
85
86 for (i = 0; i < nb_voisins2; i++)
87 tmpbis[i] = items_to_recv[selection[i]];
88 items_to_recv_.set(tmpbis);
89
90 for (i = 0; i < nb_voisins2; i++)
91 tmpbis[i] = blocs_to_recv[selection[i]];
92 blocs_to_recv_.set(tmpbis);
93 }
94
95// Calcul de nb_items_to_items_
96 {
97 nb_items_to_items_.resize_array(nb_voisins2, Array_base::NOCOPY_NOINIT);
98 tmp.resize_array(nproc(), Array_base::NOCOPY_NOINIT);
99 tmp = 0;
100 for (i = 0; i < nb_voisins2; i++)
101 tmp[pe_voisins_[i]] = items_to_recv[selection[i]].size_array();
102 ;
103 envoyer_all_to_all(tmp, tmp);
104 for (i = 0; i < nb_voisins2; i++)
105 {
106 int n = tmp[pe_voisins_[i]];
107 assert(n <= items_to_send_.get_list_size(i));
108 nb_items_to_items_[i] = n;
109 }
110 }
111// Calcul de blocs_items_count_
112 {
113 blocs_items_count_.resize_array(nb_voisins2, Array_base::NOCOPY_NOINIT);
114 for (i = 0; i < nb_voisins2; i++)
115 {
116 const int nblocs = blocs_to_recv_.get_list_size(i) / 2;
117 int count = 0;
118 for (int j = 0; j < nblocs; j++)
119 {
120 int deb = blocs_to_recv_(i, j * 2);
121 int fin = blocs_to_recv_(i, j * 2 + 1);
122 assert(deb >= 0 && fin > deb && fin <= nb_items_tot_);
123 count += fin - deb;
124 }
125 blocs_items_count_[i] = count;
126 }
127 }
128// Calcul des blocs d'items sequentiels (items non recus d'un autre proc)
129 tmp.resize_array(nb_items_tot, Array_base::NOCOPY_NOINIT);
130 {
131// Marquage des items recus a zero
132 tmp = 1;
133 const ArrOfInt& items = items_to_recv_.get_data();
134 int n = items.size_array();
135 for (i = 0; i < n; i++)
136 {
137 assert(tmp[items[i]] == 1); //!< sinon, c'est qu'on recoit deux fois le meme items
138 tmp[items[i]] = 0;
139 }
140 const ArrOfInt& items2 = blocs_to_recv_.get_data();
141 assert(items2.size_array() % 2 == 0);
142 n = items2.size_array() / 2;
143 for (i = 0; i < n; i++)
144 {
145 const int start = items2[i * 2];
146 const int end = items2[i * 2 + 1];
147 for (int j = start; j < end; j++)
148 {
149 assert(tmp[j] == 1); //!< sinon, c'est qu'on recoit deux fois le meme items
150 tmp[j] = 0;
151 }
152 }
153
154// Construction d'une liste de blocs d'items sequentiels (ceux qui sont restes a 1)
155 ArrOfInt blocs;
156 blocs.set_smart_resize(1);
157 int nb_seq = 0;
158 for (i = 0; i < nb_items_tot; i++)
159 {
160 if (tmp[i])
161 {
162 if (nb_items_reels >= 0 && i >= nb_items_reels)
163 {
164// On a un item non recu parmi les items virtuels !
165 Cerr << "Error in MD_Vector_std::MD_Vector_std(...) [pe " << me() << "]\n "
166 "an item i(" << i << ") > nb_items_reels(" << nb_items_reels << ") is not received from any other processor" << finl;
167 Process::exit();
168 }
169 append_item_to_blocs(blocs, i);
170 nb_seq++;
171 }
172 }
173 blocs_items_to_sum_ = blocs;
174 nb_items_seq_local_ = nb_seq;
175 nb_items_seq_tot_ = mp_sum(nb_seq);
176 }
177
178// Bloc items to compute:
179// Par defaut: un seul bloc
180 if (nb_items_reels_ >= 0)
181 {
182// Les operateurs sur les tableaux calculent tous les items reels
183 blocs_items_to_compute_.resize(2, Array_base::NOCOPY_NOINIT);
184 blocs_items_to_compute_[0] = 0;
185 blocs_items_to_compute_[1] = nb_items_reels_;
186 }
187 else
188 {
189// Les operateurs sur les tableaux calculent tout
190 blocs_items_to_compute_.resize(2, Array_base::NOCOPY_NOINIT);
191 blocs_items_to_compute_[0] = 0;
192 blocs_items_to_compute_[1] = nb_items_tot_;
193 }
194}

198Entree& MD_Vector_std::readOn(Entree& is)
199{
200 MD_Vector_base2::readOn(is);
201 Param p(que_suis_je());
202 ArrOfInt items_send_index, items_send_data, items_index, items_data, blocs_index, blocs_data;
203 p.ajouter("pe_voisins", &pe_voisins_);
204 p.ajouter("items_to_send_index", &items_send_index);
205 p.ajouter("items_to_send_data", &items_send_data);
206 p.ajouter("nb_items_to_items", &nb_items_to_items_);
207 p.ajouter("items_to_recv_index", &items_index);
208 p.ajouter("items_to_recv_data", &items_data);
209 p.ajouter("blocs_to_recv_index", &blocs_index);
210 p.ajouter("blocs_to_recv_data", &blocs_data);
211 p.ajouter("blocs_items_count", &blocs_items_count_);
212 p.lire_avec_accolades(is);
213 if (items_send_data.size_array())
214 items_to_send_.set_index_data(items_send_index, items_send_data);
215 if (items_data.size_array())
216 items_to_recv_.set_index_data(items_index, items_data);
217 if (blocs_data.size_array())
218 blocs_to_recv_.set_index_data(blocs_index, blocs_data);
219 return is;
220}
