41double largest_angle(const DoubleTab& coords)
42{
43 if (((coords.dimension(0)!=4)&&(coords.dimension(0)!=3))||(coords.dimension(1)!=3))
44 {
45 Cerr<<" case not provided"<<finl;
46 Process::exit();
47 }
48 int nb_face=coords.dimension(0);
49 Vecteur3 normals[4];
50 Vecteur3 edge[2],opp;
51 edge[1].set(0,0,1);
52 for (int n=0; n<nb_face; n++)
53 {
54
55 int prem=0;
56 if (n==0) prem=1;
57 int compteur=0;
58 for (int s=0; s<nb_face; s++)
59 {
60 if ((s!=n) && (s!=prem))
61 {
62 edge[compteur].set(coords(s,0)-coords(prem,0),
63 coords(s,1)-coords(prem,1),
64 coords(s,2)-coords(prem,2));
65 compteur++;
66 }
67 }
68 if (compteur!=nb_face-2) abort();
69 opp.set(coords(n,0)-coords(prem,0),
70 coords(n,1)-coords(prem,1),
71 coords(n,2)-coords(prem,2));
72 Vecteur3::produit_vectoriel(edge[0],edge[1],normals[n]);
73// normals[n]=edge[0]*edge[1];
74 double l = normals[n].length();
75 normals[n]*=(l>0 ? 1./l : 0);
76 if (Vecteur3::produit_scalaire(normals[n],opp)<0)
77 normals[n]*=-1;
78 }
79// on a les 4 normals orientes vers l'interieur
80 double max_pscal=-100;
81 for (int n1=0; n1<nb_face; n1++)
82 for (int n2=n1+1; n2<nb_face; n2++)
83 {
84 double pscal=Vecteur3::produit_scalaire(normals[n1],normals[n2]);
85// min_pscal=pscal;
86 if (pscal>max_pscal)
87 max_pscal=pscal;
88 }
89 double tet=acos(max_pscal<=1. ? max_pscal : 1.)/acos(-1.)*180; //!< acos(-1) ne compile pas avec xlC
90
91 tet=180-tet;
92 return tet;
93}

106void histogramme_angle(const Domaine& dom , Sortie& out, int nb_histo )
107{
108 out<<finl<<"Histogram of the largest angle of each element found into the mesh "<<dom.le_nom()<<" :" << finl;
109 Motcle type_elem(dom.zone(0).type_elem().valeur().que_suis_je());
110 if (((type_elem!="triangle")&& (type_elem!="tetraedre")) || ((type_elem=="triangle") && (Domaine::dimension==3)))
111 {
112 out<<"Not available for "<<type_elem<<" in dimension "<<Domaine::dimension<<finl;
113 return;
114 }
115
116 ArrOfInt histo(nb_histo+1);
117 int nb_elem=dom.zone(0).nb_elem();
118 const DoubleTab& som=dom.les_sommets();
119 const IntTab& les_elems=dom.zone(0).les_elems();
120 int dim_space=som.dimension(1);
121 int nb_som_elem=les_elems.dimension(1);
122 DoubleTab coords(nb_som_elem,3);
123 for (int elem=0; elem<nb_elem; elem++)
124 {
125 for (int s=0; s<nb_som_elem; s++)
126 {
127 int sommet=les_elems(elem,s);
128 for (int dir=0; dir<dim_space; dir++)
129 coords(s,dir)=som(sommet,dir);
130
131 }
132 double teta=largest_angle(coords);
133
134 int test=(int)(teta/180*nb_histo);
135 histo[test]++;
136 if (test==nb_histo)
137 Cerr << "Error, the mesh cell " << elem << " is flat. Fix your mesh." << finl;
138 }
139 if (histo[nb_histo]>0)
140 Process::exit();
141 int nb_elem_tot=(int)Process::mp_sum(nb_elem);
142 if (nb_elem_tot>0)
143 {
144 double obtuse_cells_proportion=0;
145 for (int h=0; h<nb_histo; h++)
146 {
147 histo[h]=Process::mp_sum(histo[h]);
148// Pas d'angles en dessous de 60 forcement
149 if (180/nb_histo*h>=60)
150 {
151 obtuse_cells_proportion=histo[h]*100./nb_elem_tot;
152 int angle=180/nb_histo*h;
153 out <<"Between "<<angle<<" degrees and " << 180/nb_histo*(h+1)<<" degrees : "<<histo[h]<<" elements ( "<<obtuse_cells_proportion<< " %)"<<finl;
154 }
155 }
156// Criteria used on the last bin (170-180 degrees)
157 if (obtuse_cells_proportion>20)
158 {
159 Cerr << finl;
160 Cerr << "Warning:" << finl;
161 Cerr << "Your mesh has too much obtuse cells for a TRUST calculation." << finl;
162 Cerr << "Check the mesh requirements." << finl;
163// Process::exit();
164 }
165 }
166 out << finl;
167}
