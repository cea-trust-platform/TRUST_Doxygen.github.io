57void Champ_Fonc_MED_Table_Temps::lire_donnees_champ(const std::string& fileName, const std::string& meshName, const std::string& fieldName,
58 ArrOfDouble& temps_sauv, int& size, int& nbcomp, Nom& type_champ)
59{
60#ifdef MEDCOUPLING_
61 MCAuto<MEDFileFieldMultiTS> ft1(MEDFileFieldMultiTS::New(fileName,fieldName));
62 std::vector<double> tps;
63 std::vector< std::pair<int,int> > tst = ft1->getTimeSteps(tps);
64
65 temps_sauv.resize_array(1);
66
67 int first_iter = tst[0].first;
68 int first_order = tst[0].second;
69 temps_sauv[0] = tps[0];
70
71// Only one MCAuto below to avoid double deletion:
72 MCAuto<MEDCouplingField> ffield = ReadField(field_type, fileName, meshName, 0, fieldName,
73 first_iter, first_order);
74 MEDCouplingFieldDouble * field = dynamic_cast<MEDCouplingFieldDouble *>((MEDCouplingField *)ffield);
75 if (field == 0)
76 {
77 Cerr << "ERROR reading MED field! Not a MEDCouplingFieldDouble!!" << finl;
78 Process::exit(-1);
79 }
80 size = field->getNumberOfTuplesExpected();
81 nbcomp = (int) field->getNumberOfComponents();
82
83 if (field_type == MEDCoupling::ON_NODES)
84 {
85 if ((cell_type == INTERP_KERNEL::NORM_QUAD4) || (cell_type == INTERP_KERNEL::NORM_HEXA8))
86 type_champ = "Champ_Fonc_Q1_MED";
87 else
88 type_champ = "Champ_Fonc_P1_MED";
89 }
90 else if (field_type == MEDCoupling::ON_CELLS)
91 type_champ = "Champ_Fonc_P0_MED";
92
93 vrai_champ0_.typer(type_champ);
94 fixer_nb_comp(nbcomp);
95 le_champ0().fixer_nb_comp(nbcomp);
96 zonebidon_inst.associer_zone(mon_dom.valeur().zone(0));
97 le_champ0().associer_zone_dis_base(zonebidon_inst);
98 if (zonebidon_inst.nb_elem()==0) size = 0;
99 le_champ0().fixer_nb_valeurs_nodales(size);
100 if (size != le_champ0().valeurs().dimension(0))
101 {
102 Cerr << "Error in " << que_suis_je() << " : nb_ddl incorrect" << finl;
103 Process::exit();
104 }
105 const double *field_values = field->getArray()->begin();
106 assert(field->getNumberOfTuplesExpected() == le_champ0().valeurs().dimension(0));
107 assert((int) field->getNumberOfComponents() ==
108 (le_champ0().valeurs().nb_dim() == 1 ? 1 : le_champ0().valeurs().dimension(1)));
109 memcpy(le_champ0().valeurs().addr(), field_values,
110 le_champ0().valeurs().size_array() * sizeof(double));
111#endif
112}
