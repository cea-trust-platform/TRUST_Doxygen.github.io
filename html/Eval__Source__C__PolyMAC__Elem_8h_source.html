56inline double Eval_Source_C_PolyMAC_Elem::calculer_terme_source(int num_elem) const
57{
58 if (sub_type(Champ_Uniforme,la_source_constituant.valeur().valeur()))
59 return source_constituant(0,0)*volumes(num_elem)*porosite_vol(num_elem);
60 else if (source_constituant.nb_dim()==2)
61 return source_constituant(num_elem,0)*volumes(num_elem)*porosite_vol(num_elem);
62 else
63 return source_constituant(num_elem)*volumes(num_elem)*porosite_vol(num_elem);
64}

66inline void Eval_Source_C_PolyMAC_Elem::calculer_terme_source(int num_elem, DoubleVect& source) const
67{
68 int size=source.size();
69 if (sub_type(Champ_Uniforme,la_source_constituant.valeur().valeur()))
70 for (int i=0; i<size; i++)
71 source(i) = source_constituant(0,i)*volumes(num_elem)*porosite_vol(num_elem);
72 else
73 for (int i=0; i<size; i++)
74 source(i) = source_constituant(num_elem,i)*volumes(num_elem)*porosite_vol(num_elem);
