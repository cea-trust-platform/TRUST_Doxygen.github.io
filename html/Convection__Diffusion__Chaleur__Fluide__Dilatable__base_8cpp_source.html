97int Convection_Diffusion_Chaleur_Fluide_Dilatable_base::sauvegarder(Sortie& os) const
98{
99 int bytes=0;
100 bytes += Equation_base::sauvegarder(os);
101// en mode ecriture special seul le maitre ecrit

158//! remplissage de la zone cl modifiee avec 1 partout au bord...
159int Convection_Diffusion_Chaleur_Fluide_Dilatable_base::remplir_cl_modifiee()
160{
161 zcl_modif_=(zone_Cl_dis());
162 Conds_lim& condlims=zcl_modif_.valeur().les_conditions_limites();
163 int nb=condlims.size();
164// pour chaque condlim on recupere le champ_front et on met 1
165// meme si la cond lim est un flux (dans ce cas la convection restera nullle.)
166 for (int i=0; i<nb; i++)
167 {
168 DoubleTab& T=condlims[i].valeur().champ_front().valeurs();
169 T=1.;
170 if (sub_type(Neumann_sortie_libre,condlims[i].valeur()))
171 ref_cast(Neumann_sortie_libre,condlims[i].valeur()).tab_ext()=1;
172 }
173 zcl_modif_.les_conditions_limites().set_modifier_val_imp(0);
174 return 1;
175}
