84//
85//! fonction outil pour tri quicksort<br>(comparaison de deux reels avec un index)
86static True_int trier_index_compare_fct(const void *i_, const void *j_)
87{
88 const int i = *((int*)i_);
89 const int j = *((int*)j_);
90 const int k = static_qsort_DoubleTab_colonne;
91 const DoubleTab& tab = *static_qsort_DoubleTab_ptr;
92 const double x = tab(i,k);
93 const double y = tab(j,k);
94 True_int resu = 0;
95 if (x > y)
96 resu = 1;
97 else if (x < y)
98 resu = -1;
99 return resu;
100}

104//! Fonction outil utilisee dans construire_partition.<br>On reordonne les valeurs du tableau index de sorte que les valeurs<br>tab(index[i], colonne) soient classees dans l'ordre croissant.
105static void trier_index_colonne_i(const DoubleTab& tab,
106 ArrOfInt& index,
107 const int colonne)
108{
109 assert(colonne >= 0 && colonne < tab.dimension(1));
110
111 static_qsort_DoubleTab_ptr = &tab;
112 static_qsort_DoubleTab_colonne = colonne;
113 int * index_ptr = index.addr();
114 const int index_size = index.size_array();
115 const int index_member_size = sizeof(int);
116
117 qsort(index_ptr, index_size, index_member_size, trier_index_compare_fct);
118}

121//
122//! Remplissage du tableau directions perio a partir des noms des bords<br>periodiques. Pour 0 <= i < Objet_U::dimension directions_perio[i] vaut 1 s'il<br>existe un bord periodique pour lequel le vecteur delta est dirige dans la direction i.
123void Partitionneur_Tranche::chercher_direction_perio(const Zone& zone,
124 const Noms& liste_bords_perio,
125 ArrOfInt& directions_perio)
126{
127 Cerr << "Search of periodic directions of domain " << zone.domaine().le_nom() << finl;
128 const int dim = Objet_U::dimension;
129 directions_perio.resize_array(dim);
130 directions_perio = 0;
131 const int n = liste_bords_perio.size();
132 for (int i = 0; i < n; i++)
133 {
134 int num_bord = 0;
135 const Nom& nom_bord = liste_bords_perio[i];
136 const int nb_bords = zone.nb_bords();
137 while (num_bord < nb_bords)
138 {
139 if (zone.bord(num_bord).le_nom() == nom_bord)
140 break;
141 num_bord++;
142 }
143 if (num_bord == nb_bords)
144 {
145 Cerr << "Error in Partitionneur_Tranche::chercher_direction_perio\n"
146 << " boundary not found : " << nom_bord << finl;
147 exit();
148 }
149 ArrOfDouble delta;
150 ArrOfDouble erreur;
151 const double epsilon = Objet_U::precision_geom;
152 Reordonner_faces_periodiques::check_faces_periodiques(zone.bord(num_bord), delta, erreur);
153 int count = 0;
154 for (int j = 0; j < dim; j++)
155 {
156 if (std::fabs(delta[j]) > epsilon)
157 {
158 directions_perio[j]++;
159 Cerr << " Boundary : " << nom_bord << " periodic direction : " << j << finl;
160 count++;
161 }
162 }
163 if (count != 1)
164 {
165 Cerr << "Error in Partitionneur_Tranche::chercher_direction_perio\n"
166 << " periodic direction not found for the boundary " << nom_bord
167 << "\n Vector delta found between faces twin : " << delta
168 << "\n with a maximum error : " << erreur << finl;
169 exit();
170 }
171 }
172 if (max_array(directions_perio) > 1)
173 {
174 Cerr << "Error in Partitionneur_Tranche::chercher_direction_perio\n"
175 << " several boundaries have the same periodic direction" << finl;
176 exit();
177 }
178}

180void Partitionneur_Tranche::construire_partition(IntVect& elem_part, int& nb_parts_tot) const
181{
182 assert(ref_domaine_.non_nul());
183 assert(nb_tranches_[0] > 0);
184
185 const Domaine& dom = ref_domaine_.valeur();
186 const Zone& zone = dom.zone(0);
187 const int nb_elem = zone.nb_elem();
188
189 const int dim = dom.dimension;
190
191// Centre de gravite des elements:
192 DoubleTab coord_g;
193// Index utilise pour trier les elements
194 ArrOfInt index;
195
196// Pour chaque dimension d'espace, cette direction est-elle
197// une direction de periodicite
198 ArrOfInt directions_perio;
199 chercher_direction_perio(zone, liste_bords_periodiques_, directions_perio);
200
201 Cerr << "Calculation of centers of gravity of the elements" << finl;
202 zone.calculer_centres_gravite(coord_g);
203 assert(coord_g.dimension(0) == nb_elem);
204 assert(coord_g.dimension(1) == dim);
205
206 Cerr << "Moving of centers of gravity" << finl;
207// Deplacement des coordonnees pour qu'il y ait une vraie relation d'ordre
208// dans chaque direction (en VDF, eviter les coordonnees identiques)
209 {
210 for (int i = 0; i < nb_elem; i++)
211 {
212 double x = coord_g(i,0);
213 double y = coord_g(i,1);
214 double z = (dim == 3) ? coord_g(i,2) : 0.;
215 coord_g(i,0) = x + y*1e-6 + z*1e-12;
216 coord_g(i,1) = y + x*1e-6 + z*1e-12;
217 if (dim == 3)
218 coord_g(i,2) = z + x*1e-6 + y*1e-12;
219 }
220 }
221
222// Nombre d'elements dans chaque partie
223// (initialisation: une partie contenant nb_elem elements)
224 ArrOfInt nb_elem_part(1);
225 nb_elem_part= nb_elem;
226
227// Dans l'ordre croissant des parties, liste des elements
228// contenus dans la partie
229// (exemple:
230// de 0..nb_elem_part[0]-1 (elements de la partie 0)
231// de nb_elem_part[0]..nb_elem_part[0]+nb_elem_part[1]-1 (partie 1)
232// Initialisation: une seule grosse partie, tous les elements
233 ArrOfInt listes_elem(nb_elem);
234 for (int i = 0; i < nb_elem; i++)
235 listes_elem[i] = i;
236
237 ArrOfInt new_nb_elem_part;
238 new_nb_elem_part.set_smart_resize(1);
239
240// Algorithme: on cree nb_tranches[0] tranches dans la direction x,
241// chaque tranche compte le meme nombre d'elements du domaine.
242// Puis on prend chacune de ces tranches et on la decoupe en nb_tranches[1]
243// parties dans la direction y, elles aussi equilibrees.
244// Enfin, on prend chacune des parties obtenues et on la coupe en nb_tranches[2]
245// parties dans la direction z.
246// Attention: si le domaine n'est pas un parallelipipede, le decoupage
247// n'est pas forcement cartesien:
248// ------------------
249// | | |
250// |-------| |
251// | |--------| <- les coins des tranches ne coincident pas forcement
252// ------------------
253
254 for (int direction = 0; direction < dim; direction++)
255 {
256// Nombre de tranches a creer dans cette direction
257 const int nb_tranches = nb_tranches_[direction];
258
259 int index_debut_partie = 0;
260 new_nb_elem_part.resize_array(0);
261 const int nb_parts = nb_elem_part.size_array();
262
263 Cerr << "The mesh contains " << nb_parts << " parts. ";
264 Cerr << "Splitting in the direction " << direction << finl;
265
266// Boucle sur les parties deja creees. On divise chaque partie
267// en nb_tranches sous-parties:
268 for (int part = 0; part < nb_parts; part++)
269 {
270 Cerr << " Splitting of subpart " << part << finl;
271// Extraction de la liste des elements de cette partie:
272// on fait pointer le tableau index vers une sous-partie de listes_elem
273 const int nb_elem_partie = nb_elem_part[part];
274 assert(index_debut_partie >= 0 && index_debut_partie + nb_elem_partie <= nb_elem);
275 index.ref_data(listes_elem.addr()+index_debut_partie, nb_elem_partie);
276// Tri des elements de cette partie dans l'ordre croissant
277// de la coordonnee (index pointe sur une partie de liste_elem, on trie
278// donc en realite une partie du tableau liste_elem)
279 trier_index_colonne_i(coord_g, index, direction);
280
281// Si cette direction est periodique, permutation circulaire de
282// nb_elem_partie/(nb_tranches*2) elements pour que les elements de l'extremite
283// se retrouvent sur la meme partie que ceux de l'extremite opposee
284 if (directions_perio[direction])
285 {
286 ArrOfInt copie(nb_elem_partie);
287 copie.inject_array(index);
288 int j = nb_elem_partie/(nb_tranches*2);
289 for (int i = 0; i < nb_elem_partie; i++)
290 {
291 index[j++] = copie[i];
292 if (j >= nb_elem_partie)
293 j = 0;
294 }
295 }
296// Construction d'une partition en nb_tranches parties, par
297// ordre croissant des coordonnees
298 {
299 int n = 0;
300 for (int i = 0; i < nb_tranches; i++)
301 {
302// const int new_n = nb_elem_partie * (i+1) / nb_tranches;
303// Rewritten cause nb_elem_partie * (i+1) > 2^31 for big meshes
304 const int new_n = (int) ((double)nb_elem_partie * (i+1) / nb_tranches);
305 new_nb_elem_part.append_array(new_n - n);
306 n = new_n;
307 }
308 }
309 index_debut_partie += nb_elem_partie;
310 }
311
312 nb_elem_part = new_nb_elem_part;
313 }
314
315// Partition initiale du domaine: au depart tout dans 0
316 elem_part.resize(nb_elem);
317 elem_part = 0;
318// Construction de elem_part
319 const int nb_parts = nb_elem_part.size_array();
320 int index_fin = 0;
321 int i = 0;
322 for (int part = 0; part < nb_parts; part++)
323 {
324 index_fin += nb_elem_part[part];
325 for (; i < index_fin; i++)
326 {
327 const int elem = listes_elem[i];
328 elem_part[elem] = part;
329 }
330 }
331
332 if (liste_bords_periodiques_.size() > 0)
333 corriger_bords_avec_liste(dom, liste_bords_periodiques_, 0, elem_part);
334
335 Cerr << "Correction elem0 on processor 0" << finl;
336 corriger_elem0_sur_proc0(elem_part);
337}
