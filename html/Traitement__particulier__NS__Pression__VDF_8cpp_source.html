167void Traitement_particulier_NS_Pression_VDF::post_traitement_particulier_calcul_pression()
168{
169 const Zone_VDF& zvdf=ref_cast(Zone_VDF, mon_equation->zone_dis().valeur());
170 const DoubleVect& porosite_face = zvdf.porosite_face();
171 int i;
172 int nb_face = zvdf.nb_faces();
173 Champ_Inc gradient_P;
174 Champ_Inc la_pression = mon_equation->pression();
175 Operateur_Div divergence = mon_equation->operateur_divergence();
176 Operateur_Grad gradient = mon_equation->operateur_gradient();
177 SolveurSys solveur_pression_ = mon_equation->solveur_pression();
178
179 DoubleTab& pression=la_pression.valeurs();
180 DoubleTrav inc_pre(pression);
181 DoubleTrav secmem(pression);
182 gradient.calculer(la_pression.valeurs(),gradient_P.valeurs());
183
184// on veut BM-1Bt(spi*Pression)
185 DoubleTab& grad=gradient_P.valeurs();
186 mon_equation->solv_masse().appliquer(grad);
187 DoubleTab& grad_temp = grad;
188 for(i=0; i<nb_face; i++)
189 {
190 grad_temp(i) /=porosite_face(i);
191 }
192 divergence.calculer(grad_temp, secmem);
193 secmem *= -1; //!< car div =-B
194 solveur_pression_.resoudre_systeme(mon_equation->matrice_pression().valeur(),secmem, inc_pre);
195 Cerr <<"inc_pre " << inc_pre << finl;
196}
